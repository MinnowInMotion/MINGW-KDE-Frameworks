54 void KWindowInfo::Private::updateData()
55 {
56  ProcessInfoRec pinfo;
57  char processName[512];
58 #ifdef Q_OS_MAC32
59  FSSpec appSpec;
60 #else
61  FSRef ref;
62 #endif
63  pinfo.processInfoLength = sizeof pinfo;
64  pinfo.processName = (unsigned char *) processName;
65 #ifdef Q_OS_MAC32
66  pinfo.processAppSpec = &appSpec;
67 #else
68  pinfo.processAppRef = &ref;
69 #endif
70  GetProcessInformation(&m_psn, &pinfo);
71  name = QString::fromLatin1(processName + 1, processName[0]);
72 
73  if (m_axWin) {
74  CFStringRef title;
75  if (AXUIElementCopyAttributeValue(m_axWin, kAXTitleAttribute, (CFTypeRef *)&title) == noErr) {
76  CFStringGetCString(title, processName, sizeof processName, kCFStringEncodingUTF8);
77  name = QString::fromUtf8(processName);
78  }
79  }
80 
81 #ifdef Q_OS_MAC32
82  iconSpec = appSpec;
83 
84  FSRef ref;
85  FSpMakeFSRef(&appSpec, &ref);
86 #else
87  iconSpec = ref;
88 #endif
89  // check if it is in an application bundle (foo.app/Contents/MacOS/plasma)
90  HFSUniStr255 name;
91  FSRef parentRef;
92  FSGetCatalogInfo(&ref, kFSCatInfoNone, 0, &name, 0, &parentRef);
93  ref = parentRef;
94  FSGetCatalogInfo(&ref, kFSCatInfoNone, 0, &name, 0, &parentRef);
95  if (QString::fromUtf16(name.unicode, name.length) == "MacOS") {
96  ref = parentRef;
97  FSGetCatalogInfo(&ref, kFSCatInfoNone, 0, &name, 0, &parentRef);
98  if (QString::fromUtf16(name.unicode, name.length) == "Contents") {
99 #ifdef Q_OS_MAC32
100  FSSpec spec;
101  ref = parentRef;
102  FSGetCatalogInfo(&ref, kFSCatInfoNone, 0, &name, &spec, &parentRef);
103  iconSpec = spec;
104 #else
105  iconSpec = parentRef;
106 #endif
107  }
108  }
109 
110  loadedData = true;
111 }

344 // see NETWM spec section 7.6
345 bool KWindowInfo::isMinimized() const
346 {
347  if (mappingState() != NET::Iconic) {
348  return false;
349  }
350  // NETWM 1.2 compliant WM - uses NET::Hidden for minimized windows
351  if ((state() & NET::Hidden) != 0
352  && (state() & NET::Shaded) == 0) { // shaded may have NET::Hidden too
353  return true;
354  }
355  // older WMs use WithdrawnState for other virtual desktops
356  // and IconicState only for minimized
357  return KWindowSystem::icccmCompliantMappingState() ? false : true;
358 }
