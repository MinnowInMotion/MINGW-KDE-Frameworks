320 void KStartupInfo::Private::new_startup_info_internal(const KStartupInfoId &id_P,
321  KStartupInfo::Data &data_P, bool update_P)
322 {
323  if (id_P.isNull()) {
324  return;
325  }
326  if (startups.contains(id_P)) {
327  // already reported, update
328  startups[ id_P ].update(data_P);
329  startups[ id_P ].age = 0; // CHECKME
330  //qCDebug(LOG_KWINDOWSYSTEM) << "updating";
331  if (startups[ id_P ].silent() == KStartupInfo::Data::Yes
332  && !(flags & AnnounceSilenceChanges)) {
333  silent_startups[ id_P ] = startups[ id_P ];
334  startups.remove(id_P);
335  emit q->gotRemoveStartup(id_P, silent_startups[ id_P ]);
336  return;
337  }
338  emit q->gotStartupChange(id_P, startups[ id_P ]);
339  return;
340  }
341  if (silent_startups.contains(id_P)) {
342  // already reported, update
343  silent_startups[ id_P ].update(data_P);
344  silent_startups[ id_P ].age = 0; // CHECKME
345  //qCDebug(LOG_KWINDOWSYSTEM) << "updating silenced";
346  if (silent_startups[ id_P ].silent() != Data::Yes) {
347  startups[ id_P ] = silent_startups[ id_P ];
348  silent_startups.remove(id_P);
349  q->emit gotNewStartup(id_P, startups[ id_P ]);
350  return;
351  }
352  emit q->gotStartupChange(id_P, silent_startups[ id_P ]);
353  return;
354  }
355  if (uninited_startups.contains(id_P)) {
356  uninited_startups[ id_P ].update(data_P);
357  //qCDebug(LOG_KWINDOWSYSTEM) << "updating uninited";
358  if (!update_P) { // uninited finally got new:
359  startups[ id_P ] = uninited_startups[ id_P ];
360  uninited_startups.remove(id_P);
361  emit q->gotNewStartup(id_P, startups[ id_P ]);
362  return;
363  }
364  // no change announce, it's still uninited
365  return;
366  }
367  if (update_P) { // change: without any new: first
368  //qCDebug(LOG_KWINDOWSYSTEM) << "adding uninited";
369  uninited_startups.insert(id_P, data_P);
370  } else if (data_P.silent() != Data::Yes || flags & AnnounceSilenceChanges) {
371  //qCDebug(LOG_KWINDOWSYSTEM) << "adding";
372  startups.insert(id_P, data_P);
373  emit q->gotNewStartup(id_P, data_P);
374  } else { // new silenced, and silent shouldn't be announced
375  //qCDebug(LOG_KWINDOWSYSTEM) << "adding silent";
376  silent_startups.insert(id_P, data_P);
377  }
378  cleanup->start(1000); // 1 sec
379 }

439 void KStartupInfo::Private::remove_startup_pids(const KStartupInfoId &id_P,
440  const KStartupInfoData &data_P)
441 {
442  if (data_P.pids().count() == 0) {
443  qFatal("data_P.pids().count() == 0");
444  }
445  Data *data = NULL;
446  if (startups.contains(id_P)) {
447  data = &startups[ id_P ];
448  } else if (silent_startups.contains(id_P)) {
449  data = &silent_startups[ id_P ];
450  } else if (uninited_startups.contains(id_P)) {
451  data = &uninited_startups[ id_P ];

457  ++it2) {
458  data->d->remove_pid(*it2); // remove all pids from the info
459  }
460  if (data->pids().count() == 0) { // all pids removed -> remove info
461  removeAllStartupInfoInternal(id_P);
462  }

826 KStartupInfo::startup_t KStartupInfo::Private::check_startup_internal(WId w_P, KStartupInfoId *id_O,
827  KStartupInfoData *data_O)
828 {
829  if (startups.count() == 0) {
830  return NoMatch; // no startups
831  }
832  // Strategy:
833  //
834  // Is this a compliant app ?
835  // - Yes - test for match
836  // - No - Is this a NET_WM compliant app ?
837  // - Yes - test for pid match
838  // - No - test for WM_CLASS match
839  qCDebug(LOG_KWINDOWSYSTEM) << "check_startup";
840  QByteArray id = windowStartupId(w_P);
841  if (!id.isNull()) {
842  if (id.isEmpty() || id == "0") { // means ignore this window
843  qCDebug(LOG_KWINDOWSYSTEM) << "ignore";
844  return NoMatch;
845  }
846  return find_id(id, id_O, data_O) ? Match : NoMatch;
847  }
848 #if KWINDOWSYSTEM_HAVE_X11
849  if (!QX11Info::isPlatformX11()) {
850  qCDebug(LOG_KWINDOWSYSTEM) << "check_startup:cantdetect";
851  return CantDetect;
852  }
853  NETWinInfo info(QX11Info::connection(), w_P, QX11Info::appRootWindow(),
854  NET::WMWindowType | NET::WMPid | NET::WMState,
855  NET::WM2WindowClass | NET::WM2ClientMachine | NET::WM2TransientFor);
856  pid_t pid = info.pid();
857  if (pid > 0) {
858  QByteArray hostname = info.clientMachine();
859  if (!hostname.isEmpty()
860  && find_pid(pid, hostname, id_O, data_O)) {
861  return Match;
862  }
863  // try XClass matching , this PID stuff sucks :(
864  }
865  if (find_wclass(info.windowClassName(), info.windowClassClass(), id_O, data_O)) {
866  return Match;
867  }
868  // ignore NET::Tool and other special window types, if they can't be matched
869  NET::WindowType type = info.windowType(NET::NormalMask | NET::DesktopMask
870  | NET::DockMask | NET::ToolbarMask | NET::MenuMask | NET::DialogMask
871  | NET::OverrideMask | NET::TopMenuMask | NET::UtilityMask | NET::SplashMask);
872  if (type != NET::Normal
873  && type != NET::Override
874  && type != NET::Unknown
875  && type != NET::Dialog
876  && type != NET::Utility)
877 // && type != NET::Dock ) why did I put this here?
878  {
879  return NoMatch;
880  }
881  // lets see if this is a transient
882  xcb_window_t transient_for = info.transientFor();
883  if (transient_for != QX11Info::appRootWindow() && transient_for != XCB_WINDOW_NONE) {
884  return NoMatch;
885  }
886 #endif
887  qCDebug(LOG_KWINDOWSYSTEM) << "check_startup:cantdetect";
888  return CantDetect;
889 }
