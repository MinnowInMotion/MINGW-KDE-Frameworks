107 int KXErrorHandler::handle(Display *dpy, XErrorEvent *e)
108 {
109  if (dpy == d->display
110  // e->serial >= d->first_request , compare like X timestamps to handle wrapping
111  && NET::timestampCompare(e->serial, d->first_request) >= 0) {
112  // it's for us
113  //qDebug( "Handling: %p", static_cast< void* >( this ));
114  bool error = false;
115  if (user_handler1 != NULL) {
116  if (user_handler1(e->request_code, e->error_code, e->resourceid)) {
117  error = true;
118  }
119  } else if (user_handler2 != NULL) {
120  if (user_handler2(dpy, e) != 0) {
121  error = true;
122  }
123  } else { // no handler set, simply set that there was an error
124  error = true;
125  }
126  if (error && !d->was_error) {
127  // only remember the first
128  d->was_error = true;
129  d->error_event = *e;
130  }
131  return 0;
132  }
133  //qDebug( "Going deeper: %p", static_cast< void* >( this ));
134  return old_handler(dpy, e);
135 }

158  }
159  }
160 #if 0

162  // XGetErrorText() currently has a bug that makes it fail to find text
163  // for some errors (when error==error_base), also XGetErrorDatabaseText()
164  // requires the right extension name, so it is needed to get info about
165  // all extensions. However that is almost impossible:
