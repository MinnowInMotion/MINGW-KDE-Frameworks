69 void updateScancodes()
70 {
71 #ifdef QT_MAC_USE_COCOA
72  TISInputSourceRef layout = TISCopyCurrentKeyboardLayoutInputSource();
73  if (!layout) {
74  qWarning() << "Error retrieving current layout";
75  return;
76  }
77  if (layout == lastLayout) {
78  CFRelease(layout);
79  } else {
80  // keyboard layout changed
81 #ifndef NDEBUG
82  const void *name = TISGetInputSourceProperty(layout, kTISPropertyLocalizedName);
83  qDebug() << "Layout changed to: " << CFStringGetCStringPtr((CFStringRef)name, 0);
84 #endif
85  lastLayout = layout;
86  scancodes.clear();
87 
88  CFDataRef data = static_cast<CFDataRef>(TISGetInputSourceProperty(layout,
89  kTISPropertyUnicodeKeyLayoutData));
90  const UCKeyboardLayout *ucData = data ? reinterpret_cast<const UCKeyboardLayout *>(CFDataGetBytePtr(data)) : 0;
91 
92  if (!ucData) {
93  qWarning() << "Error retrieving current layout character data";
94  return;
95  }
96 
97  for (int i = 0; i < 128; ++i) {
98  UInt32 tmpState = 0;
99  UniChar str[4];
100  UniCharCount actualLength = 0;
101  OSStatus err = UCKeyTranslate(ucData, i, kUCKeyActionDown, 0, LMGetKbdType(),
102  kUCKeyTranslateNoDeadKeysMask, &tmpState, 4, &actualLength, str);
103  if (err != noErr) {
104  qWarning() << "Error translating unicode key" << err;
105  } else {
106  if (str[0] && str[0] != kFunctionKeyCharCode) {
107  scancodes.insert(str[0], i);
108  }
109  }
110  }
111  }
112 #else
113  KeyboardLayoutRef layout;
114  if (KLGetCurrentKeyboardLayout(&layout) != noErr) {
115  qWarning() << "Error retrieving current layout";
116  }
117  if (layout != lastLayout) {
118 #ifndef NDEBUG
119  void *name;
120  KLGetKeyboardLayoutProperty(layout, kKLName, const_cast<const void **>(&name));
121  qDebug() << "Layout changed to: " << CFStringGetCStringPtr((CFStringRef) name, 0);
122 #endif
123  lastLayout = layout;
124  scancodes.clear();
125  void *kchr;
126  if (KLGetKeyboardLayoutProperty(layout, kKLKCHRData, const_cast<const void **>(&kchr)) != noErr) {
127  qWarning() << "Couldn't load active keyboard layout";
128  } else {
129  for (int i = 0; i < 128; i++) {
130  UInt32 tmpState = 0;
131  UInt32 chr = KeyTranslate(kchr, i, &tmpState);
132  if (chr && chr != kFunctionKeyCharCode) {
133  scancodes.insert(chr, i);
134  }
135  }
136  }
137  }
138 #endif
139 }

158 bool keyQtToSymMac(int keyQt, int &sym)
159 {
160  // Printable ascii values, before A
161  if (keyQt >= 0x20 && keyQt < Qt::Key_A) {
162  sym = keyQt;
163  return true;
164  }
165  // Letters, return lower-case equivalent
166  if (keyQt >= Qt::Key_A && keyQt <= Qt::Key_Z) {
167  sym = keyQt - Qt::Key_A + 'a';
168  return true;
169  }
170  // Printable ascii values up to lower-case a
171  if (keyQt > Qt::Key_Z && keyQt <= 0x60) {
172  sym = keyQt;
173  return true;
174  }
175  // Remainder of printable ascii values
176  if (keyQt >= 0x7B && keyQt < 0x7F) {
177  sym = keyQt;
178  return true;
179  }
180  // Function keys
181  if (keyQt >= Qt::Key_F1 && keyQt <= Qt::Key_F35) {
182  sym = kFunctionKeyCharCode;
183  return true;
184  }
185  // Try to find in lookup table
186  for (int i = 0; qtKeyToChar[i].qt_code; i++) {
187  if (qtKeyToChar[i].qt_code == keyQt) {
188  sym = qtKeyToChar[i].mac_code;
189  return true;
190  }
191  }
192 
193  // Not found
194  return false;
195 }

197 bool keyQtToCodeMac(int keyQt, QList<uint> &keyCodes)
198 {
199  updateScancodes();
200  keyCodes.clear();
201  keyQt &= ~Qt::KeyboardModifierMask;
202  int chr;
203  if (!keyQtToSymMac(keyQt, chr)) {
204  return false;
205  }
206 
207  if (chr == kFunctionKeyCharCode) {
208  for (int i = 0; functionKeys[i].qt_code; i++) {
209  if (functionKeys[i].qt_code == keyQt) {
210  keyCodes.append(functionKeys[i].mac_code);
211  }
212  }
213  } else {
214  keyCodes += scancodes.values(chr);
215  }
216 
217  return keyCodes.count() > 0;
218 }

220 bool keyQtToModMac(int keyQt, uint &mod)
221 {
222  mod = 0;
223  if (keyQt & Qt::ShiftModifier) {
224  mod |= shiftKey;
225  }
226  if (keyQt & Qt::ControlModifier) {
227  mod |= cmdKey;
228  }
229  if (keyQt & Qt::AltModifier) {
230  mod |= optionKey;
231  }
232  if (keyQt & Qt::MetaModifier) {
233  mod |= controlKey;
234  }
235  if (keyQt & Qt::KeypadModifier) {
236  mod |= kEventKeyModifierNumLockMask;
237  }
238  // Special case for Qt::Key_Backtab
239  if ((keyQt & ~Qt::KeyboardModifierMask) == Qt::Key_Backtab) {
240  mod |= shiftKey;
241  }
242 
243  return true;
244 }
