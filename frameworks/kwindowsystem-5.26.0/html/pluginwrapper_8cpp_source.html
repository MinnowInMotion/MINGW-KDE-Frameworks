37 static QStringList pluginCandidates()
38 {
39  QStringList ret;
40  foreach (const QString &path, QCoreApplication::libraryPaths()) {
41  QDir pluginDir(path + QLatin1Literal("/kf5/org.kde.kwindowsystem.platforms"));
42  if (!pluginDir.exists()) {
43  continue;
44  }
45  foreach (const QString &entry, pluginDir.entryList(QDir::Files | QDir::NoDotAndDotDot)) {
46  ret << pluginDir.absoluteFilePath(entry);
47  }
48  }
49  return ret;
50 }

52 static KWindowSystemPluginInterface *loadPlugin()
53 {
54  foreach (const QString &candidate, pluginCandidates()) {
55  if (!QLibrary::isLibrary(candidate)) {
56  continue;
57  }
58  QPluginLoader loader(candidate);
59  QJsonObject metaData = loader.metaData();
60  const QJsonArray platforms = metaData.value(QStringLiteral("MetaData")).toObject().value(QStringLiteral("platforms")).toArray();
61  for (auto it = platforms.begin(); it != platforms.end(); ++it) {
62  if (QString::compare(QGuiApplication::platformName(), (*it).toString(), Qt::CaseInsensitive) == 0) {
63  KWindowSystemPluginInterface *interface = qobject_cast< KWindowSystemPluginInterface* >(loader.instance());
64  if (interface) {
65  qCDebug(LOG_KWINDOWSYSTEM) << "Loaded plugin" << candidate << "for platform" << QGuiApplication::platformName();
66  return interface;
67  }
68  }
69  }
70  }
71  qCWarning(LOG_KWINDOWSYSTEM) << "Could not find any platform plugin";
72  return Q_NULLPTR;
73 }
