96 static HBITMAP QPixmapMask2HBitmap(const QPixmap &pix)
97 {
98  QBitmap bm = pix.mask();
99  if (bm.isNull()) {
100  bm = QBitmap(pix.size());
101  bm.fill(Qt::color1);
102  }
103  QImage im = bm.toImage().convertToFormat(QImage::Format_Mono);
104  im.invertPixels(); // funny blank'n'white games on windows
105  int w = im.width();
106  int h = im.height();
107  int bpl = ((w + 15) / 16) * 2; // bpl, 16 bit alignment
108  QByteArray bits(bpl * h, '\0');
109  for (int y = 0; y < h; y++) {
110  memcpy(bits.data() + y * bpl, im.scanLine(y), bpl);
111  }
112  return CreateBitmap(w, h, 1, 1, bits.constData());
113 }

168 bool KWindowSystemPrivate::nativeEvent(const QByteArray &eventType, void *message_, long *result)
169 {
170  if (eventType != QByteArrayLiteral("windows_generic_MSG")) {
171  return QWidget::nativeEvent(eventType, message_, result);
172  }
173 
174  MSG *message = static_cast<MSG *>(message_);
175 
176  /*
177  check winuser.h for the following codes
178  HSHELL_WINDOWCREATED 1
179  HSHELL_WINDOWDESTROYED 2
180  HSHELL_ACTIVATESHELLWINDOW 3
181  HSHELL_WINDOWACTIVATED 4
182  HSHELL_GETMINRECT 5
183  HSHELL_RUDEAPPACTIVATED 32768 + 4 = 32772
184  HSHELL_REDRAW 6
185  HSHELL_FLASH 32768 + 6 = 32774
186  HSHELL_TASKMAN 7
187  HSHELL_LANGUAGE 8
188  HSHELL_SYSMENU 9
189  HSHELL_ENDTASK 10
190  HSHELL_ACCESSIBILITYSTATE 11
191  HSHELL_APPCOMMAND 12
192  HSHELL_WINDOWREPLACED 13
193  HSHELL_WINDOWREPLACING 14
194  */
195  if (message->message == WM_SHELLHOOK) {
196 // qDebug() << "what has happened?:" << message->wParam << message->message;
197 
198  switch (message->wParam) {
199  case HSHELL_WINDOWCREATED:
200  KWindowSystem::s_d_func()->windowAdded(static_cast<WId>(message->lParam));
201  break;
202  case HSHELL_WINDOWDESTROYED:
203  KWindowSystem::s_d_func()->windowRemoved(static_cast<WId>(message->lParam));
204  break;
205  case HSHELL_WINDOWACTIVATED:
206 #ifndef _WIN32_WCE
207  case HSHELL_RUDEAPPACTIVATED:
208 #endif
209  KWindowSystem::s_d_func()->windowActivated(static_cast<WId>(message->lParam));
210  break;
211 #ifndef _WIN32_WCE
212  case HSHELL_GETMINRECT:
213  KWindowSystem::s_d_func()->windowStateChanged(static_cast<WId>(message->lParam));
214  break;
215  case HSHELL_REDRAW: //the caption has changed
216  KWindowSystem::s_d_func()->windowRedraw(static_cast<WId>(message->lParam));
217  break;
218  case HSHELL_FLASH:
219  KWindowSystem::s_d_func()->windowFlash(static_cast<WId>(message->lParam));
220  break;
221 #endif
222  }
223  }
224  return QWidget::nativeEvent(eventType, message_, result);
225 }

233  KWindowSystemPrivate *p = KWindowSystem::s_d_func();
234 
235  QString add;
236  if (!QString::fromWCharArray((wchar_t *)windowText.data()).trimmed().isEmpty() && IsWindowVisible(hWnd) && !(ex_style & WS_EX_TOOLWINDOW)
237  && !GetParent(hWnd) && !GetWindow(hWnd, GW_OWNER) && !p->winInfos.contains(win)) {
238 
239 // qDebug()<<"Adding window to windowList " << add + QString(windowText).trimmed();
240 
241  InternalWindowInfo winfo;

432 void KWindowSystem::forceActiveWindow(WId win, long time)
433 {
434  HWND hwnd = reinterpret_cast<HWND>(win);
435  // FIXME restoring a hidden window doesn't work: the window contents just appear white.
436  // But the mouse cursor still acts as if the widgets were there (e.g. button clicking works),
437  // which indicates the issue is at the window/backingstore level.
438  // This is probably a side effect of bypassing Qt's internal window state handling.
439 #ifndef _WIN32_WCE
440  if (IsIconic(hwnd) /*|| !IsWindowVisible( win ) */) {
441  // Do not activate the window as we restore it,
442  // otherwise the window appears see-through (contents not updated).
443  ShowWindow(hwnd, SW_SHOWNOACTIVATE);
444  }
445 #endif
446  // Puts the window in front and activates it.
447  //to bring a window to the front while the user is active in a different apllication we
448  //have to atach our self to the current active window
449  HWND hwndActiveWin = GetForegroundWindow();
450  int idActive = GetWindowThreadProcessId(hwndActiveWin, NULL);
451  if (AttachThreadInput(GetCurrentThreadId(), idActive, TRUE)) {
452  SetForegroundWindow(hwnd);
453  SetFocus(hwnd);
454  AttachThreadInput(GetCurrentThreadId(), idActive, FALSE);
455  }
456 
457 }

527 void KWindowSystem::setState(WId win, NET::States state)
528 {
529  HWND hwnd = reinterpret_cast<HWND>(win);
530  bool got = false;
531 #ifndef _WIN32_WCE
532  if (state & NET::SkipTaskbar) {
533  got = true;
534  LONG_PTR lp = GetWindowLongPtr(hwnd, GWL_EXSTYLE);
535  SetWindowLongPtr(hwnd, GWL_EXSTYLE, lp | WS_EX_TOOLWINDOW);
536  }
537 #endif
538  if (state & NET::KeepAbove) {
539  got = true;
540  SetWindowPos(hwnd, HWND_TOPMOST, 0, 0, 0, 0, SWP_NOMOVE | SWP_NOSIZE);
541  }
542  if (state & NET::KeepBelow) {
543  got = true;
544  SetWindowPos(hwnd, HWND_BOTTOM, 0, 0, 0, 0, SWP_NOMOVE | SWP_NOSIZE);
545  }
546  if (state & NET::Max) {
547  got = true;
548  ShowWindow(hwnd, SW_MAXIMIZE);
549  }
550  if (!got) {
551  qDebug() << "KWindowSystem::setState( WId win, unsigned long state ) isn't yet implemented for the state you requested!";
552  }
