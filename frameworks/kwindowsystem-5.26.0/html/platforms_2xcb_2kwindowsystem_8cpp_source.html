51 static inline const QRect &displayGeometry()
52 {
53  static QRect displayGeometry;
54  static bool isDirty = true;
55 
56  if (isDirty) {
57 
58  static QList<QMetaObject::Connection> connections;
59  auto dirtify = [&] {
60  isDirty = true;
61  foreach (const QMetaObject::Connection &con, connections)
62  QObject::disconnect(con);
63  connections.clear();
64  };
65 
66  QObject::connect(qApp, &QGuiApplication::screenAdded, dirtify);
67  QObject::connect(qApp, &QGuiApplication::screenRemoved, dirtify);
68  const QList<QScreen *> screenList = QGuiApplication::screens();
69  QRegion region;
70  for (int i = 0; i < screenList.count(); ++i) {
71  const QScreen *screen = screenList.at(i);
72  connections << QObject::connect(screen, &QScreen::geometryChanged, dirtify);
73  region += screen->geometry();
74  }
75  displayGeometry = region.boundingRect();
76  isDirty = false;
77  }
78 
79  return displayGeometry;
80 }

186 bool NETEventFilter::nativeEventFilter(xcb_generic_event_t *ev)
187 {
188  KWindowSystem *s_q = KWindowSystem::self();
189  const uint8_t eventType = ev->response_type & ~0x80;
190 
191  if (eventType == xfixesEventBase + XCB_XFIXES_SELECTION_NOTIFY) {
192  xcb_xfixes_selection_notify_event_t *event = reinterpret_cast<xcb_xfixes_selection_notify_event_t *>(ev);
193  if (event->window == winId) {
194  bool haveOwner = event->owner != XCB_WINDOW_NONE;
195  if (compositingEnabled != haveOwner) {
196  compositingEnabled = haveOwner;
197  emit s_q->compositingChanged(compositingEnabled);
198  }
199  return true;
200  }
201  // Qt compresses XFixesSelectionNotifyEvents without caring about the actual window
202  // gui/kernel/qapplication_x11.cpp
203  // until that can be assumed fixed, we also react on events on the root (caused by Qts own compositing tracker)
204  if (event->window == m_appRootWindow) {
205  if (event->selection == net_wm_cm) {
206  bool haveOwner = event->owner != XCB_WINDOW_NONE;
207  if (compositingEnabled != haveOwner) {
208  compositingEnabled = haveOwner;
209  emit s_q->compositingChanged(compositingEnabled);
210  }
211  // NOTICE this is not our event, we just randomly captured it from Qt -> pass on
212  return false;
213  }
214  }
215  return false;
216  }
217 
218  xcb_window_t eventWindow = XCB_WINDOW_NONE;
219  switch (eventType) {
220  case XCB_CLIENT_MESSAGE:
221  eventWindow = reinterpret_cast<xcb_client_message_event_t *>(ev)->window;
222  break;
223  case XCB_PROPERTY_NOTIFY:
224  eventWindow = reinterpret_cast<xcb_property_notify_event_t *>(ev)->window;
225  break;
226  case XCB_CONFIGURE_NOTIFY:
227  eventWindow = reinterpret_cast<xcb_configure_notify_event_t *>(ev)->window;
228  break;
229  }
230 
231  if (eventWindow == m_appRootWindow) {
232  int old_current_desktop = currentDesktop();
233  xcb_window_t old_active_window = activeWindow();
234  int old_number_of_desktops = numberOfDesktops();
235  bool old_showing_desktop = showingDesktop();
236  unsigned long m[ 5 ];
237  NETRootInfo::event(ev, m, 5);
238 
239  if ((m[ PROTOCOLS ] & CurrentDesktop) && currentDesktop() != old_current_desktop) {
240  emit s_q->currentDesktopChanged(currentDesktop());
241  }
242  if ((m[ PROTOCOLS ] & DesktopViewport) && mapViewport() && currentDesktop() != old_current_desktop) {
243  emit s_q->currentDesktopChanged(currentDesktop());
244  }
245  if ((m[ PROTOCOLS ] & ActiveWindow) && activeWindow() != old_active_window) {
246  emit s_q->activeWindowChanged(activeWindow());
247  }
248  if (m[ PROTOCOLS ] & DesktopNames) {
249  emit s_q->desktopNamesChanged();
250  }
251  if ((m[ PROTOCOLS ] & NumberOfDesktops) && numberOfDesktops() != old_number_of_desktops) {
252  emit s_q->numberOfDesktopsChanged(numberOfDesktops());
253  }
254  if ((m[ PROTOCOLS ] & DesktopGeometry) && mapViewport() && numberOfDesktops() != old_number_of_desktops) {
255  emit s_q->numberOfDesktopsChanged(numberOfDesktops());
256  }
257  if (m[ PROTOCOLS ] & WorkArea) {
258  emit s_q->workAreaChanged();
259  }
260  if (m[ PROTOCOLS ] & ClientListStacking) {
261  updateStackingOrder();
262  emit s_q->stackingOrderChanged();
263  }
264  if ((m[ PROTOCOLS2 ] & WM2ShowingDesktop) && showingDesktop() != old_showing_desktop) {
265  emit s_q->showingDesktopChanged(showingDesktop());
266  }
267  } else if (windows.contains(eventWindow)) {
268  NETWinInfo ni(QX11Info::connection(), eventWindow, m_appRootWindow, 0, 0);
269  NET::Properties dirtyProperties;
270  NET::Properties2 dirtyProperties2;
271  ni.event(ev, &dirtyProperties, &dirtyProperties2);
272  if (eventType == XCB_PROPERTY_NOTIFY) {
273  xcb_property_notify_event_t *event = reinterpret_cast<xcb_property_notify_event_t *>(ev);
274  if (event->atom == XCB_ATOM_WM_HINTS) {
275  dirtyProperties |= NET::WMIcon; // support for old icons
276  } else if (event->atom == XCB_ATOM_WM_NAME) {
277  dirtyProperties |= NET::WMName; // support for old name
278  } else if (event->atom == XCB_ATOM_WM_ICON_NAME) {
279  dirtyProperties |= NET::WMIconName; // support for old iconic name
280  }
281  }
282  if (mapViewport() && (dirtyProperties & (NET::WMState | NET::WMGeometry))) {
283  /* geometry change -> possible viewport change
284  * state change -> possible NET::Sticky change
285  */
286  dirtyProperties |= NET::WMDesktop;
287  }
288  if ((dirtyProperties & NET::WMStrut) != 0) {
289  removeStrutWindow(eventWindow);
290  if (!possibleStrutWindows.contains(eventWindow)) {
291  possibleStrutWindows.append(eventWindow);
292  }
293  }
294  if (dirtyProperties || dirtyProperties2) {
295  emit s_q->windowChanged(eventWindow);
296  emit s_q->windowChanged(eventWindow, dirtyProperties, dirtyProperties2);
297 #ifndef KWINDOWSYSTEM_NO_DEPRECATED
298  unsigned long dirty[ 2 ] = {dirtyProperties, dirtyProperties2};
299  emit s_q->windowChanged(eventWindow, dirty);
300  emit s_q->windowChanged(eventWindow, dirtyProperties);
301 #endif
302  if ((dirtyProperties & NET::WMStrut) != 0) {
303  emit s_q->strutChanged();
304  }
305  }
306  }
307 
308  return false;
309 }

331 void NETEventFilter::addClient(xcb_window_t w)
332 {
333  KWindowSystem *s_q = KWindowSystem::self();
334 
335  if ((what >= KWindowSystemPrivateX11::INFO_WINDOWS)) {
336  xcb_connection_t *c = QX11Info::connection();
337  QScopedPointer<xcb_get_window_attributes_reply_t, QScopedPointerPodDeleter> attr(xcb_get_window_attributes_reply(c,
338  xcb_get_window_attributes_unchecked(c, w), Q_NULLPTR));
339 
340  uint32_t events = XCB_EVENT_MASK_PROPERTY_CHANGE | XCB_EVENT_MASK_STRUCTURE_NOTIFY;
341  if (!attr.isNull()) {
342  events = events | attr->your_event_mask;
343  }
344  xcb_change_window_attributes(c, w, XCB_CW_EVENT_MASK, &events);
345  }
346 
347  bool emit_strutChanged = false;
348 
349  if (strutSignalConnected) {
350  NETWinInfo info(QX11Info::connection(), w, QX11Info::appRootWindow(), NET::WMStrut | NET::WMDesktop, 0);
351  NETStrut strut = info.strut();
352  if (strut.left || strut.top || strut.right || strut.bottom) {
353  strutWindows.append(StrutData(w, strut, info.desktop()));
354  emit_strutChanged = true;
355  }
356  } else {
357  possibleStrutWindows.append(w);
358  }
359 
360  windows.append(w);
361  emit s_q->windowAdded(w);
362  if (emit_strutChanged) {
363  emit s_q->strutChanged();
364  }
365 }

367 void NETEventFilter::removeClient(xcb_window_t w)
368 {
369  KWindowSystem *s_q = KWindowSystem::self();
370 
371  bool emit_strutChanged = removeStrutWindow(w);
372  if (strutSignalConnected && possibleStrutWindows.contains(w)) {
373  NETWinInfo info(QX11Info::connection(), w, QX11Info::appRootWindow(), NET::WMStrut, 0);
374  NETStrut strut = info.strut();
375  if (strut.left || strut.top || strut.right || strut.bottom) {
376  emit_strutChanged = true;
377  }
378  }
379 
380  possibleStrutWindows.removeAll(w);
381  windows.removeAll(w);
382  emit s_q->windowRemoved(w);
383  if (emit_strutChanged) {
384  emit s_q->strutChanged();
385  }

408 static void create_atoms()
409 {
410  if (!atoms_created) {
411  const int max = 20;
412  Atom *atoms[max];
413  const char *names[max];
414  Atom atoms_return[max];
415  int n = 0;
416 
417  atoms[n] = &_wm_protocols;
418  names[n++] = "WM_PROTOCOLS";
419 
420  atoms[n] = &_wm_change_state;
421  names[n++] = "WM_CHANGE_STATE";
422 
423  atoms[n] = &kwm_utf8_string;
424  names[n++] = "UTF8_STRING";
425 
426  char net_wm_cm_name[ 100 ];
427  sprintf(net_wm_cm_name, "_NET_WM_CM_S%d", QX11Info::appScreen());
428  atoms[n] = &net_wm_cm;
429  names[n++] = net_wm_cm_name;
430 
431  // we need a const_cast for the shitty X API
432  XInternAtoms(QX11Info::display(), const_cast<char **>(names), n, false, atoms_return);
433  for (int i = 0; i < n; i++) {
434  *atoms[i] = atoms_return[i];
435  }
436 
437  atoms_created = True;
438  }

470 // WARNING
471 // you have to call s_d_func() again after calling this function if you want a valid pointer!
472 void KWindowSystemPrivateX11::init(FilterInfo what)
473 {
474  NETEventFilter *const s_d = s_d_func();
475 
476  if (what >= INFO_WINDOWS) {
477  what = INFO_WINDOWS;
478  } else {
479  what = INFO_BASIC;
480  }
481 
482  if (!s_d || s_d->what < what) {
483  const bool wasCompositing = s_d ? s_d->compositingEnabled : false;
484  MainThreadInstantiator instantiator(what);
485  NETEventFilter *filter;
486  if (instantiator.thread() == QCoreApplication::instance()->thread()) {
487  filter = instantiator.createNETEventFilter();
488  } else {
489  // the instantiator is not in the main app thread, which implies
490  // we are being called in a thread that is not the main app thread
491  // so we move the instantiator to the main app thread and invoke
492  // the method with a blocking call
493  instantiator.moveToThread(QCoreApplication::instance()->thread());
494  QMetaObject::invokeMethod(&instantiator, "createNETEventFilter",
495  Qt::BlockingQueuedConnection,
496  Q_RETURN_ARG(NETEventFilter *, filter));
497  }
498  d.reset(filter);
499  d->activate();
500  if (wasCompositing != s_d_func()->compositingEnabled) {
501  emit KWindowSystem::self()->compositingChanged(s_d_func()->compositingEnabled);
502  }
503  }
504 }

608  unsigned int w, h, b, dp;
609  XGetGeometry(QX11Info::display(), win, &dummy, &x, &y, &w, &h, &b, &dp);
610  // get global position
611  XTranslateCoordinates(QX11Info::display(), win, QX11Info::appRootWindow(), 0, 0, &x, &y, &dummy);
612  x += w / 2; // center
613  y += h / 2;
614  // transform to coordinates on the current "desktop"
615  x = x % displayWidth();
616  y = y % displayHeight();
617  if (x < 0) {
618  x = x + displayWidth();
619  }
620  if (y < 0) {
621  y = y + displayHeight();
622  }
623  x += p.x(); // move to given "desktop"
624  y += p.y();
625  x -= w / 2; // from center back to topleft
626  y -= h / 2;
627  p = constrainViewportRelativePosition(QPoint(x, y));
628  int flags = (NET::FromTool << 12) | (0x03 << 8) | 10; // from tool(?), x/y, static gravity
629  NETEventFilter *const s_d = s_d_func();
630  s_d->moveResizeWindowRequest(win, flags, p.x(), p.y(), w, h);
631  return;
632  }
633  NETWinInfo info(QX11Info::connection(), win, QX11Info::appRootWindow(), NET::WMDesktop, 0);
634  info.setDesktop(desktop, true);

700 QPixmap KWindowSystemPrivateX11::iconFromNetWinInfo(int width, int height, bool scale, int flags, NETWinInfo *info)
701 {
702  QPixmap result;
703  if (!info) {
704  return result;
705  }
706  if (flags & KWindowSystem::NETWM) {
707  NETIcon ni = info->icon(width, height);
708  if (ni.data && ni.size.width > 0 && ni.size.height > 0) {
709  QImage img((uchar *) ni.data, (int) ni.size.width, (int) ni.size.height, QImage::Format_ARGB32);
710  if (scale && width > 0 && height > 0 && img.size() != QSize(width, height) && !img.isNull()) {
711  img = img.scaled(width, height, Qt::IgnoreAspectRatio, Qt::SmoothTransformation);
712  }
713  if (!img.isNull()) {
714  result = QPixmap::fromImage(img);
715  }
716  return result;
717  }
718  }
719 
720  if (flags & KWindowSystem::WMHints) {
721  xcb_pixmap_t p = info->icccmIconPixmap();
722  xcb_pixmap_t p_mask = info->icccmIconPixmapMask();
723 
724  if (p != XCB_PIXMAP_NONE) {
725  QPixmap pm = KXUtils::createPixmapFromHandle(info->xcbConnection(), p, p_mask);
726  if (scale && width > 0 && height > 0 && !pm.isNull()
727  && (pm.width() != width || pm.height() != height)) {
728  result = QPixmap::fromImage(pm.toImage().scaled(width, height, Qt::IgnoreAspectRatio, Qt::SmoothTransformation));
729  } else {
730  result = pm;
731  }
732  }
733  }
734 
735  // Since width can be any arbitrary size, but the icons cannot,
736  // take the nearest value for best results (ignoring 22 pixel
737  // icons as they don't exist for apps):
738  int iconWidth;
739  if (width < 24) {
740  iconWidth = 16;
741  } else if (width < 40) {
742  iconWidth = 32;
743  } else if (width < 56) {
744  iconWidth = 48;
745  } else if (width < 96) {
746  iconWidth = 64;
747  } else if (width < 192) {
748  iconWidth = 128;
749  } else {
750  iconWidth = 256;
751  }
752 
753  if (flags & KWindowSystem::ClassHint) {
754  // Try to load the icon from the classhint if the app didn't specify
755  // its own:
756  if (result.isNull()) {
757  const QIcon icon = QIcon::fromTheme(QString::fromUtf8(info->windowClassClass()).toLower());
758  const QPixmap pm = icon.isNull() ? QPixmap() : icon.pixmap(iconWidth, iconWidth);
759  if (scale && !pm.isNull()) {
760  result = QPixmap::fromImage(pm.toImage().scaled(width, height, Qt::IgnoreAspectRatio, Qt::SmoothTransformation));
761  } else {
762  result = pm;
763  }
764  }
765  }
766 
767  if (flags & KWindowSystem::XApp) {
768  // If the icon is still a null pixmap, load the icon for X applications
769  // as a last resort:
770  if (result.isNull()) {
771  const QIcon icon = QIcon::fromTheme(QLatin1String("xorg"));
772  const QPixmap pm = icon.isNull() ? QPixmap() : icon.pixmap(iconWidth, iconWidth);
773  if (scale && !pm.isNull()) {
774  result = QPixmap::fromImage(pm.toImage().scaled(width, height, Qt::IgnoreAspectRatio, Qt::SmoothTransformation));
775  } else {
776  result = pm;
777  }
778  }
779  }
780  return result;
781 }

907 QRect KWindowSystemPrivateX11::workArea(const QList<WId> &exclude, int desktop)
908 {
909  init(INFO_WINDOWS); // invalidates s_d_func's return value
910  NETEventFilter *const s_d = s_d_func();
911 
912  QRect all = displayGeometry();
913  QRect a = all;
914 
915  if (desktop == -1) {
916  desktop = s_d->currentDesktop();
917  }
918 
919  QList<WId>::ConstIterator it1;
920  for (it1 = s_d->windows.constBegin(); it1 != s_d->windows.constEnd(); ++it1) {
921 
922  if (exclude.contains(*it1)) {
923  continue;
924  }
925 
926 // Kicker (very) extensively calls this function, causing hundreds of roundtrips just
927 // to repeatedly find out struts of all windows. Therefore strut values for strut
928 // windows are cached here.
929  NETStrut strut;
930  auto it2 = s_d->strutWindows.begin();
931  for (; it2 != s_d->strutWindows.end(); ++it2)
932  if ((*it2).window == *it1) {
933  break;
934  }
935 
936  if (it2 != s_d->strutWindows.end()) {
937  if (!((*it2).desktop == desktop || (*it2).desktop == NETWinInfo::OnAllDesktops)) {
938  continue;
939  }
940 
941  strut = (*it2).strut;
942  } else if (s_d->possibleStrutWindows.contains(*it1)) {
943 
944  NETWinInfo info(QX11Info::connection(), (*it1), QX11Info::appRootWindow(), NET::WMStrut | NET::WMDesktop, 0);
945  strut = info.strut();
946  s_d->possibleStrutWindows.removeAll(*it1);
947  s_d->strutWindows.append(NETEventFilter::StrutData(*it1, info.strut(), info.desktop()));
948 
949  if (!(info.desktop() == desktop || info.desktop() == NETWinInfo::OnAllDesktops)) {
950  continue;
951  }
952  } else {
953  continue; // not a strut window
954  }
955 
956  QRect r = all;
957  if (strut.left > 0) {
958  r.setLeft(r.left() + (int) strut.left);
959  }
960  if (strut.top > 0) {
961  r.setTop(r.top() + (int) strut.top);
962  }
963  if (strut.right > 0) {
964  r.setRight(r.right() - (int) strut.right);
965  }
966  if (strut.bottom > 0) {
967  r.setBottom(r.bottom() - (int) strut.bottom);
968  }
969 
970  a = a.intersected(r);
971  }
972  return a;
973 }
