319 static void readIcon(xcb_connection_t *c, const xcb_get_property_cookie_t cookie,
320  NETRArray<NETIcon> &icons, int &icon_count)
321 {
322 #ifdef NETWMDEBUG
323  fprintf(stderr, "NET: readIcon\n");
324 #endif
325 
326  // reset
327  for (int i = 0; i < icons.size(); i++) {
328  delete [] icons[i].data;
329  }
330 
331  icons.reset();
332  icon_count = 0;
333 
334  xcb_get_property_reply_t *reply = xcb_get_property_reply(c, cookie, 0);
335 
336  if (!reply || reply->value_len < 3 || reply->format != 32 || reply->type != XCB_ATOM_CARDINAL) {
337  if (reply) {
338  free(reply);
339  }
340 
341  return;
342  }
343 
344  uint32_t *data = (uint32_t *) xcb_get_property_value(reply);
345 
346  for (unsigned int i = 0, j = 0; j < reply->value_len - 2; i++) {
347  uint32_t width = data[j++];
348  uint32_t height = data[j++];
349  uint32_t size = width * height * sizeof(uint32_t);
350  if (j + width * height > reply->value_len) {
351  fprintf(stderr, "Ill-encoded icon data; proposed size leads to out of bounds access. Skipping. (%d x %d)\n", width, height);
352  break;
353  }
354  if (width > 1024 || height > 1024) {
355  fprintf(stderr, "Warning: found huge icon. The icon data may be ill-encoded. (%d x %d)\n", width, height);
356  // do not break nor continue - the data may likely be junk, but causes no harm (yet) and might actually be just a huge icon, eg. when the icon system is abused to transfer wallpapers or such.
357  }
358 
359  icons[i].size.width = width;
360  icons[i].size.height = height;
361  icons[i].data = new unsigned char[size];
362 
363  memcpy((void *)icons[i].data, (const void *)&data[j], size);
364 
365  j += width * height;
366  icon_count++;
367  }
368 
369  free(reply);
370 
371 #ifdef NETWMDEBUG
372  fprintf(stderr, "NET: readIcon got %d icons\n", icon_count);
373 #endif
374 }

416 template <class Z>
417 Z &NETRArray<Z>::operator[](int index)
418 {
419  if (index >= capacity) {
420  // allocate space for the new data
421  // open table has amortized O(1) access time
422  // when N elements appended consecutively -- exa
423  int newcapacity = 2 * capacity > index + 1 ? 2 * capacity : index + 1; // max
424  // copy into new larger memory block using realloc
425  d = (Z *) realloc(d, sizeof(Z) * newcapacity);
426  memset((void *) &d[capacity], 0, sizeof(Z) * (newcapacity - capacity));
427  capacity = newcapacity;
428  }
429  if (index >= sz) { // at this point capacity>index
430  sz = index + 1;
431  }
432 
433  return d[index];
