72 static QString modToString(uint mod, bool bUserSpace)
73 {
74  if (bUserSpace && !g_bInitializedKKeyLabels) {
75  intializeKKeyLabels();
76  }
77 
78  QString s;
79  for (int i = 3; i >= 0; i--) {
80  if (mod & g_rgModInfo[i].modQt) {
81  if (!s.isEmpty()) {
82  s += QLatin1Char('+');
83  }
84  s += (bUserSpace)
85  ? *g_rgModInfo[i].sLabel
86  : QLatin1String(g_rgModInfo[i].psName);
87  }
88  }
89  return s;
90 }

97 uint stringUserToMod(const QString &mod)

102  }
103  }
104  return 0;
