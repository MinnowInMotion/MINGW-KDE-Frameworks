164 void KSelectionWatcher::init()
165 {
166  if (!d) {
167  return;
168  }
169  if (Private::manager_atom == XCB_NONE) {
170  xcb_connection_t *c = d->connection;
171 
172  xcb_intern_atom_cookie_t atom_cookie = xcb_intern_atom(c, false, strlen("MANAGER"), "MANAGER");
173  xcb_get_window_attributes_cookie_t attr_cookie = xcb_get_window_attributes(c, d->root);
174 
175  xcb_intern_atom_reply_t *atom_reply = xcb_intern_atom_reply(c, atom_cookie, 0);
176  Private::manager_atom = atom_reply->atom;
177  free(atom_reply);
178 
179  xcb_get_window_attributes_reply_t *attr = xcb_get_window_attributes_reply(c, attr_cookie, 0);
180  uint32_t event_mask = attr->your_event_mask;
181  free(attr);
182 
183  if (!(event_mask & XCB_EVENT_MASK_STRUCTURE_NOTIFY)) {
184  // We need XCB_EVENT_MASK_STRUCTURE_NORITY on the root window
185  event_mask |= XCB_EVENT_MASK_STRUCTURE_NOTIFY;
186  xcb_change_window_attributes(c, d->root, XCB_CW_EVENT_MASK, &event_mask);
187  }
188  }
189 
190  owner(); // trigger reading of current selection status
191 }

193 xcb_window_t KSelectionWatcher::owner()
194 {
195  if (!d) {
196  return XCB_WINDOW_NONE;
197  }
198  xcb_connection_t *c = d->connection;
199 
200  xcb_window_t current_owner = get_selection_owner(c, d->selection);
201  if (current_owner == XCB_NONE) {
202  return XCB_NONE;
203  }
204 
205  if (current_owner == d->selection_owner) {
206  return d->selection_owner;
207  }
208 
209  // We have a new selection owner - select for structure notify events
210  uint32_t mask = XCB_EVENT_MASK_STRUCTURE_NOTIFY;
211  xcb_void_cookie_t cookie = xcb_change_window_attributes_checked(c, current_owner, XCB_CW_EVENT_MASK, &mask);
212 
213  // Verify that the owner didn't change again while selecting for events
214  xcb_window_t new_owner = get_selection_owner(c, d->selection);
215  xcb_generic_error_t *err = xcb_request_check(c, cookie);
216 
217  if (!err && current_owner == new_owner) {
218  d->selection_owner = current_owner;
219  emit newOwner(d->selection_owner);
220  } else {
221  // ### This doesn't look right - the selection could have an owner
222  d->selection_owner = XCB_NONE;
223  }
224 
225  if (err) {
226  free(err);
227  }
228 
229  return d->selection_owner;
230 }

232 void KSelectionWatcher::filterEvent(void *ev_P)
233 {
234  if (!d) {
235  return;
236  }
237  xcb_generic_event_t *event = reinterpret_cast<xcb_generic_event_t *>(ev_P);
238  const uint response_type = event->response_type & ~0x80;
239  if (response_type == XCB_CLIENT_MESSAGE) {
240  xcb_client_message_event_t *cm_event = reinterpret_cast<xcb_client_message_event_t *>(event);
241 
242  if (cm_event->type != Private::manager_atom || cm_event->data.data32[ 1 ] != d->selection) {
243  return;
244  }
245  // owner() checks whether the owner changed and emits newOwner()
246  owner();
247  return;
248  }
249  if (response_type == XCB_DESTROY_NOTIFY) {
250  xcb_destroy_notify_event_t *ev = reinterpret_cast<xcb_destroy_notify_event_t *>(event);
251  if (d->selection_owner == XCB_NONE || ev->window != d->selection_owner) {
252  return;
253  }
254 
255  d->selection_owner = XCB_NONE; // in case the exactly same ID gets reused as the owner
256 
257  if (owner() == XCB_NONE) {
258  emit lostOwner(); // it must be safe to delete 'this' in a slot
259  }
260  return;
261  }
262 }
