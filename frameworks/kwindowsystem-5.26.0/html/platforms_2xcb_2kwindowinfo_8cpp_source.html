31 KWindowInfoPrivateX11::KWindowInfoPrivateX11(WId _win, NET::Properties properties, NET::Properties2 properties2)
32  : KWindowInfoPrivate(_win, properties, properties2)
33 {
34  KXErrorHandler handler;
35  if (properties & NET::WMVisibleIconName) {
36  properties |= NET::WMIconName | NET::WMVisibleName; // force, in case it will be used as a fallback
37  }
38  if (properties & NET::WMVisibleName) {
39  properties |= NET::WMName; // force, in case it will be used as a fallback
40  }
41  if (properties2 & NET::WM2ExtendedStrut) {
42  properties |= NET::WMStrut; // will be used as fallback
43  }
44  if (properties & NET::WMWindowType) {
45  properties2 |= NET::WM2TransientFor; // will be used when type is not set
46  }
47  if ((properties & NET::WMDesktop) && KWindowSystem::mapViewport()) {
48  properties |= NET::WMGeometry; // for viewports, the desktop (workspace) is determined from the geometry
49  }
50  properties |= NET::XAWMState; // force to get error detection for valid()
51  m_info.reset(new NETWinInfo(QX11Info::connection(), _win, QX11Info::appRootWindow(), properties, properties2));
52  if (properties & NET::WMName) {
53  if (m_info->name() && m_info->name()[ 0 ] != '\0') {
54  m_name = QString::fromUtf8(m_info->name());
55  } else {
56  m_name = KWindowSystem::readNameProperty(_win, XA_WM_NAME);
57  }
58  }
59  if (properties & NET::WMIconName) {
60  if (m_info->iconName() && m_info->iconName()[ 0 ] != '\0') {
61  m_iconic_name = QString::fromUtf8(m_info->iconName());
62  } else {
63  m_iconic_name = KWindowSystem::readNameProperty(_win, XA_WM_ICON_NAME);
64  }
65  }
66  if (properties & (NET::WMGeometry | NET::WMFrameExtents)) {
67  NETRect frame, geom;
68  m_info->kdeGeometry(frame, geom);
69  m_geometry.setRect(geom.pos.x, geom.pos.y, geom.size.width, geom.size.height);
70  m_frame_geometry.setRect(frame.pos.x, frame.pos.y, frame.size.width, frame.size.height);
71  }
72  m_valid = !handler.error(false); // no sync - NETWinInfo did roundtrips
73 }

110 NETExtendedStrut KWindowInfoPrivateX11::extendedStrut() const
111 {
112 #if !defined(KDE_NO_WARNING_OUTPUT)
113  if (!(m_info->passedProperties2() & NET::WM2ExtendedStrut)) {
114  qWarning() << "Pass NET::WM2ExtendedStrut to KWindowInfo";
115  }
116 #endif
117  NETExtendedStrut ext = m_info->extendedStrut();
118  NETStrut str = m_info->strut();
119  if (ext.left_width == 0 && ext.right_width == 0 && ext.top_width == 0 && ext.bottom_width == 0
120  && (str.left != 0 || str.right != 0 || str.top != 0 || str.bottom != 0)) {
121  // build extended from simple
122  if (str.left != 0) {
123  ext.left_width = str.left;
124  ext.left_start = 0;
125  ext.left_end = XDisplayHeight(QX11Info::display(), DefaultScreen(QX11Info::display()));
126  }
127  if (str.right != 0) {
128  ext.right_width = str.right;
129  ext.right_start = 0;
130  ext.right_end = XDisplayHeight(QX11Info::display(), DefaultScreen(QX11Info::display()));
131  }
132  if (str.top != 0) {
133  ext.top_width = str.top;
134  ext.top_start = 0;
135  ext.top_end = XDisplayWidth(QX11Info::display(), DefaultScreen(QX11Info::display()));
136  }
137  if (str.bottom != 0) {
138  ext.bottom_width = str.bottom;
139  ext.bottom_start = 0;
140  ext.bottom_end = XDisplayWidth(QX11Info::display(), DefaultScreen(QX11Info::display()));
141  }
142  }
143  return ext;
144 }

401 // see NETWM spec section 7.6
402 bool KWindowInfoPrivateX11::isMinimized() const
403 {
404  if (mappingState() != NET::Iconic) {
405  return false;
406  }
407  // NETWM 1.2 compliant WM - uses NET::Hidden for minimized windows
408  if ((state() & NET::Hidden) != 0
409  && (state() & NET::Shaded) == 0) { // shaded may have NET::Hidden too
410  return true;
411  }
412  // older WMs use WithdrawnState for other virtual desktops
413  // and IconicState only for minimized
414  return KWindowSystem::icccmCompliantMappingState() ? false : true;
415 }
