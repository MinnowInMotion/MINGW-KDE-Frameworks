31 template <typename T> T fromNative(xcb_pixmap_t pixmap, xcb_connection_t *c)
32 {
33  const xcb_get_geometry_cookie_t geoCookie = xcb_get_geometry_unchecked(c, pixmap);
34  ScopedCPointer<xcb_get_geometry_reply_t> geo(xcb_get_geometry_reply(c, geoCookie, Q_NULLPTR));
35  if (geo.isNull()) {
36  // getting geometry for the pixmap failed
37  return T();
38  }
39 
40  const xcb_get_image_cookie_t imageCookie = xcb_get_image_unchecked(c, XCB_IMAGE_FORMAT_Z_PIXMAP, pixmap,
41  0, 0, geo->width, geo->height, ~0);
42  ScopedCPointer<xcb_get_image_reply_t> xImage(xcb_get_image_reply(c, imageCookie, Q_NULLPTR));
43  if (xImage.isNull()) {
44  // request for image data failed
45  return T();
46  }
47  QImage::Format format = QImage::Format_Invalid;
48  switch (xImage->depth) {
49  case 1:
50  format = QImage::Format_MonoLSB;
51  break;
52  case 16:
53  format = QImage::Format_RGB16;
54  break;
55  case 24:
56  format = QImage::Format_RGB32;
57  break;
58  case 30: {
59  // Qt doesn't have a matching image format. We need to convert manually
60  uint32_t *pixels = reinterpret_cast<uint32_t *>(xcb_get_image_data(xImage.data()));
61  for (uint i = 0; i < xImage.data()->length; ++i) {
62  int r = (pixels[i] >> 22) & 0xff;
63  int g = (pixels[i] >> 12) & 0xff;
64  int b = (pixels[i] >> 2) & 0xff;
65 
66  pixels[i] = qRgba(r, g, b, 0xff);
67  }
68  // fall through, Qt format is still Format_ARGB32_Premultiplied
69  }
70  case 32:
71  format = QImage::Format_ARGB32_Premultiplied;
72  break;
73  default:
74  return T(); // we don't know
75  }
76  QImage image(xcb_get_image_data(xImage.data()), geo->width, geo->height,
77  xcb_get_image_data_length(xImage.data()) / geo->height, format, free, xImage.data());
78  xImage.take();
79  if (image.isNull()) {
80  return T();
81  }
82  if (image.format() == QImage::Format_MonoLSB) {
83  // work around an abort in QImage::color
84  image.setColorCount(2);
85  image.setColor(0, QColor(Qt::white).rgb());
86  image.setColor(1, QColor(Qt::black).rgb());
87  }
88  return T::fromImage(image);
89 }

97 QPixmap createPixmapFromHandle(xcb_connection_t *c, WId pixmap, WId pixmap_mask)
98 {
99 #if Q_BYTE_ORDER == Q_BIG_ENDIAN
100  qDebug() << "Byte order not supported";
101  return QPixmap();
102 #endif
103  const xcb_setup_t *setup = xcb_get_setup(c);
104  if (setup->image_byte_order != XCB_IMAGE_ORDER_LSB_FIRST) {
105  qDebug() << "Byte order not supported";
106  return QPixmap();
107  }
108 
109  QPixmap pix = fromNative<QPixmap>(pixmap, c);
110  if (pixmap_mask != XCB_PIXMAP_NONE) {
111  QBitmap mask = fromNative<QBitmap>(pixmap_mask, c);
112  if (mask.size() != pix.size()) {
113  return QPixmap();
114  }
115  pix.setMask(mask);
116  }
117  return pix;
118 }
