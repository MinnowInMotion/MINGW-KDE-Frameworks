471 bool initializeMods()
472 {
473  // Reinitialize the masks
474  g_modXNumLock = 0;
475  g_modXScrollLock = 0;
476  g_modXModeSwitch = 0;
477  g_alt_mask = 0;
478  g_meta_mask = 0;
479  g_super_mask = 0;
480  g_hyper_mask = 0;
481 
482  if (!QX11Info::isPlatformX11()) {
483  qCWarning(LOG_KKEYSERVER_X11) << "X11 implementation of KKeyServer accessed from non-X11 platform! This is an application bug.";
484  g_bInitializedMods = true;
485  return false;
486  }
487 
488  checkDisplay();
489  XModifierKeymap *xmk = XGetModifierMapping(QX11Info::display());
490 
491  int min_keycode, max_keycode;
492  int keysyms_per_keycode = 0;
493 
494  XDisplayKeycodes(QX11Info::display(), &min_keycode, &max_keycode);
495  XFree(XGetKeyboardMapping(QX11Info::display(), min_keycode, 1, &keysyms_per_keycode));
496 
497  for (int i = Mod1MapIndex; i < 8; i++) {
498  uint mask = (1 << i);
499  uint keySymX = NoSymbol;
500 
501  // This used to be only XKeycodeToKeysym( ... , 0 ), but that fails with XFree4.3.99
502  // and X.org R6.7 , where for some reason only ( ... , 1 ) works. I have absolutely no
503  // idea what the problem is, but searching all possibilities until something valid is
504  // found fixes the problem.
505  for (int j = 0; j < xmk->max_keypermod; ++j) {
506 
507  for (int k = 0; k < keysyms_per_keycode; ++k) {
508 
509  keySymX = XKeycodeToKeysym(QX11Info::display(), xmk->modifiermap[xmk->max_keypermod * i + j], k);
510 
511  switch (keySymX) {
512  case XK_Alt_L:
513  case XK_Alt_R: g_alt_mask |= mask; break;
514 
515  case XK_Super_L:
516  case XK_Super_R: g_super_mask |= mask; break;
517 
518  case XK_Hyper_L:
519  case XK_Hyper_R: g_hyper_mask |= mask; break;
520 
521  case XK_Meta_L:
522  case XK_Meta_R: g_meta_mask |= mask; break;
523 
524  case XK_Num_Lock: g_modXNumLock |= mask; break;
525  case XK_Scroll_Lock: g_modXScrollLock |= mask; break;
526  case XK_Mode_switch: g_modXModeSwitch |= mask; break;
527  }
528  }
529  }
530  }
531 
532 #ifdef KKEYSERVER_DEBUG
533  qCDebug(LOG_KKEYSERVER_X11) << "Alt:" << g_alt_mask;
534  qCDebug(LOG_KKEYSERVER_X11) << "Meta:" << g_meta_mask;
535  qCDebug(LOG_KKEYSERVER_X11) << "Super:" << g_super_mask;
536  qCDebug(LOG_KKEYSERVER_X11) << "Hyper:" << g_hyper_mask;
537  qCDebug(LOG_KKEYSERVER_X11) << "NumLock:" << g_modXNumLock;
538  qCDebug(LOG_KKEYSERVER_X11) << "ScrollLock:" << g_modXScrollLock;
539  qCDebug(LOG_KKEYSERVER_X11) << "ModeSwitch:" << g_modXModeSwitch;
540 #endif
541 
542  // Check if hyper overlaps with super or meta or alt
543  if (g_hyper_mask & (g_super_mask | g_meta_mask | g_alt_mask)) {
544 #ifdef KKEYSERVER_DEBUG
545  qCDebug(LOG_KKEYSERVER_X11) << "Hyper conflicts with super, meta or alt.";
546 #endif
547  // Remove the conflicting masks
548  g_hyper_mask &= ~(g_super_mask | g_meta_mask | g_alt_mask);
549  }
550 
551  // Check if super overlaps with meta or alt
552  if (g_super_mask & (g_meta_mask | g_alt_mask)) {
553 #ifdef KKEYSERVER_DEBUG
554  qCDebug(LOG_KKEYSERVER_X11) << "Super conflicts with meta or alt.";
555 #endif
556  // Remove the conflicting masks
557  g_super_mask &= ~(g_meta_mask | g_alt_mask);
558  }
559 
560  // Check if meta overlaps with alt
561  if (g_meta_mask | g_alt_mask) {
562 #ifdef KKEYSERVER_DEBUG
563  qCDebug(LOG_KKEYSERVER_X11) << "Meta conflicts with alt.";
564 #endif
565  // Remove the conflicting masks
566  g_meta_mask &= ~(g_alt_mask);
567  }
568 
569  if (!g_meta_mask) {
570 #ifdef KKEYSERVER_DEBUG
571  qCDebug(LOG_KKEYSERVER_X11) << "Meta is not set or conflicted with alt.";
572 #endif
573  if (g_super_mask) {
574 #ifdef KKEYSERVER_DEBUG
575  qCDebug(LOG_KKEYSERVER_X11) << "Using super for meta";
576 #endif
577  // Use Super
578  g_meta_mask = g_super_mask;
579  } else if (g_hyper_mask) {
580 #ifdef KKEYSERVER_DEBUG
581  qCDebug(LOG_KKEYSERVER_X11) << "Using hyper for meta";
582 #endif
583  // User Hyper
584  g_meta_mask = g_hyper_mask;
585  } else {
586  // ???? Nothing left
587  g_meta_mask = 0;
588  }
589  }
590 
591 #ifdef KKEYSERVER_DEBUG
592  qCDebug(LOG_KKEYSERVER_X11) << "Alt:" << g_alt_mask;
593  qCDebug(LOG_KKEYSERVER_X11) << "Meta:" << g_meta_mask;
594  qCDebug(LOG_KKEYSERVER_X11) << "Super:" << g_super_mask;
595  qCDebug(LOG_KKEYSERVER_X11) << "Hyper:" << g_hyper_mask;
596  qCDebug(LOG_KKEYSERVER_X11) << "NumLock:" << g_modXNumLock;
597  qCDebug(LOG_KKEYSERVER_X11) << "ScrollLock:" << g_modXScrollLock;
598  qCDebug(LOG_KKEYSERVER_X11) << "ModeSwitch:" << g_modXModeSwitch;
599 #endif
600 
601  if (!g_meta_mask) {
602  qCWarning(LOG_KKEYSERVER_X11) << "Your keyboard setup doesn't provide a key to use for meta. See 'xmodmap -pm' or 'xkbcomp $DISPLAY'";
603  }
604 
605  g_rgX11ModInfo[2].modX = g_alt_mask;
606  g_rgX11ModInfo[3].modX = g_meta_mask;
607 
608  XFreeModifiermap(xmk);
609  g_bInitializedMods = true;
610 
611  return true;

667 uint getModsRequired(uint sym)
668 {
669  if (!QX11Info::isPlatformX11()) {
670  qCWarning(LOG_KKEYSERVER_X11) << "X11 implementation of KKeyServer accessed from non-X11 platform! This is an application bug.";
671  return 0;
672  }
673  uint mod = 0;
674 
675  // FIXME: This might not be true on all keyboard layouts!
676  if (sym == XK_Sys_Req) {
677  return Qt::ALT;
678  }
679  if (sym == XK_Break) {
680  return Qt::CTRL;
681  }
682 
683  if (sym < 0x3000) {
684  QChar c(sym);
685  if (c.isLetter() && c.toLower() != c.toUpper() && sym == c.toUpper().unicode()) {
686  return Qt::SHIFT;
687  }
688  }
689 
690  uchar code = XKeysymToKeycode(QX11Info::display(), sym);
691  if (code) {
692  // need to check index 0 before the others, so that a null-mod
693  // can take precedence over the others, in case the modified
694  // key produces the same symbol.
695  if (sym == XKeycodeToKeysym(QX11Info::display(), code, 0))
696  ;
697  else if (sym == XKeycodeToKeysym(QX11Info::display(), code, 1)) {
698  mod = Qt::SHIFT;
699  } else if (sym == XKeycodeToKeysym(QX11Info::display(), code, 2)) {
700  mod = MODE_SWITCH;
701  } else if (sym == XKeycodeToKeysym(QX11Info::display(), code, 3)) {
702  mod = Qt::SHIFT | MODE_SWITCH;
703  }
704  }
705  return mod;
706 }

848 bool xEventToQt(XEvent *e, int *keyQt)
849 {
850  Q_ASSERT(e->type == KeyPress || e->type == KeyRelease);
851 
852  uchar keyCodeX = e->xkey.keycode;
853  uint keyModX = e->xkey.state & (accelModMaskX() | MODE_SWITCH);
854 
855  KeySym keySym;
856  char buffer[16];
857  XLookupString((XKeyEvent *) e, buffer, 15, &keySym, 0);
858  uint keySymX = (uint)keySym;
859 
860  // If numlock is active and a keypad key is pressed, XOR the SHIFT state.
861  // e.g., KP_4 => Shift+KP_Left, and Shift+KP_4 => KP_Left.
862  if (e->xkey.state & modXNumLock()) {
863  uint sym = XKeycodeToKeysym(QX11Info::display(), keyCodeX, 0);
864  // TODO: what's the xor operator in c++?
865  // If this is a keypad key,
866  if (sym >= XK_KP_Space && sym <= XK_KP_9) {
867  switch (sym) {
868  // Leave the following keys unaltered
869  // FIXME: The proper solution is to see which keysyms don't change when shifted.
870  case XK_KP_Multiply:
871  case XK_KP_Add:
872  case XK_KP_Subtract:
873  case XK_KP_Divide:
874  break;
875  default:
876  if (keyModX & modXShift()) {
877  keyModX &= ~modXShift();
878  } else {
879  keyModX |= modXShift();
880  }
881  }
882  }
883  }
884 
885  int keyCodeQt;
886  int keyModQt;
887  symXToKeyQt(keySymX, &keyCodeQt);
888  modXToQt(keyModX, &keyModQt);
889 
890  *keyQt = keyCodeQt | keyModQt;
891  return true;
892 }
