198 bool keyQtToModWin(int keyQt, uint *mod)
199 {
200  *mod = 0;
201  if (keyQt & Qt::ShiftModifier) {
202  *mod |= MOD_SHIFT;
203  }
204  if (keyQt & Qt::ControlModifier) {
205  *mod |= MOD_CONTROL;
206  }
207  if (keyQt & Qt::AltModifier) {
208  *mod |= MOD_ALT;
209  }
210  if (keyQt & Qt::MetaModifier) {
211  *mod |= MOD_WIN;
212  }
213 
214  return true;
215 }

217 bool modWinToKeyQt(uint mod, int *keyQt)
218 {
219  *keyQt = 0;
220  if (mod & MOD_SHIFT) {
221  *keyQt |= Qt::ShiftModifier;
222  }
223  if (mod & MOD_CONTROL) {
224  *keyQt |= Qt::ControlModifier;
225  }
226  if (mod & MOD_ALT) {
227  *keyQt |= Qt::AltModifier;
228  }
229  if (mod & MOD_WIN) {
230  *keyQt |= Qt::MetaModifier;
231  }
232 
233  return true;
234 }

236 bool keyQtToCodeWin(int keyQt, uint *sym)
237 {
238  int symQt = keyQt & ~Qt::KeyboardModifierMask;
239 
240  if (symQt >= Qt::Key_A && symQt <= Qt::Key_Z) {
241  *sym = symQt;
242  return true;
243  }
244 
245  for (uint i = 0; i < sizeof(KeyTbl); i++) {
246  if (KeyTbl[i].keySymQt == symQt) {
247  *sym = KeyTbl[i].keySymWin;
248  return true;
249  }
250  }
251 
252  *sym = 0;
253  if (symQt != Qt::Key_Shift && symQt != Qt::Key_Control && symQt != Qt::Key_Alt &&
254  symQt != Qt::Key_Meta && symQt != Qt::Key_Direction_L && symQt != Qt::Key_Direction_R) {
255  // qDebug() << "Sym::initQt( " << QString::number(keyQt,16) << " ): failed to convert key.";
256  }
257  return false;
258 }

260 bool codeWinToKeyQt(uint sym, int *keyQt)
261 {
262  if (sym >= Qt::Key_A && sym <= Qt::Key_Z) {
263  *keyQt = sym;
264  return true;
265  }
266 
267  for (uint i = 0; i < sizeof(KeyTbl); i++) {
268  if (KeyTbl[i].keySymWin == sym) {
269  *keyQt = KeyTbl[i].keySymQt;
270  return true;
271  }
272  }
273 
274  *keyQt = 0;
275  return false;
276 }
