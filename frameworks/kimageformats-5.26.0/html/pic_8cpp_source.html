50 static QDataStream &operator>> (QDataStream &s, PicHeader &header)
51 {
52  s.setFloatingPointPrecision(QDataStream::SinglePrecision);
53  s >> header.magic;
54  s >> header.version;
55 
56  // the comment should be truncated to the first null byte
57  char comment[81] = {};
58  s.readRawData(comment, 80);
59  header.comment = QByteArray(comment);
60 
61  header.id.resize(4);
62  s.readRawData(header.id.data(), 4);
63 
64  s >> header.width;
65  s >> header.height;
66  s >> header.ratio;
67  qint16 fields;
68  s >> fields;
69  header.fields = static_cast<PicFields>(fields);
70  qint16 pad;
71  s >> pad;
72  return s;
73 }

84 static QDataStream &operator<< (QDataStream &s, const PicHeader &header)
85 {
86  s.setFloatingPointPrecision(QDataStream::SinglePrecision);
87  s << header.magic;
88  s << header.version;
89 
90  char comment[80] = {};
91  strncpy(comment, header.comment.constData(), sizeof(comment));
92  s.writeRawData(comment, sizeof(comment));
93 
94  char id[4] = {};
95  strncpy(id, header.id.constData(), sizeof(id));
96  s.writeRawData(id, sizeof(id));
97 
98  s << header.width;
99  s << header.height;
100  s << header.ratio;
101  s << quint16(header.fields);
102  s << quint16(0);
103  return s;
104 }

120 static QDataStream &operator>> (QDataStream &s, QList<PicChannel> &channels)
121 {
122  const unsigned maxChannels = 8;
123  unsigned count = 0;
124  quint8 chained = 1;
125  channels.clear();
126  while (chained && count < maxChannels && s.status() == QDataStream::Ok) {
127  PicChannel channel;
128  s >> chained;
129  s >> channel.size;
130  quint8 encoding;
131  s >> encoding;
132  channel.encoding = PicChannelEncoding(encoding);
133  s >> channel.code;
134  channels << channel;
135  ++count;
136  }
137  if (chained) {
138  // too many channels!
139  s.setStatus(QDataStream::ReadCorruptData);
140  }
141  return s;
142 }

157 static QDataStream &operator<< (QDataStream &s, const QList<PicChannel> &channels)

173 static bool readRow(QDataStream &stream, QRgb *row, quint16 width, QList<PicChannel> channels)
174 {
175  Q_FOREACH(const PicChannel &channel, channels) {
176  auto readPixel = [&] (QDataStream &str) -> QRgb {
177  quint8 red = 0;
178  if (channel.code & RED) {
179  str >> red;
180  }
181  quint8 green = 0;
182  if (channel.code & GREEN) {
183  str >> green;
184  }
185  quint8 blue = 0;
186  if (channel.code & BLUE) {
187  str >> blue;
188  }
189  quint8 alpha = 0;
190  if (channel.code & ALPHA) {
191  str >> alpha;
192  }
193  return qRgba(red, green, blue, alpha);
194  };
195  auto updatePixel = [&] (QRgb oldPixel, QRgb newPixel) -> QRgb {
196  return qRgba(
197  qRed((channel.code & RED) ? newPixel : oldPixel),
198  qGreen((channel.code & GREEN) ? newPixel : oldPixel),
199  qBlue((channel.code & BLUE) ? newPixel : oldPixel),
200  qAlpha((channel.code & ALPHA) ? newPixel : oldPixel));
201  };
202  if (channel.encoding == MixedRLE) {
203  bool success = decodeRLEData(RLEVariant::PIC, stream, row, width,
204  readPixel, updatePixel);
205  if (!success) {
206  qDebug() << "decodeRLEData failed";
207  return false;
208  }
209  } else if (channel.encoding == Uncompressed) {
210  for (quint16 i = 0; i < width; ++i) {
211  QRgb pixel = readPixel(stream);
212  row[i] = updatePixel(row[i], pixel);
213  }
214  } else {
215  // unknown encoding
216  qDebug() << "Unknown encoding";
217  return false;
218  }
219  }
220  if (stream.status() != QDataStream::Ok) {
221  qDebug() << "DataStream status was" << stream.status();
222  }
223  return stream.status() == QDataStream::Ok;
224 }

235 bool SoftimagePICHandler::read(QImage *image)
236 {
237  if (!readChannels()) {
238  return false;
239  }
240 
241  QImage::Format fmt = QImage::Format_RGB32;
242  Q_FOREACH(const PicChannel &channel, m_channels) {
243  if (channel.size != 8) {
244  // we cannot read images that do not come in bytes
245  qDebug() << "Channel size was" << channel.size;
246  m_state = Error;
247  return false;
248  }
249  if (channel.code & ALPHA) {
250  fmt = QImage::Format_ARGB32;
251  }
252  }
253 
254  QImage img(m_header.width, m_header.height, fmt);
255  img.fill(qRgb(0,0,0));
256 
257  for (int y = 0; y < m_header.height; y++) {
258  QRgb *row = reinterpret_cast<QRgb*>(img.scanLine(y));
259  if (!readRow(m_dataStream, row, m_header.width, m_channels)) {
260  qDebug() << "readRow failed";
261  m_state = Error;
262  return false;
263  }
264  }
265 
266  *image = img;
267  m_state = Ready;
268 
269  return true;
270 }

272 bool SoftimagePICHandler::write(const QImage &_image)
273 {
274  bool alpha = _image.hasAlphaChannel();
275  const QImage image = _image.convertToFormat(
276  alpha ? QImage::Format_ARGB32
277  : QImage::Format_RGB32);
278 
279  if (image.width() < 0 || image.height() < 0) {
280  qDebug() << "Image size invalid:" << image.width() << image.height();
281  return false;
282  }
283  if (image.width() > 65535 || image.height() > 65535) {
284  qDebug() << "Image too big:" << image.width() << image.height();
285  // there are only two bytes for each dimension
286  return false;
287  }
288 
289  QDataStream stream(device());
290 
291  stream << PicHeader(image.width(), image.height(), m_description);
292 
293  PicChannelEncoding encoding = m_compression ? MixedRLE : Uncompressed;
294  QList<PicChannel> channels;
295  channels << PicChannel(encoding, RED | GREEN | BLUE);
296  if (alpha) {
297  channels << PicChannel(encoding, ALPHA);
298  }
299  stream << channels;
300 
301  for (int r = 0; r < image.height(); r++) {
302  const QRgb *row = reinterpret_cast<const QRgb*>(image.scanLine(r));
303 
304  /* Write the RGB part of the scanline */
305  auto rgbEqual = [] (QRgb p1, QRgb p2) -> bool {
306  return qRed(p1) == qRed(p2) &&
307  qGreen(p1) == qGreen(p2) &&
308  qBlue(p1) == qBlue(p2);
309  };
310  auto writeRgb = [] (QDataStream &str, QRgb pixel) -> void {
311  str << quint8(qRed(pixel))
312  << quint8(qGreen(pixel))
313  << quint8(qBlue(pixel));
314  };
315  if (m_compression) {
316  encodeRLEData(RLEVariant::PIC, stream, row, image.width(),
317  rgbEqual, writeRgb);
318  } else {
319  for (int i = 0; i < image.width(); ++i) {
320  writeRgb(stream, row[i]);
321  }
322  }
323 
324  /* Write the alpha channel */
325  if (alpha) {
326  auto alphaEqual = [] (QRgb p1, QRgb p2) -> bool {
327  return qAlpha(p1) == qAlpha(p2);
328  };
329  auto writeAlpha = [] (QDataStream &str, QRgb pixel) -> void {
330  str << quint8(qAlpha(pixel));
331  };
332  if (m_compression) {
333  encodeRLEData(RLEVariant::PIC, stream, row, image.width(),
334  alphaEqual, writeAlpha);
335  } else {
336  for (int i = 0; i < image.width(); ++i) {
337  writeAlpha(stream, row[i]);
338  }
339  }
340  }
341  }
342  return stream.status() == QDataStream::Ok;
343 }

380 void SoftimagePICHandler::setOption(ImageOption option, const QVariant &value)
381 {
382  switch (option) {
383  case CompressionRatio:
384  m_compression = value.toBool();
385  break;
386  case Description: {
387  m_description.clear();
388  QStringList entries = value.toString().split(QStringLiteral("\n\n"));
389  Q_FOREACH(const QString entry, entries) {
390  if (entry.startsWith(QStringLiteral("Description: "))) {
391  m_description = entry.mid(13).simplified().toUtf8();
392  }
393  }
394  break;
395  }
396  default:
397  break;
398  }
399 }

401 QVariant SoftimagePICHandler::option(ImageOption option) const
402 {
403  const_cast<SoftimagePICHandler*>(this)->readHeader();
404  switch (option) {
405  case Size:
406  if (const_cast<SoftimagePICHandler*>(this)->readHeader()) {
407  return QSize(m_header.width, m_header.height);
408  } else {
409  return QVariant();
410  }
411  case CompressionRatio:
412  return m_compression;
413  case Description:
414  if (const_cast<SoftimagePICHandler*>(this)->readHeader()) {
415  QString descStr = QString::fromUtf8(m_header.comment);
416  if (!descStr.isEmpty()) {
417  return QString(QStringLiteral("Description: ") +
418  descStr +
419  QStringLiteral("\n\n"));
420  }
421  }
422  return QString();
423  case ImageFormat:
424  if (const_cast<SoftimagePICHandler*>(this)->readChannels()) {
425  Q_FOREACH (const PicChannel &channel, m_channels) {
426  if (channel.code & ALPHA) {
427  return QImage::Format_ARGB32;
428  }
429  }
430  return QImage::Format_RGB32;
431  }
432  return QVariant();
433  default:
434  return QVariant();
435  }
436 }

446 QImageIOPlugin::Capabilities SoftimagePICPlugin::capabilities(QIODevice *device, const QByteArray &format) const
447 {
448  if (format == "pic") {
449  return Capabilities(CanRead | CanWrite);
450  }
451  if (!format.isEmpty()) {
452  return 0;
453  }
454  if (!device->isOpen()) {
455  return 0;
456  }
457 
458  Capabilities cap;
459  if (device->isReadable() && SoftimagePICHandler::canRead(device)) {
460  cap |= CanRead;
461  }
462  if (device->isWritable()) {
463  cap |= CanWrite;
