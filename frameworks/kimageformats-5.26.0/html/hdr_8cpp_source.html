27 static inline uchar ClipToByte(float value)
28 {
29  if (value > 255.0f) {
30  return 255;
31  }
32  //else if (value < 0.0f) return 0; // we know value is positive.
33  return uchar(value);
34 }

36 // read an old style line from the hdr image file
37 // if 'first' is true the first byte is already read
38 static bool Read_Old_Line(uchar *image, int width, QDataStream &s)
39 {
40  int rshift = 0;
41  int i;
42 
43  while (width > 0) {
44  s >> image[0];
45  s >> image[1];
46  s >> image[2];
47  s >> image[3];
48 
49  if (s.atEnd()) {
50  return false;
51  }
52 
53  if ((image[0] == 1) && (image[1] == 1) && (image[2] == 1)) {
54  for (i = image[3] << rshift; i > 0; i--) {
55  //memcpy(image, image-4, 4);
56  (uint &)image[0] = (uint &)image[0 - 4];
57  image += 4;
58  width--;
59  }
60  rshift += 8;
61  } else {
62  image += 4;
63  width--;
64  rshift = 0;
65  }
66  }
67  return true;
68 }

70 static void RGBE_To_QRgbLine(uchar *image, QRgb *scanline, int width)
71 {
72  for (int j = 0; j < width; j++) {
73  // v = ldexp(1.0, int(image[3]) - 128);
74  float v;
75  int e = int(image[3]) - 128;
76  if (e > 0) {
77  v = float(1 << e);
78  } else {
79  v = 1.0f / float(1 << -e);
80  }
81 
82  scanline[j] = qRgb(ClipToByte(float(image[0]) * v),
83  ClipToByte(float(image[1]) * v),
84  ClipToByte(float(image[2]) * v));
85 
86  image += 4;
87  }
88 }

90 // Load the HDR image.
91 static bool LoadHDR(QDataStream &s, const int width, const int height, QImage &img)
92 {
93  uchar val, code;
94 
95  // Create dst image.
96  if (!img.create(width, height, 32)) {
97  return false;
98  }
99 
100  QMemArray<uchar> image(width * 4);
101 
102  for (int cline = 0; cline < height; cline++) {
103  QRgb *scanline = (QRgb *) img.scanLine(cline);
104 
105  // determine scanline type
106  if ((width < MINELEN) || (MAXELEN < width)) {
107  Read_Old_Line(image.data(), width, s);
108  RGBE_To_QRgbLine(image.data(), scanline, width);
109  continue;
110  }
111 
112  s >> val;
113 
114  if (s.atEnd()) {
115  return true;
116  }
117 
118  if (val != 2) {
119  s.device()->at(s.device()->at() - 1);
120  Read_Old_Line(image.data(), width, s);
121  RGBE_To_QRgbLine(image.data(), scanline, width);
122  continue;
123  }
124 
125  s >> image[1];
126  s >> image[2];
127  s >> image[3];
128 
129  if (s.atEnd()) {
130  return true;
131  }
132 
133  if ((image[1] != 2) || (image[2] & 128)) {
134  image[0] = 2;
135  Read_Old_Line(image.data() + 4, width - 1, s);
136  RGBE_To_QRgbLine(image.data(), scanline, width);
137  continue;
138  }
139 
140  if ((image[2] << 8 | image[3]) != width) {
141  return false;
142  }
143 
144  // read each component
145  for (int i = 0; i < 4; i++) {
146  for (int j = 0; j < width;) {
147  s >> code;
148  if (s.atEnd()) {
149  return false;
150  }
151  if (code > 128) {
152  // run
153  code &= 127;
154  s >> val;
155  while (code != 0) {
156  image[i + j * 4] = val;
157  j++;
158  code--;
159  }
160  } else {
161  // non-run
162  while (code != 0) {
163  s >> image[i + j * 4];
164  j++;
165  code--;
166  }
167  }
168  }
169  }
170 
171  RGBE_To_QRgbLine(image.data(), scanline, width);
172  }
173 
174  return true;
175 }

179 Q_DECL_EXPORT void kimgio_hdr_read(QImageIO *io)
180 {
181  int len;
182  char line[MAXLINE];
183  //bool validHeader = false;
184  bool validFormat = false;
185 
186  // Parse header
187  do {
188  len = io->ioDevice()->readLine(line, MAXLINE);
189 
190  /*if (strcmp(line, "#?RADIANCE\n") == 0 || strcmp(line, "#?RGBE\n") == 0)
191  {
192  validHeader = true;
193  }*/
194  if (strcmp(line, "FORMAT=32-bit_rle_rgbe\n") == 0) {
195  validFormat = true;
196  }
197 
198  } while ((len > 0) && (line[0] != '\n'));
199 
200  if ( !validFormat) {
201  // qDebug() << "Unknown HDR format.";
202  io->setImage(0);
203  io->setStatus(-1);
204  return;
205  }
206 
207  io->ioDevice()->readLine(line, MAXLINE);
208 
209  char s1[3], s2[3];
210  int width, height;
211  if (sscanf(line, "%2[+-XY] %d %2[+-XY] %d\n", s1, &height, s2, &width) != 4)
212  //if( sscanf(line, "-Y %d +X %d", &height, &width) < 2 )
213  {
214  // qDebug() << "Invalid HDR file.";
215  io->setImage(0);
216  io->setStatus(-1);
217  return;
218  }
219 
220  QDataStream s(io->ioDevice());
221 
222  QImage img;
223  if (!LoadHDR(s, width, height, img)) {
224  // qDebug() << "Error loading HDR file.";
225  io->setImage(0);
226  io->setStatus(-1);
227  return;
228  }
229 
230  io->setImage(img);
231  io->setStatus(0);
232 }
