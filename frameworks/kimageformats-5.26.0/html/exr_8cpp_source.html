81 /* this does a conversion from the ILM Half (equal to Nvidia Half)
82  * format into the normal 32 bit pixel format. Process is from the
83  * ILM code.
84  */
85 QRgb RgbaToQrgba(struct Imf::Rgba imagePixel)
86 {
87  float r, g, b, a;
88 
89  // 1) Compensate for fogging by subtracting defog
90  // from the raw pixel values.
91  // Response: We work with defog of 0.0, so this is a no-op
92 
93  // 2) Multiply the defogged pixel values by
94  // 2^(exposure + 2.47393).
95  // Response: We work with exposure of 0.0.
96  // (2^2.47393) is 5.55555
97  r = imagePixel.r * 5.55555;
98  g = imagePixel.g * 5.55555;
99  b = imagePixel.b * 5.55555;
100  a = imagePixel.a * 5.55555;
101 
102  // 3) Values, which are now 1.0, are called "middle gray".
103  // If defog and exposure are both set to 0.0, then
104  // middle gray corresponds to a raw pixel value of 0.18.
105  // In step 6, middle gray values will be mapped to an
106  // intensity 3.5 f-stops below the display's maximum
107  // intensity.
108  // Response: no apparent content.
109 
110  // 4) Apply a knee function. The knee function has two
111  // parameters, kneeLow and kneeHigh. Pixel values
112  // below 2^kneeLow are not changed by the knee
113  // function. Pixel values above kneeLow are lowered
114  // according to a logarithmic curve, such that the
115  // value 2^kneeHigh is mapped to 2^3.5 (in step 6,
116  // this value will be mapped to the display's
117  // maximum intensity).
118  // Response: kneeLow = 0.0 (2^0.0 => 1); kneeHigh = 5.0 (2^5 =>32)
119  if (r > 1.0) {
120  r = 1.0 + Imath::Math<float>::log((r - 1.0) * 0.184874 + 1) / 0.184874;
121  }
122  if (g > 1.0) {
123  g = 1.0 + Imath::Math<float>::log((g - 1.0) * 0.184874 + 1) / 0.184874;
124  }
125  if (b > 1.0) {
126  b = 1.0 + Imath::Math<float>::log((b - 1.0) * 0.184874 + 1) / 0.184874;
127  }
128  if (a > 1.0) {
129  a = 1.0 + Imath::Math<float>::log((a - 1.0) * 0.184874 + 1) / 0.184874;
130  }
131 //
132 // 5) Gamma-correct the pixel values, assuming that the
133 // screen's gamma is 0.4545 (or 1/2.2).
134  r = Imath::Math<float>::pow(r, 0.4545);
135  g = Imath::Math<float>::pow(g, 0.4545);
136  b = Imath::Math<float>::pow(b, 0.4545);
137  a = Imath::Math<float>::pow(a, 0.4545);
138 
139 // 6) Scale the values such that pixels middle gray
140 // pixels are mapped to 84.66 (or 3.5 f-stops below
141 // the display's maximum intensity).
142 //
143 // 7) Clamp the values to [0, 255].
144  return qRgba((unsigned char)(Imath::clamp(r * 84.66f, 0.f, 255.f)),
145  (unsigned char)(Imath::clamp(g * 84.66f, 0.f, 255.f)),
146  (unsigned char)(Imath::clamp(b * 84.66f, 0.f, 255.f)),
147  (unsigned char)(Imath::clamp(a * 84.66f, 0.f, 255.f)));
148 }

163 bool EXRHandler::read(QImage *outImage)
164 {
165  try {
166  int width, height;
167 
168  K_IStream istr(device(), QByteArray());
169  Imf::RgbaInputFile file(istr);
170  Imath::Box2i dw = file.dataWindow();
171 
172  width = dw.max.x - dw.min.x + 1;
173  height = dw.max.y - dw.min.y + 1;
174 
175  Imf::Array2D<Imf::Rgba> pixels;
176  pixels.resizeErase(height, width);
177 
178  file.setFrameBuffer(&pixels[0][0] - dw.min.x - dw.min.y * width, 1, width);
179  file.readPixels(dw.min.y, dw.max.y);
180 
181  QImage image(width, height, QImage::Format_RGB32);
182  if (image.isNull()) {
183  return false;
184  }
185 
186  // somehow copy pixels into image
187  for (int y = 0; y < height; y++) {
188  for (int x = 0; x < width; x++) {
189  // copy pixels(x,y) into image(x,y)
190  image.setPixel(x, y, RgbaToQrgba(pixels[y][x]));
191  }
192  }
193 
194  *outImage = image;
195 
196  return true;
197  } catch (const std::exception &exc) {
198 // qDebug() << exc.what();
199  return false;
200  }
201 }

215 QImageIOPlugin::Capabilities EXRPlugin::capabilities(QIODevice *device, const QByteArray &format) const
216 {
217  if (format == "exr") {
218  return Capabilities(CanRead);
219  }
220  if (!format.isEmpty()) {
221  return 0;
222  }
223  if (!device->isOpen()) {
224  return 0;
225  }
226 
227  Capabilities cap;
