33 bool OraHandler::read(QImage *image)
34 {
35  KZip zip(device());
36  if (!zip.open(QIODevice::ReadOnly)) return false;
37 
38  const KArchiveEntry *entry = zip.directory()->entry(QLatin1String("mergedimage.png"));
39  if (!entry || !entry->isFile()) return false;
40 
41  const KZipFileEntry* fileZipEntry = static_cast<const KZipFileEntry*>(entry);
42 
43  image->loadFromData(fileZipEntry->data(), "PNG");
44 
45  return true;
46 }

48 bool OraHandler::canRead(QIODevice *device)
49 {
50  if (!device) {
51  qWarning("OraHandler::canRead() called with no device");
52  return false;
53  }
54 
55  char buff[54];
56  if (device->peek(buff, sizeof(buff)) == sizeof(buff))
57  return qstrcmp(buff + 0x26, "image/openraster") == 0;
58 
59  return false;
60 }

62 QImageIOPlugin::Capabilities OraPlugin::capabilities(QIODevice *device, const QByteArray &format) const
63 {
64  if (format == "ora" || format == "ORA") {
65  return Capabilities(CanRead);
66  }
67  if (!format.isEmpty()) {
68  return 0;
69  }
70  if (!device->isOpen()) {
71  return 0;
72  }
73 
74  Capabilities cap;
75  if (device->isReadable() && OraHandler::canRead(device)) {
76  cap |= CanRead;
77  }
78  return cap;
79 }
