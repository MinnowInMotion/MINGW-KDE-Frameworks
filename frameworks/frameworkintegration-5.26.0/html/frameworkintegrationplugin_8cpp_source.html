33 bool KMessageBoxDontAskAgainConfigStorage::shouldBeShownYesNo(const QString &dontShowAgainName,
34  KMessageBox::ButtonCode &result)
35 {
36  KConfigGroup cg(KMessageBox_againConfig ? KMessageBox_againConfig : KSharedConfig::openConfig().data(), "Notification Messages");
37  const QString dontAsk = cg.readEntry(dontShowAgainName, QString()).toLower();
38  if (dontAsk == QLatin1String("yes") || dontAsk == QLatin1String("true")) {
39  result = KMessageBox::Yes;
40  return false;
41  }
42  if (dontAsk == QLatin1String("no") || dontAsk == QLatin1String("false")) {
43  result = KMessageBox::No;
44  return false;
45  }
46  return true;
47 }

55 void KMessageBoxDontAskAgainConfigStorage::saveDontShowAgainYesNo(const QString &dontShowAgainName,
56  KMessageBox::ButtonCode result)
57 {
58  KConfigGroup::WriteConfigFlags flags = KConfig::Persistent;
59  if (dontShowAgainName[0] == QLatin1Char(':')) {
60  flags |= KConfigGroup::Global;
61  }
62  KConfigGroup cg(KMessageBox_againConfig ? KMessageBox_againConfig : KSharedConfig::openConfig().data(), "Notification Messages");
63  cg.writeEntry(dontShowAgainName, result == KMessageBox::Yes, flags);
64  cg.sync();
65 }

78 void KMessageBoxDontAskAgainConfigStorage::enableAllMessages()
79 {
80  KConfig *config = KMessageBox_againConfig ? KMessageBox_againConfig : KSharedConfig::openConfig().data();
81  if (!config->hasGroup("Notification Messages")) {
82  return;
83  }
84 
85  KConfigGroup cg(config, "Notification Messages");
86 
87  typedef QMap<QString, QString> configMap;
88 
89  const configMap map = cg.entryMap();
90 
91  configMap::ConstIterator it;
92  for (it = map.begin(); it != map.end(); ++it) {
93  cg.deleteEntry(it.key());
94  }
95 }

110 void KMessageBoxNotify::sendNotification(QMessageBox::Icon notificationType, const QString &message, QWidget *parent)
111 {
112  QString messageType;
113  switch (notificationType) {
114  case QMessageBox::Warning:
115  messageType = QStringLiteral("messageWarning");
116  break;
117  case QMessageBox::Critical:
118  messageType = QStringLiteral("messageCritical");

123  default:
124  messageType = QStringLiteral("messageInformation");
