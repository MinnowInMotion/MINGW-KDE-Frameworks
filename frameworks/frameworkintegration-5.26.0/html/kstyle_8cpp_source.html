104 void ColorSchemeFilter::installColorScheme(QWidget *w)
105 {
106  if (!w || !w->isTopLevel()) {
107  return;
108  }
109 #if HAVE_X11
110  static const bool s_x11 = QX11Info::isPlatformX11();
111  if (!s_x11) {
112  return;
113  }
114  static xcb_atom_t atom = XCB_ATOM_NONE;
115  xcb_connection_t *c = QX11Info::connection();
116  if (atom == XCB_ATOM_NONE) {
117  const QByteArray name = QByteArrayLiteral("_KDE_NET_WM_COLOR_SCHEME");
118  const xcb_intern_atom_cookie_t cookie = xcb_intern_atom(c, false, name.length(), name.constData());
119  QScopedPointer<xcb_intern_atom_reply_t, QScopedPointerPodDeleter> reply(xcb_intern_atom_reply(c, cookie, Q_NULLPTR));
120  if (!reply.isNull()) {
121  atom = reply->atom;
122  } else {
123  // no point in continuing, we don't have the atom
124  return;
125  }
126  }
127  const QString path = qApp->property(s_schemePropertyName).toString();
128  if (path.isEmpty()) {
129  xcb_delete_property(c, w->winId(), atom);
130  } else {
131  xcb_change_property(c, XCB_PROP_MODE_REPLACE, w->winId(), atom, XCB_ATOM_STRING,
132  8, path.size(), qPrintable(path));
133  }
134 #endif
135 }

196 /*
197  Custom Style Element runtime extension:
198  We reserve one StyleHint to let the effective style inform widgets whether it supports certain
199  string based style elements.
200  As this could lead to number conflicts (i.e. an app utilizing one of the hints itself for other
201  purposes) there're various safety mechanisms to rule out such interference.
202 
203  1) It's most unlikely that a widget in some 3rd party app will accidentally call a general
204  QStyle/KStyle styleHint() or draw*() and (unconditionally) expect a valid return, however:
205  a. The StyleHint is not directly above Qt's custom base, assuming most 3rd party apps would
206  - in case - make use of such
207  b. In order to be accepted, the StyleHint query must pass a widget with a perfectly matching
208  name, containing the typical element prefix ("CE_", etc.) and being supported by the current style
209  c. Instead using Qt's fragile qstyleoption_cast on the QStyleOption provided to the StyleHint
210  query, try to dump out a string and hope for the best, we now manipulate the widgets objectName().
211  Plain Qt dependent widgets can do that themselves and if a widget uses KStyle's convenience access
212  functions, it won't notice this at all
213 
214  2) The key problem is that a common KDE widget will run into an apps custom style which will then
215  falsely respond to the styleHint() call with an invalid value.
216  To prevent this, supporting styles *must* set a Q_CLASSINFO "X-KDE-CustomElements".
217 
218  3) If any of the above traps snaps, the returned id is 0 - the QStyle default, indicating
219  that this element is not supported by the current style.
220 
221  Obviously, this contains the "diminished clean" action to (temporarily) manipulate the
222  objectName() of a const QWidget* - but this happens completely inside KStyle or the widget, if
223  it does not make use of KStyles static convenience functions.
224  My biggest worry here would be, that in a multithreaded environment a thread (usually not being
225  owner of the widget) does something crucially relying on the widgets name property...
226  This however would also have to happen during the widget construction or stylechanges, when
227  the functions in doubt will typically be called.
228  So this is imho unlikely causing any trouble, ever.
229 */
230 
231 /*
232  The functions called by the real style implementation to add support for a certain element.
233  Checks for well-formed string (containing the element prefix) and returns 0 otherwise.
234  Checks whether the element is already supported or inserts it otherwise; Returns the proper id
235  NOTICE: We could check for "X-KDE-CustomElements", but this would bloat style start up times
236  (if they e.g. register 100 elements or so)
237 */

239 static inline int newStyleElement(const QString &element, const char *check, int &counter, QHash<QString, int> *elements)
240 {
241  if (!element.contains(check)) {
242  return 0;
243  }
244  int id = elements->value(element, 0);
245  if (!id) {
246  ++counter;
247  id = counter;
248  elements->insert(element, id);
249  }
250  return id;
251 }

418 int KStyle::styleHint(StyleHint hint, const QStyleOption *option, const QWidget *widget, QStyleHintReturn *returnData) const
419 {
420  switch (hint) {
421  case SH_DialogButtonBox_ButtonsHaveIcons: {
422  // was KGlobalSettings::showIconsOnPushButtons() :
423  KConfigGroup g(KSharedConfig::openConfig(), "KDE");
424  return g.readEntry("ShowIconsOnPushButtons", true);
425  }
426 
427  case SH_ItemView_ArrowKeysNavigateIntoChildren:
428  return true;
429 
430  case SH_Widget_Animate: {
431  KConfigGroup g(KSharedConfig::openConfig(), "KDE-Global GUI Settings");
432  return g.readEntry("GraphicEffectsLevel", true);
433  }
434 
435  #if QT_VERSION >= QT_VERSION_CHECK(5,5,0)
436  case QStyle::SH_Menu_SubMenuSloppyCloseTimeout: return 300;
437  #endif
438 
439  case SH_ToolButtonStyle: {
440  KConfigGroup g(KSharedConfig::openConfig(), "Toolbar style");
441 
442  bool useOthertoolbars = false;
443  const QWidget *parent = widget ? widget->parentWidget() : nullptr;
444 
445  //If the widget parent is a QToolBar and the magic property is set
446  if (parent && qobject_cast< const QToolBar * >(parent)) {
447  if (parent->property("otherToolbar").isValid()) {
448  useOthertoolbars = true;
449  }
450  }
451 
452  QString buttonStyle;
453  if (useOthertoolbars) {
454  buttonStyle = g.readEntry("ToolButtonStyleOtherToolbars", "NoText").toLower();
455  } else {
456  buttonStyle = g.readEntry("ToolButtonStyle", "TextBesideIcon").toLower();
457  }
458 
459  return buttonStyle == QLatin1String("textbesideicon") ? Qt::ToolButtonTextBesideIcon
460  : buttonStyle == QLatin1String("icontextright") ? Qt::ToolButtonTextBesideIcon
461  : buttonStyle == QLatin1String("textundericon") ? Qt::ToolButtonTextUnderIcon
462  : buttonStyle == QLatin1String("icontextbottom") ? Qt::ToolButtonTextUnderIcon
463  : buttonStyle == QLatin1String("textonly") ? Qt::ToolButtonTextOnly
464  : Qt::ToolButtonIconOnly;
465  }
466 
467  case SH_KCustomStyleElement:
468  if (!widget) {
469  return 0;
470  }
471 
472  return d->styleElements.value(widget->objectName(), 0);
473 
474  default:
475  break;
476  };
477 
478  return QCommonStyle::styleHint(hint, option, widget, returnData);
479 }
