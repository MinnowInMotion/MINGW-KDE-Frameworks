150 Value UnitCategory::convert(const Value &value, const QString &toUnit)
151 {
152  if (d && (toUnit.isEmpty() || d->m_unitMap.contains(toUnit)) && value.unit().isValid()) {
153  Unit to = toUnit.isEmpty() ? defaultUnit() : d->m_unitMap[toUnit];
154  return convert(value, to);
155  }
156  return Value();
157 }

227 void UnitCategory::addUnit(const Unit &unit)
228 {
229  if (d) {
230  unit.d->m_category = *this;
231  const QStringList list = unit.d->m_matchString.split(';');
232  foreach (const QString &name, list)
233  d->m_unitMap[name] = unit;
234  d->m_idMap[unit.id()] = unit;
235  d->m_units.append(unit);
236  }
237 }
