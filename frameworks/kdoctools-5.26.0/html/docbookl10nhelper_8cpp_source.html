33 {
34 public:
35  int searchLang(QString el)
36  {
37 
38  for (int i = 0; i < size(); ++i) {
39  if (at(i).first == el) {
40  return i;
41  }
42  }
43  return -1;

47 int writeLangFile(const QString &fname, const QString &dtdPath,
48  const LangListType &langMap)
49 {
50 
51  QFile outFile(fname);
52  if (! outFile.open(QIODevice::WriteOnly)) {
53  qCritical() << QStringLiteral("Could not write %1")
54  .arg(outFile.fileName());
55  return (1);
56  }
57 
58  QTextStream outStream(&outFile);
59  outStream << "<?xml version='1.0'?>" << endl;
60  outStream << QStringLiteral("<!DOCTYPE l:i18n SYSTEM \"%1\" [")
61  .arg(dtdPath) << endl;
62 
63  LangListType::const_iterator i = langMap.constBegin();
64  while (i != langMap.constEnd()) {
65  //qDebug() << (*i).first << ": " << (*i).second;
66  outStream << QStringLiteral("<!ENTITY %1 SYSTEM \"%2\">")
67  .arg((*i).first).arg((*i).second) << endl;
68  ++i;
69  }
70  outStream << "]>" << endl;
71 
72  if (langMap.size() > 0) {
73  outStream
74  << "<l:i18n xmlns:l=\"http://docbook.sourceforge.net/xmlns/l10n/1.0\">"
75  << endl;
76  i = langMap.constBegin();
77  while (i != langMap.constEnd()) {
78  outStream << QStringLiteral("&%1;")
79  .arg((*i).first) << endl;
80  ++i;
81  }
82  outStream << "</l:i18n>" << endl;
83  }
84 
85  outFile.close();
86 
87  return (0);
88 }

124 inline const QString addTrailingSlash(const QString &p)
125 {
126  return p.endsWith(QStringLiteral("/")) ? p : p + QStringLiteral("/");
127 }

129 int main(int argc, char **argv)
130 {
131  QCoreApplication app(argc, argv);
132 
133  const QStringList arguments = app.arguments();
134  if (arguments.count() != 4) {
135  qCritical() << "wrong argument count";
136  return (1);
137  }
138 
139  const QString l10nDir = addTrailingSlash(arguments[1]);
140  const QString l10nCustomDir = addTrailingSlash(arguments[2]);
141  const QString destDir = addTrailingSlash(arguments[3]);
142 
143  QFile i18nFile(l10nDir + QStringLiteral("common/l10n.xml"));
144 
145  if (! i18nFile.open(QIODevice::ReadOnly)) {
146  qCritical() << i18nFile.fileName() << " not found";
147  return (1);
148  }
149 
150  const QString all10nFName = destDir + QStringLiteral("all-l10n.xml");
151  const QString customl10nFName = destDir + QStringLiteral("kde-custom-l10n.xml");
152 
153  /*
154  * for each language defined in the original l10n.xml, copy
155  * it into all-l10n.xml and store it in a list;
156  **/
157  QRegExp rxEntity, rxEntity2, rxDocType, rxDocType2;
158  rxDocType.setPattern(QStringLiteral("^\\s*<!DOCTYPE\\s+l:i18n\\s+SYSTEM\\s+\"l10n\\.dtd\"\\s+\\[\\s*$"));
159  rxDocType2.setPattern(QStringLiteral("^\\s*<!DOCTYPE\\s+l:i18n\\s+SYSTEM\\s+\"l10n\\.dtd\"\\s*>$"));
160  rxEntity.setPattern(QStringLiteral("^\\s*<!ENTITY\\s+([^\\s]+)\\s+SYSTEM\\s+\"([^\\s]+)\">\\s*$"));
161  rxEntity2.setPattern(QStringLiteral("^\\s*<l:l10n language=\"([^\\s]+)\"\\s+href=\"([^\\s]+)\"/>\\s*$"));
162  QTextStream inStream(&i18nFile);
163  int parsingState = 0;
164 
165  LangListType allLangs, customLangs;
166 
167  bool foundRxEntity = false;
168  bool foundRxEntity2 = false;
169  while (! inStream.atEnd()) {
170  QString line = inStream.readLine();
171 
172  switch (parsingState) {
173  case 0:
174  if (rxDocType.indexIn(line) != -1) {
175  parsingState = 1;
176  //qDebug() << "DTD found";
177  } else if (rxDocType2.indexIn(line) != -1) {
178  parsingState = 1;
179  //qDebug() << "DTD found";
180  }
181  break;
182  case 1:
183  QString langCode, langFile;
184  if (rxEntity.indexIn(line) != -1 && !foundRxEntity2) {
185  foundRxEntity = true;
186  langCode = rxEntity.cap(1);
187  langFile = l10nDir + QStringLiteral("common/") + rxEntity.cap(2);
188  allLangs += qMakePair(langCode, langFile);
189  //qDebug() << langCode << " - " << langFile;
190  } else if (rxEntity2.indexIn(line) != -1 && !foundRxEntity) {
191  foundRxEntity2 = true;
192  langCode = rxEntity2.cap(1);
193  langFile = l10nDir + QStringLiteral("common/") + rxEntity2.cap(2);
194  allLangs += qMakePair(langCode, langFile);
195  //qDebug() << langCode << " - " << langFile;
196  }
197  break;
198  }
199 
200  }
201  i18nFile.close();
202 
203  /* read the list of locally-available custom languages */
204  QDir outDir(l10nCustomDir);
205 
206  QStringList dirFileFilters;
207  dirFileFilters << QStringLiteral("*.xml");
208  QStringList customLangFiles = outDir.entryList(dirFileFilters,
209  QDir::Files, QDir::Name);
210  /* the following two calls to removeOne should not be needed, as
211  * the customization directory from the sources should not contain
212  * those files
213  */
214  customLangFiles.removeOne(QStringLiteral("all-l10n.xml"));
215  customLangFiles.removeOne(QStringLiteral("kde-custom-l10n.xml"));
216  //qDebug() << "customLangFiles:" << customLangFiles;
217 
218  /*
219  * for each custom language (from directory listing), if it is not
220  * in the list of upstream languages, add it to all-l10n.xml,
221  * otherwise add it to kde-custom-l10n.xml
222  */
223  QStringList::const_iterator i = customLangFiles.constBegin();
224  while (i != customLangFiles.constEnd()) {
225  QString langFile = (*i);
226  /* remove trailing .xml */
227  QString langCode = langFile.left(langFile.length() - 4);
228 
229  QPair<QString, QString> cl = qMakePair(langCode, langFile);
230  if ((allLangs.searchLang(langCode)) > 0) {
231  /* custom language found in upstream list */
232  customLangs += cl;
233  } else {
234  /* custom language not found in upstream list */
235  allLangs += cl;
236  }
237  ++i;
238  }
239 
240  int res = 0;
241 
242  if (foundRxEntity) {
243  /* old style (docbook-xsl<=1.75) */
244  res = writeLangFile(all10nFName, l10nDir + QStringLiteral("common/l10n.dtd"),
245  allLangs);
246  } else {
247  res = writeLangFileNew(all10nFName, l10nDir + QStringLiteral("common/l10n.dtd"),
248  allLangs);
249  }
250 
251  return (res);
252 }
