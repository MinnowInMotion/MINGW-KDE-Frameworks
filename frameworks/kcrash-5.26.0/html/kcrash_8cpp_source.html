160 // Set the default crash handler in 10 seconds
161 // This is used after an autorestart, the second instance of the application
162 // is started with KCRASH_AUTO_RESTARTED=1, and we
163 // set the defaultCrashHandler (to handle autorestart) after 10s.
164 // The delay is to see if we stay up for more than 10s time, to avoid infinite
165 // respawning if the app crashes on startup.
166 class KCrashDelaySetHandler : public QObject
167 {
168 public:

171  startTimer(10000); // 10 s
172  }
173 protected:

176  if (!s_crashHandler) { // not set meanwhile
177  KCrash::setCrashHandler(KCrash::defaultCrashHandler);
178  }

200 void KCrash::setApplicationFilePath(const QString &filePath)
201 {
202  const int pos = filePath.lastIndexOf(QLatin1Char('/'));
203  const QString appName = filePath.mid(pos + 1);
204  const QString appPath = filePath.left(pos); // could be empty, in theory
205 
206  s_appName = qstrdup(QFile::encodeName(appName).constData());
207  s_appPath = qstrdup(QFile::encodeName(appPath).constData());

209  // Prepare the auto-restart command
210  delete[] s_autoRestartCommand;
211  s_autoRestartCommand = qstrdup(QFile::encodeName(filePath).constData());
212 
213  QStringList args = QCoreApplication::arguments();
214  args[0] = filePath; // replace argv[0] with full path above
215  delete[] s_autoRestartCommandLine;

217  s_autoRestartCommandLine = new char *[args.count() + 1];

224 void KCrash::setDrKonqiEnabled(bool enabled)
225 {
226  const int launchDrKonqi = enabled ? 1 : 0;
227  if (s_launchDrKonqi == launchDrKonqi) {
228  return;
229  }
230  s_launchDrKonqi = launchDrKonqi;
231  if (s_launchDrKonqi && !s_drkonqiPath) {
232  // search paths
233  const QStringList paths = QStringList()
234  << QFile::decodeName(qgetenv("LIBEXEC_PATH")).split(QLatin1Char(':'), QString::SkipEmptyParts) // env var is used first
235  << QCoreApplication::applicationDirPath() // then look where our application binary is located
236  << QLibraryInfo::location(QLibraryInfo::LibraryExecutablesPath) // look where libexec path is (can be set in qt.conf)
237  << QFile::decodeName(CMAKE_INSTALL_FULL_LIBEXECDIR); // look at our installation location
238  const QString exec = QStandardPaths::findExecutable(QStringLiteral("drkonqi"), paths);
239  if (exec.isEmpty()) {
240  qCDebug(LOG_KCRASH) << "Could not find drkonqi in search paths:" << paths;
241  s_launchDrKonqi = 0;
242  } else {
243  s_drkonqiPath = qstrdup(qPrintable(exec));
244  }
245  }
246 
247  //we need at least the default crash handler to launch drkonqi
248  if (s_launchDrKonqi && !s_crashHandler) {
249  setCrashHandler(defaultCrashHandler);
250  }
251 }

372 void
373 KCrash::defaultCrashHandler(int sig)
374 {
375  // WABA: Do NOT use qDebug() in this function because it is much too risky!
376  // Handle possible recursions
377  static int crashRecursionCounter = 0;
378  crashRecursionCounter++; // Nothing before this, please !
379 
380 #if !defined(Q_OS_WIN)
381  signal(SIGALRM, SIG_DFL);
382  alarm(3); // Kill me... (in case we deadlock in malloc)
383 #endif
384 
385 #ifdef Q_OS_SOLARIS
386  (void) printstack(2 /* stderr, assuming it's still open. */);
387 #endif
388 
389  if (crashRecursionCounter < 2) {
390  if (s_emergencySaveFunction) {
391  s_emergencySaveFunction(sig);
392  }
393  if ((s_flags & AutoRestart) && s_autoRestartCommand) {
394  QThread::sleep(1);
395  startProcess(s_autoRestartArgc, const_cast<const char **>(s_autoRestartCommandLine), false);
396  }
397  crashRecursionCounter++;
398  }
399 
400 #if !defined(Q_OS_WIN)
401  if (!(s_flags & KeepFDs)) {
402  closeAllFDs();
403  }
404 # if HAVE_X11
405  else if (QX11Info::display()) {
406  close(ConnectionNumber(QX11Info::display()));
407  }
408 # endif
409 #endif
410 
411  if (crashRecursionCounter < 3) {
412 #ifndef NDEBUG
413  fprintf(stderr, "KCrash: crashing... crashRecursionCounter = %d\n",
414  crashRecursionCounter);
415  fprintf(stderr, "KCrash: Application Name = %s path = %s pid = %lld\n",
416  s_appName ? s_appName : "<unknown>",
417  s_appPath ? s_appPath : "<unknown>", QCoreApplication::applicationPid());
418  fprintf(stderr, "KCrash: Arguments: ");
419  for (int i = 0; s_autoRestartCommandLine[i]; ++i) {
420  fprintf(stderr, "%s ", s_autoRestartCommandLine[i]);
421  }
422  fprintf(stderr, "\n");
423 #else
424  fprintf(stderr, "KCrash: Application '%s' crashing...\n",
425  s_appName ? s_appName : "<unknown>");
426 #endif
427 
428  if (s_launchDrKonqi != 1) {
429  setCrashHandler(0);
430 #if !defined(Q_OS_WIN)
431  raise(sig); // dump core, or whatever is the default action for this signal.
432 #endif
433  return;
434  }
435 
436  const char *argv[29]; // don't forget to update this
437  int i = 0;
438 
439  // argument 0 has to be drkonqi
440  argv[i++] = s_drkonqiPath;
441 
442  const QByteArray platformName = QGuiApplication::platformName().toUtf8();
443  if (!platformName.isEmpty()) {
444  argv[i++] = "-platform";
445  argv[i++] = platformName.constData();
446  }
447 
448 #if HAVE_X11
449  if (platformName == QByteArrayLiteral("xcb")) {
450  // start up on the correct display
451  argv[i++] = "-display";
452  if (QX11Info::display()) {
453  argv[i++] = XDisplayString(QX11Info::display());
454  } else {
455  argv[i++] = getenv("DISPLAY");
456  }
457  }
458 #endif
459 
460  argv[i++] = "--appname";
461  argv[i++] = s_appName ? s_appName : "<unknown>";
462 
463  if (loadedByKdeinit) {
464  argv[i++] = "--kdeinit";
465  }
466 
467  // only add apppath if it's not NULL
468  if (s_appPath && *s_appPath) {
469  argv[i++] = "--apppath";
470  argv[i++] = s_appPath;
471  }
472 
473  // signal number -- will never be NULL

504  if (!startupId.isNull()) {
505  argv[i++] = "--startupid";
506  argv[i++] = startupId.constData();
507  }

510  argv[i++] = "--safer";
511  }

567 LONG WINAPI KCrash::win32UnhandledExceptionFilter(_EXCEPTION_POINTERS *exceptionInfo)
568 {
569  // kdbgwin needs the context inside exceptionInfo because if getting the context after the
570  // exception happened, it will walk down the stack and will stop at KiUserEventDispatch in
571  // ntdll.dll, which is supposed to dispatch the exception from kernel mode back to user mode
572  // so... let's create some shared memory
573  HANDLE hMapFile = NULL;
574  hMapFile = CreateFileMapping(
575  INVALID_HANDLE_VALUE,
576  NULL,
577  PAGE_READWRITE,
578  0,
579  sizeof(CONTEXT),
580  TEXT("Local\\KCrashShared"));
581 
582  LPCTSTR pBuf = NULL;
583  pBuf = (LPCTSTR) MapViewOfFile(
584  hMapFile,
585  FILE_MAP_ALL_ACCESS,
586  0,
587  0,
588  sizeof(CONTEXT));
589  CopyMemory((PVOID) pBuf, exceptionInfo->ContextRecord, sizeof(CONTEXT));
590 
591  if (s_crashHandler) {
592  s_crashHandler(exceptionInfo->ExceptionRecord->ExceptionCode);
593  }
594 
595  CloseHandle(hMapFile);
596  return EXCEPTION_EXECUTE_HANDLER; //allow windows to do the default action (terminate)

607 void KCrash::startProcess(int argc, const char *argv[], bool waitAndExit)
608 {
609  bool startDirectly = true;
610 
611  // First try to start the app via kdeinit, if the AlwaysDirectly flag hasn't been specified.
612  // This is done because it is dangerous to use fork() in the crash handler
613  // (there can be functions registered to be performed before fork(), for example handling
614  // of malloc locking, which doesn't work when malloc crashes because of heap corruption).
615  if (!(s_flags & AlwaysDirectly)) {
616  startDirectly = !startProcessInternal(argc, argv, waitAndExit, false);
617  }
618 
619  // If we can't reach kdeinit, we can still at least try to fork()
620  if (startDirectly) {
621  startProcessInternal(argc, argv, waitAndExit, true);
622  }
623 }
624 
625 static bool startProcessInternal(int argc, const char *argv[], bool waitAndExit, bool directly)
626 {
627  fprintf(stderr, "KCrash: Attempting to start %s %s\n", argv[0], directly ? "directly" : "from kdeinit");
628 
629  pid_t pid = directly ? startDirectly(argv) : startFromKdeinit(argc, argv);
630 
631  if (pid > 0 && waitAndExit) {
632  // Seems we made it....
633  alarm(0); //stop the pending alarm that was set at the top of the defaultCrashHandler
634 
635  // Wait forever until the started process exits. This code path is executed
636  // when launching drkonqi. Note that drkonqi will stop this process in the meantime.
637  if (directly) {
638  //if the process was started directly, use waitpid(), as it's a child...
639  while (waitpid(-1, NULL, 0) != pid) {}
640  } else {
641 #ifdef Q_OS_LINUX
642  // Declare the process that will be debugging the crashed KDE app (#245529)
643 #ifndef PR_SET_PTRACER
644 # define PR_SET_PTRACER 0x59616d61
645 #endif
646  prctl(PR_SET_PTRACER, pid, 0, 0, 0);
647 #endif
648  //...else poll its status using kill()
649  while (kill(pid, 0) >= 0) {
650  sleep(1);
651  }
652  }
653  _exit(253);
654  }
655 
656  return (pid > 0); //return true on success
657 }

659 static pid_t startFromKdeinit(int argc, const char *argv[])
660 {
661  int socket = openSocket();
662  if (socket < -1) {
663  return 0;
664  }
665  kcrash_launcher_header header;
666  header.cmd = LAUNCHER_EXEC_NEW;
667  const int BUFSIZE = 8192; // make sure this is big enough
668  const int CWDSIZE = 2000;
669  char buffer[ BUFSIZE + 10 + 24 /*the env var*/ + CWDSIZE+1 ];
670  int pos = 0;
671  long argcl = argc;
672  memcpy(buffer + pos, &argcl, sizeof(argcl));
673  pos += sizeof(argcl);
674  for (int i = 0;
675  i < argc;
676  ++i) {
677  int len = strlen(argv[ i ]) + 1; // include terminating \0
678  if (pos + len >= BUFSIZE) {
679  fprintf(stderr, "BUFSIZE in KCrash not big enough!\n");
680  return 0;
681  }
682  memcpy(buffer + pos, argv[ i ], len);
683  pos += len;
684  }
685 
686  long env = 1; // 1 env var
687  memcpy(buffer + pos, &env, sizeof(env));
688  pos += sizeof(env);
689  static const char s_envVar[] = "KCRASH_AUTO_RESTARTED=1";
690  int len = strlen(s_envVar) + 1;
691  memcpy(buffer + pos, s_envVar, len);
692  pos += len;
693 
694  long avoid_loops = 0;
695  memcpy(buffer + pos, &avoid_loops, sizeof(avoid_loops));
696  pos += sizeof(avoid_loops);
697 
698  char cwd[CWDSIZE];
699  if (getcwd(cwd, CWDSIZE-1)) {
700  len = strlen(cwd) + 1; // include terminating \0
701  memcpy(buffer + pos, cwd, len);
702  pos += len;
703  }
704 
705  header.arg_length = pos;
706  write_socket(socket, (char *) &header, sizeof(header));
707  write_socket(socket, buffer, pos);
708 
709  if (read_socket(socket, (char *) &header, sizeof(header)) < 0
710  || header.cmd != LAUNCHER_OK) {
711  return 0;
712  }
713  long pid;
714  read_socket(socket, (char *) &pid, sizeof(pid));
715  return static_cast<pid_t>(pid);
716 }

718 static pid_t startDirectly(const char *argv[])
719 {
720  pid_t pid = fork();
721  switch (pid) {
722  case -1:
723  fprintf(stderr, "KCrash failed to fork(), errno = %d\n", errno);
724  return 0;
725  case 0:
726  setgroups(0, 0); // Remove any extraneous groups
727  if (setgid(getgid()) < 0 || setuid(getuid()) < 0) {
728  _exit(253); // This cannot happen. Theoretically.
729  }
730  closeAllFDs(); // We are in the child now. Close FDs unconditionally.
731  setenv("KCRASH_AUTO_RESTARTED", "1", 1);
732  execvp(argv[0], const_cast< char ** >(argv));
733  fprintf(stderr, "KCrash failed to exec(), errno = %d\n", errno);
734  _exit(253);
735  default:
736  return pid;
737  }

740 /*
741  * Write 'len' bytes from 'buffer' into 'sock'.
742  * returns 0 on success, -1 on failure.
743  */
744 static int write_socket(int sock, char *buffer, int len)
745 {
746  ssize_t result;
747  int bytes_left = len;
748  while (bytes_left > 0) {
749  result = write(sock, buffer, bytes_left);
750  if (result > 0) {
751  buffer += result;
752  bytes_left -= result;
753  } else if (result == 0) {
754  return -1;
755  } else if ((result == -1) && (errno != EINTR) && (errno != EAGAIN)) {
756  return -1;
757  }
758  }
759  return 0;
760 }
