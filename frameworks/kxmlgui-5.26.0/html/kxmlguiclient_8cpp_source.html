218 void KXMLGUIClient::setXMLFile(const QString &_file, bool merge, bool setXMLDoc)
219 {
220  // store our xml file name
221  if (!_file.isNull()) {
222  d->m_xmlFile = _file;
223  }
224 
225  if (!setXMLDoc) {
226  return;
227  }
228 
229  QString file = _file;
230  QStringList allFiles;
231  if (!QDir::isRelativePath(file)) {
232  allFiles.append(file);
233  } else {
234  const QString filter = componentName() + QLatin1Char('/') + _file;
235 
236  // files on filesystem
237  allFiles << QStandardPaths::locateAll(QStandardPaths::GenericDataLocation, QStringLiteral("kxmlgui5/") + filter); // KF >= 5.1
238 
239  // KF >= 5.4 (resource file)
240  const QString qrcFile(QStringLiteral(":/kxmlgui5/") + filter);
241  if (QFile::exists(qrcFile)) {
242  allFiles << qrcFile;
243  }
244 
245  // then compat locations
246  const QStringList compatFiles =
247  QStandardPaths::locateAll(QStandardPaths::GenericDataLocation, filter) + // kdelibs4, KF 5.0
248  QStandardPaths::locateAll(QStandardPaths::GenericDataLocation, _file); // kdelibs4, KF 5.0, caller passes component name
249 
250  if (allFiles.isEmpty() && !compatFiles.isEmpty()) {
251  qWarning() << "KXMLGUI file found at deprecated location" << compatFiles << "-- please use ${KXMLGUI_INSTALL_DIR} to install this file instead.";
252  }
253  allFiles += compatFiles;
254  }
255  if (allFiles.isEmpty() && !_file.isEmpty()) {
256  // if a non-empty file gets passed and we can't find it,
257  // inform the developer using some debug output
258  qWarning() << "cannot find .rc file" << _file << "for component" << componentName();
259  }
260 
261  // make sure to merge the settings from any file specified by setLocalXMLFile()
262  if (!d->m_localXMLFile.isEmpty() && !file.endsWith(QStringLiteral("ui_standards.rc"))) {
263  const bool exists = QDir::isRelativePath(d->m_localXMLFile) || QFile::exists(d->m_localXMLFile);
264  if (exists && !allFiles.contains(d->m_localXMLFile)) {
265  allFiles.prepend(d->m_localXMLFile);
266  }
267  }
268 
269  QString doc;
270  if (!allFiles.isEmpty()) {
271  file = findMostRecentXMLFile(allFiles, doc);
272  }
273 
274  // Always call setXML, even on error, so that we don't keep all ui_standards.rc menus.
275  setXML(doc, merge);
276 }

293 // The top document element may have translation domain attribute set,
294 // or the translation domain may be implicitly the application domain.
295 // This domain must be used to fetch translations for all text elements
296 // in the document that do not have their own domain attribute.
297 // In order to preserve this semantics through document mergings,
298 // the top or application domain must be propagated to all text elements
299 // lacking their own domain attribute.
300 static void propagateTranslationDomain(QDomDocument &doc, const QStringList tagNames)
301 {
302  const QLatin1String attrDomain("translationDomain");
303  QDomElement base = doc.documentElement();
304  QString domain = base.attribute(attrDomain);
305  if (domain.isEmpty()) {
306  domain = QString::fromUtf8(KLocalizedString::applicationDomain());
307  if (domain.isEmpty()) {
308  return;
309  }
310  }
311  foreach (const QString &tagName, tagNames) {
312  QDomNodeList textNodes = base.elementsByTagName(tagName);
313  for (int i = 0; i < textNodes.length(); ++i) {
314  QDomElement e = textNodes.item(i).toElement();
315  QString localDomain = e.attribute(attrDomain);
316  if (localDomain.isEmpty()) {
317  e.setAttribute(attrDomain, domain);
318  }
319  }
320  }
321 }

323 void KXMLGUIClient::setXML(const QString &document, bool merge)
324 {
325  QDomDocument doc;
326  QString errorMsg;
327  int errorLine = 0, errorColumn = 0;
328  // QDomDocument raises a parse error on empty document, but we accept no app-specific document,
329  // in which case you only get ui_standards.rc layout.
330  bool result = document.isEmpty() || doc.setContent(document, &errorMsg, &errorLine, &errorColumn);
331  if (result) {
332  propagateTranslationDomain(doc, d->m_textTagNames);
333  setDOMDocument(doc, merge);
334  } else {
335 #ifdef NDEBUG
336  qCritical() << "Error parsing XML document:" << errorMsg << "at line" << errorLine << "column" << errorColumn;
337  setDOMDocument(QDomDocument(), merge); // otherwise empty menus from ui_standards.rc stay around
338 #else
339  qCritical() << "Error parsing XML document:" << errorMsg << "at line" << errorLine << "column" << errorColumn;
340  abort();
341 #endif
342  }
343 }

382 bool KXMLGUIClientPrivate::mergeXML(QDomElement &base, QDomElement &additive, KActionCollection *actionCollection)
383 {
384  const QLatin1String tagAction("Action");
385  const QLatin1String tagMerge("Merge");
386  const QLatin1String tagSeparator("Separator");
387  const QLatin1String tagMergeLocal("MergeLocal");
388  const QLatin1String tagText("text");
389  const QLatin1String attrAppend("append");
390  const QString attrName(QStringLiteral("name"));
391  const QString attrWeakSeparator(QStringLiteral("weakSeparator"));
392  const QString attrAlreadyVisited(QStringLiteral("alreadyVisited"));
393  const QString attrNoMerge(QStringLiteral("noMerge"));
394  const QLatin1String attrOne("1");
395 
396  // there is a possibility that we don't want to merge in the
397  // additive.. rather, we might want to *replace* the base with the
398  // additive. this can be for any container.. either at a file wide
399  // level or a simple container level. we look for the 'noMerge'
400  // tag, in any event and just replace the old with the new
401  if (additive.attribute(attrNoMerge) == attrOne) { // ### use toInt() instead? (Simon)
402  base.parentNode().replaceChild(additive, base);
403  return true;
404  } else {
405  // Merge attributes
406  {
407  const QDomNamedNodeMap attribs = additive.attributes();
408  const uint attribcount = attribs.count();
409 
410  for (uint i = 0; i < attribcount; ++i) {
411  const QDomNode node = attribs.item(i);
412  base.setAttribute(node.nodeName(), node.nodeValue());
413  }
414  }
415 
416  // iterate over all elements in the container (of the global DOM tree)
417  QDomNode n = base.firstChild();
418  while (!n.isNull()) {
419  QDomElement e = n.toElement();
420  n = n.nextSibling(); // Advance now so that we can safely delete e
421  if (e.isNull()) {
422  continue;
423  }
424 
425  const QString tag = e.tagName();
426 
427  // if there's an action tag in the global tree and the action is
428  // not implemented, then we remove the element
429  if (equalstr(tag, tagAction)) {
430  const QString name = e.attribute(attrName);
431  if (!actionCollection->action(name) ||
432  !KAuthorized::authorizeAction(name)) {
433  // remove this child as we aren't using it
434  base.removeChild(e);
435  continue;
436  }
437  }
438 
439  // if there's a separator defined in the global tree, then add an
440  // attribute, specifying that this is a "weak" separator
441  else if (equalstr(tag, tagSeparator)) {
442  e.setAttribute(attrWeakSeparator, uint(1));
443 
444  // okay, hack time. if the last item was a weak separator OR
445  // this is the first item in a container, then we nuke the
446  // current one
447  QDomElement prev = e.previousSibling().toElement();
448  if (prev.isNull() ||
449  (equalstr(prev.tagName(), tagSeparator) && !prev.attribute(attrWeakSeparator).isNull()) ||
450  (equalstr(prev.tagName(), tagText))) {
451  // the previous element was a weak separator or didn't exist
452  base.removeChild(e);
453  continue;
454  }
455  }
456 
457  // the MergeLocal tag lets us specify where non-standard elements
458  // of the local tree shall be merged in. After inserting the
459  // elements we delete this element
460  else if (equalstr(tag, tagMergeLocal)) {
461  QDomNode it = additive.firstChild();
462  while (!it.isNull()) {
463  QDomElement newChild = it.toElement();
464  it = it.nextSibling();
465  if (newChild.isNull()) {
466  continue;
467  }
468 
469  if (equalstr(newChild.tagName(), tagText)) {
470  continue;
471  }
472 
473  if (newChild.attribute(attrAlreadyVisited) == attrOne) {
474  continue;
475  }
476 
477  QString itAppend(newChild.attribute(attrAppend));
478  QString elemName(e.attribute(attrName));
479 
480  if ((itAppend.isNull() && elemName.isEmpty()) ||
481  (itAppend == elemName)) {
482  // first, see if this new element matches a standard one in
483  // the global file. if it does, then we skip it as it will
484  // be merged in, later
485  QDomElement matchingElement = findMatchingElement(newChild, base);
486  if (matchingElement.isNull() || equalstr(newChild.tagName(), tagSeparator)) {
487  base.insertBefore(newChild, e);
488  }
489  }
490  }
491 
492  base.removeChild(e);
493  continue;
494  }
495 
496  else if (equalstr(tag, tagText)) {
497  continue;
498  } else if (equalstr(tag, tagMerge)) {
499  continue;
500  }
501 
502  // in this last case we check for a separator tag and, if not, we
503  // can be sure that it is a container --> proceed with child nodes
504  // recursively and delete the just proceeded container item in
505  // case it is empty (if the recursive call returns true)
506  else {
507  QDomElement matchingElement = findMatchingElement(e, additive);
508  if (!matchingElement.isNull()) {
509  matchingElement.setAttribute(attrAlreadyVisited, uint(1));
510 
511  if (mergeXML(e, matchingElement, actionCollection)) {
512  base.removeChild(e);
513  additive.removeChild(matchingElement); // make sure we don't append it below
514  continue;
515  }
516 
517  continue;
518  } else {
519  // this is an important case here! We reach this point if the
520  // "local" tree does not contain a container definition for
521  // this container. However we have to call mergeXML recursively
522  // and make it check if there are actions implemented for this
523  // container. *If* none, then we can remove this container now
524  QDomElement dummy;
525  if (mergeXML(e, dummy, actionCollection)) {
526  base.removeChild(e);
527  }
528  continue;
529  }
530  }
531  }
532 
533  //here we append all child elements which were not inserted
534  //previously via the LocalMerge tag
535  n = additive.firstChild();
536  while (!n.isNull()) {
537  QDomElement e = n.toElement();
538  n = n.nextSibling(); // Advance now so that we can safely delete e
539  if (e.isNull()) {
540  continue;
541  }
542 
543  QDomElement matchingElement = findMatchingElement(e, base);
544 
545  if (matchingElement.isNull()) {
546  base.appendChild(e);
547  }
548  }
549 
550  // do one quick check to make sure that the last element was not
551  // a weak separator
552  QDomElement last = base.lastChild().toElement();
553  if (equalstr(last.tagName(), tagSeparator) &&
554  (!last.attribute(attrWeakSeparator).isNull())) {
555  base.removeChild(last);
556  }
557  }
558 
559  return isEmptyContainer(base, actionCollection);
560 }

562 bool KXMLGUIClientPrivate::isEmptyContainer(const QDomElement &base, KActionCollection *actionCollection) const
563 {
564  // now we check if we are empty (in which case we return "true", to
565  // indicate the caller that it can delete "us" (the base element
566  // argument of "this" call)
567  QDomNode n = base.firstChild();
568  while (!n.isNull()) {
569  const QDomElement e = n.toElement();
570  n = n.nextSibling(); // Advance now so that we can safely delete e
571  if (e.isNull()) {
572  continue;
573  }
574 
575  const QString tag = e.tagName();
576 
577  if (equalstr(tag, QLatin1String("Action"))) {
578  // if base contains an implemented action, then we must not get
579  // deleted (note that the actionCollection contains both,
580  // "global" and "local" actions)
581  if (actionCollection->action(e.attribute(QStringLiteral("name")))) {
582  return false;
583  }
584  } else if (equalstr(tag, QLatin1String("Separator"))) {
585  // if we have a separator which has *not* the weak attribute
586  // set, then it must be owned by the "local" tree in which case
587  // we must not get deleted either
588  const QString weakAttr = e.attribute(QStringLiteral("weakSeparator"));
589  if (weakAttr.isEmpty() || weakAttr.toInt() != 1) {
590  return false;
591  }
592  }
593 
594  else if (equalstr(tag, QLatin1String("merge"))) {
595  continue;
596  }
597 
598  // a text tag is NOT enough to spare this container
599  else if (equalstr(tag, QLatin1String("text"))) {
600  continue;
601  }
602 
603  // what's left are non-empty containers! *don't* delete us in this
604  // case (at this position we can be *sure* that the container is
605  // *not* empty, as the recursive call for it was in the first loop
606  // which deleted the element in case the call returned "true"
607  else {
608  return false;
609  }
610  }
611 
612  return true; // I'm empty, please delete me.
613 }

615 QDomElement KXMLGUIClientPrivate::findMatchingElement(const QDomElement &base, const QDomElement &additive)
616 {
617  QDomNode n = additive.firstChild();
618  while (!n.isNull()) {
619  QDomElement e = n.toElement();
620  n = n.nextSibling(); // Advance now so that we can safely delete e -- TODO we don't, so simplify this
621  if (e.isNull()) {
622  continue;
623  }
624 
625  const QString tag = e.tagName();
626  // skip all action and merge tags as we will never use them
627  if (equalstr(tag, QLatin1String("Action"))
628  || equalstr(tag, QLatin1String("MergeLocal"))) {
629  continue;
630  }
631 
632  // now see if our tags are equivalent
633  if (equalstr(tag, base.tagName()) &&
634  e.attribute(QStringLiteral("name")) == base.attribute(QStringLiteral("name"))) {
635  return e;
636  }
637  }
638 
639  // nope, return a (now) null element
640  return QDomElement();
641 }
