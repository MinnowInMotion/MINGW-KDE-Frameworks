123 QWidget *KXMLGUIBuilder::createContainer(QWidget *parent, int index, const QDomElement &element, QAction *&containerAction)
124 {
125  containerAction = 0;
126 
127  if (element.attribute(QStringLiteral("deleted")).toLower() == QLatin1String("true")) {
128  return 0;
129  }
130 
131  const QString tagName = element.tagName().toLower();
132  if (tagName == d->tagMainWindow) {
133  KMainWindow *mainwindow = qobject_cast<KMainWindow *>(d->m_widget); // could be 0
134  return mainwindow;
135  }
136 
137  if (tagName == d->tagMenuBar) {
138  KMainWindow *mainWin = qobject_cast<KMainWindow *>(d->m_widget);
139  QMenuBar *bar = 0;
140  if (mainWin) {
141  bar = mainWin->menuBar();
142  }
143  if (!bar) {
144  bar = new QMenuBar(d->m_widget);
145  }
146  bar->show();
147  return bar;
148  }
149 
150  if (tagName == d->tagMenu) {
151  // Look up to see if we are inside a mainwindow. If yes, then
152  // use it as parent widget (to get kaction to plug itself into the
153  // mainwindow). Don't use a popupmenu as parent widget, otherwise
154  // the popup won't be hidden if it is used as a standalone menu as well.
155  // And we don't want to set the parent for a standalone popupmenu,
156  // otherwise its shortcuts appear.
157  //
158  // Note: menus with a parent of 0, coming from child clients, can be
159  // leaked if the child client is deleted without a proper removeClient call, though.
160  QWidget *p = parent;
161  while (p && !qobject_cast<QMainWindow *>(p)) {
162  p = p->parentWidget();
163  }
164 
165  QString name = element.attribute(d->attrName);
166 
167  if (!KAuthorized::authorizeAction(name)) {
168  return 0;
169  }
170 
171  QMenu *popup = new QMenu(p);
172  popup->setObjectName(name);
173 
174  d->m_menumenuhandler->insertMenu(popup);
175 
176  QString i18nText;
177  QDomElement textElem = element.namedItem(d->attrText1).toElement();
178  if (textElem.isNull()) { // try with capital T
179  textElem = element.namedItem(d->attrText2).toElement();
180  }
181  const QString text = textElem.text();
182  const QString context = textElem.attribute(d->attrContext);
183 
184  //qCDebug(DEBUG_KXMLGUI) << "DOMAIN" << KLocalizedString::applicationDomain();
185  //qCDebug(DEBUG_KXMLGUI) << "ELEMENT TEXT:" << text;
186 
187  if (text.isEmpty()) { // still no luck
188  i18nText = i18n("No text");
189  } else {
190  QByteArray domain = textElem.attribute(d->attrDomain).toUtf8();
191  if (domain.isEmpty()) {
192  domain = element.ownerDocument().documentElement().attribute(d->attrDomain).toUtf8();
193  if (domain.isEmpty()) {
194  domain = KLocalizedString::applicationDomain();
195  }
196  }
197  if (context.isEmpty()) {
198  i18nText = i18nd(domain.constData(), text.toUtf8().constData());
199  } else {
200  i18nText = i18ndc(domain.constData(), context.toUtf8().constData(), text.toUtf8().constData());
201  }
202  }
203 
204  //qCDebug(DEBUG_KXMLGUI) << "ELEMENT i18n TEXT:" << i18nText;
205 
206  const QString icon = element.attribute(d->attrIcon);
207  QIcon pix;
208  if (!icon.isEmpty()) {
209  pix = QIcon::fromTheme(icon);
210  }
211 
212  if (parent) {
213  QAction *act = popup->menuAction();
214  if (!icon.isEmpty()) {
215  act->setIcon(pix);
216  }
217  act->setText(i18nText);
218  if (index == -1 || index >= parent->actions().count()) {
219  parent->addAction(act);
220  } else {
221  parent->insertAction(parent->actions().value(index), act);
222  }
223  containerAction = act;
224  containerAction->setObjectName(name);
225  }
226 
227  return popup;
228  }
229 
230  if (tagName == d->tagToolBar) {
231  QString name = element.attribute(d->attrName);
232 
233  KToolBar *bar = static_cast<KToolBar *>(d->m_widget->findChild<KToolBar *>(name));
234  if (!bar) {
235  bar = new KToolBar(name, d->m_widget, false);
236  }
237 
238  if (qobject_cast<KMainWindow *>(d->m_widget)) {
239  if (d->m_client && !d->m_client->xmlFile().isEmpty()) {
240  bar->addXMLGUIClient(d->m_client);
241  }
242  }
243 
244  bar->loadState(element);
245 
246  return bar;
247  }
248 
249  if (tagName == d->tagStatusBar) {
250  KMainWindow *mainWin = qobject_cast<KMainWindow *>(d->m_widget);
251  if (mainWin) {
252  mainWin->statusBar()->show();
253  return mainWin->statusBar();
254  }
255  QStatusBar *bar = new QStatusBar(d->m_widget);
256  return bar;
257  }
258 
259  return 0L;
260 }

262 void KXMLGUIBuilder::removeContainer(QWidget *container, QWidget *parent, QDomElement &element, QAction *containerAction)
263 {
264  // Warning parent can be 0L
265 
266  if (qobject_cast<QMenu *>(container)) {
267  if (parent) {
268  parent->removeAction(containerAction);
269  }
270 
271  delete container;
272  } else if (qobject_cast<KToolBar *>(container)) {
273  KToolBar *tb = static_cast<KToolBar *>(container);
274 
275  tb->saveState(element);
276  delete tb;
277  } else if (qobject_cast<QMenuBar *>(container)) {
278  QMenuBar *mb = static_cast<QMenuBar *>(container);
279  mb->hide();
280  // Don't delete menubar - it can be reused by createContainer.
281  // If you decide that you do need to delete the menubar, make
282  // sure that QMainWindow::d->mb does not point to a deleted
283  // menubar object.
284  } else if (qobject_cast<QStatusBar *>(container)) {
285  if (qobject_cast<KMainWindow *>(d->m_widget)) {
286  container->hide();
287  } else {
288  delete static_cast<QStatusBar *>(container);
289  }
290  } else {
291  qWarning() << "Unhandled container to remove : " << container->metaObject()->className();
292  }
293 }

302 QAction *KXMLGUIBuilder::createCustomElement(QWidget *parent, int index, const QDomElement &element)
303 {
304  QAction *before = 0L;
305  if (index > 0 && index < parent->actions().count()) {
306  before = parent->actions().at(index);
307  }
308 
309  const QString tagName = element.tagName().toLower();
310  if (tagName == d->tagSeparator) {
311  if (QMenu *menu = qobject_cast<QMenu *>(parent)) {
312  // QMenu already cares for leading/trailing/repeated separators
313  // no need to check anything
314  return menu->insertSeparator(before);
315  } else if (QMenuBar *bar = qobject_cast<QMenuBar *>(parent)) {
316  QAction *separatorAction = new QAction(bar);
317  separatorAction->setSeparator(true);
318  bar->insertAction(before, separatorAction);
319  return separatorAction;
320  } else if (KToolBar *bar = qobject_cast<KToolBar *>(parent)) {
321  /* FIXME KAction port - any need to provide a replacement for lineSeparator/normal separator?
322  bool isLineSep = true;
323 
324  QDomNamedNodeMap attributes = element.attributes();
325  unsigned int i = 0;
326  for (; i < attributes.length(); i++ )
327  {
328  QDomAttr attr = attributes.item( i ).toAttr();
329 
330  if ( attr.name().toLower() == d->attrLineSeparator &&
331  attr.value().toLower() == QStringLiteral("false") )
332  {
333  isLineSep = false;
334  break;
335  }
336  }
337 
338  if ( isLineSep )
339  return bar->insertSeparator( index ? bar->actions()[index - 1] : 0L );
340  else*/
341 
342  return bar->insertSeparator(before);
343  }
344  } else if (tagName == d->tagTearOffHandle) {
345  static_cast<QMenu *>(parent)->setTearOffEnabled(true);
346  } else if (tagName == d->tagMenuTitle) {
347  if (QMenu *m = qobject_cast<QMenu *>(parent)) {
348  QString i18nText;
349  const QString text = element.text();
350 
351  if (text.isEmpty()) {
352  i18nText = i18n("No text");
353  } else {
354  QByteArray domain = element.attribute(d->attrDomain).toUtf8();
355  if (domain.isEmpty()) {
356  domain = element.ownerDocument().documentElement().attribute(d->attrDomain).toUtf8();
357  if (domain.isEmpty()) {
358  domain = KLocalizedString::applicationDomain();
359  }
360  }
361  i18nText = i18nd(domain.constData(), qPrintable(text));
362  }
363 
364  QString icon = element.attribute(d->attrIcon);
365  QIcon pix;
366 
367  if (!icon.isEmpty()) {
368  pix = QIcon::fromTheme(icon);
369  }
370 
371  if (!icon.isEmpty()) {
372  return m->insertSection(before, pix, i18nText);
373  } else {
374  return m->insertSection(before, i18nText);
375  }
376  }
377  }
378 
379  QAction *blank = new QAction(parent);
380  blank->setVisible(false);
381  parent->insertAction(before, blank);
382  return blank;
383 }
