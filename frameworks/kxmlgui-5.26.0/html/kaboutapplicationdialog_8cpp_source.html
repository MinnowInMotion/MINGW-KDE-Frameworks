103 
104  //Set up the first page...
105  QString aboutPageText = aboutData.shortDescription() + QLatin1Char('\n');
106 
107  if (!aboutData.otherText().isEmpty()) {
108  aboutPageText += QLatin1Char('\n') + aboutData.otherText() + QLatin1Char('\n');
109  }
110 
111  if (!aboutData.copyrightStatement().isEmpty()) {
112  aboutPageText += QLatin1Char('\n') + aboutData.copyrightStatement() + QLatin1Char('\n');
113  }
114 
115  if (!aboutData.homepage().isEmpty()) {
116  aboutPageText += QLatin1Char('\n') + QStringLiteral("<a href=\"%1\">%1</a>").arg(aboutData.homepage()) + QLatin1Char('\n');

118  aboutPageText = aboutPageText.trimmed();

120  QLabel *aboutLabel = new QLabel;

140  aboutLayout->addWidget(showLicenseLabel);

147 
148  tabWidget->addTab(aboutWidget, i18n("&About"));

154  versionLabel->setTextInteractionFlags(Qt::TextSelectableByMouse);

185  bugsLabel->setOpenExternalLinks(true);

314 void KAboutApplicationDialog::Private::_k_showLicense(const QString &number)
315 {
316  QDialog *dialog = new QDialog(q);
317  dialog->setAttribute(Qt::WA_DeleteOnClose);
318  QVBoxLayout *layout = new QVBoxLayout;
319  dialog->setLayout(layout);
320 
321  dialog->setWindowTitle(i18n("License Agreement"));
322 
323  const QFont font = QFontDatabase::systemFont(QFontDatabase::FixedFont);
324  QFontMetrics metrics(font);
325 
326  const QString licenseText = aboutData.licenses().at(number.toInt()).text();
327  QTextBrowser *licenseBrowser = new QTextBrowser(dialog);
328  licenseBrowser->setFont(font);
329  licenseBrowser->setLineWrapMode(QTextEdit::NoWrap);
330  licenseBrowser->setText(licenseText);
331  layout->addWidget(licenseBrowser);
332 
333  QDialogButtonBox *buttonBox = new QDialogButtonBox(dialog);
334  buttonBox->setStandardButtons(QDialogButtonBox::Close);
335  connect(buttonBox, SIGNAL(accepted()), dialog, SLOT(accept()));
336  connect(buttonBox, SIGNAL(rejected()), dialog, SLOT(reject()));
337  layout->addWidget(buttonBox);
338 
339  // try to set up the dialog such that the full width of the
340  // document is visible without horizontal scroll-bars being required
341  const int marginHint = dialog->style()->pixelMetric(QStyle::PM_DefaultChildMargin);
342  const qreal idealWidth = licenseBrowser->document()->idealWidth() + (2 * marginHint)
343  + licenseBrowser->verticalScrollBar()->width() * 2;
344 
345  // try to allow enough height for a reasonable number of lines to be shown
346  const int idealHeight = metrics.height() * 30;
347 
348  dialog->resize(dialog->sizeHint().expandedTo(QSize(qRound(idealWidth), idealHeight)));
349  dialog->adjustSize();
350  dialog->show();
351 }
