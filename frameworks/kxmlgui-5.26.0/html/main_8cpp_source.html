39 void BugMailer::slotError(int errornum)
40 {
41  QString lstr;
42 
43  switch (errornum) {
44  case SMTP::ConnectError:
45  lstr = i18n("Error connecting to server.");
46  break;
47  case SMTP::NotConnected:
48  lstr = i18n("Not connected.");
49  break;
50  case SMTP::ConnectTimeout:
51  lstr = i18n("Connection timed out.");
52  break;
53  case SMTP::InteractTimeout:
54  lstr = i18n("Time out waiting for server interaction.");
55  break;
56  default:
57  lstr = sm->getLastLine().trimmed();
58  lstr = i18n("Server said: \"%1\"", lstr);
59  }
60  // qDebug() << lstr;
61 
62  fputs(lstr.toUtf8().data(), stdout);
63  fflush(stdout);
64 
65  qApp->exit(1);
66 }

74 int main(int argc, char **argv)
75 {
76 
77  QCoreApplication a(argc, argv);
78  a.setApplicationName(QStringLiteral("ksendbugmail"));
79  a.setApplicationVersion(QStringLiteral("1.0"));
80 
81  KLocalizedString::setApplicationDomain("kxmlgui5");
82 
83  //d.addAuthor(ki18n("Stephan Kulow"), ki18n("Author"), "coolo@kde.org");
84 
85  QString subject, recipient;
86  {
87  QCommandLineParser parser;
88  parser.addVersionOption();
89  parser.setApplicationDescription(i18n("Sends a bug report by email."));
90  parser.addHelpOption();
91  parser.addOption(QCommandLineOption(QStringList() << QStringLiteral("subject"), i18n("The subject line of the email."), QStringLiteral("argument")));
92  parser.addOption(QCommandLineOption(QStringList() << QStringLiteral("recipient"), i18n("The email address to send the bug report to."), QStringLiteral("argument"), QStringLiteral("submit@bugs.kde.org")));
93  parser.process(a);
94  recipient = parser.value(QStringLiteral("recipient"));
95  subject = parser.value(QStringLiteral("subject"));
96  }
97  if (recipient.isEmpty()) {
98  recipient = QStringLiteral("submit@bugs.kde.org");
99  } else {
100  if (recipient.at(0) == '\'') {
101  recipient = recipient.mid(1).left(recipient.length() - 2);
102  }
103  }
104  // qDebug() << "recp" << recipient;
105 
106  if (subject.isEmpty()) {
107  subject = QStringLiteral("(no subject)");
108  } else {
109  if (subject.at(0) == '\'') {
110  subject = subject.mid(1).left(subject.length() - 2);
111  }
112  }
113  QTextStream input(stdin, QIODevice::ReadOnly);
114  input.setCodec("UTF-8");
115  QString text, line;
116  while (!input.atEnd()) {
117  line = input.readLine();
118  text += line + "\r\n";
119  }
120  // qDebug() << text;
121 
122  KEMailSettings emailConfig;
123  emailConfig.setProfile(emailConfig.defaultProfileName());
124  QString fromaddr = emailConfig.getSetting(KEMailSettings::EmailAddress);
125  if (!fromaddr.isEmpty()) {
126  QString name = emailConfig.getSetting(KEMailSettings::RealName);
127  if (!name.isEmpty()) {
128  fromaddr = name + QLatin1String(" <") + fromaddr + QLatin1String(">");
129  }
130  } else {
131  fromaddr = SystemInformation::userName();
132  fromaddr += '@';
133  fromaddr += QHostInfo::localHostName();
134  }
135  // qDebug() << "fromaddr \"" << fromaddr << "\"";
136 
137  QString server = emailConfig.getSetting(KEMailSettings::OutServer);
138  if (server.isEmpty()) {
139  server = QStringLiteral("bugs.kde.org");
140  }
141 
142  SMTP *sm = new SMTP;
143  BugMailer bm(sm);
144 
145  QObject::connect(sm, SIGNAL(messageSent()), &bm, SLOT(slotSend()));
146  QObject::connect(sm, SIGNAL(error(int)), &bm, SLOT(slotError(int)));
147  sm->setServerHost(server);
148  sm->setPort(25);
149  sm->setSenderAddress(fromaddr);
150  sm->setRecipientAddress(recipient);
151  sm->setMessageSubject(subject);
152  sm->setMessageHeader(QStringLiteral("From: %1\r\nTo: %2\r\n").arg(fromaddr).arg(QString(recipient)));
153  sm->setMessageBody(text);
154  sm->sendMessage();
155 
156  int r = a.exec();
157  // qDebug() << "execing " << r;
158  delete sm;
159  return r;
160 }
