244 QAction *KActionCollection::addAction(const QString &name, QAction *action)
245 {
246  if (!action) {
247  return action;
248  }
249 
250  const QString objectName = action->objectName();
251  QString indexName = name;
252 
253  if (indexName.isEmpty()) {
254  // No name provided. Use the objectName.
255  indexName = objectName;
256 
257  } else {
258 
259  // A name was provided. Check against objectName.
260  if ((!objectName.isEmpty()) && (objectName != indexName)) {
261  // The user specified a new name and the action already has a
262  // different one. The objectName is used for saving shortcut
263  // settings to disk. Both for local and global shortcuts.
264  qCDebug(DEBUG_KXMLGUI) << "Registering action " << objectName << " under new name " << indexName;
265  // If there is a global shortcuts it's a very bad idea.
266 #if HAVE_GLOBALACCEL
267  if (KGlobalAccel::self()->hasShortcut(action)) {
268  // In debug mode assert
269  Q_ASSERT(!KGlobalAccel::self()->hasShortcut(action));
270  // In release mode keep the old name
271  qCritical() << "Changing action name from " << objectName << " to " << indexName << "\nignored because of active global shortcut.";
272  indexName = objectName;
273  }
274 #endif
275  }
276 
277  // Set the new name
278  action->setObjectName(indexName);
279  }
280 
281  // No name provided and the action had no name. Make one up. This will not
282  // work when trying to save shortcuts. Both local and global shortcuts.
283  if (indexName.isEmpty()) {
284  indexName = indexName.sprintf("unnamed-%p", (void *)action);
285  action->setObjectName(indexName);
286  }
287 
288  // From now on the objectName has to have a value. Else we cannot safely
289  // remove actions.
290  Q_ASSERT(!action->objectName().isEmpty());
291 
292  // look if we already have THIS action under THIS name ;)
293  if (d->actionByName.value(indexName, 0) == action) {
294  // This is not a multi map!
295  Q_ASSERT(d->actionByName.count(indexName) == 1);
296  return action;
297  }
298 
299  if (!KAuthorized::authorizeAction(indexName)) {
300  // Disable this action
301  action->setEnabled(false);
302  action->setVisible(false);
303  action->blockSignals(true);
304  }
305 
306  // Check if we have another action under this name
307  if (QAction *oldAction = d->actionByName.value(indexName)) {
308  takeAction(oldAction);
309  }
310 
311  // Check if we have this action under a different name.
312  // Not using takeAction because we don't want to remove it from categories,
313  // and because it has the new name already.
314  const int oldIndex = d->actions.indexOf(action);
315  if (oldIndex != -1) {
316  d->actionByName.remove(d->actionByName.key(action));
317  d->actions.removeAt(oldIndex);
318  }
319 
320  // Add action to our lists.
321  d->actionByName.insert(indexName, action);
322  d->actions.append(action);
323 
324  Q_FOREACH (QWidget *widget, d->associatedWidgets) {
325  widget->addAction(action);
326  }
327 
328  connect(action, SIGNAL(destroyed(QObject*)), SLOT(_k_actionDestroyed(QObject*)));
329 
330  d->setComponentForAction(action);
331 
332  if (d->connectHovered) {
333  connect(action, SIGNAL(hovered()), SLOT(slotActionHovered()));
334  }
335 
336  if (d->connectTriggered) {
337  connect(action, SIGNAL(triggered(bool)), SLOT(slotActionTriggered()));
338  }
339 
340  emit inserted(action);
341  return action;
342 }

521 void KActionCollection::exportGlobalShortcuts(KConfigGroup *config, bool writeAll) const
522 {
523 #if HAVE_GLOBALACCEL
524  Q_ASSERT(config);
525  if (!config) {
526  return;
527  }
528 
529  QList<QAction *> writeActions = actions();
530 
531  for (QMap<QString, QAction *>::ConstIterator it = d->actionByName.constBegin();
532  it != d->actionByName.constEnd(); ++it) {
533 
534  QAction *action = it.value();
535  if (!action) {
536  continue;
537  }
538  QString actionName = it.key();
539 
540  // If the action name starts with unnamed- spit out a warning. That name
541  // will change at will and will break loading writing
542  if (actionName.startsWith(QLatin1String("unnamed-"))) {
543  qCritical() << "Skipped exporting Shortcut for action without name " << action->text() << "!";
544  continue;
545  }
546 
547  if (isShortcutsConfigurable(action) && KGlobalAccel::self()->hasShortcut(action)) {
548  bool bConfigHasAction = !config->readEntry(actionName, QString()).isEmpty();
549  bool bSameAsDefault = (KGlobalAccel::self()->shortcut(action) == KGlobalAccel::self()->defaultShortcut(action));
550  // If we're using a global config or this setting
551  // differs from the default, then we want to write.
552  KConfigGroup::WriteConfigFlags flags = KConfigGroup::Persistent;
553  if (configIsGlobal()) {
554  flags |= KConfigGroup::Global;
555  }
556  if (writeAll || !bSameAsDefault) {
557  QString s = QKeySequence::listToString(KGlobalAccel::self()->shortcut(action));
558  if (s.isEmpty()) {
559  s = QStringLiteral("none");
560  }
561  qCDebug(DEBUG_KXMLGUI) << "\twriting " << actionName << " = " << s;
562  config->writeEntry(actionName, s, flags);
563  }
564  // Otherwise, this key is the same as default
565  // but exists in config file. Remove it.
566  else if (bConfigHasAction) {
567  qCDebug(DEBUG_KXMLGUI) << "\tremoving " << actionName << " because == default";
568  config->deleteEntry(actionName, flags);
569  }
570  }
571  }
572 
573  config->sync();
574 #else
575  Q_UNUSED(config);
576  Q_UNUSED(writeAll);
577 #endif
578 }

580 bool KActionCollectionPrivate::writeKXMLGUIConfigFile()
581 {
582  const KXMLGUIClient *kxmlguiClient = q->parentGUIClient();
583  // return false if there is no KXMLGUIClient
584  if (!kxmlguiClient || kxmlguiClient->xmlFile().isEmpty()) {
585  return false;
586  }
587 
588  qCDebug(DEBUG_KXMLGUI) << "xmlFile=" << kxmlguiClient->xmlFile();
589 
590  QString attrShortcut = QStringLiteral("shortcut");
591 
592  // Read XML file
593  QString sXml(KXMLGUIFactory::readConfigFile(kxmlguiClient->xmlFile(), q->componentName()));
594  QDomDocument doc;
595  doc.setContent(sXml);
596 
597  // Process XML data
598 
599  // Get hold of ActionProperties tag
600  QDomElement elem = KXMLGUIFactory::actionPropertiesElement(doc);
601 
602  // now, iterate through our actions
603  for (QMap<QString, QAction *>::ConstIterator it = actionByName.constBegin();
604  it != actionByName.constEnd(); ++it) {
605  QAction *action = it.value();
606  if (!action) {
607  continue;
608  }
609 
610  QString actionName = it.key();
611 
612  // If the action name starts with unnamed- spit out a warning and ignore
613  // it. That name will change at will and will break loading writing
614  if (actionName.startsWith(QLatin1String("unnamed-"))) {
615  qCritical() << "Skipped writing shortcut for action " << actionName << "(" << action->text() << ")!";
616  continue;
617  }
618 
619  bool bSameAsDefault = (action->shortcuts() == q->defaultShortcuts(action));
620  qCDebug(DEBUG_KXMLGUI) << "name = " << actionName
621  << " shortcut = " << QKeySequence::listToString(action->shortcuts())
622 #if HAVE_GLOBALACCEL
623  << " globalshortcut = " << QKeySequence::listToString(KGlobalAccel::self()->shortcut(action))
624 #endif
625  << " def = " << QKeySequence::listToString(q->defaultShortcuts(action));
626 
627  // now see if this element already exists
628  // and create it if necessary (unless bSameAsDefault)
629  QDomElement act_elem = KXMLGUIFactory::findActionByName(elem, actionName, !bSameAsDefault);
630  if (act_elem.isNull()) {
631  continue;
632  }
633 
634  if (bSameAsDefault) {
635  act_elem.removeAttribute(attrShortcut);
636  //qCDebug(DEBUG_KXMLGUI) << "act_elem.attributes().count() = " << act_elem.attributes().count();
637  if (act_elem.attributes().count() == 1) {
638  elem.removeChild(act_elem);
639  }
640  } else {
641  act_elem.setAttribute(attrShortcut, QKeySequence::listToString(action->shortcuts()));
642  }
643  }
644 
645  // Write back to XML file
646  KXMLGUIFactory::saveConfigFile(doc, kxmlguiClient->localXMLFile(), q->componentName());
647  return true;
648 }

650 void KActionCollection::writeSettings(KConfigGroup *config, bool writeAll, QAction *oneAction) const
651 {
652  // If the caller didn't provide a config group we try to save the KXMLGUI
653  // Configuration file. If that succeeds we are finished.
654  if (config == 0 && d->writeKXMLGUIConfigFile()) {
655  return;
656  }
657 
658  KConfigGroup cg(KSharedConfig::openConfig(), configGroup());
659  if (!config) {
660  config = &cg;
661  }
662 
663  QList<QAction *> writeActions;
664  if (oneAction) {
665  writeActions.append(oneAction);
666  } else {
667  writeActions = actions();
668  }
669 
670  for (QMap<QString, QAction *>::ConstIterator it = d->actionByName.constBegin();
671  it != d->actionByName.constEnd(); ++it) {
672 
673  QAction *action = it.value();
674  if (!action) {
675  continue;
676  }
677 
678  QString actionName = it.key();
679 
680  // If the action name starts with unnamed- spit out a warning and ignore
681  // it. That name will change at will and will break loading writing
682  if (actionName.startsWith(QLatin1String("unnamed-"))) {
683  qCritical() << "Skipped saving Shortcut for action without name " << action->text() << "!";
684  continue;
685  }
686 
687  // Write the shortcut
688  if (isShortcutsConfigurable(action)) {
689  bool bConfigHasAction = !config->readEntry(actionName, QString()).isEmpty();
690  bool bSameAsDefault = (action->shortcuts() == defaultShortcuts(action));
691  // If we're using a global config or this setting
692  // differs from the default, then we want to write.
693  KConfigGroup::WriteConfigFlags flags = KConfigGroup::Persistent;
694 
695  // Honor the configIsGlobal() setting
696  if (configIsGlobal()) {
697  flags |= KConfigGroup::Global;
698  }
699 
700  if (writeAll || !bSameAsDefault) {
701  // We are instructed to write all shortcuts or the shortcut is
702  // not set to its default value. Write it
703  QString s = QKeySequence::listToString(action->shortcuts());
704  if (s.isEmpty()) {
705  s = QStringLiteral("none");
706  }
707  qCDebug(DEBUG_KXMLGUI) << "\twriting " << actionName << " = " << s;
708  config->writeEntry(actionName, s, flags);
709 
710  } else if (bConfigHasAction) {
711  // Otherwise, this key is the same as default but exists in
712  // config file. Remove it.
713  qCDebug(DEBUG_KXMLGUI) << "\tremoving " << actionName << " because == default";
714  config->deleteEntry(actionName, flags);
715  }
716  }
717  }
718 
719  config->sync();
720 }

819 QAction *KActionCollectionPrivate::unlistAction(QAction *action)
820 {
821  // ATTENTION:
822  // This method is called with an QObject formerly known as a QAction
823  // during _k_actionDestroyed(). So don't do fancy stuff here that needs a
824  // real QAction!
825 
826  // Get the index for the action
827  int index = actions.indexOf(action);
828 
829  // Action not found.
830  if (index == -1) {
831  return NULL;
832  }
833 
834  // An action collection can't have the same action twice.
835  Q_ASSERT(actions.indexOf(action, index + 1) == -1);
836 
837  // Get the actions name
838  const QString name = action->objectName();
839 
840  // Remove the action
841  actionByName.remove(name);
842  actions.removeAt(index);
843 
844  // Remove the action from the categories. Should be only one
845  QList<KActionCategory *> categories = q->findChildren<KActionCategory *>();
846  Q_FOREACH (KActionCategory *category, categories) {
847  category->unlistAction(action);
848  }
849 
850  return action;
851 }
