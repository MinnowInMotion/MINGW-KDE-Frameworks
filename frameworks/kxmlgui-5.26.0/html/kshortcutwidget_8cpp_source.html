102 //slot
103 void KShortcutWidget::setShortcut(const QList<QKeySequence> &newSc)
104 {
105  if (newSc == d->cut) {
106  return;
107  }
108 
109  d->holdChangedSignal = true;
110 
111  if (!newSc.isEmpty()) {
112  d->ui.priEditor->setKeySequence(newSc.first());
113  }
114 
115  if (newSc.size() > 1) {
116  d->ui.altEditor->setKeySequence(newSc.at(1));
117  }
118 
119  d->holdChangedSignal = false;
120 
121  emit shortcutChanged(d->cut);
122 }
123

130 //private slot
131 void KShortcutWidgetPrivate::priKeySequenceChanged(const QKeySequence &seq)
132 {
133  if (cut.isEmpty()) {
134  cut << seq;
135  } else {
136  cut[0] = seq;
137  }
138 
139  if (!holdChangedSignal) {
140  emit q->shortcutChanged(cut);
141  }
142 }
143 
144 //private slot
145 void KShortcutWidgetPrivate::altKeySequenceChanged(const QKeySequence &seq)
146 {
147  if (cut.size() <= 1) {
148  cut << seq;
149  } else {
150  cut[1] = seq;
151  }
152 
153  if (!holdChangedSignal) {
154  emit q->shortcutChanged(cut);
155  }
156 }
