43 KShortcutSchemesEditor::KShortcutSchemesEditor(KShortcutsDialog *parent)
44  : QGroupBox(i18n("Shortcut Schemes"), parent), m_dialog(parent)
45 {
46  KConfigGroup group(KSharedConfig::openConfig(), "Shortcut Schemes");
47 
48  QStringList schemes;
49  schemes << QStringLiteral("Default");
50  // List files in the shortcuts subdir, each one is a scheme. See KShortcutSchemesHelper::{shortcutSchemeFileName,exportActionCollection}
51  const QStringList shortcutsDirs = QStandardPaths::locateAll(QStandardPaths::GenericDataLocation, QCoreApplication::applicationName() + QStringLiteral("/shortcuts"), QStandardPaths::LocateDirectory);
52  qCDebug(DEBUG_KXMLGUI) << "shortcut scheme dirs:" << shortcutsDirs;
53  Q_FOREACH (const QString &dir, shortcutsDirs) {
54  Q_FOREACH (const QString &file, QDir(dir).entryList(QDir::Files | QDir::NoDotAndDotDot)) {
55  qCDebug(DEBUG_KXMLGUI) << "shortcut scheme file:" << file;
56  schemes << file;
57  }
58  }
59 
60  const QString currentScheme = group.readEntry("Current Scheme", "Default");
61  qCDebug(DEBUG_KXMLGUI) << "Current Scheme" << currentScheme;
62 
63  QHBoxLayout *l = new QHBoxLayout(this);
64 
65  QLabel *schemesLabel = new QLabel(i18n("Current scheme:"), this);
66  l->addWidget(schemesLabel);
67 
68  m_schemesList = new QComboBox(this);
69  m_schemesList->setEditable(false);
70  m_schemesList->addItems(schemes);
71  m_schemesList->setSizeAdjustPolicy(QComboBox::AdjustToContents);
72  const int schemeIdx = m_schemesList->findText(currentScheme);
73  if (schemeIdx > -1) {
74  m_schemesList->setCurrentIndex(schemeIdx);
75  } else {
76  qCWarning(DEBUG_KXMLGUI) << "Current scheme" << currentScheme << "not found in" << shortcutsDirs;
77  }
78  schemesLabel->setBuddy(m_schemesList);
79  l->addWidget(m_schemesList);
80 
81  m_newScheme = new QPushButton(i18n("New..."));
82  l->addWidget(m_newScheme);
83 
84  m_deleteScheme = new QPushButton(i18n("Delete"));
85  l->addWidget(m_deleteScheme);
86 
87  QPushButton *moreActions = new QPushButton(i18n("More Actions"));
88  l->addWidget(moreActions);
89 
90  QMenu *moreActionsMenu = new QMenu(this);
91  moreActionsMenu->addAction(i18n("Save shortcuts to scheme"),
92  this, SLOT(saveAsDefaultsForScheme()));
93  moreActionsMenu->addAction(i18n("Export Scheme..."),
94  this, SLOT(exportShortcutsScheme()));
95  moreActionsMenu->addAction(i18n("Import Scheme..."),
96  this, SLOT(importShortcutsScheme()));
97  moreActions->setMenu(moreActionsMenu);
98 
99  l->addStretch(1);
100 
101  connect(m_schemesList, SIGNAL(activated(QString)),
102  this, SIGNAL(shortcutsSchemeChanged(QString)));
103  connect(m_newScheme, SIGNAL(clicked()), this, SLOT(newScheme()));
104  connect(m_deleteScheme, SIGNAL(clicked()), this, SLOT(deleteScheme()));
105  updateDeleteButton();
106 }

108 void KShortcutSchemesEditor::newScheme()
109 {
110  bool ok;
111  const QString newName = QInputDialog::getText(this, i18n("Name for New Scheme"),
112  i18n("Name for new scheme:"), QLineEdit::Normal, i18n("New Scheme"), &ok);
113  if (!ok) {
114  return;
115  }
116 
117  if (m_schemesList->findText(newName) != -1) {
118  KMessageBox::sorry(this, i18n("A scheme with this name already exists."));
119  return;
120  }
121 
122  const QString newSchemeFileName = KShortcutSchemesHelper::writableApplicationShortcutSchemeFileName(newName);
123  QDir().mkpath(QFileInfo(newSchemeFileName).absolutePath());
124  QFile schemeFile(newSchemeFileName);
125  if (!schemeFile.open(QFile::WriteOnly | QFile::Truncate)) {
126  qCWarning(DEBUG_KXMLGUI) << "Couldn't write to" << newSchemeFileName;
127  return;
128  }
129 
130  QDomDocument doc;
131  QDomElement docElem = doc.createElement(QStringLiteral("kpartgui"));
132  doc.appendChild(docElem);
133  QDomElement elem = doc.createElement(QStringLiteral("ActionProperties"));
134  docElem.appendChild(elem);
135 
136  QTextStream out(&schemeFile);
137  out << doc.toString(4);
138 
139  m_schemesList->addItem(newName);
140  m_schemesList->setCurrentIndex(m_schemesList->findText(newName));
141  updateDeleteButton();
142  emit shortcutsSchemeChanged(newName);
143 }

145 void KShortcutSchemesEditor::deleteScheme()
146 {
147  if (KMessageBox::questionYesNo(this,
148  i18n("Do you really want to delete the scheme %1?\n\
149 Note that this will not remove any system wide shortcut schemes.", currentScheme())) == KMessageBox::No) {
150  return;
151  }
152 
153  //delete the scheme for the app itself
154  QFile::remove(KShortcutSchemesHelper::writableApplicationShortcutSchemeFileName(currentScheme()));
155 
156  //delete all scheme files we can find for xmlguiclients in the user directories
157  foreach (KActionCollection *collection, m_dialog->actionCollections()) {
158  const KXMLGUIClient *client = collection->parentGUIClient();
159  if (!client) {
160  continue;
161  }
162  QFile::remove(KShortcutSchemesHelper::writableShortcutSchemeFileName(client->componentName(), currentScheme()));
163  }
164 
165  m_schemesList->removeItem(m_schemesList->findText(currentScheme()));
166  updateDeleteButton();
167  emit shortcutsSchemeChanged(currentScheme());
168 }
