395 void KKeySequenceWidget::applyStealShortcut()
396 {
397  QSet<KActionCollection *> changedCollections;
398 
399  Q_FOREACH (QAction *stealAction, d->stealActions) {
400 
401  // Stealing a shortcut means setting it to an empty one.
402  stealAction->setShortcuts(QList<QKeySequence>());
403 
404  // The following code will find the action we are about to
405  // steal from and save it's actioncollection.
406  KActionCollection *parentCollection = 0;
407  foreach (KActionCollection *collection, d->checkActionCollections) {
408  if (collection->actions().contains(stealAction)) {
409  parentCollection = collection;
410  break;
411  }
412  }
413 
414  // Remember the changed collection
415  if (parentCollection) {
416  changedCollections.insert(parentCollection);
417  }
418  }
419 
420  Q_FOREACH (KActionCollection *col, changedCollections) {
421  col->writeSettings();
422  }
423 
424  d->stealActions.clear();
425 }

445 {
446  modifierlessTimeout.stop();

468 bool KKeySequenceWidgetPrivate::conflictWithGlobalShortcuts(const QKeySequence &keySequence)
469 {
470 #ifdef Q_OS_WIN
471  //on windows F12 is reserved by the debugger at all times, so we can't use it for a global shortcut
472  if (KKeySequenceWidget::GlobalShortcuts && keySequence.toString().contains(QStringLiteral("F12"))) {
473  QString title = i18n("Reserved Shortcut");
474  QString message = i18n("The F12 key is reserved on Windows, so cannot be used for a global shortcut.\n"
475  "Please choose another one.");
476 
477  KMessageBox::sorry(q, message, title);
478  return false;
479  }
480 #endif
481 
482 #if HAVE_GLOBALACCEL
483  if (!(checkAgainstShortcutTypes & KKeySequenceWidget::GlobalShortcuts)) {
484  return false;
485  }
486 
487  // Global shortcuts are on key+modifier shortcuts. They can clash with
488  // each of the keys of a multi key shortcut.
489  QHash<QKeySequence, QList<KGlobalShortcutInfo> > others;
490  for (int i = 0; i < keySequence.count(); ++i) {
491  QKeySequence tmp(keySequence[i]);
492 
493  if (!KGlobalAccel::isGlobalShortcutAvailable(tmp, componentName)) {
494  others.insert(tmp, KGlobalAccel::getGlobalShortcutsByKey(tmp));
495  }
496  }
497 
498  if (!others.isEmpty()
499  && !promptStealShortcutSystemwide(q, others, keySequence)) {
500  return true;
501  }
502 
503  // The user approved stealing the shortcut. We have to steal
504  // it immediately because KAction::setGlobalShortcut() refuses
505  // to set a global shortcut that is already used. There is no
506  // error it just silently fails. So be nice because this is
507  // most likely the first action that is done in the slot
508  // listening to keySequenceChanged().
509  for (int i = 0; i < keySequence.count(); ++i) {
510  KGlobalAccel::stealShortcutSystemwide(keySequence[i]);
511  }
512  return false;
513 #else
514  Q_UNUSED(keySequence);
515  return false;
516 #endif
517 }

519 static bool shortcutsConflictWith(const QList<QKeySequence> &shortcuts, const QKeySequence &needle)
520 {
521  if (needle.isEmpty()) {
522  return false;
523  }
524 
525  foreach (const QKeySequence &sequence, shortcuts) {
526  if (sequence.isEmpty()) {
527  continue;
528  }
529 
530  if (sequence.matches(needle) != QKeySequence::NoMatch
531  || needle.matches(sequence) != QKeySequence::NoMatch) {
532  return true;
533  }
534  }
535 
536  return false;
537 }

539 bool KKeySequenceWidgetPrivate::conflictWithLocalShortcuts(const QKeySequence &keySequence)
540 {
541  if (!(checkAgainstShortcutTypes & KKeySequenceWidget::LocalShortcuts)) {
542  return false;
543  }
544 
545  // We have actions both in the deprecated checkList and the
546  // checkActionCollections list. Add all the actions to a single list to
547  // be able to process them in a single loop below.
548  // Note that this can't be done in setCheckActionCollections(), because we
549  // keep pointers to the action collections, and between the call to
550  // setCheckActionCollections() and this function some actions might already be
551  // removed from the collection again.
552  QList<QAction *> allActions;
553  allActions += checkList;
554  foreach (KActionCollection *collection, checkActionCollections) {
555  allActions += collection->actions();
556  }
557 
558  // Because of multikey shortcuts we can have clashes with many shortcuts.
559  //
560  // Example 1:
561  //
562  // Application currently uses 'CTRL-X,a', 'CTRL-X,f' and 'CTRL-X,CTRL-F'
563  // and the user wants to use 'CTRL-X'. 'CTRL-X' will only trigger as
564  // 'activatedAmbiguously()' for obvious reasons.
565  //
566  // Example 2:
567  //
568  // Application currently uses 'CTRL-X'. User wants to use 'CTRL-X,CTRL-F'.
569  // This will shadow 'CTRL-X' for the same reason as above.
570  //
571  // Example 3:
572  //
573  // Some weird combination of Example 1 and 2 with three shortcuts using
574  // 1/2/3 key shortcuts. I think you can imagine.
575  QList<QAction *> conflictingActions;
576 
577  //find conflicting shortcuts with existing actions
578  foreach (QAction *qaction, allActions) {
579  if (shortcutsConflictWith(qaction->shortcuts(), keySequence)) {
580  // A conflict with a KAction. If that action is configurable
581  // ask the user what to do. If not reject this keySequence.
582  if (checkActionCollections.first()->isShortcutsConfigurable(qaction)) {
583  conflictingActions.append(qaction);
584  } else {
585  wontStealShortcut(qaction, keySequence);
586  return true;
587  }
588  }
589  }
590 
591  if (conflictingActions.isEmpty()) {
592  // No conflicting shortcuts found.
593  return false;
594  }
595 
596  if (stealShortcuts(conflictingActions, keySequence)) {
597  stealActions = conflictingActions;
598  // Announce that the user
599  // agreed
600  Q_FOREACH (QAction *stealAction, stealActions) {
601  emit q->stealShortcut(
602  keySequence,
603  stealAction);
604  }
605  return false;
606  } else {
607  return true;
608  }
609 }

638 void KKeySequenceWidgetPrivate::updateShortcutDisplay()
639 {
640  //empty string if no non-modifier was pressed
641  QString s = keySequence.toString(QKeySequence::NativeText);
642  s.replace(QLatin1Char('&'), QStringLiteral("&&"));
643 
644  if (isRecording) {
645  if (modifierKeys) {
646  if (!s.isEmpty()) {
647  s.append(QLatin1Char(','));
648  }
649  if (modifierKeys & Qt::META) {
650  s += KKeyServer::modToStringUser(Qt::META) + QLatin1Char('+');
651  }
652 #if defined(Q_OS_MAC)
653  if (modifierKeys & Qt::ALT) {
654  s += KKeyServer::modToStringUser(Qt::ALT) + QLatin1Char('+');
655  }
656  if (modifierKeys & Qt::CTRL) {
657  s += KKeyServer::modToStringUser(Qt::CTRL) + QLatin1Char('+');
658  }
659 #else
660  if (modifierKeys & Qt::CTRL) {
661  s += KKeyServer::modToStringUser(Qt::CTRL) + QLatin1Char('+');
662  }
663  if (modifierKeys & Qt::ALT) {
664  s += KKeyServer::modToStringUser(Qt::ALT) + QLatin1Char('+');
665  }
666 #endif
667  if (modifierKeys & Qt::SHIFT) {
668  s += KKeyServer::modToStringUser(Qt::SHIFT) + QLatin1Char('+');
669  }
670 
671  } else if (nKey == 0) {
672  s = i18nc("What the user inputs now will be taken as the new shortcut", "Input");
673  }
674  //make it clear that input is still going on
675  s.append(QStringLiteral(" ..."));
676  }
677 
678  if (s.isEmpty()) {
679  s = i18nc("No shortcut defined", "None");
680  }
681 
682  s.prepend(QLatin1Char(' '));
683  s.append(QLatin1Char(' '));
684  keyButton->setText(s);
685 }

719  int keyQt = e->key();
720  if (keyQt == -1) {
721  // Qt sometimes returns garbage keycodes, I observed -1, if it doesn't know a key.
722  // We cannot do anything useful with those (several keys have -1, indistinguishable)
723  // and QKeySequence.toString() will also yield a garbage string.
724  KMessageBox::sorry(this,
725  i18n("The key you just pressed is not supported by Qt."),
726  i18n("Unsupported Key"));
727  return d->cancelRecording();
728  }
729 
730  uint newModifiers = e->modifiers() & (Qt::SHIFT | Qt::CTRL | Qt::ALT | Qt::META);
731 
732  //don't have the return or space key appear as first key of the sequence when they
733  //were pressed to start editing - catch and them and imitate their effect
734  if (!d->isRecording && ((keyQt == Qt::Key_Return || keyQt == Qt::Key_Space))) {
735  d->startRecording();
736  d->modifierKeys = newModifiers;
737  d->updateShortcutDisplay();
738  return;
739  }
740 
741  // We get events even if recording isn't active.
742  if (!d->isRecording) {
743  return QPushButton::keyPressEvent(e);
744  }
745 
746  e->accept();
747  d->modifierKeys = newModifiers;
748 
749  switch (keyQt) {
750  case Qt::Key_AltGr: //or else we get unicode salad
751  return;
752  case Qt::Key_Shift:
753  case Qt::Key_Control:
754  case Qt::Key_Alt:
755  case Qt::Key_Meta:
756  case Qt::Key_Super_L:
757  case Qt::Key_Super_R:
758  d->controlModifierlessTimout();
759  d->updateShortcutDisplay();
760  break;
761  default:
762 
763  if (d->nKey == 0 && !(d->modifierKeys & ~Qt::SHIFT)) {
764  // It's the first key and no modifier pressed. Check if this is
765  // allowed
766  if (!(KKeySequenceWidgetPrivate::isOkWhenModifierless(keyQt)
767  || d->allowModifierless)) {
768  // No it's not
769  return;
770  }
771  }
772 
773  // We now have a valid key press.
774  if (keyQt) {
775  if ((keyQt == Qt::Key_Backtab) && (d->modifierKeys & Qt::SHIFT)) {
776  keyQt = Qt::Key_Tab | d->modifierKeys;
777  } else if (KKeyServer::isShiftAsModifierAllowed(keyQt)) {
778  keyQt |= d->modifierKeys;
779  } else {
780  keyQt |= (d->modifierKeys & ~Qt::SHIFT);
781  }
782 
783  if (d->nKey == 0) {
784  d->keySequence = QKeySequence(keyQt);
785  } else {
786  d->keySequence =
787  KKeySequenceWidgetPrivate::appendToSequence(d->keySequence, keyQt);
788  }
789 
790  d->nKey++;
791  if ((!d->multiKeyShortcutsAllowed) || (d->nKey >= 4)) {
792  d->doneRecording();
793  return;
794  }
795  d->controlModifierlessTimout();
796  d->updateShortcutDisplay();
797  }
798  }
799 }
