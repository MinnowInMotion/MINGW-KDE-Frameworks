109 void ShortcutEditWidget::defaultToggled(bool checked)
110 {
111  if (m_isUpdating) {
112  return;
113  }
114 
115  m_isUpdating = true;
116  if (checked) {
117  // The default key sequence should be activated. We check first if this is
118  // possible.
119  if (m_customEditor->isKeySequenceAvailable(m_defaultKeySequence)) {
120  // Clear the customs widget
121  m_customEditor->clearKeySequence();
122  emit keySequenceChanged(m_defaultKeySequence);
123  } else {
124  // We tried to switch to the default key sequence and failed. Go
125  // back.
126  m_customRadio->setChecked(true);
127  }
128  } else {
129  // The empty key sequence is always valid
130  emit keySequenceChanged(QKeySequence());
131  }
132  m_isUpdating = false;
133 }

168 //slot
169 void ShortcutEditWidget::setCustom(const QKeySequence &seq)
170 {
171  if (m_isUpdating) {
172  return;
173  }
174 
175  // seq is a const reference to a private variable of KKeySequenceWidget.
176  // Somewhere below we possible change that one. But we want to emit seq
177  // whatever happens. So we make a copy.
178  QKeySequence original = seq;
179 
180  m_isUpdating = true;
181 
182  // Check if the user typed in the default sequence into the custom field.
183  // We do this by calling setKeySequence which will do the right thing.
184  setKeySequence(original);
185 
186  emit keySequenceChanged(original);
187  m_isUpdating = false;
188 }

190 void ShortcutEditWidget::setKeySequence(const QKeySequence &activeSeq)
191 {
192  if (activeSeq.toString(QKeySequence::NativeText) == m_defaultLabel->text()) {
193  m_defaultRadio->setChecked(true);
194  m_customEditor->clearKeySequence();
195  } else {
196  m_customRadio->setChecked(true);
197  // m_customEditor->setKeySequence does some stuff we only want to
198  // execute when the sequence really changes.
199  if (activeSeq != m_customEditor->keySequence()) {
200  m_customEditor->setKeySequence(activeSeq);
201  }
202  }
203 }
