102 void KShortcutsEditor::addCollection(KActionCollection *collection, const QString &title)
103 {
104  // KXmlGui add action collections unconditionally. If some plugin doesn't
105  // provide actions we don't want to create empty subgroups.
106  if (collection->isEmpty()) {
107  return;
108  }
109 
110  // We add a bunch of items. Prevent the treewidget from permanently
111  // updating.
112  setUpdatesEnabled(false);
113 
114  d->actionCollections.append(collection);
115  // Forward our actionCollections to the delegate which does the conflict
116  // checking.
117  d->delegate->setCheckActionCollections(d->actionCollections);
118  QString displayTitle = title;
119  if (displayTitle.isEmpty()) {
120  // Use the programName (Translated).
121  displayTitle = collection->componentDisplayName();
122  }
123 
124  QTreeWidgetItem *hier[3];
125  hier[KShortcutsEditorPrivate::Root] = d->ui.list->invisibleRootItem();
126  hier[KShortcutsEditorPrivate::Program] = d->findOrMakeItem(hier[KShortcutsEditorPrivate::Root], displayTitle);
127  hier[KShortcutsEditorPrivate::Action] = NULL;
128 
129  // Set to remember which actions we have seen.
130  QSet<QAction *> actionsSeen;
131 
132  // Add all categories in their own subtree below the collections root node
133  QList<KActionCategory *> categories = collection->findChildren<KActionCategory *>();
134  foreach (KActionCategory *category, categories) {
135  hier[KShortcutsEditorPrivate::Action] = d->findOrMakeItem(hier[KShortcutsEditorPrivate::Program], category->text());
136  foreach (QAction *action, category->actions()) {
137  // Set a marker that we have seen this action
138  actionsSeen.insert(action);
139  d->addAction(action, hier, KShortcutsEditorPrivate::Action);
140  }
141  }
142 
143  // The rest of the shortcuts is added as a direct shild of the action
144  // collections root node
145  foreach (QAction *action, collection->actions()) {
146  if (actionsSeen.contains(action)) {
147  continue;
148  }
149 
150  d->addAction(action, hier, KShortcutsEditorPrivate::Program);
151  }
152 
153  // sort the list
154  d->ui.list->sortItems(Name, Qt::AscendingOrder);
155 
156  // reenable updating
157  setUpdatesEnabled(true);
158 
159  QTimer::singleShot(0, this, SLOT(resizeColumns()));
160 }

360 bool KShortcutsEditorPrivate::addAction(QAction *action, QTreeWidgetItem *hier[], hierarchyLevel level)
361 {
362  // If the action name starts with unnamed- spit out a warning and ignore
363  // it. That name will change at will and will break loading and writing
364  QString actionName = action->objectName();
365  if (actionName.isEmpty() || actionName.startsWith(QStringLiteral("unnamed-"))) {
366  qCritical() << "Skipping action without name " << action->text() << "," << actionName << "!";
367  return false;
368  }
369 
370  // This code doesn't allow editing of QAction. It can not distinguish
371  // between default and active shortcuts. This breaks many assumptions the
372  // editor makes.
373  const QVariant value = action->property("isShortcutConfigurable");
374  if (!value.isValid() || value.toBool()) {
375  new KShortcutsEditorItem((hier[level]), action);
376  return true;
377  }
378 
379  return false;
380 }

382 void KShortcutsEditorPrivate::allDefault()
383 {
384  for (QTreeWidgetItemIterator it(ui.list); (*it); ++it) {
385  if (!(*it)->parent() || (*it)->type() != ActionItem) {
386  continue;
387  }
388 
389  KShortcutsEditorItem *item = static_cast<KShortcutsEditorItem *>(*it);
390  QAction *act = item->m_action;
391 
392  QList<QKeySequence> defaultShortcuts = act->property("defaultShortcuts").value<QList<QKeySequence> >();
393  if (act->shortcuts() != defaultShortcuts) {
394  QKeySequence primary = defaultShortcuts.isEmpty() ? QKeySequence() : defaultShortcuts.at(0);
395  QKeySequence alternate = defaultShortcuts.size() <= 1 ? QKeySequence() : defaultShortcuts.at(1);
396  changeKeyShortcut(item, LocalPrimary, primary);
397  changeKeyShortcut(item, LocalAlternate, alternate);
398  }
399 
400 #if HAVE_GLOBALACCEL
401  if (KGlobalAccel::self()->shortcut(act) != KGlobalAccel::self()->defaultShortcut(act)) {
402  QList<QKeySequence> defaultShortcut = KGlobalAccel::self()->defaultShortcut(act);
403  changeKeyShortcut(item, GlobalPrimary, primarySequence(defaultShortcut));
404  changeKeyShortcut(item, GlobalAlternate, alternateSequence(defaultShortcut));
405  }
406 #endif
407 
408 #if 0
409  KShapeGesture actShapeGesture = KGestureMap::self()->shapeGesture(act);
410  KShapeGesture actDefaultShapeGesture = KGestureMap::self()->defaultShapeGesture(act);
411  if (actShapeGesture != actDefaultShapeGesture) {
412  changeShapeGesture(item, actDefaultShapeGesture);

668 - adjust the general page borders, IMHO they're too wide
669 
670 - add a custom printer options page that allows to filter out all
671  actions that don't have a shortcut set to reduce this list. IMHO this
672  should be optional as people might want to simply print all and when
673  they find a new action that they assign a shortcut they can simply use
674  a pen to fill out the empty space
675 
676 - find a way to align the Main/Alternate/Global entries in the shortcuts
677  column without adding borders. I first did this without a nested table
678  but instead simply added 3 rows and merged the 3 cells in the Action
679  name and description column, but unfortunately I didn't find a way to
680  remove the borders between the 6 shortcut cells.
681 */
682 void KShortcutsEditorPrivate::printShortcuts() const
683 {
684 // One cant print on wince
685 #ifndef _WIN32_WCE
686  QTreeWidgetItem *root = ui.list->invisibleRootItem();
687  QTextDocument doc;
688 
689  doc.setDefaultFont(QFontDatabase::systemFont(QFontDatabase::GeneralFont));
690 
691  QTextCursor cursor(&doc);
692  cursor.beginEditBlock();
693  QTextCharFormat headerFormat;
694  headerFormat.setProperty(QTextFormat::FontSizeAdjustment, 3);
695  headerFormat.setFontWeight(QFont::Bold);
696  cursor.insertText(i18nc("header for an applications shortcut list", "Shortcuts for %1",
697  QGuiApplication::applicationDisplayName()),
698  headerFormat);
699  QTextCharFormat componentFormat;
700  componentFormat.setProperty(QTextFormat::FontSizeAdjustment, 2);
701  componentFormat.setFontWeight(QFont::Bold);
702  QTextBlockFormat componentBlockFormat = cursor.blockFormat();
703  componentBlockFormat.setTopMargin(16);
704  componentBlockFormat.setBottomMargin(16);
705 
706  QTextTableFormat tableformat;
707  tableformat.setHeaderRowCount(1);
708  tableformat.setCellPadding(4.0);
709  tableformat.setCellSpacing(0);
710  tableformat.setBorderStyle(QTextFrameFormat::BorderStyle_Solid);
711  tableformat.setBorder(0.5);
712 
713  QList<QPair<QString, ColumnDesignation> > shortcutTitleToColumn;
714  shortcutTitleToColumn << qMakePair(i18n("Main:"), LocalPrimary);
715  shortcutTitleToColumn << qMakePair(i18n("Alternate:"), LocalAlternate);
716  shortcutTitleToColumn << qMakePair(i18n("Global:"), GlobalPrimary);
717  shortcutTitleToColumn << qMakePair(i18n("Global alternate:"), GlobalAlternate);
718 
719  for (int i = 0; i < root->childCount(); i++) {
720  QTreeWidgetItem *item = root->child(i);
721  cursor.insertBlock(componentBlockFormat, componentFormat);
722  cursor.insertText(item->text(0));
723 
724  QTextTable *table = cursor.insertTable(1, 3);
725  table->setFormat(tableformat);
726  int currow = 0;
727 
728  QTextTableCell cell = table->cellAt(currow, 0);
729  QTextCharFormat format = cell.format();
730  format.setFontWeight(QFont::Bold);
731  cell.setFormat(format);
732  cell.firstCursorPosition().insertText(i18n("Action Name"));
733 
734  cell = table->cellAt(currow, 1);
735  cell.setFormat(format);
736  cell.firstCursorPosition().insertText(i18n("Shortcuts"));
737 
738  cell = table->cellAt(currow, 2);
739  cell.setFormat(format);
740  cell.firstCursorPosition().insertText(i18n("Description"));
741  currow++;
742 
743  for (QTreeWidgetItemIterator it(item); *it; ++it) {
744  if ((*it)->type() != ActionItem) {
745  continue;
746  }
747 
748  KShortcutsEditorItem *editoritem = static_cast<KShortcutsEditorItem *>(*it);
749  table->insertRows(table->rows(), 1);
750  QVariant data = editoritem->data(Name, Qt::DisplayRole);
751  table->cellAt(currow, 0).firstCursorPosition().insertText(data.toString());
752 
753  QTextTable *shortcutTable = 0;
754  for (int k = 0; k < shortcutTitleToColumn.count(); k++) {
755  data = editoritem->data(shortcutTitleToColumn.at(k).second, Qt::DisplayRole);
756  QString key = data.value<QKeySequence>().toString();
757 
758  if (!key.isEmpty()) {
759  if (!shortcutTable) {
760  shortcutTable = table->cellAt(currow, 1).firstCursorPosition().insertTable(1, 2);
761  QTextTableFormat shortcutTableFormat = tableformat;
762  shortcutTableFormat.setCellSpacing(0.0);
763  shortcutTableFormat.setHeaderRowCount(0);
764  shortcutTableFormat.setBorder(0.0);
765  shortcutTable->setFormat(shortcutTableFormat);
766  } else {
767  shortcutTable->insertRows(shortcutTable->rows(), 1);
768  }
769  shortcutTable->cellAt(shortcutTable->rows() - 1, 0).firstCursorPosition().insertText(shortcutTitleToColumn.at(k).first);
770  shortcutTable->cellAt(shortcutTable->rows() - 1, 1).firstCursorPosition().insertText(key);
771  }
772  }
773 
774  QAction *action = editoritem->m_action;
775  cell = table->cellAt(currow, 2);
776  format = cell.format();
777  format.setProperty(QTextFormat::FontSizeAdjustment, -1);
778  cell.setFormat(format);
779  cell.firstCursorPosition().insertHtml(action->whatsThis());
780 
781  currow++;
782  }
783  cursor.movePosition(QTextCursor::End);
784  }
785  cursor.endEditBlock();
786 
787  QPrinter printer;
788  QPrintDialog *dlg = new QPrintDialog(&printer, q);
789  if (dlg->exec() == QDialog::Accepted) {
790  doc.print(&printer);
791  }
792  delete dlg;
793 #endif
794 }
