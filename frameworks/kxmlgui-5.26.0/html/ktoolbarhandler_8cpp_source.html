48 class BarActionBuilder
49 {
50 public:
51  BarActionBuilder(KActionCollection *actionCollection, KXmlGuiWindow *mainWindow,
52  QLinkedList<KToolBar *> &oldToolBarList)
53  : m_actionCollection(actionCollection), m_mainWindow(mainWindow), m_needsRebuild(false)
54  {
55  QList<KToolBar *> toolBars = m_mainWindow->findChildren<KToolBar *>();
56 
57  Q_FOREACH (KToolBar *toolBar, toolBars) {
58  if (toolBar->mainWindow() != m_mainWindow) {
59  continue;
60  }
61 
62  if (!oldToolBarList.contains(toolBar)) {
63  m_needsRebuild = true;
64  }
65 
66  m_toolBars.append(toolBar);
67  }
68 
69  if (!m_needsRebuild) {
70  m_needsRebuild = (oldToolBarList.count() != m_toolBars.count());
71  }
72  }
73 
74  bool needsRebuild() const
75  {
76  return m_needsRebuild;
77  }
78 
79  QList<QAction *> create()
80  {
81  QList<QAction *> actions;
82 
83  if (!m_needsRebuild) {
84  return actions;
85  }
86 
87  Q_FOREACH (KToolBar *bar, m_toolBars) {
88  handleToolBar(bar);
89  }
90 
91  if (m_toolBarActions.count() == 0) {
92  return actions;
93  }
94 
95  if (m_toolBarActions.count() == 1) {
96  KToggleToolBarAction *action = static_cast<KToggleToolBarAction *>(m_toolBarActions.first());
97  action->setText(KStandardShortcut::label(KStandardShortcut::ShowToolbar));
98  return m_toolBarActions;
99  }
100 
101  KActionMenu *menuAction = new KActionMenu(i18n("Toolbars Shown"), m_actionCollection);
102  m_actionCollection->addAction(QStringLiteral("toolbars_submenu_action"), menuAction);
103 
104  Q_FOREACH (QAction *action, m_toolBarActions) {
105  menuAction->menu()->addAction(action);
106  }
107 
108  actions.append(menuAction);
109 
110  return actions;
111  }
112 
113  const QLinkedList<KToolBar *> &toolBars() const
114  {
115  return m_toolBars;
116  }
117 
118 private:
119  void handleToolBar(KToolBar *toolBar)
120  {
121  KToggleToolBarAction *action = new KToggleToolBarAction(
122  toolBar,
123  toolBar->windowTitle(),
124  m_actionCollection);
125  m_actionCollection->addAction(toolBar->objectName(), action);
126 
127  // ## tooltips, whatsthis?
128  m_toolBarActions.append(action);
129  }

133 
134  QLinkedList<KToolBar *> m_toolBars;
135  QList<QAction *> m_toolBarActions;
136 
137  bool m_needsRebuild : 1;

239 void ToolBarHandler::setupActions()
240 {
241  if (!factory() || !d->mainWindow) {
242  return;
243  }
244 
245  BarActionBuilder builder(actionCollection(), d->mainWindow, d->toolBars);
246 
247  if (!builder.needsRebuild()) {
248  return;
249  }
250 
251  unplugActionList(QLatin1String(actionListName));
252 
253  qDeleteAll(d->actions);
254  d->actions.clear();
255 
256  d->actions = builder.create();
257 
258  d->toolBars = builder.toolBars();
259 
260  // We have no XML file associated with our action collection, so load settings from KConfig
261  actionCollection()->readSettings(); // #233712
262 
263  if (KAuthorized::authorizeAction(QStringLiteral("options_show_toolbar"))) {
264  plugActionList(QLatin1String(actionListName), d->actions);
265  }
266 
267  d->connectToActionContainers();
268 }
