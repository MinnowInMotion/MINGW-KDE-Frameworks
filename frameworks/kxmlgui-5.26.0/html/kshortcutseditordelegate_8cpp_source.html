69 void KShortcutsEditorDelegate::stealShortcut(
70  const QKeySequence &seq,
71  QAction *action)
72 {
73  QTreeWidget *view = static_cast<QTreeWidget *>(parent());
74 
75  // Iterate over all items
76  QTreeWidgetItemIterator it(view, QTreeWidgetItemIterator::NoChildren);
77 
78  for (; (*it); ++it) {
79  KShortcutsEditorItem *item = dynamic_cast<KShortcutsEditorItem *>(*it);
80  if (item && item->data(0, ObjectRole).value<QObject *>() == action) {
81 
82  // We found the action, snapshot the current state. Steal the
83  // shortcut. We will save the change later.
84  const QList<QKeySequence> cut = action->shortcuts();
85  const QKeySequence primary = cut.isEmpty() ? QKeySequence() : cut.at(0);
86  const QKeySequence alternate = cut.size() <= 1 ? QKeySequence() : cut.at(1);
87 
88  if (primary.matches(seq) != QKeySequence::NoMatch
89  || seq.matches(primary) != QKeySequence::NoMatch) {
90  item->setKeySequence(LocalPrimary, QKeySequence());
91  }
92 
93  if (alternate.matches(seq) != QKeySequence::NoMatch
94  || seq.matches(alternate) != QKeySequence::NoMatch) {
95  item->setKeySequence(LocalAlternate, QKeySequence());
96  }
97  break;
98  }
99  }
100 
101 }

112 void KShortcutsEditorDelegate::itemActivated(QModelIndex index)
113 {
114  //As per our constructor our parent *is* a QTreeWidget
115  QTreeWidget *view = static_cast<QTreeWidget *>(parent());
116 
117  KShortcutsEditorItem *item = KShortcutsEditorPrivate::itemFromIndex(view, index);
118  if (!item) {
119  //that probably was a non-leaf (type() !=ActionItem) item
120  return;
121  }
122 
123  int column = index.column();
124  if (column == Name) {
125  // If user click in the name column activate the (Global|Local)Primary
126  // column if possible.
127  if (!view->header()->isSectionHidden(LocalPrimary)) {
128  column = LocalPrimary;
129  } else if (!view->header()->isSectionHidden(GlobalPrimary)) {
130  column = GlobalPrimary;
131  } else {
132  // do nothing.
133  }
134  index = index.sibling(index.row(), column);
135  view->selectionModel()->select(index, QItemSelectionModel::SelectCurrent);
136  }
137 
138  // Check if the models wants us to edit the item at index
139  if (!index.data(ShowExtensionIndicatorRole).value<bool>()) {
140  return;
141  }
142 
143  if (!isExtended(index)) {
144  //we only want maximum ONE extender open at any time.
145  if (m_editingIndex.isValid()) {
146  KShortcutsEditorItem *oldItem = KShortcutsEditorPrivate::itemFromIndex(view,
147  m_editingIndex);
148  Q_ASSERT(oldItem); //here we really expect nothing but a real KShortcutsEditorItem
149 
150  oldItem->setNameBold(false);
151  contractItem(m_editingIndex);
152  }
153 
154  m_editingIndex = index;
155  QWidget *viewport = static_cast<QAbstractItemView *>(parent())->viewport();
156 
157  if (column >= LocalPrimary && column <= GlobalAlternate) {
158  ShortcutEditWidget *editor = new ShortcutEditWidget(viewport,
159  index.data(DefaultShortcutRole).value<QKeySequence>(),
160  index.data(ShortcutRole).value<QKeySequence>(),
161  m_allowLetterShortcuts);
162  if (column == GlobalPrimary) {
163  QObject *action = index.data(ObjectRole).value<QObject *>();
164  editor->setAction(action);
165  editor->setMultiKeyShortcutsAllowed(false);
166  QString componentName = action->property("componentName").toString();
167  if (componentName.isEmpty()) {
168  componentName = QCoreApplication::applicationName();
169  }
170  editor->setComponentName(componentName);
171  }
172 
173  m_editor = editor;
174  // For global shortcuts check against the kde standard shortcuts
175  if (column == GlobalPrimary || column == GlobalAlternate) {
176  editor->setCheckForConflictsAgainst(
177  KKeySequenceWidget::LocalShortcuts
178  | KKeySequenceWidget::GlobalShortcuts
179  | KKeySequenceWidget::StandardShortcuts);
180  }
181 
182  editor->setCheckActionCollections(m_checkActionCollections);
183 
184  connect(m_editor, SIGNAL(keySequenceChanged(QKeySequence)),
185  this, SLOT(keySequenceChanged(QKeySequence)));
186  connect(m_editor, SIGNAL(stealShortcut(QKeySequence,QAction*)),
187  this, SLOT(stealShortcut(QKeySequence,QAction*)));
188 
189  } else if (column == RockerGesture) {
190  m_editor = new QLabel(QStringLiteral("A lame placeholder"), viewport);
191 
192  } else if (column == ShapeGesture) {
193  m_editor = new QLabel(QStringLiteral("<i>A towel</i>"), viewport);
194 
195  } else {
196  return;
197  }
198 
199  m_editor->installEventFilter(this);
200  item->setNameBold(true);
201  extendItem(m_editor, index);
202 
203  } else {
204  //the item is extended, and clicking on it again closes it
205  item->setNameBold(false);
206  contractItem(index);
207  view->selectionModel()->select(index, QItemSelectionModel::Clear);
208  m_editingIndex = QModelIndex();
209  m_editor = 0;
210  }
211 }

214 void KShortcutsEditorDelegate::itemCollapsed(QModelIndex index)
215 {
216  if (!m_editingIndex.isValid()) {
217  return;
218  }
219 
220  const QAbstractItemModel *model = index.model();
221  for (int row = 0; row < model->rowCount(index); ++row) {
222  for (int col = 0; col < index.model()->columnCount(index); ++col) {
223  QModelIndex colIndex = model->index(row, col, index);
224 
225  if (colIndex == m_editingIndex) {
226  itemActivated(m_editingIndex); //this will *close* the item's editor because it's already open
227  }
228  }

245 bool KShortcutsEditorDelegate::eventFilter(QObject *o, QEvent *e)
246 {
247  if (o == m_editor) {
248  //Prevent clicks in the empty part of the editor widget from closing the editor
249  //because they would propagate to the itemview and be interpreted as a click in
250  //an item's rect. That in turn would lead to an itemActivated() call, closing
251  //the current editor.
252 
253  switch (e->type()) {
254  case QEvent::MouseButtonPress:
255  case QEvent::MouseButtonRelease:
256  case QEvent::MouseButtonDblClick:
257  return true;
258  default:
259  return false;
260  }
261  } else if (o == parent()) {
262  // Make left/right cursor keys switch items instead of operate the scroll bar
263  // (subclassing QtreeView/Widget would be cleaner but much more of a hassle)
264  // Note that in our case we know that the selection mode is SingleSelection,
265  // so we don't have to ask QAbstractItemView::selectionCommand() et al.
266 
267  if (e->type() != QEvent::KeyPress) {
268  return false;
269  }
270  QKeyEvent *ke = static_cast<QKeyEvent *>(e);
271  QTreeWidget *view = static_cast<QTreeWidget *>(parent());
272  QItemSelectionModel *selection = view->selectionModel();
273  QModelIndex index = selection->currentIndex();
274 
275  switch (ke->key()) {
276  case Qt::Key_Space:
277  case Qt::Key_Select:
278  // we are not using the standard "open editor" mechanism of QAbstractItemView,
279  // so let's emulate that here.
280  itemActivated(index);
281  return true;
282  case Qt::Key_Left:
283  index = index.sibling(index.row(), index.column() - 1);
284  break;
285  case Qt::Key_Right:
286  index = index.sibling(index.row(), index.column() + 1);
287  break;
288  default:
289  return false;
290  }
291  // a cursor key was pressed
292  if (index.isValid()) {
293  selection->setCurrentIndex(index, QItemSelectionModel::ClearAndSelect);
294  //### using PositionAtCenter for now;
295  // EnsureVisible has no effect which seems to be a bug.
296  view->scrollTo(index, QAbstractItemView::PositionAtCenter);
297  }
298  return true;
299  }
300  return false;
301 }
