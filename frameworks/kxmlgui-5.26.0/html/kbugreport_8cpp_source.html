407 void KBugReport::accept()
408 {
409  if (d->submitBugWeb) {
410  QDesktopServices::openUrl(d->url);
411  return;
412  }
413 
414  if (d->m_lineedit->toPlainText().isEmpty() ||
415  d->m_subject->text().isEmpty()) {
416  QString msg = i18n("You must specify both a subject and a description "
417  "before the report can be sent.");
418  KMessageBox::error(this, msg);
419  return;
420  }
421 
422  switch (d->currentSeverity()) {
423  case 0: // critical
424  if (KMessageBox::questionYesNo(this, i18n(
425  "<p>You chose the severity <b>Critical</b>. "
426  "Please note that this severity is intended only for bugs that:</p>"
427  "<ul><li>break unrelated software on the system (or the whole system)</li>"
428  "<li>cause serious data loss</li>"
429  "<li>introduce a security hole on the system where the affected package is installed</li></ul>\n"
430  "<p>Does the bug you are reporting cause any of the above damage? "
431  "If it does not, please select a lower severity. Thank you.</p>"), QString(), KStandardGuiItem::cont(), KStandardGuiItem::cancel()) == KMessageBox::No) {
432  return;
433  }
434  break;
435  case 1: // grave
436  if (KMessageBox::questionYesNo(this, i18n(
437  "<p>You chose the severity <b>Grave</b>. "
438  "Please note that this severity is intended only for bugs that:</p>"
439  "<ul><li>make the package in question unusable or mostly so</li>"
440  "<li>cause data loss</li>"
441  "<li>introduce a security hole allowing access to the accounts of users who use the affected package</li></ul>\n"
442  "<p>Does the bug you are reporting cause any of the above damage? "
443  "If it does not, please select a lower severity. Thank you.</p>"), QString(), KStandardGuiItem::cont(), KStandardGuiItem::cancel()) == KMessageBox::No) {
444  return;
445  }
446  break;
447  default:
448  break;
449  }
450  if (!sendBugReport()) {
451  QString msg = i18n("Unable to send the bug report.\n"
452  "Please submit a bug report manually....\n"
453  "See http://bugs.kde.org/ for instructions.");
454  KMessageBox::error(this, msg + QStringLiteral("\n\n") + d->lastError);
455  return;
456  }
457 
458  KMessageBox::information(this,
459  i18n("Bug report sent, thank you for your input."));
460  QDialog::accept();
461 }

512  .arg(appname).arg(d->m_strVersion).arg(severity) +

514  os + QLatin1String("\n") + bodyText;

518 bool KBugReport::sendBugReport()
519 {
520  QString recipient = d->m_aboutData.bugAddress();
521  if (recipient.isEmpty()) {
522  recipient = QStringLiteral("submit@bugs.kde.org");
523  }
524 
525  QString command = QStandardPaths::findExecutable(QStringLiteral("ksendbugmail"));
526  if (command.isEmpty()) {
527  command = QFile::decodeName(CMAKE_INSTALL_PREFIX "/" KF5_LIBEXEC_INSTALL_DIR "/ksendbugmail");
528  }
529 
530  QProcess proc;
531  QStringList args;
532  args << QStringLiteral("--subject") << d->m_subject->text() << QStringLiteral("--recipient") << recipient;
533  proc.start(command, args);
534  //qDebug() << command << args;
535  if (!proc.waitForStarted()) {
536  qCritical() << "Unable to open a pipe to " << command << endl;
537  return false;
538  }
539  proc.write(text().toUtf8());
540  proc.closeWriteChannel();
541 
542  proc.waitForFinished();
543  //qDebug() << "kbugreport: sendbugmail exit, status " << proc.exitStatus() << " code " << proc.exitCode();
544 
545  QByteArray line;
546  if (proc.exitStatus() == QProcess::NormalExit && proc.exitCode() != 0) {
547  // XXX not stderr?
548  while (!proc.atEnd()) {
549  line = proc.readLine();
550  }
551  d->lastError = QString::fromUtf8(line);
552  return false;
553  }
554  return true;
555 }
