36 bool KShortcutSchemesHelper::saveShortcutScheme(const QList<KActionCollection *> &collections,
37  const QString &schemeName)
38 {
39  // Every action collection is associated with a KXMLGUIClient
40  // (at least if it was added by KXMLGUIFactory::configureShortcuts)
41 
42  // Some GUI clients have the same name (e.g. the child client for a mainwindow
43  // holding the actions for hiding/showing toolbars), so we need to save them
44  // together, otherwise they will overwrite each other's file on disk.
45 
46  // For cases like kdevelop (many guiclients not reused in other apps) it's simpler
47  // to even save all shortcuts to a single shortcuts file -> set the boolean below to true
48  // to create an all-in-one scheme.
49  // Maybe we need a checkbox for this? Or an env var for contributors to set, rather? End users don't care.
50  const bool saveToApplicationFile = false;
51 
52  QMap<QString, KActionCollection *> collectionsByClientName;
53  foreach (KActionCollection *coll, collections) {
54  const KXMLGUIClient *client = coll->parentGUIClient();
55  if (client) {
56  const QString key = saveToApplicationFile ? QCoreApplication::applicationName() : client->componentName();
57  collectionsByClientName.insertMulti(key, coll);
58  }
59  }
60  foreach (const QString &componentName, collectionsByClientName.uniqueKeys()) {
61 
62  qCDebug(DEBUG_KXMLGUI) << "Considering component" << componentName;
63  QDomDocument doc;
64  QDomElement docElem = doc.createElement(QStringLiteral("kpartgui"));
65  docElem.setAttribute(QStringLiteral("version"), QStringLiteral("1"));
66  docElem.setAttribute(QStringLiteral("name"), componentName);
67  doc.appendChild(docElem);
68  QDomElement elem = doc.createElement(QStringLiteral("ActionProperties"));
69  docElem.appendChild(elem);
70 
71  foreach (KActionCollection *collection, collectionsByClientName.values(componentName)) {
72  qCDebug(DEBUG_KXMLGUI) << "Saving shortcut scheme for action collection with" << collection->actions().count() << "actions";
73 
74  foreach (QAction *action, collection->actions()) {
75  if (!action) {
76  continue;
77  }
78 
79  const QString actionName = action->objectName();
80  const QString shortcut = QKeySequence::listToString(action->shortcuts());
81  //qCDebug(DEBUG_KXMLGUI) << "action" << actionName << "has shortcut" << shortcut;
82  if (!shortcut.isEmpty()) {
83  QDomElement act_elem = doc.createElement(QStringLiteral("Action"));
84  act_elem.setAttribute(QStringLiteral("name"), actionName);
85  act_elem.setAttribute(QStringLiteral("shortcut"), shortcut);
86  elem.appendChild(act_elem);
87  }
88  }
89  }
90 
91 
92  const QString schemeFileName = writableShortcutSchemeFileName(componentName, schemeName);
93  if (elem.childNodes().isEmpty()) {
94  QFile::remove(schemeFileName);
95  } else {
96  qCDebug(DEBUG_KXMLGUI) << "saving to" << schemeFileName;
97  QDir().mkpath(QFileInfo(schemeFileName).absolutePath());
98  QFile schemeFile(schemeFileName);
99  if (!schemeFile.open(QFile::WriteOnly | QFile::Truncate)) {
100  qCDebug(DEBUG_KXMLGUI) << "COULD NOT WRITE" << schemeFileName;
101  return false;
102  }
103 
104  QTextStream out(&schemeFile);
105  out << doc.toString(2);
106  }
107  }
108  return true;
109 }
