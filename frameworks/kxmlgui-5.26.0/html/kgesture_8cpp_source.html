60 KShapeGesture::KShapeGesture(const QString &description)
61  : d(new KShapeGesturePrivate)
62 {
63  QStringList sl = description.split(QLatin1Char(','));
64  d->m_friendlyName = sl.takeFirst();
65 
66  bool ok = true;
67  QPolygon poly;
68  int x, y;
69  QStringList::const_iterator it = sl.constBegin();
70  while (it != sl.constEnd()) {
71  x = (*it).toInt(&ok);
72  ++it;
73  if (!ok || it == sl.constEnd()) {
74  break;
75  }
76  y = (*it).toInt(&ok);
77  if (!ok) {
78  break;
79  }
80  ++it;
81  poly.append(QPoint(x, y));
82  }
83  if (!ok) {
84  d->m_friendlyName.clear();
85  return;
86  }
87 
88  setShape(poly);
89 }

101 void KShapeGesture::setShape(const QPolygon &shape)
102 {
103  //Scale and translate into a 100x100 square with its
104  //upper left corner at origin.
105  d->m_shape = shape;
106  QRect bounding = shape.boundingRect();
107  //TODO: don't change aspect ratio "too much" to avoid problems with straight lines
108  //TODO: catch all bad input, like null height/width
109 
110  //compensate for QRect weirdness
111  bounding.setWidth(bounding.width() - 1);
112  bounding.setHeight(bounding.height() - 1);
113 
114  float xScale = bounding.width() ? 100.0 / bounding.width() : 1.0;
115  float yScale = bounding.height() ? 100.0 / bounding.height() : 1.0;
116  d->m_shape.translate(-bounding.left(), -bounding.top());
117  for (int i = 0; i < d->m_shape.size(); i++) {
118  d->m_shape[i].setX((int)(xScale * (float)d->m_shape[i].x()));
119  d->m_shape[i].setY((int)(yScale * (float)d->m_shape[i].y()));
120  }
121 
122  //calculate accumulated lengths of lines making up the polygon
123  Q_ASSERT(d->m_shape.size() > 1);
124  d->m_curveLength = 0.0;
125  d->m_lengthTo.clear();
126  d->m_lengthTo.reserve(d->m_shape.size());
127  d->m_lengthTo.append(d->m_curveLength);
128 
129  int prevX = d->m_shape[0].x();
130  int prevY = d->m_shape[0].y();
131  for (int i = 1; i < d->m_shape.size(); i++) {
132  int curX = d->m_shape[i].x();
133  int curY = d->m_shape[i].y();
134  d->m_curveLength += metric(curX - prevX, curY - prevY);
135  d->m_lengthTo.append(d->m_curveLength);
136  prevX = curX;
137  prevY = curY;
138  }
139 }

208 /*
209  algorithm: iterate in order over 30 points on our shape and measure the
210  minimum distance to any point on the other shape. never go backwards on
211  the other shape to also check direction of movement.
212  This algorithm is best applied like a->distance(b) + b->distance(a).
213  fabs(a->distance(b) - b->distance(a)) might turn out to be very interesting,
214  too. in fact, i think it's the most interesting value.
215  */
216 float KShapeGesture::distance(const KShapeGesture &other, float abortThreshold) const
217 {
218  Q_UNUSED(abortThreshold); //for optimizations, later
219  const QPolygon &o_shape = other.d->m_shape;
220  const QVector<float> &o_lengthTo = other.d->m_lengthTo;
221  float x = 0;
222  float y = 0;
223  float mx = 0;
224  float my = 0;
225  float position = 0;
226  float ox = 0;
227  float oy = 0;
228  float oposition = 0;
229  float omx = 0;
230  float omy = 0;
231  float oxB = 0;
232  float oyB = 0;
233  float opositionB = 0;
234  float omxB = 0;
235  float omyB = 0;
236  float dist = 0;
237  float distB = 0;
238  float desiredPosition = 0;
239  float strokeLength = 0;
240  float retval = 0.0;
241  int pointIndex = 0, opointIndex = 0, opointIndexB = 0;
242 
243  //set up starting point on our shape
244  x = d->m_shape[0].x();
245  y = d->m_shape[0].y();
246  strokeLength = d->m_lengthTo[1];
247  mx = (d->m_shape[1].x() - x) / strokeLength;
248  my = (d->m_shape[1].y() - y) / strokeLength;
249  position = 0.0;
250 
251  //set up lower bound of search interval on other shape
252  ox = o_shape[0].x();
253  oy = o_shape[0].y();
254  strokeLength = o_lengthTo[1];
255  omx = (o_shape[1].x() - ox) / strokeLength;
256  omy = (o_shape[1].y() - oy) / strokeLength;
257  oposition = 0.0;
258  dist = metric(ox - x, oy - y);
259 
260  for (int i = 0; i <= 30; i++) {
261  //go to comparison point on our own polygon
262  //30.0001 to prevent getting out-of-bounds pointIndex
263  desiredPosition = d->m_curveLength / 30.0001 * (float)i;
264  if (desiredPosition > d->m_lengthTo[pointIndex + 1]) {
265 
266  while (desiredPosition > d->m_lengthTo[pointIndex + 1]) {
267  pointIndex++;
268  }
269 
270  x = d->m_shape[pointIndex].x();
271  y = d->m_shape[pointIndex].y();
272  position = d->m_lengthTo[pointIndex];
273  strokeLength = d->m_lengthTo[pointIndex + 1] - position;
274  mx = (d->m_shape[pointIndex + 1].x() - x) / strokeLength;
275  my = (d->m_shape[pointIndex + 1].y() - y) / strokeLength;
276  }
277  x += mx * (desiredPosition - position);
278  y += my * (desiredPosition - position);
279  position = desiredPosition;
280 
281  //set up upper bound of search interval on other shape
282  desiredPosition = qMin(oposition + other.d->m_curveLength / 15.00005,
283  other.d->m_curveLength - 0.0001);
284  if (i == 0 || desiredPosition > o_lengthTo[opointIndexB + 1]) {
285 
286  while (desiredPosition > o_lengthTo[opointIndexB + 1]) {
287  opointIndexB++;
288  }
289 
290  oxB = o_shape[opointIndexB].x();
291  oyB = o_shape[opointIndexB].y();
292  opositionB = o_lengthTo[opointIndexB];
293  strokeLength = o_lengthTo[opointIndexB + 1] - opositionB;
294  omxB = (o_shape[opointIndexB + 1].x() - oxB) / strokeLength;
295  omyB = (o_shape[opointIndexB + 1].y() - oyB) / strokeLength;
296  }
297  oxB += omxB * (desiredPosition - opositionB);
298  oyB += omyB * (desiredPosition - opositionB);
299  opositionB = desiredPosition;
300  distB = metric(oxB - x, oyB - y);
301 
302  //binary search for nearest point on other shape
303  for (int j = 0; j < 6; j++) {
304  desiredPosition = (oposition + opositionB) * 0.5;
305  if (dist < distB) {
306  //retract upper bound to desiredPosition
307  //copy state of lower bound to upper bound, advance it from there
308  oxB = ox; oyB = oy;
309  omxB = omx; omyB = omy;
310  opointIndexB = opointIndex; opositionB = oposition;
311 
312  if (desiredPosition > o_lengthTo[opointIndexB + 1]) {
313 
314  while (desiredPosition > o_lengthTo[opointIndexB + 1]) {
315  opointIndexB++;
316  }
317 
318  oxB = o_shape[opointIndexB].x();
319  oyB = o_shape[opointIndexB].y();
320  opositionB = o_lengthTo[opointIndexB];
321  strokeLength = o_lengthTo[opointIndexB + 1] - opositionB;
322  omxB = (o_shape[opointIndexB + 1].x() - oxB) / strokeLength;
323  omyB = (o_shape[opointIndexB + 1].y() - oyB) / strokeLength;
324  }
325  oxB += omxB * (desiredPosition - opositionB);
326  oyB += omyB * (desiredPosition - opositionB);
327  opositionB = desiredPosition;
328  distB = metric(oxB - x, oyB - y);
329  } else {
330  //advance lower bound to desiredPosition
331  if (desiredPosition > o_lengthTo[opointIndex + 1]) {
332 
333  while (desiredPosition > o_lengthTo[opointIndex + 1]) {
334  opointIndex++;
335  }
336 
337  ox = o_shape[opointIndex].x();
338  oy = o_shape[opointIndex].y();
339  oposition = o_lengthTo[opointIndex];
340  strokeLength = o_lengthTo[opointIndex + 1] - oposition;
341  omx = (o_shape[opointIndex + 1].x() - ox) / strokeLength;
342  omy = (o_shape[opointIndex + 1].y() - oy) / strokeLength;
343  }
344  ox += omx * (desiredPosition - oposition);
345  oy += omy * (desiredPosition - oposition);
346  oposition = desiredPosition;
347  dist = metric(ox - x, oy - y);
348  }
349  }
350  retval += qMin(dist, distB);
351  }
352  //scale value to make it roughly invariant against step width
353  return retval / 30.0;
354 }

421 KRockerGesture::KRockerGesture(const QString &description)
422  : d(new KRockerGesturePrivate)
423 {
424  if (description.length() != 2) {
425  return;
426  }
427 
428  Qt::MouseButton hold, thenPush;
429  Qt::MouseButton *current = &hold;
430  for (int i = 0; i < 2; i++) {
431  switch (description[i].toLatin1()) {
432  case 'L':
433  *current = Qt::LeftButton;
434  break;
435  case 'R':
436  *current = Qt::RightButton;
437  break;
438  case 'M':
439  *current = Qt::MidButton;
440  break;
441  case '1':
442  *current = Qt::XButton1;
443  break;
444  case '2':
445  *current = Qt::XButton2;
446  break;
447  default:
448  return;
449  }
450  current = &thenPush;
451  }
452  d->m_hold = hold;
453  d->m_thenPush = thenPush;
454 }
