59 /*
60  Toolbar settings (e.g. icon size or toolButtonStyle)
61  =====================================================
62 
63  We have the following stack of settings (in order of priority) :
64  - user-specified settings (loaded/saved in KConfig)
65  - developer-specified settings in the XMLGUI file (if using xmlgui) (cannot change at runtime)
66  - KDE-global default (user-configurable; can change at runtime)
67  and when switching between kparts, they are saved as xml in memory,
68  which, in the unlikely case of no-kmainwindow-autosaving, could be
69  different from the user-specified settings saved in KConfig and would have
70  priority over it.
71 
72  So, in summary, without XML:
73  Global config / User settings (loaded/saved in kconfig)
74  and with XML:
75  Global config / App-XML attributes / User settings (loaded/saved in kconfig)
76 
77  And all those settings (except the KDE-global defaults) have to be stored in memory
78  since we cannot retrieve them at random points in time, not knowing the xml document
79  nor config file that holds these settings. Hence the iconSizeSettings and toolButtonStyleSettings arrays.
80 
81  For instance, if you change the KDE-global default, whether this makes a change
82  on a given toolbar depends on whether there are settings at Level_AppXML or Level_UserSettings.
83  Only if there are no settings at those levels, should the change of KDEDefault make a difference.
84 */
85 enum SettingLevel { Level_KDEDefault, Level_AppXML, Level_UserSettings,
86  NSettingLevels
87  };
88 enum { Unset = -1 };

185  for (int level = 0; level < NSettingLevels; ++level) {
186  values[level] = Unset;
187  }
188  }
189  int currentValue() const
190  {
191  int val = Unset;
192  for (int level = 0; level < NSettingLevels; ++level) {
193  if (values[level] != Unset) {
194  val = values[level];
195  }
196  }
197  return val;
198  }
199  // Default value as far as the user is concerned is kde-global + app-xml.
200  // If currentValue()==defaultValue() then nothing to write into kconfig.
201  int defaultValue() const
202  {
203  int val = Unset;
204  for (int level = 0; level < Level_UserSettings; ++level) {

207  }
208  }
209  return val;
210  }

213  QString str;

218  }
219  int &operator[](int index)
220  {
221  return values[index];
222  }
223  private:

440 void KToolBar::Private::adjustSeparatorVisibility()
441 {
442  bool visibleNonSeparator = false;
443  int separatorToShow = -1;
444 
445  for (int index = 0; index < q->actions().count(); ++index) {
446  QAction *action = q->actions()[ index ];
447  if (action->isSeparator()) {
448  if (visibleNonSeparator) {
449  separatorToShow = index;
450  visibleNonSeparator = false;
451  } else {
452  action->setVisible(false);
453  }
454  } else if (!visibleNonSeparator) {
455  if (action->isVisible()) {
456  visibleNonSeparator = true;
457  if (separatorToShow != -1) {
458  q->actions()[ separatorToShow ]->setVisible(true);
459  separatorToShow = -1;
460  }
461  }
462  }
463  }
464 
465  if (separatorToShow != -1) {
466  q->actions()[ separatorToShow ]->setVisible(false);
467  }
468 }

585 void KToolBar::Private::slotContextAboutToShow()
586 {
595  KXmlGuiWindow *kmw = qobject_cast<KXmlGuiWindow *>(q->mainWindow());
596 
597  // try to find "configure toolbars" action
598  QAction *configureAction = 0;
599  const char *actionName = KStandardAction::name(KStandardAction::ConfigureToolbars);
600  configureAction = findAction(QLatin1String(actionName));
601 
602  if (!configureAction && kmw) {
603  configureAction = kmw->actionCollection()->action(QLatin1String(actionName));
604  }
605 
606  if (configureAction) {
607  context->addAction(configureAction);
608  }
609 
610  context->addAction(contextLockAction);
611 
612  if (kmw) {
613  kmw->setupToolbarMenuActions();
614  // Only allow hiding a toolbar if the action is also plugged somewhere else (e.g. menubar)
615  QAction *tbAction = kmw->toolBarMenuAction();
616  if (!q->toolBarsLocked() && tbAction && tbAction->associatedWidgets().count() > 0) {
617  context->addAction(tbAction);
618  }
619  }
620 
621  KEditToolBar::setGlobalDefaultToolBar(q->QObject::objectName().toLatin1().constData());
622 
623  // Check the actions that should be checked
624  switch (q->toolButtonStyle()) {
625  case Qt::ToolButtonIconOnly:
626  default:
627  contextIcons->setChecked(true);
628  break;
629  case Qt::ToolButtonTextBesideIcon:
630  contextTextRight->setChecked(true);
631  break;
632  case Qt::ToolButtonTextOnly:
633  contextText->setChecked(true);
634  break;
635  case Qt::ToolButtonTextUnderIcon:
636  contextTextUnder->setChecked(true);
637  break;
638  }
639 
640  QMapIterator< QAction *, int > it = contextIconSizes;
641  while (it.hasNext()) {
642  it.next();
643  if (it.value() == q->iconSize().width()) {
644  it.key()->setChecked(true);
645  break;
646  }
647  }
648 
649  switch (q->mainWindow()->toolBarArea(q)) {
650  case Qt::BottomToolBarArea:
651  contextBottom->setChecked(true);
652  break;
653  case Qt::LeftToolBarArea:
654  contextLeft->setChecked(true);
655  break;
656  case Qt::RightToolBarArea:
657  contextRight->setChecked(true);
658  break;
659  default:
660  case Qt::TopToolBarArea:
661  contextTop->setChecked(true);
662  break;
663  }
664 
665  const bool showButtonSettings = contextButtonAction
666  && !contextShowText->text().isEmpty()
667  && contextTextRight->isChecked();
668  contextButtonTitle->setVisible(showButtonSettings);

918 void KToolBar::loadState(const QDomElement &element)
919 {
920  QMainWindow *mw = mainWindow();
921  if (!mw) {
922  return;
923  }
924 
925  {
926  QDomNode textNode = element.namedItem(QStringLiteral("text"));
927  QByteArray domain;
928  QByteArray text;
929  QByteArray context;
930  if (textNode.isElement()) {
931  QDomElement textElement = textNode.toElement();
932  domain = textElement.attribute(QStringLiteral("translationDomain")).toUtf8();
933  text = textElement.text().toUtf8();
934  context = textElement.attribute(QStringLiteral("context")).toUtf8();
935  } else {
936  textNode = element.namedItem(QStringLiteral("Text"));
937  if (textNode.isElement()) {
938  QDomElement textElement = textNode.toElement();
939  domain = textElement.attribute(QStringLiteral("translationDomain")).toUtf8();
940  text = textElement.text().toUtf8();
941  context = textElement.attribute(QStringLiteral("context")).toUtf8();
942  }
943  }
944 
945  if (domain.isEmpty()) {
946  domain = element.ownerDocument().documentElement().attribute(QStringLiteral("translationDomain")).toUtf8();
947  if (domain.isEmpty()) {
948  domain = KLocalizedString::applicationDomain();
949  }
950  }
951  QString i18nText;
952  if (!text.isEmpty() && !context.isEmpty()) {
953  i18nText = i18ndc(domain.constData(), context.constData(), text.constData());
954  } else if (!text.isEmpty()) {
955  i18nText = i18nd(domain.constData(), text.constData());
956  }

959  setWindowTitle(i18nText);
960  }
961  }
962 
963  /*
964  This method is called in order to load toolbar settings from XML.
965  However this can be used in two rather different cases:
966  - for the initial loading of the app's XML. In that case the settings
967  are only the defaults (Level_AppXML), the user's KConfig settings will override them
968 
969  - for later re-loading when switching between parts in KXMLGUIFactory.
970  In that case the XML contains the final settings, not the defaults.
971  We do need the defaults, and the toolbar might have been completely
972  deleted and recreated meanwhile. So we store the app-default settings
973  into the XML.
974  */
975  bool loadingAppDefaults = true;
976  if (element.hasAttribute(QStringLiteral("tempXml"))) {
977  // this isn't the first time, so the app-xml defaults have been saved into the (in-memory) XML
978  loadingAppDefaults = false;
979  const QString iconSizeDefault = element.attribute(QStringLiteral("iconSizeDefault"));
980  if (!iconSizeDefault.isEmpty()) {
981  d->iconSizeSettings[Level_AppXML] = iconSizeDefault.toInt();
982  }
983  const QString toolButtonStyleDefault = element.attribute(QStringLiteral("toolButtonStyleDefault"));
984  if (!toolButtonStyleDefault.isEmpty()) {
985  d->toolButtonStyleSettings[Level_AppXML] = d->toolButtonStyleFromString(toolButtonStyleDefault);
986  }
987  } else {
988  // loading app defaults
989  bool newLine = false;
990  QString attrNewLine = element.attribute(QStringLiteral("newline")).toLower();
991  if (!attrNewLine.isEmpty()) {
992  newLine = attrNewLine == QStringLiteral("true");
993  }
994  if (newLine && mw) {
995  mw->insertToolBarBreak(this);
996  }
997  }
998 
999  int newIconSize = -1;
1000  if (element.hasAttribute(QStringLiteral("iconSize"))) {
1001  bool ok;
1002  newIconSize = element.attribute(QStringLiteral("iconSize")).trimmed().toInt(&ok);
1003  if (!ok) {
1004  newIconSize = -1;
1005  }
1006  }
1007  if (newIconSize != -1) {
1008  d->iconSizeSettings[loadingAppDefaults ? Level_AppXML : Level_UserSettings] = newIconSize;
1009  }
1010 
1011  const QString newToolButtonStyle = element.attribute(QStringLiteral("iconText"));
1012  if (!newToolButtonStyle.isEmpty()) {
1013  d->toolButtonStyleSettings[loadingAppDefaults ? Level_AppXML : Level_UserSettings] = d->toolButtonStyleFromString(newToolButtonStyle);
1014  }
1015 
1016  bool hidden = false;
1017  {
1018  QString attrHidden = element.attribute(QStringLiteral("hidden")).toLower();
1019  if (!attrHidden.isEmpty()) {
1020  hidden = attrHidden == QStringLiteral("true");
1021  }
1022  }
1023 
1024  Qt::ToolBarArea pos = Qt::NoToolBarArea;
1025  {
1026  QString attrPosition = element.attribute(QStringLiteral("position")).toLower();
1027  if (!attrPosition.isEmpty()) {
1028  pos = KToolBar::Private::positionFromString(attrPosition);
1029  }
1030  }
1031  if (pos != Qt::NoToolBarArea) {
1032  mw->addToolBar(pos, this);
1033  }
1034 
1035  setVisible(!hidden);
1036 
1037  d->applyCurrentSettings();
1038 }

1301 bool KToolBar::eventFilter(QObject *watched, QEvent *event)
1302 {
1303  // Generate context menu events for disabled buttons too...
1304  if (event->type() == QEvent::MouseButtonPress) {
1305  QMouseEvent *me = static_cast<QMouseEvent *>(event);
1306  if (me->buttons() & Qt::RightButton)
1307  if (QWidget *ww = qobject_cast<QWidget *>(watched))
1308  if (ww->parent() == this && !ww->isEnabled()) {
1309  QCoreApplication::postEvent(this, new QContextMenuEvent(QContextMenuEvent::Mouse, me->pos(), me->globalPos()));
1310  }
1311 
1312  } else if (event->type() == QEvent::ParentChange) {
1313  // Make sure we're not leaving stale event filters around,
1314  // when a child is reparented somewhere else
1315  if (QWidget *ww = qobject_cast<QWidget *>(watched)) {
1316  if (!this->isAncestorOf(ww)) {
1317  // New parent is not a subwidget - remove event filter
1318  ww->removeEventFilter(this);
1319  Q_FOREACH (QWidget *child, ww->findChildren<QWidget *>()) {
1320  child->removeEventFilter(this);
1321  }
1322  }
1323  }
1324  }
1325 
1326  QToolButton *tb;
1327  if ((tb = qobject_cast<QToolButton *>(watched))) {
1328  const QList<QAction *> tbActions = tb->actions();
1329  if (!tbActions.isEmpty()) {
1330  // Handle MMB on toolbar buttons
1331  if (event->type() == QEvent::MouseButtonPress || event->type() == QEvent::MouseButtonRelease) {
1332  QMouseEvent *me = static_cast<QMouseEvent *>(event);
1333  if (me->button() == Qt::MidButton /*&&
1334  act->receivers(SIGNAL(triggered(Qt::MouseButtons,Qt::KeyboardModifiers)))*/) {
1335  QAction *act = tbActions.first();
1336  if (me->type() == QEvent::MouseButtonPress) {
1337  tb->setDown(act->isEnabled());
1338  } else {
1339  tb->setDown(false);
1340  if (act->isEnabled()) {
1341  QMetaObject::invokeMethod(act, "triggered", Qt::DirectConnection,
1342  Q_ARG(Qt::MouseButtons, me->button()),
1343  Q_ARG(Qt::KeyboardModifiers, QApplication::keyboardModifiers()));
1344  }
1345  }
1346  }
1347  }
1348 
1349  // CJK languages use more verbose accelerator marker: they add a Latin
1350  // letter in parenthesis, and put accelerator on that. Hence, the default
1351  // removal of ampersand only may not be enough there, instead the whole
1352  // parenthesis construct should be removed. Use KLocalizedString's method to do this.
1353  if (event->type() == QEvent::Show || event->type() == QEvent::Paint || event->type() == QEvent::EnabledChange) {
1354  QAction *act = tb->defaultAction();
1355  if (act) {
1356  const QString text = KLocalizedString::removeAcceleratorMarker(act->iconText().isEmpty() ? act->text() : act->iconText());
1357  const QString toolTip = KLocalizedString::removeAcceleratorMarker(act->toolTip());
1358  // Filtering messages requested by translators (scripting).
1359  tb->setText(i18nc("@action:intoolbar Text label of toolbar button", "%1", text));
1360  tb->setToolTip(i18nc("@info:tooltip Tooltip of toolbar button", "%1", toolTip));
1361  }
1362  }
1363  }
1364  }
1365 
1366  // Redirect mouse events to the toolbar when drag + drop editing is enabled
1367  if (toolBarsEditable()) {
1368  if (QWidget *ww = qobject_cast<QWidget *>(watched)) {
1369  switch (event->type()) {
1370  case QEvent::MouseButtonPress: {
1371  QMouseEvent *me = static_cast<QMouseEvent *>(event);
1372  QMouseEvent newEvent(me->type(), mapFromGlobal(ww->mapToGlobal(me->pos())), me->globalPos(),
1373  me->button(), me->buttons(), me->modifiers());
1374  mousePressEvent(&newEvent);
1375  return true;
1376  }
1377  case QEvent::MouseMove: {
1378  QMouseEvent *me = static_cast<QMouseEvent *>(event);
1379  QMouseEvent newEvent(me->type(), mapFromGlobal(ww->mapToGlobal(me->pos())), me->globalPos(),
1380  me->button(), me->buttons(), me->modifiers());
1381  mouseMoveEvent(&newEvent);
1382  return true;
1383  }
1384  case QEvent::MouseButtonRelease: {
1385  QMouseEvent *me = static_cast<QMouseEvent *>(event);
1386  QMouseEvent newEvent(me->type(), mapFromGlobal(ww->mapToGlobal(me->pos())), me->globalPos(),
1387  me->button(), me->buttons(), me->modifiers());
1388  mouseReleaseEvent(&newEvent);
1389  return true;
1390  }
1391  default:
1392  break;
1393  }
1394  }
1395  }
1396 
1397  return QToolBar::eventFilter(watched, event);
1398 }
