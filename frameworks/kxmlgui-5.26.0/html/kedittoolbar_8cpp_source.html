70 static ToolBarList findToolBars(const QDomElement &start)
71 {
72  ToolBarList list;
73 
74  for (QDomElement elem = start; !elem.isNull(); elem = elem.nextSiblingElement()) {
75  if (elem.tagName() == QStringLiteral("ToolBar")) {
76  if (elem.attribute(QStringLiteral("noEdit")) != QLatin1String("true")) {
77  list.append(elem);
78  }
79  } else {
80  if (elem.tagName() != QStringLiteral("MenuBar")) { // there are no toolbars inside the menubar :)
81  list += findToolBars(elem.firstChildElement()); // recursive
82  }
83  }
84  }
85 
86  return list;

165 QString XmlData::toolBarText(const QDomElement &it) const
166 {
167  const QLatin1String attrName("name");
168 
169  QString name;
170  QByteArray txt(it.namedItem(QStringLiteral("text")).toElement().text().toUtf8());
171  if (txt.isEmpty()) {
172  txt = it.namedItem(QStringLiteral("text")).toElement().text().toUtf8();
173  }
174  if (txt.isEmpty()) {
175  name = it.attribute(attrName);
176  } else {
177  QByteArray domain = it.namedItem(QStringLiteral("text")).toElement().attribute(QStringLiteral("translationDomain")).toUtf8();
178  if (domain.isEmpty()) {
179  domain = it.ownerDocument().documentElement().attribute(QStringLiteral("translationDomain")).toUtf8();
180  if (domain.isEmpty()) {
181  domain = KLocalizedString::applicationDomain();
182  }
183  }
184  name = i18nd(domain.constData(), txt.constData());
185  }
186 
187  // the name of the toolbar might depend on whether or not
188  // it is in kparts
189  if ((m_type == XmlData::Shell) ||
190  (m_type == XmlData::Part)) {
191  QString doc_name(m_document.documentElement().attribute(attrName));
192  name += QStringLiteral(" <") + doc_name + QLatin1Char('>');
193  }
194  return name;
195 }

680 void KEditToolBarPrivate::defaultClicked()
681 {
682  if (KMessageBox::warningContinueCancel(q, i18n("Do you really want to reset all toolbars of this application to their default? The changes will be applied immediately."), i18n("Reset Toolbars"), KGuiItem(i18n("Reset"))) != KMessageBox::Continue) {
683  return;
684  }
685 
686  KEditToolBarWidget *oldWidget = m_widget;
687  m_widget = 0;
688  m_accept = false;
689 
690  if (m_factory) {
691  foreach (KXMLGUIClient *client, m_factory->clients()) {
692  const QString file = client->localXMLFile();
693  if (file.isEmpty()) {
694  continue;
695  }
696  //qDebug(240) << "Deleting local xml file" << file;
697  // << "for client" << client << typeid(*client).name();
698  if (QFile::exists(file))
699  if (!QFile::remove(file)) {
700  qWarning() << "Could not delete" << file;
701  }
702  }
703 
704  // Reload the xml files in all clients, now that the local files are gone
705  oldWidget->rebuildKXMLGUIClients();
706 
707  m_widget = new KEditToolBarWidget(q);
708  m_widget->load(m_factory, m_defaultToolBar);
709  } else {
710  int slash = m_file.lastIndexOf(QLatin1Char('/')) + 1;
711  if (slash) {
712  m_file = m_file.mid(slash);

949 void KEditToolBarWidget::rebuildKXMLGUIClients()
950 {
951  if (!d->m_factory) {
952  return;
953  }
954 
955  const QList<KXMLGUIClient *> clients = d->m_factory->clients();
956  //qDebug(240) << "factory: " << clients.count() << " clients";
957 
958  // remove the elements starting from the last going to the first
959  if (!clients.count()) {
960  return;
961  }
962 
963  QListIterator<KXMLGUIClient *> clientIterator = clients;
964  clientIterator.toBack();
965  while (clientIterator.hasPrevious()) {
966  KXMLGUIClient *client = clientIterator.previous();
967  //qDebug(240) << "factory->removeClient " << client;
968  d->m_factory->removeClient(client);
969  }
970 
971  KXMLGUIClient *firstClient = clients.first();
972 
973  // now, rebuild the gui from the first to the last
974  //qDebug(240) << "rebuilding the gui";
975  foreach (KXMLGUIClient *client, clients) {
976  //qDebug(240) << "updating client " << client << " " << client->componentName() << " xmlFile=" << client->xmlFile();
977  QString file(client->xmlFile()); // before setting ui_standards!
978  if (!file.isEmpty()) {
979  // passing an empty stream forces the clients to reread the XML
980  client->setXMLGUIBuildDocument(QDomDocument());
981 
982  // for the shell, merge in ui_standards.rc
983  if (client == firstClient) { // same assumption as in the ctor: first==shell
984  client->loadStandardsXmlFile();
985  }
986 
987  // and this forces it to use the *new* XML file
988  client->setXMLFile(file, client == firstClient /* merge if shell */);
989 
990  // [we can't use reloadXML, it doesn't load ui_standards.rc]
991  }
992  }
993 
994  // Now we can add the clients to the factory
995  // We don't do it in the loop above because adding a part automatically
996  // adds its plugins, so we must make sure the plugins were updated first.
997  foreach (KXMLGUIClient *client, clients) {
998  d->m_factory->addClient(client);
999  }
