86 void SMTP::setSenderAddress(const QString &sender)
87 {
88  senderAddress = sender;
89  int index = senderAddress.indexOf('<');
90  if (index == -1) {
91  return;
92  }
93  senderAddress = senderAddress.mid(index + 1);
94  index = senderAddress.indexOf('>');
95  if (index != -1) {
96  senderAddress = senderAddress.left(index);
97  }
98  senderAddress = senderAddress.simplified();
99  while (1) {
100  index = senderAddress.indexOf(' ');
101  if (index != -1) {
102  senderAddress = senderAddress.mid(index + 1); // take one side
103  } else {
104  break;
105  }
106  }
107  index = senderAddress.indexOf('@');
108  if (index == -1) {
109  senderAddress.append("@localhost"); // won't go through without a local mail system
110  }
111 
112 }

147 {
148  if (!connected) {
149  connectTimerTick();
150  }
151  if (state == Finished && connected) {

153  finished = false;

155  writeString = QStringLiteral("helo %1\r\n").arg(domainName);
156  sock->write(writeString.toLatin1().constData(), writeString.length());

158  if (connected) {
159  // qDebug() << "enabling read on sock...\n";
160  interactTimer.setSingleShot(true);
161  interactTimer.start(timeOut);

210 void SMTP::socketReadyToRead()
211 {
212  int n, nl;
213 
214  // qDebug() << "socketRead() called...";
215  interactTimer.stop();
216 
217  if (!sock) {
218  return;
219  }
220 
221  n = sock->read(readBuffer, SMTP_READ_BUFFER_SIZE - 1);
222  if (n < 0) {
223  return;
224  }
225  readBuffer[n] = 0;
226  lineBuffer += readBuffer;
227  nl = lineBuffer.indexOf('\n');
228  if (nl == -1) {
229  return;
230  }
231  lastLine = lineBuffer.left(nl);
232  lineBuffer = lineBuffer.right(lineBuffer.length() - nl - 1);
233  processLine(&lastLine);
234  if (connected) {
235  interactTimer.setSingleShot(true);
236  interactTimer.start(timeOut);
237  }
238 }

260 void SMTP::processLine(QString *line)
261 {
262  int i, stat;
263  QString tmpstr;
264 
265  i = line->indexOf(' ');
266  tmpstr = line->left(i);
267  if (i > 3) {
268  // qDebug() << "warning: SMTP status code longer than 3 digits: " << tmpstr;
269  }
270  stat = tmpstr.toInt();
271  serverState = static_cast<SMTPServerStatus>(stat);
272  lastState = state;
273 
274  // qDebug() << "smtp state: [" << stat << "][" << *line << "]";
275 
276  switch (stat) {
277  case Greet: //220
278  state = In;
279  writeString = QStringLiteral("helo %1\r\n").arg(domainName);
280  // qDebug() << "out: " << writeString;
281  sock->write(writeString.toLatin1().constData(), writeString.length());
282  break;
283  case Goodbye: //221
284  state = Quit;
285  break;
286  case Successful://250
287  switch (state) {
288  case In:
289  state = Ready;
290  writeString = QStringLiteral("mail from: %1\r\n").arg(senderAddress);
291  // qDebug() << "out: " << writeString;
292  sock->write(writeString.toLatin1().constData(), writeString.length());
293  break;
294  case Ready:
295  state = SentFrom;
296  writeString = QStringLiteral("rcpt to: %1\r\n").arg(recipientAddress);
297  // qDebug() << "out: " << writeString;
298  sock->write(writeString.toLatin1().constData(), writeString.length());
299  break;
300  case SentFrom:
301  state = SentTo;
302  writeString = QStringLiteral("data\r\n");
303  // qDebug() << "out: " << writeString;
304  sock->write(writeString.toLatin1().constData(), writeString.length());
305  break;
306  case Data:
307  state = Finished;
308  finished = true;
309  emit messageSent();
310  break;
311  default:
312  state = CError;
313  // qDebug() << "smtp error (state error): [" << lastState << "]:[" << stat << "][" << *line << "]";
314  socketClosed();
315  emit error(Command);
316  break;
317  }
318  break;
319  case ReadyData: //354
320  state = Data;
321  writeString = QStringLiteral("Subject: %1\r\n").arg(messageSubject);
322  writeString += messageHeader;
323  writeString += QLatin1String("\r\n");
324  writeString += messageBody;
325  writeString += QLatin1String(".\r\n");
326  // qDebug() << "out: " << writeString;
327  sock->write(writeString.toLatin1().constData(), writeString.length());
328  break;
329  case Error: //501
330  state = CError;
331  // qDebug() << "smtp error (command error): [" << lastState << "]:[" << stat << "][" << *line << "]\n";
332  socketClosed();
333  emit error(Command);
334  break;
335  case Unknown: //550
336  state = CError;
337  // qDebug() << "smtp error (unknown user): [" << lastState << "]:[" << stat << "][" << *line << "]";
338  socketClosed();
339  emit error(UnknownUser);
340  break;
341  default:
342  state = CError;
343  // qDebug() << "unknown response: [" << lastState << "]:[" << stat << "][" << *line << "]";
344  socketClosed();
345  emit error(UnknownResponse);
346  }
347 }
