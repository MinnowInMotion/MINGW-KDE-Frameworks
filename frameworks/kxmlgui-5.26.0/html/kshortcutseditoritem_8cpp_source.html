83 QVariant KShortcutsEditorItem::data(int column, int role) const
84 {
85  switch (role) {
86  case Qt::DisplayRole:
87  switch (column) {
88  case Name:
89  return m_actionNameInTable;
90  case Id:
91  return m_id;
92  case LocalPrimary:
93  case LocalAlternate:
94  case GlobalPrimary:
95  case GlobalAlternate:
96  return keySequence(column);
97 #if 0
98  case ShapeGesture:
99  return KGestureMap::self()->shapeGesture(m_action).shapeName();
100  case RockerGesture:
101  return KGestureMap::self()->rockerGesture(m_action).rockerName();
102 #endif
103  default:
104  break;
105  }
106  break;
107  case Qt::DecorationRole:
108  if (column == Name) {
109  return m_action->icon();
110  } else {
111  return QIcon();
112  }
113  case Qt::WhatsThisRole:
114  return m_action->whatsThis();
115  case Qt::ToolTipRole:
116  // There is no such thing as a QAction::description(). So we have
117  // nothing to display here.
118  return QVariant();
119  case Qt::FontRole:
120  if (column == Name && m_isNameBold) {
121  QFont modifiedFont = treeWidget()->font();
122  modifiedFont.setBold(true);
123  return modifiedFont;
124  }
125  break;
126  case KExtendableItemDelegate::ShowExtensionIndicatorRole:
127  switch (column) {
128  case Name:
129  return false;
130  case LocalPrimary:
131  case LocalAlternate:
132  return !m_action->property("isShortcutConfigurable").isValid()
133  || m_action->property("isShortcutConfigurable").toBool();
134 #if HAVE_GLOBALACCEL
135  case GlobalPrimary:
136  case GlobalAlternate:
137  if (!KGlobalAccel::self()->hasShortcut(m_action)) {
138  return false;
139  }
140  return true;
141 #endif
142  default:
143  return false;
144  }
145  //the following are custom roles, defined in this source file only
146  case ShortcutRole:
147  switch (column) {
148  case LocalPrimary:
149  case LocalAlternate:
150  case GlobalPrimary:
151  case GlobalAlternate:
152  return keySequence(column);
153 #if 0
154  case ShapeGesture: { //scoping for "ret"
155  QVariant ret;
156  ret.setValue(KGestureMap::self()->shapeGesture(m_action));
157  return ret;
158  }
159  case RockerGesture: {
160  QVariant ret;
161  ret.setValue(KGestureMap::self()->rockerGesture(m_action));
162  return ret;
163  }
164 #endif
165  default:
166  // Column not valid for this role
167  Q_ASSERT(false);
168  return QVariant();
169  }
170 
171  case DefaultShortcutRole: {
172  QList<QKeySequence> defaultShortcuts = m_action->property("defaultShortcuts").value<QList<QKeySequence> >();
173 #if HAVE_GLOBALACCEL
174  QList<QKeySequence> defaultGlobalShortcuts = KGlobalAccel::self()->defaultShortcut(m_action);
175 #endif
176 
177  switch (column) {
178  case LocalPrimary:
179  return primarySequence(defaultShortcuts);
180  case LocalAlternate:
181  return alternateSequence(defaultShortcuts);
182 #if HAVE_GLOBALACCEL
183  case GlobalPrimary:
184  return primarySequence(defaultGlobalShortcuts);
185  case GlobalAlternate:
186  return alternateSequence(defaultGlobalShortcuts);
187 #endif
188 #if 0
189  case ShapeGesture: {
190  QVariant ret;
191  ret.setValue(KGestureMap::self()->defaultShapeGesture(m_action));
192  return ret;
193  }
194  case RockerGesture: {
195  QVariant ret;
196  ret.setValue(KGestureMap::self()->defaultRockerGesture(m_action));
197  return ret;
198  }
199 #endif
200  default:
201  // Column not valid for this role
202  Q_ASSERT(false);
203  return QVariant();
204  }
205  }
206  case ObjectRole:

245 void KShortcutsEditorItem::setKeySequence(uint column, const QKeySequence &seq)
246 {
247  QList<QKeySequence> ks;
248 #if HAVE_GLOBALACCEL
249  if (column == GlobalPrimary || column == GlobalAlternate) {
250  ks = KGlobalAccel::self()->shortcut(m_action);
251  if (!m_oldGlobalShortcut) {
252  m_oldGlobalShortcut = new QList<QKeySequence>(ks);
253  }
254  } else
255 #endif
256  {
257  ks = m_action->shortcuts();
258  if (!m_oldLocalShortcut) {
259  m_oldLocalShortcut = new QList<QKeySequence>(ks);
260  }
261  }
262 
263  if (column == LocalAlternate || column == GlobalAlternate) {
264  if (ks.isEmpty()) {
265  ks << QKeySequence();
266  }
267 
268  if (ks.size() <= 1) {
269  ks << seq;
270  } else {
271  ks[1] = seq;
272  }
273  } else {
274  if (ks.isEmpty()) {
275  ks << seq;
276  } else {
277  ks[0] = seq;
278  }
279  }
280 
281  //avoid also setting the default shortcut - what we are setting here is custom by definition
282 #if HAVE_GLOBALACCEL
283  if (column == GlobalPrimary || column == GlobalAlternate) {
284  KGlobalAccel::self()->setShortcut(m_action, ks, KGlobalAccel::NoAutoloading);
285 
286  } else
287 #endif
288  {
289  m_action->setShortcuts(ks);
290  }
291 
292  updateModified();
293 }

344 bool KShortcutsEditorItem::isModified(uint column) const
345 {
346  switch (column) {
347  case Name:
348  return false;
349  case LocalPrimary:
350  case LocalAlternate:
351  if (!m_oldLocalShortcut) {
352  return false;
353  }
354  if (column == LocalPrimary) {
355  return primarySequence(*m_oldLocalShortcut) != primarySequence(m_action->shortcuts());
356  } else {
357  return alternateSequence(*m_oldLocalShortcut) != alternateSequence(m_action->shortcuts());
358  }
359 #if HAVE_GLOBALACCEL
360  case GlobalPrimary:
361  case GlobalAlternate:
362  if (!m_oldGlobalShortcut) {
363  return false;
364  }
365  if (column == GlobalPrimary) {
366  return primarySequence(*m_oldGlobalShortcut) != primarySequence(KGlobalAccel::self()->shortcut(m_action));
367  } else {
368  return alternateSequence(*m_oldGlobalShortcut) != alternateSequence(KGlobalAccel::self()->shortcut(m_action));
369  }
370 #endif
371 #if 0
372  case ShapeGesture:
373  return static_cast<bool>(m_oldShapeGesture);
374  case RockerGesture:
375  return static_cast<bool>(m_oldRockerGesture);
376 #endif
377  default:
378  return false;
