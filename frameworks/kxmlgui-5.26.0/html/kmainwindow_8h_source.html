697 template <typename T>
698 inline void kRestoreMainWindows()
699 {
700  for (int n = 1; KMainWindow::canBeRestored(n); ++n) {
701  const QString className = KMainWindow::classNameOfToplevel(n);
702  if (className == QLatin1String(T::staticMetaObject.className())) {
703  (new T)->restore(n);
704  }
705  }
706 }

708 template <typename T0, typename T1>
709 inline void kRestoreMainWindows()
710 {
711  const char *classNames[2];
712  classNames[0] = T0::staticMetaObject.className();
713  classNames[1] = T1::staticMetaObject.className();
714  for (int n = 1; KMainWindow::canBeRestored(n); ++n) {
715  const QString className = KMainWindow::classNameOfToplevel(n);
716  if (className == QLatin1String(classNames[0])) {
717  (new T0)->restore(n);
718  } else if (className == QLatin1String(classNames[1])) {
719  (new T1)->restore(n);
720  }
721  }
722 }
