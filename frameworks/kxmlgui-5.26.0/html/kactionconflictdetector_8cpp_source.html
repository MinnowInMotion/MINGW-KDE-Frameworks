34 class KActionConflictDetector : public QObject
35 {
36 public:
37  explicit KActionConflictDetector(QObject *parent = 0)
38  : QObject(parent)
39  {
40  }
41 
42  bool eventFilter(QObject *watched, QEvent *event) Q_DECL_OVERRIDE
43  {
44  if (qobject_cast<QAction *>(watched) && (event->type() == QEvent::Shortcut)) {
45  QShortcutEvent *se = static_cast<QShortcutEvent *>(event);
46  if (se->isAmbiguous()) {
47  KMessageBox::information(
48  Q_NULLPTR, // No widget to be seen around here
49  i18n("The key sequence '%1' is ambiguous. Use 'Configure Shortcuts'\n"
50  "from the 'Settings' menu to solve the ambiguity.\n"
51  "No action will be triggered.",
52  se->key().toString(QKeySequence::NativeText)),
53  i18n("Ambiguous shortcut detected"));
54  return true;
55  }
56  }
57 
58  return QObject::eventFilter(watched, event);
59  }
60 };
61 
62 void _k_installConflictDetector()

64  QCoreApplication *app = QCoreApplication::instance();
65  app->installEventFilter(new KActionConflictDetector(app));
66 }
