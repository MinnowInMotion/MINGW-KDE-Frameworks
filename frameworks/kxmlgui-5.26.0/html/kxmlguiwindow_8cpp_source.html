182 void KXmlGuiWindow::setupGUI(const QSize &defaultSize, StandardWindowOptions options, const QString &xmlfile)
183 {
184  K_D(KXmlGuiWindow);
185 
186  if (options & Keys) {
187  KStandardAction::keyBindings(guiFactory(),
188  SLOT(configureShortcuts()), actionCollection());
189  }
190 
191  if ((options & StatusBar) && statusBar()) {
192  createStandardStatusBarAction();
193  }
194 
195  if (options & ToolBar) {
196  setStandardToolBarMenuEnabled(true);
197  KStandardAction::configureToolbars(this,
198  SLOT(configureToolbars()), actionCollection());
199  }
200 
201  d->defaultSize = defaultSize;
202 
203  if (options & Create) {
204  createGUI(xmlfile);
205  }
206 
207  if (d->defaultSize.isValid()) {
208  resize(d->defaultSize);
209  } else if (isHidden()) {
210  adjustSize();
211  }
212 
213  if (options & Save) {
214  const KConfigGroup cg(autoSaveConfigGroup());
215  if (cg.isValid()) {
216  setAutoSaveSettings(cg);
217  } else {
218  setAutoSaveSettings();
219  }
220  }

222 void KXmlGuiWindow::createGUI(const QString &xmlfile)
223 {
224  K_D(KXmlGuiWindow);
225  // disabling the updates prevents unnecessary redraws
226  //setUpdatesEnabled( false );
227 
228  // just in case we are rebuilding, let's remove our old client
229  guiFactory()->removeClient(this);
230 
231  // make sure to have an empty GUI
232  QMenuBar *mb = menuBar();
233  if (mb) {
234  mb->clear();
235  }
236 
237  qDeleteAll(toolBars()); // delete all toolbars
238 
239  // don't build a help menu unless the user ask for it
240  if (d->showHelpMenu) {
241  delete d->helpMenu;
242  // we always want a help menu
243  d->helpMenu = new KHelpMenu(this, KAboutData::applicationData(), true);
244 
245  KActionCollection *actions = actionCollection();
246  QAction *helpContentsAction = d->helpMenu->action(KHelpMenu::menuHelpContents);
247  QAction *whatsThisAction = d->helpMenu->action(KHelpMenu::menuWhatsThis);
248  QAction *reportBugAction = d->helpMenu->action(KHelpMenu::menuReportBug);
249  QAction *switchLanguageAction = d->helpMenu->action(KHelpMenu::menuSwitchLanguage);
250  QAction *aboutAppAction = d->helpMenu->action(KHelpMenu::menuAboutApp);
251  QAction *aboutKdeAction = d->helpMenu->action(KHelpMenu::menuAboutKDE);
252  QAction *donateAction = d->helpMenu->action(KHelpMenu::menuDonate);
253 
254  if (helpContentsAction) {
255  actions->addAction(helpContentsAction->objectName(), helpContentsAction);
256  }
257  if (whatsThisAction) {
258  actions->addAction(whatsThisAction->objectName(), whatsThisAction);
259  }
260  if (reportBugAction) {
261  actions->addAction(reportBugAction->objectName(), reportBugAction);
262  }
263  if (switchLanguageAction) {
264  actions->addAction(switchLanguageAction->objectName(), switchLanguageAction);
265  }
266  if (aboutAppAction) {
267  actions->addAction(aboutAppAction->objectName(), aboutAppAction);
268  }
269  if (aboutKdeAction) {
270  actions->addAction(aboutKdeAction->objectName(), aboutKdeAction);
271  }
272  if (donateAction) {
273  actions->addAction(donateAction->objectName(), donateAction);
274  }
275  }
276 
277  const QString windowXmlFile = xmlfile.isNull() ? componentName() + QStringLiteral("ui.rc") : xmlfile;
278 
279  // Help beginners who call setXMLFile and then setupGUI...
280  if (!xmlFile().isEmpty() && xmlFile() != windowXmlFile) {
281  qWarning() << "You called setXMLFile(" << xmlFile() << ") and then createGUI or setupGUI,"
282  << "which also calls setXMLFile and will overwrite the file you have previously set.\n"
283  << "You should call createGUI(" << xmlFile() << ") or setupGUI(<options>," << xmlFile() << ") instead.";
284  }
285 
286  // we always want to load in our global standards file
287  loadStandardsXmlFile();
288 
289  // now, merge in our local xml file.
290  setXMLFile(windowXmlFile, true);
291 
292  // make sure we don't have any state saved already
293  setXMLGUIBuildDocument(QDomDocument());
294 
295  // do the actual GUI building
296  guiFactory()->reset();
297  guiFactory()->addClient(this);
298 
299  // setUpdatesEnabled( true );
300 }

314 void KXmlGuiWindow::setStandardToolBarMenuEnabled(bool enable)
315 {
316  K_D(KXmlGuiWindow);
317  if (enable) {
318  if (d->toolBarHandler) {
319  return;
320  }
321 
322  d->toolBarHandler = new KDEPrivate::ToolBarHandler(this);
323 
324  if (factory()) {
325  factory()->addClient(d->toolBarHandler);
326  }
327  } else {
328  if (!d->toolBarHandler) {
329  return;
330  }
331 
332  if (factory()) {
333  factory()->removeClient(d->toolBarHandler);
334  }
335 
336  delete d->toolBarHandler;
337  d->toolBarHandler = 0;
338  }
339 }

347 void KXmlGuiWindow::createStandardStatusBarAction()
348 {
349  K_D(KXmlGuiWindow);
350  if (!d->showStatusBarAction) {
351  d->showStatusBarAction = KStandardAction::showStatusbar(this, SLOT(setSettingsDirty()), actionCollection());
352  QStatusBar *sb = statusBar(); // Creates statusbar if it doesn't exist already.
353  connect(d->showStatusBarAction, SIGNAL(toggled(bool)), sb, SLOT(setVisible(bool)));
354  d->showStatusBarAction->setChecked(sb->isHidden());
355  } else {
356  // If the language has changed, we'll need to grab the new text and whatsThis
357  QAction *tmpStatusBar = KStandardAction::showStatusbar(NULL, NULL, NULL);
358  d->showStatusBarAction->setText(tmpStatusBar->text());
359  d->showStatusBarAction->setWhatsThis(tmpStatusBar->whatsThis());
360  delete tmpStatusBar;
361  }
362 }
