165 void KMWSessionManager::commitData(QSessionManager &sm)
166 {
167  if (!sm.allowsInteraction()) {
168  return;
169  }
170 
171  /*
172  Purpose of this exercise: invoke queryClose() without actually closing the
173  windows, because
174  - queryClose() may contain session management code, so it must be invoked
175  - actually closing windows may quit the application - cf.
176  QGuiApplication::quitOnLastWindowClosed()
177  - quitting the application and thus closing the session manager connection
178  violates the X11 XSMP protocol.
179  The exact requirement of XSMP that would be broken is,
180  in the description of the client's state machine:
181 
182  save-yourself-done: (changing state is forbidden)
183 
184  Closing the session manager connection causes a state change.
185  Worst of all, that is a real problem with ksmserver - it will not save
186  applications that quit on their own in state save-yourself-done.
187  */
188  foreach (KMainWindow *window, KMainWindow::memberList()) {
189  if (window->testAttribute(Qt::WA_WState_Hidden)) {
190  continue;
191  }
192  QCloseEvent e;
193  QApplication::sendEvent(window, &e);
194  if (!e.isAccepted()) {
195  sm.cancel();
196  return;
197  }
198  }
199 }

265 static bool endsWithHashNumber(const QString &s)
266 {
267  for (int i = s.length() - 1;
268  i > 0;
269  --i) {
270  if (s[ i ] == QLatin1Char('#') && i != s.length() - 1) {
271  return true; // ok
272  }
273  if (!s[ i ].isDigit()) {
274  break;
275  }
276  }
277  return false;
278 }

280 static inline bool isValidDBusObjectPathCharacter(const QChar &c)
281 {
282  ushort u = c.unicode();
283  return (u >= QLatin1Char('a') && u <= QLatin1Char('z'))
284  || (u >= QLatin1Char('A') && u <= QLatin1Char('Z'))
285  || (u >= QLatin1Char('0') && u <= QLatin1Char('9'))
286  || (u == QLatin1Char('_')) || (u == QLatin1Char('/'));
287 }

289 void KMainWindowPrivate::polish(KMainWindow *q)
290 {
291  // Set a unique object name. Required by session management, window management, and for the dbus interface.
292  QString objname;
293  QString s;
294  int unusedNumber = 1;
295  const QString name = q->objectName();
296  bool startNumberingImmediately = true;
297  bool tryReuse = false;
298  if (name.isEmpty()) {
299  // no name given
300  objname = QStringLiteral("MainWindow#");
301  } else if (name.endsWith(QLatin1Char('#'))) {
302  // trailing # - always add a number - KWin uses this for better grouping
303  objname = name;
304  } else if (endsWithHashNumber(name)) {
305  // trailing # with a number - like above, try to use the given number first
306  objname = name;
307  tryReuse = true;
308  startNumberingImmediately = false;
309  } else {
310  objname = name;
311  startNumberingImmediately = false;
312  }
313 
314  s = objname;
315  if (startNumberingImmediately) {
316  s += QLatin1Char('1');
317  }
318 
319  for (;;) {
320  const QList<QWidget *> list = qApp->topLevelWidgets();
321  bool found = false;
322  foreach (QWidget *w, list) {
323  if (w != q && w->objectName() == s) {
324  found = true;
325  break;
326  }
327  }
328  if (!found) {
329  break;
330  }
331  if (tryReuse) {
332  objname = name.left(name.length() - 1); // lose the hash
333  unusedNumber = 0; // start from 1 below
334  tryReuse = false;
335  }
336  s.setNum(++unusedNumber);
337  s = objname + s;
338  }
339  q->setObjectName(s);
340  q->winId(); // workaround for setWindowRole() crashing, and set also window role, just in case TT
341  q->setWindowRole(s); // will keep insisting that object name suddenly should not be used for window role
342 
343  dbusName = QLatin1Char('/') + QCoreApplication::applicationName() + QLatin1Char('/');
344  dbusName += q->objectName().replace(QLatin1Char('/'), QLatin1Char('_'));
345  // Clean up for dbus usage: any non-alphanumeric char should be turned into '_'
346  const int len = dbusName.length();
347  for (int i = 0; i < len; ++i) {
348  if (!isValidDBusObjectPathCharacter(dbusName[i])) {
349  dbusName[i] = QLatin1Char('_');
350  }
351  }
352 
353  QDBusConnection::sessionBus().registerObject(dbusName, q, QDBusConnection::ExportScriptableSlots |
354  QDBusConnection::ExportScriptableProperties |
355  QDBusConnection::ExportNonScriptableSlots |
356  QDBusConnection::ExportNonScriptableProperties |
357  QDBusConnection::ExportAdaptors);
358 }

569 void KMainWindow::saveMainWindowSettings(KConfigGroup &cg)
570 {
571  K_D(KMainWindow);
572  //qDebug(200) << "KMainWindow::saveMainWindowSettings " << cg.name();
573 
574  // Called by session management - or if we want to save the window size anyway
575  if (d->autoSaveWindowSize) {
576  KWindowConfig::saveWindowSize(windowHandle(), cg);
577  }
578 
579  // One day will need to save the version number, but for now, assume 0
580  // Utilise the QMainWindow::saveState() functionality.
581  const QByteArray state = saveState();
582  cg.writeEntry("State", state.toBase64());
583 
584  QStatusBar *sb = internalStatusBar(this);
585  if (sb) {
586  if (!cg.hasDefault("StatusBar") && !sb->isHidden()) {
587  cg.revertToDefault("StatusBar");
588  } else {
589  cg.writeEntry("StatusBar", sb->isHidden() ? "Disabled" : "Enabled");
590  }
591  }
592 
593  QMenuBar *mb = internalMenuBar(this);
594  if (mb) {
595  if (!cg.hasDefault("MenuBar") && !mb->isHidden()) {
596  cg.revertToDefault("MenuBar");
597  } else {
598  cg.writeEntry("MenuBar", mb->isHidden() ? "Disabled" : "Enabled");
599  }
600  }
601 
602  if (!autoSaveSettings() || cg.name() == autoSaveGroup()) {
603  // TODO should be cg == d->autoSaveGroup, to compare both kconfig and group name
604  if (!cg.hasDefault("ToolBarsMovable") && !KToolBar::toolBarsLocked()) {
605  cg.revertToDefault("ToolBarsMovable");
606  } else {
607  cg.writeEntry("ToolBarsMovable", KToolBar::toolBarsLocked() ? "Disabled" : "Enabled");
608  }
609  }
610 
611  int n = 1; // Toolbar counter. toolbars are counted from 1,
612  foreach (KToolBar *toolbar, toolBars()) {
613  QByteArray groupName("Toolbar");
614  // Give a number to the toolbar, but prefer a name if there is one,
615  // because there's no real guarantee on the ordering of toolbars
616  groupName += (toolbar->objectName().isEmpty() ? QByteArray::number(n) : QByteArray(" ").append(toolbar->objectName().toUtf8()));
617 
618  KConfigGroup toolbarGroup(&cg, groupName.constData());
619  toolbar->saveSettings(toolbarGroup);
620  n++;
621  }
622 }

624 bool KMainWindow::readPropertiesInternal(KConfig *config, int number)
625 {
626  K_D(KMainWindow);
627 
628  const bool oldLetDirtySettings = d->letDirtySettings;
629  d->letDirtySettings = false;
630 
631  if (number == 1) {
632  readGlobalProperties(config);
633  }
634 
635  // in order they are in toolbar list
636  KConfigGroup cg(config, QByteArray(WINDOW_PROPERTIES).append(QByteArray::number(number)).constData());
637 
638  // restore the object name (window role)
639  if (cg.hasKey("ObjectName")) {
640  setObjectName(cg.readEntry("ObjectName"));
641  }
642 
643  d->sizeApplied = false; // since we are changing config file, reload the size of the window
644  // if necessary. Do it before the call to applyMainWindowSettings.
645  applyMainWindowSettings(cg); // Menubar, statusbar and toolbar settings.
646 
647  KConfigGroup grp(config, QByteArray::number(number).constData());
648  readProperties(grp);
649 
650  d->letDirtySettings = oldLetDirtySettings;
651 
652  return true;
653 }

655 void KMainWindow::applyMainWindowSettings(const KConfigGroup &cg)
656 {
657  K_D(KMainWindow);
658  //qDebug(200) << "KMainWindow::applyMainWindowSettings " << cg.name();
659 
660  QWidget *focusedWidget = QApplication::focusWidget();
661 
662  const bool oldLetDirtySettings = d->letDirtySettings;
663  d->letDirtySettings = false;
664 
665  if (!d->sizeApplied) {
666  winId(); // ensure there's a window created
667  KWindowConfig::restoreWindowSize(windowHandle(), cg);
668  // NOTICE: QWindow::setGeometry() does NOT impact the backing QWidget geometry even if the platform
669  // window was created -> QTBUG-40584. We therefore copy the size here.
670  // TODO: remove once this was resolved in QWidget QPA
671  resize(windowHandle()->size());
672  d->sizeApplied = true;
673  }
674 
675  QStatusBar *sb = internalStatusBar(this);
676  if (sb) {
677  QString entry = cg.readEntry("StatusBar", "Enabled");
678  sb->setVisible( entry != QLatin1String("Disabled") );
679  }
680 
681  QMenuBar *mb = internalMenuBar(this);
682  if (mb) {
683  QString entry = cg.readEntry("MenuBar", "Enabled");
684  mb->setVisible( entry != QLatin1String("Disabled") );
685  }
686 
687  if (!autoSaveSettings() || cg.name() == autoSaveGroup()) { // TODO should be cg == d->autoSaveGroup, to compare both kconfig and group name
688  QString entry = cg.readEntry("ToolBarsMovable", "Disabled");
689  KToolBar::setToolBarsLocked(entry == QLatin1String("Disabled"));
690  }
691 
692  int n = 1; // Toolbar counter. toolbars are counted from 1,
693  foreach (KToolBar *toolbar, toolBars()) {
694  QByteArray groupName("Toolbar");
695  // Give a number to the toolbar, but prefer a name if there is one,
696  // because there's no real guarantee on the ordering of toolbars
697  groupName += (toolbar->objectName().isEmpty() ? QByteArray::number(n) : QByteArray(" ").append(toolbar->objectName().toUtf8()));
698 
699  KConfigGroup toolbarGroup(&cg, groupName.constData());
700  toolbar->applySettings(toolbarGroup);
701  n++;
702  }
703 
704  QByteArray state;
705  if (cg.hasKey("State")) {
706  state = cg.readEntry("State", state);
707  state = QByteArray::fromBase64(state);
708  // One day will need to load the version number, but for now, assume 0
709  restoreState(state);
710  }
711 
712  if (focusedWidget) {
713  focusedWidget->setFocus();
714  }
715 
716  d->settingsDirty = false;
717  d->letDirtySettings = oldLetDirtySettings;
718 }

804 bool KMainWindow::event(QEvent *ev)
805 {
806  K_D(KMainWindow);
807  switch (ev->type()) {
808 #ifdef Q_OS_WIN
809  case QEvent::Move:
810 #endif
811  case QEvent::Resize:
812  d->setSizeDirty();
813  break;
814  case QEvent::Polish:
815  d->polish(this);
816  break;
817  case QEvent::ChildPolished: {
818  QChildEvent *event = static_cast<QChildEvent *>(ev);
819  QDockWidget *dock = qobject_cast<QDockWidget *>(event->child());
820  KToolBar *toolbar = qobject_cast<KToolBar *>(event->child());
821  QMenuBar *menubar = qobject_cast<QMenuBar *>(event->child());
822  if (dock) {
823  connect(dock, SIGNAL(dockLocationChanged(Qt::DockWidgetArea)),
824  this, SLOT(setSettingsDirty()));
825  connect(dock, SIGNAL(visibilityChanged(bool)),
826  this, SLOT(setSettingsDirty()), Qt::QueuedConnection);
827  connect(dock, SIGNAL(topLevelChanged(bool)),
828  this, SLOT(setSettingsDirty()));
829 
830  // there is no signal emitted if the size of the dock changes,
831  // hence install an event filter instead
832  dock->installEventFilter(k_ptr->dockResizeListener);
833  } else if (toolbar) {
834  // there is no signal emitted if the size of the toolbar changes,
835  // hence install an event filter instead
836  toolbar->installEventFilter(k_ptr->dockResizeListener);
837  } else if (menubar) {
838  // there is no signal emitted if the size of the menubar changes,
839  // hence install an event filter instead
840  menubar->installEventFilter(k_ptr->dockResizeListener);
841  }
842  }
843  break;
844  case QEvent::ChildRemoved: {
845  QChildEvent *event = static_cast<QChildEvent *>(ev);
846  QDockWidget *dock = qobject_cast<QDockWidget *>(event->child());
847  KToolBar *toolbar = qobject_cast<KToolBar *>(event->child());
848  QMenuBar *menubar = qobject_cast<QMenuBar *>(event->child());
849  if (dock) {
850  disconnect(dock, SIGNAL(dockLocationChanged(Qt::DockWidgetArea)),
851  this, SLOT(setSettingsDirty()));
852  disconnect(dock, SIGNAL(visibilityChanged(bool)),
853  this, SLOT(setSettingsDirty()));
854  disconnect(dock, SIGNAL(topLevelChanged(bool)),
855  this, SLOT(setSettingsDirty()));
856  dock->removeEventFilter(k_ptr->dockResizeListener);
857  } else if (toolbar) {
858  toolbar->removeEventFilter(k_ptr->dockResizeListener);
859  } else if (menubar) {
860  menubar->removeEventFilter(k_ptr->dockResizeListener);
861  }
862  }
863  break;
864  default:
865  break;
866  }
867  return QMainWindow::event(ev);
868 }
