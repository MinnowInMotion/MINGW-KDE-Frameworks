57 void ImageLoader::slotDownload(KJob *job)
58 {
59  if (job->error()) {
60  m_buffer.clear();
61  return;
62  }
63  QImage image;
64  image.loadFromData(m_buffer);
65  m_buffer.clear();
66 
67  if (m_previewType == EntryInternal::PreviewSmall1
68  || m_previewType == EntryInternal::PreviewSmall2
69  || m_previewType == EntryInternal::PreviewSmall3) {
70  if (image.width() > PreviewWidth || image.height() > PreviewHeight) {
71  // if the preview is really big, first scale fast to a smaller size, then smooth to desired size
72  if (image.width() > 4 * PreviewWidth || image.height() > 4 * PreviewHeight) {
73  image = image.scaled(2 * PreviewWidth, 2 * PreviewHeight, Qt::KeepAspectRatio, Qt::FastTransformation);
74  }
75  image = image.scaled(PreviewWidth, PreviewHeight, Qt::KeepAspectRatio, Qt::SmoothTransformation);
76  } else if (image.width() <= PreviewWidth / 2 && image.height() <= PreviewHeight / 2) {
77  // upscale tiny previews to double size
78  image = image.scaled(2 * image.width(), 2 * image.height());
79  }
80  }
81 
82  m_entry.setPreviewImage(image, m_previewType);
83  emit signalPreviewLoaded(m_entry, m_previewType);
84  deleteLater();
85 }
