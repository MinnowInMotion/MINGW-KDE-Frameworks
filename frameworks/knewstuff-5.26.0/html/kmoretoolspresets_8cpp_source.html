114 QList<KMoreToolsService*> KMoreToolsPresetsPrivate::registerServicesByGroupingNames(QString* firstMoreSectionDesktopEntryName, KMoreTools* kmt, const QStringList& groupingNames)
115 {
116  static QHash<QString, QList<QString>> dict;
117 
118  // The corresponding desktop files are located here:
119  // 'knewstuff/data/kmoretools-desktopfiles/'
120 
121  // Use KMoreToolsTest2::testDialogForGroupingNames to see if the settings
122  // here are correct.
123 
124  // NOTE that the desktopentry names must be registered in
125  // registerServiceByDesktopEntryName above.
126 
127  // For special handlings about naming in the menu etc. see kmoretoolsmenufactory.cpp/addItemsForGroupingNameWithSpecialHandling
128 
129  //
130  // definitions begin:
131  //
132  dict.insert(_("disk-usage"), { _("kdf"), _("org.kde.filelight") });
133  dict.insert(_("disk-partitions"), { _("gparted"), _("org.kde.PartitionManager"), _("disk") });
134  dict.insert(_("files-find"), { _("org.kde.kfind"), _("more:"), _("gnome-search-tool"), _("catfish") });
135  dict.insert(_("git-clients-for-folder"), { _("git-cola-folder-handler"), _("gitk.kmt-edition"),
136  _("giggle"), _("qgit.kmt-edition"), _("gitg") });
137  dict.insert(_("git-clients-and-actions"), { _("git-cola-folder-handler"), _("git-cola-view-history.kmt-edition"),
138  _("giggle"), _("gitk.kmt-edition"), _("qgit.kmt-edition"), _("gitg") });
139  dict.insert(_("icon-browser"), { _("org.kde.plasma.cuttlefish.kmt-edition") });
140  dict.insert(_("screenshot-take"), { _("org.kde.ksnapshot"), _("org.kde.kscreengenie"), _("shutter"), _("kaption"), _("hotshots") });
141  dict.insert(_("system-monitor-processes"), { _("org.kde.ksysguard"), _("more:"), _("htop"), _("xfce4-taskmanager") });
142  dict.insert(_("system-monitor-logs"), { _("ksystemlog") });
143  dict.insert(_("time-countdown"), { _("org.gnome.clocks"), _("org.kde.ktimer") });
144  //
145  // ...definitions end
146  //
147 
148  QList<KMoreToolsService*> resultList;
149  QSet<QString> alreadyUsedDesktopEntryNames; // including the "more:" keyword
150  bool nextIsMore = false;
151 
152  Q_FOREACH (const QString &groupingName, groupingNames) {
153  auto iter = dict.constFind(groupingName);
154  if (iter != dict.constEnd()) {
155  Q_FOREACH(const QString &desktopEntryName, *iter) {
156  if (!alreadyUsedDesktopEntryNames.contains(desktopEntryName)) {
157  if (desktopEntryName == _("more:")) {
158  nextIsMore = true;
159  } else {
160  if (nextIsMore) { // this will be only set once
161  *firstMoreSectionDesktopEntryName = desktopEntryName;
162  nextIsMore = false;
163  }
164  KMoreToolsService *kmtService = KMoreToolsPresets::registerServiceByDesktopEntryName(kmt, desktopEntryName);
165  if (kmtService) { // Do not add null pointers caused by missing or broken .desktop files
166  resultList << kmtService;
167  }
168  }
169  } else {
170  alreadyUsedDesktopEntryNames.insert(desktopEntryName);
171  }
172  }
173  } else {
174  qWarning() << "KMoreToolsPresets::registerServicesByGroupingName: groupingName not found: " << groupingName;
175  }
176  }
177 
178  if (resultList.isEmpty()) {
179  qWarning() << "KMoreToolsPresets::registerServicesByGroupingName: " << groupingNames << ". Nothing found in this groupings. HINT: check for invalid grouping names.";
180  }
181 
182  return resultList;
183 }
