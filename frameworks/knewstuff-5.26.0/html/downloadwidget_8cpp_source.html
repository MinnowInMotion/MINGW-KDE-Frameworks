124 void DownloadWidgetPrivate::sortingChanged()
125 {
126  Provider::SortMode sortMode = Provider::Newest;
127  if (ui.ratingRadio->isChecked()) {
128  sortMode = Provider::Rating;
129  } else if (ui.mostDownloadsRadio->isChecked()) {
130  sortMode = Provider::Downloads;
131  } else if (ui.installedRadio->isChecked()) {
132  sortMode = Provider::Installed;
133  }
134 
135  model->clearEntries();
136  if (sortMode == Provider::Installed) {
137  ui.m_searchEdit->clear();
138  }
139  ui.m_searchEdit->setEnabled(sortMode != Provider::Installed);
140 
141  engine->setSortMode(sortMode);
142 }

310 void DownloadWidgetPrivate::setListViewMode(QListView::ViewMode mode)
311 {
312  if (ui.m_listView->viewMode() == mode) {
313  return;
314  }
315 
316  ItemsViewBaseDelegate *oldDelegate = delegate;
317  if (mode == QListView::ListMode) {
318  delegate = new ItemsViewDelegate(ui.m_listView, engine, q);
319  ui.m_listView->setViewMode(QListView::ListMode);
320  ui.m_listView->setResizeMode(QListView::Fixed);
321  } else {
322  delegate = new ItemsGridViewDelegate(ui.m_listView, engine, q);
323  ui.m_listView->setViewMode(QListView::IconMode);
324  ui.m_listView->setResizeMode(QListView::Adjust);
325  }
326  ui.m_listView->setItemDelegate(delegate);
327  delete oldDelegate;
328 
329  q->connect(ui.m_listView, SIGNAL(doubleClicked(QModelIndex)), delegate, SLOT(slotDetailsClicked(QModelIndex)));
330  q->connect(delegate, SIGNAL(signalShowDetails(KNS3::EntryInternal)), q, SLOT(slotShowDetails(KNS3::EntryInternal)));
331 }

350 void DownloadWidgetPrivate::displayMessage(const QString &msg, KTitleWidget::MessageType type, int timeOutMs)
351 {
352  if (!messageTimer) {
353  messageTimer = new QTimer;
354  messageTimer->setSingleShot(true);
355  q->connect(messageTimer, SIGNAL(timeout()), q, SLOT(slotResetMessage()));
356  }
357  // stop the pending timer if present
358  messageTimer->stop();
359 
360  // set text to messageLabel
361  ui.m_titleWidget->setComment(msg, type);
362 
363  // single shot the resetColors timer (and create it if null)
364  if (timeOutMs > 0) {
365  qCDebug(KNEWSTUFF) << "starting the message timer for " << timeOutMs;
366  messageTimer->start(timeOutMs);
367  }
368 }
