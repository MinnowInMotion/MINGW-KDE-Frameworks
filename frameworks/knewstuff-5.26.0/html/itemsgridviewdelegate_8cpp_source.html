84 void ItemsGridViewDelegate::updateItemWidgets(const QList<QWidget *> widgets,
85  const QStyleOptionViewItem &option,
86  const QPersistentModelIndex &index) const
87 {
88  const ItemsModel *model = qobject_cast<const ItemsModel *>(index.model());
89  if (!model) {
90  qCDebug(KNEWSTUFF) << "WARNING - INVALID MODEL!";
91  return;
92  }
93 
94  EntryInternal entry = index.data(Qt::UserRole).value<KNS3::EntryInternal>();
95  int elementYPos = PreviewHeight + ItemMargin + FrameThickness * 2;
96 
97  //setup rating widget
98  KRatingWidget *rating = qobject_cast<KRatingWidget *>(widgets.at(DelegateGridRatingWidget));
99  if (rating) {
100  if (entry.rating() > 0) {
101  rating->setToolTip(i18n("Rating: %1%", entry.rating()));
102  // assume all entries come with rating 0..100 but most are in the range 20 - 80, so 20 is 0 stars, 80 is 5 stars
103  rating->setRating((entry.rating() - 20) * 10 / 60);
104  //make the rating widget smaller than the one at list view
105  int newWidth = 68;
106  QSize size(newWidth, 15);
107  rating->resize(size);
108  //put rating widget under image rectangle
109  rating->move((ItemGridWidth - newWidth) / 2, elementYPos);
110  elementYPos += rating->height();
111  } else {
112  //is it better to stay visible?
113  rating->setVisible(false);
114  }
115  }
116  elementYPos += ItemMargin;
117 
118  //setup title label
119  QLabel *titleLabel = qobject_cast<QLabel *>(widgets.at(DelegateTitleLabel));
120  if (titleLabel != NULL) {
121  titleLabel->setWordWrap(true);
122  titleLabel->setAlignment(Qt::AlignHCenter);
123  //titleLabel->setFrameStyle(QFrame::Panel);
124  titleLabel->resize(QSize(option.rect.width() - (ItemMargin * 2), option.fontMetrics.height() * 2));
125  titleLabel->move((ItemGridWidth - titleLabel->width()) / 2, elementYPos);
126 
127  QString title;
128  QUrl link = qvariant_cast<QUrl>(entry.homepage());
129  if (!link.isEmpty()) {
130  title += "<b><a href=\"" + link.url() + "\">" + entry.name() + "</a></b>\n";
131  } else {
132  title += "<b>" + entry.name() + "</b>";
133  }
134  titleLabel->setText(title);
135  elementYPos += titleLabel->height();
136  }
137  //setup author label
138  QLabel *authorLabel = qobject_cast<QLabel *>(widgets.at(DelegateAuthorLabel));
139  if (authorLabel != NULL) {
140  authorLabel->setWordWrap(true);
141  authorLabel->setAlignment(Qt::AlignHCenter);
142  authorLabel->resize(QSize(option.rect.width() - (ItemMargin * 2), option.fontMetrics.height()));
143  authorLabel->move((ItemGridWidth - authorLabel->width()) / 2, elementYPos);
144 
145  QString text;
146  QString authorName = entry.author().name();
147  QString email = entry.author().email();
148  QString authorPage = entry.author().homepage();
149 
150  if (!authorName.isEmpty()) {
151  if (!authorPage.isEmpty()) {
152  text += "<p>" + i18nc("Show the author of this item in a list", "By <i>%1</i>", " <a href=\"" + authorPage + "\">" + authorName + "</a>") + "</p>\n";
153  } else if (!email.isEmpty()) {
154  text += "<p>" + i18nc("Show the author of this item in a list", "By <i>%1</i>", authorName) + " <a href=\"mailto:" + email + "\">" + email + "</a></p>\n";
155  } else {
156  text += "<p>" + i18nc("Show the author of this item in a list", "By <i>%1</i>", authorName) + "</p>\n";
157  }
158  }
159  authorLabel->setText(text);
160  elementYPos += authorLabel->height();
161  }
162  elementYPos += ItemMargin;
163 
164  //setup download label
165  QLabel *downloadLabel = qobject_cast<QLabel *>(widgets.at(DelegateDownloadCounterLabel));
166  if (downloadLabel != NULL) {
167  downloadLabel->setWordWrap(true);
168  downloadLabel->setAlignment(Qt::AlignHCenter);
169  downloadLabel->resize(QSize(option.rect.width() - (ItemMargin * 2), option.fontMetrics.height()));
170  downloadLabel->move((ItemGridWidth - downloadLabel->width()) / 2, elementYPos);
171 
172  unsigned int fans = entry.numberFans();
173  unsigned int downloads = entry.downloadCount();
174 
175  QString text;
176  QString fanString;
177  QString downloadString;
178  if (fans > 0) {
179  fanString = i18ncp("fan as in supporter", "1 fan", "%1 fans", fans);
180  }
181  if (downloads > 0) {
182  downloadString = i18np("1 download", "%1 downloads", downloads);
183  }
184  if (downloads > 0 || fans > 0) {
185  text += "<p>" + downloadString;
186  if (downloads > 0 && fans > 0) {
187  text += QLatin1String(", ");
188  }
189  text += fanString + QLatin1String("</p>");
190  }
191  downloadLabel->setText(text);
192  elementYPos += downloadLabel->height();
193  }
194  elementYPos += ItemMargin;
195  m_elementYPos = elementYPos;
196 }

198 void ItemsGridViewDelegate::paint(QPainter *painter, const QStyleOptionViewItem &option, const QModelIndex &index) const
199 {
200  if (option.state & QStyle::State_MouseOver) {
201  QModelIndex focIndex = focusedIndex();
202  if (m_oldIndex != focIndex || !m_operationBar->isVisible()) {
203  ItemsGridViewDelegate *delegate = const_cast<ItemsGridViewDelegate *>(this);
204 
205  delegate->displayOperationBar(option.rect, index);
206  delegate->m_oldIndex = focIndex;
207  }
208  } else {
209  QModelIndex focindex = focusedIndex();
210  if (!focindex.isValid()) {
211  qCDebug(KNEWSTUFF) << "INVALID hide selection";
212  m_operationBar->hide();
213  }
214  }
215 
216  QStyle *style = QApplication::style();
217  style->drawPrimitive(QStyle::PE_PanelItemViewItem, &option, painter, 0);
218 
219  painter->save();
220 
221  if (option.state & QStyle::State_Selected) {
222  painter->setPen(QPen(option.palette.highlightedText().color()));
223  } else {
224  painter->setPen(QPen(option.palette.text().color()));
225  }
226 
227  const ItemsModel *realmodel = qobject_cast<const ItemsModel *>(index.model());
228 
229  if (realmodel->hasPreviewImages()) {
230  int width = option.rect.width();
231 
232  KNS3::EntryInternal entry = index.data(Qt::UserRole).value<KNS3::EntryInternal>();
233  if (entry.previewUrl(EntryInternal::PreviewSmall1).isEmpty()) {
234  ;
235  } else {
236  QPoint centralPoint(option.rect.left() + width / 2, option.rect.top() + ItemMargin + FrameThickness + PreviewHeight / 2);
237  QImage image = entry.previewImage(EntryInternal::PreviewSmall1);
238  if (!image.isNull()) {
239  QPoint previewPoint(centralPoint.x() - image.width() / 2, centralPoint.y() - image.height() / 2);
240  painter->drawImage(previewPoint, image);
241 
242  QPixmap frameImageScaled = m_frameImage.scaled(image.width() + FrameThickness * 2, image.height() + FrameThickness * 2);
243  QPoint framePoint(centralPoint.x() - frameImageScaled.width() / 2, centralPoint.y() - frameImageScaled.height() / 2);
244  painter->drawPixmap(framePoint, frameImageScaled);
245  } else {
246  QPoint thumbnailPoint(option.rect.left() + ((width - PreviewWidth - FrameThickness * 2) / 2), option.rect.top() + ItemMargin);
247  QRect rect(thumbnailPoint, QSize(PreviewWidth + FrameThickness * 2, PreviewHeight + FrameThickness * 2));
248  painter->drawText(rect, Qt::AlignCenter | Qt::TextWordWrap, i18n("Loading Preview"));
249  }
250  }
251  }
252 
253  painter->restore();
254 }

307 void ItemsGridViewDelegate::displayOperationBar(const QRect &rect, const QModelIndex &index)
308 {
309  KNS3::EntryInternal entry = index.data(Qt::UserRole).value<KNS3::EntryInternal>();
310  if (m_installButton != 0) {
311  if (m_installButton->menu() != 0) {
312  QMenu *buttonMenu = m_installButton->menu();
313  buttonMenu->clear();
314  m_installButton->setMenu(0);
315  buttonMenu->deleteLater();
316  }
317 
318  bool installable = false;
319  bool enabled = true;
320  QString text;
321  QIcon icon;
322 
323  switch (entry.status()) {
324  case Entry::Installed:
325  text = i18n("Uninstall");
326  icon = m_iconDelete;
327  break;
328  case Entry::Updateable:
329  text = i18n("Update");
330  icon = m_iconUpdate;
331  installable = true;
332  break;
333  case Entry::Installing:
334  text = i18n("Installing");
335  enabled = false;
336  icon = m_iconUpdate;
337  break;
338  case Entry::Updating:
339  text = i18n("Updating");
340  enabled = false;
341  icon = m_iconUpdate;
342  break;
343  case Entry::Downloadable:
344  text = i18n("Install");
345  icon = m_iconInstall;
346  installable = true;
347  break;
348  case Entry::Deleted:
349  text = i18n("Install Again");
350  icon = m_iconInstall;
351  installable = true;
352  break;
353  default:
354  text = i18n("Install");
355  }
356  m_installButton->setToolTip(text);
357  m_installButton->setIcon(icon);
358  m_installButton->setEnabled(enabled);
359  if (installable && entry.downloadLinkCount() > 1) {
360  QMenu *installMenu = new QMenu(m_installButton);
361  foreach (EntryInternal::DownloadLinkInformation info, entry.downloadLinkInformationList()) {
362  QString text = info.name;
363  if (!info.distributionType.trimmed().isEmpty()) {
364  text + " (" + info.distributionType.trimmed() + ')';
365  }
366  QAction *installAction = installMenu->addAction(m_iconInstall, text);
367  installAction->setData(QPoint(index.row(), info.id));
368  }
369  m_installButton->setMenu(installMenu);
370  }
371 
372  m_operationBar->move(rect.left() + (ItemGridWidth - m_operationBar->width()) / 2, rect.top() + m_elementYPos);
373  m_operationBar->show();
374  }
375 }
376 }
