49 QList<QWidget *> ItemsViewDelegate::createItemWidgets(const QModelIndex &index) const
50 {
51  Q_UNUSED(index);
52  QList<QWidget *> list;
53 
54  QLabel *infoLabel = new QLabel();
55  infoLabel->setOpenExternalLinks(true);
56  // not so nice - work around constness to install the event filter
57  ItemsViewDelegate *delegate = const_cast<ItemsViewDelegate *>(this);
58  infoLabel->installEventFilter(delegate);
59  list << infoLabel;
60 
61  QToolButton *installButton = new QToolButton();
62  installButton->setToolButtonStyle(Qt::ToolButtonTextBesideIcon);
63  installButton->setPopupMode(QToolButton::InstantPopup);
64  list << installButton;
65  setBlockedEventTypes(installButton, QList<QEvent::Type>() << QEvent::MouseButtonPress
66  << QEvent::MouseButtonRelease << QEvent::MouseButtonDblClick);
67  connect(installButton, &QAbstractButton::clicked, this, &ItemsViewDelegate::slotInstallClicked);
68  connect(installButton, &QToolButton::triggered, this, &ItemsViewDelegate::slotInstallActionTriggered);
69 
70  QToolButton *detailsButton = new QToolButton();
71  detailsButton->setToolButtonStyle(Qt::ToolButtonTextBesideIcon);
72  list << detailsButton;
73  setBlockedEventTypes(detailsButton, QList<QEvent::Type>() << QEvent::MouseButtonPress
74  << QEvent::MouseButtonRelease << QEvent::MouseButtonDblClick);
75  connect(detailsButton, SIGNAL(clicked()), this, SLOT(slotDetailsClicked()));
76 
77  KRatingWidget *rating = new KRatingWidget();
78  rating->setMaxRating(10);
79  rating->setHalfStepsEnabled(true);
80  list << rating;
81  //connect(rating, SIGNAL(ratingChanged(uint)), this, SLOT());
82 
83  return list;
84 }

86 void ItemsViewDelegate::updateItemWidgets(const QList<QWidget *> widgets,
87  const QStyleOptionViewItem &option,
88  const QPersistentModelIndex &index) const
89 {
90  const ItemsModel *model = qobject_cast<const ItemsModel *>(index.model());
91  if (!model) {
92  qCDebug(KNEWSTUFF) << "WARNING - INVALID MODEL!";
93  return;
94  }
95 
96  EntryInternal entry = index.data(Qt::UserRole).value<KNS3::EntryInternal>();
97 
98  // setup the install button
99  int margin = option.fontMetrics.height() / 2;
100  int right = option.rect.width();
101 
102  QToolButton *installButton = qobject_cast<QToolButton *>(widgets.at(DelegateInstallButton));
103  if (installButton != 0) {
104 
105  if (installButton->menu()) {
106  QMenu *buttonMenu = installButton->menu();
107  buttonMenu->clear();
108  installButton->setMenu(0);
109  buttonMenu->deleteLater();
110  }
111 
112  bool installable = false;
113  bool enabled = true;
114  QString text;
115  QIcon icon;
116 
117  switch (entry.status()) {
118  case Entry::Installed:
119  text = i18n("Uninstall");
120  icon = m_iconDelete;
121  break;
122  case Entry::Updateable:
123  text = i18n("Update");
124  icon = m_iconUpdate;
125  installable = true;
126  break;
127  case Entry::Installing:
128  text = i18n("Installing");
129  enabled = false;
130  icon = m_iconUpdate;
131  break;
132  case Entry::Updating:
133  text = i18n("Updating");
134  enabled = false;
135  icon = m_iconUpdate;
136  break;
137  case Entry::Downloadable:
138  text = i18n("Install");
139  icon = m_iconInstall;
140  installable = true;
141  break;
142  case Entry::Deleted:
143  text = i18n("Install Again");
144  icon = m_iconInstall;
145  installable = true;
146  break;
147  default:
148  text = i18n("Install");
149  }
150  installButton->setText(text);
151  installButton->setEnabled(enabled);
152  installButton->setIcon(icon);
153  if (installable && entry.downloadLinkCount() > 1) {
154  QMenu *installMenu = new QMenu(installButton);
155  foreach (const EntryInternal::DownloadLinkInformation &info, entry.downloadLinkInformationList()) {
156  QString text = info.name;
157  if (!info.distributionType.trimmed().isEmpty()) {
158  text + " (" + info.distributionType.trimmed() + ')';
159  }
160  QAction *installAction = installMenu->addAction(m_iconInstall, text);
161  installAction->setData(QPoint(index.row(), info.id));
162  }
163  installButton->setMenu(installMenu);
164  }
165  }
166 
167  QToolButton *detailsButton = qobject_cast<QToolButton *>(widgets.at(DelegateDetailsButton));
168  if (detailsButton) {
169  detailsButton->setText(i18n("Details"));
170  detailsButton->setIcon(QIcon::fromTheme(QStringLiteral("documentinfo")));
171  }
172 
173  if (installButton && detailsButton) {
174  if (m_buttonSize.width() < installButton->sizeHint().width()) {
175  const_cast<QSize &>(m_buttonSize) = QSize(
176  qMax(option.fontMetrics.height() * 7,
177  qMax(installButton->sizeHint().width(), detailsButton->sizeHint().width())),
178  installButton->sizeHint().height());
179  }
180  installButton->resize(m_buttonSize);
181  installButton->move(right - installButton->width() - margin, option.rect.height() / 2 - installButton->height() * 1.5);
182  detailsButton->resize(m_buttonSize);
183  detailsButton->move(right - installButton->width() - margin, option.rect.height() / 2 - installButton->height() / 2);
184  }
185 
186  QLabel *infoLabel = qobject_cast<QLabel *>(widgets.at(DelegateLabel));
187  infoLabel->setWordWrap(true);
188  if (infoLabel != NULL) {
189  if (model->hasPreviewImages()) {
190  // move the text right by kPreviewWidth + margin pixels to fit the preview
191  infoLabel->move(PreviewWidth + margin * 2, 0);
192  infoLabel->resize(QSize(option.rect.width() - PreviewWidth - (margin * 6) - m_buttonSize.width(), option.fontMetrics.height() * 7));
193 
194  } else {
195  infoLabel->move(margin, 0);
196  infoLabel->resize(QSize(option.rect.width() - (margin * 4) - m_buttonSize.width(), option.fontMetrics.height() * 7));
197  }
198 
199  QString text = QLatin1String("<!DOCTYPE HTML PUBLIC \"-//W3C//DTD HTML 4.0//EN\" \"http://www.w3.org/TR/REC-html40/strict.dtd\">\n"
200  "<html><head><meta name=\"qrichtext\" content=\"1\" /><style type=\"text/css\">p, li { white-space: pre-wrap; margin:0 0 0 0;}\n"
201  "</style></head><body><p><b>");
202 
203  QUrl link = qvariant_cast<QUrl>(entry.homepage());
204  if (!link.isEmpty()) {
205  text += "<p><a href=\"" + link.url() + "\">" + entry.name() + "</a></p>\n";
206  } else {
207  text += entry.name();
208  }
209 
210  text += QLatin1String("</b></p>\n");
211 
212  QString authorName = entry.author().name();
213  QString email = entry.author().email();
214  QString authorPage = entry.author().homepage();
215 
216  if (!authorName.isEmpty()) {
217  if (!authorPage.isEmpty()) {
218  text += "<p>" + i18nc("Show the author of this item in a list", "By <i>%1</i>", " <a href=\"" + authorPage + "\">" + authorName + "</a>") + "</p>\n";
219  } else if (!email.isEmpty()) {
220  text += "<p>" + i18nc("Show the author of this item in a list", "By <i>%1</i>", authorName) + " <a href=\"mailto:" + email + "\">" + email + "</a></p>\n";
221  } else {
222  text += "<p>" + i18nc("Show the author of this item in a list", "By <i>%1</i>", authorName) + "</p>\n";
223  }
224  }
225 
226  QString summary = "<p>" + option.fontMetrics.elidedText(entry.summary(),
227  Qt::ElideRight, infoLabel->width() * 3) + "</p>\n";
228  text += summary;
229 
230  unsigned int fans = entry.numberFans();
231  unsigned int downloads = entry.downloadCount();
232 
233  QString fanString;
234  QString downloadString;
235  if (fans > 0) {
236  fanString = i18ncp("fan as in supporter", "1 fan", "%1 fans", fans);
237  }
238  if (downloads > 0) {
239  downloadString = i18np("1 download", "%1 downloads", downloads);
240  }
241  if (downloads > 0 || fans > 0) {
242  text += "<p>" + downloadString;
243  if (downloads > 0 && fans > 0) {
244  text += QLatin1String(", ");
245  }
246  text += fanString + QLatin1String("</p>\n");
247  }
248 
249  text += QLatin1String("</body></html>");
250  // use simplified to get rid of newlines etc
251  text = replaceBBCode(text).simplified();
252  infoLabel->setText(text);
253  }
254 
255  KRatingWidget *rating = qobject_cast<KRatingWidget *>(widgets.at(DelegateRatingWidget));
256  if (rating) {
257  if (entry.rating() > 0) {
258  rating->setToolTip(i18n("Rating: %1%", entry.rating()));
259  // assume all entries come with rating 0..100 but most are in the range 20 - 80, so 20 is 0 stars, 80 is 5 stars
260  rating->setRating((entry.rating() - 20) * 10 / 60);
261  // put the rating label below the install button
262  rating->move(right - installButton->width() - margin, option.rect.height() / 2 + installButton->height() / 2);
263  rating->resize(m_buttonSize);
264  } else {
265  rating->setVisible(false);
266  }
267  }
268 }

271 void ItemsViewDelegate::paint(QPainter *painter, const QStyleOptionViewItem &option, const QModelIndex &index) const
272 {
273  int margin = option.fontMetrics.height() / 2;
274 
275  QStyle *style = QApplication::style();
276  style->drawPrimitive(QStyle::PE_PanelItemViewItem, &option, painter, 0);
277 
278  painter->save();
279 
280  if (option.state & QStyle::State_Selected) {
281  painter->setPen(QPen(option.palette.highlightedText().color()));
282  } else {
283  painter->setPen(QPen(option.palette.text().color()));
284  }
285 
286  const ItemsModel *realmodel = qobject_cast<const ItemsModel *>(index.model());
287 
288  if (realmodel->hasPreviewImages()) {
289  int height = option.rect.height();
290  QPoint point(option.rect.left() + margin, option.rect.top() + ((height - PreviewHeight) / 2));
291 
292  KNS3::EntryInternal entry = index.data(Qt::UserRole).value<KNS3::EntryInternal>();
293  if (entry.previewUrl(EntryInternal::PreviewSmall1).isEmpty()) {
294  // paint the no preview icon
295  //point.setX((PreviewWidth - m_noImage.width())/2 + 5);
296  //point.setY(option.rect.top() + ((height - m_noImage.height()) / 2));
297  //painter->drawPixmap(point, m_noImage);
298  } else {
299  QImage image = entry.previewImage(EntryInternal::PreviewSmall1);
300  if (!image.isNull()) {
301  point.setX((PreviewWidth - image.width()) / 2 + 5);
302  point.setY(option.rect.top() + ((height - image.height()) / 2));
303  painter->drawImage(point, image);
304 
305  QPoint framePoint(point.x() - 5, point.y() - 5);
306  painter->drawPixmap(framePoint, m_frameImage.scaled(image.width() + 10, image.height() + 10));
307  } else {
308  QRect rect(point, QSize(PreviewWidth, PreviewHeight));
309  painter->drawText(rect, Qt::AlignCenter | Qt::TextWordWrap, i18n("Loading Preview"));
310  }
311  }
312 
313  }
314  painter->restore();
315 }
