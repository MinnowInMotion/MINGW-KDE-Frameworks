82 static void addItemFromKmtService(KMoreToolsMenuBuilder* menuBuilder,
83  QMenu* menu,
84  KMoreToolsService* kmtService,
85  const QUrl& url,
86  bool isMoreSection
87  )
88 {
89  auto menuItem = menuBuilder->addMenuItem(kmtService, isMoreSection ?
90  KMoreTools::MenuSection_More : KMoreTools::MenuSection_Main);
91 
92  if (kmtService->isInstalled()) {
93  auto kService = kmtService->installedService();
94 
95  if (!kService) {
96  // if the corresponding desktop file is not installed
97  // then the isInstalled was true because of the Exec line check
98  // and we use the desktopfile provided by KMoreTools.
99  // Otherwise *kService would crash.
100  qDebug() << "Desktop file not installed:" << kmtService->desktopEntryName() << "=> Use desktop file provided by KMoreTools";
101  kService = kmtService->kmtProvidedService();
102  }
103 
104  if (!url.isEmpty() && kmtService->maxUrlArgCount() > 0) {
105  menu->connect(menuItem->action(), &QAction::triggered, menu,
106  [kService, url](bool) {
107  KRun::runService(*kService, { url }, nullptr);
108  });
109  } else {
110  menu->connect(menuItem->action(), &QAction::triggered, menu,
111  [kService](bool) {
112  KRun::runService(*kService, { }, nullptr);
113  });
114  }
115  }
116 }

119 static void addItemsFromKmtServiceList(KMoreToolsMenuBuilder* menuBuilder,
120  QMenu* menu,
121  QList<KMoreToolsService*> kmtServiceList,
122  const QUrl& url,
123  bool isMoreSection,
124  QString firstMoreSectionDesktopEntryName
125  )
126 {
127  Q_FOREACH(auto kmtService, kmtServiceList) {
128  // Check the pointer just in case a null pointer got in somewhere
129  if (!kmtService) continue;
130  if (kmtService->desktopEntryName() == firstMoreSectionDesktopEntryName) {
131  // once we reach the potential first "more section desktop entry name"
132  // all remaining services are added to the more section by default
133  isMoreSection = true;
134  }
135  addItemFromKmtService(menuBuilder, menu, kmtService, url, isMoreSection);
136  }
137 }

145 static void addItemsForGroupingNameWithSpecialHandling(KMoreToolsMenuBuilder* menuBuilder,
146  QMenu* menu,
147  QList<KMoreToolsService*> kmtServiceList,
148  const QString& groupingName,
149  const QUrl& url,
150  bool isMoreSection,
151  QString firstMoreSectionDesktopEntryName
152  )
153 {
154  //
155  // special handlings
156  //
157  if (groupingName == QLatin1String("disk-usage") && !url.isEmpty()) {
158  //
159  // "disk-usage" plus a given URL. If no url is given there is no need
160  // for special handling
161  //
162 
163  auto filelightAppIter = std::find_if(kmtServiceList.begin(),
164  kmtServiceList.end(),
165  [](KMoreToolsService* s) {
166  return s->desktopEntryName() == QLatin1String("org.kde.filelight");
167  });
168 
169  if (filelightAppIter != kmtServiceList.end()) {
170  auto filelightApp = *filelightAppIter;
171 
172  // because we later add all remaining items
173  kmtServiceList.removeOne(filelightApp);
174 
175  if (url.isLocalFile()) { // 2015-01-12: Filelight can handle FTP connections
176  // but KIO/kioexec cannot (bug or feature?), so we
177  // don't offer it in this case
178 
179  const auto filelight1Item = menuBuilder->addMenuItem(filelightApp);
180 
181  if (filelightApp->isInstalled()) {
182  const auto filelightService = filelightApp->installedService();
183 
184  filelight1Item->action()->setText(filelightApp->formatString(
185  i18nc("@action:inmenu", "$GenericName - current folder")));
186  menu->connect(filelight1Item->action(), &QAction::triggered, menu,
187  [filelightService, url](bool) {
188  KRun::runService(*filelightService, { url }, nullptr);
189  });
190 
191  const auto filelight2Item = menuBuilder->addMenuItem(filelightApp);
192  filelight2Item->action()->setText(filelightApp->formatString(
193  i18nc("@action:inmenu", "$GenericName - current device")));
194  menu->connect(filelight2Item->action(), &QAction::triggered, menu,
195  [filelightService, url](bool) {
196  KMountPoint::Ptr mountPoint
197  = KMountPoint::currentMountPoints().findByPath(url.toLocalFile());
198  KRun::runService(*filelightService,
199  { QUrl::fromLocalFile(mountPoint->mountPoint()) },
200  nullptr);
201  });
202  }
203  }
204 
205  auto filelight3Item = menuBuilder->addMenuItem(filelightApp, KMoreTools::MenuSection_More);
206  if (filelightApp->isInstalled()) {
207  filelight3Item->action()->setText(filelightApp->formatString(
208  i18nc("@action:inmenu", "$GenericName - all devices")));
209  const auto filelightService = filelightApp->installedService();
210  menu->connect(filelight3Item->action(), &QAction::triggered, menu,
211  [filelightService](bool) {
212  KRun::runService(*filelightService, { }, nullptr);
213  });
214  }
215  } else {
216  qWarning() << "org.kde.filelight should be present in KMoreTools but it is not!";
217  }
218 
219  } else if (groupingName == QLatin1String("disk-partitions")) {
220  // better because the Partition editors all have the same GenericName
221  menuBuilder->setInitialItemTextTemplate(QStringLiteral("$GenericName ($Name)"));
222 
223  addItemsFromKmtServiceList(menuBuilder, menu, kmtServiceList, url, isMoreSection, firstMoreSectionDesktopEntryName);
224 
225  menuBuilder->setInitialItemTextTemplate(QStringLiteral("$GenericName")); // set back to default
226 
227  return; // skip processing remaining list (would result in duplicates)
228 
229  } else if (groupingName == QLatin1String("git-clients-and-actions")) {
230  // Here we change the default item text and make sure that the url
231  // argument is properly handled.
232  //
233 
234  menuBuilder->setInitialItemTextTemplate(QStringLiteral("$Name")); // just use the application name
235 
236  Q_FOREACH(auto kmtService, kmtServiceList) {
237  // Check the pointer just in case a null pointer got in somewhere
238  if (!kmtService) continue;
239  QUrl argUrl = url;
240 
241  if (url.isLocalFile()) { // this can only be done for local files, remote urls probably won't work for git clients anyway
242  // by default we need an URL pointing to a directory
243  // (this impl currently leads to wrong behaviour if the root dir of a git repo is chosen because it always goes one level up)
244  argUrl = KmtUrlUtil::localFileAbsoluteDir(url); // needs local file
245 
246  if (kmtService->desktopEntryName() == _("git-cola-view-history.kmt-edition")) {
247  // in this case we need the file because we would like to see its history
248  argUrl = url;
249  }
250  }
251 
252  addItemFromKmtService(menuBuilder, menu, kmtService, argUrl, isMoreSection);
253  }
254 
255  menuBuilder->setInitialItemTextTemplate(QStringLiteral("$GenericName")); // set back to default
256 
257  return; // skip processing remaining list (would result in duplicates)
258  }
259 
260  //
261  // default handling (or process remaining list)
262  //
263  menuBuilder->setInitialItemTextTemplate(QStringLiteral("$Name")); // just use the application name
264  addItemsFromKmtServiceList(menuBuilder, menu, kmtServiceList, url, isMoreSection, firstMoreSectionDesktopEntryName);
265  menuBuilder->setInitialItemTextTemplate(QStringLiteral("$GenericName")); // set back to default
266 }

283 void KMoreToolsMenuFactory::fillMenuFromGroupingNames(QMenu* menu, const QStringList& groupingNames, const QUrl& url)
284 {
285  const auto menuBuilder = d->kmt->menuBuilder();
286  menuBuilder->clear();
287 
288  bool isMoreSection = false;
289 
290  Q_FOREACH(const auto &groupingName, groupingNames) {
291 
292  if (groupingName == QLatin1String("more:")) {
293  isMoreSection = true;
294  continue;
295  }
296 
297  QString firstMoreSectionDesktopEntryName;
298  auto kmtServiceList = KMoreToolsPresetsPrivate::registerServicesByGroupingNames(
299  &firstMoreSectionDesktopEntryName, d->kmt, { groupingName });
300 
301  addItemsForGroupingNameWithSpecialHandling(menuBuilder,
302  menu,
303  kmtServiceList,
304  groupingName,
305  url,
306  isMoreSection,
307  firstMoreSectionDesktopEntryName);
308  }
309 
310  menuBuilder->buildByAppendingToMenu(menu);
311 }
