214 void UploadDialogPrivate::_k_updatePage()
215 {
216  bool firstPage = ui.stackedWidget->currentIndex() == 0;

219  bool nextEnabled = false;

223  nextEnabled = true;
224  }
225  break;
226 
227  case FileNewUpdatePage:
228  // FIXME: check if the file requester contains a valid file
229  if (!uploadFile.isEmpty() || ui.uploadFileRequester->url().isLocalFile()) {
230  if (ui.radioNewUpload->isChecked() || ui.userContentList->currentRow() >= 0) {
231  nextEnabled = true;
232  }
233  }
234  break;
235 
236  case Details1Page:
237  if (!ui.mNameEdit->text().isEmpty()) {
238  nextEnabled = true;
239  }
240  break;
241 
242  case Details2Page:
243  nextEnabled = true;
244  break;
245 
246  case UploadFinalPage:
247  break;
248  }
249 
250  nextButton->setEnabled(nextEnabled);
251  finishButton->setEnabled(finished);
252 
253  nextButton->setDefault(nextEnabled);
254  finishButton->setDefault(!nextEnabled);
255

585 void UploadDialogPrivate::_k_startUpload()
586 {
587  // FIXME: this only works if categories are set in the .knsrc file
588  // TODO: ask for confirmation when closing the dialog
589 
590  backButton->setEnabled(false);
591  buttonBox->button(QDialogButtonBox::Cancel)->setEnabled(false);
592 
593  ui.uploadButton->setEnabled(false);
594 
595  // idle back and forth, we need a fix in attica to get at real progress values
596  ui.uploadProgressBar->setMinimum(0);
597  ui.uploadProgressBar->setMaximum(0);
598  ui.uploadProgressBar->setValue(0);
599 
600  // check the category
601  QString categoryName = ui.mCategoryCombo->currentText();
602  QList<Attica::Category>::const_iterator iter = categories.constBegin();
603  Attica::Category category;
604  QList<Attica::Category>::const_iterator iterEnd = categories.constEnd();
605  while (iter != iterEnd) {
606  if (iter->name() == categoryName) {
607  category = *iter;
608  break;
609  }
610  ++iter;
611  }
612  if (!category.isValid()) {
613  KMessageBox::error(q, i18n("The selected category \"%1\" is invalid.", categoryName), i18n("Upload Failed"));
614  return;
615  }
616 
617  // fill in the content object
618  Attica::Content content;
619  content.setName(ui.mNameEdit->text());
620  QString summary = ui.mSummaryEdit->toPlainText();
621  content.addAttribute(QStringLiteral("description"), summary);
622  content.addAttribute(QStringLiteral("version"), ui.mVersionEdit->text());
623 
624  // for the license, if one of the licenses coming from the server was used, pass its id, otherwise the string
625  QString licenseId = ui.mLicenseCombo->itemData(ui.mLicenseCombo->currentIndex()).toString();
626  if (licenseId.isEmpty()) {

702 void UploadDialogPrivate::_k_contentAdded(Attica::BaseJob *baseJob)
703 {
704  if (baseJob->metadata().error()) {
705  if (baseJob->metadata().error() == Attica::Metadata::NetworkError) {
706  KMessageBox::error(q, i18n("There was a network error."), i18n("Uploading Failed"));
707  return;
708  }
709  if (baseJob->metadata().error() == Attica::Metadata::OcsError) {
710  if (baseJob->metadata().statusCode() == 102) {
711  KMessageBox::error(q, i18n("Authentication error."), i18n("Uploading Failed"));
712  }
713  }
714  return;
715  }
716 
717  ui.createContentImageLabel->setPixmap(QIcon::fromTheme(QStringLiteral("dialog-ok")).pixmap(16));
718 
719  Attica::ItemPostJob<Attica::Content> *job = static_cast<Attica::ItemPostJob<Attica::Content> *>(baseJob);
720  if (job->metadata().error() != Attica::Metadata::NoError) {
721  KMessageBox::error(q, i18n("Upload failed: %1", job->metadata().message()));
722  return;
723  }
724 
725  // only when adding new content we get an id returned, otherwise stick with the old one
726  QString id = job->result().id();
727  if (!id.isEmpty()) {
728  contentId = id;
729  }
730 
731  if (!uploadFile.isEmpty()) {
732  doUpload(QString(), uploadFile);
733  } else {
734  doUpload(QString(), ui.uploadFileRequester->url());
735  }
736 
737  // FIXME: status labels need to accommodate 3 previews
738  if (!previewFile1.isEmpty()) {
739  doUpload(QStringLiteral("1"), previewFile1);
740  }
741  if (!previewFile2.isEmpty()) {
742  doUpload(QStringLiteral("2"), previewFile2);
743  }
744  if (!previewFile3.isEmpty()) {
745  doUpload(QStringLiteral("3"), previewFile3);
746  }
747 
748  if (ui.radioNewUpload->isChecked()) {
749  atticaHelper->loadDetailsLink(contentId);
750  }
751 }

758 void UploadDialogPrivate::doUpload(const QString &index, const QUrl &path)
759 {
760  QFile file(path.toLocalFile());
761  if (!file.open(QIODevice::ReadOnly)) {
762  KMessageBox::error(q, i18n("File not found: %1", uploadFile.url()), i18n("Upload Failed"));
763  q->reject();
764  return;
765  }
766 
767  QByteArray fileContents;
768  fileContents.append(file.readAll());
769  file.close();
770 
771  QString fileName = QFileInfo(path.toLocalFile()).fileName();
772 
773  Attica::PostJob *job = 0;
774  if (index.isEmpty()) {
775  job = currentProvider().setDownloadFile(contentId, fileName, fileContents);
776  q->connect(job, SIGNAL(finished(Attica::BaseJob*)), q, SLOT(_k_fileUploadFinished(Attica::BaseJob*)));
777  } else if (index == QLatin1String("1")) {
778  job = currentProvider().setPreviewImage(contentId, index, fileName, fileContents);
779  q->connect(job, SIGNAL(finished(Attica::BaseJob*)), q, SLOT(_k_preview1UploadFinished(Attica::BaseJob*)));
780  } else if (index == QLatin1String("2")) {
781  job = currentProvider().setPreviewImage(contentId, index, fileName, fileContents);
782  q->connect(job, SIGNAL(finished(Attica::BaseJob*)), q, SLOT(_k_preview2UploadFinished(Attica::BaseJob*)));
783  } else if (index == QLatin1String("3")) {
784  job = currentProvider().setPreviewImage(contentId, index, fileName, fileContents);
785  q->connect(job, SIGNAL(finished(Attica::BaseJob*)), q, SLOT(_k_preview3UploadFinished(Attica::BaseJob*)));
786  }
787  if (job) {
788  job->start();
789  }
790 }
