64 def expand_platform_all(dct, available_platforms):
65  """If one of the keys of dct is `PLATFORM_ALL` (or `PLATFORM_UNKNOWN`),
66  remove it and add entries for all available platforms to dct
67 
68  Args:
69  dct: (dictionary) dictionary to expand
70  available_platforms: (list of string) name of platforms
71  """
72 
73  add_all_platforms = False
74  if PLATFORM_ALL in dct:
75  note = dct[PLATFORM_ALL]
76  add_all_platforms = True
77  del dct[PLATFORM_ALL]
78  if PLATFORM_UNKNOWN in dct:
79  add_all_platforms = True
80  note = dct[PLATFORM_UNKNOWN]
81  del dct[PLATFORM_UNKNOWN]
82  if add_all_platforms:
83  for platform in available_platforms:
84  if platform not in dct:
85  dct[platform] = note

88 def create_metainfo(path):
89  """Look for a `metadata.yaml` file and create a dictionary out it.
90 
91  Args:
92  path: (string) the current path to search.
93  Returns:
94  A dictionary containing all the parsed information, or `None` if it
95  did not fullfill some conditions.
96  """
97 
98  if not os.path.isdir(path):
99  return None
100 
101  metainfo_file = os.path.join(path, 'metainfo.yaml')
102  if not os.path.isfile(metainfo_file):
103  return None
104 
105  try:
106  metainfo = yaml.load(open(metainfo_file))
107  except Exception as e:
108  print(e)
109  logging.warning('Could not load metainfo.yaml for {}, skipping it'
110  .format(path))
111  return None
112 
113  if metainfo is None:
114  logging.warning('Empty metainfo.yaml for {}, skipping it'
115  .format(path))
116  return None
117 
118  if 'subgroup' in metainfo and 'group' not in metainfo:
119  logging.warning('Subgroup but no group in {}, skipping it'
120  .format(path))
121  return None
122 
123  name = os.path.basename(path)
124  if 'fancyname' in metainfo:
125  fancyname = metainfo['fancyname']
126  else:
127  fancyname = utils.parse_fancyname(path)
128 
129  if not fancyname:
130  logging.warning('Could not find fancy name for {}, skipping it'
131  .format(path))
132  return None
133  # A fancyname has 1st char capitalized
134  fancyname = fancyname[0].capitalize() + fancyname[1:]
135  metainfo.update({
136  'fancyname': fancyname,
137  'name': name,
138  'public_lib': metainfo.get('public_lib', False),
139  'dependency_diagram': None,
140  'path': path,
141  })
142 
143  return metainfo

146 def parse_tree(rootdir):
147  """Recursively call create_metainfo() in subdirs of rootdir
148 
149  Args:
150  rootdir: (string) Top level directory containing the libraries.
151 
152  Returns:
153  A list of metainfo dictionary (see create_metainfo()).
154 
155  """
156  metalist = []
157  for path, dirs, _ in os.walk(rootdir):
158  # We don't want to do the recursion in the dotdirs
159  dirs[:] = [d for d in dirs if not d[0] == '.']
160  metainfo = create_metainfo(path)
161  if metainfo is not None:
162  if metainfo['public_lib'] or 'group_info' in metainfo:
163  metalist.append(metainfo)
164  else:
165  logging.warning("{} has no public libraries"
166  .format(metainfo['name']))
167 
168  return metalist

171 def sort_metainfo(metalist, all_maintainers):
172  """Extract the structure (Product/Subproduct/Library) from the metainfo
173  list.
174 
175  Args:
176  metalist: (list of dict) lists of the metainfo extracted in
177  parse_tree().
178  all_maintainers: (dict of dict) all possible maintainers.
179 
180  Returns:
181  A list of Products, a list of groups (which are products containing
182  several libraries), a list of Libraries and the available platforms.
183  """
184  products = dict()
185  groups = []
186  libraries = []
187  available_platforms = set(['Windows', 'MacOSX', 'Linux', 'Android', 'FreeBSD'])
188 
189  # First extract the structural info
190  for metainfo in metalist:
191  product = extract_product(metainfo, all_maintainers)
192  if product is not None:
193  products[product.name] = product
194 
195  # Second extract the libraries
196  for metainfo in metalist:
197  try:
198  platforms = metainfo['platforms']
199  platform_lst = [x['name'] for x in platforms
200  if x['name'] not in (PLATFORM_ALL,
201  PLATFORM_UNKNOWN)]
202 
203  available_platforms.update(set(platform_lst))
204  except (KeyError, TypeError):
205  logging.warning('{} library lacks valid platform definitions'
206  .format(metainfo['fancyname']))
207  platforms = [dict(name=PLATFORM_UNKNOWN)]
208 
209  dct = dict((x['name'], x.get('note', '')) for x in platforms)
210 
211  expand_platform_all(dct, available_platforms)
212  platforms = dct
213 
214  if metainfo['public_lib']:
215  lib = Library(metainfo, products, platforms, all_maintainers)
216  libraries.append(lib)
217 
218  groups = []
219  for key in products.copy():
220  if len(products[key].libraries) == 0:
221  del products[key]
222  elif products[key].part_of_group:
223  groups.append(products[key])
224 
225  return list(products.values()), groups, libraries, available_platforms

228 def extract_product(metainfo, all_maintainers):
229  """Extract a product from a metainfo dictionary.
230 
231  Args:
232  metainfo: (dict) metainfo created by the create_metainfo() function.
233  all_maintainer: (dict of dict) all possible maintainers
234 
235  Returns:
236  A Product or None if the metainfo does not describe a product.
237  """
238 
239  if 'group_info' not in metainfo and 'group' in metainfo:
240  # This is not a product but a simple lib
241  return None
242 
243  try:
244  product = Product(metainfo, all_maintainers)
245  return product
246  except ValueError as e:
247  logging.error(e)
248  return None
