25 class Framework(object):
26  """
27  A framework has a tier, a name, a collection of targets it provides.
28  For each target it provides, it knows which targets they depend on.
29 
30  It also has a list of frameworks it depends on, for framework-to-framework
31  dependencies. This is useful for cases when a framework depends on macros
32  provided by another framework.
33  """
34  def __init__(self, tier, name):
35  self.tier = tier
36  self.name = name
37 
38  # A dict of target => set([targets])
39  self._target_dict = {}
40 
41  # A list of other frameworks (not targets!) this framework depends on.
42  self._fw_list = []
43 
44  def add_target(self, target):
45  # Add a new target provided by this framework
46  assert target not in self._target_dict
47  self._target_dict[target] = set()
48 
49  def add_target_dependency(self, target, dep):
50  # Add a new dependency target to a target provided by this framework
51  assert target in self._target_dict
52  self._target_dict[target].add(dep)
53 
54  def get_targets(self):
55  return self._target_dict.keys()
56 
57  def has_target(self, target):
58  return target in self._target_dict
59 
60  def get_all_target_dependencies(self):
61  deps = self._target_dict.values()
62  if not deps:
63  return set()
64  return reduce(lambda x,y: x.union(y), deps)
65 
66  def get_dependencies_for_target(self, target):
67  return self._target_dict[target]
68 
69  def add_extra_framework(self, name):
70  # Add a framework this framework depends on because of a
71  # framework-to-framework dependency
72  self._fw_list.append(name)
73 
74  def get_extra_frameworks(self):
75  # Returns frameworks this framework depends on because of a
76  # framework-to-framework dependency
77  return self._fw_list
78 
79  def __repr__(self):
80  return self.name
