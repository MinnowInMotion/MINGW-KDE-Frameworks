72 def set_maintainers(maintainer_keys, all_maintainers):
73  """ Expend the name of the maintainers.
74 
75  Args:
76  dictonary: (dict) Dictionary from which the name to expend will be read.
77  key: (string) Key of the dictionary where the name to expend is saved.
78  all_maintainers: (dict of dict) Look-up table where the names and emails of
79  the maintainers are stored.
80 
81  Examples:
82  >>> maintainer_keys = ['arthur', 'toto']
83 
84  >>> myteam = {'arthur': {'name': 'Arthur Pendragon',
85  'email': 'arthur@example.com'},
86  'toto': {'name': 'Toto',
87  'email: 'toto123@example.com'}
88  }
89 
90  >>> set_maintainers(maintainer_keys, my_team)
91  """
92 
93  if not maintainer_keys:
94  maintainers = []
95  elif isinstance(maintainer_keys, list):
96  maintainers = map(lambda x: all_maintainers.get(x, None),
97  maintainer_keys)
98  else:
99  maintainers = [all_maintainers.get(maintainer_keys, None)]
100 
101  maintainers = [x for x in maintainers if x is not None]
102  return maintainers

105 def parse_fancyname(fw_dir):
106  """Return the framework name for a given source dir
107 
108  The framework name is the name of the toplevel CMake project
109  """
110  cmakelists_path = os.path.join(fw_dir, "CMakeLists.txt")
111  if not os.path.exists(cmakelists_path):
112  logging.error("No CMakeLists.txt in {}".format(fw_dir))
113  return None
114  project_re = re.compile(r"project\s*\(\s*([\w\-\_]+)", re.I)
115  with open(cmakelists_path) as f:
116  for line in f.readlines():
117  match = project_re.search(line)
118  if match:
119  return match.group(1)
120  logging.error("Failed to find framework name: Could not find a "
121  "'project()' command in {}.".format(cmakelists_path))
122  return None

125 def cache_dir():
126  """Find/create a semi-long-term cache directory.
127 
128  We do not use tempdir, except as a fallback, because temporary directories
129  are intended for files that only last for the program's execution.
130  """
131  cachedir = None
132  if sys.platform == 'darwin':
133  try:
134  from AppKit import NSSearchPathForDirectoriesInDomains
135  # http://developer.apple.com/DOCUMENTATION/Cocoa/Reference/Foundation/Miscellaneous/Foundation_Functions/Reference/reference.html#//apple_ref/c/func/NSSearchPathForDirectoriesInDomains
136  # NSApplicationSupportDirectory = 14
137  # NSUserDomainMask = 1
138  # True for expanding the tilde into a fully qualified path
139  cachedir = os.path.join(
140  NSSearchPathForDirectoriesInDomains(14, 1, True)[0],
141  'KApiDox')
142  except:
143  pass
144  elif os.name == "posix":
145  if 'HOME' in os.environ and os.path.exists(os.environ['HOME']):
146  cachedir = os.path.join(os.environ['HOME'], '.cache', 'kapidox')
147  elif os.name == "nt":
148  if 'APPDATA' in os.environ and os.path.exists(os.environ['APPDATA']):
149  cachedir = os.path.join(os.environ['APPDATA'], 'KApiDox')
150  if cachedir is None:
151  cachedir = os.path.join(tempfile.gettempdir(), 'kapidox')
152  if not os.path.isdir(cachedir):
153  os.makedirs(cachedir)
154  return cachedir

157 def svn_export(remote, local, overwrite=False):
158  """Wraps svn export.
159 
160  Args:
161  remote: (string) the remote url.
162  local: (string) the local path where to dowload.
163  overwrite: (bool) whether to overwrite `local` or not. (optional,
164  default = False)
165 
166  Returns:
167  True if success.
168 
169  Raises:
170  FileNotFoundError: &nbsp;
171  subprocess.CalledProcessError: &nbsp;
172  """
173  try:
174  import svn.core
175  import svn.client
176  logging.debug("Using Python libsvn bindings to fetch %s", remote)
177  ctx = svn.client.create_context()
178  ctx.auth_baton = svn.core.svn_auth_open([])
179 
180  latest = svn.core.svn_opt_revision_t()
181  latest.type = svn.core.svn_opt_revision_head
182 
183  svn.client.export(remote, local, latest, True, ctx)
184  except ImportError:
185  logging.debug("Using external svn client to fetch %s", remote)
186  cmd = ['svn', 'export', '--quiet']
187  if overwrite:
188  cmd.append('--force')
189  cmd += [remote, local]
190  try:
191  subprocess.check_call(cmd, stderr=subprocess.STDOUT)
192  except subprocess.CalledProcessError as e:
193  raise subprocess.StandardException(e.output)
194  except FileNotFoundError as e:
195  logging.debug("External svn client not found")
196  return False
197  # subversion will set the timestamp to match the server
198  os.utime(local, None)
199  return True

202 def copy_dir_contents(directory, dest):
203  """Copy the contents of a directory
204 
205  Args:
206  directory: (string) the directory to copy the contents of.
207  dest: (string) the directory to copy them into.
208  """
209  ignored = ['CMakeLists.txt']
210  ignore = shutil.ignore_patterns(*ignored)
211  for fn in os.listdir(directory):
212  f = os.path.join(directory, fn)
213  if os.path.isfile(f):
214  docopy = True
215  for i in ignored:
216  if fnmatch(fn, i):
217  docopy = False
218  break
219  if docopy:
220  shutil.copy(f, dest)
221  elif os.path.isdir(f):
222  dest_f = os.path.join(dest, fn)
223  if os.path.isdir(dest_f):
224  shutil.rmtree(dest_f)
225  shutil.copytree(f, dest_f, ignore=ignore)

229 def get_kapidox_version():
230  """Get commit id of running code if it is running from git repository.
231 
232  May return an empty string if it failed to extract the commit id.
233 
234  Assumes .git/HEAD looks like this:
235 
236  ref: refs/heads/master
237 
238  and assumes .git/refs/heads/master contains the commit id
239  """
240  global _KAPIDOX_VERSION
241 
242  if _KAPIDOX_VERSION is not None:
243  return _KAPIDOX_VERSION
244 
245  _KAPIDOX_VERSION = ""
246  bin_dir = os.path.dirname(sys.argv[0])
247  git_dir = os.path.join(bin_dir, "..", ".git")
248  if not os.path.isdir(git_dir):
249  # Looks like we are not running from the git repo, exit silently
250  return _KAPIDOX_VERSION
251 
252  git_HEAD = os.path.join(git_dir, "HEAD")
253  if not os.path.isfile(git_HEAD):
254  logging.warning("Getting git info failed: {} is not a file".format(git_HEAD))
255  return _KAPIDOX_VERSION
256 
257  try:
258  line = open(git_HEAD).readline()
259  ref_name = line.split(": ")[1].strip()
260  with open(os.path.join(git_dir, ref_name)) as f:
261  _KAPIDOX_VERSION = f.read().strip()
262  except Exception as exc:
263  # Catch all exceptions here: whatever fails in this function should not
264  # cause the code to fail
265  logging.warning("Getting git info failed: {}".format(exc))
266  return _KAPIDOX_VERSION

269 def find_dot_files(dot_dir):

271  lst = []
272  for (root, dirs, files) in os.walk(dot_dir):
