55 def preprocess(fname):
56  graph_handle = gv.read(fname)
57  txt = open(fname).read()
58  targets = []
59 
60  # Replace the generated node names with their label. CMake generates a graph
61  # like this:
62  #
63  # "node0" [ label="KF5DNSSD" shape="polygon"];
64  # "node1" [ label="Qt5::Network" shape="ellipse"];
65  # "node0" -> "node1"
66  #
67  # And we turn it into this:
68  #
69  # "KF5DNSSD" [ label="KF5DNSSD" shape="polygon"];
70  # "Qt5::Network" [ label="Qt5::Network" shape="ellipse"];
71  # "KF5DNSSD" -> "Qt5::Network"
72  #
73  # Using real framework names as labels makes it possible to merge multiple
74  # .dot files.
75  for node_handle in gvutils.get_node_list(graph_handle):
76  node = gvutils.Node(node_handle)
77  label = node.label.replace("KF5::", "")
78  if node.shape in TARGET_SHAPES:
79  targets.append(label)
80  txt = txt.replace('"' + node.name + '"', '"' + label + '"')
81 
82  # Sometimes cmake will generate an entry for the target alias, something
83  # like this:
84  #
85  # "node9" [ label="KParts" shape="polygon"];
86  # ...
87  # "node15" [ label="KF5::KParts" shape="ellipse"];
88  # ...
89  #
90  # After our node renaming, this ends up with a second "KParts" node
91  # definition, which we need to get rid of.
92  for target in targets:
93  rx = r' *"' + target + '".*label="KF5::' + target + '".*shape="ellipse".*;'
94  txt = re.sub(rx, '', txt)
95  return txt

106 class DotFileParser(object):
107  def __init__(self, with_qt):
108  self._with_qt = with_qt
109 
110  def parse(self, tier, dot_file):
111  name = os.path.basename(dot_file).replace(".dot", "")
112  fw = Framework(tier, name)
113 
114  # Preprocess dot files so that they can be merged together.
115  dot_data = preprocess(dot_file)
116  self._init_fw_from_dot_data(fw, dot_data, self._with_qt)
117 
118  return fw
119 
120  def _init_fw_from_dot_data(self, fw, dot_data, with_qt):
121  def target_from_node(node):
122  return node.name.replace("KF5", "")
123 
124  src_handle = gv.readstring(dot_data)
125 
126  targets = set()
127  for node_handle in gvutils.get_node_list(src_handle):
128  node = gvutils.Node(node_handle)
129  if node.shape in TARGET_SHAPES and self._want(node):
130  target = target_from_node(node)
131  targets.add(target)
132  fw.add_target(target)
133 
134  for edge_handle in gvutils.get_edge_list(src_handle):
135  edge = gvutils.Edge(edge_handle)
136  target = target_from_node(edge.tail)
137  if target in targets and self._want(edge.head):
138  dep_target = target_from_node(edge.head)
139  fw.add_target_dependency(target, dep_target)
140 
141  def _want(self, node):
142  if node.shape not in TARGET_SHAPES and node.shape != DEPS_SHAPE:
143  return False
144  name = node.name
145 
146  for pattern in DEPS_BLACKLIST:
147  if fnmatch.fnmatchcase(node.name, pattern):
148  return False
149  if not self._with_qt and name.startswith("Qt"):
150  return False
151  return True

154 class FrameworkDb(object):
155  def __init__(self):
156  self._fw_list = []
157  self._fw_for_target = {}
158 
159  def populate(self, dot_files, with_qt=False):
160  """
161  Init db from dot files
162  """
163  parser = DotFileParser(with_qt)
164  for dot_file in dot_files:
165  yaml_file = dot_file.replace(".dot", ".yaml")
166  with open(yaml_file) as f:
167  dct = yaml.load(f)
168 
169  if 'tier' not in dct:
170  # This mean it's not a frameworks
171  continue
172 
173  tier = dct["tier"]
174  dot_file = dot_file.encode("utf-8")
175  fw = parser.parse(tier, dot_file)
176 
177  _add_extra_dependencies(fw, dct)
178  self._fw_list.append(fw)
179  self._update_fw_for_target()
180 
181  def _update_fw_for_target(self):
182  self._fw_for_target = {}
183  for fw in self._fw_list:
184  for target in fw.get_targets():
185  self._fw_for_target[target] = fw
186 
187  def find_by_name(self, name):
188  for fw in self._fw_list:
189  if fw.name == name:
190  return fw
191  return None
192 
193  def remove_unused_frameworks(self, wanted_fw):
194  def add_to_set(fw_set, wanted_fw):
195  fw_set.add(wanted_fw)
196 
197  for target in wanted_fw.get_all_target_dependencies():
198  fw = self._fw_for_target.get(target)
199  if fw is not None and not fw in fw_set:
200  add_to_set(fw_set, fw)
201 
202  for fw_name in wanted_fw.get_extra_frameworks():
203  fw = self.find_by_name(fw_name)
204  if not fw:
205  logging.warning("Framework {} has an extra dependency on {}, but there is no such framework".format(wanted_fw, fw_name))
206  continue
207  if not fw in fw_set:
208  add_to_set(fw_set, fw)
209 
210  fw_set = set()
211  add_to_set(fw_set, wanted_fw)
212  self._fw_list = list(fw_set)
213 
214  def find_external_targets(self):
215  all_targets = set([])
216  fw_targets = set([])
217  for fw in self._fw_list:
218  fw_targets.update(fw.get_targets())
219  all_targets.update(fw.get_all_target_dependencies())
220  return all_targets.difference(fw_targets)
221 
222  def get_framework_for_target(self, target):
223  return self._fw_for_target[target]
224 
225  def __iter__(self):
226  return iter(self._fw_list)
