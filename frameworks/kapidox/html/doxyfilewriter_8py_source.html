31 def _quote(txt):
32  return '"' + txt + '"'

34 class DoxyfileWriter(object):
35  """Makes it easy to write entries in a Doxyfile, correctly handling quoting
36  """
37  def __init__(self, fl):
38  self.fl = fl
39 
40  def write_entry(self, key, value):
41  """Write an entry
42 
43  Args:
44  key: the key part of the entry
45  value: the value part of the entry. Can be a string, a list, a
46  tuple or a boolean
47  """
48  if isinstance(value, (list, tuple)):
49  txt = ' '.join([_quote(x) for x in value])
50  elif isinstance(value, bool):
51  txt = ['NO', 'YES'][value]
52  else:
53  txt = _quote(str(value))
54  self.fl.write(key + ' = ' + txt + '\n')
55 
56  def write_entries(self, **kwargs):
57  """ Call write_entry for all arguments passed"""
58  for key, value in kwargs.items():
59  self.write_entry(key, value)
