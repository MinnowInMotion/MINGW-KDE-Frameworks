52 class FrameworkCmp(object):
53  def __init__(self, db):
54  self.db = db
55  self.src = set(db)
56  self.dst = []
57 
58  def __call__(self, fw1, fw2):
59  if self.depends_on(fw1, fw2):
60  return 1
61  if self.depends_on(fw2, fw1):
62  return -1
63  return 0
64 
65  def depends_on(self, depender_fw, provider_fw):
66  for dep_target in depender_fw.get_all_target_dependencies():
67  if provider_fw.has_target(dep_target):
68  return True
69 
70  try:
71  dep_fw = self.db.get_framework_for_target(dep_target)
72  except KeyError:
73  # No framework for this target, must be an external dependency,
74  # carry on
75  continue
76  if dep_fw != depender_fw and self.depends_on(dep_fw, provider_fw):
77  return True
78 
79  return False

82 class DotWriter(Block):
83  def __init__(self, db, out, wanted_fw=None, detailed=False):
84  Block.__init__(self, out)
85  self.db = db
86  self.detailed = detailed
87  self.wanted_fw = wanted_fw
88 
89  def write(self):
90  with self.curly_block("digraph Root") as root:
91  root.write_list_attrs("node", **ROOT_NODE_ATTRS)
92 
93  other_targets = self.db.find_external_targets()
94  qt_targets = set([x for x in other_targets if x.startswith("Qt")])
95  other_targets.difference_update(qt_targets)
96 
97  if qt_targets:
98  with root.cluster_block("Qt", **GROUP_ATTRS) as b:
99  b.write_list_attrs("node", **QT_ATTRS)
100  b.write_nodes(qt_targets)
101 
102  if other_targets:
103  with root.cluster_block("Others", **GROUP_ATTRS) as b:
104  b.write_list_attrs("node", **OTHER_ATTRS)
105  b.write_nodes(other_targets)
106 
107  lst = sorted([x for x in self.db], key=lambda x: x.tier)
108  for tier, frameworks in itertools.groupby(lst, lambda x: x.tier):
109  cluster_title = "Tier {}".format(tier)
110  with root.cluster_block(cluster_title, **GROUP_ATTRS) as tier_block:
111  tier_block.write_list_attrs("node", **FW_ATTRS)
112  # Sort frameworks within the tier to ensure frameworks which
113  # depend on other frameworks from that tier are listed after
114  # their dependees.
115  frameworks = list(frameworks)
116  for fw in sorted(frameworks, cmp=FrameworkCmp(self.db)):
117  if self.detailed:
118  self.write_detailed_framework(tier_block, fw)
119  else:
120  self.write_framework(tier_block, fw)
121 
122  def write_framework(self, tier_block, fw):
123  if fw == self.wanted_fw:
124  tier_block.write_list_attrs(quote(fw.name), **WANTED_FW_ATTRS)
125  else:
126  tier_block.write_nodes([fw.name])
127  edges = set([])
128  for target in fw.get_all_target_dependencies():
129  try:
130  target_fw = self.db.get_framework_for_target(target)
131  if fw == target_fw:
132  continue
133  dep = target_fw.name
134  except KeyError:
135  dep = target
136  edges.add((fw.name, dep))
137  for dep_fw in fw.get_extra_frameworks():
138  edges.add((fw.name, dep_fw))
139  for edge in edges:
140  tier_block.writeln('"{}" -> "{}";'.format(*edge))
141 
142  def write_detailed_framework(self, tier_block, fw):
143  with tier_block.cluster_block(fw.name, **FW_ATTRS) as fw_block:
144  if fw == self.wanted_fw:
145  fw_block.write_attrs(**WANTED_FW_ATTRS)
146  fw_block.write_list_attrs("node", **FW_TARGET_ATTRS)
147  targets = sorted(fw.get_targets())
148  fw_block.write_nodes(targets)
149  for target in targets:
150  deps = fw.get_dependencies_for_target(target)
151  for dep in sorted(deps):
152  fw_block.writeln('"{}" -> "{}";'.format(target, dep))

155 def generate(out, dot_files, framework=None, with_qt=False, detailed=False):
156  db = FrameworkDb()
157  db.populate(dot_files, with_qt=with_qt)
158 
159  if framework:
160  wanted_fw = db.find_by_name(framework)
161  if wanted_fw is None:
162  logging.error("No framework named {}.".format(framework))
163  return False
164  db.remove_unused_frameworks(wanted_fw)
165  else:
166  wanted_fw = None
167 
168  writer = DotWriter(db, out, wanted_fw=wanted_fw, detailed=detailed)
169  writer.write()
170
