112 bool KCompletionBox::eventFilter(QObject *o, QEvent *e)
113 {
114  Q_D(KCompletionBox);
115  int type = e->type();
116  QWidget *wid = qobject_cast<QWidget *>(o);
117 
118  if (o == this) {
119  return false;
120  }
121 
122  if (wid && wid == d->m_parent &&
123  (type == QEvent::Move || type == QEvent::Resize)) {
124  resizeAndReposition();
125  return false;
126  }
127 
128  if (wid && (wid->windowFlags() & Qt::Window) &&
129  type == QEvent::Move && wid == d->m_parent->window()) {
130  hide();
131  return false;
132  }
133 
134  if (type == QEvent::MouseButtonPress && (wid && !isAncestorOf(wid))) {
135  if (!d->emitSelected && currentItem() && !qobject_cast<QScrollBar *>(o)) {
136  Q_ASSERT(currentItem());
137  emit currentTextChanged(currentItem()->text());
138  }
139  hide();
140  e->accept();
141  return true;
142  }
143 
144  if (wid && wid->isAncestorOf(d->m_parent) && isVisible()) {
145  if (type == QEvent::KeyPress) {
146  QKeyEvent *ev = static_cast<QKeyEvent *>(e);
147  switch (ev->key()) {
148  case Qt::Key_Backtab:
149  if (d->tabHandling && (ev->modifiers() == Qt::NoButton ||
150  (ev->modifiers() & Qt::ShiftModifier))) {
151  up();
152  ev->accept();
153  return true;
154  }
155  break;
156  case Qt::Key_Tab:
157  if (d->tabHandling && (ev->modifiers() == Qt::NoButton)) {
158  down();
159  // #65877: Key_Tab should complete using the first
160  // (or selected) item, and then offer completions again
161  if (count() == 1) {
162  KLineEdit *parent = qobject_cast<KLineEdit *>(d->m_parent);
163  if (parent) {
164  parent->doCompletion(currentItem()->text());
165  } else {
166  hide();
167  }
168  }
169  ev->accept();
170  return true;
171  }
172  break;
173  case Qt::Key_Down:
174  down();
175  ev->accept();
176  return true;
177  case Qt::Key_Up:
178  // If there is no selected item and we've popped up above
179  // our parent, select the first item when they press up.
180  if (!selectedItems().isEmpty() ||
181  mapToGlobal(QPoint(0, 0)).y() >
182  d->m_parent->mapToGlobal(QPoint(0, 0)).y()) {
183  up();
184  } else {
185  down();
186  }
187  ev->accept();
188  return true;
189  case Qt::Key_PageUp:
190  pageUp();
191  ev->accept();
192  return true;
193  case Qt::Key_PageDown:
194  pageDown();
195  ev->accept();
196  return true;
197  case Qt::Key_Escape:
198  d->cancelled();
199  ev->accept();
200  return true;
201  case Qt::Key_Enter:
202  case Qt::Key_Return:
203  if (ev->modifiers() & Qt::ShiftModifier) {
204  hide();
205  ev->accept(); // Consume the Enter event
206  return true;
207  }
208  break;
209  case Qt::Key_End:
210  if (ev->modifiers() & Qt::ControlModifier) {
211  end();
212  ev->accept();
213  return true;
214  }
215  break;
216  case Qt::Key_Home:
217  if (ev->modifiers() & Qt::ControlModifier) {
218  home();
219  ev->accept();
220  return true;
221  }
222  default:
223  break;
224  }
225  } else if (type == QEvent::ShortcutOverride) {
226  // Override any accelerators that match
227  // the key sequences we use here...
228  QKeyEvent *ev = static_cast<QKeyEvent *>(e);
229  switch (ev->key()) {
230  case Qt::Key_Down:
231  case Qt::Key_Up:
232  case Qt::Key_PageUp:
233  case Qt::Key_PageDown:
234  case Qt::Key_Escape:
235  case Qt::Key_Enter:
236  case Qt::Key_Return:
237  ev->accept();
238  return true;
239  case Qt::Key_Tab:
240  case Qt::Key_Backtab:
241  if (ev->modifiers() == Qt::NoButton ||
242  (ev->modifiers() & Qt::ShiftModifier)) {
243  ev->accept();
244  return true;
245  }
246  break;
247  case Qt::Key_Home:
248  case Qt::Key_End:
249  if (ev->modifiers() & Qt::ControlModifier) {
250  ev->accept();
251  return true;
252  }
253  break;
254  default:
255  break;
256  }
257  } else if (type == QEvent::FocusOut) {
258  QFocusEvent *event = static_cast<QFocusEvent *>(e);
259  if (event->reason() != Qt::PopupFocusReason
260 #ifdef Q_OS_WIN
261  && (event->reason() != Qt::ActiveWindowFocusReason || QApplication::activeWindow() != this)
262 #endif
263  ) {
264  hide();
265  }
266  }
267  }
268 
269  return QListWidget::eventFilter(o, e);
270 }

290 void KCompletionBox::resizeAndReposition()
291 {
292  Q_D(KCompletionBox);
293  int currentGeom = height();
294  QPoint currentPos = pos();
295  QRect geom = calculateGeometry();
296  resize(geom.size());
297 
298  int x = currentPos.x(), y = currentPos.y();
299  if (d->m_parent) {
300  if (!isVisible()) {
301  QPoint orig = globalPositionHint();
302  QRect screenSize = QApplication::desktop()->screenGeometry(orig);
303 
304  x = orig.x() + geom.x();
305  y = orig.y() + geom.y();
306 
307  if (x + width() > screenSize.right()) {
308  x = screenSize.right() - width();
309  }
310  if (y + height() > screenSize.bottom()) {
311  y = y - height() - d->m_parent->height();
312  d->upwardBox = true;
313  }
314  } else {
315  // Are we above our parent? If so we must keep bottom edge anchored.
316  if (d->upwardBox) {
317  y += (currentGeom - height());
318  }
319  }
320  move(x, y);
321  }
322 }

365 QRect KCompletionBox::calculateGeometry() const
366 {
367  Q_D(const KCompletionBox);
368  QRect visualRect;
369  if (count() == 0 || !(visualRect = visualItemRect(item(0))).isValid()) {
370  return QRect();
371  }
372 
373  int x = 0, y = 0;
374  int ih = visualRect.height();
375  int h = qMin(15 * ih, (int) count() * ih) + 2 * frameWidth();
376 
377  int w = (d->m_parent) ? d->m_parent->width() : QListWidget::minimumSizeHint().width();
378  w = qMax(QListWidget::minimumSizeHint().width(), w);
379  return QRect(x, y, w, h);

493 void KCompletionBox::setItems(const QStringList &items)
494 {
495  bool block = signalsBlocked();
496  blockSignals(true);
497 
498  int rowIndex = 0;
499 
500  if (!count()) {
501  addItems(items);
502  } else {
503  // Keep track of whether we need to change anything,
504  // so we can avoid a repaint for identical updates,
505  // to reduce flicker
506  bool dirty = false;
507 
508  QStringList::ConstIterator it = items.constBegin();
509  const QStringList::ConstIterator itEnd = items.constEnd();
510 
511  for (; it != itEnd; ++it) {
512  if (rowIndex < count()) {
513  const bool changed = ((KCompletionBoxItem *)item(rowIndex))->reuse(*it);
514  dirty = dirty || changed;
515  } else {
516  dirty = true;
517  // Inserting an item is a way of making this dirty
518  addItem(*it);
519  }
520  rowIndex++;
521  }
522 
523  // If there is an unused item, mark as dirty -> less items now
524  if (rowIndex < count()) {
525  dirty = true;
526  }
527 
528  // remove unused items with an index >= rowIndex
529  for (; rowIndex < count();) {
530  QListWidgetItem *item = takeItem(rowIndex);
531  Q_ASSERT(item);
532  delete item;
533  }
534  }
535 
536  if (isVisible() && size().height() != sizeHint().height()) {
537  resizeAndReposition();
538  }
539 
540  blockSignals(block);
541 }
