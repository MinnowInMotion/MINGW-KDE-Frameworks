23 /* Fast zone memory allocator with deallocation support, for use as obstack
24  or as general purpose allocator. It does no compaction. If the usage
25  pattern is non-optimal it might waste some memory while running. E.g.
26  allocating many small things at once, and then deallocating only every
27  second one, there is a high chance, that actually no memory is freed.
28  */

36 class KZoneAllocator::MemBlock
37 {
38 public:
39  MemBlock(size_t s) : size(s), ref(0), older(0), newer(0)
40  {
41  begin = new char[s];
42  }
43  ~MemBlock()
44  {
45  delete [] begin;
46  }
47  bool is_in(void *ptr) const
48  {
49  return !(begin > (char *)ptr
50  || (begin + size) <= (char *)ptr);
51  }
52  size_t size;
53  unsigned int ref;
54  char *begin;
55  MemBlock *older;
56  MemBlock *newer;
57 };

87 KZoneAllocator::KZoneAllocator(unsigned long _blockSize)
88  : d(new Private)
89 {
90  while (d->blockSize < _blockSize) {
91  d->blockSize <<= 1;
92  d->log2++;
93  }
94 
95  /* Make sure, that a block is allocated at the first time allocate()
96  is called (even with a 0 size). */
97  d->blockOffset = d->blockSize + 1;
98 }

100 KZoneAllocator::~KZoneAllocator()
101 {
102  unsigned int count = 0;
103  if (d->hashList) {
104  /* No need to maintain the different lists in d->hashList[] anymore.
105  I.e. no need to use delBlock(). */
106  for (unsigned int i = 0; i < d->hashSize; i++) {
107  delete d->hashList[i];
108  }
109  delete [] d->hashList;
110  d->hashList = 0;
111  }
112  MemBlock *next;
113  for (; d->currentBlock; d->currentBlock = next) {
114  next = d->currentBlock->older;
115  delete d->currentBlock;
116  count++;
117  }
118 #ifndef NDEBUG // as this is called quite late in the app, we don't care
119  // to use qDebug
120  if (count > 1) {
121  fprintf(stderr, "zone still contained %d blocks", count);
122  }
123 #endif
124  delete d;
125 }

127 void KZoneAllocator::insertHash(MemBlock *b)
128 {
129  quintptr adr = ((quintptr)b->begin) & (~(d->blockSize - 1));
130  quintptr end = ((quintptr)b->begin) + d->blockSize;
131  while (adr < end) {
132  quintptr key = adr >> d->log2;
133  key = key & (d->hashSize - 1);
134  if (!d->hashList[key]) {
135  d->hashList[key] = new QList<MemBlock *>;
136  }
137  d->hashList[key]->append(b);
138  adr += d->blockSize;
139  }
140 }

147 void KZoneAllocator::addBlock(MemBlock *b)
148 {
149  b->newer = 0;
150  b->older = d->currentBlock;
151  if (d->currentBlock) {
152  b->older->newer = b;
153  }
154  d->currentBlock = b;
155  d->num_blocks++;
156  /* If we either have no d->hashList at all, or since it's last construction
157  there are now many more blocks we reconstruct the list. But don't
158  make it larger than a certain maximum. */
159  if (d->hashList && ((d->num_blocks / 4) > d->hashSize && d->hashSize < 64 * 1024)) {
160  d->hashDirty = true;
161  }
162  /* Only insert this block into the hashlists, if we aren't going to
163  reconstruct them anyway. */
164  if (d->hashList && !d->hashDirty) {
165  insertHash(b);
166  }
167 }

170 void KZoneAllocator::initHash()
171 {
172  if (d->hashList) {
173  for (unsigned int i = 0; i < d->hashSize; i++) {
174  delete d->hashList[i];
175  }
176  delete [] d->hashList;
177  d->hashList = 0;
178  }
179  d->hashSize = 1;
180  while (d->hashSize < d->num_blocks) {
181  d->hashSize <<= 1;
182  }
183  if (d->hashSize < 1024) {
184  d->hashSize = 1024;
185  }
186  if (d->hashSize > 64 * 1024) {
187  d->hashSize = 64 * 1024;
188  }
189  d->hashList = new QList<MemBlock *> *[d->hashSize];
190  memset(d->hashList, 0, sizeof(QList<MemBlock *> *) * d->hashSize);
191  d->hashDirty = false;
192  for (MemBlock *b = d->currentBlock; b; b = b->older) {
193  insertHash(b);
194  }
195 }

201 void KZoneAllocator::delBlock(MemBlock *b)
202 {
203  /* Update also the hashlists if we aren't going to reconstruct them
204  soon. */
205  if (d->hashList && !d->hashDirty) {
206  quintptr adr = ((quintptr)b->begin) & (~(d->blockSize - 1));
207  quintptr end = ((quintptr)b->begin) + d->blockSize;
208  while (adr < end) {
209  quintptr key = adr >> d->log2;
210  key = key & (d->hashSize - 1);
211  if (d->hashList[key]) {
212  QList<MemBlock *> *list = d->hashList[key];
213  QList<MemBlock *>::Iterator it = list->begin();
214  QList<MemBlock *>::Iterator endit = list->end();
215  for (; it != endit; ++it)
216  if (*it == b) {
217  list->erase(it);
218  break;
219  }
220  }
221  adr += d->blockSize;
222  }
223  }
224  if (b->older) {
225  b->older->newer = b->newer;
226  }
227  if (b->newer) {
228  b->newer->older = b->older;
229  }
230  if (b == d->currentBlock) {
231  d->currentBlock = 0;
232  d->blockOffset = d->blockSize;
233  }
234  delete b;
235  d->num_blocks--;
236 }

238 void *
239 KZoneAllocator::allocate(size_t _size)
240 {
241  // Use the size of (void *) as alignment
242  const size_t alignment = sizeof(void *) - 1;
243  _size = (_size + alignment) & ~alignment;
244 
245  if ((unsigned long) _size + d->blockOffset > d->blockSize) {
246  if (_size > d->blockSize) {
247  qDebug("KZoneAllocator: allocating more than %zu bytes", (size_t)d->blockSize);
248  return 0;
249  }
250  addBlock(new MemBlock(d->blockSize));
251  d->blockOffset = 0;
252  //qDebug ("Allocating block #%d (%x)\n", d->num_blocks, d->currentBlock->begin);
253  }
254  void *result = (void *)(d->currentBlock->begin + d->blockOffset);
255  d->currentBlock->ref++;
256  d->blockOffset += _size;
257  return result;
258 }

260 void
261 KZoneAllocator::deallocate(void *ptr)
262 {
263  if (d->hashDirty) {
264  initHash();
265  }
266 
267  quintptr key = (((quintptr)ptr) >> d->log2) & (d->hashSize - 1);
268  const QList<MemBlock *> *list = d->hashList[key];
269  if (!list) {
270  /* Can happen with certain usage pattern of intermixed free_since()
271  and deallocate(). */
272  //qDebug("Uhoh");
273  return;
274  }
275  QList<MemBlock *>::ConstIterator it = list->begin();
276  QList<MemBlock *>::ConstIterator endit = list->end();
277  for (; it != endit; ++it) {
278  MemBlock *cur = *it;
279  if (cur->is_in(ptr)) {
280  if (!--cur->ref) {
281  if (cur != d->currentBlock) {
282  delBlock(cur);
283  } else {
284  d->blockOffset = 0;
285  }
286  }
287  return;
288  }
289  }
290  /* Can happen with certain usage pattern of intermixed free_since()
291  and deallocate(). */
292  //qDebug("Uhoh2");
293 }

295 void
296 KZoneAllocator::free_since(void *ptr)
297 {
298  /* If we have a d->hashList and it's not yet dirty, see, if we will dirty
299  it by removing too many blocks. This will make the below delBlock()s
300  faster. */
301  if (d->hashList && !d->hashDirty) {
302  const MemBlock *b;
303  unsigned int removed = 0;
304  for (b = d->currentBlock; b; b = b->older, removed++)
305  if (b->is_in(ptr)) {
306  break;
307  }
308  if (d->hashSize >= 4 * (d->num_blocks - removed)) {
309  d->hashDirty = true;
310  }
311  }
312  while (d->currentBlock && !d->currentBlock->is_in(ptr)) {
313  d->currentBlock = d->currentBlock->older;
314  delBlock(d->currentBlock->newer);
315  }
316  d->blockOffset = ((char *)ptr) - d->currentBlock->begin;
317 }
