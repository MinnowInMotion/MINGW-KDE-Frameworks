67 void KComboBoxPrivate::_k_lineEditDeleted()
68 {
69  Q_Q(KComboBox);
70  // yes, we need those ugly casts due to the multiple inheritance
71  // sender() is guaranteed to be a KLineEdit (see the connect() to the
72  // destroyed() signal
73  const KCompletionBase *base = static_cast<const KCompletionBase *>(static_cast<const KLineEdit *>(q->sender()));
74 
75  // is it our delegate, that is destroyed?
76  if (base == q->delegate()) {
77  q->setDelegate(0L);
78  }
79 }

277 QSize KComboBox::minimumSizeHint() const
278 {
279  Q_D(const KComboBox);
280  QSize size = QComboBox::minimumSizeHint();
281  if (isEditable() && d->klineEdit) {
282  // if it's a KLineEdit and it's editable add the clear button size
283  // to the minimum size hint, otherwise looks ugly because the
284  // clear button will cover the last 2/3 letters of the biggest entry
285  QSize bs = d->klineEdit->clearButtonUsedSize();
286  if (bs.isValid()) {
287  size.rwidth() += bs.width();
288  size.rheight() = qMax(size.height(), bs.height());
289  }
290  }
291  return size;

294 void KComboBox::setLineEdit(QLineEdit *edit)
295 {
296  Q_D(KComboBox);
297  if (!isEditable() && edit &&
298  !qstrcmp(edit->metaObject()->className(), "QLineEdit")) {
299  // uic generates code that creates a read-only KComboBox and then
300  // calls combo->setEditable(true), which causes QComboBox to set up
301  // a dumb QLineEdit instead of our nice KLineEdit.
302  // As some KComboBox features rely on the KLineEdit, we reject
303  // this order here.
304  delete edit;
305  KLineEdit *kedit = new KLineEdit(this);
306 
307  if (isEditable()) {
308  kedit->setClearButtonShown(true);
309  }
310 
311  edit = kedit;
312  }
313 
314  QComboBox::setLineEdit(edit);
315  d->klineEdit = qobject_cast<KLineEdit *>(edit);
316  setDelegate(d->klineEdit);
317 
318  // Connect the returnPressed signal for both Q[K]LineEdits'
319  if (edit) {
320  connect(edit, SIGNAL(returnPressed()), SIGNAL(returnPressed()));
321  }
322 
323  if (d->klineEdit) {
324  // someone calling KComboBox::setEditable(false) destroys our
325  // line edit without us noticing. And KCompletionBase::delegate would
326  // be a dangling pointer then, so prevent that. Note: only do this
327  // when it is a KLineEdit!
328  connect(edit, SIGNAL(destroyed()), SLOT(_k_lineEditDeleted()));
329 
330  connect(d->klineEdit, SIGNAL(returnPressed(QString)),
331  SIGNAL(returnPressed(QString)));
332 
333  connect(d->klineEdit, SIGNAL(completion(QString)),
334  SIGNAL(completion(QString)));
335 
336  connect(d->klineEdit, SIGNAL(substringCompletion(QString)),
337  SIGNAL(substringCompletion(QString)));
338 
339  connect(d->klineEdit,
340  SIGNAL(textRotation(KCompletionBase::KeyBindingType)),
341  SIGNAL(textRotation(KCompletionBase::KeyBindingType)));
342 
343  connect(d->klineEdit,
344  SIGNAL(completionModeChanged(KCompletion::CompletionMode)),
345  SIGNAL(completionModeChanged(KCompletion::CompletionMode)));
346 
347  connect(d->klineEdit,
348  SIGNAL(aboutToShowContextMenu(QMenu*)),
349  SIGNAL(aboutToShowContextMenu(QMenu*)));
350 
351  connect(d->klineEdit,
352  SIGNAL(completionBoxActivated(QString)),
353  SIGNAL(activated(QString)));
354 
355  d->klineEdit->setTrapReturnKey(d->trapReturnKey);
356  }
357 }

359 void KComboBox::setCurrentItem(const QString &item, bool insert, int index)
360 {
361  int sel = -1;
362 
363  const int itemCount = count();
364  for (int i = 0; i < itemCount; ++i) {
365  if (itemText(i) == item) {
366  sel = i;
367  break;
368  }
369  }
370 
371  if (sel == -1 && insert) {
372  if (index >= 0) {
373  insertItem(index, item);
374  sel = index;
375  } else {
376  addItem(item);
377  sel = count() - 1;
378  }
379  }
380  setCurrentIndex(sel);
381 }

383 void KComboBox::setEditable(bool editable)
384 {
385  if (editable) {

388  // If a style needs this, then we'll need to call QComboBox::setEditable and then setLineEdit again
389  KLineEdit *edit = new KLineEdit(this);
390  edit->setClearButtonShown(true);
391  setLineEdit(edit);
392  } else {
393  QComboBox::setEditable(editable);
394  }
395 }
