37 void KCompletionPrivate::addWeightedItem(const QString &item)
38 {
39  Q_Q(KCompletion);
40  if (order != KCompletion::Weighted) {
41  q->addItem(item, 0);
42  return;
43  }
44 
45  uint len = item.length();
46  uint weight = 0;
47 
48  // find out the weighting of this item (appended to the string as ":num")
49  int index = item.lastIndexOf(QLatin1Char(':'));
50  if (index > 0) {
51  bool ok;
52  weight = item.midRef(index + 1).toUInt(&ok);
53  if (!ok) {
54  weight = 0;
55  }
56 
57  len = index; // only insert until the ':'
58  }
59 
60  q->addItem(item.left(len), weight);
61  return;
62 }

65 QString KCompletionPrivate::findCompletion(const QString &string)
66 {
67  QChar ch;
68  QString completion;
69  const KCompTreeNode *node = treeRoot;
70 
71  // start at the tree-root and try to find the search-string
72  for (int i = 0; i < string.length(); i++) {
73  ch = string.at(i);
74  node = node->find(ch);
75 
76  if (node) {
77  completion += ch;
78  } else {
79  return QString(); // no completion
80  }
81  }
82 
83  // Now we have the last node of the to be completed string.
84  // Follow it as long as it has exactly one child (= longest possible
85  // completion)
86 
87  while (node->childrenCount() == 1) {
88  node = node->firstChild();
89  if (!node->isNull()) {
90  completion += *node;
91  }
92  }
93  // if multiple matches and auto-completion mode
94  // -> find the first complete match
95  if (node && node->childrenCount() > 1) {
96  hasMultipleMatches = true;
97 
98  if (completionMode == KCompletion::CompletionAuto) {
99  rotationIndex = 1;
100  if (order != KCompletion::Weighted) {
101  while ((node = node->firstChild())) {
102  if (!node->isNull()) {
103  completion += *node;
104  } else {
105  break;
106  }
107  }
108  } else {
109  // don't just find the "first" match, but the one with the
110  // highest priority
111 
112  const KCompTreeNode *temp_node = 0L;
113  while (1) {
114  int count = node->childrenCount();
115  temp_node = node->firstChild();
116  uint weight = temp_node->weight();
117  const KCompTreeNode *hit = temp_node;
118  for (int i = 1; i < count; i++) {
119  temp_node = node->childAt(i);
120  if (temp_node->weight() > weight) {
121  hit = temp_node;
122  weight = hit->weight();
123  }
124  }
125  // 0x0 has the highest priority -> we have the best match
126  if (hit->isNull()) {
127  break;
128  }
129 
130  node = hit;
131  completion += *node;
132  }
133  }
134  }
135  }
136 
137  return completion;
138 }

237 void KCompletion::addItem(const QString &item, uint weight)
238 {
239  Q_D(KCompletion);
240  if (item.isEmpty()) {
241  return;
242  }
243 
244  KCompTreeNode *node = d->treeRoot;
245  uint len = item.length();
246 
247  bool sorted = (d->order == Sorted);
248  bool weighted = ((d->order == Weighted) && weight > 1);
249 
250  // knowing the weight of an item, we simply add this weight to all of its
251  // nodes.
252 
253  for (uint i = 0; i < len; i++) {
254  node = node->insert(item.at(i), sorted);
255  if (weighted) {
256  node->confirm(weight - 1); // node->insert() sets weighting to 1
257  }
258  }
259 
260  // add 0x0-item as delimiter with evtl. weight
261  node = node->insert(0x0, true);
262  if (weighted) {
263  node->confirm(weight - 1);
264  }
265 // qDebug("*** added: %s (%i)", item.toLatin1().constData(), node->weight());
266 }

289 QString KCompletion::makeCompletion(const QString &string)
290 {
291  Q_D(KCompletion);
292  if (d->completionMode == CompletionNone) {
293  return QString();
294  }
295 
296  //qDebug() << "KCompletion: completing: " << string;
297 
298  d->matches.clear();
299  d->rotationIndex = 0;
300  d->hasMultipleMatches = false;
301  d->lastMatch = d->currentMatch;
302 
303  // in Shell-completion-mode, emit all matches when we get the same
304  // complete-string twice
305  if (d->completionMode == CompletionShell &&
306  string == d->lastString) {
307  // Don't use d->matches since calling postProcessMatches()
308  // on d->matches here would interfere with call to
309  // postProcessMatch() during rotation
310 
311  d->matches.findAllCompletions(d->treeRoot, string, d->ignoreCase, d->hasMultipleMatches);
312  QStringList l = d->matches.list();
313  postProcessMatches(&l);
314  emit matches(l);
315 
316  return QString();
317  }
318 
319  QString completion;
320  // in case-insensitive popup mode, we search all completions at once
321  if (d->completionMode == CompletionPopup ||
322  d->completionMode == CompletionPopupAuto) {
323  d->matches.findAllCompletions(d->treeRoot, string, d->ignoreCase, d->hasMultipleMatches);
324  if (!d->matches.isEmpty()) {
325  completion = d->matches.first();
326  }
327  } else {
328  completion = d->findCompletion(string);
329  }
330 
331  if (d->hasMultipleMatches) {
332  emit multipleMatches();
333  }
334 
335  d->lastString = string;
336  d->currentMatch = completion;
337 
338  postProcessMatch(&completion);
339 
340  if (!string.isEmpty()) { // only emit match when string is not empty
341  //qDebug() << "KCompletion: Match: " << completion;
342  emit match(completion);
343  }
344 
345  return completion;
346 }

555 // Adds a child-node "ch" to this node. If such a node is already existent,
556 // it will not be created. Returns the new/existing node.
557 KCompTreeNode *KCompTreeNode::insert(const QChar &ch, bool sorted)
558 {
559  KCompTreeNode *child = find(ch);
560  if (!child) {
561  child = new KCompTreeNode(ch);
562 
563  // FIXME, first (slow) sorted insertion implementation
564  if (sorted) {
565  KCompTreeNode *prev = 0;
566  KCompTreeNode *cur = m_children.begin();
567  while (cur) {
568  if (ch > *cur) {
569  prev = cur;
570  cur = cur->m_next;
571  } else {
572  break;
573  }
574  }
575  if (prev) {
576  m_children.insert(prev, child);
577  } else {
578  m_children.prepend(child);
579  }
580  }
581 
582  else {
583  m_children.append(child);
584  }
585  }
586 
587  // implicit weighting: the more often an item is inserted, the higher
588  // priority it gets.
589  child->confirm();
590 
591  return child;
592 }

594 // Iteratively removes a string from the tree. The nicer recursive
595 // version apparently was a little memory hungry (see #56757)
596 void KCompTreeNode::remove(const QString &str)
597 {
598  QString string = str;
599  string += QChar(0x0);
600 
601  QVector<KCompTreeNode *> deletables(string.length() + 1);
602 
603  KCompTreeNode *child = 0L;
604  KCompTreeNode *parent = this;
605  deletables.replace(0, parent);
606 
607  int i = 0;
608  for (; i < string.length(); i++) {
609  child = parent->find(string.at(i));
610  if (child) {
611  deletables.replace(i + 1, child);
612  } else {
613  break;
614  }
615 
616  parent = child;
617  }
618 
619  for (; i >= 1; i--) {
620  parent = deletables.at(i - 1);
621  child = deletables.at(i);
622  if (child->m_children.count() == 0) {
623  delete parent->m_children.remove(child);
624  }
625  }
626 }

650 void KCompletionMatchesWrapper::findAllCompletions(const KCompTreeNode *treeRoot,const QString &string, bool ignoreCase, bool &hasMultipleMatches)
651 {
652  //qDebug() << "*** finding all completions for " << string;
653 
654  if (string.isEmpty()) {
655  return;
656  }
657 
658  if (ignoreCase) { // case insensitive completion
659  extractStringsFromNodeCI(treeRoot, QString(), string);
660  hasMultipleMatches = (count() > 1);
661  return;
662  }
663 
664  QChar ch;
665  QString completion;
666  const KCompTreeNode *node = treeRoot;
667 
668  // start at the tree-root and try to find the search-string
669  for (int i = 0; i < string.length(); i++) {
670  ch = string.at(i);
671  node = node->find(ch);
672 
673  if (node) {
674  completion += ch;
675  } else {
676  return; // no completion -> return empty list
677  }
678  }
679 
680  // Now we have the last node of the to be completed string.
681  // Follow it as long as it has exactly one child (= longest possible
682  // completion)
683 
684  while (node->childrenCount() == 1) {
685  node = node->firstChild();
686  if (!node->isNull()) {
687  completion += *node;
688  }
689  // qDebug() << completion << node->latin1();
690  }
691 
692  // there is just one single match)
693  if (node->childrenCount() == 0) {
694  append(node->weight(), completion);
695  }
696 
697  else {
698  // node has more than one child
699  // -> recursively find all remaining completions
700  hasMultipleMatches = true;
701  extractStringsFromNode(node, completion);
702  }
703 }

705 void KCompletionMatchesWrapper::extractStringsFromNode(const KCompTreeNode *node, const QString &beginning, bool addWeight)
706 {
707  if (!node) {
708  return;
709  }
710 
711  // qDebug() << "Beginning: " << beginning;
712  const KCompTreeChildren *list = node->children();
713  QString string;
714  QString w;
715 
716  // loop thru all children
717  for (KCompTreeNode *cur = list->begin(); cur; cur = cur->m_next) {
718  string = beginning;
719  node = cur;
720  if (!node->isNull()) {
721  string += *node;
722  }
723 
724  while (node && node->childrenCount() == 1) {
725  node = node->firstChild();
726  if (node->isNull()) {
727  break;
728  }
729  string += *node;
730  }
731 
732  if (node && node->isNull()) { // we found a leaf
733  if (addWeight) {
734  // add ":num" to the string to store the weighting
735  string += QLatin1Char(':');
736  w.setNum(node->weight());
737  string.append(w);
738  }
739  append(node->weight(), string);
740  }
741 
742  // recursively find all other strings.
743  if (node && node->childrenCount() > 1) {
744  extractStringsFromNode(node, string, addWeight);
745  }
746  }
747 }

749 void KCompletionMatchesWrapper::extractStringsFromNodeCI(const KCompTreeNode *node, const QString &beginning, const QString &restString)
750 {
751  if (restString.isEmpty()) {
752  extractStringsFromNode(node, beginning, false /*noweight*/);
753  return;
754  }
755 
756  QChar ch1 = restString.at(0);
757  QString newRest = restString.mid(1);
758  KCompTreeNode *child1, *child2;
759 
760  child1 = node->find(ch1); // the correct match
761  if (child1)
762  extractStringsFromNodeCI(child1, beginning + QChar(*child1), newRest);
763 
764  // append the case insensitive matches, if available
765  if (ch1.isLetter()) {
766  // find out if we have to lower or upper it. Is there a better way?
767  QChar ch2 = ch1.toLower();
768  if (ch1 == ch2) {
769  ch2 = ch1.toUpper();
770  }
771  if (ch1 != ch2) {
772  child2 = node->find(ch2);
773  if (child2)
774  extractStringsFromNodeCI(child2, beginning + QChar(*child2), newRest);
775  }
776  }
777 }

823 KCompTreeNode *KCompTreeNodeList::remove(KCompTreeNode *item)
824 {
825  if (!m_first || !item) {
826  return 0;
827  }
828  KCompTreeNode *cur = 0;
829 
830  if (item == m_first) {
831  m_first = m_first->m_next;
832  } else {
833  cur = m_first;
834  while (cur && cur->m_next != item) {
835  cur = cur->m_next;
836  }
837  if (!cur) {
838  return 0;
839  }
840  cur->m_next = item->m_next;
841  }
842  if (item == m_last) {
843  m_last = cur;
844  }
845  m_count--;
846  return item;
847 }
