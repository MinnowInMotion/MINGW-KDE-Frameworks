423 void KLineEdit::makeCompletion(const QString &text)
424 {
425  Q_D(KLineEdit);
426  KCompletion *comp = compObj();
427  KCompletion::CompletionMode mode = completionMode();
428 
429  if (!comp || mode == KCompletion::CompletionNone) {
430  return; // No completion object...
431  }
432 
433  const QString match = comp->makeCompletion(text);
434 
435  if (mode == KCompletion::CompletionPopup ||
436  mode == KCompletion::CompletionPopupAuto) {
437  if (match.isEmpty()) {
438  if (d->completionBox) {
439  d->completionBox->hide();
440  d->completionBox->clear();
441  }
442  } else {
443  setCompletedItems(comp->allMatches());
444  }
445  } else { // Auto, ShortAuto (Man) and Shell
446  // all other completion modes
447  // If no match or the same match, simply return without completing.
448  if (match.isEmpty() || match == text) {
449  return;

451

453  setUserSelection(false);
454  }

525 void KLineEditPrivate::setSqueezedText()
526 {
527  Q_Q(KLineEdit);
528  squeezedStart = 0;
529  squeezedEnd = 0;
530  const QString fullText = squeezedText;
531  const int fullLength = fullText.length();
532  const QFontMetrics fm(q->fontMetrics());
533  const int labelWidth = q->size().width() - 2 * q->style()->pixelMetric(QStyle::PM_DefaultFrameWidth) - 2;
534  const int textWidth = fm.width(fullText);
535 
536  if (textWidth > labelWidth) {
537  // start with the dots only
538  QString squeezedText = QStringLiteral("...");
539  int squeezedWidth = fm.width(squeezedText);
540 
541  // estimate how many letters we can add to the dots on both sides
542  int letters = fullText.length() * (labelWidth - squeezedWidth) / textWidth / 2;
543  squeezedText = fullText.left(letters) + "..." + fullText.right(letters);
544  squeezedWidth = fm.width(squeezedText);
545 
546  if (squeezedWidth < labelWidth) {
547  // we estimated too short
548  // add letters while text < label
549  do {
550  letters++;
551  squeezedText = fullText.left(letters) + "..." + fullText.right(letters);
552  squeezedWidth = fm.width(squeezedText);
553  } while (squeezedWidth < labelWidth && letters <= fullLength / 2);
554  letters--;
555  squeezedText = fullText.left(letters) + "..." + fullText.right(letters);
556  } else if (squeezedWidth > labelWidth) {
557  // we estimated too long
558  // remove letters while text > label
559  do {
560  letters--;
561  squeezedText = fullText.left(letters) + "..." + fullText.right(letters);
562  squeezedWidth = fm.width(squeezedText);
563  } while (squeezedWidth > labelWidth && letters >= 5);
564  }
565 
566  if (letters < 5) {
567  // too few letters added -> we give up squeezing
568  q->QLineEdit::setText(fullText);
569  } else {
570  q->QLineEdit::setText(squeezedText);
571  squeezedStart = letters;
572  squeezedEnd = fullText.length() - letters;
573  }
574 
575  q->setToolTip(fullText);
576 
577  } else {
578  q->QLineEdit::setText(fullText);
579 
580  q->setToolTip(QLatin1String(""));
581  QToolTip::showText(q->pos(), QString()); // hide
582  }
583 
584  q->setCursorPosition(0);
585 }

595 bool KLineEditPrivate::copySqueezedText(bool copy) const
596 {
597  Q_Q(const KLineEdit);
598  if (!squeezedText.isEmpty() && squeezedStart) {
599  KLineEdit *that = const_cast<KLineEdit *>(q);
600  if (!that->hasSelectedText()) {
601  return false;
602  }
603  int start = q->selectionStart(), end = start + q->selectedText().length();
604  if (start >= squeezedStart + 3) {
605  start = start - 3 - squeezedStart + squeezedEnd;
606  } else if (start > squeezedStart) {
607  start = squeezedStart;
608  }
609  if (end >= squeezedStart + 3) {
610  end = end - 3 - squeezedStart + squeezedEnd;
611  } else if (end > squeezedStart) {
612  end = squeezedEnd;
613  }
614  if (start == end) {
615  return false;
616  }
617  QString t = squeezedText;
618  t = t.mid(start, end - start);
619  q->disconnect(QApplication::clipboard(), SIGNAL(selectionChanged()), q, 0);
620  QApplication::clipboard()->setText(t, copy ? QClipboard::Clipboard : QClipboard::Selection);
621  q->connect(QApplication::clipboard(), SIGNAL(selectionChanged()), q,
622  SLOT(_q_clipboardChanged()));
623  return true;
624  }
625  return false;
626 }

661  if (!isReadOnly()) {

682  return;
683  } else if (KStandardShortcut::deleteWordForward().contains(key)) {

687  del();
688  }

690  e->accept();
691  return;

695  return;

697  cursorWordForward(false);
698  e->accept();
