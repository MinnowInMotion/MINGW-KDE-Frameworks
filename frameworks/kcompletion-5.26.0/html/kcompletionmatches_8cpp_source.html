81 QStringList KCompletionMatches::list(bool sort_P) const
82 {
83  Q_D(const KCompletionMatches);
84  if (d->sorting && sort_P) {
85  const_cast< KCompletionMatches * >(this)->sort();
86  }
87  QStringList stringList;
88  // high weight == sorted last -> reverse the sorting here
89  for (ConstIterator it = begin(); it != end(); ++it) {
90  stringList.prepend((*it).value());
91  }
92  return stringList;
93 }

101 void KCompletionMatches::removeDuplicates()
102 {
103  Iterator it1, it2;
104  for (it1 = begin(); it1 != end(); ++it1) {
105  for ((it2 = it1), ++it2; it2 != end();) {
106  if ((*it1).value() == (*it2).value()) {
107  // use the max height
108  (*it1).first = qMax((*it1).key(), (*it2).key());
109  it2 = erase(it2);
110  continue;
111  }
112  ++it2;
113  }
114  }
115 }
