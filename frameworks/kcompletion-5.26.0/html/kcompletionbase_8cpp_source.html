99 KCompletion *KCompletionBase::completionObject(bool handleSignals)
100 {
101  Q_D(KCompletionBase);
102  if (d->delegate) {
103  return d->delegate->completionObject(handleSignals);
104  }
105 
106  if (!d->completionObject) {
107  setCompletionObject(new KCompletion(), handleSignals);
108  d->autoDeleteCompletionObject = true;
109  }
110  return d->completionObject;
111 }

113 void KCompletionBase::setCompletionObject(KCompletion *completionObject, bool handleSignals)
114 {
115  Q_D(KCompletionBase);
116  if (d->delegate) {
117  d->delegate->setCompletionObject(completionObject, handleSignals);
118  return;
119  }
120 
121  if (d->autoDeleteCompletionObject && completionObject != d->completionObject) {
122  delete d->completionObject;
123  }
124 
125  d->completionObject = completionObject;
126 
127  setAutoDeleteCompletionObject(false);
128  setHandleSignals(handleSignals);
129 
130  // We emit rotation and completion signals
131  // if completion object is not NULL.
132  setEmitSignals(!d->completionObject.isNull());
133 }

217 bool KCompletionBase::setKeyBinding(KeyBindingType item, const QList<QKeySequence> &cut)
218 {
219  Q_D(KCompletionBase);
220  if (d->delegate) {
221  return d->delegate->setKeyBinding(item, cut);
222  }
223 
224  if (!cut.isEmpty()) {
225  for (KeyBindingMap::Iterator it = d->keyBindingMap.begin(); it != d->keyBindingMap.end(); ++it)
226  if (it.value() == cut) {
227  return false;
228  }
229  }
230  d->keyBindingMap.insert(item, cut);
231  return true;
232 }

240 void KCompletionBase::useGlobalKeyBindings()
241 {
242  Q_D(KCompletionBase);
243  if (d->delegate) {
244  d->delegate->useGlobalKeyBindings();
245  return;
246  }
247 
248  d->keyBindingMap.clear();
249  d->keyBindingMap.insert(TextCompletion, QList<QKeySequence>());
250  d->keyBindingMap.insert(PrevCompletionMatch, QList<QKeySequence>());
251  d->keyBindingMap.insert(NextCompletionMatch, QList<QKeySequence>());
252  d->keyBindingMap.insert(SubstringCompletion, QList<QKeySequence>());
