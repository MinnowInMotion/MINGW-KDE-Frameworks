30 void KWindowConfig::saveWindowSize(const QWindow *window, KConfigGroup &config, KConfigGroup::WriteConfigFlags options)
31 {
32  // QWindow::screen() shouldn't return null, but it sometimes does due to bugs.
33  if (!window || !window->screen()) {
34  return;
35  }
36  const QRect desk = window->screen()->geometry();
37 
38  const QSize sizeToSave = window->size();
39  const bool isMaximized = window->windowState() & Qt::WindowMaximized;
40 
41  const QString screenMaximizedString(QStringLiteral("Window-Maximized %1x%2").arg(desk.height()).arg(desk.width()));
42  // Save size only if window is not maximized
43  if (!isMaximized) {
44  const QSize defaultSize(window->property(s_initialSizePropertyName).toSize());
45  const QSize defaultScreenSize(window->property(s_initialScreenSizePropertyName).toSize());
46  const bool sizeValid = defaultSize.isValid() && defaultScreenSize.isValid();
47  if (!sizeValid || (sizeValid && (defaultSize != sizeToSave || defaultScreenSize != desk.size()))) {
48  const QString wString(QStringLiteral("Width %1").arg(desk.width()));
49  const QString hString(QStringLiteral("Height %1").arg(desk.height()));
50  config.writeEntry(wString, sizeToSave.width(), options);
51  config.writeEntry(hString, sizeToSave.height(), options);
52  }
53  }
54  if ((isMaximized == false) && !config.hasDefault(screenMaximizedString)) {
55  config.revertToDefault(screenMaximizedString);
56  } else {
57  config.writeEntry(screenMaximizedString, isMaximized, options);
58  }
59 
60 }

62 void KWindowConfig::restoreWindowSize(QWindow *window, const KConfigGroup &config)
63 {
64  if (!window) {
65  return;
66  }
67 
68  const QRect desk = window->screen()->geometry();
69 
70  const int width = config.readEntry(QStringLiteral("Width %1").arg(desk.width()), window->size().width());
71  const int height = config.readEntry(QStringLiteral("Height %1").arg(desk.height()), window->size().height());
72  const bool isMaximized = config.readEntry(QStringLiteral("Window-Maximized %1x%2").arg(desk.height()).arg(desk.width()), false);
73 
74  // Check default size
75  const QSize defaultSize(window->property(s_initialSizePropertyName).toSize());
76  const QSize defaultScreenSize(window->property(s_initialScreenSizePropertyName).toSize());
77  if (!defaultSize.isValid() || !defaultScreenSize.isValid()) {
78  window->setProperty(s_initialSizePropertyName, window->size());
79  window->setProperty(s_initialScreenSizePropertyName, desk.size());
80  }
81 
82  // If window is maximized set maximized state and in all case set the size
83  window->resize(width, height);
84  if (isMaximized) {
85  window->setWindowState(Qt::WindowMaximized);
86  }
87 }
