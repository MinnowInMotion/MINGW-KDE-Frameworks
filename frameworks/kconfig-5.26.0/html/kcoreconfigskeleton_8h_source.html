272  void writeConfig(KConfig *config) Q_DECL_OVERRIDE
273  {
274  if (mReference != mLoadedValue) { // Is this needed?
275  KConfigGroup cg(config, mGroup);
276  if ((mDefault == mReference) && !cg.hasDefault(mKey)) {
277  cg.revertToDefault(mKey);
278  } else {
279  cg.writeEntry(mKey, mReference);
280  }
281  mLoadedValue = mReference;
282  }
283  }

286  void readDefault(KConfig *config) Q_DECL_OVERRIDE
287  {
288  config->setReadDefaults(true);
289  readConfig(config);
290  config->setReadDefaults(false);
291  mDefault = mReference;
292  }
293 
295  void swapDefault() Q_DECL_OVERRIDE
296  {
297  T tmp = mReference;
298  mReference = mDefault;
299  mDefault = tmp;
300  }

304  T mDefault;
