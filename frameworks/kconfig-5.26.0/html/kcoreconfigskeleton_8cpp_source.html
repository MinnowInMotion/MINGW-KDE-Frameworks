27 static QString obscuredString(const QString &str)
28 {
29  QString result;
30  const QChar *unicode = str.unicode();
31  for (int i = 0; i < str.length(); ++i)
32  // yes, no typo. can't encode ' ' or '!' because
33  // they're the unicode BOM. stupid scrambling. stupid.
34  result += (unicode[ i ].unicode() <= 0x21) ? unicode[ i ]
35  : QChar(0x1001F - unicode[ i ].unicode());
36 
37  return result;
38 }

452 void KCoreConfigSkeleton::ItemEnum::readConfig(KConfig *config)
453 {
454  KConfigGroup cg(config, mGroup);
455  if (!cg.hasKey(mKey)) {
456  mReference = mDefault;
457  } else {
458  int i = 0;
459  mReference = -1;
460  QString tmp = cg.readEntry(mKey, QString()).toLower();
461  for (QList<Choice>::ConstIterator it = mChoices.constBegin();
462  it != mChoices.constEnd(); ++it, ++i) {
463  if ((*it).name.toLower() == tmp) {
464  mReference = i;
465  break;
466  }
467  }
468  if (mReference == -1) {
469  mReference = cg.readEntry(mKey, mDefault);
470  }
471  }
472  mLoadedValue = mReference;
473 
474  readImmutability(cg);
475 }

484  cg.writeEntry(mKey, mChoices[mReference].name);
