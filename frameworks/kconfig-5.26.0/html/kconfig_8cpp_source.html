120 void KConfigPrivate::copyGroup(const QByteArray &source, const QByteArray &destination,
121  KConfigGroup *otherGroup, KConfigBase::WriteConfigFlags flags) const
122 {
123  KEntryMap &otherMap = otherGroup->config()->d_ptr->entryMap;
124  const int len = source.length();
125  const bool sameName = (destination == source);
126 
127  // we keep this bool outside the foreach loop so that if
128  // the group is empty, we don't end up marking the other config
129  // as dirty erroneously
130  bool dirtied = false;
131 
132  for (KEntryMap::ConstIterator entryMapIt(entryMap.constBegin()); entryMapIt != entryMap.constEnd(); ++entryMapIt) {
133  const QByteArray &group = entryMapIt.key().mGroup;
134 
135  if (!group.startsWith(source)) { // nothing to do
136  continue;
137  }
138 
139  // don't copy groups that start with the same prefix, but are not sub-groups
140  if (group.length() > len && group[len] != '\x1d') {
141  continue;
142  }
143 
144  KEntryKey newKey = entryMapIt.key();
145 
146  if (flags & KConfigBase::Localized) {
147  newKey.bLocal = true;
148  }
149 
150  if (!sameName) {
151  newKey.mGroup.replace(0, len, destination);
152  }
153 
154  KEntry entry = entryMap[ entryMapIt.key() ];
155  dirtied = entry.bDirty = flags & KConfigBase::Persistent;
156 
157  if (flags & KConfigBase::Global) {
158  entry.bGlobal = true;
159  }
160 
161  otherMap[newKey] = entry;
162  }
163 
164  if (dirtied) {
165  otherGroup->config()->d_ptr->bDirty = true;
166  }
167 }

169 QString KConfigPrivate::expandString(const QString &value)
170 {
171  QString aValue = value;
172 
173  // check for environment variables and make necessary translations
174  int nDollarPos = aValue.indexOf(QLatin1Char('$'));
175  while (nDollarPos != -1 && nDollarPos + 1 < aValue.length()) {
176  // there is at least one $
177  if (aValue[nDollarPos + 1] == QLatin1Char('(')) {
178  int nEndPos = nDollarPos + 1;
179  // the next character is not $
180  while ((nEndPos <= aValue.length()) && (aValue[nEndPos] != QLatin1Char(')'))) {
181  nEndPos++;
182  }
183  nEndPos++;
184  QString cmd = aValue.mid(nDollarPos + 2, nEndPos - nDollarPos - 3);
185 
186  QString result;
187 
188 // FIXME: wince does not have pipes
189 #ifndef _WIN32_WCE
190  FILE *fs = popen(QFile::encodeName(cmd).data(), "r");
191  if (fs) {
192  QTextStream ts(fs, QIODevice::ReadOnly);
193  result = ts.readAll().trimmed();
194  pclose(fs);
195  }
196 #endif
197  aValue.replace(nDollarPos, nEndPos - nDollarPos, result);
198  nDollarPos += result.length();
199  } else if (aValue[nDollarPos + 1] != QLatin1Char('$')) {
200  int nEndPos = nDollarPos + 1;
201  // the next character is not $
202  QString aVarName;
203  if (aValue[nEndPos] == QLatin1Char('{')) {
204  while ((nEndPos <= aValue.length()) && (aValue[nEndPos] != QLatin1Char('}'))) {
205  nEndPos++;
206  }
207  nEndPos++;
208  aVarName = aValue.mid(nDollarPos + 2, nEndPos - nDollarPos - 3);
209  } else {
210  while (nEndPos <= aValue.length() &&
211  (aValue[nEndPos].isNumber() ||
212  aValue[nEndPos].isLetter() ||
213  aValue[nEndPos] == QLatin1Char('_'))) {
214  nEndPos++;
215  }
216  aVarName = aValue.mid(nDollarPos + 1, nEndPos - nDollarPos - 1);
217  }
218  QString env;
219  if (!aVarName.isEmpty()) {
220 #ifdef Q_OS_WIN
221  if (aVarName == QLatin1String("HOME")) {
222  env = QDir::homePath();
223  } else
224 #endif
225  {
226  QByteArray pEnv = qgetenv(aVarName.toLatin1().constData());
227  if (!pEnv.isEmpty()) {
228  env = QString::fromLocal8Bit(pEnv.constData());
229  } else {
230  if (aVarName == QStringLiteral("QT_DATA_HOME")) {
231  env = QStandardPaths::writableLocation(QStandardPaths::GenericDataLocation);
232  } else if (aVarName == QStringLiteral("QT_CONFIG_HOME")) {
233  env = QStandardPaths::writableLocation(QStandardPaths::GenericConfigLocation);
234  } else if (aVarName == QStringLiteral("QT_CACHE_HOME")) {
235  env = QStandardPaths::writableLocation(QStandardPaths::GenericCacheLocation);
236  }
237  }
238  }
239  aValue.replace(nDollarPos, nEndPos - nDollarPos, env);
240  nDollarPos += env.length();
241  } else {
242  aValue.remove(nDollarPos, nEndPos - nDollarPos);
243  }
244  } else {
245  // remove one of the dollar signs
246  aValue.remove(nDollarPos, 1);
247  nDollarPos++;
248  }
249  nDollarPos = aValue.indexOf(QLatin1Char('$'), nDollarPos);
250  }
251 
252  return aValue;
253 }

386 QMap<QString, QString> KConfig::entryMap(const QString &aGroup) const
387 {
388  Q_D(const KConfig);
389  QMap<QString, QString> theMap;
390  const QByteArray theGroup(aGroup.isEmpty() ? "<default>" : aGroup.toUtf8());
391 
392  const KEntryMapConstIterator theEnd = d->entryMap.constEnd();
393  KEntryMapConstIterator it = d->entryMap.findEntry(theGroup, 0, 0);
394  if (it != theEnd) {
395  ++it; // advance past the special group entry marker
396 
397  for (; it != theEnd && it.key().mGroup == theGroup; ++it) {
398  // leave the default values and deleted entries out
399  if (!it->bDeleted && !it.key().bDefault) {
400  const QString key = QString::fromUtf8(it.key().mKey.constData());
401  // the localized entry should come first, so don't overwrite it
402  // with the non-localized entry
403  if (!theMap.contains(key)) {
404  if (it->bExpand) {
405  theMap.insert(key, KConfigPrivate::expandString(QString::fromUtf8(it->mValue.constData())));
406  } else {
407  theMap.insert(key, QString::fromUtf8(it->mValue.constData()));
408  }
409  }
410  }
411  }
412  }
413 
414  return theMap;
415 }

417 bool KConfig::sync()
418 {
419  Q_D(KConfig);
420 
421  if (isImmutable() || name().isEmpty()) {
422  // can't write to an immutable or anonymous file.
423  return false;
424  }
425 
426  if (d->bDirty && d->mBackend) {
427  const QByteArray utf8Locale(locale().toUtf8());
428 
429  // Create the containing dir, maybe it wasn't there
430  d->mBackend->createEnclosing();
431 
432  // lock the local file
433  if (d->configState == ReadWrite && !d->lockLocal()) {
434  qWarning() << "couldn't lock local file";
435  return false;
436  }
437 
438  // Rewrite global/local config only if there is a dirty entry in it.
439  bool writeGlobals = false;
440  bool writeLocals = false;
441  Q_FOREACH (const KEntry &e, d->entryMap) {
442  if (e.bDirty) {
443  if (e.bGlobal) {
444  writeGlobals = true;
445  } else {
446  writeLocals = true;
447  }
448 
449  if (writeGlobals && writeLocals) {
450  break;
451  }
452  }
453  }
454 
455  d->bDirty = false; // will revert to true if a config write fails
456 
457  if (d->wantGlobals() && writeGlobals) {
458  QExplicitlySharedDataPointer<KConfigBackend> tmp = KConfigBackend::create(*sGlobalFileName);
459  if (d->configState == ReadWrite && !tmp->lock()) {
460  qWarning() << "couldn't lock global file";
461 
462  //unlock the local config if we're returning early
463  if (d->mBackend->isLocked()) {
464  d->mBackend->unlock();
465  }
466 
467  d->bDirty = true;
468  return false;
469  }
470  if (!tmp->writeConfig(utf8Locale, d->entryMap, KConfigBackend::WriteGlobal)) {
471  d->bDirty = true;
472  }
473  if (tmp->isLocked()) {
474  tmp->unlock();
475  }
476  }
477 
478  if (writeLocals) {
479  if (!d->mBackend->writeConfig(utf8Locale, d->entryMap, KConfigBackend::WriteOptions())) {
480  d->bDirty = true;
481  }
482  }
483  if (d->mBackend->isLocked()) {
484  d->mBackend->unlock();
485  }
486  }
487  return !d->bDirty;
488 }

588 void KConfigPrivate::changeFileName(const QString &name)
589 {
590  fileName = name;
591 
592  QString file;
593  if (name.isEmpty()) {
594  if (wantDefaults()) { // accessing default app-specific config "appnamerc"
595  fileName = KConfig::mainConfigName();
596  file = QStandardPaths::writableLocation(resourceType) + QLatin1Char('/') + fileName;
597  } else if (wantGlobals()) { // accessing "kdeglobals" by specifying no filename and NoCascade - XXX used anywhere?
598  resourceType = QStandardPaths::GenericConfigLocation;
599  fileName = QStringLiteral("kdeglobals");
600  file = *sGlobalFileName;
601  } else {
602  // anonymous config
603  openFlags = KConfig::SimpleConfig;
604  return;
605  }
606  } else if (QDir::isAbsolutePath(fileName)) {
607  fileName = QFileInfo(fileName).canonicalFilePath();
608  if (fileName.isEmpty()) { // file doesn't exist (yet)
609  fileName = name;
610  }
611  file = fileName;
612  } else {
613  file = QStandardPaths::writableLocation(resourceType) + QLatin1Char('/') + fileName;
614  }
615 
616  Q_ASSERT(!file.isEmpty());
617 
618  bSuppressGlobal = (file.compare(*sGlobalFileName, sPathCaseSensitivity) == 0);
619 
620  if (bDynamicBackend || !mBackend) { // allow dynamic changing of backend
621  mBackend = KConfigBackend::create(file);
622  } else {
623  mBackend->setFilePath(file);
624  }
625 
626  configState = mBackend->accessMode();
627 }

629 void KConfig::reparseConfiguration()
630 {
631  Q_D(KConfig);
632  if (d->fileName.isEmpty()) {
633  return;
634  }
635 
636  // Don't lose pending changes
637  if (!d->isReadOnly() && d->bDirty) {
638  sync();
639  }
640 
641  d->entryMap.clear();
642 
643  d->bFileImmutable = false;
644 
645  {
646  QMutexLocker locker(&s_globalFilesMutex);
647  s_globalFiles()->clear();
648  }
649 
650  // Parse all desired files from the least to the most specific.

683 void KConfigPrivate::parseGlobalFiles()
684 {
685  const QStringList globalFiles = getGlobalFiles();
686 // qDebug() << "parsing global files" << globalFiles;
687 
688  // TODO: can we cache the values in etc_kderc / other global files
689  // on a per-application basis?
690  const QByteArray utf8Locale = locale.toUtf8();
691  Q_FOREACH (const QString &file, globalFiles) {
692  KConfigBackend::ParseOptions parseOpts = KConfigBackend::ParseGlobal | KConfigBackend::ParseExpansions;
693 
694  if (file.compare(*sGlobalFileName, sPathCaseSensitivity) != 0)
695  parseOpts |= KConfigBackend::ParseDefaults;
696 
697  QExplicitlySharedDataPointer<KConfigBackend> backend = KConfigBackend::create(file);
698  if (backend->parseConfig(utf8Locale, entryMap, parseOpts) == KConfigBackend::ParseImmutable) {
699  break;
700  }
701  }

704 void KConfigPrivate::parseConfigFiles()
705 {
706  // can only read the file if there is a backend and a file name
707  if (mBackend && !fileName.isEmpty()) {
708 
709  bFileImmutable = false;
710 
711  QList<QString> files;
712  if (wantDefaults()) {
713  if (bSuppressGlobal) {
714  files = getGlobalFiles();
715  } else {
716  if (QDir::isAbsolutePath(fileName)) {
717  files << fileName;
718  } else {
719  Q_FOREACH (const QString &f, QStandardPaths::locateAll(resourceType, fileName)) {
720  files.prepend(f);
721  }
722 
723  // allow fallback to config files bundled in resources
724  const QString resourceFile(QStringLiteral(":/kconfig/") + fileName);
725  if (QFile::exists(resourceFile)) {
726  files.prepend(resourceFile);
727  }
728  }
729  }
730  } else {
731  files << mBackend->filePath();
732  }
733  if (!isSimple()) {
734  files = extraFiles.toList() + files;
735  }
736 
737 // qDebug() << "parsing local files" << files;
738 
739  const QByteArray utf8Locale = locale.toUtf8();
740  foreach (const QString &file, files) {
741  if (file.compare(mBackend->filePath(), sPathCaseSensitivity) == 0) {
742  switch (mBackend->parseConfig(utf8Locale, entryMap, KConfigBackend::ParseExpansions)) {
743  case KConfigBackend::ParseOk:
744  break;
745  case KConfigBackend::ParseImmutable:
746  bFileImmutable = true;
747  break;
748  case KConfigBackend::ParseOpenError:
749  configState = KConfigBase::NoAccess;
750  break;
751  }
752  } else {
753  QExplicitlySharedDataPointer<KConfigBackend> backend = KConfigBackend::create(file);
754  bFileImmutable = (backend->parseConfig(utf8Locale, entryMap,
755  KConfigBackend::ParseDefaults | KConfigBackend::ParseExpansions)
756  == KConfigBackend::ParseImmutable);
757  }
758 
759  if (bFileImmutable) {
760  break;
761  }
762  }
763 #pragma message("TODO: enable kiosk feature again (resource restrictions), but without KStandardDirs... Needs a class in the kconfig framework.")
764 #if 0
765  if (componentData.dirs()->isRestrictedResource(resourceType, fileName)) {
766  bFileImmutable = true;
767  }
768 #endif
769  }
770 }

882 {

894  }
895  }
