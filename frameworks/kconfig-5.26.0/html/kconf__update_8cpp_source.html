230 bool KonfUpdate::checkFile(const QString &filename)
231 {
232  m_currentFilename = filename;
233  int i = m_currentFilename.lastIndexOf('/');
234  if (i != -1) {
235  m_currentFilename = m_currentFilename.mid(i + 1);
236  }
237  m_skip = true;
238  QFile file(filename);
239  if (!file.open(QIODevice::ReadOnly)) {
240  return false;
241  }
242 
243  QTextStream ts(&file);
244  ts.setCodec(QTextCodec::codecForName("ISO-8859-1"));
245  int lineCount = 0;
246  resetOptions();
247  QString id;
248  bool foundVersion = false;
249  while (!ts.atEnd()) {
250  const QString line = ts.readLine().trimmed();
251  if (line.startsWith(QLatin1String("Version=5"))) {
252  foundVersion = true;
253  }
254  ++lineCount;
255  if (line.isEmpty() || (line[0] == '#')) {
256  continue;
257  }
258  if (line.startsWith(QLatin1String("Id="))) {
259  if (!foundVersion) {
260  qDebug() << QStringLiteral("Missing \"Version=5\", file \'%1\' will be skipped.").arg(filename);
261  return true;
262  }
263  id = m_currentFilename + ':' + line.mid(3);
264  } else if (line.startsWith(QLatin1String("File="))) {
265  checkGotFile(line.mid(5), id);
266  }
267  }
268 
269  return true;
270 }

313 bool KonfUpdate::updateFile(const QString &filename)
314 {
315  m_currentFilename = filename;
316  int i = m_currentFilename.lastIndexOf('/');
317  if (i != -1) {
318  m_currentFilename = m_currentFilename.mid(i + 1);
319  }
320  m_skip = true;
321  QFile file(filename);
322  if (!file.open(QIODevice::ReadOnly)) {
323  return false;
324  }
325 
326  log() << "Checking update-file '" << filename << "' for new updates" << endl;
327 
328  QTextStream ts(&file);
329  ts.setCodec(QTextCodec::codecForName("ISO-8859-1"));
330  m_lineCount = 0;
331  resetOptions();
332  bool foundVersion = false;
333  while (!ts.atEnd()) {
334  m_line = ts.readLine().trimmed();
335  if (m_line.startsWith(QLatin1String("Version=5"))) {
336  foundVersion = true;
337  }
338  m_lineCount++;
339  if (m_line.isEmpty() || (m_line[0] == QLatin1Char('#'))) {
340  continue;
341  }
342  if (m_line.startsWith(QLatin1String("Id="))) {
343  if (!foundVersion) {
344  qDebug() << QStringLiteral("Missing \"Version=5\", file \'%1\' will be skipped.").arg(filename);
345  break;
346  }
347  gotId(m_line.mid(3));
348  } else if (m_skip) {
349  continue;
350  } else if (m_line.startsWith(QLatin1String("Options="))) {
351  gotOptions(m_line.mid(8));
352  } else if (m_line.startsWith(QLatin1String("File="))) {
353  gotFile(m_line.mid(5));
354  } else if (m_skipFile) {
355  continue;
356  } else if (m_line.startsWith(QLatin1String("Group="))) {
357  gotGroup(m_line.mid(6));
358  } else if (m_line.startsWith(QLatin1String("RemoveGroup="))) {
359  gotRemoveGroup(m_line.mid(12));
360  resetOptions();
361  } else if (m_line.startsWith(QLatin1String("Script="))) {
362  gotScript(m_line.mid(7));
363  resetOptions();
364  } else if (m_line.startsWith(QLatin1String("ScriptArguments="))) {
365  gotScriptArguments(m_line.mid(16));
366  } else if (m_line.startsWith(QLatin1String("Key="))) {
367  gotKey(m_line.mid(4));
368  resetOptions();
369  } else if (m_line.startsWith(QLatin1String("RemoveKey="))) {
370  gotRemoveKey(m_line.mid(10));
371  resetOptions();
372  } else if (m_line == QLatin1String("AllKeys")) {
373  gotAllKeys();
374  resetOptions();
375  } else if (m_line == QLatin1String("AllGroups")) {
376  gotAllGroups();
377  resetOptions();
378  } else {
379  logFileError() << "Parse error" << endl;
380  }
381  }
382  // Flush.
383  gotId(QString());
384 
385  QFileInfo info(filename);
386  KConfigGroup cg(m_config, m_currentFilename);
387  cg.writeEntry("ctime", info.created().toTime_t());
388  cg.writeEntry("mtime", info.lastModified().toTime_t());
389  cg.sync();
390  return true;
391 }

430 void KonfUpdate::gotFile(const QString &_file)
431 {
432  // Reset group
433  gotGroup(QString());
434 
435  if (!m_oldFile.isEmpty()) {
436  // Close old file.
437  delete m_oldConfig1;
438  m_oldConfig1 = Q_NULLPTR;
439 
440  KConfigGroup cg(m_oldConfig2, "$Version");
441  QStringList ids = cg.readEntry("update_info", QStringList());
442  QString cfg_id = m_currentFilename + ':' + m_id;
443  if (!ids.contains(cfg_id) && !m_skip) {
444  ids.append(cfg_id);
445  cg.writeEntry("update_info", ids);
446  }
447  cg.sync();
448  delete m_oldConfig2;
449  m_oldConfig2 = Q_NULLPTR;
450 
451  QString file = QStandardPaths::writableLocation(QStandardPaths::GenericConfigLocation) + QLatin1Char('/') + m_oldFile;
452  QFileInfo info(file);
453  if (info.exists() && info.size() == 0) {
454  // Delete empty file.
455  QFile::remove(file);
456  }
457 
458  m_oldFile.clear();
459  }
460  if (!m_newFile.isEmpty()) {
461  // Close new file.
462  KConfigGroup cg(m_newConfig, "$Version");
463  QStringList ids = cg.readEntry("update_info", QStringList());
464  QString cfg_id = m_currentFilename + ':' + m_id;
465  if (!ids.contains(cfg_id) && !m_skip) {
466  ids.append(cfg_id);
467  cg.writeEntry("update_info", ids);
468  }
469  m_newConfig->sync();
470  delete m_newConfig;
471  m_newConfig = Q_NULLPTR;
472 
473  m_newFile.clear();
474  }
475  m_newConfig = Q_NULLPTR;
476 
477  int i = _file.indexOf(',');
478  if (i == -1) {
479  m_oldFile = _file.trimmed();
480  } else {
481  m_oldFile = _file.left(i).trimmed();
482  m_newFile = _file.mid(i + 1).trimmed();
483  if (m_oldFile == m_newFile) {
484  m_newFile.clear();
485  }
486  }
487 
488  if (!m_oldFile.isEmpty()) {
489  m_oldConfig2 = new KConfig(m_oldFile, KConfig::NoGlobals);
490  QString cfg_id = m_currentFilename + ':' + m_id;
491  KConfigGroup cg(m_oldConfig2, "$Version");
492  QStringList ids = cg.readEntry("update_info", QStringList());
493  if (ids.contains(cfg_id)) {
494  m_skip = true;
495  m_newFile.clear();
496  log() << m_currentFilename << ": Skipping update '" << m_id << "'" << endl;
497  }
498 
499  if (!m_newFile.isEmpty()) {
500  m_newConfig = new KConfig(m_newFile, KConfig::NoGlobals);
501  KConfigGroup cg(m_newConfig, "$Version");
502  ids = cg.readEntry("update_info", QStringList());
503  if (ids.contains(cfg_id)) {
504  m_skip = true;
505  log() << m_currentFilename << ": Skipping update '" << m_id << "'" << endl;
506  }
507  } else {
508  m_newConfig = m_oldConfig2;
509  }
510 
511  m_oldConfig1 = new KConfig(m_oldFile, KConfig::NoGlobals);
512  } else {
513  m_newFile.clear();
514  }
515  m_newFileName = m_newFile;
516  if (m_newFileName.isEmpty()) {
517  m_newFileName = m_oldFile;
518  }
519 
520  m_skipFile = false;
521  if (!m_oldFile.isEmpty()) { // if File= is specified, it doesn't exist, is empty or contains only kconf_update's [$Version] group, skip
522  if (m_oldConfig1 != Q_NULLPTR
523  && (m_oldConfig1->groupList().isEmpty()
524  || (m_oldConfig1->groupList().count() == 1 && m_oldConfig1->groupList().at(0) == QLatin1String("$Version")))) {
525  log() << m_currentFilename << ": File '" << m_oldFile << "' does not exist or empty, skipping" << endl;
526  m_skipFile = true;
527  }
528  }
529 }

600 void KonfUpdate::copyOrMoveKey(const QStringList &srcGroupPath, const QString &srcKey, const QStringList &dstGroupPath, const QString &dstKey)
601 {
602  KConfigGroup dstCg = KConfigUtils::openGroup(m_newConfig, dstGroupPath);
603  if (!m_bOverwrite && dstCg.hasKey(dstKey)) {
604  log() << m_currentFilename << ": Skipping " << m_newFileName << ":" << dstCg.name() << ":" << dstKey << ", already exists." << endl;
605  return;
606  }
607 
608  KConfigGroup srcCg = KConfigUtils::openGroup(m_oldConfig1, srcGroupPath);
609  if (!srcCg.hasKey(srcKey)) {
610  return;
611  }
612  QString value = srcCg.readEntry(srcKey, QString());
613  log() << m_currentFilename << ": Updating " << m_newFileName << ":" << dstCg.name() << ":" << dstKey << " to '" << value << "'" << endl;
614  dstCg.writeEntry(dstKey, value);
615 
616  if (m_bCopy) {
617  return; // Done.
618  }
619 
620  // Delete old entry
621  if (m_oldConfig2 == m_newConfig
622  && srcGroupPath == dstGroupPath
623  && srcKey == dstKey) {
624  return; // Don't delete!
625  }
626  KConfigGroup srcCg2 = KConfigUtils::openGroup(m_oldConfig2, srcGroupPath);
627  srcCg2.deleteEntry(srcKey);
628  log() << m_currentFilename << ": Removing " << m_oldFile << ":" << srcCg2.name() << ":" << srcKey << ", moved." << endl;
629 }

746 void KonfUpdate::gotScript(const QString &_script)
747 {
748  QString script, interpreter;
749  int i = _script.indexOf(',');
750  if (i == -1) {
751  script = _script.trimmed();
752  } else {
753  script = _script.left(i).trimmed();
754  interpreter = _script.mid(i + 1).trimmed();
755  }
756 
757  if (script.isEmpty()) {
758  logFileError() << "Script fails to specify filename";
759  m_skip = true;
760  return;
761  }
762 
763  QString path = QStandardPaths::locate(QStandardPaths::GenericDataLocation, QStringLiteral("kconf_update/") + script);
764  if (path.isEmpty()) {
765  if (interpreter.isEmpty()) {
766  path = CMAKE_INSTALL_PREFIX "/" LIB_INSTALL_DIR "/kconf_update_bin/" + script;
767  if (!QFile::exists(path)) {
768  path = QStandardPaths::findExecutable(script);
769  }
770  }
771 
772  if (path.isEmpty()) {
773  logFileError() << "Script '" << script << "' not found" << endl;
774  m_skip = true;
775  return;
776  }
777  }
778 
779  if (!m_arguments.isNull()) {
780  log() << m_currentFilename << ": Running script '" << script << "' with arguments '" << m_arguments << "'" << endl;
781  } else {
782  log() << m_currentFilename << ": Running script '" << script << "'" << endl;
783  }
784 
785  QStringList args;
786  QString cmd;
787  if (interpreter.isEmpty()) {
788  cmd = path;
789  } else {
790  QString interpreterPath = QStandardPaths::findExecutable(interpreter);
791  if (interpreterPath.isEmpty()) {
792  logFileError() << "Cannot find interpreter '" << interpreter << "'." << endl;
793  m_skip = true;
794  return;
795  }
796  cmd = interpreterPath;
797  args << path;
798  }
799 
800  if (!m_arguments.isNull()) {
801  args += m_arguments;
802  }
803 
804  QTemporaryFile scriptIn;
805  scriptIn.open();
806  QTemporaryFile scriptOut;
807  scriptOut.open();
808 
809  int result;
810  QProcess proc;
811  proc.setProcessChannelMode(QProcess::SeparateChannels);
812  proc.setStandardInputFile(scriptIn.fileName());
813  proc.setStandardOutputFile(scriptOut.fileName());
814  if (m_oldConfig1) {
815  if (m_debug) {
816  scriptIn.setAutoRemove(false);
817  log() << "Script input stored in " << scriptIn.fileName() << endl;
818  }
819  KConfig cfg(scriptIn.fileName(), KConfig::SimpleConfig);
820 
821  if (m_oldGroup.isEmpty()) {
822  // Write all entries to tmpFile;
823  const QStringList grpList = m_oldConfig1->groupList();
824  for (QStringList::ConstIterator it = grpList.begin();
825  it != grpList.end();
826  ++it) {
827  copyGroup(m_oldConfig1, *it, &cfg, *it);
828  }
829  } else {
830  KConfigGroup cg1 = KConfigUtils::openGroup(m_oldConfig1, m_oldGroup);
831  KConfigGroup cg2(&cfg, QString());
832  copyGroup(cg1, cg2);
833  }
834  cfg.sync();
835  }
836  if (m_debug) {
837  log() << "About to run " << cmd << endl;
838  QFile scriptFile(path);
839  if (scriptFile.open(QIODevice::ReadOnly)) {
840  log() << "Script contents is:" << endl << scriptFile.readAll() << endl;
841  }
842  }
843  proc.start(cmd, args);
844  if (!proc.waitForFinished(60000)) {
845  logFileError() << "update script did not terminate within 60 seconds: " << cmd << endl;
846  m_skip = true;
847  return;
848  }
849  result = proc.exitCode();
850 
851  // Copy script stderr to log file
852  {
853  QTextStream ts(proc.readAllStandardError());
854  ts.setCodec(QTextCodec::codecForName("UTF-8"));
855  while (!ts.atEnd()) {
856  QString line = ts.readLine();
857  log() << "[Script] " << line << endl;
858  }
859  }
860  proc.close();
861 
862  if (result) {
863  log() << m_currentFilename << ": !! An error occurred while running '" << cmd << "'" << endl;
864  return;
865  }
866 
867  if (m_debug) {
868  log() << "Successfully ran " << cmd << endl;
869  }
870 
871  if (!m_oldConfig1) {
872  return; // Nothing to merge
873  }
874 
875  if (m_debug) {
876  scriptOut.setAutoRemove(false);
877  log() << "Script output stored in " << scriptOut.fileName() << endl;
878  QFile output(scriptOut.fileName());
879  if (output.open(QIODevice::ReadOnly)) {
880  log() << "Script output is:" << endl << output.readAll() << endl;
881  }
882  }
883 
884  // Deleting old entries
885  {
886  QStringList group = m_oldGroup;
887  QFile output(scriptOut.fileName());
888  if (output.open(QIODevice::ReadOnly)) {
889  QTextStream ts(&output);
890  ts.setCodec(QTextCodec::codecForName("UTF-8"));
891  while (!ts.atEnd()) {
892  QString line = ts.readLine();
893  if (line.startsWith('[')) {
894  group = parseGroupString(line);
895  } else if (line.startsWith(QLatin1String("# DELETE "))) {
896  QString key = line.mid(9);
897  if (key[0] == '[') {
898  int j = key.lastIndexOf(']') + 1;
899  if (j > 0) {
900  group = parseGroupString(key.left(j));
901  key = key.mid(j);
902  }
903  }
904  KConfigGroup cg = KConfigUtils::openGroup(m_oldConfig2, group);
905  cg.deleteEntry(key);
906  log() << m_currentFilename << ": Script removes " << m_oldFile << ":" << group << ":" << key << endl;
907  /*if (m_oldConfig2->deleteGroup(group, KConfig::Normal)) { // Delete group if empty.
908  log() << m_currentFilename << ": Removing empty group " << m_oldFile << ":" << group << endl;
909  } (this should be automatic)*/
910  } else if (line.startsWith(QLatin1String("# DELETEGROUP"))) {
911  QString str = line.mid(13).trimmed();
912  if (!str.isEmpty()) {
913  group = parseGroupString(str);
914  }
915  KConfigGroup cg = KConfigUtils::openGroup(m_oldConfig2, group);
916  cg.deleteGroup();
917  log() << m_currentFilename << ": Script removes group " << m_oldFile << ":" << group << endl;
918  }
919  }
920  }
921  }
922 
923  // Merging in new entries.
924  KConfig scriptOutConfig(scriptOut.fileName(), KConfig::NoGlobals);
925  if (m_newGroup.isEmpty()) {
926  // Copy "default" keys as members of "default" keys
927  copyGroup(&scriptOutConfig, QString(), m_newConfig, QString());
928  } else {
929  // Copy default keys as members of m_newGroup
930  KConfigGroup srcCg = KConfigUtils::openGroup(&scriptOutConfig, QStringList());
931  KConfigGroup dstCg = KConfigUtils::openGroup(m_newConfig, m_newGroup);
932  copyGroup(srcCg, dstCg);
933  }
934  Q_FOREACH (const QString &group, scriptOutConfig.groupList()) {
935  copyGroup(&scriptOutConfig, group, m_newConfig, group);
936  }
937 }
