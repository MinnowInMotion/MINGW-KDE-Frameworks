126 QByteArray KConfigGroupPrivate::serializeList(const QList<QByteArray> &list)
127 {
128  QByteArray value;
129 
130  if (!list.isEmpty()) {
131  QList<QByteArray>::ConstIterator it = list.constBegin();
132  const QList<QByteArray>::ConstIterator end = list.constEnd();
133 
134  value = QByteArray(*it).replace('\\', QByteArrayLiteral("\\\\")).replace(',', QByteArrayLiteral("\\,"));
135 
136  while (++it != end) {
137  // In the loop, so it is not done when there is only one element.
138  // Doing it repeatedly is a pretty cheap operation.
139  value.reserve(4096);
140 
141  value += ',';
142  value += QByteArray(*it).replace('\\', QByteArrayLiteral("\\\\")).replace(',', QByteArrayLiteral("\\,"));
143  }
144 
145  // To be able to distinguish an empty list from a list with one empty element.
146  if (value.isEmpty()) {
147  value = QByteArrayLiteral("\\0");
148  }
149  }
150 
151  return value;
152 }

154 QStringList KConfigGroupPrivate::deserializeList(const QString &data)
155 {
156  if (data.isEmpty()) {
157  return QStringList();
158  }
159  if (data == QLatin1String("\\0")) {
160  return QStringList(QString());
161  }
162  QStringList value;
163  QString val;
164  val.reserve(data.size());
165  bool quoted = false;
166  for (int p = 0; p < data.length(); p++) {
167  if (quoted) {
168  val += data[p];
169  quoted = false;
170  } else if (data[p].unicode() == '\\') {
171  quoted = true;
172  } else if (data[p].unicode() == ',') {
173  val.squeeze(); // release any unused memory
174  value.append(val);
175  val.clear();
176  val.reserve(data.size() - p);
177  } else {
178  val += data[p];
179  }
180  }
181  value.append(val);
182  return value;
183 }

216 QVariant KConfigGroup::convertToQVariant(const char *pKey, const QByteArray &value, const QVariant &aDefault)
217 {
218  // if a type handler is added here you must add a QVConversions definition
219  // to conversioncheck.h, or ConversionCheck::to_QVariant will not allow
220  // readEntry<T> to convert to QVariant.
221  switch (static_cast<QMetaType::Type>(aDefault.type())) {
222  case QMetaType::UnknownType:
223  return QVariant();
224  case QMetaType::QString:
225  // this should return the raw string not the dollar expanded string.
226  // imho if processed string is wanted should call
227  // readEntry(key, QString) not readEntry(key, QVariant)
228  return QString::fromUtf8(value);
229  case QMetaType::QVariantList:
230  case QMetaType::QStringList:
231  return KConfigGroupPrivate::deserializeList(QString::fromUtf8(value));
232  case QMetaType::QByteArray:
233  return value;
234  case QMetaType::Bool: {
235  const QByteArray lower(value.toLower());
236  if (lower == "false" || lower == "no" || lower == "off" || lower == "0") {
237  return false;
238  }
239  return true;
240  }
241  case QMetaType::Double:
242  case QMetaType::Float:
243  case QMetaType::Int:
244  case QMetaType::UInt:
245  case QMetaType::LongLong:
246  case QMetaType::ULongLong: {
247  QVariant tmp = value;
248  if (!tmp.convert(aDefault.type())) {
249  tmp = aDefault;
250  }
251  return tmp;
252  }
253  case QMetaType::QPoint: {
254  const QList<int> list = asIntList(value);
255 
256  if (list.count() != 2) {
257  qWarning() << errString(pKey, value, aDefault)
258  << formatError(2, list.count());
259  return aDefault;
260  }
261  return QPoint(list.at(0), list.at(1));
262  }
263  case QMetaType::QPointF: {
264  const QList<qreal> list = asRealList(value);
265 
266  if (list.count() != 2) {
267  qWarning() << errString(pKey, value, aDefault)
268  << formatError(2, list.count());
269  return aDefault;
270  }
271  return QPointF(list.at(0), list.at(1));
272  }
273  case QMetaType::QRect: {
274  const QList<int> list = asIntList(value);
275 
276  if (list.count() != 4) {
277  qWarning() << errString(pKey, value, aDefault)
278  << formatError(4, list.count());
279  return aDefault;
280  }
281  const QRect rect(list.at(0), list.at(1), list.at(2), list.at(3));
282  if (!rect.isValid()) {
283  qWarning() << errString(pKey, value, aDefault);
284  return aDefault;
285  }
286  return rect;
287  }
288  case QMetaType::QRectF: {
289  const QList<qreal> list = asRealList(value);
290 
291  if (list.count() != 4) {
292  qWarning() << errString(pKey, value, aDefault)
293  << formatError(4, list.count());
294  return aDefault;
295  }
296  const QRectF rect(list.at(0), list.at(1), list.at(2), list.at(3));
297  if (!rect.isValid()) {
298  qWarning() << errString(pKey, value, aDefault);
299  return aDefault;
300  }
301  return rect;
302  }
303  case QMetaType::QSize: {
304  const QList<int> list = asIntList(value);
305 
306  if (list.count() != 2) {
307  qWarning() << errString(pKey, value, aDefault)
308  << formatError(2, list.count());
309  return aDefault;
310  }
311  const QSize size(list.at(0), list.at(1));
312  if (!size.isValid()) {
313  qWarning() << errString(pKey, value, aDefault);
314  return aDefault;
315  }
316  return size;
317  }
318  case QMetaType::QSizeF: {
319  const QList<qreal> list = asRealList(value);
320 
321  if (list.count() != 2) {
322  qWarning() << errString(pKey, value, aDefault)
323  << formatError(2, list.count());
324  return aDefault;
325  }
326  const QSizeF size(list.at(0), list.at(1));
327  if (!size.isValid()) {
328  qWarning() << errString(pKey, value, aDefault);
329  return aDefault;
330  }
331  return size;
332  }
333  case QMetaType::QDateTime: {
334  const QList<int> list = asIntList(value);
335  if (list.count() != 6) {
336  qWarning() << errString(pKey, value, aDefault)
337  << formatError(6, list.count());
338  return aDefault;
339  }
340  const QDate date(list.at(0), list.at(1), list.at(2));
341  const QTime time(list.at(3), list.at(4), list.at(5));
342  const QDateTime dt(date, time);
343  if (!dt.isValid()) {
344  qWarning() << errString(pKey, value, aDefault);
345  return aDefault;
346  }
347  return dt;
348  }
349  case QMetaType::QDate: {
350  QList<int> list = asIntList(value);
351  if (list.count() == 6) {
352  list = list.mid(0, 3); // don't break config files that stored QDate as QDateTime
353  }
354  if (list.count() != 3) {
355  qWarning() << errString(pKey, value, aDefault)
356  << formatError(3, list.count());
357  return aDefault;
358  }
359  const QDate date(list.at(0), list.at(1), list.at(2));
360  if (!date.isValid()) {
361  qWarning() << errString(pKey, value, aDefault);
362  return aDefault;
363  }
364  return date;
365  }
366  case QMetaType::QColor:
367  case QMetaType::QFont:
368  qWarning() << "KConfigGroup::readEntry was passed GUI type '"
369  << aDefault.typeName()
370  << "' but kdeui isn't linked! If it is linked to your program, "
371  "this is a platform bug. Please inform the KDE developers";
372  break;
373  case QMetaType::QUrl:
374  return QUrl(QString::fromUtf8(value));
375 
376  default:
377  break;
378  }
379 
380  qWarning() << "unhandled type " << aDefault.typeName();
381  return QVariant();
382 }

388 static bool cleanHomeDirPath(QString &path, const QString &homeDir)
389 {
390 #ifdef Q_OS_WIN //safer
391  if (!QDir::toNativeSeparators(path).startsWith(QDir::toNativeSeparators(homeDir))) {
392  return false;
393  }
394 #else
395  if (!path.startsWith(homeDir)) {
396  return false;
397  }
398 #endif
399 
400  int len = homeDir.length();
401  // replace by "$HOME" if possible
402  if (len && (path.length() == len || path[len] == QLatin1Char('/'))) {
403  path.replace(0, len, QStringLiteral("$HOME"));
404  return true;
405  }
406 
407  return false;
408 }

410 static QString translatePath(QString path) // krazy:exclude=passbyvalue
411 {
412  if (path.isEmpty()) {
413  return path;
414  }
415 
416  // only "our" $HOME should be interpreted
417  path.replace(QLatin1Char('$'), QLatin1String("$$"));
418 
419  const bool startsWithFile = path.startsWith(QLatin1String("file:"), Qt::CaseInsensitive);
420  path = startsWithFile ? QUrl(path).toLocalFile() : path;
421 
422  if (QDir::isRelativePath(path)) {
423  return path;
424  }
425 
426  // we can not use KGlobal::dirs()->relativeLocation("home", path) here,
427  // since it would not recognize paths without a trailing '/'.
428  // All of the 3 following functions to return the user's home directory
429  // can return different paths. We have to test all them.
430  const QString homeDir0 = QFile::decodeName(qgetenv("HOME"));
431  const QString homeDir1 = QDir::homePath();
432  const QString homeDir2 = QDir(homeDir1).canonicalPath();
433  if (cleanHomeDirPath(path, homeDir0) ||
434  cleanHomeDirPath(path, homeDir1) ||
435  cleanHomeDirPath(path, homeDir2)) {
436  // qDebug() << "Path was replaced\n";
437  }
438 
439  if (startsWithFile) {
440  path = QUrl::fromLocalFile(path).toString();
441  }
442 
443  return path;
444 }
