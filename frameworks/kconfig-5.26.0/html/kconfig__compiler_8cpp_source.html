652 static void preProcessDefault(QString &defaultValue, const QString &name,
653  const QString &type,
654  const CfgEntry::Choices &choices,
655  QString &code, const CfgConfig &cfg)
656 {
657  if (type == QLatin1String("String") && !defaultValue.isEmpty()) {
658  defaultValue = literalString(defaultValue);
659 
660  } else if (type == QLatin1String("Path") && !defaultValue.isEmpty()) {
661  defaultValue = literalString(defaultValue);
662  } else if (type == QLatin1String("Url") && !defaultValue.isEmpty()) {
663  // Use fromUserInput in order to support absolute paths and absolute urls, like KDE4's KUrl(QString) did.
664  defaultValue = QLatin1String("QUrl::fromUserInput( ") + literalString(defaultValue) + QLatin1Char(')');
665  } else if ((type == QLatin1String("UrlList") || type == QLatin1String("StringList") || type == QLatin1String("PathList")) && !defaultValue.isEmpty()) {
666  QTextStream cpp(&code, QIODevice::WriteOnly | QIODevice::Append);
667  if (!code.isEmpty()) {
668  cpp << endl;
669  }
670 
671  if (type == QLatin1String("UrlList")) {
672  cpp << " QList<QUrl> default" << name << ";" << endl;
673  } else {
674  cpp << " QStringList default" << name << ";" << endl;
675  }
676  const QStringList defaults = defaultValue.split(QLatin1Char(','));
677  QStringList::ConstIterator it;
678  for (it = defaults.constBegin(); it != defaults.constEnd(); ++it) {
679  cpp << " default" << name << ".append( ";
680  if (type == QLatin1String("UrlList")) {
681  cpp << "QUrl::fromUserInput(";
682  }
683  cpp << "QString::fromUtf8( \"" << *it << "\" ) ";
684  if (type == QLatin1String("UrlList")) {
685  cpp << ") ";
686  }
687  cpp << ");" << endl;
688  }
689  defaultValue = QLatin1String("default") + name;
690 
691  } else if (type == QLatin1String("Color") && !defaultValue.isEmpty()) {
692  QRegExp colorRe(QStringLiteral("\\d+,\\s*\\d+,\\s*\\d+(,\\s*\\d+)?"));
693  if (colorRe.exactMatch(defaultValue)) {
694  defaultValue = QLatin1String("QColor( ") + defaultValue + QLatin1String(" )");
695  } else {
696  defaultValue = QLatin1String("QColor( \"") + defaultValue + QLatin1String("\" )");
697  }
698 
699  } else if (type == QLatin1String("Enum")) {
700  QList<CfgEntry::Choice>::ConstIterator it;
701  for (it = choices.choices.constBegin(); it != choices.choices.constEnd(); ++it) {
702  if ((*it).name == defaultValue) {
703  if (cfg.globalEnums && choices.name().isEmpty()) {
704  defaultValue.prepend(choices.prefix);
705  } else {
706  defaultValue.prepend(enumTypeQualifier(name, choices) + choices.prefix);
707  }
708  break;
709  }
710  }
711 
712  } else if (type == QLatin1String("IntList")) {
713  QTextStream cpp(&code, QIODevice::WriteOnly | QIODevice::Append);
714  if (!code.isEmpty()) {
715  cpp << endl;
716  }
717 
718  cpp << " QList<int> default" << name << ";" << endl;
719  if (!defaultValue.isEmpty()) {
720  const QStringList defaults = defaultValue.split(QLatin1Char(','));
721  QStringList::ConstIterator it;
722  for (it = defaults.constBegin(); it != defaults.constEnd(); ++it) {
723  cpp << " default" << name << ".append( " << *it << " );"
724  << endl;
725  }
726  }
727  defaultValue = QLatin1String("default") + name;
728  }
