56 bool ConfigLoaderHandler::startElement(const QString &namespaceURI, const QString &localName,
57  const QString &qName, const QXmlAttributes &attrs)
58 {
59  Q_UNUSED(namespaceURI)
60  Q_UNUSED(qName)
61 
62 // qDebug() << "ConfigLoaderHandler::startElement(" << localName << qName;
63  int numAttrs = attrs.count();
64  QString tag = localName.toLower();
65  if (tag == QStringLiteral("group")) {
66  QString group;
67  for (int i = 0; i < numAttrs; ++i) {
68  QString name = attrs.localName(i).toLower();
69  if (name == QStringLiteral("name")) {
70  //qDebug() << "set group to" << attrs.value(i);
71  group = attrs.value(i);
72  }
73  }
74  if (group.isEmpty()) {
75  group = d->baseGroup;
76  } else {
77  d->groups.append(group);
78  if (!d->baseGroup.isEmpty()) {
79  group = d->baseGroup + QLatin1Char('\x1d') + group;
80  }
81  }
82 
83  if (m_config) {
84  m_config->setCurrentGroup(group);
85  }
86  } else if (tag == QStringLiteral("entry")) {
87  for (int i = 0; i < numAttrs; ++i) {
88  QString name = attrs.localName(i).toLower();
89  if (name == QStringLiteral("name")) {
90  m_name = attrs.value(i).trimmed();
91  } else if (name == QStringLiteral("type")) {
92  m_type = attrs.value(i).toLower();
93  } else if (name == QStringLiteral("key")) {
94  m_key = attrs.value(i).trimmed();
95  }
96  }
97  } else if (tag == QStringLiteral("choice")) {
98  m_choice.name.clear();
99  m_choice.label.clear();
100  m_choice.whatsThis.clear();
101  for (int i = 0; i < numAttrs; ++i) {
102  QString name = attrs.localName(i).toLower();
103  if (name == QStringLiteral("name")) {
104  m_choice.name = attrs.value(i);
105  }
106  }
107  m_inChoice = true;
108  }
109 
110  return true;
111 }

149 bool ConfigLoaderHandler::endElement(const QString &namespaceURI,
150  const QString &localName, const QString &qName)
151 {
152  Q_UNUSED(namespaceURI)
153  Q_UNUSED(qName)
154 
155 // qDebug() << "ConfigLoaderHandler::endElement(" << localName << qName;
156  const QString tag = localName.toLower();
157  if (tag == QStringLiteral("entry")) {
158  addItem();
159  resetState();
160  } else if (tag == QStringLiteral("label")) {
161  if (m_inChoice) {
162  m_choice.label = m_cdata.trimmed();
163  } else {
164  m_label = m_cdata.trimmed();
165  }
166  } else if (tag == QStringLiteral("whatsthis")) {
167  if (m_inChoice) {
168  m_choice.whatsThis = m_cdata.trimmed();
169  } else {
170  m_whatsThis = m_cdata.trimmed();
171  }
172  } else if (tag == QStringLiteral("default")) {
173  m_default = m_cdata.trimmed();
174  } else if (tag == QStringLiteral("min")) {
175  m_min = m_cdata.toInt(&m_haveMin);
176  } else if (tag == QStringLiteral("max")) {
177  m_max = m_cdata.toInt(&m_haveMax);
178  } else if (tag == QStringLiteral("choice")) {
179  m_enumChoices.append(m_choice);
180  m_inChoice = false;
181  }
182 
183  m_cdata.clear();
184  return true;
185 }

187 void ConfigLoaderHandler::addItem()
188 {
189  if (m_name.isEmpty()) {
190  if (m_key.isEmpty()) {
191  return;
192  }
193 
194  m_name = m_key;
195  }
196 
197  m_name.remove(QLatin1Char(' '));
198 
199  KConfigSkeletonItem *item = Q_NULLPTR;
200 
201  if (m_type == QStringLiteral("bool")) {
202  bool defaultValue = m_default.toLower() == QStringLiteral("true");
203  item = m_config->addItemBool(m_name, *d->newBool(), defaultValue, m_key);
204  } else if (m_type == QStringLiteral("color")) {
205  item = m_config->addItemColor(m_name, *d->newColor(), QColor(m_default), m_key);
206  } else if (m_type == QStringLiteral("datetime")) {
207  item = m_config->addItemDateTime(m_name, *d->newDateTime(),
208  QDateTime::fromString(m_default), m_key);
209  } else if (m_type == QStringLiteral("enum")) {
210  m_key = (m_key.isEmpty()) ? m_name : m_key;
211  KConfigSkeleton::ItemEnum *enumItem =
212  new KConfigSkeleton::ItemEnum(m_config->currentGroup(),
213  m_key, *d->newInt(),
214  m_enumChoices,
215  m_default.toUInt());
216  m_config->addItem(enumItem, m_name);
217  item = enumItem;
218  } else if (m_type == QStringLiteral("font")) {
219  item = m_config->addItemFont(m_name, *d->newFont(), QFont(m_default), m_key);
220  } else if (m_type == QStringLiteral("int")) {
221  KConfigSkeleton::ItemInt *intItem = m_config->addItemInt(m_name, *d->newInt(),
222  m_default.toInt(), m_key);
223 
224  if (m_haveMin) {
225  intItem->setMinValue(m_min);
226  }
227 
228  if (m_haveMax) {
229  intItem->setMaxValue(m_max);
230  }
231 
232  item = intItem;
233  } else if (m_type == QStringLiteral("password")) {
234  item = m_config->addItemPassword(m_name, *d->newString(), m_default, m_key);
235  } else if (m_type == QStringLiteral("path")) {
236  item = m_config->addItemPath(m_name, *d->newString(), m_default, m_key);
237  } else if (m_type == QStringLiteral("string")) {
238  item = m_config->addItemString(m_name, *d->newString(), m_default, m_key);
239  } else if (m_type == QStringLiteral("stringlist")) {
240  //FIXME: the split() is naive and will break on lists with ,'s in them
241  //empty parts are not wanted in this case
242  item = m_config->addItemStringList(m_name, *d->newStringList(),
243  m_default.split(QLatin1Char(','), QString::SkipEmptyParts), m_key);
244  } else if (m_type == QStringLiteral("uint")) {
245  KConfigSkeleton::ItemUInt *uintItem =
246  m_config->addItemUInt(m_name, *d->newUint(), m_default.toUInt(), m_key);
247  if (m_haveMin) {
248  uintItem->setMinValue(m_min);
249  }
250  if (m_haveMax) {
251  uintItem->setMaxValue(m_max);
252  }
253  item = uintItem;
254  } else if (m_type == QStringLiteral("url")) {
255  m_key = (m_key.isEmpty()) ? m_name : m_key;
256  KConfigSkeleton::ItemUrl *urlItem =
257  new KConfigSkeleton::ItemUrl(m_config->currentGroup(),
258  m_key, *d->newUrl(),
259  QUrl::fromUserInput(m_default));
260  m_config->addItem(urlItem, m_name);
261  item = urlItem;
262  } else if (m_type == QStringLiteral("double")) {
263  KConfigSkeleton::ItemDouble *doubleItem = m_config->addItemDouble(m_name,
264  *d->newDouble(), m_default.toDouble(), m_key);
265  if (m_haveMin) {
266  doubleItem->setMinValue(m_min);
267  }
268  if (m_haveMax) {
269  doubleItem->setMaxValue(m_max);
270  }
271  item = doubleItem;
272  } else if (m_type == QStringLiteral("intlist")) {
273  QStringList tmpList = m_default.split(QLatin1Char(','));
274  QList<int> defaultList;
275  foreach (const QString &tmp, tmpList) {
276  defaultList.append(tmp.toInt());
277  }
278  item = m_config->addItemIntList(m_name, *d->newIntList(), defaultList, m_key);
279  } else if (m_type == QStringLiteral("longlong")) {
280  KConfigSkeleton::ItemLongLong *longlongItem = m_config->addItemLongLong(m_name,
281  *d->newLongLong(), m_default.toLongLong(), m_key);
282  if (m_haveMin) {
283  longlongItem->setMinValue(m_min);
284  }
285  if (m_haveMax) {
286  longlongItem->setMaxValue(m_max);
287  }
288  item = longlongItem;
289  /* No addItemPathList in KConfigSkeleton ?
290  } else if (m_type == "PathList") {
291  //FIXME: the split() is naive and will break on lists with ,'s in them
292  item = m_config->addItemPathList(m_name, *d->newStringList(), m_default.split(","), m_key);
293  */
294  } else if (m_type == QStringLiteral("point")) {
295  QPoint defaultPoint;
296  QStringList tmpList = m_default.split(QLatin1Char(','));
297  if (tmpList.size() >= 2) {
298  defaultPoint.setX(tmpList[0].toInt());
299  defaultPoint.setY(tmpList[1].toInt());
300  }
301  item = m_config->addItemPoint(m_name, *d->newPoint(), defaultPoint, m_key);
302  } else if (m_type == QStringLiteral("rect")) {
303  QRect defaultRect;
304  QStringList tmpList = m_default.split(QLatin1Char(','));
305  if (tmpList.size() >= 4) {
306  defaultRect.setCoords(tmpList[0].toInt(), tmpList[1].toInt(),
307  tmpList[2].toInt(), tmpList[3].toInt());
308  }
309  item = m_config->addItemRect(m_name, *d->newRect(), defaultRect, m_key);
310  } else if (m_type == QStringLiteral("size")) {
311  QSize defaultSize;
312  QStringList tmpList = m_default.split(QLatin1Char(','));
313  if (tmpList.size() >= 2) {
314  defaultSize.setWidth(tmpList[0].toInt());
315  defaultSize.setHeight(tmpList[1].toInt());
316  }
317  item = m_config->addItemSize(m_name, *d->newSize(), defaultSize, m_key);
318  } else if (m_type == QStringLiteral("ulonglong")) {
319  KConfigSkeleton::ItemULongLong *ulonglongItem =
320  m_config->addItemULongLong(m_name, *d->newULongLong(), m_default.toULongLong(), m_key);
321  if (m_haveMin) {
322  ulonglongItem->setMinValue(m_min);
323  }
324  if (m_haveMax) {
325  ulonglongItem->setMaxValue(m_max);
326  }
327  item = ulonglongItem;
328  /* No addItemUrlList in KConfigSkeleton ?
329  } else if (m_type == "urllist") {
330  //FIXME: the split() is naive and will break on lists with ,'s in them
331  QStringList tmpList = m_default.split(",");
332  QList<QUrl> defaultList;
333  foreach (const QString& tmp, tmpList) {
334  defaultList.append(QUrl(tmp));
335  }
336  item = m_config->addItemUrlList(m_name, *d->newUrlList(), defaultList, m_key);*/
337  }
338 
339  if (item) {
340  item->setLabel(m_label);
341  item->setWhatsThis(m_whatsThis);
342  d->keysToNames.insert(item->group() + item->key(), item->name());
343  }
344 }

379 KConfigLoader::KConfigLoader(const KConfigGroup &config, QIODevice *xml, QObject *parent)
380  : KConfigSkeleton(KSharedConfig::openConfig(config.config()->name(), config.config()->openFlags()), parent),
381  d(new ConfigLoaderPrivate)

383  KConfigGroup group = config.parent();

387  group = group.parent();
388  }
