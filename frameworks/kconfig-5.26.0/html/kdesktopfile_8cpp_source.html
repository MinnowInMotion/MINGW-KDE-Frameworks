79 QString KDesktopFile::locateLocal(const QString &path)
80 {
81  QString relativePath;
82  QChar plus(QLatin1Char('/'));
83  // Relative to config? (e.g. for autostart)
84  Q_FOREACH (const QString &dir, QStandardPaths::standardLocations(QStandardPaths::GenericConfigLocation)) {
85  if (path.startsWith(dir + plus)) {
86  relativePath = dir.mid(path.length() + 1);
87  return QStandardPaths::writableLocation(QStandardPaths::GenericConfigLocation) + QLatin1Char('/') + relativePath;
88  }
89  }
90  // Relative to xdg data dir? (much more common)
91  Q_FOREACH (const QString &dir, QStandardPaths::standardLocations(QStandardPaths::GenericDataLocation)) {
92  if (path.startsWith(dir + plus)) {
93  relativePath = dir.mid(path.length() + 1);
94  }
95  }
96  if (relativePath.isEmpty()) {
97  // What now? The desktop file doesn't come from XDG_DATA_DIRS. Use filename only and hope for the best.
98  relativePath = path.mid(path.lastIndexOf(QLatin1Char('/')) + 1);
99  }
100  return QStandardPaths::writableLocation(QStandardPaths::GenericDataLocation) + QLatin1Char('/') + relativePath;
101 }

108 bool KDesktopFile::isAuthorizedDesktopFile(const QString &path)
109 {
110  if (path.isEmpty()) {
111  return false; // Empty paths are not ok.
112  }
113 
114  if (QDir::isRelativePath(path)) {
115  return true; // Relative paths are ok.
116  }
117 
118  const QString realPath = QFileInfo(path).canonicalFilePath();
119  if (realPath.isEmpty()) {
120  return false; // File doesn't exist.
121  }
122 
123 #ifndef Q_OS_WIN
124  const Qt::CaseSensitivity sensitivity = Qt::CaseSensitive;
125 #else
126  const Qt::CaseSensitivity sensitivity = Qt::CaseInsensitive;
127 #endif
128 
129  // Check if the .desktop file is installed as part of KDE or XDG.
130  const QStringList appsDirs = QStandardPaths::standardLocations(QStandardPaths::ApplicationsLocation);
131  Q_FOREACH (const QString &prefix, appsDirs) {
132  if (QDir(prefix).exists() && realPath.startsWith(QFileInfo(prefix).canonicalFilePath(), sensitivity)) {
133  return true;
134  }
135  }
136  const QString servicesDir = QStringLiteral("kservices5/"); // KGlobal::dirs()->xdgDataRelativePath("services")
137  Q_FOREACH (const QString &xdgDataPrefix, QStandardPaths::standardLocations(QStandardPaths::GenericDataLocation)) {
138  if (QDir(xdgDataPrefix).exists()) {
139  const QString prefix = QFileInfo(xdgDataPrefix).canonicalFilePath();
140  if (realPath.startsWith(prefix + QLatin1Char('/') + servicesDir, sensitivity)) {
141  return true;
142  }
143  }
144  }
145  const QString autostartDir = QStringLiteral("autostart/");
146  Q_FOREACH (const QString &xdgDataPrefix, QStandardPaths::standardLocations(QStandardPaths::GenericConfigLocation)) {
147  if (QDir(xdgDataPrefix).exists()) {
148  const QString prefix = QFileInfo(xdgDataPrefix).canonicalFilePath();
149  if (realPath.startsWith(prefix + QLatin1Char('/') + autostartDir, sensitivity)) {
150  return true;
151  }
152  }
153  }
154 
155  // Forbid desktop files outside of standard locations if kiosk is set so
156  if (!KAuthorized::authorize(QStringLiteral("run_desktop_files"))) {
157  qWarning() << "Access to '" << path << "' denied because of 'run_desktop_files' restriction." << endl;
158  return false;
159  }
160 
161  // Not otherwise permitted, so only allow if the file is executable, or if
162  // owned by root (uid == 0)
163  QFileInfo entryInfo(path);
164  if (entryInfo.isExecutable() || entryInfo.ownerId() == 0) {
165  return true;
166  }
167 
168  qWarning() << "Access to '" << path << "' denied, not owned by root, executable flag not set." << endl;
169  return false;
170 }

218 QString KDesktopFile::readUrl() const
219 {
220  Q_D(const KDesktopFile);
221  if (hasDeviceType()) {
222  return d->desktopGroup.readEntry("MountPoint", QString());
223  } else {
224  // NOT readPathEntry (see readPath())
225  QString url = d->desktopGroup.readEntry("URL", QString());
226  if (!url.isEmpty() && !QDir::isRelativePath(url)) {
227  // Handle absolute paths as such (i.e. we need to escape them)
228  return QUrl::fromLocalFile(url).toString();
229  }
230  return url;
231  }
232 }

276 bool KDesktopFile::tryExec() const
277 {
278  Q_D(const KDesktopFile);
279  // Test for TryExec and "X-KDE-AuthorizeAction"
280  // NOT readPathEntry (see readPath())
281  QString te = d->desktopGroup.readEntry("TryExec", QString());
282 
283  if (!te.isEmpty()) {
284  if (QStandardPaths::findExecutable(te).isEmpty()) {
285  return false;
286  }
287  }
288  const QStringList list = d->desktopGroup.readEntry("X-KDE-AuthorizeAction", QStringList());
289 
290  if (!list.isEmpty()) {
291  for (QStringList::ConstIterator it = list.begin();
292  it != list.end();
293  ++it) {
294  if (!KAuthorized::authorize((*it).trimmed())) {
295  return false;
296  }
297  }
298  }
299 
300  // See also KService::username()
301  bool su = d->desktopGroup.readEntry("X-KDE-SubstituteUID", false);
302  if (su) {
303  QString user = d->desktopGroup.readEntry("X-KDE-Username", QString());
304  if (user.isEmpty()) {
305  user = QString::fromLocal8Bit(qgetenv("ADMIN_ACCOUNT"));
306  }
307  if (user.isEmpty()) {
308  user = QStringLiteral("root");
309  }
310  if (!KAuthorized::authorize(QLatin1String("user/") + user)) {
311  return false;
312  }
313  }
314 
315  return true;
316 }

364 bool KDesktopFile::noDisplay() const
365 {
366  Q_D(const KDesktopFile);
367  if (d->desktopGroup.readEntry("NoDisplay", false)) {
368  return true;
369  }
370  if (d->desktopGroup.hasKey("OnlyShowIn")) {
371  if (!d->desktopGroup.readXdgListEntry("OnlyShowIn").contains(QStringLiteral("KDE"))) {
372  return true;
373  }
374  }
375  if (d->desktopGroup.hasKey("NotShowIn")) {
376  if (d->desktopGroup.readXdgListEntry("NotShowIn").contains(QStringLiteral("KDE"))) {
377  return true;
378  }
379  }
380  return false;
381 }
