79 // merging==true is the merging that happens at the beginning of writeConfig:
80 // merge changes in the on-disk file with the changes in the KConfig object.
81 KConfigBackend::ParseInfo
82 KConfigIniBackend::parseConfig(const QByteArray &currentLocale, KEntryMap &entryMap,
83  ParseOptions options, bool merging)
84 {
85  if (filePath().isEmpty() || !QFile::exists(filePath())) {
86  return ParseOk;
87  }
88 
89  const QByteArray currentLanguage = currentLocale.split('_').first();
90 
91  bool bDefault = options & ParseDefaults;
92  bool allowExecutableValues = options & ParseExpansions;
93 
94  QByteArray currentGroup("<default>");
95 
96  QFile file(filePath());
97  if (!file.open(QIODevice::ReadOnly | QIODevice::Text)) {
98  return ParseOpenError;
99  }
100 
101  QList<QByteArray> immutableGroups;
102 
103  bool fileOptionImmutable = false;
104  bool groupOptionImmutable = false;
105  bool groupSkip = false;
106 
107  int lineNo = 0;
108  // on systems using \r\n as end of line, \r will be taken care of by
109  // trim() below
110  QByteArray buffer = file.readAll();
111  BufferFragment contents(buffer.data(), buffer.size());
112  unsigned int len = contents.length();
113  unsigned int startOfLine = 0;
114 
115  // Reduce memory overhead by making use of implicit sharing
116  // This assumes that config files contain only a small amount of
117  // different fragments which are repeated often.
118  // This is often the case, especially sub groups will all have
119  // the same list of keys and similar values as well.
120  QHash<BufferFragment, QByteArray> cache;
121  cache.reserve(4096);
122 
123  while (startOfLine < len) {
124  BufferFragment line = contents.split('\n', &startOfLine);
125  line.trim();
126  lineNo++;
127 
128  // skip empty lines and lines beginning with '#'
129  if (line.isEmpty() || line.at(0) == '#') {
130  continue;
131  }
132 
133  if (line.at(0) == '[') { // found a group
134  groupOptionImmutable = fileOptionImmutable;
135 
136  QByteArray newGroup;
137  int start = 1, end;
138  do {
139  end = start;
140  for (;;) {
141  if (end == line.length()) {
142  qWarning() << warningProlog(file, lineNo) << "Invalid group header.";
143  // XXX maybe reset the current group here?
144  goto next_line;
145  }
146  if (line.at(end) == ']') {
147  break;
148  }
149  end++;
150  }
151  if (end + 1 == line.length() && start + 2 == end &&
152  line.at(start) == '$' && line.at(start + 1) == 'i') {
153  if (newGroup.isEmpty()) {
154  fileOptionImmutable = !kde_kiosk_exception;
155  } else {
156  groupOptionImmutable = !kde_kiosk_exception;
157  }
158  } else {
159  if (!newGroup.isEmpty()) {
160  newGroup += '\x1d';
161  }
162  BufferFragment namePart = line.mid(start, end - start);
163  printableToString(&namePart, file, lineNo);
164  newGroup += namePart.toByteArray();
165  }
166  } while ((start = end + 2) <= line.length() && line.at(end + 1) == '[');
167  currentGroup = newGroup;
168 
169  groupSkip = entryMap.getEntryOption(currentGroup, 0, 0, KEntryMap::EntryImmutable);
170 
171  if (groupSkip && !bDefault) {
172  continue;
173  }
174 
175  if (groupOptionImmutable)
176  // Do not make the groups immutable until the entries from
177  // this file have been added.
178  {
179  immutableGroups.append(currentGroup);
180  }
181  } else {
182  if (groupSkip && !bDefault) {
183  continue; // skip entry
184  }
185 
186  BufferFragment aKey;
187  int eqpos = line.indexOf('=');
188  if (eqpos < 0) {
189  aKey = line;
190  line.clear();
191  } else {
192  BufferFragment temp = line.left(eqpos);
193  temp.trim();
194  aKey = temp;
195  line.truncateLeft(eqpos + 1);
196  }
197  if (aKey.isEmpty()) {
198  qWarning() << warningProlog(file, lineNo) << "Invalid entry (empty key)";
199  continue;
200  }
201 
202  KEntryMap::EntryOptions entryOptions = 0;
203  if (groupOptionImmutable) {
204  entryOptions |= KEntryMap::EntryImmutable;
205  }
206 
207  BufferFragment locale;
208  int start;
209  while ((start = aKey.lastIndexOf('[')) >= 0) {
210  int end = aKey.indexOf(']', start);
211  if (end < 0) {
212  qWarning() << warningProlog(file, lineNo)
213  << "Invalid entry (missing ']')";
214  goto next_line;
215  } else if (end > start + 1 && aKey.at(start + 1) == '$') { // found option(s)
216  int i = start + 2;
217  while (i < end) {
218  switch (aKey.at(i)) {
219  case 'i':
220  if (!kde_kiosk_exception) {
221  entryOptions |= KEntryMap::EntryImmutable;
222  }
223  break;
224  case 'e':
225  if (allowExecutableValues) {
226  entryOptions |= KEntryMap::EntryExpansion;
227  }
228  break;
229  case 'd':
230  entryOptions |= KEntryMap::EntryDeleted;
231  aKey = aKey.left(start);
232  printableToString(&aKey, file, lineNo);
233  entryMap.setEntry(currentGroup, aKey.toByteArray(), QByteArray(), entryOptions);
234  goto next_line;
235  default:
236  break;
237  }
238  i++;
239  }
240  } else { // found a locale
241  if (!locale.isNull()) {
242  qWarning() << warningProlog(file, lineNo)
243  << "Invalid entry (second locale!?)";
244  goto next_line;
245  }
246 
247  locale = aKey.mid(start + 1, end - start - 1);
248  }
249  aKey.truncate(start);
250  }
251  if (eqpos < 0) { // Do this here after [$d] was checked
252  qWarning() << warningProlog(file, lineNo) << "Invalid entry (missing '=')";
253  continue;
254  }
255  printableToString(&aKey, file, lineNo);
256  if (!locale.isEmpty()) {
257  if (locale != currentLocale && locale != currentLanguage) {
258  // backward compatibility. C == en_US
259  if (locale.at(0) != 'C' || currentLocale != "en_US") {
260  if (merging) {
261  entryOptions |= KEntryMap::EntryRawKey;
262  } else {
263  goto next_line; // skip this entry if we're not merging
264  }
265  }
266  }
267  }
268 
269  if (!(entryOptions & KEntryMap::EntryRawKey)) {
270  printableToString(&aKey, file, lineNo);
271  }
272 
273  if (options & ParseGlobal) {
274  entryOptions |= KEntryMap::EntryGlobal;
275  }
276  if (bDefault) {
277  entryOptions |= KEntryMap::EntryDefault;
278  }
279  if (!locale.isNull()) {
280  entryOptions |= KEntryMap::EntryLocalized;
281  if (locale.indexOf('_') != -1) {
282  entryOptions |= KEntryMap::EntryLocalizedCountry;
283  }
284  }
285  printableToString(&line, file, lineNo);
286  if (entryOptions & KEntryMap::EntryRawKey) {
287  QByteArray rawKey;
288  rawKey.reserve(aKey.length() + locale.length() + 2);
289  rawKey.append(aKey.toVolatileByteArray());
290  rawKey.append('[').append(locale.toVolatileByteArray()).append(']');
291  entryMap.setEntry(currentGroup, rawKey, lookup(line, &cache), entryOptions);
292  } else {
293  entryMap.setEntry(currentGroup, lookup(aKey, &cache), lookup(line, &cache), entryOptions);
294  }
295  }
296  next_line:
297  continue;
298  }
299 
300  // now make sure immutable groups are marked immutable
301  Q_FOREACH (const QByteArray &group, immutableGroups) {
302  entryMap.setEntry(group, QByteArray(), QByteArray(), KEntryMap::EntryImmutable);
303  }
304 
305  return fileOptionImmutable ? ParseImmutable : ParseOk;
306 }

308 void KConfigIniBackend::writeEntries(const QByteArray &locale, QIODevice &file,
309  const KEntryMap &map, bool defaultGroup, bool &firstEntry)
310 {
311  QByteArray currentGroup;
312  bool groupIsImmutable = false;
313  const KEntryMapConstIterator end = map.constEnd();
314  for (KEntryMapConstIterator it = map.constBegin(); it != end; ++it) {
315  const KEntryKey &key = it.key();
316 
317  // Either process the default group or all others
318  if ((key.mGroup != "<default>") == defaultGroup) {
319  continue; // skip
320  }
321 
322  // the only thing we care about groups is, is it immutable?
323  if (key.mKey.isNull()) {
324  groupIsImmutable = it->bImmutable;
325  continue; // skip
326  }
327 
328  const KEntry &currentEntry = *it;
329  if (!defaultGroup && currentGroup != key.mGroup) {
330  if (!firstEntry) {
331  file.putChar('\n');
332  }
333  currentGroup = key.mGroup;
334  for (int start = 0, end;; start = end + 1) {
335  file.putChar('[');
336  end = currentGroup.indexOf('\x1d', start);
337  if (end < 0) {
338  int cgl = currentGroup.length();
339  if (currentGroup.at(start) == '$' && cgl - start <= 10) {
340  for (int i = start + 1; i < cgl; i++) {
341  char c = currentGroup.at(i);
342  if (c < 'a' || c > 'z') {
343  goto nope;
344  }
345  }
346  file.write("\\x24");
347  start++;
348  }
349  nope:
350  file.write(stringToPrintable(currentGroup.mid(start), GroupString));
351  file.putChar(']');
352  if (groupIsImmutable) {
353  file.write("[$i]", 4);
354  }
355  file.putChar('\n');
356  break;
357  } else {
358  file.write(stringToPrintable(currentGroup.mid(start, end - start), GroupString));
359  file.putChar(']');
360  }
361  }
362  }
363 
364  firstEntry = false;
365  // it is data for a group
366 
367  if (key.bRaw) { // unprocessed key with attached locale from merge
368  file.write(key.mKey);
369  } else {
370  file.write(stringToPrintable(key.mKey, KeyString)); // Key
371  if (key.bLocal && locale != "C") { // 'C' locale == untranslated
372  file.putChar('[');
373  file.write(locale); // locale tag
374  file.putChar(']');
375  }
376  }
377  if (currentEntry.bDeleted) {
378  if (currentEntry.bImmutable) {
379  file.write("[$di]", 5); // Deleted + immutable
380  } else {
381  file.write("[$d]", 4); // Deleted
382  }
383  } else {
384  if (currentEntry.bImmutable || currentEntry.bExpand) {
385  file.write("[$", 2);
386  if (currentEntry.bImmutable) {
387  file.putChar('i');
388  }
389  if (currentEntry.bExpand) {
390  file.putChar('e');
391  }
392  file.putChar(']');
393  }
394  file.putChar('=');
395  file.write(stringToPrintable(currentEntry.mValue, ValueString));
396  }
397  file.putChar('\n');
398  }
399 }

412 bool KConfigIniBackend::writeConfig(const QByteArray &locale, KEntryMap &entryMap,
413  WriteOptions options)
414 {
415  Q_ASSERT(!filePath().isEmpty());
416 
417  KEntryMap writeMap;
418  const bool bGlobal = options & WriteGlobal;
419 
420  // First, reparse the file on disk, to merge our changes with the ones done by other apps
421  // Store the result into writeMap.
422  {
423  ParseOptions opts = ParseExpansions;
424  if (bGlobal) {
425  opts |= ParseGlobal;
426  }
427  ParseInfo info = parseConfig(locale, writeMap, opts, true);
428  if (info != ParseOk) { // either there was an error or the file became immutable
429  return false;
430  }
431  }
432 
433  const KEntryMapIterator end = entryMap.end();
434  for (KEntryMapIterator it = entryMap.begin(); it != end; ++it) {
435  if (!it.key().mKey.isEmpty() && !it->bDirty) { // not dirty, doesn't overwrite entry in writeMap. skips default entries, too.
436  continue;
437  }
438 
439  const KEntryKey &key = it.key();
440 
441  // only write entries that have the same "globality" as the file
442  if (it->bGlobal == bGlobal) {
443  if (it->bReverted) {
444  writeMap.remove(key);
445  } else if (!it->bDeleted) {
446  writeMap[key] = *it;
447  } else {
448  KEntryKey defaultKey = key;
449  defaultKey.bDefault = true;
450  if (!entryMap.contains(defaultKey)) {
451  writeMap.remove(key); // remove the deleted entry if there is no default
452  //qDebug() << "Detected as deleted=>removed:" << key.mGroup << key.mKey << "global=" << bGlobal;
453  } else {
454  writeMap[key] = *it; // otherwise write an explicitly deleted entry
455  //qDebug() << "Detected as deleted=>[$d]:" << key.mGroup << key.mKey << "global=" << bGlobal;
456  }
457  }
458  it->bDirty = false;
459  }
460  }
461 
462  // now writeMap should contain only entries to be written
463  // so write it out to disk
464 
465  // check if file exists
466  QFile::Permissions fileMode = QFile::ReadUser | QFile::WriteUser;
467  bool createNew = true;
468 
469  QFileInfo fi(filePath());
470  if (fi.exists()) {
471 #ifdef Q_OS_WIN
472  //TODO: getuid does not exist on windows, use GetSecurityInfo and GetTokenInformation instead
473  createNew = false;
474 #else
475  if (fi.ownerId() == ::getuid()) {
476  // Preserve file mode if file exists and is owned by user.
477  fileMode = fi.permissions();
478  } else {
479  // File is not owned by user:
480  // Don't create new file but write to existing file instead.
481  createNew = false;
482  }
483 #endif
484  }
485 
486  if (createNew) {
487  QSaveFile file(filePath());
488  if (!file.open(QIODevice::WriteOnly)) {
489  return false;
490  }
491 
492  file.setTextModeEnabled(true); // to get eol translation
493  writeEntries(locale, file, writeMap);
494 
495  if (!file.size() && (fileMode == (QFile::ReadUser | QFile::WriteUser))) {
496  // File is empty and doesn't have special permissions: delete it.
497  file.cancelWriting();
498 
499  if (fi.exists()) {
500  // also remove the old file in case it existed. this can happen
501  // when we delete all the entries in an existing config file.
502  // if we don't do this, then deletions and revertToDefault's
503  // will mysteriously fail
504  QFile::remove(filePath());
505  }
506  } else {
507  // Normal case: Close the file
508  if (file.commit()) {
509  QFile::setPermissions(filePath(), fileMode);
510  return true;
511  }
512  // Couldn't write. Disk full?
513  qWarning() << "Couldn't write" << filePath() << ". Disk full?";
514  return false;
515  }
516  } else {
517  // Open existing file. *DON'T* create it if it suddenly does not exist!
518 #ifdef Q_OS_UNIX
519  int fd = QT_OPEN(QFile::encodeName(filePath()).constData(), O_WRONLY | O_TRUNC);
520  if (fd < 0) {
521  return false;
522  }
523  FILE *fp = ::fdopen(fd, "w");
524  if (!fp) {
525  QT_CLOSE(fd);
526  return false;
527  }
528  QFile f;
529  if (!f.open(fp, QIODevice::WriteOnly)) {
530  fclose(fp);
531  return false;
532  }
533  writeEntries(locale, f, writeMap);
534  f.close();
535  fclose(fp);
536 #else
537  QFile f(filePath());
538  // XXX This is broken - it DOES create the file if it is suddenly gone.
539  if (!f.open(QIODevice::WriteOnly | QIODevice::Truncate)) {
540  return false;
541  }
542  f.setTextModeEnabled(true);
543  writeEntries(locale, f, writeMap);
544 #endif
545  }
546  return true;
547 }

549 bool KConfigIniBackend::isWritable() const
550 {
551  const QString filePath = this->filePath();
552  if (!filePath.isEmpty()) {
553  QFileInfo file(filePath);
554  if (!file.exists()) {
555  // If the file does not exist, check if the deepest
556  // existing dir is writable.
557  QFileInfo dir(file.absolutePath());
558  while (!dir.exists()) {
559  QString parent = dir.absolutePath(); // Go up. Can't use cdUp() on non-existing dirs.
560  if (parent == dir.filePath()) {
561  // no parent
562  return false;
563  }
564  dir.setFile(parent);
565  }
566  return dir.isDir() && dir.isWritable();
567  } else {
568  return file.isWritable();
569  }
570  }
571 
572  return false;
573 }

651 QByteArray KConfigIniBackend::stringToPrintable(const QByteArray &aString, StringType type)
652 {
653  static const char nibbleLookup[] = {
654  '0', '1', '2', '3', '4', '5', '6', '7',
655  '8', '9', 'a', 'b', 'c', 'd', 'e', 'f'
656  };
657 
658  if (aString.isEmpty()) {
659  return aString;
660  }
661  const int l = aString.length();
662 
663  QByteArray result; // Guesstimated that it's good to avoid data() initialization for a length of l*4
664  result.resize(l * 4); // Maximum 4x as long as source string due to \x<ab> escape sequences
665  const char *s = aString.constData();
666  int i = 0;
667  char *data = result.data();
668  char *start = data;
669 
670  // Protect leading space
671  if (s[0] == ' ' && type != GroupString) {
672  *data++ = '\\';
673  *data++ = 's';
674  i++;
675  }
676 
677  for (; i < l; ++i/*, r++*/) {
678  switch (s[i]) {
679  default:
680  // The \n, \t, \r cases (all < 32) are handled below; we can ignore them here
681  if (((unsigned char)s[i]) < 32) {
682  goto doEscape;
683  }
684  *data++ = s[i];
685  break;
686  case '\n':
687  *data++ = '\\';
688  *data++ = 'n';
689  break;
690  case '\t':
691  *data++ = '\\';
692  *data++ = 't';
693  break;
694  case '\r':
695  *data++ = '\\';
696  *data++ = 'r';
697  break;
698  case '\\':
699  *data++ = '\\';
700  *data++ = '\\';
701  break;
702  case '=':
703  if (type != KeyString) {
704  *data++ = s[i];
705  break;
706  }
707  goto doEscape;
708  case '[':
709  case ']':
710  // Above chars are OK to put in *value* strings as plaintext
711  if (type == ValueString) {
712  *data++ = s[i];
713  break;
714  }
715  doEscape:
716  *data++ = '\\';
717  *data++ = 'x';
718  *data++ = nibbleLookup[((unsigned char)s[i]) >> 4];
719  *data++ = nibbleLookup[((unsigned char)s[i]) & 0x0f];
720  break;
721  }
722  }
723  *data = 0;
724  result.resize(data - start);
725 
726  // Protect trailing space
727  if (result.endsWith(' ') && type != GroupString) {
728  result.replace(result.length() - 1, 1, "\\s");
729  }
730  result.squeeze();
731 
732  return result;
733 }

735 char KConfigIniBackend::charFromHex(const char *str, const QFile &file, int line)
736 {
737  unsigned char ret = 0;
738  for (int i = 0; i < 2; i++) {
739  ret <<= 4;
740  quint8 c = quint8(str[i]);
741 
742  if (c >= '0' && c <= '9') {
743  ret |= c - '0';
744  } else if (c >= 'a' && c <= 'f') {
745  ret |= c - 'a' + 0x0a;
746  } else if (c >= 'A' && c <= 'F') {
747  ret |= c - 'A' + 0x0a;
748  } else {
749  QByteArray e(str, 2);
750  e.prepend("\\x");
751  qWarning() << warningProlog(file, line) << "Invalid hex character " << c
752  << " in \\x<nn>-type escape sequence \"" << e.constData() << "\".";
753  return 'x';
754  }
755  }
756  return char(ret);
757 }

759 void KConfigIniBackend::printableToString(BufferFragment *aString, const QFile &file, int line)
760 {
761  if (aString->isEmpty() || aString->indexOf('\\') == -1) {
762  return;
763  }
764  aString->trim();
765  int l = aString->length();
766  char *r = aString->data();
767  char *str = r;
768 
769  for (int i = 0; i < l; i++, r++) {
770  if (str[i] != '\\') {
771  *r = str[i];
772  } else {
773  // Probable escape sequence
774  i++;
775  if (i >= l) { // Line ends after backslash - stop.
776  *r = '\\';
777  break;
778  }
779 
780  switch (str[i]) {
781  case 's':
782  *r = ' ';
783  break;
784  case 't':
785  *r = '\t';
786  break;
787  case 'n':
788  *r = '\n';
789  break;
790  case 'r':
791  *r = '\r';
792  break;
793  case '\\':
794  *r = '\\';
795  break;
796  case ';':
797  // not really an escape sequence, but allowed in .desktop files, don't strip '\;' from the string
798  *r = '\\';
799  r++;
800  *r = ';';
801  break;
802  case 'x':
803  if (i + 2 < l) {
804  *r = charFromHex(str + i + 1, file, line);
805  i += 2;
806  } else {
807  *r = 'x';
808  i = l - 1;
809  }
810  break;
811  default:
812  *r = '\\';
813  qWarning() << warningProlog(file, line)
814  << QStringLiteral("Invalid escape sequence \"\\%1\".").arg(str[i]);
815  }
816  }
817  }
818  aString->truncate(r - aString->constData());
819 }
