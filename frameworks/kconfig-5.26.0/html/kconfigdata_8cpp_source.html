84 bool KEntryMap::setEntry(const QByteArray &group, const QByteArray &key, const QByteArray &value, KEntryMap::EntryOptions options)
85 {
86  KEntryKey k;
87  KEntry e;
88  bool newKey = false;
89 
90  const Iterator it = findExactEntry(group, key, SearchFlags(options >> 16));
91 
92  if (key.isEmpty()) { // inserting a group marker
93  k.mGroup = group;
94  e.bImmutable = (options & EntryImmutable);
95  if (options & EntryDeleted) {
96  qWarning("Internal KConfig error: cannot mark groups as deleted");
97  }
98  if (it == end()) {
99  insert(k, e);
100  return true;
101  } else if (it.value() == e) {
102  return false;
103  }
104 
105  it.value() = e;
106  return true;
107  }
108 
109  if (it != end()) {
110  if (it->bImmutable) {
111  return false; // we cannot change this entry. Inherits group immutability.
112  }
113  k = it.key();
114  e = *it;
115  //qDebug() << "found existing entry for key" << k;
116  } else {
117  // make sure the group marker is in the map
118  KEntryMap const *that = this;
119  ConstIterator cit = that->findEntry(group);
120  if (cit == constEnd()) {
121  insert(KEntryKey(group), KEntry());
122  } else if (cit->bImmutable) {
123  return false; // this group is immutable, so we cannot change this entry.
124  }
125 
126  k = KEntryKey(group, key);
127  newKey = true;
128  }
129 
130  // set these here, since we may be changing the type of key from the one we found
131  k.bLocal = (options & EntryLocalized);
132  k.bDefault = (options & EntryDefault);
133  k.bRaw = (options & EntryRawKey);
134 
135  e.mValue = value;
136  e.bDirty = e.bDirty || (options & EntryDirty);
137  e.bGlobal = (options & EntryGlobal); //we can't use || here, because changes to entries in
138  //kdeglobals would be written to kdeglobals instead
139  //of the local config file, regardless of the globals flag
140  e.bImmutable = e.bImmutable || (options & EntryImmutable);
141  if (value.isNull()) {
142  e.bDeleted = e.bDeleted || (options & EntryDeleted);
143  } else {
144  e.bDeleted = false; // setting a value to a previously deleted entry
145  }
146  e.bExpand = (options & EntryExpansion);
147  e.bReverted = false;
148  if (options & EntryLocalized) {
149  e.bLocalizedCountry = (options & EntryLocalizedCountry);
150  } else {
151  e.bLocalizedCountry = false;
152  }
153 
154  if (newKey) {
155  //qDebug() << "inserting" << k << "=" << value;
156  insert(k, e);
157  if (k.bDefault) {
158  k.bDefault = false;
159  //qDebug() << "also inserting" << k << "=" << value;
160  insert(k, e);
161  }
162  // TODO check for presence of unlocalized key
163  return true;
164  } else {
165 // KEntry e2 = it.value();
166  if (options & EntryLocalized) {
167  // fast exit checks for cases where the existing entry is more specific
168  const KEntry &e2 = it.value();
169  if (e2.bLocalizedCountry && !e.bLocalizedCountry) {
170  // lang_COUNTRY > lang
171  return false;
172  }
173  }
174  if (it.value() != e) {
175  //qDebug() << "changing" << k << "from" << e.mValue << "to" << value;
176  it.value() = e;
177  if (k.bDefault) {
178  KEntryKey nonDefaultKey(k);
179  nonDefaultKey.bDefault = false;
180  insert(nonDefaultKey, e);
181  }
182  if (!(options & EntryLocalized)) {
183  KEntryKey theKey(group, key, true, false);
184  //qDebug() << "non-localized entry, remove localized one:" << theKey;
185  remove(theKey);
186  if (k.bDefault) {
187  theKey.bDefault = true;
188  remove(theKey);
189  }
190  }
191  return true;
192  } else {
193  //qDebug() << k << "was already set to" << e.mValue;
194  if (!(options & EntryLocalized)) {
195  //qDebug() << "unchanged non-localized entry, remove localized one.";
196  KEntryKey theKey(group, key, true, false);
197  bool ret = false;
198  Iterator cit = find(theKey);
199  if (cit != end()) {
200  erase(cit);
201  ret = true;
202  }
203  if (k.bDefault) {
204  theKey.bDefault = true;
205  Iterator cit = find(theKey);
206  if (cit != end()) {
207  erase(cit);
208  return true;
209  }
210  }
211  return ret;
212  }
213  //qDebug() << "localized entry, unchanged, return false";
214  // When we are writing a default, we know that the non-
215  // default is the same as the default, so we can simply
216  // use the same branch.
217  return false;
218  }
219  }
220 }

222 QString KEntryMap::getEntry(const QByteArray &group, const QByteArray &key, const QString &defaultValue, KEntryMap::SearchFlags flags, bool *expand) const
223 {
224  const ConstIterator it = findEntry(group, key, flags);
225  QString theValue = defaultValue;
226 
227  if (it != constEnd() && !it->bDeleted) {
228  if (!it->mValue.isNull()) {
229  const QByteArray data = it->mValue;
230  theValue = QString::fromUtf8(data.constData(), data.length());
231  if (expand) {
232  *expand = it->bExpand;
233  }
234  }
235  }
236 
237  return theValue;
238 }

240 bool KEntryMap::hasEntry(const QByteArray &group, const QByteArray &key, KEntryMap::SearchFlags flags) const
241 {
242  const ConstIterator it = findEntry(group, key, flags);
243  if (it == constEnd()) {
244  return false;
245  }
246  if (it->bDeleted) {
247  return false;
248  }
249  if (key.isNull()) { // looking for group marker
250  return it->mValue.isNull();
251  }
252  // if it->bReverted, we'll just return true; the real answer depends on lookup up with SearchDefaults, though.
253  return true;
254 }

305 bool KEntryMap::revertEntry(const QByteArray &group, const QByteArray &key, KEntryMap::SearchFlags flags)
306 {
307  Q_ASSERT((flags & KEntryMap::SearchDefaults) == 0);
308  Iterator entry = findEntry(group, key, flags);
309  if (entry != end()) {
310  //qDebug() << "reverting" << entry.key() << " = " << entry->mValue;
311  if (entry->bReverted) { // already done before
312  return false;
313  }
314 
315  KEntryKey defaultKey(entry.key());
316  defaultKey.bDefault = true;
317  //qDebug() << "looking up default entry with key=" << defaultKey;
318  const ConstIterator defaultEntry = constFind(defaultKey);
319  if (defaultEntry != constEnd()) {
320  Q_ASSERT(defaultEntry.key().bDefault);
321  //qDebug() << "found, update entry";
322  *entry = *defaultEntry; // copy default value, for subsequent lookups
323  } else {
324  entry->mValue = QByteArray();
325  }
326  entry->bDirty = true;
327  entry->bReverted = true; // skip it when writing out to disk
328 
329  //qDebug() << "Here's what we have now:" << *this;
330  return true;
331  }
332  return false;
333 }
