179 static KStandardShortcutInfo *guardedStandardShortcutInfo(StandardShortcut id)
180 {
181  if (id >= static_cast<int>(sizeof(g_infoStandardShortcut) / sizeof(KStandardShortcutInfo)) ||
182  id < 0) {
183  qWarning() << "KStandardShortcut: id not found!";
184  return &g_infoStandardShortcut[AccelNone];
185  } else {
186  return &g_infoStandardShortcut[id];
187  }
188 }

195 static void initialize(StandardShortcut id)
196 {
197  KStandardShortcutInfo *info = guardedStandardShortcutInfo(id);
198 
199  // All three are needed.
200  if (info->id != AccelNone) {
201  Q_ASSERT(info->description.text);
202  Q_ASSERT(info->description.context);
203  Q_ASSERT(info->name);
204  }
205 
206  KConfigGroup cg(KSharedConfig::openConfig(), "Shortcuts");
207 
208  if (cg.hasKey(info->name)) {
209  QString s = cg.readEntry(info->name);
210  if (s != QLatin1String("none")) {
211  info->cut = QKeySequence::listFromString(s);
212  } else {
213  info->cut = QList<QKeySequence>();
214  }
215  } else {
216  info->cut = hardcodedDefaultShortcut(id);
217  }
218 
219  info->isInitialized = true;
220 }
221 
222 void saveShortcut(StandardShortcut id, const QList<QKeySequence> &newShortcut)
223 {
224  KStandardShortcutInfo *info = guardedStandardShortcutInfo(id);
225  // If the action has no standard shortcut associated there is nothing to
226  // save
227  if (info->id == AccelNone) {
228  return;
229  }
230 
231  KConfigGroup cg(KSharedConfig::openConfig(), "Shortcuts");
232 
233  info->cut = newShortcut;
234  bool sameAsDefault = (newShortcut == hardcodedDefaultShortcut(id));
235 
236  if (sameAsDefault) {
237  // If the shortcut is the equal to the hardcoded one we remove it from
238  // kdeglobal if necessary and return.
239  if (cg.hasKey(info->name)) {
240  cg.deleteEntry(info->name, KConfig::Global | KConfig::Persistent);
241  }
242 
243  return;
244  }
245 
246  // Write the changed shortcut to kdeglobals
247  cg.writeEntry(info->name, QKeySequence::listToString(info->cut), KConfig::Global | KConfig::Persistent);
248 }

284 StandardShortcut find(const QKeySequence &seq)
285 {
286  if (!seq.isEmpty()) {
287  for (uint i = 0; i < sizeof(g_infoStandardShortcut) / sizeof(KStandardShortcutInfo); i++) {
288  StandardShortcut id = g_infoStandardShortcut[i].id;
289  if (id != AccelNone) {
290  if (!g_infoStandardShortcut[i].isInitialized) {
291  initialize(id);
292  }
293  if (g_infoStandardShortcut[i].cut.contains(seq)) {
294  return id;
295  }
296  }
297  }
298  }
299  return AccelNone;
300 }

302 StandardShortcut find(const char *keyName)
303 {
304  for (uint i = 0; i < sizeof(g_infoStandardShortcut) / sizeof(KStandardShortcutInfo); i++)
305  if (qstrcmp(g_infoStandardShortcut[i].name, keyName)) {
306  return g_infoStandardShortcut[i].id;
307  }
308 
309  return AccelNone;
310 }

312 QList<QKeySequence> hardcodedDefaultShortcut(StandardShortcut id)
313 {
314  QList<QKeySequence> cut;
315  KStandardShortcutInfo *info = guardedStandardShortcutInfo(id);
316 
317  if (info->cutDefault != 0) {
318  cut << info->cutDefault;
319  }
320 
321  if (info->cutDefault2 != 0) {
322  if (cut.isEmpty()) {
323  cut << QKeySequence();
324  }
325 
326  cut << info->cutDefault2;
327  }
328 
329  return cut;
330 }
