241 KEMailSettings::KEMailSettings()
242  : p(new KEMailSettingsPrivate())
243 {
244  p->m_sCurrentProfile.clear();
245 
246  p->m_pConfig = new KConfig(QStringLiteral("emaildefaults"));
247 
248  const QStringList groups = p->m_pConfig->groupList();
249  for (QStringList::ConstIterator it = groups.begin(); it != groups.end(); ++it) {
250  if ((*it).startsWith(QLatin1String("PROFILE_"))) {
251  p->profiles += (*it).mid(8, (*it).length());
252  }
253  }
254 
255  KConfigGroup cg(p->m_pConfig, "Defaults");
256  p->m_sDefaultProfile = cg.readEntry("Profile", tr("Default"));
257  if (!p->m_sDefaultProfile.isNull()) {
258  if (!p->m_pConfig->hasGroup(QStringLiteral("PROFILE_") + p->m_sDefaultProfile)) {
259  setDefault(tr("Default"));
260  } else {
261  setDefault(p->m_sDefaultProfile);
262  }
263  } else {
264  if (!p->profiles.isEmpty()) {
265  setDefault(p->profiles[0]);
266  } else {
267  setDefault(tr("Default"));
268  }
269  }
270  setProfile(defaultProfileName());
271 }
