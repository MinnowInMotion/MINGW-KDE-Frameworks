74 // These operators are used to check whether an entry which is about

130 inline bool operator <(const KEntryKey &k1, const KEntryKey &k2)
131 {
132  int result = qstrcmp(k1.mGroup, k2.mGroup);
133  if (result != 0) {
134  return result < 0;
135  }
136 
137  result = qstrcmp(k1.mKey, k2.mKey);
138  if (result != 0) {
139  return result < 0;
140  }
141 
142  if (k1.bLocal != k2.bLocal) {
143  return k1.bLocal;
144  }
145  return (!k1.bDefault && k2.bDefault);
146 }
