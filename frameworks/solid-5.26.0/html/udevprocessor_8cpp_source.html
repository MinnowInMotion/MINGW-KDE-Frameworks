53 int Processor::maxSpeed() const
54 {
55  if (m_maxSpeed == -1) {
56  QFile cpuMaxFreqFile(m_device->deviceName() + prefix() + "/cpufreq/cpuinfo_max_freq");
57  if (cpuMaxFreqFile.open(QIODevice::ReadOnly)) {
58  QString value = cpuMaxFreqFile.readAll().trimmed();
59  // cpuinfo_max_freq is in kHz
60  m_maxSpeed = static_cast<int>(value.toLongLong() / 1000);
61  }
62  if (m_maxSpeed <= 0) {
63  // couldn't get the info from /sys, try /proc instead
64  m_maxSpeed = extractCurrentCpuSpeed(number());
65  }
66  }
67  return m_maxSpeed;
68 }

70 bool Processor::canChangeFrequency() const
71 {
72  if (m_canChangeFrequency == NotChecked) {
73  /* Note that cpufreq is the right information source here, rather than
74  * anything to do with throttling (ACPI T-states). */
75 
76  m_canChangeFrequency = CannotChangeFreq;
77 
78  QFile cpuMinFreqFile(m_device->deviceName() + prefix() + "/cpufreq/cpuinfo_min_freq");
79  QFile cpuMaxFreqFile(m_device->deviceName() + prefix() + "/cpufreq/cpuinfo_max_freq");
80  if (cpuMinFreqFile.open(QIODevice::ReadOnly) && cpuMaxFreqFile.open(QIODevice::ReadOnly)) {
81  qlonglong minFreq = cpuMinFreqFile.readAll().trimmed().toLongLong();
82  qlonglong maxFreq = cpuMaxFreqFile.readAll().trimmed().toLongLong();
83  if (minFreq > 0 && maxFreq > minFreq) {
84  m_canChangeFrequency = CanChangeFreq;
85  }
86  }
87  }
88 
89  return m_canChangeFrequency == CanChangeFreq;
90 }
