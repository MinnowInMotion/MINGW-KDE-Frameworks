51 std::ostream &operator<<(std::ostream &out, const QVariant &value)
52 {
53  switch (value.type())
54  {
55  case QVariant::StringList:
56  {
57  out << "{";
58 
59  const QStringList list = value.toStringList();
60 
61  QStringList::ConstIterator it = list.constBegin();
62  QStringList::ConstIterator end = list.constEnd();
63 
64  for (; it!=end; ++it)
65  {
66  out << "'" << *it << "'";
67 
68  if (it+1!=end)
69  {
70  out << ", ";
71  }
72  }
73 
74  out << "} (string list)";
75  break;
76  }
77  case QVariant::Bool:
78  out << (value.toBool()?"true":"false") << " (bool)";
79  break;
80  case QVariant::Int:
81  case QVariant::LongLong:
82  out << value.toString()
83  << " (0x" << QString::number(value.toLongLong(), 16) << ") (" << QVariant::typeToName(value.type()) << ")";
84  break;
85  case QVariant::UInt:
86  case QVariant::ULongLong:
87  out << value.toString()
88  << " (0x" << QString::number(value.toULongLong(), 16) << ") (" << QVariant::typeToName(value.type()) << ")";
89  break;
90  case QVariant::Double:
91  out << value.toString() << " (double)";
92  break;
93  case QVariant::UserType:
94  {
95  //qDebug() << "got variant type:" << value.typeName();
96  if (value.canConvert<QList<int> >())
97  {
98  QList<int> intlist = value.value<QList<int> >();
99  QStringList tmp;
100  foreach (int val, intlist)
101  tmp.append(QString::number(val));
102  out << "{" << tmp.join(",") << "} (int list)";
103  }
104  break;
105  }
106  default:
107  out << "'" << value.toString() << "' (string)";
108  break;
109  }
110 
111  return out;
112 }

114 std::ostream &operator<<(std::ostream &out, const Solid::Device &device)
115 {
116  out << " parent = " << QVariant(device.parentUdi()) << endl;
117  out << " vendor = " << QVariant(device.vendor()) << endl;
118  out << " product = " << QVariant(device.product()) << endl;
119  out << " description = " << QVariant(device.description()) << endl;
120 
121  int index = Solid::DeviceInterface::staticMetaObject.indexOfEnumerator("Type");
122  QMetaEnum typeEnum = Solid::DeviceInterface::staticMetaObject.enumerator(index);
123 
124  for (int i=0; i<typeEnum.keyCount(); i++)
125  {
126  Solid::DeviceInterface::Type type = (Solid::DeviceInterface::Type)typeEnum.value(i);
127  const Solid::DeviceInterface *interface = device.asDeviceInterface(type);
128 
129  if (interface)
130  {
131  const QMetaObject *meta = interface->metaObject();
132 
133  for (int i=meta->propertyOffset(); i<meta->propertyCount(); i++)
134  {
135  QMetaProperty property = meta->property(i);
136  out << " " << QString(meta->className()).mid(7) << "." << property.name()
137  << " = ";
138 
139  QVariant value = property.read(interface);
140 
141  if (property.isEnumType()) {
142  QMetaEnum metaEnum = property.enumerator();
143  if (metaEnum.isFlag()) {
144  out << "'" << metaEnum.valueToKeys(value.toInt()).constData() << "'"
145  << " (0x" << QString::number(value.toInt(), 16) << ") (flag)";
146  } else {
147  out << "'" << metaEnum.valueToKey(value.toInt()) << "'"
148  << " (0x" << QString::number(value.toInt(), 16) << ") (enum)";
149  }
150  out << endl;
151  } else {
152  out << value << endl;
153  }
154  }
155  }
156  }
157 
158  return out;
159 }

272  parser.addPositionalArgument("udi", QCoreApplication::translate("solid-hardware", "Device udi"));
273  parser.addPositionalArgument("parent", QCoreApplication::translate("solid-hardware", "Parent device udi"));
274  parser.process(app);
275  if (parser.positionalArguments().count() < 2 || parser.positionalArguments().count() > 3) {
276  parser.showHelp(1);
277  }
278 
279  QString query = args.at(1);
280  QString parent;
281 
282  if (args.count() == 3)
283  {
284  parent = args.at(2);
285  }
286 
287  return app.hwQuery(parent, query);
288  } else if (command == "mount") {
289  const QString udi = getUdiFromArguments(app, parser);
290  return app.hwVolumeCall(SolidHardware::Mount, udi);
291  } else if (command == "unmount") {
292  const QString udi = getUdiFromArguments(app, parser);
293  return app.hwVolumeCall(SolidHardware::Unmount, udi);
294  } else if (command == "eject") {
295  const QString udi = getUdiFromArguments(app, parser);
296  return app.hwVolumeCall(SolidHardware::Eject, udi);
297  } else if (command == "listen") {
298  return app.listen();
299  }
300 
301  cerr << QCoreApplication::translate("solid-hardware", "Syntax Error: Unknown command '%1'").arg(command) << endl;
302 
303  return 1;

364 bool SolidHardware::hwVolumeCall(SolidHardware::VolumeCallType type, const QString &udi)
365 {
366  Solid::Device device(udi);
367 
368  if (!device.is<Solid::StorageAccess>() && type!=Eject)
369  {
370  cerr << tr("Error: %1 does not have the interface StorageAccess.").arg(udi) << endl;
371  return false;
372  }
373  else if (!device.is<Solid::OpticalDrive>() && type==Eject)
374  {
375  cerr << tr("Error: %1 does not have the interface OpticalDrive.").arg(udi) << endl;
376  return false;
377  }
378 
379  switch(type)
380  {
381  case Mount:
382  connect(device.as<Solid::StorageAccess>(),
383  SIGNAL(setupDone(Solid::ErrorType, QVariant, const QString &)),
384  this,
385  SLOT(slotStorageResult(Solid::ErrorType, QVariant)));
386  device.as<Solid::StorageAccess>()->setup();
387  break;
388  case Unmount:
389  connect(device.as<Solid::StorageAccess>(),
390  SIGNAL(teardownDone(Solid::ErrorType, QVariant, const QString &)),
391  this,
392  SLOT(slotStorageResult(Solid::ErrorType, QVariant)));
393  device.as<Solid::StorageAccess>()->teardown();
394  break;
395  case Eject:
396  connect(device.as<Solid::OpticalDrive>(),
397  SIGNAL(ejectDone(Solid::ErrorType, QVariant, const QString &)),
398  this,
399  SLOT(slotStorageResult(Solid::ErrorType, QVariant)));
400  device.as<Solid::OpticalDrive>()->eject();
401  break;
402  }
403 
404  m_loop.exec();
405 
406  if (m_error)
407  {
408  cerr << tr("Error: %1").arg(m_errorString) << endl;
409  return false;
410  }
411 
412  return true;
413 }
