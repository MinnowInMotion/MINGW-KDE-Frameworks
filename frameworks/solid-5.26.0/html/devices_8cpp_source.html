34 QSharedPointer<DevicesQueryPrivate> DevicesQueryPrivate::forQuery(const QString &query)
35 {
36  if (handlers.contains(query)) {
37  return handlers[query].toStrongRef();
38  }
39 
40  // Creating a new shared backend instance
41  QSharedPointer<DevicesQueryPrivate> backend(new DevicesQueryPrivate(query));
42 
43  // Storing a weak pointer to the backend
44  handlers[query] = backend;
45 
46  // Returns the newly created backend
47  // TODO: It would be nicer with std::move and STL's smart pointers,
48  // but RVO should optimize this out.
49  return backend;
50 }

52 DevicesQueryPrivate::DevicesQueryPrivate(const QString &query)
53  : query(query)
54  , predicate(Solid::Predicate::fromString(query))
55  , notifier(Solid::DeviceNotifier::instance())
56 {
57  connect(notifier, &Solid::DeviceNotifier::deviceAdded,
58  this, &DevicesQueryPrivate::addDevice);
59  connect(notifier, &Solid::DeviceNotifier::deviceRemoved,
60  this, &DevicesQueryPrivate::removeDevice);
61 
62  if (!query.isEmpty() && !predicate.isValid()) {
63  return;
64  }
65 
66  Q_FOREACH (const Solid::Device &device, Solid::Device::listFromQuery(predicate)) {
67  matchingDevices << device.udi();
68  }
69 }

97 void Devices::initialize() const
98 {
99  if (m_backend) {
100  return;
101  }
102 
103  m_backend = DevicesQueryPrivate::forQuery(m_query);
104 
105  connect(m_backend.data(), &DevicesQueryPrivate::deviceAdded,
106  this, &Devices::addDevice);
107  connect(m_backend.data(), &DevicesQueryPrivate::deviceAdded,
108  this, &Devices::addDevice);
109  connect(m_backend.data(), &DevicesQueryPrivate::deviceRemoved,
110  this, &Devices::removeDevice);
111 
112  const int matchesCount = m_backend->devices().count();
113 
114  if (matchesCount != 0) {
115  emit emptyChanged(false);
116  emit countChanged(matchesCount);
117  emit devicesChanged(m_backend->devices());
118  }
119 }
