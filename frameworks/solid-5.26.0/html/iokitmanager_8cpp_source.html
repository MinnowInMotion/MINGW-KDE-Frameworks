56 QStringList IOKitManagerPrivate::devicesFromRegistry(io_iterator_t it)
57 {
58  QStringList result;
59  io_object_t obj;
60  io_string_t pathName;
61  while ((obj = IOIteratorNext(it))) {
62  kern_return_t ret = IORegistryEntryGetPath(obj, kIOServicePlane, pathName);
63  if (ret != KERN_SUCCESS) {
64  qWarning() << Q_FUNC_INFO << "IORegistryEntryGetPath failed";
65  continue;
66  }
67  result += QString::fromUtf8(pathName);
68  ret = IOObjectRelease(obj);
69  if (ret != KERN_SUCCESS) {
70  // very unlikely to happen - keep it a qDebug just in case.
71  // compiler will nuke this code in release builds.
72  qDebug() << Q_FUNC_INFO << "Unable to release object reference";
73  }
74  }
75  IOObjectRelease(it);
76 
77  return result;
78 }

104 IOKitManager::IOKitManager(QObject *parent)
105  : Solid::Ifaces::DeviceManager(parent), d(new IOKitManagerPrivate)
106 {
107  d->port = IONotificationPortCreate(kIOMasterPortDefault);
108  if (!d->port) {
109  qWarning() << Q_FUNC_INFO << "Unable to create notification port";
110  return;
111  }
112 
113  d->source = IONotificationPortGetRunLoopSource(d->port);
114  if (!d->source) {
115  qWarning() << Q_FUNC_INFO << "Unable to create notification source";
116  return;
117  }
118 
119  CFRunLoopAddSource(CFRunLoopGetCurrent(), d->source, kCFRunLoopDefaultMode);
120 
121  d->supportedInterfaces << Solid::DeviceInterface::GenericInterface
122  << Solid::DeviceInterface::Processor
123  << Solid::DeviceInterface::Block
124  << Solid::DeviceInterface::StorageAccess
125  << Solid::DeviceInterface::StorageDrive
126  << Solid::DeviceInterface::OpticalDrive
127  << Solid::DeviceInterface::StorageVolume
128  << Solid::DeviceInterface::OpticalDisc
129  << Solid::DeviceInterface::Camera
130  << Solid::DeviceInterface::PortableMediaPlayer
131  << Solid::DeviceInterface::Battery;

174 QStringList IOKitManager::devicesFromQuery(const QString &parentUdi,
175  Solid::DeviceInterface::Type type)
176 {
177  QStringList result;
178 
179  if (type == Solid::DeviceInterface::Unknown) {
180  // match all device interfaces
181  result = allDevices();
182  } else {
183  const char *deviceClassName = IOKitManagerPrivate::typeToName(type);
184  if (!deviceClassName) {
185  return QStringList();
186  }
187 
188  CFMutableDictionaryRef matchingDict = IOServiceMatching(deviceClassName);
189 
190  if (!matchingDict) {
191  return QStringList();
192  }
193 
194  io_iterator_t it = 0;
195 
196  // note - IOServiceGetMatchingServices dereferences the dict
197  kern_return_t ret = IOServiceGetMatchingServices(
198  kIOMasterPortDefault,
199  matchingDict,
200  &it);
201 
202  result = IOKitManagerPrivate::devicesFromRegistry(it);
203  }
204 
205  // if the parentUdi is an empty string, return all matches
206  if (parentUdi.isEmpty()) {
207  return result;
208  }
209 
210  // return only matches that start with the parent's UDI
211  QStringList filtered;
212  Q_FOREACH (const QString &udi, result) {
213  if (udi.startsWith(parentUdi)) {
214  filtered += udi;
215  }
216  }
217 
218  return filtered;
219 }
