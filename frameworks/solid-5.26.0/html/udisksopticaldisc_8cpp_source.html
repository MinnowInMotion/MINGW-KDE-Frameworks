40 static Solid::OpticalDisc::ContentType advancedDiscDetect(const QByteArray &device_file)
41 {
42  /* the discs block size */
43  unsigned short bs;
44  /* the path table size */
45  unsigned short ts;
46  /* the path table location (in blocks) */
47  unsigned int tl;
48  /* length of the directory name in current path table entry */
49  unsigned char len_di = 0;
50  /* the number of the parent directory's path table entry */
51  unsigned int parent = 0;
52  /* filename for the current path table entry */
53  char dirname[256];
54  /* our position into the path table */
55  int pos = 0;
56  /* the path table record we're on */
57  int curr_record = 1;
58 
59  Solid::OpticalDisc::ContentType result = Solid::OpticalDisc::NoContent;
60 
61  int fd = open(device_file.constData(), O_RDONLY);
62 
63  /* read the block size */
64  lseek(fd, 0x8080, SEEK_CUR);
65  if (read(fd, &bs, 2) != 2) {
66  qDebug("Advanced probing on %s failed while reading block size", qPrintable(device_file));
67  goto out;
68  }
69 
70  /* read in size of path table */
71  lseek(fd, 2, SEEK_CUR);
72  if (read(fd, &ts, 2) != 2) {
73  qDebug("Advanced probing on %s failed while reading path table size", qPrintable(device_file));
74  goto out;
75  }
76 
77  /* read in which block path table is in */
78  lseek(fd, 6, SEEK_CUR);
79  if (read(fd, &tl, 4) != 4) {
80  qDebug("Advanced probing on %s failed while reading path table block", qPrintable(device_file));
81  goto out;
82  }
83 
84  /* seek to the path table */
85  lseek(fd, bs * tl, SEEK_SET);
86 
87  /* loop through the path table entries */
88  while (pos < ts) {
89  /* get the length of the filename of the current entry */
90  if (read(fd, &len_di, 1) != 1) {
91  qDebug("Advanced probing on %s failed, cannot read more entries", qPrintable(device_file));
92  break;
93  }
94 
95  /* get the record number of this entry's parent
96  i'm pretty sure that the 1st entry is always the top directory */
97  lseek(fd, 5, SEEK_CUR);
98  if (read(fd, &parent, 2) != 2) {
99  qDebug("Advanced probing on %s failed, couldn't read parent entry", qPrintable(device_file));
100  break;
101  }
102 
103  /* read the name */
104  if (read(fd, dirname, len_di) != len_di) {
105  qDebug("Advanced probing on %s failed, couldn't read the entry name", qPrintable(device_file));
106  break;
107  }
108  dirname[len_di] = 0;
109 
110  /* if we found a folder that has the root as a parent, and the directory name matches
111  one of the special directories then set the properties accordingly */
112  if (parent == 1) {
113  if (!strcasecmp(dirname, "VIDEO_TS")) {
114  qDebug("Disc in %s is a Video DVD", qPrintable(device_file));
115  result = Solid::OpticalDisc::VideoDvd;
116  break;
117  } else if (!strcasecmp(dirname, "BDMV")) {
118  qDebug("Disc in %s is a Blu-ray video disc", qPrintable(device_file));
119  result = Solid::OpticalDisc::VideoBluRay;
120  break;
121  } else if (!strcasecmp(dirname, "VCD")) {
122  qDebug("Disc in %s is a Video CD", qPrintable(device_file));
123  result = Solid::OpticalDisc::VideoCd;
124  break;
125  } else if (!strcasecmp(dirname, "SVCD")) {
126  qDebug("Disc in %s is a Super Video CD", qPrintable(device_file));
127  result = Solid::OpticalDisc::SuperVideoCd;
128  break;
129  }
130  }
131 
132  /* all path table entries are padded to be even,
133  so if this is an odd-length table, seek a byte to fix it */
134  if (len_di % 2 == 1) {
135  lseek(fd, 1, SEEK_CUR);
136  pos++;
137  }
138 
139  /* update our position */
140  pos += 8 + len_di;
141  curr_record++;
142  }
143 
144  close(fd);
145  return result;
146 
147 out:
148  /* go back to the start of the file */
149  lseek(fd, 0, SEEK_SET);
150  close(fd);
151  return result;
152 }

156 class ContentTypesCache
157 {
158 public:
159  ContentTypesCache() : m_n(0) { }
160 
161  void add(const OpticalDisc::Identity &key,
162  Solid::OpticalDisc::ContentTypes content)
163  {
164  if (!find(key)) {
165  m_n = qMin(m_n + 1, sizeof(m_info) / sizeof(*m_info));
166  moveToFront(m_n - 1);
167  front().first = key;
168  }
169  front().second = content;
170  }
171 
172  bool find(const OpticalDisc::Identity &key)
173  {
174  for (size_t i = 0; i < m_n; i++) {
175  if (m_info[i].first == key) {
176  moveToFront(i);
177  return true;
178  }
179  }
180  return false;
181  }
182 
183  QPair<OpticalDisc::Identity, Solid::OpticalDisc::ContentTypes> &front()
184  {
185  return *m_info;
186  }
187 
188 private:
189 
190  void moveToFront(size_t i)
191  {
192  while (i) {
193  qSwap(m_info[i - 1], m_info[i]);
194  --i;
195  }
196  }
197 
198  size_t m_n;
199  QPair<OpticalDisc::Identity, Solid::OpticalDisc::ContentTypes> m_info[100];
200 };

231  static QString getKey()
232  {

240 public:
241  SharedContentTypesCache()
242  : m_pointer(0),
243  m_semaphore(getKey() + "sem", 1), m_shmem(getKey() + "mem")
244  {
245  if (!m_semaphore.acquire()) {
246  return;
247  }
248  Releaser releaser(&m_semaphore);
249 
250  if (m_shmem.attach()) {
251  m_pointer = reinterpret_cast<ContentTypesCache *>(m_shmem.data());
252  return;
253  }
254 
255  if (!m_shmem.create(sizeof(ContentTypesCache))) {
256  return;
257  }
258 
259  if (!m_shmem.lock()) {
260  m_shmem.detach();
261  return;
262  }
263  Unlocker unlocker(&m_shmem);
264 
265  m_pointer = new(m_shmem.data()) ContentTypesCache;
266  }

268  Solid::OpticalDisc::ContentTypes getContent(const OpticalDisc::Identity &info,
269  const QByteArray &file)
270  {
271  if (!m_pointer) {
272  return advancedDiscDetect(file);
273  }
274 
275  if (!m_semaphore.acquire()) {
276  return advancedDiscDetect(file);
277  }
278  Releaser releaser(&m_semaphore);
279 
280  if (!m_shmem.lock()) {
281  return advancedDiscDetect(file);
282  }
283  Unlocker unlocker(&m_shmem);
284 
285  if (!m_pointer->find(info)) {
286  m_pointer->add(info, advancedDiscDetect(file));
287  }
288 
289  Solid::OpticalDisc::ContentTypes content = m_pointer->front().second;
290  return content;
291  }
292 
293  ~SharedContentTypesCache()
294  {
295  m_semaphore.acquire();
296  Releaser releaser(&m_semaphore);
297  m_shmem.detach();
298  }
