143 const Solid::DeviceInterface *Solid::Device::asDeviceInterface(const DeviceInterface::Type &type) const
144 {
145  Ifaces::Device *device = qobject_cast<Ifaces::Device *>(d->backendObject());
146 
147  if (device != 0) {
148  DeviceInterface *iface = d->interface(type);
149 
150  if (iface != 0) {
151  return iface;
152  }
153 
154  QObject *dev_iface = device->createDeviceInterface(type);
155 
156  if (dev_iface != 0) {
157  switch (type) {
158  case DeviceInterface::GenericInterface:
159  iface = deviceinterface_cast(Ifaces::GenericInterface, GenericInterface, dev_iface);
160  break;
161  case DeviceInterface::Processor:
162  iface = deviceinterface_cast(Ifaces::Processor, Processor, dev_iface);
163  break;
164  case DeviceInterface::Block:
165  iface = deviceinterface_cast(Ifaces::Block, Block, dev_iface);
166  break;
167  case DeviceInterface::StorageAccess:
168  iface = deviceinterface_cast(Ifaces::StorageAccess, StorageAccess, dev_iface);
169  break;
170  case DeviceInterface::StorageDrive:
171  iface = deviceinterface_cast(Ifaces::StorageDrive, StorageDrive, dev_iface);
172  break;
173  case DeviceInterface::OpticalDrive:
174  iface = deviceinterface_cast(Ifaces::OpticalDrive, OpticalDrive, dev_iface);
175  break;
176  case DeviceInterface::StorageVolume:
177  iface = deviceinterface_cast(Ifaces::StorageVolume, StorageVolume, dev_iface);
178  break;
179  case DeviceInterface::OpticalDisc:
180  iface = deviceinterface_cast(Ifaces::OpticalDisc, OpticalDisc, dev_iface);
181  break;
182  case DeviceInterface::Camera:
183  iface = deviceinterface_cast(Ifaces::Camera, Camera, dev_iface);
184  break;
185  case DeviceInterface::PortableMediaPlayer:
186  iface = deviceinterface_cast(Ifaces::PortableMediaPlayer, PortableMediaPlayer, dev_iface);
187  break;
188  case DeviceInterface::Battery:
189  iface = deviceinterface_cast(Ifaces::Battery, Battery, dev_iface);
190  break;
191  case DeviceInterface::NetworkShare:
192  iface = deviceinterface_cast(Ifaces::NetworkShare, NetworkShare, dev_iface);
193  break;
194  case DeviceInterface::Unknown:
195  case DeviceInterface::Last:
196  break;
197  }
198  }
199 
200  if (iface != 0) {
201  // Lie on the constness since we're simply doing caching here
202  const_cast<Device *>(this)->d->setInterface(type, iface);
203  iface->d_ptr->setDevicePrivate(d.data());
204  }
205 
206  return iface;
207  } else {
208  return 0;
209  }
210 }

233 void Solid::DevicePrivate::setBackendObject(Ifaces::Device *object)
234 {
235 
236  if (m_backendObject) {
237  m_backendObject.data()->disconnect(this);
238  }
239 
240  delete m_backendObject.data();
241  m_backendObject = object;
242 
243  if (object) {
244  connect(object, SIGNAL(destroyed(QObject*)),
245  this, SLOT(_k_destroyed(QObject*)));
246  }
247 
248  if (!m_ifaces.isEmpty()) {
249  Q_FOREACH (DeviceInterface *iface, m_ifaces) {
250  delete iface;
251  }
252 
253  m_ifaces.clear();
254  if (!ref.deref()) {
255  deleteLater();
256  }
257  }
258 }
