83 QObject *FstabManager::createDevice(const QString &udi)
84 {
85  if (udi == udiPrefix()) {
86  RootDevice *root = new RootDevice(FSTAB_UDI_PREFIX);
87 
88  root->setProduct(tr("Network Shares"));
89  root->setDescription(tr("NFS and SMB shares declared in your system"));
90  root->setIcon("folder-remote");
91 
92  return root;
93 
94  } else {
95  // global device manager makes sure udi starts with udi prefix + '/'
96  QString internalName = udi.mid(udiPrefix().length() + 1, -1);
97  if (!m_deviceList.contains(internalName)) {
98  return 0;
99  }
100 
101  QObject *device = new FstabDevice(udi);
102  connect(this, SIGNAL(mtabChanged(QString)), device, SLOT(onMtabChanged(QString)));
103  return device;
104 
105  }

114 void FstabManager::_k_updateDeviceList()
115 {
116  QStringList deviceList = FstabHandling::deviceList();
117  QSet<QString> newlist = deviceList.toSet();
118  QSet<QString> oldlist = m_deviceList.toSet();
119 
120  Q_FOREACH (const QString &device, newlist) {
121  if (!oldlist.contains(device)) {
122  emit deviceAdded(udiPrefix() + "/" + device);
123  }
124  }
125 
126  Q_FOREACH (const QString &device, oldlist) {
127  if (!newlist.contains(device)) {
128  emit deviceRemoved(udiPrefix() + "/" + device);
129  }
130  }
131 
132  m_deviceList = deviceList;
133 
134  Q_FOREACH (const QString &device, newlist) {
135  if (!oldlist.contains(device)) {
136  emit deviceAdded(udiPrefix() + "/" + device);
137  }
138  }
139 }
