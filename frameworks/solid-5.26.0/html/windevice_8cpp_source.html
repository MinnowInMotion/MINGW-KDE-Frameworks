43 WinDevice::WinDevice(const QString &udi) :
44  Device(),
45  m_udi(udi),
46  m_type(Solid::DeviceInterface::Unknown)
47 {
48  /*
49  /org/kde/solid/win/volume/disk #%1, partition #%2
50  /org/kde/solid/win/storage.cdrom/disk #%0
51  */
52  QStringList data = udi.split("/");
53  QString parentName = data[6].split(",")[0].trimmed();
54  QString type = data[5];
55 
56  if (type == "storage") {
57  m_type = Solid::DeviceInterface::StorageDrive;
58  } else if (type == "volume") {
59  m_type = Solid::DeviceInterface::StorageVolume;
60  } else if (type == "storage.cdrom") {
61  m_type = Solid::DeviceInterface::OpticalDrive;
62  } else if (type == "volume.cdrom") {
63  m_type = Solid::DeviceInterface::OpticalDisc;
64  } else if (type == "cpu") {
65  m_type = Solid::DeviceInterface::Processor;
66  } else if (type == "power.battery") {
67  m_type = Solid::DeviceInterface::Battery;
68  } else if (type == "volume.virtual") {
69  m_type = Solid::DeviceInterface::StorageAccess;
70  }
71 
72  switch (m_type) {
73  case Solid::DeviceInterface::StorageVolume: {
74  m_parentUdi = QLatin1String("/org/kde/solid/win/storage/") + parentName;
75  }
76  break;
77  case Solid::DeviceInterface::OpticalDisc: {
78  m_parentUdi = QLatin1String("/org/kde/solid/win/storage.cdrom/") + parentName;
79  }
80  break;
81  case Solid::DeviceInterface::StorageAccess: {
82  m_parentUdi = WinBlock::udiFromDriveLetter(WinBlock::resolveVirtualDrive(udi).mid(0, 2));
83  if (m_parentUdi.isEmpty()) {
84  m_parentUdi = QLatin1String("/org/kde/solid/win/") + type;
85  }
86  }
87  break;
88  default:
89  m_parentUdi = QLatin1String("/org/kde/solid/win/") + type;
90  }
91 
92  switch (m_type) {
93  case Solid::DeviceInterface::Processor:
94  initCpuDevice();
95  break;
96  case Solid::DeviceInterface::Battery:
97  initBatteryDevice();
98  break;
99  default:
100  if (queryDeviceInterface(Solid::DeviceInterface::StorageAccess) || queryDeviceInterface(Solid::DeviceInterface::StorageDrive)) {
101  initStorageDevice();
102  } else {
103  qWarning() << "Unknown device" << udi;
104  }
105  }
106 
107 }

109 void WinDevice::initStorageDevice()
110 {
111  QString dev;
112  switch (m_type) {
113  case Solid::DeviceInterface::StorageAccess:
114  dev = WinBlock::driveLetterFromUdi(udi());
115  m_product = QString("Virtual drive %1").arg(dev);
116  m_description = QString("%1 (%2)").arg(dev, WinBlock::resolveVirtualDrive(udi()));
117  return;
118  case Solid::DeviceInterface::OpticalDrive:
119  dev = WinBlock::driveLetterFromUdi(udi());
120  break;
121  case Solid::DeviceInterface::StorageDrive:
122  dev = QString("PhysicalDrive%1").arg(WinBlock(this).deviceMajor());
123  break;
124  default:
125  dev = WinBlock::driveLetterFromUdi(udi());
126  m_description = QString("%1 (%2)").arg(dev, WinStorageVolume(this).label());
127  }
128 
129  STORAGE_PROPERTY_QUERY query;
130  ZeroMemory(&query, sizeof(STORAGE_PROPERTY_QUERY));
131  query.PropertyId = StorageDeviceProperty;
132  query.QueryType = PropertyStandardQuery;
133 
134  char buff[1024];
135  WinDeviceManager::getDeviceInfo<char, STORAGE_PROPERTY_QUERY>(dev, IOCTL_STORAGE_QUERY_PROPERTY, buff, 1024, &query);
136  STORAGE_DEVICE_DESCRIPTOR *info = ((STORAGE_DEVICE_DESCRIPTOR *)buff);
137  if (info->VendorIdOffset != 0) {
138  m_vendor = QString((char *)buff + info->VendorIdOffset).trimmed();
139  if (info->ProductIdOffset != 0) {
140  m_product = QString((char *)buff + info->ProductIdOffset).trimmed();
141  }
142  } else if (info->ProductIdOffset != 0) { //fallback doesnt work for all devices
143  QStringList tmp = QString((char *)buff + info->ProductIdOffset).trimmed().split(" ");
144  m_vendor = tmp.takeFirst();
145  m_product = tmp.join(" ");
146  }
147 }

149 void WinDevice::initBatteryDevice()
150 {
151  WinBattery::Battery battery = WinBattery::batteryInfoFromUdi(m_udi);
152  BATTERY_QUERY_INFORMATION query;
153  ZeroMemory(&query, sizeof(query));
154  query.BatteryTag = battery.second;
155 
156  size_t size = 1024;
157  wchar_t buff[1024];
158 
159  query.InformationLevel = BatteryDeviceName;
160  WinDeviceManager::getDeviceInfo<wchar_t, BATTERY_QUERY_INFORMATION>(battery.first, IOCTL_BATTERY_QUERY_INFORMATION, buff, size, &query);
161  m_product = QString::fromWCharArray(buff);
162 
163  query.InformationLevel = BatteryManufactureName;
164  WinDeviceManager::getDeviceInfo<wchar_t, BATTERY_QUERY_INFORMATION>(battery.first, IOCTL_BATTERY_QUERY_INFORMATION, buff, size, &query);
165  m_vendor = QString::fromWCharArray(buff);
166 
167  switch (WinBattery(this).technology()) {
168  case Solid::Battery::LithiumIon:
169  m_description = tr("Lithium Ion", "battery technology");
170  break;
171  case Solid::Battery::LeadAcid:
172  m_description = tr("Lead Acid", "battery technology");
173  break;
174  case Solid::Battery::NickelCadmium:
175  m_description = tr("Nickel Cadmium", "battery technology");
176  break;
177  case Solid::Battery::NickelMetalHydride:
178  m_description = tr("Nickel Metal Hydride", "battery technology");
179  break;
180  default:
181  m_description = tr("Unknown", "battery technology");
182  }
183 }

219 QString WinDevice::icon() const
220 {
221  if (parentUdi().isEmpty()) {
222  return QLatin1String("computer");
223  }
224 
225  QString icon;
226  switch (type()) {
227  case Solid::DeviceInterface::OpticalDrive:
228  icon = QLatin1String("drive-optical");
229  break;
230  case Solid::DeviceInterface::OpticalDisc: {
231  WinOpticalDisc disk(const_cast<WinDevice *>(this));
232  if (disk.availableContent() & Solid::OpticalDisc::Audio) {
233  icon = QLatin1String("media-optical-audio");
234  } else {
235  icon = QLatin1String("drive-optical");
236  }
237  break;
238  }
239  case Solid::DeviceInterface::StorageDrive:
240  case Solid::DeviceInterface::StorageVolume: {
241  WinStorageDrive storage(const_cast<WinDevice *>(this));
242  if (storage.bus() == Solid::StorageDrive::Usb) {
243  icon = QLatin1String("drive-removable-media-usb-pendrive");
244  } else {
245  icon = QLatin1String("drive-harddisk");
246  }
247  }
248  break;
249  case Solid::DeviceInterface::Processor:
250  icon = QLatin1String("cpu");
251  break;
252  case Solid::DeviceInterface::Battery:
253  icon = QLatin1String("battery");
254  break;
255  case Solid::DeviceInterface::StorageAccess:
256  icon = QLatin1String("drive-harddisk");
257  break;
258  default:
259  break;
260  }
261  return icon;
262 }

277 bool WinDevice::queryDeviceInterface(const Solid::DeviceInterface::Type &queryType) const
278 {
279  if (queryType == Solid::DeviceInterface::GenericInterface) {
280  return true;
281  }
282 
283  QList<Solid::DeviceInterface::Type> interfaceList;
284  interfaceList << type();
285 
286  switch (type()) {
287  case Solid::DeviceInterface::GenericInterface:
288  break;
289  case Solid::DeviceInterface::Block:
290  break;
291  case Solid::DeviceInterface::StorageAccess:
292  break;
293  case Solid::DeviceInterface::StorageDrive:
294  break;
295  case Solid::DeviceInterface::OpticalDrive:
296  interfaceList << Solid::DeviceInterface::Block << Solid::DeviceInterface::StorageDrive;
297  break;
298  case Solid::DeviceInterface::StorageVolume:
299  interfaceList << Solid::DeviceInterface::Block << Solid::DeviceInterface::StorageAccess;
300  break;
301  case Solid::DeviceInterface::OpticalDisc:
302  interfaceList << Solid::DeviceInterface::Block << Solid::DeviceInterface::StorageVolume << Solid::DeviceInterface::StorageAccess;
303  break;
304  case Solid::DeviceInterface::PortableMediaPlayer:
305  break;
306  case Solid::DeviceInterface::Unknown:
307  case Solid::DeviceInterface::Last:
308  default:
309  break;
310  }
311 
312  if (interfaceList.size() == 0) {
313  qWarning() << "no interface found for type" << type();
314  }
315  return interfaceList.contains(queryType);
316 }
