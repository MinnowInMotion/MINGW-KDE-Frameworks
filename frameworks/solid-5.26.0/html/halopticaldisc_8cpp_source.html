36 Solid::OpticalDisc::ContentTypes OpticalDisc::availableContent() const
37 {
38  Solid::OpticalDisc::ContentTypes content;
39 
40  QMap<Solid::OpticalDisc::ContentType, QString> map;
41  map[Solid::OpticalDisc::Audio] = "volume.disc.has_audio";
42  map[Solid::OpticalDisc::Data] = "volume.disc.has_data";
43  map[Solid::OpticalDisc::VideoCd] = "volume.disc.is_vcd";
44  map[Solid::OpticalDisc::SuperVideoCd] = "volume.disc.is_svcd";
45  map[Solid::OpticalDisc::VideoDvd] = "volume.disc.is_videodvd";
46  map[Solid::OpticalDisc::VideoBluRay] = "volume.disc.is_blurayvideo";
47 
48  Q_FOREACH (const Solid::OpticalDisc::ContentType type, map.keys()) {
49  if (m_device->prop(map[type]).toBool()) {
50  content |= type;
51  }
52  }
53 
54  return content;
55 }

57 Solid::OpticalDisc::DiscType OpticalDisc::discType() const
58 {
59  QString type = m_device->prop("volume.disc.type").toString();
60 
61  if (type == "cd_rom") {
62  return Solid::OpticalDisc::CdRom;
63  } else if (type == "cd_r") {
64  return Solid::OpticalDisc::CdRecordable;
65  } else if (type == "cd_rw") {
66  return Solid::OpticalDisc::CdRewritable;
67  } else if (type == "dvd_rom") {
68  return Solid::OpticalDisc::DvdRom;
69  } else if (type == "dvd_ram") {
70  return Solid::OpticalDisc::DvdRam;
71  } else if (type == "dvd_r") {
72  return Solid::OpticalDisc::DvdRecordable;
73  } else if (type == "dvd_rw") {
74  return Solid::OpticalDisc::DvdRewritable;
75  } else if (type == "dvd_plus_r") {
76  return Solid::OpticalDisc::DvdPlusRecordable;
77  } else if (type == "dvd_plus_rw") {
78  return Solid::OpticalDisc::DvdPlusRewritable;
79  } else if (type == "dvd_plus_r_dl") {
80  return Solid::OpticalDisc::DvdPlusRecordableDuallayer;
81  } else if (type == "dvd_plus_rw_dl") {
82  return Solid::OpticalDisc::DvdPlusRewritableDuallayer;
83  } else if (type == "bd_rom") {
84  return Solid::OpticalDisc::BluRayRom;
85  } else if (type == "bd_r") {
86  return Solid::OpticalDisc::BluRayRecordable;
87  } else if (type == "bd_re") {
88  return Solid::OpticalDisc::BluRayRewritable;
89  } else if (type == "hddvd_rom") {
90  return Solid::OpticalDisc::HdDvdRom;
91  } else if (type == "hddvd_r") {
92  return Solid::OpticalDisc::HdDvdRecordable;
93  } else if (type == "hddvd_rw") {
94  return Solid::OpticalDisc::HdDvdRewritable;
95  } else {
96  return Solid::OpticalDisc::UnknownDiscType;
97  }
98 }
