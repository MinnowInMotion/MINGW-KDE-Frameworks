96 Solid::Battery::ChargeState Battery::chargeState() const
97 {
98  Solid::Battery::ChargeState result = Solid::Battery::NoCharge;
99  const uint state = m_device.data()->prop("State").toUInt();
100  switch (state) {
101  case 0:
102  result = Solid::Battery::NoCharge; // stable or unknown
103  break;
104  case 1:
105  result = Solid::Battery::Charging;
106  break;
107  case 2:
108  result = Solid::Battery::Discharging;
109  break;
110  case 3: // TODO "Empty"
111  break;
112  case 4:
113  result = Solid::Battery::FullyCharged;
114  break;
115  case 5: // TODO "Pending charge"
116  break;
117  case 6: // TODO "Pending discharge"
118  break;
119  }
120  return result;
121 }

204 qlonglong Battery::remainingTime() const
205 {
206  if (chargeState() == Solid::Battery::Charging) {
207  return timeToFull();
208  } else if (chargeState() == Solid::Battery::Discharging) {
209  return timeToEmpty();
210  }
211 
212  return -1;

215 void Battery::slotChanged()
216 {
217  if (m_device) {
218  const bool old_isPresent = m_isPresent;
219  const int old_chargePercent = m_chargePercent;
220  const int old_capacity = m_capacity;
221  const bool old_isPowerSupply = m_isPowerSupply;
222  const Solid::Battery::ChargeState old_chargeState = m_chargeState;
223  const qlonglong old_timeToEmpty = m_timeToEmpty;
224  const qlonglong old_timeToFull = m_timeToFull;
225  const double old_energy = m_energy;
226  const double old_energyFull = m_energyFull;
227  const double old_energyFullDesign = m_energyFullDesign;
228  const double old_energyRate = m_energyRate;
229  const double old_voltage = m_voltage;
230  const double old_temperature = m_temperature;
231  updateCache();
232 
233  if (old_isPresent != m_isPresent) {
234  emit presentStateChanged(m_isPresent, m_device.data()->udi());
235  }
236 
237  if (old_chargePercent != m_chargePercent) {
238  emit chargePercentChanged(m_chargePercent, m_device.data()->udi());
239  }
240 
241  if (old_capacity != m_capacity) {
242  emit capacityChanged(m_capacity, m_device.data()->udi());
243  }
244 
245  if (old_isPowerSupply != m_isPowerSupply) {
246  emit powerSupplyStateChanged(m_isPowerSupply, m_device.data()->udi());
247  }
248 
249  if (old_chargeState != m_chargeState) {
250  emit chargeStateChanged(m_chargeState, m_device.data()->udi());
251  }
252 
253  if (old_timeToEmpty != m_timeToEmpty) {
254  emit timeToEmptyChanged(m_timeToEmpty, m_device.data()->udi());
255  }
256 
257  if (old_timeToFull != m_timeToFull) {
258  emit timeToFullChanged(m_timeToFull, m_device.data()->udi());
259  }
260 
261  if (old_energy != m_energy) {
262  emit energyChanged(m_energy, m_device.data()->udi());
263  }
264 
265  if (old_energyFull != m_energyFull) {
266  emit energyFullChanged(m_energyFull, m_device.data()->udi());
267  }
268 
269  if (old_energyFullDesign != m_energyFullDesign) {
270  emit energyFullChanged(m_energyFullDesign, m_device.data()->udi());
271  }
272 
273  if (old_energyRate != m_energyRate) {
274  emit energyRateChanged(m_energyRate, m_device.data()->udi());
275  }
276 
277  if (old_voltage != m_voltage) {
278  emit voltageChanged(m_voltage, m_device.data()->udi());
279  }
280 
281  if (old_temperature != m_temperature) {
282  emit temperatureChanged(m_temperature, m_device.data()->udi());
283  }
284 
285  if (old_timeToFull != m_timeToFull || old_timeToEmpty != m_timeToEmpty) {
