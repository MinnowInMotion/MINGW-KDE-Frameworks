95 bool StorageAccess::isAccessible() const
96 {
97  if (m_device->prop("info.interfaces").toStringList().contains("org.freedesktop.Hal.Device.Volume.Crypto")) {
98 
99  // Might be a bit slow, but I see no cleaner way to do this with HAL...
100  QDBusInterface manager("org.freedesktop.Hal",
101  "/org/freedesktop/Hal/Manager",
102  "org.freedesktop.Hal.Manager",
103  QDBusConnection::systemBus());
104 
105  QDBusReply<QStringList> reply = manager.call("FindDeviceStringMatch",
106  "volume.crypto_luks.clear.backing_volume",
107  m_device->udi());
108 
109  QStringList list = reply;
110 
111  return reply.isValid() && !list.isEmpty();
112 
113  } else {
114  return m_device->prop("volume.is_mounted").toBool();
115  }
116 }

133 bool StorageAccess::isIgnored() const
134 {
135  HalDevice lock("/org/freedesktop/Hal/devices/computer");
136  bool isLocked = lock.prop("info.named_locks.Global.org.freedesktop.Hal.Device.Storage.locked").toBool();
137 
138  if (m_device->prop("volume.ignore").toBool() || isLocked) {
139  return true;
140  }
141 
142  const QString mount_point = StorageAccess(m_device).filePath();
143  const bool mounted = m_device->prop("volume.is_mounted").toBool();
144  if (!mounted) {
145  return false;
146  } else if (mount_point.startsWith(QLatin1String("/media/")) || mount_point.startsWith(QLatin1String("/mnt/"))) {
147  return false;
148  }
149 
150  /* Now be a bit more aggressive on what we want to ignore,
151  * the user generally need to check only what's removable or in /media
152  * the volumes mounted to make the system (/, /boot, /var, etc.)
153  * are useless to him.
154  */
155  HalDevice drive(m_device->prop("block.storage_device").toString());
156  Hal::GenericInterface gif(&drive);
157 
158  const bool removable = gif.property("storage.removable").toBool();
159  const bool hotpluggable = gif.property("storage.hotpluggable").toBool();
160 
161  return !removable && !hotpluggable;
162 }

164 bool StorageAccess::setup()
165 {
166  if (m_teardownInProgress || m_setupInProgress || isAccessible()) {
167  return false;
168  }
169  m_setupInProgress = true;
170  m_device->broadcastActionRequested("setup");
171 
172  if (m_device->prop("info.interfaces").toStringList().contains("org.freedesktop.Hal.Device.Volume.Crypto")) {
173  return requestPassphrase();
174  } else if (FstabHandling::isInFstab(m_device->prop("block.device").toString())) {
175  return callSystemMount();
176  } else {
177  return callHalVolumeMount();
178  }
179 }

205 void StorageAccess::slotDBusReply(const QDBusMessage &/*reply*/)
206 {

208  m_setupInProgress = false;
209  m_device->broadcastActionDone("setup");
210  } else if (m_teardownInProgress) {
211  m_teardownInProgress = false;
212  m_device->broadcastActionDone("teardown");
213 
214  HalDevice drive(m_device->prop("block.storage_device").toString());
215  if (drive.prop("storage.drive_type").toString() != "cdrom"
216  && drive.prop("storage.requires_eject").toBool()) {
217 
218  QString devnode = m_device->prop("block.device").toString();
219 
220 #if defined(Q_OS_OPENBSD)
221  QString program = "cdio";
222  QStringList args;
223  args << "-f" << devnode << "eject";
224 #elif defined(Q_OS_FREEBSD) || defined(Q_OS_NETBSD)
225  devnode.remove("/dev/").replace("([0-9]).", "\\1");
226  QString program = "cdcontrol";
227  QStringList args;
228  args << "-f" << devnode << "eject";
229 #else
230  QString program = "eject";
231  QStringList args;
232  args << devnode;
233 #endif
234 
235  m_ejectInProgress = true;
236  m_device->broadcastActionRequested("eject");
237  m_process = FstabHandling::callSystemCommand("eject", args,
238  this, SLOT(slotProcessFinished(int,QProcess::ExitStatus)));
239  }
240  } else if (m_ejectInProgress) {
241  m_ejectInProgress = false;
242  m_device->broadcastActionDone("eject");
243  }

329  QWidget *activeWindow = QApplication::activeWindow();

362 bool StorageAccess::callHalVolumeMount()
363 {
364  QDBusConnection c = QDBusConnection::systemBus();
365  QString udi = m_device->udi();
366  QDBusMessage msg = QDBusMessage::createMethodCall("org.freedesktop.Hal", udi,
367  "org.freedesktop.Hal.Device.Volume",
368  "Mount");
369 
370  // HAL 0.5.12 supports using alternative drivers for the same filesystem.
371  // This is mainly used to integrate the ntfs-3g driver.
372  // Unfortunately, the primary driver gets used unless we
373  // specify some other driver (fstype) to the Mount method.
374  // TODO: Allow the user to choose the driver.
375 
376  QString fstype = m_device->prop("volume.fstype").toString();
377  QStringList halOptions = m_device->prop("volume.mount.valid_options").toStringList();
378 
379  QString alternativePreferred = m_device->prop("volume.fstype.alternative.preferred").toString();
380  if (!alternativePreferred.isEmpty()) {
381  QStringList alternativeFstypes = m_device->prop("volume.fstype.alternative").toStringList();
382  if (alternativeFstypes.contains(alternativePreferred)) {
383  fstype = alternativePreferred;
384  halOptions = m_device->prop("volume.mount." + fstype + ".valid_options").toStringList();
385  }
386  }
387 
388  QStringList options;
389 
390 #ifdef Q_OS_FREEBSD
391  QString uid = "-u=";
392 #else
393  QString uid = "uid=";
394 #endif
395  if (halOptions.contains(uid)) {
396  options << uid + QString::number(::getuid());
397  }
398 
399 #ifdef Q_OS_FREEBSD
400  char *cType;
401  if (fstype == "vfat" && halOptions.contains("-L=")) {
402  if ((cType = getenv("LC_ALL")) || (cType = getenv("LC_CTYPE")) || (cType = getenv("LANG"))) {
403  options << "-L=" + QString(cType);
404  }
405  } else if ((fstype.startsWith(QLatin1String("ntfs")) || fstype == "iso9660" || fstype == "udf") && halOptions.contains("-C=")) {
406  if ((cType = getenv("LC_ALL")) || (cType = getenv("LC_CTYPE")) || (cType = getenv("LANG"))) {
407  options << "-C=" + QString(nl_langinfo(CODESET));
408  }
409  }
410 #else
411  if (fstype == "vfat" || fstype == "ntfs" || fstype == "iso9660" || fstype == "udf") {
412  if (halOptions.contains("utf8")) {
413  options << "utf8";
414  } else if (halOptions.contains("iocharset=")) {
415  options << "iocharset=utf8";
416  }
417  if (halOptions.contains("shortname=")) {
418  options << "shortname=mixed";
419  }
420  if (halOptions.contains("flush")) {
421  options << "flush";
422  }
423  }
424  // pass our locale to the ntfs-3g driver so it can translate local characters
425  else if (halOptions.contains("locale=")) {
426  // have to obtain LC_CTYPE as returned by the `locale` command
427  // check in the same order as `locale` does
428  char *cType;
429  if ((cType = getenv("LC_ALL")) || (cType = getenv("LC_CTYPE")) || (cType = getenv("LANG"))) {
430  options << "locale=" + QString(cType);
431  }
432  }
433 #endif
434 
435  msg << "" << fstype << options;
436 
437  return c.callWithCallback(msg, this,
438  SLOT(slotDBusReply(QDBusMessage)),
439  SLOT(slotDBusError(QDBusError)));
440 }
