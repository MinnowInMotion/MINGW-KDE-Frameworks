85 bool UPowerDevice::queryDeviceInterface(const Solid::DeviceInterface::Type &type) const
86 {
87  const uint uptype = prop("Type").toUInt();
88  switch (type) {
89  case Solid::DeviceInterface::GenericInterface:
90  return true;
91  case Solid::DeviceInterface::Battery:
92  return (uptype == 2 || uptype == 3 || uptype == 5 || uptype == 6 || uptype == 7 || uptype == 8);
93  default:
94  return false;

172 void UPowerDevice::checkCache(const QString &key) const
173 {
174  if (m_cache.isEmpty()) { // recreate the cache
175  allProperties();
176  }
177 
178  if (m_cache.contains(key)) {
179  return;
180  }
181 
182  QVariant reply = m_device.property(key.toUtf8());
183 
184  if (reply.isValid()) {
185  m_cache[key] = reply;
186  } else {
187  m_cache[key] = QVariant();
188  }
189 }

203 QMap<QString, QVariant> UPowerDevice::allProperties() const
204 {
205  QDBusMessage call = QDBusMessage::createMethodCall(m_device.service(), m_device.path(),
206  "org.freedesktop.DBus.Properties", "GetAll");
207  call << m_device.interface();
208  QDBusPendingReply< QVariantMap > reply = QDBusConnection::systemBus().asyncCall(call);
209  reply.waitForFinished();
210 
211  if (reply.isValid()) {
212  m_cache = reply.value();
213  } else {
214  m_cache.clear();
215  }
216 
217  return m_cache;
218 }
