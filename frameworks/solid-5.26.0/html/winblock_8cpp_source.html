74 QStringList WinBlock::drivesFromMask(const DWORD unitmask)
75 {
76  QStringList result;
77  DWORD localUnitmask(unitmask);
78  for (int i = 0; i <= 25; ++i) {
79  if (0x01 == (localUnitmask & 0x1)) {
80  result << QString("%1:").arg((char)(i + 'A'));
81  }
82  localUnitmask >>= 1;
83  }
84  return result;

117 QSet<QString> WinBlock::updateUdiFromBitMask(const DWORD unitmask)
118 {
119  QStringList drives = drivesFromMask(unitmask);
120  QSet<QString> list;
121  wchar_t driveWCHAR[MAX_PATH];
122  wchar_t bufferOut[MAX_PATH];
123  QString dosPath;
124  foreach (const QString &drive, drives) {
125  QSet<QString> udis;
126  driveWCHAR[drive.toWCharArray(driveWCHAR)] = 0;
127  if (GetDriveType(driveWCHAR) == DRIVE_REMOTE) { //network drive
128  QSettings settings(QLatin1String("HKEY_CURRENT_USER\\Network\\") + drive.at(0), QSettings::NativeFormat);
129  QString path = settings.value("RemotePath").toString();
130  if (!path.isEmpty()) {
131  QString key = QLatin1String("/org/kde/solid/win/volume.virtual/") + drive.at(0);
132  m_virtualDrives[key] = path;
133  udis << key;
134  }
135 
136  } else {
137  QueryDosDeviceW(driveWCHAR, bufferOut, MAX_PATH);
138  dosPath = QString::fromWCharArray(bufferOut);
139  if (dosPath.startsWith("\\??\\")) { //subst junction
140  dosPath = dosPath.mid(4);
141  QString key = QLatin1String("/org/kde/solid/win/volume.virtual/") + drive.at(0);
142  m_virtualDrives[key] = dosPath;
143  udis << key;
144  } else {
145  STORAGE_DEVICE_NUMBER info = WinDeviceManager::getDeviceInfo<STORAGE_DEVICE_NUMBER>(drive, IOCTL_STORAGE_GET_DEVICE_NUMBER);
146 
147  switch (info.DeviceType) {
148  case FILE_DEVICE_DISK: {
149  udis << QString("/org/kde/solid/win/volume/disk#%1,partition#%2").arg(info.DeviceNumber).arg(info.PartitionNumber);
150  udis << QString("/org/kde/solid/win/storage/disk#%1").arg(info.DeviceNumber);
151 
152  }
153  break;
154  case FILE_DEVICE_CD_ROM:
155  case FILE_DEVICE_DVD: {
156  udis << QString("/org/kde/solid/win/storage.cdrom/disk#%1").arg(info.DeviceNumber);
157  DISK_GEOMETRY_EX out = WinDeviceManager::getDeviceInfo<DISK_GEOMETRY_EX>(drive, IOCTL_DISK_GET_DRIVE_GEOMETRY_EX);
158  if (out.DiskSize.QuadPart != 0) {
159  udis << QString("/org/kde/solid/win/volume.cdrom/disk#%1").arg(info.DeviceNumber);
160  }
161  }
162  break;
163  default:
164  qDebug() << "unknown device" << drive << info.DeviceType << info.DeviceNumber << info.PartitionNumber;
165  }
166  }
167  }
168  m_driveUDIS[drive] = udis;
169  foreach (const QString &s, udis) {
170  m_driveLetters[s] = drive;
171  }
172  list += udis;
173  }
174  return list;
175 }
