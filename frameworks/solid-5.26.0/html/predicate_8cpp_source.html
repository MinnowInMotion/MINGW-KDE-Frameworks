183 bool Solid::Predicate::matches(const Device &device) const
184 {
185  if (!d->isValid) {
186  return false;
187  }
188 
189  switch (d->type) {
190  case Disjunction:
191  return d->operand1->matches(device)
192  || d->operand2->matches(device);
193  case Conjunction:
194  return d->operand1->matches(device)
195  && d->operand2->matches(device);
196  case PropertyCheck: {
197  const DeviceInterface *iface = device.asDeviceInterface(d->ifaceType);
198 
199  if (iface != 0) {
200  const int index = iface->metaObject()->indexOfProperty(d->property.toLatin1());
201  QMetaProperty metaProp = iface->metaObject()->property(index);
202  QVariant value = metaProp.isReadable() ? metaProp.read(iface) : QVariant();
203  QVariant expected = d->value;
204 
205  if (metaProp.isEnumType() && expected.type() == QVariant::String) {
206  QMetaEnum metaEnum = metaProp.enumerator();
207  int value = metaEnum.keysToValue(d->value.toString().toLatin1());
208  if (value >= 0) { // No value found for these keys, resetting expected to invalid
209  expected = value;
210  } else {
211  expected = QVariant();
212  }
213  }
214 
215  if (d->compOperator == Mask) {
216  bool v_ok;
217  int v = value.toInt(&v_ok);
218  bool e_ok;
219  int e = expected.toInt(&e_ok);
220 
221  return (e_ok && v_ok && (v & e));
222  } else {
223  return (value == expected);
224  }
225  }
226  break;
227  }
228  case InterfaceCheck:
229  return device.isDeviceInterface(d->ifaceType);
230  }
231 
232  return false;
233 }

258 QString Solid::Predicate::toString() const
259 {
260  if (!d->isValid) {
261  return "False";
262  }
263 
264  if (d->type != PropertyCheck && d->type != InterfaceCheck) {
265  QString op = " AND ";
266  if (d->type == Disjunction) {
267  op = " OR ";
268  }
269 
270  return '[' + d->operand1->toString() + op + d->operand2->toString() + ']';
271  } else {
272  QString ifaceName = DeviceInterface::typeToString(d->ifaceType);
273 
274  if (ifaceName.isEmpty()) {
275  ifaceName = "Unknown";
276  }
277 
278  if (d->type == InterfaceCheck) {
279  return "IS " + ifaceName;
280  }
281 
282  QString value;
283 
284  switch (d->value.type()) {
285  case QVariant::StringList: {
286  value = '{';
287 
288  const QStringList list = d->value.toStringList();
289 
290  QStringList::ConstIterator it = list.begin();
291  QStringList::ConstIterator end = list.end();
292 
293  for (; it != end; ++it) {
294  value += '\'' + *it + '\'';
295 
296  if (it + 1 != end) {
297  value += ", ";
298  }
299  }
300 
301  value += '}';
302  break;
303  }
304  case QVariant::Bool:
305  value = (d->value.toBool() ? "true" : "false");
306  break;
307  case QVariant::Int:
308  case QVariant::UInt:
309  case QVariant::LongLong:
310  case QVariant::ULongLong:
311  value = d->value.toString();
312  break;
313  default:
314  value = '\'' + d->value.toString() + '\'';
315  break;
316  }
317 
318  QString str_operator = "==";
319  if (d->compOperator != Equals) {
320  str_operator = " &";
321  }
322 
323  return ifaceName + '.' + d->property + ' ' + str_operator + ' ' + value;
324  }
