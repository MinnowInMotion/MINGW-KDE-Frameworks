57 nsProbingState nsEscCharSetProber::HandleData(const char *aBuf, unsigned int aLen)
58 {
59  nsSMState codingState;
60  int j;
61  unsigned int i;
62 
63  for (i = 0; i < aLen && mState == eDetecting; i++) {
64  for (j = mActiveSM - 1; j >= 0; j--) {
65  //byte is feed to all active state machine
66  codingState = mCodingSM[j]->NextState(aBuf[i]);
67  if (codingState == eError) {
68  //got negative answer for this state machine, make it inactive
69  mActiveSM--;
70  if (mActiveSM == 0) {
71  mState = eNotMe;
72  return mState;
73  } else if (j != (int)mActiveSM) {
74  nsCodingStateMachine *t;
75  t = mCodingSM[mActiveSM];
76  mCodingSM[mActiveSM] = mCodingSM[j];
77  mCodingSM[j] = t;
78  }
79  } else if (codingState == eItsMe) {
80  mState = eFoundIt;
81  mDetectedCharset = mCodingSM[j]->GetCodingStateMachine();
82  return mState;
83  }
84  }
85  }
86 
87  return mState;
88 }
89 }
