47  * and have to use some Stastics methods.

50  */

92  void unicodeTest(const char *aBuf, int aLen)
93  {
94  if (mStart) {
95  mStart = false;
96  if (aLen > 3)
97  switch (aBuf[0]) {
98  case '\xEF':
99  if (('\xBB' == aBuf[1]) && ('\xBF' == aBuf[2]))
100  // EF BB BF UTF-8 encoded BOM
101  {
102  proberState = KEncodingProber::FoundIt;
103  }
104  break;
105  case '\xFE':
106  if (('\xFF' == aBuf[1]) && ('\x00' == aBuf[2]) && ('\x00' == aBuf[3]))
107  // FE FF 00 00 UCS-4, unusual octet order BOM (3412)
108  {
109  proberState = KEncodingProber::FoundIt;
110  } else if ('\xFF' == aBuf[1])
111  // FE FF UTF-16, big endian BOM
112  {
113  proberState = KEncodingProber::FoundIt;
114  }
115  break;
116  case '\x00':
117  if (('\x00' == aBuf[1]) && ('\xFE' == aBuf[2]) && ('\xFF' == aBuf[3]))
118  // 00 00 FE FF UTF-32, big-endian BOM
119  {
120  proberState = KEncodingProber::FoundIt;
121  } else if (('\x00' == aBuf[1]) && ('\xFF' == aBuf[2]) && ('\xFE' == aBuf[3]))
122  // 00 00 FF FE UCS-4, unusual octet order BOM (2143)
123  {
124  proberState = KEncodingProber::FoundIt;
125  }
126  break;
127  case '\xFF':
128  if (('\xFE' == aBuf[1]) && ('\x00' == aBuf[2]) && ('\x00' == aBuf[3]))
129  // FF FE 00 00 UTF-32, little-endian BOM
130  {
131  proberState = KEncodingProber::FoundIt;
132  } else if ('\xFE' == aBuf[1])
133  // FF FE UTF-16, little endian BOM
134  {
135  proberState = KEncodingProber::FoundIt;
136  }
137  break;
138  } // switch
139 
140  }
141  }
142  KEncodingProber::ProberType proberType;
143  KEncodingProber::ProberState proberState;
144  kencodingprober::nsCharSetProber *prober;
145  bool mStart;
146 };

169 KEncodingProber::ProberState KEncodingProber::feed(const char *data, int len)
170 {
171  if (!d->prober) {
172  return d->proberState;
173  }
174  if (d->proberState == Probing) {
175  if (d->mStart) {
176  d->unicodeTest(data, len);
177  if (d->proberState == FoundIt) {
178  return d->proberState;
179  }
180  }
181  d->prober->HandleData(data, len);
182  switch (d->prober->GetState()) {
183  case kencodingprober::eNotMe:
184  d->proberState = NotMe;
185  break;
186  case kencodingprober::eFoundIt:
187  d->proberState = FoundIt;
188  break;
189  default:
190  d->proberState = Probing;
191  break;
192  }
193  }
194 #ifdef DEBUG_PROBE
195  d->prober->DumpStatus();
196 #endif
197  return d->proberState;
198 }
