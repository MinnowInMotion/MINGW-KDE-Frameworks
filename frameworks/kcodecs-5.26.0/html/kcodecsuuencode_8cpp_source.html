99 void UUDecoder::searchForBegin(const char *&scursor, const char *const send)
100 {
101  static const char begin[] = "begin\n";
102  static const uint beginLength = 5; // sic!
103 
104  assert(!mSawBegin || mIntoBeginLine > 0);
105 
106  while (scursor != send) {
107  uchar ch = *scursor++;
108  if (ch == begin[mIntoBeginLine]) {
109  if (mIntoBeginLine < beginLength) {
110  // found another char
111  ++mIntoBeginLine;
112  if (mIntoBeginLine == beginLength) {
113  mSawBegin = true; // "begin" complete, now search the next \n...
114  }
115  } else { // mIntoBeginLine == beginLength
116  // found '\n': begin line complete
117  mLastWasCRLF = true;
118  mIntoBeginLine = 0;
119  return;
120  }
121  } else if (mSawBegin) {
122  // OK, skip stuff until the next \n
123  } else {
124  //qWarning() << "UUDecoder: garbage before \"begin\", resetting parser";
125  mIntoBeginLine = 0;
126  }
127  }
128 
129 }

131 // uuencoding just shifts all 6-bit octets by 32 (SP/' '), except NUL,
132 // which gets mapped to 0x60
133 static inline uchar uuDecode(uchar c)
134 {
135  return (c - ' ') // undo shift and
136  & 0x3F; // map 0x40 (0x60-' ') to 0...

139 bool UUDecoder::decode(const char *&scursor, const char *const send,
140  char *&dcursor, const char *const dend)
141 {
142  // First, check whether we still need to find the "begin" line:
143  if (!mSawBegin || mIntoBeginLine != 0) {
144  searchForBegin(scursor, send);
145  } else if (mSawEnd) {
146  // or if we are past the end line:
147  scursor = send; // do nothing anymore...
148  return true;
149  }
150 
151  while (dcursor != dend && scursor != send) {
152  uchar ch = *scursor++;
153  uchar value;
154 
155  // Check whether we need to look for the "end" line:
156  if (mIntoEndLine > 0) {
157  static const char end[] = "end";
158  static const uint endLength = 3;
159 
160  if (ch == end[mIntoEndLine]) {
161  ++mIntoEndLine;
162  if (mIntoEndLine == endLength) {
163  mSawEnd = true;
164  scursor = send; // shortcut to the end
165  return true;
166  }
167  continue;
168  } else {
169  //qWarning() << "UUDecoder: invalid line octet count looks like \"end\" (mIntoEndLine ="
170  // << mIntoEndLine << ")!";
171  mIntoEndLine = 0;
172  // fall through...
173  }
174  }
175 
176  // Normal parsing:
177 
178  // The first char of a line is an encoding of the length of the
179  // current line. We simply ignore it:
180  if (mLastWasCRLF) {
181  // reset char-per-line counter:
182  mLastWasCRLF = false;
183  mCurrentOctetCount = 0;
184 
185  // try to decode the chars-on-this-line announcement:
186  if (ch == 'e') { // maybe the beginning of the "end"? ;-)
187  mIntoEndLine = 1;
188  } else if (ch > 0x60) {
189  // ### invalid line length char: what shall we do??
190  } else if (ch > ' ') {
191  mAnnouncedOctetCount = uuDecode(ch);
192  } else if (ch == '\n') {
193  mLastWasCRLF = true; // oops, empty line
194  }
195 
196  continue;
197  }
198 
199  // try converting ch to a 6-bit value:
200  if (ch > 0x60) {
201  continue; // invalid char
202  } else if (ch > ' ') {
203  value = uuDecode(ch);
204  } else if (ch == '\n') { // line end
205  mLastWasCRLF = true;
206  continue;
207  } else {
208  continue;
209  }
210 
211  // add the new bits to the output stream and flush full octets:
212  switch (mStepNo) {
213  case 0:
214  mOutbits = value << 2;
215  break;
216  case 1:
217  if (mCurrentOctetCount < mAnnouncedOctetCount) {
218  *dcursor++ = (char)(mOutbits | value >> 4);
219  }
220  ++mCurrentOctetCount;
221  mOutbits = value << 4;
222  break;
223  case 2:
224  if (mCurrentOctetCount < mAnnouncedOctetCount) {
225  *dcursor++ = (char)(mOutbits | value >> 2);
226  }
227  ++mCurrentOctetCount;
228  mOutbits = value << 6;
229  break;
230  case 3:
231  if (mCurrentOctetCount < mAnnouncedOctetCount) {
232  *dcursor++ = (char)(mOutbits | value);
233  }
234  ++mCurrentOctetCount;
235  mOutbits = 0;
236  break;
237  default:
238  assert(0);
239  }
240  mStepNo = (mStepNo + 1) % 4;
241 
242  // check whether we ran over the announced octet count for this line:
243  if (mCurrentOctetCount == mAnnouncedOctetCount + 1) {
244  // qWarning()
245  // << "UUDecoder: mismatch between announced ("
246  // << mAnnouncedOctetCount << ") and actual line octet count!";
247  }
248 
249  }
250 
251  // return false when caller should call us again:
252  return scursor == send;
253 } // UUDecoder::decode()
