87 #define MINIMUM_THRESHOLD (float)0.20

89 nsProbingState nsUniversalDetector::HandleData(const char *aBuf, unsigned int aLen)
90 {
91  if (mDone) {
92  return eFoundIt;
93  }
94 
95  if (aLen > 0) {
96  mGotData = true;
97  }
98 
99  unsigned int i;
100  for (i = 0; i < aLen; i++) {
101  //other than 0xa0, if every othe character is ascii, the page is ascii
102  if (aBuf[i] & '\x80' && aBuf[i] != '\xA0') { //Since many Ascii only page contains NBSP
103  //we got a non-ascii byte (high-byte)
104  if (mInputState != eHighbyte) {
105  //adjust state
106  mInputState = eHighbyte;
107 
108  //kill mEscCharSetProber if it is active
109  delete mEscCharSetProber;
110  mEscCharSetProber = 0;
111 
112  //start multibyte and singlebyte charset prober
113  if (0 == mCharSetProbers[0]) {
114  mCharSetProbers[0] = new nsMBCSGroupProber;
115  }
116  if (0 == mCharSetProbers[1]) {
117  mCharSetProbers[1] = new nsSBCSGroupProber;
118  }
119  if (0 == mCharSetProbers[2]) {
120  mCharSetProbers[2] = new nsLatin1Prober;
121  }
122  }
123  } else {
124  //ok, just pure ascii so far
125  if (ePureAscii == mInputState &&
126  (aBuf[i] == '\033' || (aBuf[i] == '{' && mLastChar == '~'))) {
127  //found escape character or HZ "~{"
128  mInputState = eEscAscii;
129  }
130 
131  mLastChar = aBuf[i];
132  }
133  }
134 
135  nsProbingState st = eDetecting;
136  switch (mInputState) {
137  case eEscAscii:
138  if (0 == mEscCharSetProber) {
139  mEscCharSetProber = new nsEscCharSetProber;
140  }
141  st = mEscCharSetProber->HandleData(aBuf, aLen);
142  if (st == eFoundIt) {
143  mDone = true;
144  mDetectedCharset = mEscCharSetProber->GetCharSetName();
145  }
146  break;
147  case eHighbyte:
148  for (i = 0; i < NUM_OF_CHARSET_PROBERS; ++i) {
149  st = mCharSetProbers[i]->HandleData(aBuf, aLen);
150  if (st == eFoundIt) {
151  mDone = true;
152  mDetectedCharset = mCharSetProbers[i]->GetCharSetName();
153  }
154  }
155  break;
156 
157  default: //pure ascii
158  mDetectedCharset = "UTF-8";
159  }
160  return st;
161 }

163 //---------------------------------------------------------------------
164 const char *nsUniversalDetector::GetCharSetName()
165 {
166  if (mDetectedCharset) {
167  return mDetectedCharset;
168  }
169  switch (mInputState) {
170  case eHighbyte: {
171  float proberConfidence;
172  float maxProberConfidence = (float)0.0;
173  int maxProber = 0;
174 
175  for (int i = 0; i < NUM_OF_CHARSET_PROBERS; i++) {
176  proberConfidence = mCharSetProbers[i]->GetConfidence();
177  if (proberConfidence > maxProberConfidence) {
178  maxProberConfidence = proberConfidence;
179  maxProber = i;
180  }
181  }
182  //do not report anything because we are not confident of it, that's in fact a negative answer
183  if (maxProberConfidence > MINIMUM_THRESHOLD) {
184  return mCharSetProbers[maxProber]->GetCharSetName();
185  }
186  }
187  case eEscAscii:
188  break;
189  default: // pure ascii
190  ;
191  }
192  return "UTF-8";
193 
194 }

197 float nsUniversalDetector::GetConfidence()
198 {
199  if (!mGotData) {
200  // we haven't got any data yet, return immediately
201  // caller program sometimes call DataEnd before anything has been sent to detector
202  return MINIMUM_THRESHOLD;
203  }
204  if (mDetectedCharset) {
205  return 0.99f;
206  }
207  switch (mInputState) {
208  case eHighbyte: {
209  float proberConfidence;
210  float maxProberConfidence = (float)0.0;
211  int maxProber = 0;
212 
213  for (int i = 0; i < NUM_OF_CHARSET_PROBERS; i++) {
214  proberConfidence = mCharSetProbers[i]->GetConfidence();
215  if (proberConfidence > maxProberConfidence) {
216  maxProberConfidence = proberConfidence;
217  maxProber = i;
218  }
219  }
220  //do not report anything because we are not confident of it, that's in fact a negative answer
221  if (maxProberConfidence > MINIMUM_THRESHOLD) {
222  return mCharSetProbers[maxProber]->GetConfidence();
223  }
224  }
225  case eEscAscii:
226  break;
227  default: // pure ascii
228  ;
229  }
230  return MINIMUM_THRESHOLD;
231 }
