70 QAction* _k_createInternal(StandardAction id, QObject *parent)
71 {
72  static bool stdNamesInitialized = false;
73 
74  if (!stdNamesInitialized) {
75  KAcceleratorManager::addStandardActionNames(stdNames());
76  stdNamesInitialized = true;
77  }
78 
79  QAction *pAction = 0;
80  const KStandardActionInfo *pInfo = infoPtr(id);
81 
82  // qDebug() << "KStandardAction::create( " << id << "=" << (pInfo ? pInfo->psName : (const char*)0) << ", " << parent << " )"; // ellis
83 
84  if (pInfo) {
85  QString sLabel, iconName = pInfo->psIconName;
86  switch (id) {
87  case Back:
88  sLabel = i18nc("go back", "&Back");
89  if (QApplication::isRightToLeft()) {
90  iconName = QStringLiteral("go-next");
91  }
92  break;
93 
94  case Forward:
95  sLabel = i18nc("go forward", "&Forward");
96  if (QApplication::isRightToLeft()) {
97  iconName = QStringLiteral("go-previous");
98  }
99  break;
100 
101  case Home:
102  sLabel = i18nc("home page", "&Home");
103  break;
104  case Help:
105  sLabel = i18nc("show help", "&Help");
106  break;
107  case Preferences:
108  case AboutApp:
109  case HelpContents: {
110  QString appDisplayName = QGuiApplication::applicationDisplayName();
111  if (appDisplayName.isEmpty()) {
112  appDisplayName = QCoreApplication::applicationName();
113  }
114  sLabel = i18n(pInfo->psLabel, appDisplayName);
115  }
116  break;
117  default:
118  sLabel = i18n(pInfo->psLabel);
119  }
120 
121  if (QApplication::isRightToLeft()) {
122  switch (id) {
123  case Prior: iconName = QStringLiteral("go-next-view-page"); break;
124  case Next: iconName = QStringLiteral("go-previous-view-page"); break;
125  case FirstPage: iconName = QStringLiteral("go-last-view-page"); break;
126  case LastPage: iconName = QStringLiteral("go-first-view-page"); break;
127  case DocumentBack: iconName = QStringLiteral("go-next"); break;
128  case DocumentForward: iconName = QStringLiteral("go-previous"); break;
129  default: break;
130  }
131  }
132 
133  QIcon icon = iconName.isEmpty() ? QIcon() : QIcon::fromTheme(iconName);
134 
135  switch (id) {
136  case OpenRecent:
137  pAction = new KRecentFilesAction(parent);
138  break;
139  case ShowMenubar:
140  pAction = new KToggleAction(parent);
141  pAction->setWhatsThis(i18n("Show Menubar<p>"
142  "Shows the menubar again after it has been hidden</p>"));
143  pAction->setChecked(true);
144  break;
145  case ShowToolbar:
146  pAction = new KToggleAction(parent);
147  pAction->setChecked(true);
148  break;
149  case ShowStatusbar:
150  pAction = new KToggleAction(parent);
151  pAction->setWhatsThis(i18n("Show Statusbar<p>"
152  "Shows the statusbar, which is the bar at the bottom of the window used for status information.</p>"));
153  pAction->setChecked(true);
154  break;
155  case FullScreen:
156  pAction = new KToggleFullScreenAction(parent);
157  pAction->setCheckable(true);
158  break;
159  case PasteText:
160  pAction = new KPasteTextAction(parent);
161  break;
162  // Same as default, but with the app icon
163  case AboutApp:
164  {
165  pAction = new QAction(parent);
166  icon = qApp->windowIcon();
167  if (icon.isNull()) {
168  const KAboutData data = KAboutData::applicationData();
169  if (!data.programIconName().isEmpty()) {
170  icon = QIcon::fromTheme(data.programIconName());
171  }
172  }
173  break;
174  }
175 
176  default:
177  pAction = new QAction(parent);
178  break;
179  }
180 
181  switch (id) {
182  case Quit:
183  pAction->setMenuRole(QAction::QuitRole);
184  break;
185 
186  case Preferences:
187  pAction->setMenuRole(QAction::PreferencesRole);
188  break;
189 
190  case AboutApp:
191  pAction->setMenuRole(QAction::AboutRole);
192  break;
193 
194  default:
195  pAction->setMenuRole(QAction::NoRole);
196  break;
197  }
198 
199  pAction->setText(sLabel);
200  if (pInfo->psToolTip) {
201  pAction->setToolTip(i18n(pInfo->psToolTip));
202  }
203  pAction->setIcon(icon);
204 
205  QList<QKeySequence> cut = KStandardShortcut::shortcut(pInfo->idAccel);
206  if (!cut.isEmpty()) {
207  // emulate KActionCollection::setDefaultShortcuts to allow the use of "configure shortcuts"
208  pAction->setShortcuts(cut);
209  pAction->setProperty("defaultShortcuts", QVariant::fromValue(cut));
210  }
211 
212  pAction->setObjectName(pInfo->psName);
213  }
214 
215  if (pAction && parent && parent->inherits("KActionCollection")) {
216  QMetaObject::invokeMethod(parent, "addAction", Q_ARG(QString, pAction->objectName()), Q_ARG(QAction *, pAction));
217  }
218 
219  return pAction;
220 }

222 QAction *create(StandardAction id, const QObject *recvr, const char *slot, QObject *parent)
223 {
224  QAction* pAction = _k_createInternal(id, parent);
225  if (recvr && slot) {
226  if (id == OpenRecent) {
227  // FIXME QAction port: probably a good idea to find a cleaner way to do this
228  // Open Recent is a special case - provide the selected URL
229  QObject::connect(pAction, SIGNAL(urlSelected(QUrl)), recvr, slot);
230  } else if (id == ConfigureToolbars) { // #200815
231  QObject::connect(pAction, SIGNAL(triggered(bool)), recvr, slot, Qt::QueuedConnection);
232  } else {
233  QObject::connect(pAction, SIGNAL(triggered(bool)), recvr, slot);
234  }
235  }
236  return pAction;
237 }

485 static QAction *buildAutomaticAction(QObject *parent, StandardAction id, const char *slot)
486 {
487  const KStandardActionInfo *p = infoPtr(id);
488  if (!p) {
489  return 0;
490  }
491 
492  AutomaticAction *action = new AutomaticAction(
493  QIcon::fromTheme(p->psIconName),
494  i18n(p->psLabel),
495  KStandardShortcut::shortcut(p->idAccel),
496  slot,
497  parent);
498 
499  action->setObjectName(p->psName);
500  if (p->psToolTip) {
501  action->setToolTip(i18n(p->psToolTip));
502  }
503 
504  if (parent && parent->inherits("KActionCollection")) {
505  QMetaObject::invokeMethod(parent, "addAction", Q_ARG(QString, action->objectName()), Q_ARG(QAction *, action));
506  }
507 
508  return action;
509 }
