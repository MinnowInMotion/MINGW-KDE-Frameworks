44  KConfigDialogPrivate(KConfigDialog *q, const QString &name, KCoreConfigSkeleton *config)
45  : q(q), shown(false), manager(0)
46  {
47  q->setObjectName(name);
48  q->setWindowTitle(i18nc("@title:window", "Configure"));
49  q->setFaceType(List);
50 
51  if (!name.isEmpty()) {
52  openDialogs.insert(name, q);
53  } else {
54  QString genericName;
55  genericName.sprintf("SettingsDialog-%p", static_cast<void *>(q));
56  openDialogs.insert(genericName, q);

58  }

60  QDialogButtonBox *buttonBox = q->buttonBox();
61  buttonBox->setStandardButtons(QDialogButtonBox::RestoreDefaults

63  | QDialogButtonBox::Apply
64  | QDialogButtonBox::Cancel

69  connect(buttonBox->button(QDialogButtonBox::Cancel), SIGNAL(clicked()), q, SLOT(updateWidgets()));
70  connect(buttonBox->button(QDialogButtonBox::RestoreDefaults), SIGNAL(clicked()), q, SLOT(updateWidgetsDefault()));
71  connect(buttonBox->button(QDialogButtonBox::RestoreDefaults), SIGNAL(clicked()), q, SLOT(_k_updateButtons()));

120 KPageWidgetItem *KConfigDialog::addPage(QWidget *page,
121  const QString &itemName,
122  const QString &pixmapName,
123  const QString &header,
124  bool manage)
125 {
126  Q_ASSERT(page);
127  if (!page) {
128  return 0;
129  }
130 
131  KPageWidgetItem *item = d->addPageInternal(page, itemName, pixmapName, header);
132  if (manage) {
133  d->manager->addWidget(page);
134  }
135 
136  if (d->shown && manage) {
137  // update the default button if the dialog is shown
138  QPushButton *defaultButton = buttonBox()->button(QDialogButtonBox::RestoreDefaults);
139  if (defaultButton) {
140  bool is_default = defaultButton->isEnabled() && d->manager->isDefault();
141  defaultButton->setEnabled(!is_default);
142  }
143  }
144  return item;
145 }

173 KPageWidgetItem *KConfigDialog::KConfigDialogPrivate::addPageInternal(QWidget *page,
174  const QString &itemName,
175  const QString &pixmapName,
176  const QString &header)
177 {
178  QWidget *frame = new QWidget(q);
179  QVBoxLayout *boxLayout = new QVBoxLayout(frame);
180  boxLayout->setMargin(0);
181  boxLayout->setMargin(0);
182 
183  QScrollArea *scroll = new QScrollArea(q);
184  scroll->setFrameShape(QFrame::NoFrame);
185  scroll->setHorizontalScrollBarPolicy(Qt::ScrollBarAsNeeded);
186  scroll->setVerticalScrollBarPolicy(Qt::ScrollBarAsNeeded);
187  scroll->setWidget(page);
188  scroll->setWidgetResizable(true);
189  scroll->setSizePolicy( QSizePolicy::MinimumExpanding, QSizePolicy::MinimumExpanding );
190 
191  boxLayout->addWidget(scroll);
192  KPageWidgetItem *item = new KPageWidgetItem(frame, itemName);
193  item->setHeader(header);
194  if (!pixmapName.isEmpty()) {
195  item->setIcon(QIcon::fromTheme(pixmapName));
196  }
197 
198  q->KPageDialog::addPage(item);
199  return item;
200 }

230 void KConfigDialog::onPageRemoved(KPageWidgetItem *item)
231 {
232  QMap<QWidget *, KConfigDialogManager *>::iterator j = d->managerForPage.begin();
233  while (j != d->managerForPage.end()) {
234  // there is a manager for this page, so remove it
235  if (item->widget()->isAncestorOf(j.key())) {
236  KConfigDialogManager *manager = j.value();
237  d->managerForPage.erase(j);
238  delete manager;
239  d->_k_updateButtons();
240  break;
241  }
242  ++j;
243  }
244 }

264 void KConfigDialog::KConfigDialogPrivate::_k_updateButtons()
265 {
266  static bool only_once = false;
267  if (only_once) {
268  return;
269  }
270  only_once = true;
271 
272  QMap<QWidget *, KConfigDialogManager *>::iterator it;
273 
274  bool has_changed = manager->hasChanged() || q->hasChanged();
275  for (it = managerForPage.begin();
276  it != managerForPage.end() && !has_changed;
277  ++it) {
278  has_changed |= (*it)->hasChanged();
279  }
280 
281  setApplyButtonEnabled(has_changed);
282 
283  bool is_default = manager->isDefault() && q->isDefault();
284  for (it = managerForPage.begin();
285  it != managerForPage.end() && is_default;
286  ++it) {
287  is_default &= (*it)->isDefault();
288  }
289 
290  setRestoreDefaultsButtonEnabled(!is_default);
291 
292  emit q->widgetModified();
293  only_once = false;
294 }

303 void KConfigDialog::showEvent(QShowEvent *e)
304 {
305  if (!d->shown) {
306  QMap<QWidget *, KConfigDialogManager *>::iterator it;
307 
308  updateWidgets();
309  d->manager->updateWidgets();
310  for (it = d->managerForPage.begin(); it != d->managerForPage.end(); ++it) {
311  (*it)->updateWidgets();
312  }
313 
314  bool has_changed = d->manager->hasChanged() || hasChanged();
315  for (it = d->managerForPage.begin();
316  it != d->managerForPage.end() && !has_changed;
317  ++it) {
318  has_changed |= (*it)->hasChanged();
319  }
320 
321  d->setApplyButtonEnabled(has_changed);
322 
323  bool is_default = d->manager->isDefault() && isDefault();
324  for (it = d->managerForPage.begin();
325  it != d->managerForPage.end() && is_default;
326  ++it) {
327  is_default &= (*it)->isDefault();
328  }
329 
330  d->setRestoreDefaultsButtonEnabled(!is_default);
331 
332  d->shown = true;
333  }
334  KPageDialog::showEvent(e);
335 }
