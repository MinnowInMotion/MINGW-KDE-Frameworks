34 static void checkInsertPos(QMenu *popup, const QString &str, int &index)
35 {
36  if (index != -1) {
37  return;
38  }
39 
40  int a = 0;
41  const QList<QAction *> actions = popup->actions();
42  int b = actions.count();
43 
44  while (a < b) {
45  int w = (a + b) / 2;
46  QAction *ac = actions[ w ];
47  int j = str.localeAwareCompare(ac->text());
48  if (j > 0) {
49  a = w + 1;
50  } else {
51  b = w;
52  }
53  }
54 
55  index = a; // it doesn't really matter ... a == b here.
56 
57  Q_ASSERT(a == b);
58 }

136 void KLanguageButton::insertLanguage(const QString &languageCode, const QString &name, int index)
137 {
138  QString text;
139  bool showCodes = d->showCodes;
140  if (name.isEmpty()) {
141  text = languageCode;
142  QLocale locale(languageCode);
143  if (locale != QLocale::c()) {
144  text = locale.nativeLanguageName();
145  } else {
146  showCodes = false;
147  }
148  } else {
149  text = name;
150  }
151  if (showCodes) {
152  text += QLatin1String(" (") + languageCode + QLatin1Char(')');
153  }
154 
155  checkInsertPos(d->popup, text, index);
156  QAction *a = new QAction(QIcon(), text, this);
157  a->setData(languageCode);
158  if (index >= 0 && index < d->popup->actions().count() - 1) {
159  d->popup->insertAction(d->popup->actions()[index], a);
160  } else {
161  d->popup->addAction(a);
162  }
163  d->ids.append(languageCode);
164 }

175 void KLanguageButton::loadAllLanguages()
176 {
177  QStringList langlist;
178  const QStringList localeDirs = QStandardPaths::locateAll(QStandardPaths::GenericDataLocation, QStringLiteral("locale"), QStandardPaths::LocateDirectory);
179  Q_FOREACH (const QString &localeDir, localeDirs) {
180  const QStringList entries = QDir(localeDir).entryList(QDir::Dirs);
181  Q_FOREACH (const QString &d, entries) {
182  const QString entryFile = localeDir + '/' + d + "/kf5_entry.desktop";
183  if (QFile::exists(entryFile)) {
184  langlist.append(entryFile);
185  }
186  }
187  }
188  langlist.sort();
189  for (int i = 0, count = langlist.count(); i < count; ++i) {
190  QString fpath = langlist[i].left(langlist[i].length() - 14);
191  QString code = fpath.mid(fpath.lastIndexOf('/') + 1);
192  KConfig entry(langlist[i], KConfig::SimpleConfig);
193  KConfigGroup group(&entry, "KCM Locale");
194  QString name = group.readEntry("Name", i18n("without name"));
195  insertLanguage(code, name);
196  }
197 
198  setCurrentItem(d->locale);
199 }
