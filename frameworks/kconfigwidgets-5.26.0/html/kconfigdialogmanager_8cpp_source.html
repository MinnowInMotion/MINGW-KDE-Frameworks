173 void KConfigDialogManager::setupWidget(QWidget *widget, KConfigSkeletonItem *item)
174 {
175  QVariant minValue = item->minValue();
176  if (minValue.isValid()) {
177  // Only q3datetimeedit is using this property we can remove it if we stop supporting Qt3Support
178  if (widget->metaObject()->indexOfProperty("minValue") != -1) {
179  widget->setProperty("minValue", minValue);
180  }
181  if (widget->metaObject()->indexOfProperty("minimum") != -1) {
182  widget->setProperty("minimum", minValue);
183  }
184  }
185  QVariant maxValue = item->maxValue();
186  if (maxValue.isValid()) {
187  // Only q3datetimeedit is using that property we can remove it if we stop supporting Qt3Support
188  if (widget->metaObject()->indexOfProperty("maxValue") != -1) {
189  widget->setProperty("maxValue", maxValue);
190  }
191  if (widget->metaObject()->indexOfProperty("maximum") != -1) {
192  widget->setProperty("maximum", maxValue);
193  }
194  }
195 
196  if (widget->whatsThis().isEmpty()) {
197  QString whatsThis = item->whatsThis();
198  if (!whatsThis.isEmpty()) {
199  widget->setWhatsThis(whatsThis);
200  }
201  }
202 
203  if (widget->toolTip().isEmpty()) {
204  QString toolTip = item->toolTip();
205  if (!toolTip.isEmpty()) {
206  widget->setToolTip(toolTip);
207  }
208  }
209 
210  // If it is a QGroupBox with only autoExclusive buttons
211  // and has no custom property and the config item type
212  // is an integer, assume we want to save the index like we did with
213  // KButtonGroup instead of if it is checked or not
214  QGroupBox *gb = qobject_cast<QGroupBox *>(widget);
215  if (gb && getCustomProperty(gb).isEmpty()) {
216  const KConfigSkeletonItem *item = d->m_conf->findItem(widget->objectName().mid(5));
217  if (item->property().type() == QVariant::Int) {
218  QObjectList children = gb->children();
219  children.removeAll(gb->layout());
220  const QList<QAbstractButton *> buttons = gb->findChildren<QAbstractButton *>();
221  if (children.count() == buttons.count()) {
222  bool allAutoExclusive = true;
223  foreach(QAbstractButton *button, buttons) {
224  allAutoExclusive = allAutoExclusive && button->autoExclusive();
225  }
226  if (allAutoExclusive) {
227  d->allExclusiveGroupBoxes << widget;
228  }
229  }
230  }
231  }
232 
233  if (!item->isEqual(property(widget))) {
234  setProperty(widget, item->property());
235  }
236 }

238 bool KConfigDialogManager::parseChildren(const QWidget *widget, bool trackChanges)
239 {
240  bool valueChanged = false;
241  const QList<QObject *> listOfChildren = widget->children();
242  if (listOfChildren.count() == 0) { //?? XXX
243  return valueChanged;
244  }
245 
246  foreach (QObject *object, listOfChildren) {
247  if (!object->isWidgetType()) {
248  continue; // Skip non-widgets
249  }
250 
251  QWidget *childWidget = static_cast<QWidget *>(object);
252 
253  QString widgetName = childWidget->objectName();
254  bool bParseChildren = true;
255  bool bSaveInsideGroupBox = d->insideGroupBox;
256 
257  if (widgetName.startsWith(QLatin1String("kcfg_"))) {
258  // This is one of our widgets!
259  QString configId = widgetName.mid(5);
260  KConfigSkeletonItem *item = d->m_conf->findItem(configId);
261  if (item) {
262  d->knownWidget.insert(configId, childWidget);
263 
264  setupWidget(childWidget, item);
265 
266  if (trackChanges) {
267 
268  if (d->allExclusiveGroupBoxes.contains(childWidget)) {
269  const QList<QAbstractButton *> buttons = childWidget->findChildren<QAbstractButton *>();
270  foreach(QAbstractButton *button, buttons) {
271  connect(button, SIGNAL(toggled(bool)), this, SIGNAL(widgetModified()));
272  }
273  }
274 
275  QHash<QString, QByteArray>::const_iterator changedIt = s_changedMap()->constFind(childWidget->metaObject()->className());
276 
277  if (changedIt == s_changedMap()->constEnd()) {
278  // If the class name of the widget wasn't in the monitored widgets map, then look for
279  // it again using the super class name. This fixes a problem with using QtRuby/Korundum
280  // widgets with KConfigXT where 'Qt::Widget' wasn't being seen a the real deal, even
281  // though it was a 'QWidget'.
282  if (childWidget->metaObject()->superClass()) {
283  changedIt = s_changedMap()->constFind(childWidget->metaObject()->superClass()->className());
284  } else {
285  changedIt = s_changedMap()->constFind(0);
286  }
287  }
288 
289  if (changedIt == s_changedMap()->constEnd()) {
290  qWarning() << "Don't know how to monitor widget '" << childWidget->metaObject()->className() << "' for changes!";
291  } else {
292  connect(childWidget, *changedIt,
293  this, SIGNAL(widgetModified()));
294 
295  QComboBox *cb = qobject_cast<QComboBox *>(childWidget);
296  if (cb && cb->isEditable())
297  connect(cb, SIGNAL(editTextChanged(QString)),
298  this, SIGNAL(widgetModified()));
299  }
300  }
301  QGroupBox *gb = qobject_cast<QGroupBox *>(childWidget);
302  if (!gb) {
303  bParseChildren = false;
304  } else {
305  d->insideGroupBox = true;
306  }
307  } else {
308  qWarning() << "A widget named '" << widgetName << "' was found but there is no setting named '" << configId << "'";
309  }
310  } else if (QLabel *label = qobject_cast<QLabel *>(childWidget)) {
311  QWidget *buddy = label->buddy();
312  if (!buddy) {
313  continue;
314  }
315  QString buddyName = buddy->objectName();
316  if (buddyName.startsWith(QLatin1String("kcfg_"))) {
317  // This is one of our widgets!
318  QString configId = buddyName.mid(5);
319  d->buddyWidget.insert(configId, childWidget);
320  }
321  }
322 //kf5: commented out to reduce debug output
323 // #ifndef NDEBUG
324 // else if (!widgetName.isEmpty() && trackChanges)
325 // {
326 // QHash<QString, QByteArray>::const_iterator changedIt = s_changedMap()->constFind(childWidget->metaObject()->className());
327 // if (changedIt != s_changedMap()->constEnd())
328 // {
329 // if ((!d->insideGroupBox || !qobject_cast<QRadioButton*>(childWidget)) &&
330 // !qobject_cast<QGroupBox*>(childWidget) &&!qobject_cast<QTabWidget*>(childWidget) )
331 // qDebug() << "Widget '" << widgetName << "' (" << childWidget->metaObject()->className() << ") remains unmanaged.";
332 // }
333 // }
334 // #endif
335 
336  if (bParseChildren) {
337  // this widget is not known as something we can store.
338  // Maybe we can store one of its children.
339  valueChanged |= parseChildren(childWidget, trackChanges);
340  }
341  d->insideGroupBox = bSaveInsideGroupBox;
342  }
343  return valueChanged;
344 }

346 void KConfigDialogManager::updateWidgets()
347 {
348  bool changed = false;
349  bool bSignalsBlocked = signalsBlocked();
350  blockSignals(true);
351 
352  QWidget *widget;
353  QHashIterator<QString, QWidget *> it(d->knownWidget);
354  while (it.hasNext()) {
355  it.next();
356  widget = it.value();
357 
358  KConfigSkeletonItem *item = d->m_conf->findItem(it.key());
359  if (!item) {
360  qWarning() << "The setting '" << it.key() << "' has disappeared!";
361  continue;
362  }
363 
364  if (!item->isEqual(property(widget))) {
365  setProperty(widget, item->property());
366 // qDebug() << "The setting '" << it.key() << "' [" << widget->className() << "] has changed";
367  changed = true;
368  }
369  if (item->isImmutable()) {
370  widget->setEnabled(false);
371  QWidget *buddy = d->buddyWidget.value(it.key(), 0);
372  if (buddy) {
373  buddy->setEnabled(false);
374  }
375  }
376  }
377  blockSignals(bSignalsBlocked);
378 
379  if (changed) {
380  QTimer::singleShot(0, this, SIGNAL(widgetModified()));
381  }
382 }

419 QByteArray KConfigDialogManager::getUserProperty(const QWidget *widget) const
420 {
421  if (!s_propertyMap()->contains(widget->metaObject()->className())) {
422  const QMetaObject *metaObject = widget->metaObject();
423  const QMetaProperty user = metaObject->userProperty();
424  if (user.isValid()) {
425  s_propertyMap()->insert(widget->metaObject()->className(), user.name());
426  //qDebug() << "class name: '" << widget->metaObject()->className()
427  //<< " 's USER property: " << metaProperty.name() << endl;
428  } else {
429  return QByteArray(); //no USER property
430  }
431  }
432  const QComboBox *cb = qobject_cast<const QComboBox *>(widget);
433  if (cb) {
434  const char *qcomboUserPropertyName = cb->QComboBox::metaObject()->userProperty().name();
435  const int qcomboUserPropertyIndex = qcomboUserPropertyName ? cb->QComboBox::metaObject()->indexOfProperty(qcomboUserPropertyName) : -1;
436  const char *widgetUserPropertyName = widget->metaObject()->userProperty().name();
437  const int widgetUserPropertyIndex = widgetUserPropertyName ? cb->metaObject()->indexOfProperty(widgetUserPropertyName) : -1;
438 
439  if (qcomboUserPropertyIndex == widgetUserPropertyIndex) {
440  return QByteArray(); // use the q/kcombobox special code
441  }
442  }
443 
444  return s_propertyMap()->value(widget->metaObject()->className());
445 }

461 void KConfigDialogManager::setProperty(QWidget *w, const QVariant &v)
462 {
463  if (d->allExclusiveGroupBoxes.contains(w)) {
464  const QList<QAbstractButton *> buttons = w->findChildren<QAbstractButton *>();
465  if (v.toInt() < buttons.count()) {
466  buttons[v.toInt()]->setChecked(true);
467  }
468  return;
469  }
470 
471  QByteArray userproperty = getCustomProperty(w);
472  if (userproperty.isEmpty()) {
473  userproperty = getUserProperty(w);
474  }
475  if (userproperty.isEmpty()) {
476  QComboBox *cb = qobject_cast<QComboBox *>(w);
477  if (cb) {
478  if (cb->isEditable()) {
479  int i = cb->findText(v.toString());
480  if (i != -1) {
481  cb->setCurrentIndex(i);
482  } else {
483  cb->setEditText(v.toString());
484  }
485  } else {
486  cb->setCurrentIndex(v.toInt());
487  }
488  return;
489  }
490  }
491  if (userproperty.isEmpty()) {
492  qWarning() << w->metaObject()->className() << " widget not handled!";
493  return;
494  }
495 
496  w->setProperty(userproperty, v);
497 }

499 QVariant KConfigDialogManager::property(QWidget *w) const
500 {
501  if (d->allExclusiveGroupBoxes.contains(w)) {
502  const QList<QAbstractButton *> buttons = w->findChildren<QAbstractButton *>();
503  for (int i = 0; i < buttons.count(); ++i) {
504  if (buttons[i]->isChecked())
505  return i;
506  }
507  return -1;
508  }
509 
510  QByteArray userproperty = getCustomProperty(w);
511  if (userproperty.isEmpty()) {
512  userproperty = getUserProperty(w);
513  }
514  if (userproperty.isEmpty()) {
515  QComboBox *cb = qobject_cast<QComboBox *>(w);
516  if (cb) {
517  if (cb->isEditable()) {
518  return QVariant(cb->currentText());
519  } else {
520  return QVariant(cb->currentIndex());
521  }
522  }
523  }
524  if (userproperty.isEmpty()) {
525  qWarning() << w->metaObject()->className() << " widget not handled!";
526  return QVariant();
527  }
528 
529  return w->property(userproperty);
530 }
