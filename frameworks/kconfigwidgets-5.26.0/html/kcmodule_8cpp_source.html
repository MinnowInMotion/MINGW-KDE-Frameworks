121 void KCModule::setNeedsAuthorization(bool needsAuth)
122 {
123  d->_needsAuthorization = needsAuth;
124  if (needsAuth && d->_about) {
125  d->_authAction = KAuth::Action(QString("org.kde.kcontrol." + d->_about->componentName() + ".save"));
126  d->_needsAuthorization = d->_authAction.isValid();
127  d->_authAction.setHelperId("org.kde.kcontrol." + d->_about->componentName());
128  d->_authAction.setParentWidget(this);
129  authStatusChanged(d->_authAction.status());
130  } else {
131  d->_authAction = KAuth::Action();
132  }
133 }

140 void KCModule::setAuthAction(const KAuth::Action &action)
141 {
142  if (!d->_authAction.isValid()) {
143  qWarning() << "Auth action" << action.name() << "is invalid";
144  d->_needsAuthorization = false;
145  return;
146  }
147  d->_authAction = action;
148  d->_needsAuthorization = true;
149  d->_authAction.setParentWidget(this);
150  authStatusChanged(d->_authAction.status());
151 }

158 void KCModule::authStatusChanged(KAuth::Action::AuthStatus status)
159 {
160  switch (status) {
161  case KAuth::Action::AuthorizedStatus:
162  setUseRootOnlyMessage(false);
163  break;
164  case KAuth::Action::AuthRequiredStatus:
165  setUseRootOnlyMessage(true);
166  setRootOnlyMessage(i18n("You will be asked to authenticate before saving"));
167  break;
168  default:
169  setUseRootOnlyMessage(true);
170  setRootOnlyMessage(i18n("You are not allowed to save the configuration"));
171  break;
172  }
173 
174  qDebug() << useRootOnlyMessage();
175 }
