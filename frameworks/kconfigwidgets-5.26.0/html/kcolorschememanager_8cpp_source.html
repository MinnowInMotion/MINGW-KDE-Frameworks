74 void KColorSchemeModel::init()
75 {
76  beginResetModel();
77  m_data.clear();
78 
79  const QStringList dirs = QStandardPaths::locateAll(QStandardPaths::GenericDataLocation,
80  QStringLiteral("color-schemes"),
81  QStandardPaths::LocateDirectory);
82  QStringList schemeFiles;
83  Q_FOREACH (const QString &dir, dirs) {
84  const QStringList fileNames = QDir(dir).entryList(QStringList() << QStringLiteral("*.colors"));
85  Q_FOREACH (const QString &file, fileNames) {
86  schemeFiles << dir + QDir::separator() + file;
87  }
88  }
89  Q_FOREACH (const QString &schemeFile, schemeFiles) {
90  KSharedConfigPtr config = KSharedConfig::openConfig(schemeFile);
91  KConfigGroup group(config, QStringLiteral("General"));
92  const QString name = group.readEntry("Name", QFileInfo(schemeFile).baseName());
93  const KColorSchemeModelData data = {name, schemeFile, createPreview(schemeFile)};
94  m_data.append(data);
95  }
96  std::sort(m_data.begin(), m_data.end(), [](const KColorSchemeModelData & first, const KColorSchemeModelData & second) {
97  return first.name < second.name;
98  });
99  endResetModel();
100 }

102 QIcon KColorSchemeModel::createPreview(const QString &path)
103 {
104  KSharedConfigPtr schemeConfig = KSharedConfig::openConfig(path);
105  QIcon result;
106  KColorScheme activeWindow(QPalette::Active, KColorScheme::Window, schemeConfig);
107  KColorScheme activeButton(QPalette::Active, KColorScheme::Button, schemeConfig);
108  KColorScheme activeView(QPalette::Active, KColorScheme::View, schemeConfig);
109  KColorScheme activeSelection(QPalette::Active, KColorScheme::Selection, schemeConfig);
110 
111  auto pixmap = [&](int size) {
112  QPixmap pix(size, size);
113  pix.fill(Qt::black);
114  QPainter p;
115  p.begin(&pix);
116  const int itemSize = size / 2 - 1;
117  p.fillRect(1, 1, itemSize, itemSize, activeWindow.background());
118  p.fillRect(1 + itemSize, 1, itemSize, itemSize, activeButton.background());
119  p.fillRect(1, 1 + itemSize, itemSize, itemSize, activeView.background());
120  p.fillRect(1 + itemSize, 1 + itemSize, itemSize, itemSize, activeSelection.background());
121  p.end();
122  result.addPixmap(pix);
123  };
124  // 16x16
125  pixmap(16);
126  // 24x24
127  pixmap(24);
128 
129  return result;
130 }

158 KActionMenu *KColorSchemeManager::createSchemeSelectionMenu(const QIcon &icon, const QString &name, const QString &selectedSchemeName, QObject *parent)
159 {
160  KActionMenu *menu = new KActionMenu(icon, name, parent);
161  QActionGroup *group = new QActionGroup(menu);
162  connect(group, &QActionGroup::triggered, [](QAction * action) {
163  // hint for the style to synchronize the color scheme with the window manager/compositor
164  qApp->setProperty("KDE_COLOR_SCHEME_PATH", action->data());
165  qApp->setPalette(KColorScheme::createApplicationPalette(KSharedConfig::openConfig(action->data().toString())));
166  });
167  for (int i = 0; i < d->model->rowCount(); ++i) {
168  QModelIndex index = d->model->index(i);
169  QAction *action = new QAction(index.data(Qt::DecorationRole).value<QIcon>(), index.data().toString(), menu);
170  action->setData(index.data(Qt::UserRole));
171  action->setActionGroup(group);
172  action->setCheckable(true);
173  if (index.data().toString() == selectedSchemeName) {
174  action->setChecked(true);
175  }
176  menu->addAction(action);
177  }
178  return menu;
179 }

191 void KColorSchemeManager::activateScheme(const QModelIndex &index)
