68 StateEffects::StateEffects(QPalette::ColorGroup state, const KSharedConfigPtr &config)
69  : _color(0, 0, 0, 0) //, _chain(0) not needed yet
70 {
71  QString group;
72  if (state == QPalette::Disabled) {
73  group = QStringLiteral("ColorEffects:Disabled");
74  } else if (state == QPalette::Inactive) {
75  group = QStringLiteral("ColorEffects:Inactive");
76  }
77 
78  _effects[0] = 0;
79  _effects[1] = 0;
80  _effects[2] = 0;
81 
82  // NOTE: keep this in sync with kdebase/workspace/kcontrol/colors/colorscm.cpp
83  if (! group.isEmpty()) {
84  KConfigGroup cfg(config, group);
85  const bool enabledByDefault = (state == QPalette::Disabled);
86  if (cfg.readEntry("Enable", enabledByDefault)) {
87  _effects[Intensity] = cfg.readEntry("IntensityEffect",
88  (int)(state == QPalette::Disabled ? IntensityDarken : IntensityNoEffect));
89  _effects[Color] = cfg.readEntry("ColorEffect",
90  (int)(state == QPalette::Disabled ? ColorNoEffect : ColorDesaturate));
91  _effects[Contrast] = cfg.readEntry("ContrastEffect",
92  (int)(state == QPalette::Disabled ? ContrastFade : ContrastTint));
93  _amount[Intensity] = cfg.readEntry("IntensityAmount", state == QPalette::Disabled ? 0.10 : 0.0);
94  _amount[Color] = cfg.readEntry("ColorAmount", state == QPalette::Disabled ? 0.0 : -0.9);
95  _amount[Contrast] = cfg.readEntry("ContrastAmount", state == QPalette::Disabled ? 0.65 : 0.25);
96  if (_effects[Color] > ColorNoEffect) {
97  _color = cfg.readEntry("Color", state == QPalette::Disabled ? QColor(56, 56, 56) : QColor(112, 111, 110));
98  }
99  }
100  }
101 }

103 QBrush StateEffects::brush(const QBrush &background) const
104 {
105  QColor color = background.color(); // TODO - actually work on brushes
106  switch (_effects[Intensity]) {
107  case IntensityShade:
108  color = KColorUtils::shade(color, _amount[Intensity]);
109  break;
110  case IntensityDarken:
111  color = KColorUtils::darken(color, _amount[Intensity]);
112  break;
113  case IntensityLighten:
114  color = KColorUtils::lighten(color, _amount[Intensity]);
115  break;
116  }
117  switch (_effects[Color]) {
118  case ColorDesaturate:
119  color = KColorUtils::darken(color, 0.0, 1.0 - _amount[Color]);
120  break;
121  case ColorFade:
122  color = KColorUtils::mix(color, _color, _amount[Color]);
123  break;
124  case ColorTint:
125  color = KColorUtils::tint(color, _color, _amount[Color]);
126  break;
127  }
128  return QBrush(color);
129 }

315 void KColorSchemePrivate::init(const KSharedConfigPtr &config,
316  QPalette::ColorGroup state,
317  const char *group,
318  SetDefaultColors defaults)
319 {
320  KConfigGroup cfg(config, group);
321 
322  // loaded-from-config colors
323  _brushes.fg[0] = cfg.readEntry("ForegroundNormal", SET_DEFAULT(NormalText));
324  _brushes.fg[1] = cfg.readEntry("ForegroundInactive", SET_DEFAULT(InactiveText));
325  _brushes.fg[2] = cfg.readEntry("ForegroundActive", SET_DEFAULT(ActiveText));
326  _brushes.fg[3] = cfg.readEntry("ForegroundLink", SET_DEFAULT(LinkText));
327  _brushes.fg[4] = cfg.readEntry("ForegroundVisited", SET_DEFAULT(VisitedText));
328  _brushes.fg[5] = cfg.readEntry("ForegroundNegative", SET_DEFAULT(NegativeText));
329  _brushes.fg[6] = cfg.readEntry("ForegroundNeutral", SET_DEFAULT(NeutralText));
330  _brushes.fg[7] = cfg.readEntry("ForegroundPositive", SET_DEFAULT(PositiveText));
331 
332  _brushes.deco[0] = cfg.readEntry("DecorationHover", DECO_DEFAULT(Hover));
333  _brushes.deco[1] = cfg.readEntry("DecorationFocus", DECO_DEFAULT(Focus));
334 
335  // apply state adjustments
336  if (state != QPalette::Active) {
337  StateEffects effects(state, config);
338  for (int i = 0; i < 8; i++) {
339  _brushes.fg[i] = effects.brush(_brushes.fg[i], _brushes.bg[0]);
340  }
341  _brushes.deco[0] = effects.brush(_brushes.deco[0], _brushes.bg[0]);
342  _brushes.deco[1] = effects.brush(_brushes.deco[1], _brushes.bg[0]);
343  _brushes.bg[0] = effects.brush(_brushes.bg[0]);
344  _brushes.bg[1] = effects.brush(_brushes.bg[1]);
345  }
346 
347  // calculated backgrounds
348  _brushes.bg[2] = KColorUtils::tint(_brushes.bg[0].color(), _brushes.fg[2].color());
349  _brushes.bg[3] = KColorUtils::tint(_brushes.bg[0].color(), _brushes.fg[3].color());

351  _brushes.bg[5] = KColorUtils::tint(_brushes.bg[0].color(), _brushes.fg[5].color());
352  _brushes.bg[6] = KColorUtils::tint(_brushes.bg[0].color(), _brushes.fg[6].color());

431 KColorScheme::KColorScheme(QPalette::ColorGroup state, ColorSet set, KSharedConfigPtr config)
432 {
433  if (!config) {
434  config = KSharedConfig::openConfig();
435  }
436 
437  switch (set) {
438  case Window:
439  d = new KColorSchemePrivate(config, state, "Colors:Window", defaultWindowColors);
440  break;
441  case Button:
442  d = new KColorSchemePrivate(config, state, "Colors:Button", defaultButtonColors);
443  break;
444  case Selection: {
445  KConfigGroup group(config, "ColorEffects:Inactive");
446  // NOTE: keep this in sync with kdebase/workspace/kcontrol/colors/colorscm.cpp
447  bool inactiveSelectionEffect = group.readEntry("ChangeSelectionColor", group.readEntry("Enable", true));
448  // if enabled, inactiver/disabled uses Window colors instead, ala gtk
449  // ...except tinted with the Selection:NormalBackground color so it looks more like selection
450  if (state == QPalette::Active || (state == QPalette::Inactive && !inactiveSelectionEffect)) {
451  d = new KColorSchemePrivate(config, state, "Colors:Selection", defaultSelectionColors);
452  } else if (state == QPalette::Inactive)
453  d = new KColorSchemePrivate(config, state, "Colors:Window", defaultWindowColors,
454  KColorScheme(QPalette::Active, Selection, config).background());
455  else { // disabled (...and still want this branch when inactive+disabled exists)
456  d = new KColorSchemePrivate(config, state, "Colors:Window", defaultWindowColors);
457  }
458  } break;
459  case Tooltip:
460  d = new KColorSchemePrivate(config, state, "Colors:Tooltip", defaultTooltipColors);
461  break;
462  case Complementary:
463  d = new KColorSchemePrivate(config, state, "Colors:Complementary", defaultComplementaryColors);
464  break;
465  default:
466  d = new KColorSchemePrivate(config, state, "Colors:View", defaultViewColors);
467  }
468 }

512 QColor KColorScheme::shade(const QColor &color, ShadeRole role, qreal contrast, qreal chromaAdjust)
513 {
514  // nan -> 1.0
515  contrast = (1.0 > contrast ? (-1.0 < contrast ? contrast : -1.0) : 1.0);
516  qreal y = KColorUtils::luma(color), yi = 1.0 - y;
517 
518  // handle very dark colors (base, mid, dark, shadow == midlight, light)
519  if (y < 0.006) {
520  switch (role) {
521  case KColorScheme::LightShade:
522  return KColorUtils::shade(color, 0.05 + 0.95 * contrast, chromaAdjust);
523  case KColorScheme::MidShade:
524  return KColorUtils::shade(color, 0.01 + 0.20 * contrast, chromaAdjust);
525  case KColorScheme::DarkShade:
526  return KColorUtils::shade(color, 0.02 + 0.40 * contrast, chromaAdjust);
527  default:
528  return KColorUtils::shade(color, 0.03 + 0.60 * contrast, chromaAdjust);
529  }
530  }
531 
532  // handle very light colors (base, midlight, light == mid, dark, shadow)
533  if (y > 0.93) {
534  switch (role) {
535  case KColorScheme::MidlightShade:
536  return KColorUtils::shade(color, -0.02 - 0.20 * contrast, chromaAdjust);
537  case KColorScheme::DarkShade:
538  return KColorUtils::shade(color, -0.06 - 0.60 * contrast, chromaAdjust);
539  case KColorScheme::ShadowShade:
540  return KColorUtils::shade(color, -0.10 - 0.90 * contrast, chromaAdjust);
541  default:
542  return KColorUtils::shade(color, -0.04 - 0.40 * contrast, chromaAdjust);
543  }
544  }
545 
546  // handle everything else
547  qreal lightAmount = (0.05 + y * 0.55) * (0.25 + contrast * 0.75);
548  qreal darkAmount = (- y) * (0.55 + contrast * 0.35);
549  switch (role) {
550  case KColorScheme::LightShade:
551  return KColorUtils::shade(color, lightAmount, chromaAdjust);
552  case KColorScheme::MidlightShade:
553  return KColorUtils::shade(color, (0.15 + 0.35 * yi) * lightAmount, chromaAdjust);
554  case KColorScheme::MidShade:
555  return KColorUtils::shade(color, (0.35 + 0.15 * y) * darkAmount, chromaAdjust);
556  case KColorScheme::DarkShade:
557  return KColorUtils::shade(color, darkAmount, chromaAdjust);
558  default:
559  return KColorUtils::darken(KColorUtils::shade(color, darkAmount, chromaAdjust), 0.5 + 0.3 * y);
560  }
561 }

579 QPalette KColorScheme::createApplicationPalette(const KSharedConfigPtr &config)
580 {
581  QPalette palette;
582 
583  static const QPalette::ColorGroup states[3] = { QPalette::Active, QPalette::Inactive, QPalette::Disabled };
584 
585  // TT thinks tooltips shouldn't use active, so we use our active colors for all states
586  KColorScheme schemeTooltip(QPalette::Active, KColorScheme::Tooltip, config);
587 
588  for (int i = 0; i < 3; i++) {
