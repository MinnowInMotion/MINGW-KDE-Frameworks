120 static QString titleWithSensibleWidth(const QString &nameValue, const QString &value)
121 {
122  // Calculate 3/4 of screen geometry, we do not want
123  // action titles to be bigger than that
124  // Since we do not know in which screen we are going to show
125  // we choose the min of all the screens
126  const QDesktopWidget desktopWidget;
127  int maxWidthForTitles = INT_MAX;
128  for (int i = 0; i < desktopWidget.screenCount(); ++i) {
129  maxWidthForTitles = qMin(maxWidthForTitles, desktopWidget.availableGeometry(i).width() * 3 / 4);
130  }
131  const QFontMetrics fontMetrics = QFontMetrics(QFont());
132 
133  QString title = nameValue + " [" + value + ']';
134  if (fontMetrics.width(title) > maxWidthForTitles) {
135  // If it does not fit, try to cut only the whole path, though if the
136  // name is too long (more than 3/4 of the whole text) we cut it a bit too
137  const int nameValueMaxWidth = maxWidthForTitles * 3 / 4;
138  const int nameWidth = fontMetrics.width(nameValue);
139  QString cutNameValue, cutValue;
140  if (nameWidth > nameValueMaxWidth) {
141  cutNameValue = fontMetrics.elidedText(nameValue, Qt::ElideMiddle, nameValueMaxWidth);
142  cutValue = fontMetrics.elidedText(value, Qt::ElideMiddle, maxWidthForTitles - nameValueMaxWidth);
143  } else {
144  cutNameValue = nameValue;
145  cutValue = fontMetrics.elidedText(value, Qt::ElideMiddle, maxWidthForTitles - nameWidth);
146  }
147  title = cutNameValue + " [" + cutValue + ']';
148  }
149  return title;
150 }

152 void KRecentFilesAction::addUrl(const QUrl &_url, const QString &name)
153 {
154  Q_D(KRecentFilesAction);
160  const QUrl url(_url);
161 
162  if (url.isLocalFile() && url.toLocalFile().startsWith(QDir::tempPath())) {
163  return;
164  }
165  const QString tmpName = name.isEmpty() ? url.fileName() : name;
166  const QString pathOrUrl(url.toDisplayString(QUrl::PreferLocalFile));
167 
168 #ifdef Q_OS_WIN
169  const QString file = url.isLocalFile() ? QDir::toNativeSeparators(pathOrUrl) : pathOrUrl;
170 #else
171  const QString file = pathOrUrl;
172 #endif
173 
174  // remove file if already in list
175  foreach (QAction *action, selectableActionGroup()->actions()) {
176  const QString urlStr = d->m_urls[action].toDisplayString(QUrl::PreferLocalFile);
177 #ifdef Q_OS_WIN
178  const QString tmpFileName = url.isLocalFile() ? QDir::toNativeSeparators(urlStr) : urlStr;
179  if (tmpFileName.endsWith(file, Qt::CaseInsensitive))
180 #else
181  if (urlStr.endsWith(file))
182 #endif
183  {
184  removeAction(action)->deleteLater();
185  break;
186  }
187  }
188  // remove oldest item if already maxitems in list
189  if (d->m_maxItems && selectableActionGroup()->actions().count() == d->m_maxItems) {
190  // remove oldest added item
191  delete removeAction(selectableActionGroup()->actions().first());
192  }
193 
194  d->m_noEntriesAction->setVisible(false);
195  d->clearSeparator->setVisible(true);
196  d->clearAction->setVisible(true);
197  setEnabled(true);
198  // add file to list
199  const QString title = titleWithSensibleWidth(tmpName, file);
200  QAction *action = new QAction(title, selectableActionGroup());
201  addAction(action, url, tmpName);
202 }

258 void KRecentFilesAction::loadEntries(const KConfigGroup &_config)
259 {
260  Q_D(KRecentFilesAction);
261  clearEntries();
262 
263  QString key;
264  QString value;
265  QString nameKey;
266  QString nameValue;
267  QString title;
268  QUrl url;
269 
270  KConfigGroup cg = _config;
271  if (cg.name().isEmpty()) {
272  cg = KConfigGroup(cg.config(), "RecentFiles");
273  }
274 
275  bool thereAreEntries = false;
276  // read file list
277  for (int i = 1; i <= d->m_maxItems; i++) {
278  key = QStringLiteral("File%1").arg(i);
279  value = cg.readPathEntry(key, QString());
280  if (value.isEmpty()) {
281  continue;
282  }
283  url = QUrl::fromUserInput(value);
284 
285  // Don't restore if file doesn't exist anymore
286  if (url.isLocalFile() && !QFile::exists(url.toLocalFile())) {
287  continue;
288  }
289 
290  // Don't restore where the url is already known (eg. broken config)
291  if (d->m_urls.values().contains(url)) {
292  continue;
293  }
294 
295 #ifdef Q_OS_WIN
296  // convert to backslashes
297  if (url.isLocalFile()) {
298  value = QDir::toNativeSeparators(value);
299  }
300 #endif
301 
302  nameKey = QStringLiteral("Name%1").arg(i);
303  nameValue = cg.readPathEntry(nameKey, url.fileName());
304  title = titleWithSensibleWidth(nameValue, value);
305  if (!value.isNull()) {
306  thereAreEntries = true;
307  addAction(new QAction(title, selectableActionGroup()), url, nameValue);
308  }
309  }
310  if (thereAreEntries) {
311  d->m_noEntriesAction->setVisible(false);
312  d->clearSeparator->setVisible(true);
313  d->clearAction->setVisible(true);
314  setEnabled(true);
315  }
316 }

318 void KRecentFilesAction::saveEntries(const KConfigGroup &_cg)
319 {
320  Q_D(KRecentFilesAction);
321  QString key;
322  QString value;
323  QStringList lst = items();
324 
325  KConfigGroup cg = _cg;
326  if (cg.name().isEmpty()) {
327  cg = KConfigGroup(cg.config(), "RecentFiles");
328  }
329 
330  cg.deleteGroup();
331 
332  // write file list
333  for (int i = 1; i <= selectableActionGroup()->actions().count(); i++) {
334  key = QStringLiteral("File%1").arg(i);
335  // i - 1 because we started from 1
336  value = d->m_urls[ selectableActionGroup()->actions()[ i - 1 ] ].toDisplayString(QUrl::PreferLocalFile);
337  cg.writePathEntry(key, value);
338  key = QStringLiteral("Name%1").arg(i);
339  value = d->m_shortNames[ selectableActionGroup()->actions()[ i - 1 ] ];
340  cg.writePathEntry(key, value);
341  }
342 
343 }
