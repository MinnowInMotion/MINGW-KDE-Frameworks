139 void KPlotWidget::Private::calcDataRectLimits(double x1, double x2, double y1, double y2)
140 {
141  double XA1, XA2, YA1, YA2;
142  if (x2 < x1) {
143  XA1 = x2;
144  XA2 = x1;
145  } else {
146  XA1 = x1;
147  XA2 = x2;
148  }
149  if (y2 < y1) {
150  YA1 = y2;
151  YA2 = y1;
152  } else {
153  YA1 = y1;
154  YA2 = y2;
155  }
156 
157  if (XA2 == XA1) {
158  //qWarning() << "x1 and x2 cannot be equal. Setting x2 = x1 + 1.0";
159  XA2 = XA1 + 1.0;
160  }
161  if (YA2 == YA1) {
162  //qWarning() << "y1 and y2 cannot be equal. Setting y2 = y1 + 1.0";
163  YA2 = YA1 + 1.0;
164  }
165  dataRect = QRectF(XA1, YA1, XA2 - XA1, YA2 - YA1);
166 
167  q->axis(LeftAxis)->setTickMarks(dataRect.y(), dataRect.height());
168  q->axis(BottomAxis)->setTickMarks(dataRect.x(), dataRect.width());
169 
170  if (secondDataRect.isNull()) {
171  q->axis(RightAxis)->setTickMarks(dataRect.y(), dataRect.height());
172  q->axis(TopAxis)->setTickMarks(dataRect.x(), dataRect.width());
173  }

472 void KPlotWidget::maskAlongLine(const QPointF &p1, const QPointF &p2, float fvalue)
473 {
474  if (! d->pixRect.contains(p1.toPoint()) && ! d->pixRect.contains(p2.toPoint())) {
475  return;
476  }
477 
478  int value = int(fvalue);
479 
480  //Determine slope and zeropoint of line
481  double m = (p2.y() - p1.y()) / (p2.x() - p1.x());
482  double y0 = p1.y() - m * p1.x();
483  QColor newColor;
484 
485  //Mask each pixel along the line joining p1 and p2
486  if (m > 1.0 || m < -1.0) { //step in y-direction
487  int y1 = int(p1.y());
488  int y2 = int(p2.y());
489  if (y1 > y2) {
490  y1 = int(p2.y());
491  y2 = int(p1.y());
492  }
493 
494  for (int y = y1; y <= y2; ++y) {
495  int x = int((y - y0) / m);
496  if (d->pixRect.contains(x, y)) {
497  newColor = QColor(d->plotMask.pixel(x, y));
498  newColor.setAlpha(100);
499  newColor.setRed(qMin(newColor.red() + value, 255));
500  d->plotMask.setPixel(x, y, newColor.rgba());
501  }
502  }
503 
504  } else { //step in x-direction
505  int x1 = int(p1.x());
506  int x2 = int(p2.x());
507  if (x1 > x2) {
508  x1 = int(p2.x());
509  x2 = int(p1.x());
510  }
511 
512  for (int x = x1; x <= x2; ++x) {
513  int y = int(y0 + m * x);
514  if (d->pixRect.contains(x, y)) {
515  newColor = QColor(d->plotMask.pixel(x, y));
516  newColor.setAlpha(100);
517  newColor.setRed(qMin(newColor.red() + value, 255));
518  d->plotMask.setPixel(x, y, newColor.rgba());
519  }
520  }
521  }
522 }

524 //Determine optimal placement for a text label for point pp. We want
525 //the label to be near point pp, but we don't want it to overlap with
526 //other labels or plot elements. We will use a "downhill simplex"
527 //algorithm to find a label position that minimizes the pixel values
528 //in the plotMask image over the label's rect(). The sum of pixel
529 //values in the label's rect is the "cost" of placing the label there.
530 //
531 //Because a downhill simplex follows the local gradient to find low
532 //values, it can get stuck in local minima. To mitigate this, we will
533 //iteratively attempt each of the initial path offset directions (up,
534 //down, right, left) in the order of increasing cost at each location.
535 void KPlotWidget::placeLabel(QPainter *painter, KPlotPoint *pp)
536 {
537  int textFlags = Qt::TextSingleLine | Qt::AlignCenter;
538 
539  QPointF pos = mapToWidget(pp->position());
540  if (! d->pixRect.contains(pos.toPoint())) {
541  return;
542  }
543 
544  QFontMetricsF fm(painter->font(), painter->device());
545  QRectF bestRect = fm.boundingRect(QRectF(pos.x(), pos.y(), 1, 1), textFlags, pp->label());
546  float xStep = 0.5 * bestRect.width();
547  float yStep = 0.5 * bestRect.height();
548  float maxCost = 0.05 * bestRect.width() * bestRect.height();
549  float bestCost = d->rectCost(bestRect);
550 
551  //We will travel along a path defined by the maximum decrease in
552  //the cost at each step. If this path takes us to a local minimum
553  //whose cost exceeds maxCost, then we will restart at the
554  //beginning and select the next-best path. The indices of
555  //already-tried paths are stored in the TriedPathIndex list.
556  //
557  //If we try all four first-step paths and still don't get below
558  //maxCost, then we'll adopt the local minimum position with the
559  //best cost (designated as bestBadCost).
560  int iter = 0;
561  QList<int> TriedPathIndex;
562  float bestBadCost = 10000;
563  QRectF bestBadRect;
564 
565  //needed to halt iteration from inside the switch
566  bool flagStop = false;
567 
568  while (bestCost > maxCost) {
569  //Displace the label up, down, left, right; determine which
570  //step provides the lowest cost
571  QRectF upRect = bestRect;
572  upRect.moveTop(upRect.top() + yStep);
573  float upCost = d->rectCost(upRect);
574  QRectF downRect = bestRect;
575  downRect.moveTop(downRect.top() - yStep);
576  float downCost = d->rectCost(downRect);
577  QRectF leftRect = bestRect;
578  leftRect.moveLeft(leftRect.left() - xStep);
579  float leftCost = d->rectCost(leftRect);
580  QRectF rightRect = bestRect;
581  rightRect.moveLeft(rightRect.left() + xStep);
582  float rightCost = d->rectCost(rightRect);
583 
584  //which direction leads to the lowest cost?
585  QList<float> costList;
586  costList << upCost << downCost << leftCost << rightCost;
587  int imin = -1;
588  for (int i = 0; i < costList.size(); ++i) {
589  if (iter == 0 && TriedPathIndex.contains(i)) {
590  continue; //Skip this first-step path, we already tried it!
591  }
592 
593  //If this first-step path doesn't improve the cost,
594  //skip this direction from now on
595  if (iter == 0 && costList[i] >= bestCost) {
596  TriedPathIndex.append(i);
597  continue;
598  }
599 
600  if (costList[i] < bestCost && (imin < 0 || costList[i] < costList[imin])) {
601 
602  imin = i;
603  }
604  }
605 
606  //Make a note that we've tried the current first-step path
607  if (iter == 0 && imin >= 0) {
608  TriedPathIndex.append(imin);
609  }
610 
611  //Adopt the step that produced the best cost
612  switch (imin) {
613  case 0: //up
614  bestRect.moveTop(upRect.top());
615  bestCost = upCost;
616  break;
617  case 1: //down
618  bestRect.moveTop(downRect.top());
619  bestCost = downCost;
620  break;
621  case 2: //left
622  bestRect.moveLeft(leftRect.left());
623  bestCost = leftCost;
624  break;
625  case 3: //right
626  bestRect.moveLeft(rightRect.left());
627  bestCost = rightCost;
628  break;
629  case -1: //no lower cost found!
630  //We hit a local minimum. Keep the best of these as bestBadRect
631  if (bestCost < bestBadCost) {
632  bestBadCost = bestCost;
633  bestBadRect = bestRect;
634  }
635 
636  //If all of the first-step paths have now been searched, we'll
637  //have to adopt the bestBadRect
638  if (TriedPathIndex.size() == 4) {
639  bestRect = bestBadRect;
640  flagStop = true; //halt iteration
641  break;
642  }
643 
644  //If we haven't yet tried all of the first-step paths, start over
645  if (TriedPathIndex.size() < 4) {
646  iter = -1; //anticipating the ++iter below
647  bestRect = fm.boundingRect(QRectF(pos.x(), pos.y(), 1, 1), textFlags, pp->label());
648  bestCost = d->rectCost(bestRect);
649  }
650  break;
651  }
652 
653  //Halt iteration, because we've tried all directions and
654  //haven't gotten below maxCost (we'll adopt the best
655  //local minimum found)
656  if (flagStop) {
657  break;
658  }
659 
660  ++iter;
661  }
662 
663  painter->drawText(bestRect, textFlags, pp->label());
664 
665  //Is a line needed to connect the label to the point?
666  float deltax = pos.x() - bestRect.center().x();
667  float deltay = pos.y() - bestRect.center().y();
668  float rbest = sqrt(deltax * deltax + deltay * deltay);
669  if (rbest > 20.0) {
670  //Draw a rectangle around the label
671  painter->setBrush(QBrush());
672  //QPen pen = painter->pen();
673  //pen.setStyle( Qt::DotLine );
674  //painter->setPen( pen );
675  painter->drawRoundRect(bestRect);
676 
677  //Now connect the label to the point with a line.
678  //The line is drawn from the center of the near edge of the rectangle
679  float xline = bestRect.center().x();
680  if (bestRect.left() > pos.x()) {
681  xline = bestRect.left();
682  }
683  if (bestRect.right() < pos.x()) {
684  xline = bestRect.right();
685  }
686 
687  float yline = bestRect.center().y();

702 float KPlotWidget::Private::rectCost(const QRectF &r) const
703 {
704  if (! plotMask.rect().contains(r.toRect())) {
705  return 10000.;
706  }
707 
708  //Compute sum of mask values in the rect r
709  QImage subMask = plotMask.copy(r.toRect());
710  int cost = 0;
711  for (int ix = 0; ix < subMask.width(); ++ix) {
712  for (int iy = 0; iy < subMask.height(); ++iy) {
713  cost += QColor(subMask.pixel(ix, iy)).red();
714  }
715  }
716 
717  return float(cost);
718 }

750 void KPlotWidget::drawAxes(QPainter *p)
751 {
752  if (d->showGrid) {
753  p->setPen(gridColor());
754 
755  //Grid lines are placed at locations of primary axes' major tickmarks
756  //vertical grid lines
757  Q_FOREACH (double xx, axis(BottomAxis)->majorTickMarks()) {
758  double px = d->pixRect.width() * (xx - d->dataRect.x()) / d->dataRect.width();
759  p->drawLine(QPointF(px, 0.0), QPointF(px, double(d->pixRect.height())));
760  }
761  //horizontal grid lines
762  Q_FOREACH (double yy, axis(LeftAxis)->majorTickMarks()) {
763  double py = d->pixRect.height() * (1.0 - (yy - d->dataRect.y()) / d->dataRect.height());
764  p->drawLine(QPointF(0.0, py), QPointF(double(d->pixRect.width()), py));
765  }
766  }
767 
768  p->setPen(foregroundColor());
769  p->setBrush(Qt::NoBrush);
770 
771  //set small font for tick labels
772  QFont f = p->font();
773  int s = f.pointSize();
774  f.setPointSize(s - 2);
775  p->setFont(f);
776 
777  /*** BottomAxis ***/
778  KPlotAxis *a = axis(BottomAxis);
779  if (a->isVisible()) {
780  //Draw axis line
781  p->drawLine(0, d->pixRect.height(), d->pixRect.width(), d->pixRect.height());
782 
783  // Draw major tickmarks
784  Q_FOREACH (double xx, a->majorTickMarks()) {
785  double px = d->pixRect.width() * (xx - d->dataRect.x()) / d->dataRect.width();
786  if (px > 0 && px < d->pixRect.width()) {
787  p->drawLine(QPointF(px, double(d->pixRect.height() - TICKOFFSET)),
788  QPointF(px, double(d->pixRect.height() - BIGTICKSIZE - TICKOFFSET)));
789 
790  //Draw ticklabel
791  if (a->areTickLabelsShown()) {
792  QRect r(int(px) - BIGTICKSIZE, d->pixRect.height() + BIGTICKSIZE, 2 * BIGTICKSIZE, BIGTICKSIZE);
793  p->drawText(r, Qt::AlignCenter | Qt::TextDontClip, a->tickLabel(xx));
794  }
795  }
796  }
797 
798  // Draw minor tickmarks
799  Q_FOREACH (double xx, a->minorTickMarks()) {
800  double px = d->pixRect.width() * (xx - d->dataRect.x()) / d->dataRect.width();
801  if (px > 0 && px < d->pixRect.width()) {
802  p->drawLine(QPointF(px, double(d->pixRect.height() - TICKOFFSET)),
803  QPointF(px, double(d->pixRect.height() - SMALLTICKSIZE - TICKOFFSET)));
804  }
805  }
806 
807  // Draw BottomAxis Label
808  if (! a->label().isEmpty()) {
809  QRect r(0, d->pixRect.height() + 2 * YPADDING, d->pixRect.width(), YPADDING);
810  p->drawText(r, Qt::AlignCenter, a->label());
811  }
812  } //End of BottomAxis
813 
814  /*** LeftAxis ***/
815  a = axis(LeftAxis);
816  if (a->isVisible()) {
817  //Draw axis line
818  p->drawLine(0, 0, 0, d->pixRect.height());
819 
820  // Draw major tickmarks
821  Q_FOREACH (double yy, a->majorTickMarks()) {
822  double py = d->pixRect.height() * (1.0 - (yy - d->dataRect.y()) / d->dataRect.height());
823  if (py > 0 && py < d->pixRect.height()) {
824  p->drawLine(QPointF(TICKOFFSET, py), QPointF(double(TICKOFFSET + BIGTICKSIZE), py));
825 
826  //Draw ticklabel
827  if (a->areTickLabelsShown()) {
828  QRect r(-2 * BIGTICKSIZE - SMALLTICKSIZE, int(py) - SMALLTICKSIZE, 2 * BIGTICKSIZE, 2 * SMALLTICKSIZE);
829  p->drawText(r, Qt::AlignRight | Qt::AlignVCenter | Qt::TextDontClip, a->tickLabel(yy));
830  }
831  }
832  }
833 
834  // Draw minor tickmarks
835  Q_FOREACH (double yy, a->minorTickMarks()) {
836  double py = d->pixRect.height() * (1.0 - (yy - d->dataRect.y()) / d->dataRect.height());
837  if (py > 0 && py < d->pixRect.height()) {
838  p->drawLine(QPointF(TICKOFFSET, py), QPointF(double(TICKOFFSET + SMALLTICKSIZE), py));
839  }
840  }
841 
842  //Draw LeftAxis Label. We need to draw the text sideways.
843  if (! a->label().isEmpty()) {
844  //store current painter translation/rotation state
845  p->save();
846 
847  //translate coord sys to left corner of axis label rectangle, then rotate 90 degrees.
848  p->translate(-3 * XPADDING, d->pixRect.height());
849  p->rotate(-90.0);
850 
851  QRect r(0, 0, d->pixRect.height(), XPADDING);
852  p->drawText(r, Qt::AlignCenter, a->label()); //draw the label, now that we are sideways
853 
854  p->restore(); //restore translation/rotation state
855  }
856  } //End of LeftAxis
857 
858  //Prepare for top and right axes; we may need the secondary data rect
859  double x0 = d->dataRect.x();
860  double y0 = d->dataRect.y();
861  double dw = d->dataRect.width();
862  double dh = d->dataRect.height();
863  if (secondaryDataRect().isValid()) {
864  x0 = secondaryDataRect().x();
865  y0 = secondaryDataRect().y();
866  dw = secondaryDataRect().width();
867  dh = secondaryDataRect().height();
868  }
869

871  a = axis(TopAxis);
872  if (a->isVisible()) {
873  //Draw axis line
874  p->drawLine(0, 0, d->pixRect.width(), 0);
875 
876  // Draw major tickmarks
877  Q_FOREACH (double xx, a->majorTickMarks()) {
878  double px = d->pixRect.width() * (xx - x0) / dw;
879  if (px > 0 && px < d->pixRect.width()) {
880  p->drawLine(QPointF(px, TICKOFFSET), QPointF(px, double(BIGTICKSIZE + TICKOFFSET)));
881 
882  //Draw ticklabel
883  if (a->areTickLabelsShown()) {
884  QRect r(int(px) - BIGTICKSIZE, (int) - 1.5 * BIGTICKSIZE, 2 * BIGTICKSIZE, BIGTICKSIZE);
885  p->drawText(r, Qt::AlignCenter | Qt::TextDontClip, a->tickLabel(xx));
886  }
887  }
888  }
889 
890  // Draw minor tickmarks
891  Q_FOREACH (double xx, a->minorTickMarks()) {
892  double px = d->pixRect.width() * (xx - x0) / dw;
893  if (px > 0 && px < d->pixRect.width()) {
894  p->drawLine(QPointF(px, TICKOFFSET), QPointF(px, double(SMALLTICKSIZE + TICKOFFSET)));
895  }
896  }
897 
898  // Draw TopAxis Label
899  if (! a->label().isEmpty()) {
900  QRect r(0, 0 - 3 * YPADDING, d->pixRect.width(), YPADDING);
901  p->drawText(r, Qt::AlignCenter, a->label());
902  }
903  } //End of TopAxis
904 
905  /*** RightAxis ***/
906  a = axis(RightAxis);
907  if (a->isVisible()) {
908  //Draw axis line
909  p->drawLine(d->pixRect.width(), 0, d->pixRect.width(), d->pixRect.height());
910 
911  // Draw major tickmarks
912  Q_FOREACH (double yy, a->majorTickMarks()) {
913  double py = d->pixRect.height() * (1.0 - (yy - y0) / dh);
914  if (py > 0 && py < d->pixRect.height()) {
915  p->drawLine(QPointF(double(d->pixRect.width() - TICKOFFSET), py),
916  QPointF(double(d->pixRect.width() - TICKOFFSET - BIGTICKSIZE), py));
917 
918  //Draw ticklabel
919  if (a->areTickLabelsShown()) {
920  QRect r(d->pixRect.width() + SMALLTICKSIZE, int(py) - SMALLTICKSIZE, 2 * BIGTICKSIZE, 2 * SMALLTICKSIZE);
921  p->drawText(r, Qt::AlignLeft | Qt::AlignVCenter | Qt::TextDontClip, a->tickLabel(yy));
922  }
923  }
924  }
925 
926  // Draw minor tickmarks
927  Q_FOREACH (double yy, a->minorTickMarks()) {
928  double py = d->pixRect.height() * (1.0 - (yy - y0) / dh);
929  if (py > 0 && py < d->pixRect.height()) {
930  p->drawLine(QPointF(double(d->pixRect.width() - 0.0), py),
931  QPointF(double(d->pixRect.width() - 0.0 - SMALLTICKSIZE), py));
932  }
933  }
934 
935  //Draw RightAxis Label. We need to draw the text sideways.
936  if (! a->label().isEmpty()) {
937  //store current painter translation/rotation state
938  p->save();
939 
940  //translate coord sys to left corner of axis label rectangle, then rotate 90 degrees.
941  p->translate(d->pixRect.width() + 2 * XPADDING, d->pixRect.height());
942  p->rotate(-90.0);
943 
944  QRect r(0, 0, d->pixRect.height(), XPADDING);
945  p->drawText(r, Qt::AlignCenter, a->label()); //draw the label, now that we are sideways
946 
947  p->restore(); //restore translation/rotation state
948  }
949  } //End of RightAxis
