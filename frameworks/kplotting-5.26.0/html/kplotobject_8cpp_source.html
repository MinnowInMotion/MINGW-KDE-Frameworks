225 void KPlotObject::draw(QPainter *painter, KPlotWidget *pw)
226 {
227  //Order of drawing determines z-distance: Bars in the back, then lines,
228  //then points, then labels.
229 
230  if (d->type & Bars) {
231  painter->setPen(barPen());
232  painter->setBrush(barBrush());
233 
234  double w = 0;
235  for (int i = 0; i < d->pList.size(); ++i) {
236  if (d->pList[i]->barWidth() == 0.0) {
237  if (i < d->pList.size() - 1) {
238  w = d->pList[i + 1]->x() - d->pList[i]->x();
239  }
240  //For the last bin, we'll just keep the previous width
241 
242  } else {
243  w = d->pList[i]->barWidth();
244  }
245 
246  QPointF pp = d->pList[i]->position();
247  QPointF p1(pp.x() - 0.5 * w, 0.0);
248  QPointF p2(pp.x() + 0.5 * w, pp.y());
249  QPointF sp1 = pw->mapToWidget(p1);
250  QPointF sp2 = pw->mapToWidget(p2);
251 
252  QRectF barRect = QRectF(sp1.x(), sp1.y(), sp2.x() - sp1.x(), sp2.y() - sp1.y()).normalized();
253  painter->drawRect(barRect);
254  pw->maskRect(barRect, 0.25);
255  }
256  }
257 
258  //Draw lines:
259  if (d->type & Lines) {
260  painter->setPen(linePen());
261 
262  QPointF Previous = QPointF(); //Initialize to null
263 
264  Q_FOREACH (KPlotPoint *pp, d->pList) {
265  //q is the position of the point in screen pixel coordinates
266  QPointF q = pw->mapToWidget(pp->position());
267 
268  if (! Previous.isNull()) {
269  painter->drawLine(Previous, q);
270  pw->maskAlongLine(Previous, q);
271  }
272 
273  Previous = q;
274  }
275  }
276 
277  //Draw points:
278  if (d->type & Points) {
279 
280  Q_FOREACH (KPlotPoint *pp, d->pList) {
281  //q is the position of the point in screen pixel coordinates
282  QPointF q = pw->mapToWidget(pp->position());
283  if (pw->pixRect().contains(q.toPoint(), false)) {
284  double x1 = q.x() - size();
285  double y1 = q.y() - size();
286  QRectF qr = QRectF(x1, y1, 2 * size(), 2 * size());
287 
288  //Mask out this rect in the plot for label avoidance
289  pw->maskRect(qr, 2.0);
290 
291  painter->setPen(pen());
292  painter->setBrush(brush());
293 
294  switch (pointStyle()) {
295  case Circle:
296  painter->drawEllipse(qr);
297  break;
298 
299  case Letter:
300  painter->drawText(qr, Qt::AlignCenter, pp->label().left(1));
301  break;
302 
303  case Triangle: {
304  QPolygonF tri;
305  tri << QPointF(q.x() - size(), q.y() + size())
306  << QPointF(q.x(), q.y() - size())
307  << QPointF(q.x() + size(), q.y() + size());
308  painter->drawPolygon(tri);
309  break;
310  }
311 
312  case Square:
313  painter->drawRect(qr);
314  break;
315 
316  case Pentagon: {
317  QPolygonF pent;
318  pent << QPointF(q.x(), q.y() - size())
319  << QPointF(q.x() + size(), q.y() - 0.309 * size())
320  << QPointF(q.x() + 0.588 * size(), q.y() + size())
321  << QPointF(q.x() - 0.588 * size(), q.y() + size())
322  << QPointF(q.x() - size(), q.y() - 0.309 * size());
323  painter->drawPolygon(pent);
324  break;
325  }
326 
327  case Hexagon: {
328  QPolygonF hex;
329  hex << QPointF(q.x(), q.y() + size())
330  << QPointF(q.x() + size(), q.y() + 0.5 * size())
331  << QPointF(q.x() + size(), q.y() - 0.5 * size())
332  << QPointF(q.x(), q.y() - size())
333  << QPointF(q.x() - size(), q.y() + 0.5 * size())
334  << QPointF(q.x() - size(), q.y() - 0.5 * size());
335  painter->drawPolygon(hex);
336  break;
337  }
338 
339  case Asterisk:
340  painter->drawLine(q, QPointF(q.x(), q.y() + size()));
341  painter->drawLine(q, QPointF(q.x() + size(), q.y() + 0.5 * size()));
342  painter->drawLine(q, QPointF(q.x() + size(), q.y() - 0.5 * size()));
343  painter->drawLine(q, QPointF(q.x(), q.y() - size()));
344  painter->drawLine(q, QPointF(q.x() - size(), q.y() + 0.5 * size()));
345  painter->drawLine(q, QPointF(q.x() - size(), q.y() - 0.5 * size()));
346  break;
347 
348  case Star: {
349  QPolygonF star;
350  star << QPointF(q.x(), q.y() - size())
351  << QPointF(q.x() + 0.2245 * size(), q.y() - 0.309 * size())
352  << QPointF(q.x() + size(), q.y() - 0.309 * size())
353  << QPointF(q.x() + 0.363 * size(), q.y() + 0.118 * size())
354  << QPointF(q.x() + 0.588 * size(), q.y() + size())
355  << QPointF(q.x(), q.y() + 0.382 * size())
356  << QPointF(q.x() - 0.588 * size(), q.y() + size())
357  << QPointF(q.x() - 0.363 * size(), q.y() + 0.118 * size())
358  << QPointF(q.x() - size(), q.y() - 0.309 * size())
359  << QPointF(q.x() - 0.2245 * size(), q.y() - 0.309 * size());
360  painter->drawPolygon(star);
361  break;
362  }
363 
364  default:
365  break;
366  }
367  }
368  }
369  }
370 
371  //Draw labels
372  painter->setPen(labelPen());
373 
374  Q_FOREACH (KPlotPoint *pp, d->pList) {
375  QPoint q = pw->mapToWidget(pp->position()).toPoint();
376  if (pw->pixRect().contains(q, false) && ! pp->label().isEmpty()) {
377  pw->placeLabel(painter, pp);
378  }
379  }
380 
381 }
