86 int CipherBlockChain::encrypt(void *block, int len)
87 {
88  if (_cipher && !_reader) {
89  int rc;
90 
91  _writer |= 1;
92 
93  initRegister();
94 
95  if ((len % _len) >0) {
96  qDebug() << "Block length given encrypt (" << len << ") is not a multiple of " << _len;
97  return -1;
98  }
99 
100  char *elemBlock = static_cast<char*>(block);
101  for (int b = 0; b < len/_len; b++) {
102 
103  // This might be optimizable
104  char *tb = static_cast<char*>(elemBlock);
105  for (int i = 0; i < _len; i++) {
106  *tb++ ^= ((char *)_register)[i];
107  }
108 
109  rc = _cipher->encrypt(elemBlock, _len);
110 
111  if (rc != -1) {
112  memcpy(_register, elemBlock, _len);
113  }
114  elemBlock += _len;
115  }
116 
117  return rc;
118  }
119  return -1;
120 }

124 int CipherBlockChain::decryptECB(void *block, int len) {
125  if (_cipher && !_writer) {
126  int rc;
127 
128  _reader |= 1;
129 
130  if (!_register) {
131  _register = new unsigned char[len];
132  _len = len;
133  memset(_register, 0, len);
134  } else if (len > _len) {
135  return -1;
136  }
137 
138  if (!_next) {
139  _next = new unsigned char[_len];
140  }
141  memcpy(_next, block, _len);
142 
143  rc = _cipher->decrypt(block, len);
144 
145  if (rc != -1) {
146  // This might be optimizable
147  char *tb = (char *)block;
148  for (int i = 0; i < len; i++) {
149  tb[i] ^= ((char *)_register)[i];
150  }
151  }
152 
153  void *temp;
154  temp = _next;
155  _next = _register;
156  _register = temp;
157 
158  return rc;
159  }
160  return -1;
161 }

163 int CipherBlockChain::decrypt(void *block, int len)
164 {
165  if (_useECBforReading) {
166  qDebug() << "decrypting using ECB!";
167  return decryptECB(block, len);
168  }
169 
170  if (_cipher && !_writer) {
171  int rc = 0;
172 
173  _reader |= 1;
174 
175  initRegister();
176 
177  if ((len % _len) >0) {
178  qDebug() << "Block length given for decrypt (" << len << ") is not a multiple of " << _len;
179  return -1;
180  }
181 
182  char *elemBlock = static_cast<char*>(block);
183  for (int b = 0; b < len/_len; b++) {
184  if (_next == nullptr) {
185  _next = new unsigned char[_len];
186  }
187  memcpy(_next, elemBlock, _len);
188 
189  int bytesDecrypted = _cipher->decrypt(elemBlock, _len);
190 
191  if (bytesDecrypted != -1) {
192  rc += bytesDecrypted;
193  // This might be optimizable
194  char *tb = (char *)elemBlock;
195  for (int i = 0; i < _len; i++) {
196  *tb++ ^= ((char *)_register)[i];
197  }
198  }
199 
200  void *temp;
201  temp = _next;
202  _next = _register;
203  _register = temp;
204 
205  elemBlock += _len;
206  }
207 
208  return rc;
209  }
210  return -1;
211 }
