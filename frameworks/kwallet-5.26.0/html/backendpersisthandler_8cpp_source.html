192 int BlowfishPersistHandler::write(Backend *wb, QSaveFile &sf, QByteArray &version, WId)
193 {
194  assert(wb->_cipherType == BACKEND_CIPHER_BLOWFISH);
195 
196  if (_useECBforReading) {
197  qDebug() << "This wallet used ECB and is now saved using CBC";
198  _useECBforReading = false;
199  }
200 
201  version[2] = KWALLET_CIPHER_BLOWFISH_CBC;
202  if (!wb->_useNewHash) {
203  version[3] = KWALLET_HASH_SHA1;
204  } else {
205  version[3] = KWALLET_HASH_PBKDF2_SHA512;//Since 4.13 we always use PBKDF2_SHA512
206  }
207 
208  if (sf.write(version) != 4) {
209  sf.cancelWriting();
210  return -4; // write error
211  }
212 
213  // Holds the hashes we write out
214  QByteArray hashes;
215  QDataStream hashStream(&hashes, QIODevice::WriteOnly);
216  QCryptographicHash md5(QCryptographicHash::Md5);
217  hashStream << static_cast<quint32>(wb->_entries.count());
218 
219  // Holds decrypted data prior to encryption
220  QByteArray decrypted;
221 
222  // FIXME: we should estimate the amount of data we will write in each
223  // buffer and resize them approximately in order to avoid extra
224  // resizes.
225 
226  // populate decrypted
227  QDataStream dStream(&decrypted, QIODevice::WriteOnly);
228  for (Backend::FolderMap::ConstIterator i = wb->_entries.constBegin(); i != wb->_entries.constEnd(); ++i) {
229  dStream << i.key();
230  dStream << static_cast<quint32>(i.value().count());
231 
232  md5.reset();
233  md5.addData(i.key().toUtf8());
234  hashStream.writeRawData(md5.result().constData(), 16);
235  hashStream << static_cast<quint32>(i.value().count());
236 
237  for (Backend::EntryMap::ConstIterator j = i.value().constBegin(); j != i.value().constEnd(); ++j) {
238  dStream << j.key();
239  dStream << static_cast<qint32>(j.value()->type());
240  dStream << j.value()->value();
241 
242  md5.reset();
243  md5.addData(j.key().toUtf8());
244  hashStream.writeRawData(md5.result().constData(), 16);
245  }
246  }
247 
248  if (sf.write(hashes) != hashes.size()) {
249  sf.cancelWriting();
250  return -4; // write error
251  }
252 
253  // calculate the hash of the file
254  SHA1 sha;
255  BlowFish _bf;
256  CipherBlockChain bf(&_bf);
257 
258  sha.process(decrypted.data(), decrypted.size());
259 
260  // prepend and append the random data
261  QByteArray wholeFile;
262  long blksz = bf.blockSize();
263  long newsize = decrypted.size() +
264  blksz + // encrypted block
265  4 + // file size
266  20; // size of the SHA hash
267 
268  int delta = (blksz - (newsize % blksz));
269  newsize += delta;
270  wholeFile.resize(newsize);
271 
272  QByteArray randBlock;
273  randBlock.resize(blksz + delta);
274  if (getRandomBlock(randBlock) < 0) {
275  sha.reset();
276  decrypted.fill(0);
277  sf.cancelWriting();
278  return -3; // Fatal error: can't get random
279  }
280 
281  for (int i = 0; i < blksz; i++) {
282  wholeFile[i] = randBlock[i];
283  }
284 
285  for (int i = 0; i < 4; i++) {
286  wholeFile[(int)(i + blksz)] = (decrypted.size() >> 8 * (3 - i)) & 0xff;
287  }
288 
289  for (int i = 0; i < decrypted.size(); i++) {
290  wholeFile[(int)(i + blksz + 4)] = decrypted[i];
291  }
292 
293  for (int i = 0; i < delta; i++) {
294  wholeFile[(int)(i + blksz + 4 + decrypted.size())] = randBlock[(int)(i + blksz)];
295  }
296 
297  const char *hash = (const char *)sha.hash();
298  for (int i = 0; i < 20; i++) {
299  wholeFile[(int)(newsize - 20 + i)] = hash[i];
300  }
301 
302  sha.reset();
303  decrypted.fill(0);
304 
305  // encrypt the data
306  if (!bf.setKey(wb->_passhash.data(), wb->_passhash.size() * 8)) {
307  wholeFile.fill(0);
308  sf.cancelWriting();
309  return -2; // encrypt error
310  }
311 
312  int rc = bf.encrypt(wholeFile.data(), wholeFile.size());
313  if (rc < 0) {
314  wholeFile.fill(0);
315  sf.cancelWriting();
316  return -2; // encrypt error
317  }
318 
319  // write the file
320  auto written = sf.write(wholeFile);
321  if (written != wholeFile.size()) {
322  wholeFile.fill(0);
323  sf.cancelWriting();
324  return -4; // write error
325  }
326  if (!sf.commit()) {
327  qDebug() << "WARNING: wallet sync to disk failed! QSaveFile status was " << sf.errorString();
328  wholeFile.fill(0);
329  return -4; // write error
330  }
331 
332  wholeFile.fill(0);
333 
334  return 0;

337 int BlowfishPersistHandler::read(Backend *wb, QFile &db, WId)
338 {
339  wb->_cipherType = BACKEND_CIPHER_BLOWFISH;
340  wb->_hashes.clear();
341  // Read in the hashes
342  QDataStream hds(&db);
343  quint32 n;
344  hds >> n;
345  if (n > 0xffff) { // sanity check
346  return -43;
347  }
348 
349  for (size_t i = 0; i < n; ++i) {
350  Digest d, d2; // judgment day
351  MD5Digest ba;
352  QMap<MD5Digest, QList<MD5Digest> >::iterator it;
353  quint32 fsz;
354  if (hds.atEnd()) {
355  return -43;
356  }
357  hds.readRawData(d, 16);
358  hds >> fsz;
359  ba = MD5Digest(reinterpret_cast<char *>(d));
360  it = wb->_hashes.insert(ba, QList<MD5Digest>());
361  for (size_t j = 0; j < fsz; ++j) {
362  hds.readRawData(d2, 16);
363  ba = MD5Digest(d2);
364  (*it).append(ba);
365  }
366  }
367 
368  // Read in the rest of the file.
369  QByteArray encrypted = db.readAll();
370  assert(encrypted.size() < db.size());
371 
372  BlowFish _bf;
373  CipherBlockChain bf(&_bf, _useECBforReading);
374  int blksz = bf.blockSize();
375  if ((encrypted.size() % blksz) != 0) {
376  return -5; // invalid file structure
377  }
378 
379  bf.setKey((void *)wb->_passhash.data(), wb->_passhash.size() * 8);
380 
381  if (!encrypted.data()) {
382  wb->_passhash.fill(0);
383  encrypted.fill(0);
384  return -7; // file structure error
385  }
386 
387  int rc = bf.decrypt(encrypted.data(), encrypted.size());
388  if (rc < 0) {
389  wb->_passhash.fill(0);
390  encrypted.fill(0);
391  return -6; // decrypt error
392  }
393 
394  const char *t = encrypted.data();
395 
396  // strip the leading data
397  t += blksz; // one block of random data
398 
399  // strip the file size off
400  long fsize = 0;
401 
402  fsize |= (long(*t) << 24) & 0xff000000;
403  t++;
404  fsize |= (long(*t) << 16) & 0x00ff0000;
405  t++;
406  fsize |= (long(*t) << 8) & 0x0000ff00;
407  t++;
408  fsize |= long(*t) & 0x000000ff;
409  t++;
410 
411  if (fsize < 0 || fsize > long(encrypted.size()) - blksz - 4) {
412  qDebug() << "fsize: " << fsize << " encrypted.size(): " << encrypted.size() << " blksz: " << blksz;
413  encrypted.fill(0);
414  return -9; // file structure error.
415  }
416 
417  // compute the hash ourself
418  SHA1 sha;
419  sha.process(t, fsize);
420  const char *testhash = (const char *)sha.hash();
421 
422  // compare hashes
423  int sz = encrypted.size();
424  for (int i = 0; i < 20; i++) {
425  if (testhash[i] != encrypted[sz - 20 + i]) {
426  encrypted.fill(0);
427  sha.reset();
428  return -8; // hash error.
429  }
430  }
431 
432  sha.reset();
433 
434  // chop off the leading blksz+4 bytes
435  QByteArray tmpenc(encrypted.data() + blksz + 4, fsize);
436  encrypted = tmpenc;
437  tmpenc.fill(0);
438 
439  // Load the data structures up
440  QDataStream eStream(encrypted);
441 
442  while (!eStream.atEnd()) {
443  QString folder;
444  quint32 n;
445 
446  eStream >> folder;
447  eStream >> n;
448 
449  // Force initialisation
450  wb->_entries[folder].clear();
451 
452  for (size_t i = 0; i < n; ++i) {
453  QString key;
454  KWallet::Wallet::EntryType et = KWallet::Wallet::Unknown;
455  Entry *e = new Entry;
456  eStream >> key;
457  qint32 x = 0; // necessary to read properly
458  eStream >> x;
459  et = static_cast<KWallet::Wallet::EntryType>(x);
460 
461  switch (et) {
462  case KWallet::Wallet::Password:
463  case KWallet::Wallet::Stream:
464  case KWallet::Wallet::Map:
465  break;
466  default: // Unknown entry
467  delete e;
468  continue;
469  }
470 
471  QByteArray a;
472  eStream >> a;
473  e->setValue(a);
474  e->setType(et);
475  e->setKey(key);
476  wb->_entries[folder][key] = e;
477  }
478  }
479 
480  wb->_open = true;
481  encrypted.fill(0);
482  return 0;
483 }
