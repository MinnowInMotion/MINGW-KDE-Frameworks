45 class MD5Digest : public QByteArray

47 public:
48  MD5Digest() : QByteArray(16, 0) {}
49  MD5Digest(const char *data) : QByteArray(data, 16) {}
50  MD5Digest(const QByteArray &digest) : QByteArray(digest) {}
51  virtual ~MD5Digest() {}
52 
53  int operator<(const MD5Digest &r) const
54  {
55  int i = 0;
56  char x, y;
57  for (; i < 16; ++i) {
58  x = at(i);
59  y = r.at(i);
60  if (x != y) {
61  break;
62  }
63  }
64  if (i < 16 && x < y) {
65  return 1;
66  }
67  return 0;
68  }
69 };
