171  template <typename V>
172  int forEachItemThatMatches(const QString &key, V verb)
173  {
174  int rc = -1;
175  KSecretsService::StringStringMap attrs;
176  attrs[KSS_ATTR_ENTRYFOLDER] = folder;
177  KSecretsService::SearchCollectionItemsJob *searchItemsJob = secretsCollection->searchItems(attrs);
178  if (searchItemsJob->exec()) {
179  QRegExp re(key, Qt::CaseSensitive, QRegExp::Wildcard);
180  foreach (KSecretsService::SearchCollectionItemsJob::Item item, searchItemsJob->items()) {
181  KSecretsService::ReadItemPropertyJob *readLabelJob = item->label();
182  if (readLabelJob->exec()) {
183  QString label = readLabelJob->propertyValue().toString();
184  if (re.exactMatch(label)) {
185  if (verb(this, label, item.data())) {
186  rc = 0; // one successful iteration already produced results, so success return
187  }
188  }
189  } else {
190  qDebug() << "Cannot execute ReadItemPropertyJob " << readLabelJob->errorString();
191  }
192  }
193  } else {
194  qDebug() << "Cannot execute KSecretsService::SearchCollectionItemsJob " << searchItemsJob->errorString();
195  }

429 Wallet *Wallet::openWallet(const QString &name, WId w, OpenType ot)
430 {
431  if (w == 0) {
432  qDebug() << "Pass a valid window to KWallet::Wallet::openWallet().";
433  }
434 
435  if (!walletLauncher()->m_walletEnabled) {
436  qDebug() << "User disabled the wallet system so returning 0 here.";
437  return 0;
438  }
439 
440 #if HAVE_KSECRETSSERVICE
441  if (walletLauncher()->m_useKSecretsService) {
442  Wallet *wallet = new Wallet(-1, name);
443  // FIXME: should we specify CreateCollection or OpenOnly here?
444  wallet->d->secretsCollection = KSecretsService::Collection::findCollection(name, KSecretsService::Collection::CreateCollection, QVariantMap(), w);
445  connect(wallet->d->secretsCollection, SIGNAL(statusChanged(int)), wallet, SLOT(slotCollectionStatusChanged(int)));
446  connect(wallet->d->secretsCollection, SIGNAL(deleted()), wallet, SLOT(slotCollectionDeleted()));
447  if (ot == Synchronous) {
448  qDebug() << "WARNING openWallet OpenType=Synchronous requested";
449  // TODO: not sure what to do with in this case; however, all other KSecretsService API methods are already
450  // async and will perform sync inside this API because of it's design
451  }
452  return wallet;
453  } else {
454 #endif
455  Wallet *wallet = new Wallet(-1, name);
456 
457  // connect the daemon's opened signal to the slot filtering the
458  // signals we need
459  connect(&walletLauncher()->getInterface(), SIGNAL(walletAsyncOpened(int,int)),
460  wallet, SLOT(walletAsyncOpened(int,int)));
461 
462  // Make sure the password prompt window will be visible and activated
463  KWindowSystem::allowExternalProcessWindowActivation();
464 
465  org::kde::KWallet &interface = walletLauncher->getInterface();
466 
467  // do the call
468  QDBusReply<int> r;
469  if (ot == Synchronous) {
470  interface.setTimeout(0x7FFFFFFF); // Don't timeout after 25s, but 24 days
471  r = interface.open(name, (qlonglong)w, appid());
472  interface.setTimeout(-1); // Back to the default 25s
473  // after this call, r would contain a transaction id >0 if OK or -1 if NOK
474  // if OK, the slot walletAsyncOpened should have been received, but the transaction id
475  // will not match. We'll get that handle from the reply - see below
476  } else if (ot == Asynchronous) {
477  r = interface.openAsync(name, (qlonglong)w, appid(), true);
478  } else if (ot == Path) {
479  r = interface.openPathAsync(name, (qlonglong)w, appid(), true);
480  } else {
481  delete wallet;
482  return 0;
483  }
484  // error communicating with the daemon (maybe not running)
485  if (!r.isValid()) {
486  qDebug() << "Invalid DBus reply: " << r.error();
487  delete wallet;
488  return 0;
489  }
490  wallet->d->transactionId = r.value();
491 
492  if (ot == Synchronous || ot == Path) {
493  // check for an immediate error
494  if (wallet->d->transactionId < 0) {
495  delete wallet;
496  wallet = 0;
497  } else {
498  wallet->d->handle = r.value();
499  }
500  } else if (ot == Asynchronous) {
501  if (wallet->d->transactionId < 0) {
502  QTimer::singleShot(0, wallet, SLOT(emitWalletAsyncOpenError()));
503  // client code is responsible for deleting the wallet
504  }
505  }
506 
507  return wallet;
508 #if HAVE_KSECRETSSERVICE
509  }
510 #endif
511 }
