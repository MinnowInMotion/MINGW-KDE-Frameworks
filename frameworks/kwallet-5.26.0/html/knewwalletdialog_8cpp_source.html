96 struct AddKeyToList {
97  QTableWidget *_list;
98  int _row;
99  AddKeyToList(QTableWidget *list) : _list(list), _row(0) {}
100  void operator()(const GpgME::Key &k)
101  {
102  GpgME::UserID uid = k.userID(0);
103  QString name(uid.name());
104  if (uid.comment()) {
105  name = QStringLiteral("%1 (%2)").arg(name).arg(uid.comment());
106  }
107  _list->setItem(_row, 0, new QTableWidgetItem(name));
108  _list->setItem(_row, 1, new QTableWidgetItem(uid.email()));
109  _list->setItem(_row, 2, new QTableWidgetItem(k.shortKeyID()));
110  QVariant varKey;
111  varKey.setValue(k);
112  _list->item(_row, 0)->setData(Qt::UserRole, varKey);
113  ++_row;
114  }
115 };

117 void KNewWalletDialogGpg::initializePage()
118 {
119  if (_alreadyInitialized) {
120  return;
121  }
122 
123  registerField(QStringLiteral("key"), this);
124 
125  GpgME::initializeLibrary();
126  GpgME::Error err = GpgME::checkEngine(GpgME::OpenPGP);
127  if (err) {
128  qDebug() << "OpenPGP not supported on your system!";
129  KMessageBox::error(this, i18n("The QGpgME library failed to initialize for the OpenPGP protocol. Please check your system's configuration then try again."));
130  emit completeChanged();
131  return;
132  }
133  boost::shared_ptr< GpgME::Context > _ctx(GpgME::Context::createForProtocol(GpgME::OpenPGP));
134  if (0 == _ctx) {
135  KMessageBox::error(this, i18n("The QGpgME library failed to initialize for the OpenPGP protocol. Please check your system's configuration then try again."));
136  emit completeChanged();
137  return;
138  }
139  _ctx->setKeyListMode(GpgME::Local);
140 
141  std::vector< GpgME::Key > keys;
142  int row = 0;
143  err = _ctx->startKeyListing();
144  while (!err) {
145  GpgME::Key k = _ctx->nextKey(err);
146  if (err) {
147  break;
148  }
149  if (!k.isInvalid() && k.canEncrypt() && (k.ownerTrust() == GpgME::Key::Ultimate)) {
150  keys.push_back(k);
151  }
152  }
153  _ctx->endKeyListing();
154 
155  if (keys.size() == 0) {
156  KMessageBox::error(this, i18n("Seems that your system has no keys suitable for encryption. Please set-up at least one encryption key, then try again."));
157  emit completeChanged();
158  return;
159  }
160 
161  _ui.listCertificates->setRowCount(keys.size());
162  std::for_each(keys.begin(), keys.end(), AddKeyToList(_ui.listCertificates));
163  _ui.listCertificates->resizeColumnsToContents();
164  _ui.listCertificates->setCurrentCell(0, 0);
165 
166  _alreadyInitialized = true;
167 }
