218 void KWalletD::processTransactions()
219 {
220  if (_processing) {
221  return;
222  }
223 
224  _processing = true;
225 
226  // Process remaining transactions
227  while (!_transactions.isEmpty()) {
228  _curtrans = _transactions.takeFirst();
229  int res;
230 
231  assert(_curtrans->tType != KWalletTransaction::Unknown);
232 
233  switch (_curtrans->tType) {
234  case KWalletTransaction::Open:
235  res = doTransactionOpen(_curtrans->appid, _curtrans->wallet,
236  _curtrans->isPath, _curtrans->wId, _curtrans->modal,
237  _curtrans->service);
238 
239  // multiple requests from the same client
240  // should not produce multiple password
241  // dialogs on a failure
242  if (res < 0) {
243  QList<KWalletTransaction*>::iterator it;
244  for (it = _transactions.begin(); it != _transactions.end();
245  ++it) {
246  KWalletTransaction* x = *it;
247  if (_curtrans->appid == x->appid
248  && x->tType == KWalletTransaction::Open
249  && x->wallet == _curtrans->wallet
250  && x->wId == _curtrans->wId) {
251  x->tType = KWalletTransaction::OpenFail;
252  }
253  }
254  }
255  else if (_curtrans->cancelled) {
256  // the wallet opened successfully but the application
257  // opening exited/crashed while the dialog was still shown.
258  KWalletTransaction* _xact
259  = new KWalletTransaction(_curtrans->connection);
260  _xact->tType = KWalletTransaction::CloseCancelled;
261  _xact->appid = _curtrans->appid;
262  _xact->wallet = _curtrans->wallet;
263  _xact->service = _curtrans->service;
264  _transactions.append(_xact);
265  }
266 
267  // emit the AsyncOpened signal as a reply
268  _curtrans->res = res;
269  emit walletAsyncOpened(_curtrans->tId, res);
270  break;
271 
272  case KWalletTransaction::OpenFail:
273  // emit the AsyncOpened signal with an invalid handle
274  _curtrans->res = -1;
275  emit walletAsyncOpened(_curtrans->tId, -1);
276  break;
277 
278  case KWalletTransaction::ChangePassword:
279  doTransactionChangePassword(
280  _curtrans->appid, _curtrans->wallet, _curtrans->wId);
281  break;
282 
283  case KWalletTransaction::CloseCancelled:
284  doTransactionOpenCancelled(
285  _curtrans->appid, _curtrans->wallet, _curtrans->service);
286  break;
287 
288  case KWalletTransaction::Unknown:
289  break;
290  default:
291  break;
292  }
293 
294  // send delayed dbus message reply to the caller
295  if (_curtrans->message.type() != QDBusMessage::InvalidMessage) {
296  if (_curtrans->connection.isConnected()) {
297  QDBusMessage reply = _curtrans->message.createReply();
298  reply << _curtrans->res;
299  _curtrans->connection.send(reply);
300  }
301  }
302 
303  delete _curtrans;
304  _curtrans = 0;
305  }
306 
307  _processing = false;
308 }
