47 void MigrationAgent::migrateWallets()
48 {
49  // the migration is done in several steps :
50  // is the setting "alreadyMigrated" present and true ?
51  // if yes, then stop here
52  // if not, then is kwalletd present (the daemon from KDE4) ?
53  // if not, then stop here
54  // if yes, then start the migration wizard
55  // if the migration wizard returns without error
56  // create "alreadyMigrated=true" setting
57  qDebug() << "Migration agent starting...";
58  if (!isAlreadyMigrated()) {
59  if (connectOldDaemon()) {
60  if (!isEmptyOldWallet()) {
61  if (isMigrationWizardOk()) {
62  setAlreadyMigrated();
63  } else {
64  qDebug() << "Migration wizard returned an error or has been canceled. The migration agent will resume upon next daemon start";
65  }
66  } else {
67  qDebug() << "Old wallet is empty. No need to migrate.";
68  setAlreadyMigrated();
69  }
70  } else {
71  qDebug() << "KDE4 kwalletd not present, stopping migration agent";
72  }
73  } else {
74  qDebug() << "old wallets were already migrated";
75  }
76  qDebug() << "Migration agent stop.";
77 }

93 bool MigrationAgent::connectOldDaemon()
94 {
95  // the old daemon may not have been started, so attempt start it
96  // NOTE we provide a "fake" org.kde.kwalletd.service file - see the project structure
97  // however, this thing is a HACK as we cannot tell here, in KF5, user's KDE4 install prefix
98  // the provided .service file assumes /usr/bin
99  QDBusConnectionInterface *bus = QDBusConnection::sessionBus().interface();
100  if (!bus->isServiceRegistered(QStringLiteral(SERVICE_KWALLETD4))) {
101  qDebug() << "kwalletd not started. Attempting start...";
102  QDBusReply<void> reply = bus->startService(SERVICE_KWALLETD4);
103  if (!reply.isValid()) {
104  qDebug() << "Couldn't start kwalletd: " << reply.error();
105  return false;
106  }
107  if (!bus->isServiceRegistered(QStringLiteral(SERVICE_KWALLETD4))) {
108  qDebug() << "The kwalletd service is still not registered after start attemtp. Aborting migration";
109  return false;
110  } else {
111  qDebug() << "The kwalletd service has been registered";
112  }
113  }
114 
115  _kde4_daemon = new org::kde::KWallet(QLatin1String(SERVICE_KWALLETD4), QStringLiteral("/modules/kwalletd"), QDBusConnection::sessionBus());
116  return _kde4_daemon->isValid();
117 }

119 bool MigrationAgent::isMigrationWizardOk()
120 {
121  bool ok = false;
122 
123  // The migration wizard would no longer been shown by default.
124  // see BUG 351056
125  // NOTE if user wants to show the migration wizard, then he should add the
126  // following setting to the kwalletrc:
127  // [Migration]
128  // showMigrationWizard=true
129  KConfig kwalletrc(QStringLiteral("kwalletrc"));
130  KConfigGroup cfg(&kwalletrc, "Migration");
131  bool showMigrationWizard = cfg.readEntry<bool>(ENTRY_SHOW_MIGRATION_WIZARD, false);
132 
133  if (showMigrationWizard) {
134  MigrationWizard *wizard = new MigrationWizard(this);
135  int result = wizard->exec();
136  if (QDialog::Accepted == result) {
137  // the user either migrated the wallets, or choose not to be prompted again
138  ok = true;
139  }
140  } else {
141  if (performMigration(0, true)) {
142  ok = true;
143  } else {
144  qDebug() << "Migration failed.";
145  }
146  }
147 
148  return ok;
149 }

161 template <typename R, typename F> R invokeAndCheck(MigrationAgent *migrationAgent, F f, QString errorMsg) {
162  QDBusPendingReply<R> reply = f();
163  reply.waitForFinished();
164  QApplication::processEvents(); // keep UI responsive to show progress messages
165  if (!reply.isValid()) {
166  migrationAgent->emitProgressMessage(errorMsg);
167  throw MigrationException(errorMsg);
168  }
169  return reply.value();
170 }

172 bool MigrationAgent::isEmptyOldWallet() const {
173  QStringList wallets;
174  try {
175  wallets = invokeAndCheck<QStringList>(
176  const_cast<MigrationAgent*>(this), [this] { return _kde4_daemon->wallets(); },
177  i18n("Cannot read old wallet list. Aborting."));
178  } catch (MigrationException ex) {
179  return true;
180  }
181 
182  return wallets.length() == 0;
183 }

185 bool MigrationAgent::performMigration(WId wid, bool withoutWizard)
186 {
187  auto appId = i18n("KDE Wallet Migration Agent");
188  try {
189  QStringList wallets = invokeAndCheck<QStringList>(
190  this, [this] { return _kde4_daemon->wallets(); },
191  i18n("Cannot read old wallet list. Aborting."));
192 
193  foreach (const QString &wallet, wallets) {
194  emit progressMessage(i18n("Migrating wallet: %1", wallet));
195  emit progressMessage(i18n("* Creating KF5 wallet: %1", wallet));
196 
197  int handle5 = -1;
198  if (withoutWizard && (_pam_hash != nullptr)) {
199  // see BUG 351056 for why this hacky code
200  // If the user has several wallets, all the values will be
201  // merged into the single LocalWallet
202  handle5 = _kf5_daemon->pamOpen(KWallet::Wallet::LocalWallet(), _pam_hash, 0);
203  } else {
204  handle5 = _kf5_daemon->internalOpen(appId, wallet, false, 0, true, QString());
205  }
206  if (handle5 <0) {
207  emit progressMessage(i18n("ERROR when attempting new wallet creation. Aborting."));
208  return false;
209  }
210 
211  int handle4 = invokeAndCheck<int>(
212  this, [this, wallet, wid, appId] { return _kde4_daemon->open(wallet, wid, appId); },
213  i18n("Cannot open KDE4 wallet named: %1", wallet));
214  emit progressMessage(i18n("* Opened KDE4 wallet: %1", wallet));
215 
216  const QStringList folders = invokeAndCheck<QStringList>(
217  this, [this, handle4, appId] { return _kde4_daemon->folderList(handle4, appId); },
218  i18n("Cannot retrieve folder list. Aborting."));
219 
220  foreach (const QString &folder, folders) {
221  emit progressMessage(i18n("* Migrating folder %1", folder));
222 
223  QStringList entries = invokeAndCheck<QStringList>(
224  this, [this, handle4, folder, appId] { return _kde4_daemon->entryList(handle4, folder, appId); },
225  i18n("Cannot retrieve folder %1 entries. Aborting.", folder));
226 
227  foreach (const QString &key, entries) {
228 
229  int entryType = invokeAndCheck<int>(
230  this, [this, handle4, folder, key, appId] { return _kde4_daemon->entryType(handle4, folder, key, appId); },
231  i18n("Cannot retrieve key %1 info. Aborting.", key));
232 
233  // handle the case where the entries are already there
234  if (_kf5_daemon->hasEntry(handle5, folder, key, appId)) {
235  emit progressMessage(i18n("* SKIPPING entry %1 in folder %2 as it seems already migrated", key, folder));
236  } else {
237  QByteArray entry = invokeAndCheck<QByteArray>(
238  this, [this, handle4, folder, key, appId] { return _kde4_daemon->readEntry(handle4, folder, key, appId); },
239  i18n("Cannot retrieve key %1 data. Aborting.", key));
240  if ( _kf5_daemon->writeEntry(handle5, folder, key, entry, entryType, appId) != 0 ) {
241  auto msg = i18n("Cannot write entry %1 in the new wallet. Aborting.", key);
242  emit progressMessage(msg);
243  throw MigrationException(msg);
244  }
245  }
246  }
247  }
248 
249  //_kde4_daemon->close(handle4, false, appId);
250  //_kf5_daemon->close(handle5, true, appId);
251  _kf5_daemon->sync(handle5, appId);
252  emit progressMessage(i18n("DONE migrating wallet\n"));
253  }
254  } catch (MigrationException ex) {
255  return false;
256  }
257  return true;
258 }
