58 static char *waitForHash()
59 {
60  printf("kwalletd5: Waiting for hash on %d-\n", pipefd);
61  int totalRead = 0;
62  int readBytes = 0;
63  int attempts = 0;
64  char *buf = (char*)malloc(sizeof(char) * PBKDF2_SHA512_KEYSIZE);
65  memset(buf, '\0', PBKDF2_SHA512_KEYSIZE);
66  while(totalRead != PBKDF2_SHA512_KEYSIZE) {
67  readBytes = read(pipefd, buf + totalRead, PBKDF2_SHA512_KEYSIZE - totalRead);
68  if (readBytes == -1 || attempts > 5) {
69  free(buf);
70  return NULL;
71  }
72  totalRead += readBytes;
73  ++attempts;
74  }
75 
76  close(pipefd);
77  return buf;
78 }

114 char* checkPamModule(int argc, char **argv)
115 {
116  printf("kwalletd5: Checking for pam module\n");
117  char *hash = NULL;
118  int x = 1;
119  for (; x < argc; ++x) {
120  if (strcmp(argv[x], "--pam-login") != 0) {
121  continue;
122  }
123  printf("kwalletd5: Got pam-login param\n");
124  argv[x] = NULL;
125  x++;
126  //We need at least 2 extra arguments after --pam-login
127  if (x + 1 > argc) {
128  printf("kwalletd5: Invalid arguments (less than needed)\n");
129  return NULL;
130  }
131 
132  //first socket for the hash, comes from a pipe
133  pipefd = atoi(argv[x]);
134  argv[x] = NULL;
135  x++;
136  //second socket for environment, comes from a localsocket
137  socketfd = atoi(argv[x]);
138  argv[x] = NULL;
139  break;
140  }
141 
142  if (!pipefd || !socketfd) {
143  printf("Lacking a socket, pipe: %d, env:%d\n", pipefd, socketfd);
144  return NULL;
145  }
146 
147  hash = waitForHash();
148 
149  if (hash == NULL || waitForEnvironment() == -1) {
150  printf("kwalletd5: Hash or environment not received\n");
151  free(hash);
152  return NULL;
153  }
154 
155  return hash;
156 }
