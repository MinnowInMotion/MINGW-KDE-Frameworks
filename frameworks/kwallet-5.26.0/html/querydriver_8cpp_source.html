99 void QueryDriver::readEntries() {
100  theWallet = Wallet::openWallet(walletName, 0);
101  if (entryFolder.isEmpty()) {
102  auto fl = theWallet->folderList();
103  for (auto f: fl) {
104  std::cout << f.toUtf8().constData() << std::endl;
105  Q_ASSERT(theWallet->setFolder(f));
106  auto el = theWallet->entryList();
107  for (auto e: el) {
108  std::cout << "\t" << e.toUtf8().constData() << std::endl;
109  }
110  }
111  } else {
112  if (!theWallet->setFolder(entryFolder)) {
113  std::cout << i18n("The folder %1 does not exist!", entryFolder).toUtf8().constData() << std::endl;
114  exit(4);
115  }
116  auto el = theWallet->entryList();
117  for (auto e: el) {
118  std::cout << e.toUtf8().constData() << std::endl;
119  }
120  }
121  quit();
122 }

124 void QueryDriver::readValue() {
125  if (verbose) qDebug() << "reading" << entryName << "from" << entryFolder << "from" << walletName;
126  if (!theWallet->setFolder(entryFolder)) {
127  std::cout << i18n("The folder %1 does not exist!", entryFolder).toUtf8().constData() << std::endl;
128  exit(4);
129  }
130  Wallet::EntryType kind = theWallet->entryType(entryName);
131  if (kind == Wallet::Password) {
132  readPasswordValue();
133  } else if (kind == Wallet::Map) {
134  readMapValue();
135  } else {
136  std::cout << i18n("Failed to read entry %1 value from the %2 wallet.", entryName, walletName).toUtf8().constData() << std::endl;
137  exit(4);
138  }
139  quit();
140 }

169 void QueryDriver::writeValue() {
170  if (verbose) qDebug() << "writing" << entryName << "to" << entryFolder << "to" << walletName;
171  theWallet->setFolder(entryFolder);
172 
173  QString passwordContents;
174  for (std::string line; std::getline(std::cin, line); ) {
175  if (!passwordContents.isEmpty()) passwordContents += '\n';
176  passwordContents += QString::fromStdString(line);
177  if (!std::cin) break;
178  }
179  Wallet::EntryType kind = theWallet->entryType(entryName);
180  if (kind == Wallet::Password) {
181  if (verbose) qDebug() << "about to write" << passwordContents;
182  int rc = theWallet->writePassword(entryName, passwordContents);
183  if (rc != 0) {
184  std::cout << i18n("Failed to write entry %1 value to %2 wallet", entryName, walletName).toUtf8().constData() << std::endl;
185  exit(4);
186  }
187  } else if (kind == Wallet::Map) {
188  const QJsonDocument json = QJsonDocument::fromJson(passwordContents.toLatin1());
189  if (!json.isNull()) {
190  QJsonObject values = json.object();
191  QMap<QString, QString> map;
192  for (auto e : values.keys()) {
193  map.insert(e, values.value(e).toString());
194  }
195  if (verbose) qDebug() << "about to write" << map;
196  int rc = theWallet->writeMap(entryName, map);
197  if (rc != 0) {
198  std::cout << i18n("Failed to write entry %1 value to %2 wallet", entryName, walletName).toUtf8().constData() << std::endl;
199  exit(4);
200  }
201  }
202  }
203  quit();
204 }
