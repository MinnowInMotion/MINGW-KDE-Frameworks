146 class PageGpgKey : public QWizardPage
147 {
148 public:
149  PageGpgKey(QWidget *parent)
150  : QWizardPage(parent)
151  , userHasGpgKeys(false)
152  {
153  ui.setupUi(this);
154 
155  registerField(QStringLiteral("gpgKey"), ui._gpgKey);
156 
157  KeysVector keys;
158  GpgME::initializeLibrary();
159  GpgME::Error err = GpgME::checkEngine(GpgME::OpenPGP);
160  if (err) {
161  qDebug() << "OpenPGP not supported on your system!";
162  KMessageBox::error(this, i18n("The QGpgME library failed to initialize for the OpenPGP protocol. Please check your system's configuration then try again."));
163  } else {
164  boost::shared_ptr< GpgME::Context > ctx(GpgME::Context::createForProtocol(GpgME::OpenPGP));
165  if (0 == ctx) {
166  KMessageBox::error(this, i18n("The QGpgME library failed to initialize for the OpenPGP protocol. Please check your system's configuration then try again."));
167  } else {
168 
169  ctx->setKeyListMode(GPGME_KEYLIST_MODE_LOCAL);
170  err = ctx->startKeyListing();
171  while (!err) {
172  GpgME::Key k = ctx->nextKey(err);
173  if (err) {
174  break;
175  }
176  if (!k.isInvalid() && k.canEncrypt() && (k.ownerTrust() == GpgME::Key::Ultimate)) {
177  keys.push_back(k);
178  }
179  }
180  ctx->endKeyListing();
181  }
182  }
183  std::for_each(keys.begin(), keys.end(), AddKeyToCombo(ui._gpgKey));
184 
185  userHasGpgKeys = keys.size() > 0;
186  if (userHasGpgKeys) {
187  ui.stackedWidget->setCurrentWidget(ui._pageWhenHasKeys);
188  } else {
189  ui.stackedWidget->setCurrentWidget(ui._pageNoKeys);
190  setFinalPage(true);
191  }
192  emit completeChanged();
193  }
194 
195  virtual int nextId() const
196  {
197  return static_cast<KWalletWizard *>(wizard())->wizardType() == KWalletWizard::Basic ? -1 : KWalletWizard::PageOptionsId;
198  }
199 
200  virtual bool isComplete() const
201  {
202  return userHasGpgKeys;
203  }

270 void KWalletWizard::passwordPageUpdate()
271 {
272  bool complete = true;
273  if (field(QStringLiteral("useWallet")).toBool()) {
274 #ifdef HAVE_QGPGME
275  if (field(QStringLiteral("useBlowfish")).toBool()) {
276  m_pagePasswd->setFinalPage(wizardType() == Basic);
277  button(NextButton)->setVisible(wizardType() != Basic);
278 #endif
279  if (field(QStringLiteral("pass1")).toString() == field(QStringLiteral("pass2")).toString()) {
280  if (field(QStringLiteral("pass1")).toString().isEmpty()) {
281  m_pagePasswd->setMatchLabelText(i18n("<qt>Password is empty. <b>(WARNING: Insecure)</b></qt>"));
282  } else {
283  m_pagePasswd->setMatchLabelText(i18n("Passwords match."));
284  }
285  } else {
286  m_pagePasswd->setMatchLabelText(i18n("Passwords do not match."));
287  complete = false;
288  }
289 #ifdef HAVE_QGPGME
290  } else {
291  m_pagePasswd->setFinalPage(false);
292  button(NextButton)->setEnabled(true);
293  return;
294  }
295 #endif
296  } else {
297  m_pagePasswd->setMatchLabelText(QString());
298  }
299  button(wizardType() == Basic ? FinishButton : NextButton)->setEnabled(complete);
300 }
