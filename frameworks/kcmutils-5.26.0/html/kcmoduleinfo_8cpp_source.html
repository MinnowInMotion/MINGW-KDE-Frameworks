117 void KCModuleInfo::Private::loadAll()
118 {
119  allLoaded = true;
120 
121  if (!service) { /* We have a bogus service. All get functions will return empty/zero values */
122  return;
123  }
124 
125  // get the documentation path
126  doc = service->property(QStringLiteral("X-DocPath"), QVariant::String).toString();
127  if (doc.isEmpty()) {
128  doc = service->property(QStringLiteral("DocPath"), QVariant::String).toString();
129  }
130 
131  // read weight
132  QVariant tmp = service->property(QStringLiteral("X-KDE-Weight"), QVariant::Int);
133  weight = tmp.isValid() ? tmp.toInt() : 100;
134 
135  // factory handle
136  tmp = service->property(QStringLiteral("X-KDE-FactoryName"), QVariant::String);
137  handle = tmp.isValid() ? tmp.toString() : lib;
138 
139 }
