153 void KPluginSelector::Private::DependenciesWidget::addDependency(const QString &dependency, const QString &pluginCausant, bool added)
154 {
155  if (!isVisible()) {
156  setVisible(true);
157  }
158 
159  struct FurtherInfo furtherInfo;
160  furtherInfo.added = added;
161  furtherInfo.pluginCausant = pluginCausant;
162 
163  if (dependencyMap.contains(dependency)) { // The dependency moved from added to removed or vice-versa
164  if (added && removedByDependencies) {
165  removedByDependencies--;
166  } else if (addedByDependencies) {
167  addedByDependencies--;
168  }
169 
170  dependencyMap[dependency] = furtherInfo;
171  } else {
172  dependencyMap.insert(dependency, furtherInfo);
173  }
174 
175  if (added) {
176  addedByDependencies++;
177  } else {
178  removedByDependencies++;
179  }
180 
181  updateDetails();
182 }

429 void KPluginSelector::Private::PluginModel::addPlugins(const QList<KPluginInfo> &pluginList, const QString &categoryName, const QString &categoryKey, const KConfigGroup &cfgGroup, PluginLoadMethod pluginLoadMethod, bool manuallyAdded)
430 {
431  QList<PluginEntry> listToAdd;
432 
433  foreach (const KPluginInfo &pluginInfo, pluginList) {
434  PluginEntry pluginEntry;
435  pluginEntry.category = categoryName;
436  pluginEntry.pluginInfo = pluginInfo;
437  if (pluginLoadMethod == ReadConfigFile) {
438  pluginEntry.pluginInfo.load(cfgGroup);
439  }
440  pluginEntry.checked = pluginInfo.isPluginEnabled();
441  pluginEntry.manuallyAdded = manuallyAdded;
442  if (cfgGroup.isValid()) {
443  pluginEntry.cfgGroup = cfgGroup;
444  } else {
445  pluginEntry.cfgGroup = pluginInfo.config();
446  }
447 
448  // this is where kiosk will set if a plugin is checkable or not (pluginName + "Enabled")
449  pluginEntry.isCheckable = !pluginInfo.isValid() || !pluginEntry.cfgGroup.isEntryImmutable(pluginInfo.pluginName() + QLatin1String("Enabled"));
450 
451  if (!pluginEntryList.contains(pluginEntry) && !listToAdd.contains(pluginEntry) &&
452  (categoryKey.isEmpty() || !pluginInfo.category().compare(categoryKey, Qt::CaseInsensitive)) &&
453  (!pluginInfo.service() || !pluginInfo.service()->noDisplay())) {
454  listToAdd << pluginEntry;
455 
456  if (!pluginSelector_d->showIcons && !pluginInfo.icon().isEmpty()) {
457  pluginSelector_d->showIcons = true;
458  }
459  }
460  }
461 
462  if (listToAdd.count()) {
463  beginInsertRows(QModelIndex(), pluginEntryList.count(), pluginEntryList.count() + listToAdd.count() - 1);
464  pluginEntryList << listToAdd;
465  endInsertRows();
466  }
467 }

482 QVariant KPluginSelector::Private::PluginModel::data(const QModelIndex &index, int role) const
483 {
484  if (!index.isValid() || !index.internalPointer()) {
485  return QVariant();
486  }
487 
488  PluginEntry *pluginEntry = static_cast<PluginEntry *>(index.internalPointer());
489 
490  switch (role) {
491  case Qt::DisplayRole:
492  return pluginEntry->pluginInfo.name();
493  case PluginEntryRole:
494  return QVariant::fromValue(pluginEntry);
495  case ServicesCountRole:
496  return pluginEntry->pluginInfo.kcmServices().count();
497  case NameRole:
498  return pluginEntry->pluginInfo.name();
499  case CommentRole:
500  return pluginEntry->pluginInfo.comment();
501  case AuthorRole:
502  return pluginEntry->pluginInfo.author();
503  case EmailRole:
504  return pluginEntry->pluginInfo.email();
505  case WebsiteRole:
506  return pluginEntry->pluginInfo.website();
507  case VersionRole:
508  return pluginEntry->pluginInfo.version();
509  case LicenseRole:
510  return pluginEntry->pluginInfo.license();
511  case DependenciesRole:
512  return pluginEntry->pluginInfo.dependencies();
513  case IsCheckableRole:
514  return pluginEntry->isCheckable;
515  case Qt::DecorationRole:
516  return pluginEntry->pluginInfo.icon();
517  case Qt::CheckStateRole:
518  return pluginEntry->checked;
519  case KCategorizedSortFilterProxyModel::CategoryDisplayRole: // fall through
520  case KCategorizedSortFilterProxyModel::CategorySortRole:
521  return pluginEntry->category;
522  default:
523  return QVariant();
524  }

797 void KPluginSelector::Private::PluginDelegate::slotConfigureClicked()
798 {
799  const QModelIndex index = focusedIndex();
800  const QAbstractItemModel *model = index.model();
801 
802  PluginEntry *pluginEntry = model->data(index, PluginEntryRole).value<PluginEntry *>();
803  KPluginInfo pluginInfo = pluginEntry->pluginInfo;
804 
805  QDialog configDialog(itemView());
806  configDialog.setWindowTitle(model->data(index, NameRole).toString());
807  // The number of KCModuleProxies in use determines whether to use a tabwidget
808  QTabWidget *newTabWidget = 0;
809  // Widget to use for the setting dialog's main widget,
810  // either a QTabWidget or a KCModuleProxy
811  QWidget *mainWidget = 0;
812  // Widget to use as the KCModuleProxy's parent.
813  // The first proxy is owned by the dialog itself
814  QWidget *moduleProxyParentWidget = &configDialog;
815 
816  foreach (const KService::Ptr &servicePtr, pluginInfo.kcmServices()) {
817  if (!servicePtr->noDisplay()) {
818  KCModuleInfo moduleInfo(servicePtr);
819  KCModuleProxy *currentModuleProxy = new KCModuleProxy(moduleInfo, moduleProxyParentWidget, pluginSelector_d->kcmArguments);
820  if (currentModuleProxy->realModule()) {
821  moduleProxyList << currentModuleProxy;
822  if (mainWidget && !newTabWidget) {
823  // we already created one KCModuleProxy, so we need a tab widget.
824  // Move the first proxy into the tab widget and ensure this and subsequent
825  // proxies are in the tab widget
826  newTabWidget = new QTabWidget(&configDialog);
827  moduleProxyParentWidget = newTabWidget;
828  mainWidget->setParent(newTabWidget);
829  KCModuleProxy *moduleProxy = qobject_cast<KCModuleProxy *>(mainWidget);
830  if (moduleProxy) {
831  newTabWidget->addTab(mainWidget, moduleProxy->moduleInfo().moduleName());
832  mainWidget = newTabWidget;
833  } else {
834  delete newTabWidget;
835  newTabWidget = 0;
836  moduleProxyParentWidget = &configDialog;
837  mainWidget->setParent(0);
838  }
839  }
840 
841  if (newTabWidget) {
842  newTabWidget->addTab(currentModuleProxy, servicePtr->name());
843  } else {
844  mainWidget = currentModuleProxy;
845  }
846  } else {
847  delete currentModuleProxy;
848  }
849  }
850  }
851 
852  // it could happen that we had services to show, but none of them were real modules.
853  if (moduleProxyList.count()) {
854  QVBoxLayout *layout = new QVBoxLayout;
855  layout->addWidget(mainWidget);
856  const int marginHint = configDialog.style()->pixelMetric(QStyle::PM_DefaultChildMargin);
857  layout->insertSpacing(-1, marginHint);
858 
859  QDialogButtonBox *buttonBox = new QDialogButtonBox(&configDialog);
860  buttonBox->setStandardButtons(QDialogButtonBox::Ok | QDialogButtonBox::Cancel | QDialogButtonBox::RestoreDefaults);
861  KGuiItem::assign(buttonBox->button(QDialogButtonBox::Ok), KStandardGuiItem::ok());
862  KGuiItem::assign(buttonBox->button(QDialogButtonBox::Cancel), KStandardGuiItem::cancel());
863  KGuiItem::assign(buttonBox->button(QDialogButtonBox::RestoreDefaults), KStandardGuiItem::defaults());
864  connect(buttonBox, &QDialogButtonBox::accepted, &configDialog, &QDialog::accept);
865  connect(buttonBox, &QDialogButtonBox::rejected, &configDialog, &QDialog::reject);
866  connect(buttonBox->button(QDialogButtonBox::RestoreDefaults), &QAbstractButton::clicked, this, &PluginDelegate::slotDefaultClicked);
867  layout->addWidget(buttonBox);
868 
869  configDialog.setLayout(layout);
870 
871  if (configDialog.exec() == QDialog::Accepted) {
872  foreach (KCModuleProxy *moduleProxy, moduleProxyList) {
873  QStringList parentComponents = moduleProxy->moduleInfo().service()->property(QStringLiteral("X-KDE-ParentComponents")).toStringList();
874  moduleProxy->save();
875  foreach (const QString &parentComponent, parentComponents) {
876  emit configCommitted(parentComponent.toLatin1());
877  }
878  }
879  } else {
880  foreach (KCModuleProxy *moduleProxy, moduleProxyList) {
881  moduleProxy->load();
882  }
883  }
884 
885  qDeleteAll(moduleProxyList);
886  moduleProxyList.clear();
887  }
888 }
