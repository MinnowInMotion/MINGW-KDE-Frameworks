69 KCModule *KCModuleLoader::loadModule(const KCModuleInfo &mod, ErrorReporting report, QWidget *parent, const QStringList &args)
70 {
71  /*
72  * Simple libraries as modules are the easiest case:
73  * We just have to load the library and get the module
74  * from the factory.
75  */
76 
77  if (!mod.service())
78  return reportError(report,
79  i18n("The module %1 could not be found.",
80  mod.moduleName()), i18n("<qt><p>The diagnosis is:<br />The desktop file %1 could not be found.</p></qt>", mod.fileName()), parent);
81  if (mod.service()->noDisplay())
82  return reportError(report, i18n("The module %1 is disabled.", mod.moduleName()),
83  i18n("<qt><p>Either the hardware/software the module configures is not available or the module has been disabled by the administrator.</p></qt>"),
84  parent);
85 
86  if (!mod.library().isEmpty()) {
87  QString error;
88  QVariantList args2;
89  foreach (const QString &arg, args) {
90  args2 << arg;
91  }
92  KCModule *module = mod.service()->createInstance<KCModule>(parent, args2, &error);
93  if (module) {
94  return module;
95  } else {
96  KPluginLoader loader(KPluginLoader::findPlugin(QLatin1String("kcms/") + mod.service()->library()));
97  KPluginFactory* factory = loader.factory();
98  if (!factory) {
99  qWarning() << "Error loading plugin:" << loader.errorString();
100  } else {
101  KQuickAddons::ConfigModule *cm = factory->create<KQuickAddons::ConfigModule >();
102  if (!cm) {
103  qWarning() << "Error creating object from plugin" << loader.fileName();
104  }
105  module = new KCModuleQml(cm, parent, args2);
106  return module;
107  }
108  }
109 //#ifndef NDEBUG
110  {
111  // get the create_ function
112  QLibrary lib(KPluginLoader::findPlugin(mod.library()));
113  if (lib.load()) {
114  KCModule *(*create)(QWidget *, const char *);
115  QByteArray factorymethod("create_");
116  factorymethod += mod.handle().toLatin1();
117  create = reinterpret_cast<KCModule *(*)(QWidget *, const char *)>(lib.resolve(factorymethod.constData()));
118  if (create) {
119  return create(parent, mod.handle().toLatin1().constData());
120  qFatal("This module still uses a custom factory method (%s). This is not supported anymore. Please fix the module.", factorymethod.constData());
121  } else {
122  qWarning() << "This module has no valid entry symbol at all. The reason could be that it's still using K_EXPORT_COMPONENT_FACTORY with a custom X-KDE-FactoryName which is not supported anymore";
123  }
124  lib.unload();
125  }
126  }
127 //#endif // NDEBUG
128  return reportError(report, error, QString(), parent);
129  }
130 
131  /*
132  * Ok, we could not load the library.
133  * Try to run it as an executable.
134  * This must not be done when calling from kcmshell, or you'll
135  * have infinite recursion
136  * (startService calls kcmshell which calls modloader which calls startService...)
137  *
138  */
139  return reportError(report,
140  i18n("The module %1 is not a valid configuration module.", mod.moduleName()),
141  i18n("<qt>The diagnosis is:<br />The desktop file %1 does not specify a library.</qt>", mod.fileName()), parent);
142 }

151 KCModule *KCModuleLoader::reportError(ErrorReporting report, const QString &text,
152  const QString &details, QWidget *parent)
153 {
154  QString realDetails = details;
155  if (realDetails.isNull()) {
156  realDetails = i18n("<qt><p>Possible reasons:<ul><li>An error occurred during your last "
157  "KDE upgrade leaving an orphaned control module</li><li>You have old third party "
158  "modules lying around.</li></ul></p><p>Check these points carefully and try to remove "
159  "the module mentioned in the error message. If this fails, consider contacting "
160  "your distributor or packager.</p></qt>");
161  }
162  if (report & KCModuleLoader::Dialog) {
163  KMessageBox::detailedError(parent, text, realDetails);
164  }
165  if (report & KCModuleLoader::Inline) {
166  return new KCMError(text, realDetails, parent);
167  }
168  return 0;
169 }
