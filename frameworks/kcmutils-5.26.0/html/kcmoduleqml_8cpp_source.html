55 
56  connect(configModule, &KQuickAddons::ConfigModule::quickHelpChanged,

58  //HACK:Here is important those two enums keep having the exact same values
59  //but the kdeclarative one can't use the KCModule's enum
60  setButtons((KCModule::Buttons)(int)d->configModule->buttons());
61  connect(configModule, &KQuickAddons::ConfigModule::buttonsChanged, [=] {
62  setButtons((KCModule::Buttons)(int)d->configModule->buttons());
63  });
64 
65  if (d->configModule->needsSave()) {
66  emit changed(true);
67  }
68  connect(configModule, &KQuickAddons::ConfigModule::needsSaveChanged, [=] {
69  emit changed(d->configModule->needsSave());
70  });
71 
72  setNeedsAuthorization(d->configModule->needsAuthorization());
73  connect(configModule, &KQuickAddons::ConfigModule::needsAuthorizationChanged, [=] {
74  setNeedsAuthorization(d->configModule->needsAuthorization());
75  });
76 
77  setRootOnlyMessage(d->configModule->rootOnlyMessage());
78  setUseRootOnlyMessage(d->configModule->useRootOnlyMessage());
79  connect(configModule, &KQuickAddons::ConfigModule::rootOnlyMessageChanged, [=] {
80  setRootOnlyMessage(d->configModule->rootOnlyMessage());

82  connect(configModule, &KQuickAddons::ConfigModule::useRootOnlyMessageChanged, [=] {
83  setUseRootOnlyMessage(d->configModule->useRootOnlyMessage());
84  });
85

100 void KCModuleQml::showEvent(QShowEvent *event)
101 {
102  if (d->quickWidget || !d->configModule->mainUi()) {
103  KCModule::showEvent(event);
104  return;
105  }
106 
107  QVBoxLayout* layout = new QVBoxLayout(this);
108 
109  d->quickWidget = new QQuickWidget(d->configModule->engine(), this);
110  d->quickWidget->setResizeMode(QQuickWidget::SizeRootObjectToView);
111  d->quickWidget->setClearColor(QGuiApplication::palette().window().color());
112  connect(qApp, &QGuiApplication::paletteChanged, [=]() {
113  d->quickWidget->setClearColor(QGuiApplication::palette().window().color());
114  });
115 
116  QQmlComponent *component = new QQmlComponent(d->configModule->engine(), this);
117  component->setData(QByteArrayLiteral("import QtQuick 2.3\nItem{}"), QUrl());
118  QObject *root = component->create();
119  d->quickWidget->setContent(QUrl(), component, root);
120 
121  d->configModule->mainUi()->setParentItem(d->quickWidget->rootObject());
122 
123  //set anchors
124  QQmlExpression expr(d->configModule->engine()->rootContext(), d->configModule->mainUi(), QStringLiteral("parent"));
125  QQmlProperty prop(d->configModule->mainUi(), QStringLiteral("anchors.fill"));
126  prop.write(expr.evaluate());
127 
128  layout->addWidget(d->quickWidget);
129  KCModule::showEvent(event);
130 }
