62 void reparseConfiguration(const QString &componentName)
63 {
64  // qDebug() << componentName;
65  // check if the componentName is valid:
66  if (! d()->m_componentInfo.contains(componentName)) {
67  return;
68  }
69  // first we reparse the config so that the KConfig object will be up to date
70  KSharedConfig::Ptr config = configForComponentName(componentName);
71  config->reparseConfiguration();
72 
73  foreach (const ComponentInfo::Slot &slot, d()->m_componentInfo[componentName].slotList) {
74  QMetaObject::invokeMethod(slot.first, slot.second);
75  }
76 }

78 void syncConfiguration()
79 {
80  for (QMap<QString, ComponentInfo>::ConstIterator it = d()->m_componentInfo.constBegin(); it != d()->m_componentInfo.constEnd(); ++it) {
81  KSharedConfig::Ptr config = configForComponentName(it.key());
82  config->sync();
83  }

86 void DispatcherPrivate::unregisterComponent(QObject *obj)
87 {
88  if (!m_componentName.contains(obj)) {
89  qWarning() << Q_FUNC_INFO << "Tried to unregister an object which is not already registered.";
90  return;
91  }
92 
93  QString name = m_componentName[obj];
94  m_componentName.remove(obj); //obj will be destroyed when we return, so we better remove this entry
95  --(m_componentInfo[name].count);
96  // qDebug() << "componentName=" << name << "refcount=" << m_componentInfo[name].count;
97  Q_ASSERT(m_componentInfo[name].count >= 0);
98  if (m_componentInfo[name].count == 0) {
99  m_componentInfo.remove(name);
100  }
101 }
