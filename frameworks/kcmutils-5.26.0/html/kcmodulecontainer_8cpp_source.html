80 void KCModuleContainer::init()
81 {
82  d->topLayout = new QVBoxLayout(this);
83  d->topLayout->setMargin(0);
84  d->topLayout->setObjectName(QStringLiteral("topLayout"));
85  d->tabWidget = new QTabWidget(this);
86  d->tabWidget->setObjectName(QStringLiteral("tabWidget"));
87  connect(d->tabWidget, &QTabWidget::currentChanged, this, &KCModuleContainer::tabSwitched);
88  d->topLayout->addWidget(d->tabWidget);
89 
90  if (!d->modules.isEmpty()) {
91  /* Add our modules */
92  for (QStringList::const_iterator it = d->modules.constBegin(); it != d->modules.constEnd(); ++it) {
93  addModule((*it));
94  }
95  }
96 }

98 void KCModuleContainer::addModule(const QString &module)
99 {
100  /* In case it doesn't exist we just silently drop it.
101  * This allows people to easily extend containers.
102  * For example, KCM monitor gamma can be in kdegraphics.
103  */
104  KService::Ptr service = KService::serviceByDesktopName(module);
105  if (!service) {
106  // qDebug() << "KCModuleContainer: module '" <<
107  // module << "' was not found and thus not loaded" << endl;
108  return;
109  }
110 
111  if (service->noDisplay()) {
112  return;
113  }
114 
115  KCModuleProxy *proxy = new KCModuleProxy(service, d->tabWidget);
116  d->allModules.append(proxy);
117 
118  proxy->setObjectName(module);
119 
120  d->tabWidget->addTab(proxy, QIcon::fromTheme(proxy->moduleInfo().icon()),
121  /* Qt eats ampersands for dinner. But not this time. */
122  proxy->moduleInfo().moduleName().replace(QLatin1Char('&'), QStringLiteral("&&")));
123 
124  d->tabWidget->setTabToolTip(d->tabWidget->indexOf(proxy), proxy->moduleInfo().comment());
125 
126  connect(proxy, SIGNAL(changed(KCModuleProxy*)), SLOT(moduleChanged(KCModuleProxy*)));
127 
128  /* Collect our buttons - we go for the common deliminator */
129  setButtons(buttons() | proxy->realModule()->buttons());
130 }
