126 void ComponentsDialog::show()
127 {
128  // clear the treelist
129  d->listview->clear();
130  d->plugininfomap.clear();
131 
132  // construct the treelist
133  for (QList<KPluginInfo *>::ConstIterator it = d->plugininfolist.constBegin();
134  it != d->plugininfolist.constEnd(); ++it) {
135  (*it)->load();
136  QTreeWidgetItem *item = new QTreeWidgetItem(d->listview, QStringList((*it)->name()));
137  if (!(*it)->icon().isEmpty()) {
138  item->setIcon(0, SmallIcon((*it)->icon(), IconSize(KIconLoader::Small)));
139  }
140  item->setCheckState(0, (*it)->isPluginEnabled() ? Qt::Checked : Qt::Unchecked);
141  d->plugininfomap[ item ] = (*it);
142  }
143  QDialog::show();
144 }
