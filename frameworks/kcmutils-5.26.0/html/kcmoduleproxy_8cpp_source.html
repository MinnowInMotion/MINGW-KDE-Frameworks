71 KCModule *KCModuleProxy::realModule() const
72 {
73  Q_D(const KCModuleProxy);
74  /*
75  * Note, don't call any function that calls realModule() since
76  * that leads to an infinite loop.
77  */
78 
79  /* Already loaded */
80  if (!d->kcm) {
81  QApplication::setOverrideCursor(Qt::WaitCursor);
82  const_cast<KCModuleProxyPrivate *>(d)->loadModule();
83  QApplication::restoreOverrideCursor();
84  }
85  return d->kcm;
86 }

88 void KCModuleProxyPrivate::loadModule()
89 {
90  if (!topLayout) {
91  topLayout = new QVBoxLayout(parent);
92  topLayout->setMargin(0);
93 
94  QString name = modInfo.handle();
95  name.replace(QLatin1Char('-'), QLatin1Char('_')); // hyphen is not allowed in dbus, only [A-Z][a-z][0-9]_
96  name.replace(QLatin1Char('/'), QLatin1Char('_')); // same goes for '/'
97  dbusService = QLatin1String("org.kde.internal.KSettingsWidget_") + name;
98 
99  // for dbus path, we also need to convert '.' characters
100  name.replace(QLatin1Char('.'), QLatin1Char('_'));
101  dbusPath = QLatin1String("/internal/KSettingsWidget/") + name;
102  }
103 
104  if (QDBusConnection::sessionBus().registerService(dbusService) || bogusOccupier) {
105  /* We got the name we requested, because no one was before us,
106  * or, it was an random application which had picked that name */
107  // qDebug() << "Module not already loaded, loading module " << modInfo.moduleName() << " from library " << modInfo.library() << " using symbol " << modInfo.handle();
108 
109  kcm = KCModuleLoader::loadModule(modInfo, KCModuleLoader::Inline, parent, args);
110 
111  QObject::connect(kcm, SIGNAL(changed(bool)), parent, SLOT(_k_moduleChanged(bool)));
112  QObject::connect(kcm, SIGNAL(destroyed()), parent, SLOT(_k_moduleDestroyed()));
113  QObject::connect(kcm, &KCModule::quickHelpChanged, parent, &KCModuleProxy::quickHelpChanged);
114  parent->setWhatsThis(kcm->quickHelp());
115 
116  if (kcm->layout()) {
117  kcm->layout()->setMargin(0);
118  }
119  topLayout->addWidget(kcm);
120  if (!modInfo.handle().isEmpty()) {
121  QDBusConnection::sessionBus().registerObject(dbusPath, new KSettingsWidgetAdaptor(parent), QDBusConnection::ExportAllSlots);
122  }
123 
124  if (!rootInfo && /* If it's not already done */
125  kcm->useRootOnlyMessage() && /* kcm wants root message */
126  !KUser().isSuperUser()) { /* Not necessary if we're root */
127  /*rootInfo = new QLabel( parent );
128  topLayout->insertWidget( 0, rootInfo );
129 
130  QPalette palette = rootInfo->palette();
131  KStatefulBrush stbrush(KColorScheme::Window, KColorScheme::NeutralBackground);
132  qDebug() << stbrush.brush(rootInfo);
133  palette.setBrush(QPalette::Window, stbrush.brush(rootInfo));
134  rootInfo->setPalette(palette);
135  rootInfo->setAutoFillBackground(true);
136 
137  const QString message = kcm->rootOnlyMessage();
138  if( message.isEmpty() )
139  rootInfo->setText( i18n(
140  "<b>Changes in this section require root access.</b><br />"
141  "On applying your changes you will have to supply your root "
142  "password." ) );
143  else
144  rootInfo->setText(message);
145 
146  rootInfo->setWhatsThis( i18n(
147  "This section requires special permissions, probably "
148  "for system-wide changes; therefore, it is "
149  "required that you provide the root password to be "
150  "able to change the module's properties. If "
151  "you cannot provide the password, the changes of the "
152  "module cannot be saved " ) );*/
153  }
154  } else {
155  // qDebug() << "Module already loaded, loading KCMError";
156 
157  /* Figure out the name of where the module is already loaded */
158  QDBusInterface proxy(dbusService, dbusPath, QStringLiteral("org.kde.internal.KSettingsWidget"));
159  QDBusReply<QString> reply = proxy.call(QStringLiteral("applicationName"));
160 
161  if (reply.isValid()) {
162  QObject::connect(QDBusConnection::sessionBus().interface(), SIGNAL(serviceOwnerChanged(QString,QString,QString)),
163  parent, SLOT(_k_ownerChanged(QString,QString,QString)));
164  kcm = KCModuleLoader::reportError(KCModuleLoader::Inline,
165  i18nc("Argument is application name", "This configuration section is "
166  "already opened in %1", reply.value()), QStringLiteral(" "), parent);
167  topLayout->addWidget(kcm);
168  } else {
169  // qDebug() << "Calling KCModuleProxy's DBus interface for fetching the name failed.";
170  bogusOccupier = true;
171  loadModule();
172  }
173  }
174 }

176 void KCModuleProxyPrivate::_k_ownerChanged(const QString &service, const QString &oldOwner, const QString &)
177 {
178  if (service == dbusService && !oldOwner.isEmpty()) {
179  // Violence: Get rid of KCMError & CO, so that
180  // realModule() attempts to reload the module
181  delete kcm;
182  kcm = 0;
183  Q_Q(KCModuleProxy);
184  q->realModule();
185 
186  Q_ASSERT(kcm);
187  kcm->show();
188  }
189 }
