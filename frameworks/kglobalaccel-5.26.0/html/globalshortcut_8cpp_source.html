71 GlobalShortcut::operator KGlobalShortcutInfo () const
72  {
73  KGlobalShortcutInfo info;
74  info.d->uniqueName = _uniqueName;
75  info.d->friendlyName = _friendlyName;
76  info.d->contextUniqueName = context()->uniqueName();
77  info.d->contextFriendlyName = context()->friendlyName();
78  info.d->componentUniqueName = context()->component()->uniqueName();
79  info.d->componentFriendlyName = context()->component()->friendlyName();
80  Q_FOREACH (int key, _keys)

82  info.d->keys.append(QKeySequence(key));
83  }
84  Q_FOREACH (int key, _defaultKeys)
85  {
86  info.d->defaultKeys.append(QKeySequence(key));
87  }
88  return info;

174 void GlobalShortcut::setKeys(const QList<int> newKeys)
175  {
176  bool active = _isRegistered;
177  if (active)
178  {
179  setInactive();
180  }
181 
182  _keys = QList<int>();
183 
184  Q_FOREACH(int key, newKeys)
185  {
186  if (key!=0 && !GlobalShortcutsRegistry::self()->getShortcutByKey(key))
187  {
188  _keys.append(key);
189  }
190  else
191  {
192  qCDebug(KGLOBALACCELD) << _uniqueName << "skipping because key" << QKeySequence(key).toString() << "is already taken";
193  _keys.append(0);
194  }
195  }
196 
197  if (active)
198  {
199  setActive();
200  }
201  }

216 void GlobalShortcut::setActive()
217  {
218  if (!_isPresent || _isRegistered)
219  {
220  // The corresponding application is not present or the keys are
221  // already grabbed
222  return;
223  }
224 
225  Q_FOREACH( int key, _keys)
226  {
227  if (key != 0 && !GlobalShortcutsRegistry::self()->registerKey(key, this))
228  {
229  qCDebug(KGLOBALACCELD) << uniqueName() << ": Failed to register " << QKeySequence(key).toString();
230  }
231  }
232 
233  _isRegistered = true;
234  }
235 
236 
237 void GlobalShortcut::setInactive()
238  {
239  if (!_isRegistered)
240  {
241  // The keys are not grabbed currently
242  return;
243  }
244 
245  Q_FOREACH( int key, _keys)
246  {
247  if (key != 0 && !GlobalShortcutsRegistry::self()->unregisterKey(key, this))
248  {
249  qCDebug(KGLOBALACCELD) << uniqueName() << ": Failed to unregister " << QKeySequence(key).toString();
250  }
251  }
252 
253  _isRegistered = false;
254  }
