36 static QList<int> keysFromString(const QString &str)
37 {
38  QList<int> ret;
39  if (str == QLatin1String("none")) {
40  return ret;
41  }
42  const QStringList strList = str.split('\t');
43  foreach (const QString &s, strList) {
44  int key = QKeySequence(s)[0];
45  if (key != -1) { //sanity check just in case
46  ret.append(key);
47  }
48  }
49  return ret;
50 }

53 static QString stringFromKeys(const QList<int> &keys)
54 {
55  if (keys.isEmpty()) {
56  return QStringLiteral("none");
57  }
58  QString ret;
59  foreach (int key, keys) {
60  ret.append(QKeySequence(key).toString());
61  ret.append('\t');
62  }
63  ret.chop(1);
64  return ret;
65 }

208  QString dbusPath = _uniqueName;

236 void Component::emitGlobalShortcutPressed( const GlobalShortcut &shortcut )
237  {
238 #if HAVE_X11
239  // pass X11 timestamp
240  long timestamp = QX11Info::appTime();
241  // Make sure kglobalacceld has ungrabbed the keyboard after receiving the
242  // keypress, otherwise actions in application that try to grab the
243  // keyboard (e.g. in kwin) may fail to do so. There is still a small race
244  // condition with this being out-of-process.
245  if (_registry->_manager) {
246  _registry->_manager->syncWindowingSystem();
247  }
248 #else
249  long timestamp = 0;
250 #endif
251 
252  // Make sure it is one of ours
253  if (shortcut.context()->component() != this)
254  {
255  Q_ASSERT(shortcut.context()->component() == this);
256  // In production mode do nothing
257  return;
258  }
259 
260  emit globalShortcutPressed(
261  shortcut.context()->component()->uniqueName(),
262  shortcut.uniqueName(),
263  timestamp);
264  }

327 bool Component::isShortcutAvailable(
328  int key,
329  const QString &component,
330  const QString &context) const
331  {
332  qCDebug(KGLOBALACCELD) << QKeySequence(key).toString() << component;
333 
334  // if this component asks for the key. only check the keys in the same
335  // context
336  if (component==uniqueName())
337  {
338  Q_FOREACH(GlobalShortcut *sc, shortcutContext(context)->_actions)
339  {
340  if (sc->keys().contains(key)) return false;
341  }
342  }
343  else
344  {
345  Q_FOREACH(GlobalShortcutContext *ctx, _contexts)
346  {
347  Q_FOREACH(GlobalShortcut *sc, ctx->_actions)
348  {
349  if (sc->keys().contains(key)) return false;
350  }
351  }
352  }
353  return true;
354  }

356 GlobalShortcut *Component::registerShortcut(const QString &uniqueName, const QString &friendlyName, const QString &shortcutString, const QString &defaultShortcutString)
357  {
358  // The shortcut will register itself with us
359  GlobalShortcut *shortcut = new GlobalShortcut(
360  uniqueName,
361  friendlyName,
362  currentContext());
363 
364  QList<int> keys = keysFromString(shortcutString);
365  shortcut->setDefaultKeys(keysFromString(defaultShortcutString));
366  shortcut->setIsFresh(false);
367 
368  Q_FOREACH (int key, keys)
369  {
370  if (key != 0)
371  {
372  if (GlobalShortcutsRegistry::self()->getShortcutByKey(key))
373  {
374  // The shortcut is already used. The config file is
375  // broken. Ignore the request.
376  keys.removeAll(key);
377  qCWarning(KGLOBALACCELD) << "Shortcut found twice in kglobalshortcutsrc."<<key;
378  }
379  }
380  }
381  shortcut->setKeys(keys);
382  return shortcut;
383  }

387  {
388  // GlobalShortcutsRegistry::loadSettings handles contexts.
389  Q_FOREACH (const QString &confKey, configGroup.keyList())

397  GlobalShortcut *shortcut = registerShortcut(confKey, entry[2], entry[0], entry[1]);
398  if (configGroup.name().endsWith(QLatin1String(".desktop"))) {
399  shortcut->setIsPresent(true);
400  }

455 void Component::writeSettings(KConfigGroup& configGroup) const
456  {
457  // If we don't delete the current content global shortcut
458  // registrations will never not deleted after forgetGlobalShortcut()
459  configGroup.deleteGroup();
460 
461 
462  // Now write all contexts
463  Q_FOREACH( GlobalShortcutContext *context, _contexts)
464  {
465  KConfigGroup contextGroup;
466 
467  if (context->uniqueName() == QLatin1String("default"))
468  {
469  contextGroup = configGroup;
470  // Write the friendly name
471  contextGroup.writeEntry("_k_friendly_name", friendlyName());
472  }
473  else
474  {
475  contextGroup = KConfigGroup(&configGroup, context->uniqueName());
476  // Write the friendly name
477  contextGroup.writeEntry("_k_friendly_name", context->friendlyName());
478  }
479 
480  // qCDebug(KGLOBALACCELD) << "writing group " << _uniqueName << ":" << context->uniqueName();
481 
482  Q_FOREACH(const GlobalShortcut *shortcut, context->_actions)
483  {
484  // qCDebug(KGLOBALACCELD) << "writing" << shortcut->uniqueName();
485 
486  // We do not write fresh shortcuts.
487  // We do not write session shortcuts
488  if (shortcut->isFresh() || shortcut->isSessionShortcut())
489  {
490  continue;
491  }
492  // qCDebug(KGLOBALACCELD) << "really writing" << shortcut->uniqueName();
493 
494  QStringList entry(stringFromKeys(shortcut->keys()));
495  entry.append(stringFromKeys(shortcut->defaultKeys()));
496  entry.append(shortcut->friendlyName());
497 
498  contextGroup.writeEntry(shortcut->uniqueName(), entry);
499  }
500  }
501  }
