44  // get a hang on kglobalaccel restart (kglobalaccel tries to register with ksmserver,
45  // ksmserver tries to register with kglobalaccel).

54  KAboutLicense::LGPL,

56  aboutdata.addAuthor("Andreas Hartmetz", QObject::tr("Maintainer"), "ahartmetz@gmail.com");
57  aboutdata.addAuthor("Michael Jansen", QObject::tr("Maintainer"), "kde@michael-jansen.biz");
58 
59  app.setWindowIcon(QIcon::fromTheme(QStringLiteral("kglobalaccel")));
60  KAboutData::setApplicationData(aboutdata);
61 
62  // check if kglobalaccel is disabled
