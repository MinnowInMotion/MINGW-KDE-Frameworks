34 OSStatus hotKeyEventHandler(EventHandlerCallRef inHandlerCallRef, EventRef inEvent, void * inUserData)
35 {
36  UInt32 eventKind = GetEventKind(inEvent);
37  if (eventKind == kEventRawKeyDown) {
38  UInt32 keycode;
39  if (GetEventParameter(inEvent, kEventParamKeyCode, typeUInt32, NULL, sizeof(keycode), NULL, &keycode) != noErr) {
40  qCWarning(KGLOBALACCELD) << "Error retrieving keycode parameter from event";
41  }
42  qCDebug(KGLOBALACCELD) << " key down, keycode = " << keycode;
43  } else if (eventKind == kEventHotKeyPressed) {
44  KGlobalAccelImpl* impl = static_cast<KGlobalAccelImpl *>(inUserData);
45  EventHotKeyID hotkey;
46  if (GetEventParameter(inEvent, kEventParamDirectObject, typeEventHotKeyID, NULL, sizeof(hotkey), NULL, &hotkey) != noErr) {
47  qCWarning(KGLOBALACCELD) << "Error retrieving hotkey parameter from event";
48  return eventNotHandledErr;
49  }
50  // Typecasts necesary to prevent a warning from gcc
51  return (impl->keyPressed(hotkey.id) ? (OSStatus) noErr : (OSStatus) eventNotHandledErr);
52  }
53  return eventNotHandledErr;
54 }

60 KGlobalAccelImpl::KGlobalAccelImpl(GlobalShortcutsRegistry* owner)
61  : m_owner(owner)
62  , m_eventTarget(GetApplicationEventTarget())
63  , m_eventHandler(NewEventHandlerUPP(hotKeyEventHandler))
64 {
65  m_eventType[0].eventClass = kEventClassKeyboard;
66  m_eventType[0].eventKind = kEventHotKeyPressed;
67  m_eventType[1].eventClass = kEventClassKeyboard; // only useful for testing, is not used because count passed in call to InstallEventHandler is 1
68  m_eventType[1].eventKind = kEventRawKeyDown;
69  refs = new QMap<int, QList<EventHotKeyRef> >();
70 
71  CFStringRef str = CFStringCreateWithCString(NULL, "AppleKeyboardPreferencesChangedNotification", kCFStringEncodingASCII);
72  if (str) {
73  CFNotificationCenterAddObserver(CFNotificationCenterGetDistributedCenter(), this, layoutChanged, str, NULL, CFNotificationSuspensionBehaviorHold);
74  CFRelease(str);
75  } else {
76  qCWarning(KGLOBALACCELD) << "Couldn't create CFString to register for keyboard notifications";
77  }
78 }

87 bool KGlobalAccelImpl::grabKey( int keyQt, bool grab )
88 {
89  if (grab) {
90  qCDebug(KGLOBALACCELD) << "Grabbing key " << keyQt;
91  QList<uint> keyCodes;
92  uint mod;
93  KKeyServer::keyQtToCodeMac( keyQt, keyCodes );
94  KKeyServer::keyQtToModMac( keyQt, mod );
95 
96  qCDebug(KGLOBALACCELD) << "keyQt: " << keyQt << " mod: " << mod;
97  foreach (uint keyCode, keyCodes) {
98  qCDebug(KGLOBALACCELD) << " keyCode: " << keyCode;
99  }
100 
101  EventHotKeyID ehkid;
102  ehkid.signature = 'Kgai';
103  ehkid.id = keyQt;
104  QList<EventHotKeyRef> hotkeys;
105  foreach (uint keyCode, keyCodes) {
106  EventHotKeyRef ref;
107  if (RegisterEventHotKey(keyCode, mod, ehkid, m_eventTarget, 0, &ref) != noErr) {
108  qCWarning(KGLOBALACCELD) << "RegisterEventHotKey failed!";
109  }
110  hotkeys.append(ref);
111  }
112  refs->insert(keyQt, hotkeys);
113  } else {
114  qCDebug(KGLOBALACCELD) << "Ungrabbing key " << keyQt;
115  if (refs->count(keyQt) == 0) qCWarning(KGLOBALACCELD) << "Trying to ungrab a key thas is not grabbed";
116  foreach (const EventHotKeyRef &ref, refs->value(keyQt)) {
117  if (UnregisterEventHotKey(ref) != noErr) {
118  qCWarning(KGLOBALACCELD) << "UnregisterEventHotKey should not fail!";
119  }
120  }
121  refs->remove(keyQt);
122  }
123  return true;
124 }

142 void KGlobalAccelImpl::keyboardLayoutChanged()
143 {
144  // Keyboard layout might have changed, first ungrab all keys
145  QList<int> keys; // Array to store all the keys that were grabbed
146  while (!refs->empty()) {
147  int key = refs->begin().key();
148  keys.append(key);
149  grabKey(key, false);
150  }
151  // Now re-grab all the keys
152  foreach (int key, keys) {
153  grabKey(key, true);
154  }
155 }
