40 static KGlobalAccelInterface *loadPlugin(GlobalShortcutsRegistry *parent)
41 {
42  const QVector<KPluginMetaData> candidates = KPluginLoader::findPlugins(QStringLiteral("org.kde.kglobalaccel5.platforms"));
43  QString platformName = QString::fromLocal8Bit(qgetenv("KGLOBALACCELD_PLATFORM"));
44  if (platformName.isEmpty()) {
45  platformName = QGuiApplication::platformName();
46  }
47  foreach (const KPluginMetaData &candidate, candidates) {
48  const QJsonArray platforms = candidate.rawData().value(QStringLiteral("platforms")).toArray();
49  for (auto it = platforms.begin(); it != platforms.end(); ++it) {
50  if (QString::compare(platformName, (*it).toString(), Qt::CaseInsensitive) == 0) {
51  KGlobalAccelInterface *interface = qobject_cast< KGlobalAccelInterface* >(candidate.instantiate());
52  if (interface) {
53  qCDebug(KGLOBALACCELD) << "Loaded plugin" << candidate.fileName() << "for platform" << platformName;
54  interface->setRegistry(parent);
55  return interface;
56  }
57  }
58  }
59  }
60  qCWarning(KGLOBALACCELD) << "Could not find any platform plugin";
61  return Q_NULLPTR;
62 }

212 bool GlobalShortcutsRegistry::keyPressed(int keyQt)
213  {
214  GlobalShortcut *shortcut = getShortcutByKey(keyQt);
215  if (!shortcut)
216  {
217  // This can happen for example with the ALT-Print shortcut of kwin.
218  // ALT+PRINT is SYSREQ on my keyboard. So we grab something we think
219  // is ALT+PRINT but symXToKeyQt and modXToQt make ALT+SYSREQ of it
220  // when pressed (correctly). We can't match that.
221 #ifdef KDEDGLOBALACCEL_TRACE
222  qCDebug(KGLOBALACCELD) << "Got unknown key" << QKeySequence(keyQt).toString();
223 #endif
224 
225  // In production mode just do nothing.
226  return false;
227  }
228  else if (!shortcut->isActive())
229  {
230 #ifdef KDEDGLOBALACCEL_TRACE
231  qCDebug(KGLOBALACCELD) << "Got inactive key" << QKeySequence(keyQt).toString();
232 #endif
233 
234  // In production mode just do nothing.
235  return false;
236  }
237 
238 #ifdef KDEDGLOBALACCEL_TRACE
239  qCDebug(KGLOBALACCELD) << QKeySequence(keyQt).toString() << "=" << shortcut->uniqueName();
240 #endif
241 
242  QStringList data(shortcut->context()->component()->uniqueName());
243  data.append(shortcut->uniqueName());
244  data.append(shortcut->context()->component()->friendlyName());
245  data.append(shortcut->friendlyName());
246 
247  // Make sure kglobalacceld has ungrabbed the keyboard after receiving the
248  // keypress, otherwise actions in application that try to grab the
249  // keyboard (e.g. in kwin) may fail to do so. There is still a small race
250  // condition with this being out-of-process.
251  if (_manager) {
252  _manager->syncWindowingSystem();
253  }
254 
255  // 1st Invoke the action
256  shortcut->context()->component()->emitGlobalShortcutPressed( *shortcut );
257 
258  return true;
259 }

262 void GlobalShortcutsRegistry::loadSettings()
263  {
264  foreach (const QString &groupName, _config.groupList())
265  {
266  qCDebug(KGLOBALACCELD) << "Loading group " << groupName;
267 
268  Q_ASSERT(groupName.indexOf('\x1d')==-1);
269 
270  // loadSettings isn't designed to be called in between. Only at the
271  // beginning.
272  Q_ASSERT(!getComponent(groupName));
273 
274  KConfigGroup configGroup(&_config, groupName);
275 
276  // We previously stored the friendly name in a separate group. migrate
277  // that
278  const QString friendlyName = configGroup.readEntry("_k_friendly_name");
279 
280  // Create the component
281  KdeDGlobalAccel::Component *component = nullptr;
282  if (groupName.endsWith(QLatin1String(".desktop"))) {
283  component = new KdeDGlobalAccel::KServiceActionComponent(
284  groupName,
285  friendlyName,
286  this);
287  } else {
288  component = new KdeDGlobalAccel::Component(
289  groupName,
290  friendlyName,
291  this);
292  }
293 
294  // Now load the contexts
295  Q_FOREACH(const QString& context, configGroup.groupList())
296  {
297  // Skip the friendly name group
298  if (context==QLatin1String("Friendly Name")) continue;
299 
300  KConfigGroup contextGroup(&configGroup, context);
301  QString contextFriendlyName = contextGroup.readEntry("_k_friendly_name");
302  component->createGlobalShortcutContext(context, contextFriendlyName);
303  component->activateGlobalShortcutContext(context);
304  component->loadSettings(contextGroup);
305  }
306 
307  // Load the default context
308  component->activateGlobalShortcutContext("default");
309  component->loadSettings(configGroup);
310  }
311 
312  // Load the configured KServiceActions
313  const QStringList desktopPaths = QStandardPaths::locateAll(QStandardPaths::GenericDataLocation, QStringLiteral("kglobalaccel"), QStandardPaths::LocateDirectory);
314  foreach (const QString &path, desktopPaths) {
315  QDir dir(path);
316  if (!dir.exists()) {
317  continue;
318  }
319  const QStringList patterns = {QStringLiteral("*.desktop")};
320  foreach (const QString &desktopFile, dir.entryList(patterns)) {
321  if (_components.contains(desktopFile)) {
322  continue;
323  }
324 
325  KDesktopFile f(dir.filePath(desktopFile));
326  if (f.noDisplay()) {
327  continue;
328  }
329 
330  KdeDGlobalAccel::KServiceActionComponent *component = new KdeDGlobalAccel::KServiceActionComponent(
331  desktopFile,
332  f.readName(),
333  this);
334  component->activateGlobalShortcutContext(QStringLiteral("default"));
335  component->loadFromService();
336  }
337  }
338  }

347 bool GlobalShortcutsRegistry::registerKey(int key, GlobalShortcut *shortcut)
348  {
349  if (!_manager) {
350  return false;
351  }
352  if (key == 0)
353  {
354  qCDebug(KGLOBALACCELD) << shortcut->uniqueName() << ": Key '" << QKeySequence(key).toString()
355  << "' already taken by " << _active_keys.value(key)->uniqueName() << ".";
356  return false;
357  }
358  else if (_active_keys.value(key))
359  {
360  qCDebug(KGLOBALACCELD) << shortcut->uniqueName() << ": Attempt to register key 0.";
361  return false;
362  }
363 
364  qCDebug(KGLOBALACCELD) << "Registering key" << QKeySequence(key).toString() << "for"
365  << shortcut->context()->component()->uniqueName() << ":" << shortcut->uniqueName();
366 
367  _active_keys.insert(key, shortcut);
368  return _manager->grabKey(key, true);
369  }

398 bool GlobalShortcutsRegistry::unregisterKey(int key, GlobalShortcut *shortcut)
399  {
400  if (!_manager) {
401  return false;
402  }
403  if (_active_keys.value(key)!=shortcut)
404  {
405  // The shortcut doesn't own the key or the key isn't grabbed
406  return false;
407  }
408 
409  qCDebug(KGLOBALACCELD) << "Unregistering key" << QKeySequence(key).toString() << "for"
410  << shortcut->context()->component()->uniqueName() << ":" << shortcut->uniqueName();
411 
412  _manager->grabKey(key, false);
413  _active_keys.take(key);
414  return true;
415  }
