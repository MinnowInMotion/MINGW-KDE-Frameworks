41 org::kde::kglobalaccel::Component *KGlobalAccelPrivate::getComponent(const QString &componentUnique, bool remember = false)
42 {
43  // Check if we already have this component
44  if (components.contains(componentUnique)) {
45  return components[componentUnique];
46  }
47 
48  // Connect to the kglobalaccel daemon
49  org::kde::KGlobalAccel kglobalaccel(
50  QStringLiteral("org.kde.kglobalaccel"),
51  QStringLiteral("/kglobalaccel"),
52  QDBusConnection::sessionBus());
53  if (!kglobalaccel.isValid()) {
54  qDebug() << "Failed to connect to the kglobalaccel daemon" << QDBusConnection::sessionBus().lastError();
55  return NULL;
56  }
57 
58  // Get the path for our component. We have to do that because
59  // componentUnique is probably not a valid dbus object path
60  QDBusReply<QDBusObjectPath> reply = kglobalaccel.getComponent(componentUnique);
61  if (!reply.isValid()) {
62 
63  if (reply.error().name() == QLatin1String("org.kde.kglobalaccel.NoSuchComponent")) {
64  // No problem. The component doesn't exists. That's normal
65  return NULL;
66  }
67 
68  // An unknown error.
69  qDebug() << "Failed to get dbus path for component " << componentUnique << reply.error();
70  return NULL;
71  }
72 
73  // Now get the component
74  org::kde::kglobalaccel::Component *component = new org::kde::kglobalaccel::Component(
75  QStringLiteral("org.kde.kglobalaccel"),
76  reply.value().path(),
77  QDBusConnection::sessionBus(),
78  q);
79 
80  // No component no cleaning
81  if (!component->isValid()) {
82  qDebug() << "Failed to get component" << componentUnique << QDBusConnection::sessionBus().lastError();
83  return NULL;
84  }
85 
86  if (remember) {
87  // Connect to the signals we are interested in.
88  q->connect(component, SIGNAL(globalShortcutPressed(QString,QString,qlonglong)),
89  SLOT(_k_invokeAction(QString,QString,qlonglong)));
90 
91  components[componentUnique] = component;
92  }
93 
94  return component;
95 }

235 bool KGlobalAccelPrivate::doRegister(QAction *action)
236 {
237  if (!action || action->objectName().isEmpty() || action->objectName().startsWith(QLatin1String("unnamed-"))) {
238  qWarning() << "Attempt to set global shortcut for action without objectName()."
239  " Read the setGlobalShortcut() documentation.";
240  return false;
241  }
242 
243  const bool isRegistered = actions.contains(action);
244  if (isRegistered) {
245  return true;
246  }
247 
248  QStringList actionId = makeActionId(action);
249 
250  nameToAction.insertMulti(actionId.at(KGlobalAccel::ActionUnique), action);
251  actions.insert(action);
252  iface()->doRegister(actionId);
253 
254  QObject::connect(action, &QObject::destroyed, [this, action](QObject *) {
255  if (actions.contains(action) && (actionShortcuts.contains(action) || actionDefaultShortcuts.contains(action))) {
256  remove(action, KGlobalAccelPrivate::SetInactive);
257  }
258  });
259 
260  return true;
261 }

263 void KGlobalAccelPrivate::remove(QAction *action, Removal removal)
264 {
265  if (!action || action->objectName().isEmpty()) {
266  return;
267  }
268 
269  const bool isRegistered = actions.contains(action);
270  if (!isRegistered) {
271  return;
272  }
273 
274  QStringList actionId = makeActionId(action);
275 
276  nameToAction.remove(actionId.at(KGlobalAccel::ActionUnique), action);
277  actions.remove(action);
278 
279  if (removal == UnRegister) {
280  // Complete removal of the shortcut is requested
281  // (forgetGlobalShortcut)
282  iface()->unRegister(actionId);
283  } else {
284  // If the action is a configurationAction wen only remove it from our
285  // internal registry. That happened above.
286  if (!action->property("isConfigurationAction").toBool()) {
287  // If it's a session shortcut unregister it.
288  action->objectName().startsWith(QStringLiteral("_k_session:"))
289  ? iface()->unRegister(actionId)
290  : iface()->setInactive(actionId);
291  }
292  }
293 
294  actionDefaultShortcuts.remove(action);
295  actionShortcuts.remove(action);
296 }

298 void KGlobalAccelPrivate::updateGlobalShortcut(/*const would be better*/QAction* action,
299  ShortcutTypes actionFlags,
300  KGlobalAccel::GlobalShortcutLoading globalFlags)
301 {
302  // No action or no objectname -> Do nothing
303  if (!action || action->objectName().isEmpty()) {
304  return;
305  }
306 
307  QStringList actionId = makeActionId(action);
308  const QList<QKeySequence> activeShortcut = actionShortcuts.value(action);
309  const QList<QKeySequence> defaultShortcut = actionDefaultShortcuts.value(action);
310 
311  uint setterFlags = 0;
312  if (globalFlags & NoAutoloading) {
313  setterFlags |= NoAutoloading;
314  }
315 
316  if (actionFlags & ActiveShortcut) {
317  bool isConfigurationAction = action->property("isConfigurationAction").toBool();
318  uint activeSetterFlags = setterFlags;
319 
320  // setPresent tells kglobalaccel that the shortcut is active
321  if (!isConfigurationAction) {
322  activeSetterFlags |= SetPresent;
323  }
324 
325  // Sets the shortcut, returns the active/real keys
326  const QList<int> result = iface()->setShortcut(
327  actionId,
328  intListFromShortcut(activeShortcut),
329  activeSetterFlags);
330 
331  // Make sure we get informed about changes in the component by kglobalaccel
332  getComponent(componentUniqueForAction(action), true);
333 
334  // Create a shortcut from the result
335  const QList<QKeySequence> scResult(shortcutFromIntList(result));
336 
337  if (isConfigurationAction && (globalFlags & NoAutoloading)) {
338  // If this is a configuration action and we have set the shortcut,
339  // inform the real owner of the change.
340  // Note that setForeignShortcut will cause a signal to be sent to applications
341  // even if it did not "see" that the shortcut has changed. This is Good because
342  // at the time of comparison (now) the action *already has* the new shortcut.
343  // We called setShortcut(), remember?
344  // Also note that we will see our own signal so we may not need to call
345  // setActiveGlobalShortcutNoEnable - _k_shortcutGotChanged() does it.
346  // In practice it's probably better to get the change propagated here without
347  // DBus delay as we do below.
348  iface()->setForeignShortcut(actionId, result);
349  }
350  if (scResult != activeShortcut) {
351  // If kglobalaccel returned a shortcut that differs from the one we
352  // sent, use that one. There must have been clashes or some other problem.
353  actionShortcuts.insert(action, scResult);
354  emit q->globalShortcutChanged(action, scResult.isEmpty() ? QKeySequence() : scResult.first());
355  }
356  }
357 
358  if (actionFlags & DefaultShortcut) {
359  iface()->setShortcut(actionId, intListFromShortcut(defaultShortcut),
360  setterFlags | IsDefault);
361  }
362 }

430 void KGlobalAccelPrivate::_k_invokeAction(
431  const QString &componentUnique,
432  const QString &actionUnique,
433  qlonglong timestamp)
434 {
435  QAction *action = 0;
436  QList<QAction *> candidates = nameToAction.values(actionUnique);
437  Q_FOREACH (QAction *const a, candidates) {
438  if (componentUniqueForAction(a) == componentUnique) {
439  action = a;
440  }
441  }
442 
443  // We do not trigger if
444  // - there is no action
445  // - the action is not enabled
446  // - the action is an configuration action
447  if (!action || !action->isEnabled() || action->property("isConfigurationAction").toBool()) {
448  return;
449  }
450 
451 #if HAVE_X11
452  // Update this application's X timestamp if needed.
453  // TODO The 100%-correct solution should probably be handling this action
454  // in the proper place in relation to the X events queue in order to avoid
455  // the possibility of wrong ordering of user events.
456  if (QX11Info::isPlatformX11()) {
457  if (_k_timestampCompare(timestamp, QX11Info::appTime()) > 0) {
458  QX11Info::setAppTime(timestamp);
459  }
460  if (_k_timestampCompare(timestamp, QX11Info::appUserTime()) > 0) {
461  QX11Info::setAppUserTime(timestamp);
462  }
463  }
464 #endif
465  action->setProperty("org.kde.kglobalaccel.activationTimestamp", timestamp);
466 
467  action->trigger();
468 }

494 void KGlobalAccelPrivate::reRegisterAll()
495 {
496  //We clear all our data, assume that all data on the other side is clear too,
497  //and register each action as if it just was allowed to have global shortcuts.
498  //If the kded side still has the data it doesn't matter because of the
499  //autoloading mechanism. The worst case I can imagine is that an action's
500  //shortcut was changed but the kded side died before it got the message so
501  //autoloading will now assign an old shortcut to the action. Particularly
502  //picky apps might assert or misbehave.
503  QSet<QAction *> allActions = actions;
504  nameToAction.clear();
505  actions.clear();
506  Q_FOREACH (QAction *const action, allActions) {
507  if (doRegister(action)) {
508  updateGlobalShortcut(action, ActiveShortcut, KGlobalAccel::Autoloading);
509  }
510  }
511 }
