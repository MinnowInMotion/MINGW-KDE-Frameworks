34 void KNSBookmarkImporterImpl::parse()
35 {
36  QFile f(m_fileName);
37  QTextCodec *codec = m_utf8 ? QTextCodec::codecForName("UTF-8") : QTextCodec::codecForLocale();
38  Q_ASSERT(codec);
39  if (!codec) {
40  return;
41  }
42 
43  if (f.open(QIODevice::ReadOnly)) {
44 
45  static const int g_lineLimit = 16 * 1024;
46  QByteArray s(g_lineLimit, 0);
47  // skip header
48  while (f.readLine(s.data(), g_lineLimit) >= 1 && !s.contains("<DL>")) {
49  ;
50  }
51 
52  while (int size = f.readLine(s.data(), g_lineLimit) >= 1) {
53  if (size == g_lineLimit) { // Gosh, this line is longer than g_lineLimit. Skipping.
54  qWarning() << "Netscape bookmarks contain a line longer than " << g_lineLimit << ". Skipping.";
55  continue;
56  }
57  QByteArray t = s.trimmed();
58 
59  if (t.left(4).toUpper() == "<HR>") {
60  emit newSeparator();
61  t = t.mid(4).trimmed();
62  if (t.isEmpty()) {
63  continue;
64  }
65  }
66 
67  if (t.left(12).toUpper() == "<DT><A HREF=" ||
68  t.left(16).toUpper() == "<DT><H3><A HREF=") {
69 
70  int firstQuotes = t.indexOf('"') + 1;
71  int secondQuotes = t.indexOf('"', firstQuotes);
72  if (firstQuotes != -1 && secondQuotes != -1) {
73  QByteArray link = t.mid(firstQuotes, secondQuotes - firstQuotes);
74  int endTag = t.indexOf('>', secondQuotes + 1);
75 
76  int closeTag = t.indexOf('<', endTag + 1);
77 
78  QByteArray name = t.mid(endTag + 1, closeTag - endTag - 1);
79  QString qname = KCharsets::resolveEntities(codec->toUnicode(name));
80  QByteArray additionalInfo = t.mid(secondQuotes + 1, endTag - secondQuotes - 1);
81 
82  emit newBookmark(qname,
83  codec->toUnicode(link),
84  QByteArray());
85  }
86  } else if (t.left(7).toUpper() == "<DT><H3") {
87  int endTag = t.indexOf('>', 7);
88  QByteArray name = t.mid(endTag + 1);
89  name = name.left(name.indexOf('<'));
90  QString qname = KCharsets::resolveEntities(codec->toUnicode(name));
91  QByteArray additionalInfo = t.mid(8, endTag - 8);
92  bool folded = (additionalInfo.left(6) == "FOLDED");
93  if (folded) {
94  additionalInfo.remove(0, 7);
95  }
96 
97  emit newFolder(qname,
98  !folded,
99  QByteArray());
100  } else if (t.left(8).toUpper() == "</DL><P>") {
101  emit endFolder();
102  }
103  }
104 
105  f.close();
106  }
107 }

132 void KNSBookmarkExporterImpl::write(const KBookmarkGroup &parent)
133 {
134  if (!QFile::exists(m_fileName)) {
135  QString errorMsg = KNSBookmarkImporterImpl::tr("Could not find %1. Netscape is probably not installed. "
136  "Aborting the export.").arg(m_fileName);
137  QMessageBox::critical(0, KNSBookmarkImporterImpl::tr("Netscape not found"), errorMsg);
138  return;
139  }
140  if (QFile::exists(m_fileName)) {
141  (void)QFile::rename(m_fileName, m_fileName + ".beforekde");
142  }
143 
144  QFile file(m_fileName);
145 
146  if (!file.open(QIODevice::WriteOnly)) {
147  qCritical() << "Can't write to file " << m_fileName << endl;
148  return;
149  }
150 
151  QTextStream fstream(&file);
152  fstream.setCodec(m_utf8 ? QTextCodec::codecForName("UTF-8") : QTextCodec::codecForLocale());
153 
154  QString charset
155  = m_utf8 ? QStringLiteral("UTF-8") : QString::fromLatin1(QTextCodec::codecForLocale()->name()).toUpper();
156 
157  fstream << "<!DOCTYPE NETSCAPE-Bookmark-file-1>" << endl
158  << KNSBookmarkImporterImpl::tr("<!-- This file was generated by Konqueror -->") << endl
159  << "<META HTTP-EQUIV=\"Content-Type\" CONTENT=\"text/html; charset="
160  << charset << "\">" << endl
161  << "<TITLE>" << KNSBookmarkImporterImpl::tr("Bookmarks") << "</TITLE>" << endl
162  << "<H1>" << KNSBookmarkImporterImpl::tr("Bookmarks") << "</H1>" << endl
163  << "<DL><p>" << endl
164  << folderAsString(parent)
165  << "</DL><P>" << endl;
166 }

168 QString KNSBookmarkExporterImpl::folderAsString(const KBookmarkGroup &parent) const
169 {
170  QString str;
171  QTextStream fstream(&str, QIODevice::WriteOnly);
172 
173  for (KBookmark bk = parent.first(); !bk.isNull(); bk = parent.next(bk)) {
174  if (bk.isSeparator()) {
175  fstream << "<HR>" << endl;
176  continue;
177  }
178 
179  QString text = bk.fullText().toHtmlEscaped();
180 
181  if (bk.isGroup()) {
182  fstream << "<DT><H3 "
183  << (!bk.toGroup().isOpen() ? "FOLDED " : "")
184  << bk.internalElement().attribute(QStringLiteral("netscapeinfo")) << ">"
185  << text << "</H3>" << endl
186  << "<DL><P>" << endl
187  << folderAsString(bk.toGroup())
188  << "</DL><P>" << endl;
189  continue;
190 
191  } else {
192  // note - netscape seems to use local8bit for url...
193  fstream << "<DT><A HREF=\"" << bk.url().toString() << "\""
194  << bk.internalElement().attribute(QStringLiteral("netscapeinfo")) << ">"
195  << text << "</A>" << endl;
196  continue;
197  }
198  }
199 
200  return str;
201 }
