61 void KIEBookmarkImporter::parseIEBookmarks_url_file(const QString &filename, const QString &name)
62 {
63  static const int g_lineLimit = 16 * 1024;
64 
65  QFile f(filename);
66 
67  if (f.open(QIODevice::ReadOnly)) {
68 
69  QByteArray s(g_lineLimit, 0);
70 
71  while (f.readLine(s.data(), g_lineLimit) >= 0) {
72  if (s[s.length() - 1] != '\n') { // Gosh, this line is longer than g_lineLimit. Skipping.
73  qWarning() << "IE bookmarks contain a line longer than " << g_lineLimit << ". Skipping.";
74  continue;
75  }
76  QByteArray t = s.trimmed();
77  QRegExp rx(QStringLiteral("URL=(.*)"));
78  if (rx.exactMatch(t)) {
79  emit newBookmark(name, rx.cap(1), QLatin1String(""));
80  }
81  }
82 
83  f.close();
84  }
85 }

87 void KIEBookmarkImporter::parseIEBookmarks_dir(const QString &dirname, const QString &foldername)
88 {
89 
90  QDir dir(dirname);
91  dir.setFilter(QDir::Files | QDir::Dirs | QDir::AllDirs);
92  dir.setSorting(QFlags<QDir::SortFlag>(QDir::Name | QDir::DirsFirst));
93  dir.setNameFilters(QStringList(QStringLiteral("*.url"))); // AK - possibly add ";index.ini" ?
94 
95  const QFileInfoList list = dir.entryInfoList();
96  if (list.isEmpty()) {
97  return;
98  }
99 
100  if (dirname != m_fileName) {
101  emit newFolder(foldername, false, QLatin1String(""));
102  }
103 
104  foreach (const QFileInfo &fi, list) {
105  if (fi.fileName() == QLatin1String(".") || fi.fileName() == QLatin1String("..")) {
106  continue;
107  }
108 
109  if (fi.isDir()) {
110  parseIEBookmarks_dir(fi.absoluteFilePath(), fi.fileName());
111 
112  } else if (fi.isFile()) {
113  if (fi.fileName().endsWith(QLatin1String(".url"))) {
114  QString name = fi.fileName();
115  name.truncate(name.length() - 4); // .url
116  parseIEBookmarks_url_file(fi.absoluteFilePath(), name);
117  }
118  // AK - add index.ini
119  }
120  }
121 
122  if (dirname != m_fileName) {
123  emit endFolder();
124  }
125 }

189 void IEExporter::visit(const KBookmark &bk)

191  QString fname = m_currentDir.path() + '/' + ieStyleQuote(bk.fullText()) + ".url";
