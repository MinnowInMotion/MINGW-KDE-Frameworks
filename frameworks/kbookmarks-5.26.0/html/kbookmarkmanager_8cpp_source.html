374 void KBookmarkManager::parse() const
375 {
376  d->m_docIsLoaded = true;
377  // qDebug() << "KBookmarkManager::parse " << d->m_bookmarksFile;
378  QFile file(d->m_bookmarksFile);
379  if (!file.open(QIODevice::ReadOnly)) {
380  qWarning() << "Can't open " << d->m_bookmarksFile;
381  return;
382  }
383  d->m_doc = QDomDocument(QStringLiteral("xbel"));
384  d->m_doc.setContent(&file);
385 
386  if (d->m_doc.documentElement().isNull()) {
387  qWarning() << "KBookmarkManager::parse : main tag is missing, creating default " << d->m_bookmarksFile;
388  QDomElement element = d->m_doc.createElement(QStringLiteral("xbel"));
389  d->m_doc.appendChild(element);
390  }
391 
392  QDomElement docElem = d->m_doc.documentElement();
393 
394  QString mainTag = docElem.tagName();
395  if (mainTag != QLatin1String("xbel")) {
396  qWarning() << "KBookmarkManager::parse : unknown main tag " << mainTag;
397  }
398 
399  if (d->m_dbusObjectName.isNull()) {
400  d->m_dbusObjectName = docElem.attribute(QStringLiteral("dbusName"));
401  } else if (docElem.attribute(QStringLiteral("dbusName")) != d->m_dbusObjectName) {
402  docElem.setAttribute(QStringLiteral("dbusName"), d->m_dbusObjectName);
403  save();
404  }
405 
406  QDomNode n = d->m_doc.documentElement().previousSibling();
407  if (n.isProcessingInstruction()) {
408  QDomProcessingInstruction pi = n.toProcessingInstruction();
409  pi.parentNode().removeChild(pi);
410  }
411 
412  QDomProcessingInstruction pi;
413  pi = d->m_doc.createProcessingInstruction(QStringLiteral("xml"), PI_DATA);
414  d->m_doc.insertBefore(pi, docElem);
415 
416  file.close();
417 
418  d->m_map.setNeedsUpdate();
419 }

426 bool KBookmarkManager::saveAs(const QString &filename, bool toolbarCache) const
427 {
428  // qDebug() << "KBookmarkManager::save " << filename;
429 
430  // Save the bookmark toolbar folder for quick loading
431  // but only when it will actually make things quicker
432  const QString cacheFilename = filename + QLatin1String(".tbcache");
433  if (toolbarCache && !root().isToolbarGroup()) {
434  QSaveFile cacheFile(cacheFilename);
435  if (cacheFile.open(QIODevice::WriteOnly)) {
436  QString str;
437  QTextStream stream(&str, QIODevice::WriteOnly);
438  stream << root().findToolbar();
439  const QByteArray cstr = str.toUtf8();
440  cacheFile.write(cstr.data(), cstr.length());
441  cacheFile.commit();
442  }
443  } else { // remove any (now) stale cache
444  QFile::remove(cacheFilename);
445  }
446 
447  // Create parent dirs
448  QFileInfo info(filename);
449  QDir().mkpath(info.absolutePath());
450 
451  QSaveFile file(filename);
452  if (file.open(QIODevice::WriteOnly)) {
453  KBackup::simpleBackupFile(file.fileName(), QString(), QStringLiteral(".bak"));
454  QTextStream stream(&file);
455  stream.setCodec(QTextCodec::codecForName("UTF-8"));
456  stream << internalDocument().toString();
457  stream.flush();
458  if (file.commit()) {
459  return true;
460  }
461  }
462 
463  static int hadSaveError = false;
464  if (!hadSaveError) {
465  QString err = tr("Unable to save bookmarks in %1. Reported error was: %2. "
466  "This error message will only be shown once. The cause "
467  "of the error needs to be fixed as quickly as possible, "
468  "which is most likely a full hard drive.").arg(filename).arg(file.errorString());
469 
470  if (d->m_dialogAllowed && qobject_cast<QApplication *>(qApp) && QThread::currentThread() == qApp->thread()) {
471  QMessageBox::critical(QApplication::activeWindow(), QApplication::applicationName(), err);
472  }
473 
474  qCritical() << QStringLiteral("Unable to save bookmarks in %1. File reported the following error-code: %2.").arg(filename).arg(file.error());
475  emit const_cast<KBookmarkManager *>(this)->error(err);
476  }
477  hadSaveError = true;
478  return false;
479 }

491 KBookmarkGroup KBookmarkManager::toolbar()
492 {
493  // qDebug() << "KBookmarkManager::toolbar begin";
494  // Only try to read from a toolbar cache if the full document isn't loaded
495  if (!d->m_docIsLoaded) {
496  // qDebug() << "KBookmarkManager::toolbar trying cache";
497  const QString cacheFilename = d->m_bookmarksFile + QLatin1String(".tbcache");
498  QFileInfo bmInfo(d->m_bookmarksFile);
499  QFileInfo cacheInfo(cacheFilename);
500  if (d->m_toolbarDoc.isNull() &&
501  QFile::exists(cacheFilename) &&
502  bmInfo.lastModified() < cacheInfo.lastModified()) {
503  // qDebug() << "KBookmarkManager::toolbar reading file";
504  QFile file(cacheFilename);
505 
506  if (file.open(QIODevice::ReadOnly)) {
507  d->m_toolbarDoc = QDomDocument(QStringLiteral("cache"));
508  d->m_toolbarDoc.setContent(&file);
509  // qDebug() << "KBookmarkManager::toolbar opened";
510  }
511  }
512  if (!d->m_toolbarDoc.isNull()) {
513  // qDebug() << "KBookmarkManager::toolbar returning element";
514  QDomElement elem = d->m_toolbarDoc.firstChild().toElement();
515  return KBookmarkGroup(elem);
516  }
517  }
518 
519  // Fallback to the normal way if there is no cache or if the bookmark file
520  // is already loaded
521  QDomElement elem = root().findToolbar();
522  if (elem.isNull()) {
523  // Root is the bookmark toolbar if none has been set.
524  // Make it explicit to speed up invocations of findToolbar()
525  root().internalElement().setAttribute(QStringLiteral("toolbar"), QStringLiteral("yes"));
526  return root();
527  } else {
528  return KBookmarkGroup(elem);
529  }
530 }

532 KBookmark KBookmarkManager::findByAddress(const QString &address)
533 {
534  // qDebug() << "KBookmarkManager::findByAddress " << address;
535  KBookmark result = root();
536  // The address is something like /5/10/2+
537  const QStringList addresses = address.split(QRegExp(QStringLiteral("[/+]")), QString::SkipEmptyParts);
538  // qWarning() << addresses.join(",");
539  for (QStringList::const_iterator it = addresses.begin(); it != addresses.end();) {
540  bool append = ((*it) == QLatin1String("+"));
541  uint number = (*it).toUInt();
542  Q_ASSERT(result.isGroup());
543  KBookmarkGroup group = result.toGroup();
544  KBookmark bk = group.first(), lbk = bk; // last non-null bookmark
545  for (uint i = 0; ((i < number) || append) && !bk.isNull(); ++i) {
546  lbk = bk;
547  bk = group.next(bk);
548  // qWarning() << i;
549  }
550  it++;
551  // qWarning() << "found section";
552  result = bk;
553  }
554  if (result.isNull()) {
555  qWarning() << "KBookmarkManager::findByAddress: couldn't find item " << address;
556  }
557  // qWarning() << "found " << result.address();
558  return result;
559 }
