102 void KBookmarkDialogPrivate::fillGroup(QTreeWidgetItem *parentItem, const KBookmarkGroup &group, const KBookmarkGroup &selectGroup)
103 {
104  for (KBookmark bk = group.first(); !bk.isNull(); bk = group.next(bk)) {
105  if (bk.isGroup()) {
106  const KBookmarkGroup bkGroup = bk.toGroup();
107  QTreeWidgetItem* item = new KBookmarkTreeItem(parentItem, folderTree, bkGroup);
108  if (selectGroup == bkGroup) {
109  folderTree->setCurrentItem(item);
110  }
111  fillGroup(item, bkGroup, selectGroup);
112  }
113  }
114 }

116 void KBookmarkDialogPrivate::setParentBookmark(const KBookmark &bm)
117 {
118  QString address = bm.address();
119  KBookmarkTreeItem *item = static_cast<KBookmarkTreeItem *>(folderTree->topLevelItem(0));
120  while (true) {
121  if (item->address() == bm.address()) {
122  folderTree->setCurrentItem(item);
123  return;
124  }
125  for (int i = 0; i < item->childCount(); ++i) {
126  KBookmarkTreeItem *child = static_cast<KBookmarkTreeItem *>(item->child(i));
127  if (KBookmark::commonParent(child->address(), address) == child->address()) {
128  item = child;
129  break;
130  }
131  }
132  }
133 }

145 void KBookmarkDialog::accept()
146 {
147  if (d->mode == KBookmarkDialogPrivate::NewFolder) {
148  KBookmarkGroup parent = d->parentBookmark();
149  if (d->title->text().isEmpty()) {
150  d->title->setText(QStringLiteral("New Folder"));
151  }
152  d->bm = parent.createNewFolder(d->title->text());
153  d->bm.setDescription(d->comment->text());
154  d->mgr->emitChanged(parent);
155  } else if (d->mode == KBookmarkDialogPrivate::NewBookmark) {
156  KBookmarkGroup parent = d->parentBookmark();
157  if (d->title->text().isEmpty()) {
158  d->title->setText(QStringLiteral("New Bookmark"));
159  }
160  d->bm = parent.addBookmark(d->title->text(), QUrl(d->url->text()), d->icon);
161  d->bm.setDescription(d->comment->text());
162  d->mgr->emitChanged(parent);
163  } else if (d->mode == KBookmarkDialogPrivate::NewMultipleBookmarks) {
164  KBookmarkGroup parent = d->parentBookmark();
165  if (d->title->text().isEmpty()) {
166  d->title->setText(QStringLiteral("New Folder"));
167  }
168  d->bm = parent.createNewFolder(d->title->text());
169  d->bm.setDescription(d->comment->text());
170  foreach (const KBookmarkOwner::FutureBookmark &fb, d->list) {
171  d->bm.toGroup().addBookmark(fb.title(), fb.url(), fb.icon());
172  }
173  d->mgr->emitChanged(parent);
174  } else if (d->mode == KBookmarkDialogPrivate::EditBookmark) {
175  d->bm.setFullText(d->title->text());
176  d->bm.setUrl(QUrl(d->url->text()));
177  d->bm.setDescription(d->comment->text());
178  d->mgr->emitChanged(d->bm.parentGroup());
179  } else if (d->mode == d->SelectFolder) {
180  d->bm = d->parentBookmark();
181  }
182  QDialog::accept();
183 }

358 void KBookmarkDialog::newFolderButton()
359 {
360 
361  QString caption = d->parentBookmark().fullText().isEmpty() ?
362  tr("Create New Bookmark Folder", "@title:window") :
363  tr("Create New Bookmark Folder in %1", "@title:window").arg(d->parentBookmark().text());
364  bool ok;
365  QString text = QInputDialog::getText(this, caption, tr("New folder:", "@label:textbox"),
366  QLineEdit::Normal, QString(), &ok);
367  if (!ok) {
368  return;
369  }
370 
371  KBookmarkGroup group = d->parentBookmark().createNewFolder(text);
372  if (!group.isNull()) {
373  KBookmarkGroup parentGroup = group.parentGroup();
374  d->mgr->emitChanged(parentGroup);
375  d->folderTree->clear();
376  QTreeWidgetItem *root = new KBookmarkTreeItem(d->folderTree);
377  d->fillGroup(root, d->mgr->root(), group);
378  }
379 }
