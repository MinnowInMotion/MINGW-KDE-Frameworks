65 static QDomNode findMetadata(const QString &forOwner, QDomNode &parent, bool create)
66 {
67  bool forOwnerIsKDE = forOwner == METADATA_KDE_OWNER;
68 
69  QDomElement metadataElement;
70  for (QDomNode _node = parent.firstChild(); !_node.isNull(); _node = _node.nextSibling()) {
71  QDomElement elem = _node.toElement();
72  if (!elem.isNull() && elem.tagName() == QLatin1String("metadata")) {
73  const QString owner = elem.attribute(QStringLiteral("owner"));
74  if (owner == forOwner) {
75  return elem;
76  }
77  if (owner.isEmpty() && forOwnerIsKDE) {
78  metadataElement = elem;
79  }
80  }
81  }
82  if (create && metadataElement.isNull()) {
83  metadataElement = parent.ownerDocument().createElement(QStringLiteral("metadata"));
84  parent.appendChild(metadataElement);
85  metadataElement.setAttribute(QStringLiteral("owner"), forOwner);
86 
87  } else if (!metadataElement.isNull() && forOwnerIsKDE) {
88  // i'm not sure if this is good, we shouln't take over foreign metatdata
89  metadataElement.setAttribute(QStringLiteral("owner"), METADATA_KDE_OWNER);
90  }
91  return metadataElement;
92 }

181 bool KBookmarkGroup::moveBookmark(const KBookmark &item, const KBookmark &after)
182 {
183  QDomNode n;
184  if (!after.isNull()) {
185  n = element.insertAfter(item.element, after.element);
186  } else { // first child
187  if (element.firstChild().isNull()) { // Empty element -> set as real first child
188  n = element.insertBefore(item.element, QDomElement());
189  }
190 
191  // we have to skip everything up to the first valid child
192  QDomElement firstChild = nextKnownTag(element.firstChild().toElement(), true);
193  if (!firstChild.isNull()) {
194  if (firstChild == item.element) { // item is already the first child, done
195  return true;
196  }
197  n = element.insertBefore(item.element, firstChild);
198  } else {
199  // No real first child -> append after the <title> etc.
200  n = element.appendChild(item.element);
201  }
202  }
203  return (!n.isNull());
204 }

345 QString KBookmark::icon() const
346 {
347  QDomNode metaDataNode = metaData(METADATA_FREEDESKTOP_OWNER, false);
348  QDomElement iconElement = cd(metaDataNode, QStringLiteral("bookmark:icon"), false).toElement();
349 
350  QString icon = iconElement.attribute(QStringLiteral("name"));
351 
352  // migration code
353  if (icon.isEmpty()) {
354  icon = element.attribute(QStringLiteral("icon"));
355  }
356  if (icon == QLatin1String("www")) { // common icon for kde3 bookmarks
357  return QStringLiteral("internet-web-browser");
358  }
359  // end migration code
360 
361  if (icon == QLatin1String("bookmark_folder")) {
362  return QStringLiteral("folder-bookmarks");
363  }
364  if (icon.isEmpty()) {
365  // Default icon depends on URL for bookmarks, and is default directory
366  // icon for groups.
367  if (isGroup()) {
368  icon = QStringLiteral("folder-bookmarks");
369  } else {
370  if (isSeparator()) {
371  icon = QStringLiteral("edit-clear"); // whatever
372  } else {
373  // get icon from mimeType
374  QMimeDatabase db;
375  QMimeType mime;
376  QString _mimeType = mimeType();
377  if (!_mimeType.isEmpty()) {
378  mime = db.mimeTypeForName(_mimeType);
379  } else {
380  mime = db.mimeTypeForUrl(url());
381  }
382  if (mime.isValid()) {
383  icon = mime.iconName();
384  }
385  }
386  }
387  }
388  return icon;
389 }

509 QString KBookmark::commonParent(const QString &first, const QString &second)
510 {
511  QString A = first;
512  QString B = second;
513  QString error(QStringLiteral("ERROR"));
514  if (A == error || B == error) {
515  return error;
516  }
517 
518  A += '/';
519  B += '/';
520 
521  uint lastCommonSlash = 0;
522  uint lastPos = A.length() < B.length() ? A.length() : B.length();
523  for (uint i = 0; i < lastPos; ++i) {
524  if (A[i] != B[i]) {
525  return A.left(lastCommonSlash);
526  }
527  if (A[i] == '/') {
528  lastCommonSlash = i;
529  }
530  }
531  return A.left(lastCommonSlash);
532 }

621 void KBookmarkGroupTraverser::traverse(const KBookmarkGroup &root)
622 {
623  QStack<KBookmarkGroup> stack;
624  stack.push(root);
625  KBookmark bk = root.first();
626  for (;;) {
627  if (bk.isNull()) {
628  if (stack.count() == 1) { // only root is on the stack
629  return;
630  }
631  if (stack.count() > 0) {
632  visitLeave(stack.top());
633  bk = stack.pop();
634  }
635  bk = stack.top().next(bk);
636  } else if (bk.isGroup()) {
637  KBookmarkGroup gp = bk.toGroup();
638  visitEnter(gp);
639  bk = gp.first();
640  stack.push(gp);
641  } else {
642  visit(bk);
643  bk = stack.top().next(bk);
644  }
645  }
646 }

703  QByteArray payload = mimeData->data(QStringLiteral("application/x-xbel"));
704  if (!payload.isEmpty()) {
705  doc.setContent(payload);
706  QDomElement elem = doc.documentElement();
707  const QDomNodeList children = elem.childNodes();
708  for (int childno = 0; childno < children.count(); childno++) {
709  bookmarks.append(KBookmark(children.item(childno).toElement()));
710  }
711  return bookmarks;
712  }

717  }
