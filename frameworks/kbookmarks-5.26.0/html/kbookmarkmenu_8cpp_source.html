212 void KBookmarkMenu::slotBookmarksChanged(const QString &groupAddress)

342 void KBookmarkMenu::fillBookmarks()
343 {
344  KBookmarkGroup parentBookmark = m_pManager->findByAddress(m_parentAddress).toGroup();
345  Q_ASSERT(!parentBookmark.isNull());
346 
347  if (m_bIsRoot && !parentBookmark.first().isNull()) { // at least one bookmark
348  m_parentMenu->addSeparator();
349  }
350 
351  for (KBookmark bm = parentBookmark.first(); !bm.isNull(); bm = parentBookmark.next(bm)) {
352  m_parentMenu->addAction(actionForBookmark(bm));
353  }
354 }

356 QAction *KBookmarkMenu::actionForBookmark(const KBookmark &bm)
357 {
358  if (bm.isGroup()) {
359  //qDebug() << "Creating bookmark submenu named " << bm.text();
360  KActionMenu *actionMenu = new KBookmarkActionMenu(bm, this);
361  m_actions.append(actionMenu);
362  KBookmarkMenu *subMenu = new KBookmarkMenu(m_pManager, m_pOwner, actionMenu->menu(), bm.address());
363  m_lstSubMenus.append(subMenu);
364  return actionMenu;
365  } else if (bm.isSeparator()) {
366  QAction *sa = new QAction(this);
367  sa->setSeparator(true);
368  m_actions.append(sa);
369  return sa;
370  } else {
371  //qDebug() << "Creating bookmark menu item for " << bm.text();
372  QAction *action = new KBookmarkAction(bm, m_pOwner, this);
373  m_actions.append(action);
374  return action;
375  }

391 void KBookmarkMenu::slotAddBookmark()
392 {
393  if (!m_pOwner) {
394  return;
395  }
396  if (m_pOwner->currentTitle().isEmpty() && m_pOwner->currentUrl().isEmpty()) {
397  return;
398  }
399  KBookmarkGroup parentBookmark = m_pManager->findByAddress(m_parentAddress).toGroup();
400 
401  if (KBookmarkSettings::self()->m_advancedaddbookmark) {
402  KBookmarkDialog *dlg = m_pOwner->bookmarkDialog(m_pManager, QApplication::activeWindow());
403  dlg->addBookmark(m_pOwner->currentTitle(), m_pOwner->currentUrl(), m_pOwner->currentIcon(), parentBookmark);
404  delete dlg;
405  } else {
406  parentBookmark.addBookmark(m_pOwner->currentTitle(), m_pOwner->currentUrl(), m_pOwner->currentIcon());
407  m_pManager->emitChanged(parentBookmark);
408  }
409 
410 }
