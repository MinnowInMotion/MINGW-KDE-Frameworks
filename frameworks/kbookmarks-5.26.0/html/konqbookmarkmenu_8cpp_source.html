49 void KonqBookmarkContextMenu::addActions()
50 {
51  KConfigGroup config = KSharedConfig::openConfig(QStringLiteral("kbookmarkrc"), KConfig::NoGlobals)->group("Bookmarks");
52  bool filteredToolbar = config.readEntry("FilteredToolbar", false);
53 
54  if (bookmark().isGroup()) {
55  addOpenFolderInTabs();
56  addBookmark();
57 
58  if (filteredToolbar) {
59  QString text = bookmark().showInToolbar() ? tr("Hide in toolbar") : tr("Show in toolbar");
60  addAction(QIcon::fromTheme(QLatin1String("")), text, this, SLOT(toggleShowInToolbar()));
61  }
62 
63  addFolderActions();
64  } else {
65  if (owner()) {
66  addAction(QIcon::fromTheme(QStringLiteral("window-new")), tr("Open in New Window"), this, SLOT(openInNewWindow()));
67  addAction(QIcon::fromTheme(QStringLiteral("tab-new")), tr("Open in New Tab"), this, SLOT(openInNewTab()));
68  }
69  addBookmark();
70 
71  if (filteredToolbar) {
72  QString text = bookmark().showInToolbar() ? tr("Hide in toolbar") : tr("Show in toolbar");
73  addAction(QIcon::fromTheme(QLatin1String("")), text, this, SLOT(toggleShowInToolbar()));
74  }
75 
76  addBookmarkActions();
77  }
78 }

100 void KonqBookmarkMenu::fillDynamicBookmarks()
101 {
102  if (isDirty()
103  && KBookmarkManager::userBookmarksManager()->path() == manager()->path()) {
104  bool haveSep = false;
105 
106  const QStringList keys = KonqBookmarkMenu::dynamicBookmarksList();
107  for (QStringList::const_iterator it = keys.begin(); it != keys.end(); ++it) {
108  DynMenuInfo info;
109  info = showDynamicBookmarks((*it));
110 
111  if (!info.show || !QFile::exists(info.location)) {
112  continue;
113  }
114 
115  if (!haveSep) {
116  parentMenu()->addSeparator();
117  haveSep = true;
118  }
119 
120  KActionMenu *actionMenu;
121  actionMenu = new KActionMenu(QIcon::fromTheme(info.type), info.name, this);
122  m_actionCollection->addAction(QStringLiteral("kbookmarkmenu"), actionMenu);
123 
124  parentMenu()->addAction(actionMenu);
125  m_actions.append(actionMenu);
126 
127  KImportedBookmarkMenu *subMenu =
128  new KImportedBookmarkMenu(manager(), owner(), actionMenu->menu(),
129  info.type, info.location);
130  m_lstSubMenus.append(subMenu);
131  }
132  }
133 }

147 QAction *KonqBookmarkMenu::actionForBookmark(const KBookmark &bm)
148 {
149  if (bm.isGroup()) {
150  // qDebug() << "Creating Konq bookmark submenu named " << bm.text();
151  KBookmarkActionMenu *actionMenu = new KBookmarkActionMenu(bm, this);
152  m_actionCollection->addAction(QStringLiteral("kbookmarkmenu"), actionMenu);
153  m_actions.append(actionMenu);
154 
155  KBookmarkMenu *subMenu = new KonqBookmarkMenu(manager(), owner(), actionMenu, bm.address());
156 
157  m_lstSubMenus.append(subMenu);
158  return actionMenu;
159  } else if (bm.isSeparator()) {
160  return KBookmarkMenu::actionForBookmark(bm);
161  } else {
162  // qDebug() << "Creating Konq bookmark action named " << bm.text();
163  KBookmarkAction *action = new KBookmarkAction(bm, owner(), this);
164  m_actionCollection->addAction(action->objectName(), action);
165  m_actions.append(action);
166  return action;
167  }
168 }

170 KonqBookmarkMenu::DynMenuInfo KonqBookmarkMenu::showDynamicBookmarks(const QString &id)
171 {
172  KConfig bookmarkrc(QStringLiteral("kbookmarkrc"), KConfig::NoGlobals);
173  KConfigGroup config(&bookmarkrc, "Bookmarks");
174 
175  DynMenuInfo info;
176  info.show = false;
177  info.d = Q_NULLPTR;
178 
179  if (!config.hasKey("DynamicMenus")) {
180  if (bookmarkrc.hasGroup("DynamicMenu-" + id)) {
181  KConfigGroup dynGroup(&bookmarkrc, "DynamicMenu-" + id);
182  info.show = dynGroup.readEntry("Show", false);
183  info.location = dynGroup.readPathEntry("Location", QString());
184  info.type = dynGroup.readEntry("Type");
185  info.name = dynGroup.readEntry("Name");
186  }
187  }
188  return info;
189 }

203 void KonqBookmarkMenu::setDynamicBookmarks(const QString &id, const DynMenuInfo &newMenu)
204 {
205  KSharedConfig::Ptr kbookmarkrc = KSharedConfig::openConfig(QStringLiteral("kbookmarkrc"), KConfig::NoGlobals);
206  KConfigGroup dynConfig = kbookmarkrc->group(QString("DynamicMenu-" + id));
207 
208  // add group unconditionally
209  dynConfig.writeEntry("Show", newMenu.show);
210  dynConfig.writePathEntry("Location", newMenu.location);
211  dynConfig.writeEntry("Type", newMenu.type);
212  dynConfig.writeEntry("Name", newMenu.name);
213 
214  QStringList elist;
215  KConfigGroup config = kbookmarkrc->group("Bookmarks");
216  if (config.hasKey("DynamicMenus")) {
217  elist = config.readEntry("DynamicMenus", QStringList());
218  }
219 
220  // make sure list includes type
221  if (!elist.contains(id)) {
222  elist << id;
223  config.writeEntry("DynamicMenus", elist);
224  }
225 
226  config.sync();
227 }
