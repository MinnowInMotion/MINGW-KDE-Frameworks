73 void KDialogJobUiDelegate::Private::next()
74 {
75  if (queue.isEmpty()) {
76  running = false;
77  return;
78  }
79 
80  QSharedPointer<MessageBoxData> data = queue.dequeue();
81 
82  //kmessagebox starts a new event loop which can cause this to get deleted
83  //https://bugs.kde.org/show_bug.cgi?id=356321#c16
84  QPointer<KDialogJobUiDelegate::Private> thisGuard(this);
85  KMessageBox::messageBox(data->widget, data->type, data->msg);
86 
87  if (!thisGuard) {
88  return;
89  }
90 
91  QMetaObject::invokeMethod(this, "next", Qt::QueuedConnection);
92 }
93 
94 void KDialogJobUiDelegate::Private::queuedMessageBox(QWidget *widget, KMessageBox::DialogType type, const QString &msg)
95 {
96  QSharedPointer<MessageBoxData> data(new MessageBoxData);
97  data->type = type;
98  data->widget = widget;
99  data->msg = msg;
100 
101  queue.enqueue(data);
102 
103  if (!running) {
104  running = true;
105  QMetaObject::invokeMethod(this, "next", Qt::QueuedConnection);
106  }
107 }
