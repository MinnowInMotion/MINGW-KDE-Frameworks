46 void KStatusBarJobTracker::registerJob(KJob *job)
47 {
48  KAbstractWidgetJobTracker::registerJob(job);
49 
50  if (d->progressWidget.contains(job)) {
51  return;

57  d->progressWidget.insert(job, vi);

60 void KStatusBarJobTracker::unregisterJob(KJob *job)
61 {
62  KAbstractWidgetJobTracker::unregisterJob(job);
63 
64  if (!d->progressWidget.contains(job)) {
65  return;
66  }
67 
68  if (d->currentProgressWidget == d->progressWidget[job]) {
69  d->currentProgressWidget = 0;
70  }
71 
72  if (!d->progressWidget[job]->beingDeleted) {
73  delete d->progressWidget[job];
74  }
75 
76  d->progressWidget.remove(job);
77 }

149 void KStatusBarJobTracker::Private::ProgressWidget::init(KJob *job, QWidget *parent)
150 {
151  widget = new QWidget(parent);
152 
153  int w = fontMetrics().width(QStringLiteral(" 999.9 kB/s 00:00:01 ")) + 8;
154  box = new QHBoxLayout(widget);
155  box->setMargin(0);
156  box->setSpacing(0);
157  widget->setLayout(box);
158 
159  stack = new QStackedWidget(widget);
160  box->addWidget(stack);
161 
162  if (q->d->showStopButton) {
163  button = new QPushButton(QCoreApplication::translate("KStatusBarJobTracker", "Stop"), widget);
164  box->addWidget(button);
165  connect(button, SIGNAL(clicked(bool)),
166  this, SLOT(killJob()));
167  } else {
168  button = 0;
169  }
170 
171  progressBar = new QProgressBar(widget);
172  progressBar->installEventFilter(this);
173  progressBar->setMinimumWidth(w);
174  stack->insertWidget(1, progressBar);
175 
176  label = new QLabel(widget);
177  label->setAlignment(Qt::AlignHCenter | Qt::AlignVCenter);
178  label->installEventFilter(this);
179  label->setMinimumWidth(w);
180  stack->insertWidget(2, label);
181  setMinimumSize(sizeHint());
182 
183  setMode(KStatusBarJobTracker::LabelOnly);
184 
185  q->setAutoDelete(job, true);
186 
187  QVBoxLayout *layout = new QVBoxLayout;
188  layout->addWidget(widget);
189  setLayout(layout);
190 }

192 void KStatusBarJobTracker::Private::ProgressWidget::setMode(StatusBarModes newMode)
193 {
194  mode = newMode;
195 
196  if (newMode == KStatusBarJobTracker::NoInformation) {
197  stack->hide();
198 
199  return;
200  }
201 
202  if (newMode & KStatusBarJobTracker::LabelOnly) {
203  stack->show();
204  stack->setCurrentWidget(label);
205 
206  return; // TODO: we should make possible to show an informative label and the progress bar
207  }
208 
209  if (newMode & KStatusBarJobTracker::ProgressOnly) {
210  stack->show();
211  stack->setCurrentWidget(progressBar);
212  }
213 }

259 bool KStatusBarJobTracker::Private::ProgressWidget::eventFilter(QObject *obj, QEvent *event)
260 {
261  if (obj == progressBar || obj == label) {
262 
263  if (event->type() == QEvent::MouseButtonPress) {
264  QMouseEvent *e = static_cast<QMouseEvent *>(event);
265 
266  // TODO: we should make possible to show an informative label and the progress bar
267  if (e->button() == Qt::LeftButton) { // toggle view on left mouse button
268  if (mode == KStatusBarJobTracker::LabelOnly) {
269  setMode(KStatusBarJobTracker::ProgressOnly);
270  } else if (mode == KStatusBarJobTracker::ProgressOnly) {
271  setMode(KStatusBarJobTracker::LabelOnly);
272  }
273  return true;
274  }
275  }
276 
277  return false;
278  }
279 
280  return QWidget::eventFilter(obj, event);
281 }
