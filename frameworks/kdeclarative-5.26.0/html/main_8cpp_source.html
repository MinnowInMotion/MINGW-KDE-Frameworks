38 int main(int argc, char **argv)
39 {
40  QCommandLineParser parser;
41  QApplication app(argc, argv);
42 
43  const QString description = i18n("KPackage QML application shell");
44  const char version[] = "0.1";
45 
46  app.setApplicationVersion(version);
47  parser.addVersionOption();
48  parser.addHelpOption();
49  parser.setApplicationDescription(description);
50 
51  QCommandLineOption appPluginOption(QCommandLineOption(QStringList() << "a" << "app", i18n("The unique name of the application (mandatory)"), "app"));
52 
53  parser.addOption(appPluginOption);
54 
55  parser.process(app);
56 
57  QString packagePath;
58  if (parser.isSet(appPluginOption)) {
59  packagePath = parser.value(appPluginOption);
60  } else {
61  parser.showHelp(1);
62  }
63 
64  //usually we have an ApplicationWindow here, so we do not need to create a window by ourselves
65  KDeclarative::QmlObject obj;
66  obj.setTranslationDomain(packagePath);
67  obj.setInitializationDelayed(true);
68  obj.loadPackage(packagePath);
69  if (!obj.package().isValid()) {
70  qWarning() << "The specified Application package is not valid.";
71  return 1;
72  }
73  obj.engine()->rootContext()->setContextProperty("commandlineArguments", parser.positionalArguments());
74  QVariantMap paths;
75  paths["desktop"] = QStandardPaths::writableLocation(QStandardPaths::DesktopLocation);
76  paths["documents"] = QStandardPaths::writableLocation(QStandardPaths::DocumentsLocation);
77  paths["music"] = QStandardPaths::writableLocation(QStandardPaths::MusicLocation);
78  paths["movies"] = QStandardPaths::writableLocation(QStandardPaths::MoviesLocation);
79  paths["pictures"] = QStandardPaths::writableLocation(QStandardPaths::PicturesLocation);
80  paths["home"] = QStandardPaths::writableLocation(QStandardPaths::HomeLocation);
81  obj.engine()->rootContext()->setContextProperty("userPaths", paths);
82  obj.completeInitialization();
83 
84  if (!obj.package().metadata().isValid()) {
85  return -1;
86  }
87 
88  KPluginMetaData data = obj.package().metadata();
89  // About data
90  KAboutData aboutData(data.pluginId(), data.name(), data.version(), data.description(), KAboutLicense::byKeyword(data.license()).key());
91 
92  for (auto author : data.authors()) {
93  aboutData.addAuthor(author.name(), author.task(), author.emailAddress(), author.webAddress(), author.ocsUsername());
94  }
95  // We assume that desktop file in applicaitons dir is named similar as
96  // the pluginId, this is required to get icons working on wayland for example
97  aboutData.setDesktopFileName(data.pluginId());
98 
99  KAboutData::setApplicationData(aboutData);
100 
101  //The root is not a window?
102  //have to use a normal QQuickWindow since the root item is already created
103  QQuickItem *item = qobject_cast<QQuickItem *>(obj.rootObject());
104  QWindow *window = qobject_cast<QWindow *>(obj.rootObject());
105  if (item) {
106  QQuickWindow view;
107  item->setParentItem(view.contentItem());
108  if (item->implicitWidth() > 0 && item->implicitHeight() > 0) {
109  view.resize(item->implicitWidth(), item->implicitHeight());
110  } else {
111  view.resize(item->width(), item->height());
112  }
113 
114  //set anchors
115  QQmlExpression expr(obj.engine()->rootContext(), obj.rootObject(), "parent");
116  QQmlProperty prop(obj.rootObject(), "anchors.fill");
117  prop.write(expr.evaluate());
118  view.setTitle(obj.package().metadata().name());
119  view.setIcon(QIcon::fromTheme(obj.package().metadata().iconName()));
120 
121  // Forward Qt.quit() in QML code to QApplication::quit()
122  QObject::connect(obj.engine(), &QQmlEngine::quit, &app, &QCoreApplication::quit);
123 
124  view.show();
125  return app.exec();
126  } else if (window) {
127  window->setTitle(obj.package().metadata().name());
128  window->setIcon(QIcon::fromTheme(obj.package().metadata().iconName()));
129  } else {
130  qWarning() << "The root Qml Item should be either a kind of window or a QQuickItem";
131  return 1;
132  }
133 
134  return app.exec();
135 }
