38 void KWindowSystemProxy::connectNotify(const QMetaMethod &signal)
39 {
40  if (m_initialized) {
41  return;
42  }
43 
44  if (signal == QMetaMethod::fromSignal(&KWindowSystemProxy::currentDesktopChanged) ||
45  signal == QMetaMethod::fromSignal(&KWindowSystemProxy::desktopNamesChanged) ||
46  signal == QMetaMethod::fromSignal(&KWindowSystemProxy::numberOfDesktopsChanged) ||
47  signal == QMetaMethod::fromSignal(&KWindowSystemProxy::showingDesktopChanged) ||
48  signal == QMetaMethod::fromSignal(&KWindowSystemProxy::compositingActiveChanged)) {
49 
50  connect(KWindowSystem::self(), &KWindowSystem::currentDesktopChanged, this, [this](int currentDesktop) {
51  Q_EMIT currentDesktopChanged(currentDesktop);
52  Q_EMIT currentDesktopNameChanged();
53  });
54 
55  connect(KWindowSystem::self(), &KWindowSystem::desktopNamesChanged, this, [this]() {
56  Q_EMIT desktopNamesChanged();
57  Q_EMIT currentDesktopNameChanged();
58  });
59 
60  connect(KWindowSystem::self(), &KWindowSystem::numberOfDesktopsChanged,
61  this, &KWindowSystemProxy::numberOfDesktopsChanged);
62  connect(KWindowSystem::self(), &KWindowSystem::showingDesktopChanged,
63  this, &KWindowSystemProxy::showingDesktopChanged);
64  connect(KWindowSystem::self(), &KWindowSystem::compositingChanged,
65  this, &KWindowSystemProxy::compositingActiveChanged);
66 
67  m_initialized = true;
68  }
69 }
