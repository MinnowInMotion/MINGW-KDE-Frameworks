57 QVariant Clipboard::contentFormat(const QString &format) const
58 {
59  const QMimeData* data = m_clipboard->mimeData(m_mode);
60  QVariant ret;
61  if(format == QStringLiteral("text/uri-list")) {
62  QVariantList retList;
63  foreach(const QUrl& url, data->urls())
64  retList += url;
65  ret = retList;
66  } else if(format.startsWith(QStringLiteral("text/"))) {
67  ret = data->text();
68  } else if(format.startsWith(QStringLiteral("image/"))) {
69  ret = data->imageData();
70  } else
71  ret = data->data(format.isEmpty() ? data->formats().first(): format);
72 
73  return ret;
74 }

81 void Clipboard::setContent(const QVariant &content)
82 {
83  QMimeData* mimeData = new QMimeData;
84  switch(content.type())
85  {
86  case QVariant::String:
87  mimeData->setText(content.toString());
88  break;
89  case QVariant::Color:
90  mimeData->setColorData(content.toString());
91  break;
92  case QVariant::Pixmap:
93  case QVariant::Image:
94  mimeData->setImageData(content);
95  break;
96  default:
97  if (content.type() == QVariant::List) {
98  QVariantList list = content.toList();
99  QList<QUrl> urls;
100  bool wasUrlList = true;
101  foreach (const QVariant& url, list) {
102  if (url.type() != QVariant::Url) {
103  wasUrlList = false;
104  break;
105  }
106  urls += url.toUrl();
107  }
108  if(wasUrlList) {
109  mimeData->setUrls(urls);
110  break;
111  }
112  }
113 
114  if (content.canConvert(QVariant::String)) {
115  mimeData->setText(content.toString());
116  } else {
117  mimeData->setData("application/octet-stream", content.toByteArray());
118  qWarning() << "Couldn't figure out the content type, storing as application/octet-stream";
119  }
120  break;
121  }
122  m_clipboard->setMimeData(mimeData, m_mode);
123 }
