28 void ColumnProxyModel::setSourceModel(QAbstractItemModel* sourceModel)
29 {
30  if(sourceModel==m_sourceModel) {
31  return;
32  }
33 
34  beginResetModel();
35  if(m_sourceModel) {
36  disconnect(m_sourceModel, SIGNAL(destroyed(QObject*)),
37  this, SLOT(sourceDestroyed(QObject*)));
38 
39  disconnect(m_sourceModel, SIGNAL(dataChanged(QModelIndex,QModelIndex)),
40  this, SLOT(considerDataChanged(QModelIndex,QModelIndex)));
41  disconnect(m_sourceModel, SIGNAL(rowsAboutToBeInserted(QModelIndex,int,int)),
42  this, SLOT(considerRowsAboutToBeInserted(QModelIndex,int,int)));
43  disconnect(m_sourceModel, SIGNAL(rowsAboutToBeMoved(QModelIndex,int,int,QModelIndex,int)),
44  this, SLOT(considerRowsAboutToBeMoved(QModelIndex,int,int,QModelIndex,int)));
45  disconnect(m_sourceModel, SIGNAL(rowsAboutToBeRemoved(QModelIndex,int,int)),
46  this, SLOT(considerRowsAboutToBeRemoved(QModelIndex,int,int)));
47  disconnect(m_sourceModel, SIGNAL(rowsInserted(QModelIndex,int,int)),
48  this, SLOT(considerRowsInserted(QModelIndex,int,int)));
49  disconnect(m_sourceModel, SIGNAL(rowsMoved(QModelIndex,int,int,QModelIndex,int)),
50  this, SLOT(considerRowsMoved(QModelIndex,int,int,QModelIndex,int)));
51  disconnect(m_sourceModel, SIGNAL(rowsRemoved(QModelIndex,int,int)),
52  this, SLOT(considerRowsRemoved(QModelIndex,int,int)));
53 
54  disconnect(m_sourceModel, SIGNAL(modelAboutToBeReset()),
55  this, SIGNAL(modelAboutToBeReset()));
56  disconnect(m_sourceModel, SIGNAL(modelReset()),
57  this, SIGNAL(modelReset()));
58  disconnect(m_sourceModel, SIGNAL(headerDataChanged(Qt::Orientation,int,int)),
59  this, SIGNAL(headerDataChanged(Qt::Orientation,int,int)));
60  disconnect(m_sourceModel, SIGNAL(layoutAboutToBeChanged()),
61  this, SIGNAL(layoutAboutToBeChanged()));
62  disconnect(m_sourceModel, SIGNAL(layoutChanged()),
63  this, SIGNAL(layoutChanged()));
64  }
65  m_sourceModel = sourceModel;
66  if(m_sourceModel) {
67  connect(m_sourceModel, SIGNAL(destroyed(QObject*)),
68  this, SLOT(sourceDestroyed(QObject*)));
69 
70  connect(m_sourceModel, SIGNAL(dataChanged(QModelIndex,QModelIndex)),
71  this, SLOT(considerDataChanged(QModelIndex,QModelIndex)));
72  connect(m_sourceModel, SIGNAL(rowsAboutToBeInserted(QModelIndex,int,int)),
73  this, SLOT(considerRowsAboutToBeInserted(QModelIndex,int,int)));
74  connect(m_sourceModel, SIGNAL(rowsAboutToBeMoved(QModelIndex,int,int,QModelIndex,int)),
75  this, SLOT(considerRowsAboutToBeMoved(QModelIndex,int,int,QModelIndex,int)));
76  connect(m_sourceModel, SIGNAL(rowsAboutToBeRemoved(QModelIndex,int,int)),
77  this, SLOT(considerRowsAboutToBeRemoved(QModelIndex,int,int)));
78  connect(m_sourceModel, SIGNAL(rowsInserted(QModelIndex,int,int)),
79  this, SLOT(considerRowsInserted(QModelIndex,int,int)));
80  connect(m_sourceModel, SIGNAL(rowsMoved(QModelIndex,int,int,QModelIndex,int)),
81  this, SLOT(considerRowsMoved(QModelIndex,int,int,QModelIndex,int)));
82  connect(m_sourceModel, SIGNAL(rowsRemoved(QModelIndex,int,int)),
83  this, SLOT(considerRowsRemoved(QModelIndex,int,int)));
84 
85  connect(m_sourceModel, SIGNAL(modelAboutToBeReset()),
86  this, SIGNAL(modelAboutToBeReset()));
87  connect(m_sourceModel, SIGNAL(modelReset()),
88  this, SIGNAL(modelReset()));
89  connect(m_sourceModel, SIGNAL(headerDataChanged(Qt::Orientation,int,int)),
90  this, SIGNAL(headerDataChanged(Qt::Orientation,int,int)));
91  connect(m_sourceModel, SIGNAL(layoutAboutToBeChanged()),
92  this, SIGNAL(layoutAboutToBeChanged()));
93  connect(m_sourceModel, SIGNAL(layoutChanged()),
94  this, SIGNAL(layoutChanged()));
95  }
96  endResetModel();
97 }

116 void ColumnProxyModel::setRootIndex(const QModelIndex& index)
117 {
118  if (index == m_index) {
119  return;
120  }
121 
122  if(index.isValid()) {
123  setSourceModel(const_cast<QAbstractItemModel*>(index.model()));
124  }
125  beginResetModel();
126  m_index = index;
127  endResetModel();
128 
129  emit rootIndexChanged();
130 }

195 void ColumnProxyModel::considerRowsAboutToBeMoved(const QModelIndex &sourceParent, int rA, int rB, const QModelIndex& destParent, int rD)
196 {
197  if(sourceParent==m_index && destParent==m_index) {
198  beginMoveRows(QModelIndex(), rA, rB, QModelIndex(), rD);
199  } else if(sourceParent==m_index) {
200  beginRemoveRows(sourceParent, rA, rB);
201  } else if(destParent==m_index) {
202  beginInsertRows(destParent, rD, rD+(rB-rA));
203  }
204 }
