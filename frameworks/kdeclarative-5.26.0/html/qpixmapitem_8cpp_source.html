106 void QPixmapItem::paint(QPainter *painter)
107 {
108  if (m_pixmap.isNull()) {
109  return;
110  }
111  painter->save();
112  painter->setRenderHint(QPainter::Antialiasing, m_smooth);
113  painter->setRenderHint(QPainter::SmoothPixmapTransform, m_smooth);
114 
115  if (m_fillMode == TileVertically) {
116  painter->scale(width()/(qreal)m_pixmap.width(), 1);
117  }
118 
119  if (m_fillMode == TileHorizontally) {
120  painter->scale(1, height()/(qreal)m_pixmap.height());
121  }
122 
123  if (m_fillMode >= Tile) {
124  painter->drawTiledPixmap(m_paintedRect, m_pixmap);
125  } else {
126  painter->drawPixmap(m_paintedRect, m_pixmap, m_pixmap.rect());
127  }
128 
129  painter->restore();
130 }

156 void QPixmapItem::updatePaintedRect()
157 {
158 
159  if (m_pixmap.isNull()) {
160  return;
161  }
162 
163  QRect sourceRect = m_paintedRect;
164 
165  QRect destRect;
166 
167  switch (m_fillMode) {
168  case PreserveAspectFit: {
169  QSize scaled = m_pixmap.size();
170 
171  scaled.scale(boundingRect().size().toSize(), Qt::KeepAspectRatio);
172  destRect = QRect(QPoint(0, 0), scaled);
173  destRect.moveCenter(boundingRect().center().toPoint());
174  break;
175  }
176  case PreserveAspectCrop: {
177  QSize scaled = m_pixmap.size();
178 
179  scaled.scale(boundingRect().size().toSize(), Qt::KeepAspectRatioByExpanding);
180  destRect = QRect(QPoint(0, 0), scaled);
181  destRect.moveCenter(boundingRect().center().toPoint());
182  break;
183  }
184  case TileVertically: {
185  destRect = boundingRect().toRect();
186  destRect.setWidth(destRect.width() / (width()/(qreal)m_pixmap.width()));
187  break;
188  }
189  case TileHorizontally: {
190  destRect = boundingRect().toRect();
191  destRect.setHeight(destRect.height() / (height()/(qreal)m_pixmap.height()));
192  break;
193  }
194  case Stretch:
195  case Tile:
196  default:
197  destRect = boundingRect().toRect();
198  }
199 
200  if (destRect != sourceRect) {
201  m_paintedRect = destRect;
202  emit paintedHeightChanged();
203  emit paintedWidthChanged();
204  }
205 }
