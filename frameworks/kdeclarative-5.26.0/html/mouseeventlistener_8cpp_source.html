144 void MouseEventListener::mousePressEvent(QMouseEvent *me)
145 {
146  if (m_lastEvent == me || !(me->buttons() & m_acceptedButtons)) {
147  me->setAccepted(false);
148  return;
149  }
150 
151  //FIXME: when a popup window is visible: a click anywhere hides it: but the old qquickitem will continue to think it's under the mouse
152  //doesn't seem to be any good way to properly reset this.
153  //this msolution will still caused a missed click after the popup is gone, but gets the situation unblocked.
154  QPoint viewPosition;
155  if (window()) {
156  viewPosition = window()->position();
157  }
158 
159  if (!QRectF(mapToScene(QPoint(0, 0)) + viewPosition, QSizeF(width(), height())).contains(me->screenPos())) {
160  me->ignore();
161  return;
162  }
163  m_buttonDownPos = me->screenPos();
164 
165  KDeclarativeMouseEvent dme(me->pos().x(), me->pos().y(), me->screenPos().x(), me->screenPos().y(), me->button(), me->buttons(), me->modifiers(), screenForGlobalPos(me->globalPos()));
166  if (!m_pressAndHoldEvent) {
167  m_pressAndHoldEvent = new KDeclarativeMouseEvent(me->pos().x(), me->pos().y(), me->screenPos().x(), me->screenPos().y(), me->button(), me->buttons(), me->modifiers(), screenForGlobalPos(me->globalPos()));
168  }
169 
170  emit pressed(&dme);
171  m_pressed = true;
172  emit pressedChanged();
173 
174 #if (QT_VERSION >= QT_VERSION_CHECK(5, 3, 0))
175  m_pressAndHoldTimer->start(QGuiApplication::styleHints()->mousePressAndHoldInterval());
176 #else
177  m_pressAndHoldTimer->start(1000);
178 #endif
179 }

234 bool MouseEventListener::childMouseEventFilter(QQuickItem *item, QEvent *event)
235 {
236  if (!isEnabled()) {
237  return false;
238  }
239 
240  //don't filter other mouseeventlisteners
241  if (qobject_cast<MouseEventListener *>(item)) {
242  return false;
243  }
244 
245  switch (event->type()) {
246  case QEvent::MouseButtonPress: {
247  m_lastEvent = event;
248  QMouseEvent *me = static_cast<QMouseEvent *>(event);
249 
250  if (!(me->buttons() & m_acceptedButtons)) {
251  break;
252  }
253 
254  //the parent will receive events in its own coordinates
255  const QPointF myPos = mapFromScene(me->windowPos());
256 
257  KDeclarativeMouseEvent dme(myPos.x(), myPos.y(), me->screenPos().x(), me->screenPos().y(), me->button(), me->buttons(), me->modifiers(), screenForGlobalPos(me->globalPos()));
258  delete m_pressAndHoldEvent;
259  m_pressAndHoldEvent = new KDeclarativeMouseEvent(myPos.x(), myPos.y(), me->screenPos().x(), me->screenPos().y(), me->button(), me->buttons(), me->modifiers(), screenForGlobalPos(me->globalPos()));
260 
261  //qDebug() << "pressed in sceneEventFilter";
262  m_buttonDownPos = me->screenPos();
263  emit pressed(&dme);
264  m_pressed = true;
265  emit pressedChanged();
266 
267 #if (QT_VERSION >= QT_VERSION_CHECK(5, 3, 0))
268  m_pressAndHoldTimer->start(QGuiApplication::styleHints()->mousePressAndHoldInterval());
269 #else
270  m_pressAndHoldTimer->start(1000);
271 #endif
272  break;
273  }
274  case QEvent::HoverMove: {
275  if (!acceptHoverEvents()) {
276  break;
277  }
278  m_lastEvent = event;
279  QHoverEvent *he = static_cast<QHoverEvent *>(event);
280  const QPointF myPos = item->mapToItem(this, he->pos());
281 
282  QQuickWindow *w = window();
283  QPoint screenPos;
284  if (w) {
285  screenPos = w->mapToGlobal(myPos.toPoint());
286  }
287 
288  KDeclarativeMouseEvent dme(myPos.x(), myPos.y(), screenPos.x(), screenPos.y(), Qt::NoButton, Qt::NoButton, he->modifiers(), 0);
289  //qDebug() << "positionChanged..." << dme.x() << dme.y();
290  emit positionChanged(&dme);
291  break;
292  }
293  case QEvent::MouseMove: {
294  m_lastEvent = event;
295  QMouseEvent *me = static_cast<QMouseEvent *>(event);
296  if (!(me->buttons() & m_acceptedButtons)) {
297  break;
298  }
299 
300  const QPointF myPos = mapFromScene(me->windowPos());
301  KDeclarativeMouseEvent dme(myPos.x(), myPos.y(), me->screenPos().x(), me->screenPos().y(), me->button(), me->buttons(), me->modifiers(), screenForGlobalPos(me->globalPos()));
302  //qDebug() << "positionChanged..." << dme.x() << dme.y();
303 
304  //stop the pressandhold if mouse moved enough
305  if (QPointF(me->screenPos() - m_buttonDownPos).manhattanLength() > QGuiApplication::styleHints()->startDragDistance() && m_pressAndHoldTimer->isActive()) {
306  m_pressAndHoldTimer->stop();
307 
308  //if the mouse moves and we are waiting to emit a press and hold event, update the co-ordinates
309  //as there is no update function, delete the old event and create a new one
310  } else if (m_pressAndHoldEvent) {
311  delete m_pressAndHoldEvent;
312  m_pressAndHoldEvent = new KDeclarativeMouseEvent(myPos.x(), myPos.y(), me->screenPos().x(), me->screenPos().y(), me->button(), me->buttons(), me->modifiers(), screenForGlobalPos(me->globalPos()));
313  }
314  emit positionChanged(&dme);
315  break;
316  }
317  case QEvent::MouseButtonRelease: {
318  m_lastEvent = event;
319  QMouseEvent *me = static_cast<QMouseEvent *>(event);
320 
321  const QPointF myPos = mapFromScene(me->windowPos());
322  KDeclarativeMouseEvent dme(myPos.x(), myPos.y(), me->screenPos().x(), me->screenPos().y(), me->button(), me->buttons(), me->modifiers(), screenForGlobalPos(me->globalPos()));
323  m_pressed = false;
324 
325  emit released(&dme);
326  emit pressedChanged();
327 
328  if (QPointF(me->screenPos() - m_buttonDownPos).manhattanLength() <= QGuiApplication::styleHints()->startDragDistance() && m_pressAndHoldTimer->isActive()) {
329  emit clicked(&dme);
330  m_pressAndHoldTimer->stop();
331  }
332  break;
333  }
334  case QEvent::UngrabMouse: {
335  m_lastEvent = event;
336  break;
337  }
338  case QEvent::Wheel: {
339  m_lastEvent = event;
340  QWheelEvent *we = static_cast<QWheelEvent *>(event);
341  KDeclarativeWheelEvent dwe(we->pos(), we->globalPos(), we->delta(), we->buttons(), we->modifiers(), we->orientation());
342  emit wheelMoved(&dwe);
343  break;
344  }
345  default:
346  break;
347  }
348 
349  return QQuickItem::childMouseEventFilter(item, event);
350 // return false;
351 }
