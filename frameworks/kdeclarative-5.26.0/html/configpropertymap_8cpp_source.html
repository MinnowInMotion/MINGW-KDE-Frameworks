84 void ConfigPropertyMapPrivate::loadConfig()
85 {
86  if (!config) {
87  return;
88  }
89 
90  foreach (KConfigSkeletonItem *item, config.data()->items()) {
91  q->insert(item->key(), item->property());
92  emit q->valueChanged(item->key(), item->property());
93  }
94 }
95 
96 void ConfigPropertyMapPrivate::writeConfig()
97 {
98  if (!config) {
99  return;
100  }
101 
102  foreach (KConfigSkeletonItem *item, config.data()->items()) {
103  item->setProperty(q->value(item->key()));
104  }
105 
106  config.data()->blockSignals(true);
107  config.data()->save();
108  config.data()->blockSignals(false);
109 }
110 
111 void ConfigPropertyMapPrivate::writeConfigValue(const QString &key, const QVariant &value)
112 {
113  KConfigSkeletonItem *item = config.data()->findItem(key);
114  if (item) {
115  item->setProperty(value);
116  config.data()->blockSignals(true);
117  config.data()->save();
118  //why read? read will update KConfigSkeletonItem::mLoadedValue,
119  //allowing a write operation to be performed next time
120  config.data()->read();
121  config.data()->blockSignals(false);
122  }
123 }
