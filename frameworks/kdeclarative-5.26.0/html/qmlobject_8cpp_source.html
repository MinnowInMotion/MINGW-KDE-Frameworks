100 void QmlObjectPrivate::errorPrint(QQmlComponent *component)
101 {
102  QString errorStr = QStringLiteral("Error loading QML file.\n");
103  if (component->isError()) {
104  QList<QQmlError> errors = component->errors();
105  foreach (const QQmlError &error, errors) {
106  errorStr += (error.line() > 0 ? QString(QString::number(error.line()) + QLatin1String(": ")) : QLatin1String(""))
107  + error.description() + QLatin1Char('\n');
108  }
109  }
110  qWarning() << component->url().toString() << '\n' << errorStr;
111 }

113 void QmlObjectPrivate::execute(const QUrl &source)
114 {
115  if (source.isEmpty()) {
116  qWarning() << "File name empty!";
117  return;
118  }
119 
120  delete component;
121  component = new QQmlComponent(engine, q);
122  QObject::connect(component, &QQmlComponent::statusChanged,
123  q, &QmlObject::statusChanged, Qt::QueuedConnection);
124  delete incubator.object();
125 
126  component->loadUrl(source);
127 
128  if (delay) {
129  executionEndTimer->start(0);
130  } else {
131  scheduleExecutionEnd();
132  }
133 }

135 void QmlObjectPrivate::scheduleExecutionEnd()
136 {
137  if (component->isReady() || component->isError()) {
138  q->completeInitialization();
139  } else {
140  QObject::connect(component, SIGNAL(statusChanged(QQmlComponent::Status)), q, SLOT(completeInitialization()));
141  }

290 void QmlObjectPrivate::checkInitializationCompleted()
291 {
292  if (!incubator.isReady() && incubator.status() != QQmlIncubator::Error) {
293  QTimer::singleShot(0, q, SLOT(checkInitializationCompleted()));
294  return;
295  }
296 
297  if (!incubator.object()) {
298  errorPrint(component);
299  }
300 
301  emit q->finished();
302 }

304 void QmlObject::completeInitialization(const QVariantHash &initialProperties)
305 {
306  d->executionEndTimer->stop();
307  if (d->incubator.object()) {
308  return;
309  }
310 
311  if (!d->component) {
312  qWarning() << "No component for" << source();
313  return;
314  }
315 
316 
317  if (d->component->status() != QQmlComponent::Ready || d->component->isError()) {
318  d->errorPrint(d->component);
319  return;
320  }
321 
322  d->incubator.m_initialProperties = initialProperties;
323  d->component->create(d->incubator, d->rootContext);
324 
325  if (d->delay) {
326  d->checkInitializationCompleted();
327  } else {
328  d->incubator.forceCompletion();
329 
330  if (!d->incubator.object()) {
331  d->errorPrint(d->component);
332  }
333  emit finished();
334  }
335 }

345 QObject *QmlObject::createObjectFromComponent(QQmlComponent *component, QQmlContext *context, const QVariantHash &initialProperties)
346 {
347  QmlObjectIncubator incubator;
348  incubator.m_initialProperties = initialProperties;
349  component->create(incubator, context ? context : d->rootContext);
350  incubator.forceCompletion();
351 
352  QObject *object = incubator.object();
353 
354  if (!component->isError() && object) {
355  //memory management
356  component->setParent(object);
357  //reparent to root object if wasn't specified otherwise by initialProperties
358  if (!initialProperties.contains("parent")) {
359  if (qobject_cast<QQuickItem *>(rootObject())) {
360  object->setProperty("parent", QVariant::fromValue(rootObject()));
361  } else {
362  object->setParent(rootObject());
363  }
364  }
365 
366  return object;
367 
368  } else {
369  d->errorPrint(component);
370  delete object;
371  return 0;
372  }
373 }
