115 ConfigModule *ConfigModule::qmlAttachedProperties(QObject *object)
116 {
117  //at the moment of the attached object creation, the root item is the only one that hasn't a parent

126 QQuickItem *ConfigModule::mainUi()
127 {
128  if (d->_qmlObject) {
129  return qobject_cast<QQuickItem *>(d->_qmlObject->rootObject());
130  }
131  d->_qmlObject = new KDeclarative::QmlObject(this);
132  ConfigModulePrivate::s_rootObjects[d->_qmlObject->engine()] = this;
133  d->_qmlObject->setTranslationDomain(aboutData()->componentName());
134  d->_qmlObject->setInitializationDelayed(true);
135 
136  KPackage::Package package = KPackage::PackageLoader::self()->loadPackage("KPackage/GenericQML");
137  package.setDefaultPackageRoot("kpackage/kcms");
138  package.setPath(aboutData()->componentName());
139 
140  if (!package.filePath("mainscript").isEmpty()) {
141  d->_qmlObject->setSource(QUrl::fromLocalFile(package.filePath("mainscript")));
142  d->_qmlObject->engine()->rootContext()->setContextProperty("kcm", this);
143  d->_qmlObject->completeInitialization();
144 
145  return qobject_cast<QQuickItem *>(d->_qmlObject->rootObject());
146  } else {
147  qWarning() << "Error loading the module" << aboutData()->componentName() << ": no QML file provided";
148  return 0;
149  }
150 }

167 void ConfigModule::setNeedsAuthorization(bool needsAuth)
168 {
169  if (d->_needsAuthorization == needsAuth) {
170  return;
171  }
172 
173  d->_needsAuthorization = needsAuth;
174  if (needsAuth && d->_about) {
175  d->_authActionName = QString("org.kde.kcontrol." + d->_about->componentName() + ".save");
176  d->_needsAuthorization = true;
177 
178  } else {
179  d->_authActionName = QString();
180  }
181 
182  emit needsAuthorizationChanged();
183  emit authActionNameChanged();
184 }
