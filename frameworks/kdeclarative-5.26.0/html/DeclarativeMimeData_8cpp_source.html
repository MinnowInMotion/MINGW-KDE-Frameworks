50 DeclarativeMimeData::DeclarativeMimeData(const QMimeData* copy)
51  : QMimeData(),
52  m_source(0)
53 {
54  // Copy the standard MIME data
55  foreach(QString format, copy->formats()) {
56  QMimeData::setData(format, copy->data(format));
57  }
58 
59  // If the object we are copying actually is a DeclarativeMimeData, copy our extended properties as well
60  const DeclarativeMimeData* declarativeMimeData = qobject_cast<const DeclarativeMimeData*>(copy);
61  if (declarativeMimeData) {
62  this->setSource(declarativeMimeData->source());
63  }
64 }
