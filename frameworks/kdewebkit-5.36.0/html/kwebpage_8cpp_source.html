77 static bool isMimeTypeAssociatedWithSelf(const KService::Ptr &offer)
78 {
79  if (!offer) {
80  return false;
81  }
82 
83  // qDebug() << offer->desktopEntryName();
84 
85  const QString &appName = QCoreApplication::applicationName();
86 
87  if (appName == offer->desktopEntryName() || offer->exec().trimmed().startsWith(appName)) {
88  return true;
89  }
90 
91  // konqueror exception since it uses kfmclient to open html content...
92  if (appName == QL1S("konqueror") && offer->exec().trimmed().startsWith(QL1S("kfmclient"))) {
93  return true;
94  }
95 
96  return false;
97 }

99 static void extractMimeType(const QNetworkReply *reply, QString &mimeType)
100 {
101  mimeType.clear();
102  const KIO::MetaData &metaData = reply->attribute(static_cast<QNetworkRequest::Attribute>(KIO::AccessManager::MetaData)).toMap();
103  if (metaData.contains(QL1S("content-type"))) {
104  mimeType = metaData.value(QL1S("content-type"));
105  }
106 
107  if (!mimeType.isEmpty()) {
108  return;
109  }
110 
111  if (!reply->hasRawHeader("Content-Type")) {
112  return;
113  }
114 
115  const QString value(QL1S(reply->rawHeader("Content-Type").simplified().constData()));
116  const int index = value.indexOf(QL1C(';'));
117  mimeType = ((index == -1) ? value : value.left(index));
118 }

120 static bool downloadResource(const QUrl &srcUrl, const QString &suggestedName = QString(),
121  QWidget *parent = nullptr, const KIO::MetaData &metaData = KIO::MetaData())
122 {
123  const QString fileName = suggestedName.isEmpty() ? srcUrl.fileName() : suggestedName;
124  // convert filename to URL using fromPath to avoid trouble with ':' in filenames (#184202)
125  QUrl destUrl = QUrl::fromLocalFile(QFileDialog::getSaveFileName(parent, QString(), fileName));
126  if (!destUrl.isValid()) {
127  return false;
128  }
129 
130  // Using KIO::copy rather than file_copy, to benefit from "dest already exists" dialogs.
131  KIO::Job *job = KIO::copy(srcUrl, destUrl);
132 
133  if (!metaData.isEmpty()) {
134  job->setMetaData(metaData);
135  }
136 
137  job->addMetaData(QL1S("MaxCacheSize"), QL1S("0")); // Don't store in http cache.
138  job->addMetaData(QL1S("cache"), QL1S("cache")); // Use entry from cache if available.
139  KJobWidgets::setWindow(job, parent ? parent->window() : nullptr);
140  job->ui()->setAutoErrorHandlingEnabled(true);
141  return true;
142 }

144 static bool isReplyStatusOk(const QNetworkReply *reply)
145 {
146  if (!reply || reply->error() != QNetworkReply::NoError) {
147  return false;
148  }
149 
150  // Check HTTP status code only for http and webdav protocols...
151  const QString scheme = reply->url().scheme();
152  if (scheme.startsWith(QLatin1String("http"), Qt::CaseInsensitive) ||
153  scheme.startsWith(QLatin1String("webdav"), Qt::CaseInsensitive)) {
154  bool ok = false;
155  const int statusCode = reply->attribute(QNetworkRequest::HttpStatusCodeAttribute).toInt(&ok);
156  if (!ok || statusCode < 200 || statusCode > 299) {
157  return false;
158  }
159  }
160 
161  return true;
162 }

353 void KWebPage::downloadResponse(QNetworkReply *reply)
354 {
355  Q_ASSERT(reply);
356 
357  if (!reply) {
358  return;
359  }
360 
361  // Put the job on hold only for the protocols we know about (read: http).
362  KIO::Integration::AccessManager::putReplyOnHold(reply);
363 
364  QString mimeType;
365  KIO::MetaData metaData;
366 
367  if (handleReply(reply, &mimeType, &metaData)) {
368  return;
369  }
370 
371  const QUrl replyUrl(reply->url());
372 
373  // Ask KRun to handle the response when mimetype is unknown
374  if (mimeType.isEmpty()) {
375  (void)new KRun(replyUrl, d->windowWidget());
376  return;
377  }
378 
379  // Ask KRun::runUrl to handle the response when mimetype is inode/*
380  if (mimeType.startsWith(QL1S("inode/"), Qt::CaseInsensitive) &&
381  KRun::runUrl(replyUrl, mimeType, d->windowWidget(), false, false,
382  metaData.value(QL1S("content-disposition-filename")))) {
383  return;
384  }

500 bool KWebPage::handleReply(QNetworkReply *reply, QString *contentType, KIO::MetaData *metaData)
501 {
502  // Reply url...
503  const QUrl replyUrl(reply->url());
504 
505  // Get suggested file name...
506  const KIO::MetaData &data = reply->attribute(static_cast<QNetworkRequest::Attribute>(KIO::AccessManager::MetaData)).toMap();
507  const QString suggestedFileName = data.value(QL1S("content-disposition-filename"));
508  if (metaData) {
509  *metaData = data;
510  }
511 
512  // Get the mime-type...
513  QString mimeType;
514  extractMimeType(reply, mimeType);
515  if (contentType) {
516  *contentType = mimeType;
517  }
518 
519  // Let the calling function deal with handling empty or inode/* mimetypes...
520  if (mimeType.isEmpty() || mimeType.startsWith(QL1S("inode/"), Qt::CaseInsensitive)) {
521  return false;
522  }
523 
524  // Convert executable text files to plain text...
525  if (KParts::BrowserRun::isTextExecutable(mimeType)) {
526  mimeType = QL1S("text/plain");
527  }
528 
529  //qDebug() << "Content-disposition:" << suggestedFileName;
530  //qDebug() << "Got unsupported content of type:" << mimeType << "URL:" << replyUrl;
531  //qDebug() << "Error code:" << reply->error() << reply->errorString();
532 
533  if (isReplyStatusOk(reply)) {
534  while (true) {
535  KParts::BrowserOpenOrSaveQuestion::Result result;
536  KParts::BrowserOpenOrSaveQuestion dlg(d->windowWidget(), replyUrl, mimeType);
537  dlg.setSuggestedFileName(suggestedFileName);
538  dlg.setFeatures(KParts::BrowserOpenOrSaveQuestion::ServiceSelection);
539  result = dlg.askOpenOrSave();
540 
541  switch (result) {
542  case KParts::BrowserOpenOrSaveQuestion::Open:
543  // Handle Post operations that return content...
544  if (reply->operation() == QNetworkAccessManager::PostOperation) {
545  d->mimeType = mimeType;
546  QFileInfo finfo(suggestedFileName.isEmpty() ? replyUrl.fileName() : suggestedFileName);
547  QTemporaryFile tempFile(QDir::tempPath() + QLatin1String("/kwebpage_XXXXXX.") + finfo.suffix());
548  tempFile.setAutoRemove(false);
549  tempFile.open();
550  const QUrl destUrl = QUrl::fromLocalFile(tempFile.fileName());
551  KIO::Job *job = KIO::file_copy(replyUrl, destUrl, 0600, KIO::Overwrite);
552  KJobWidgets::setWindow(job, d->windowWidget());
553  job->ui()->setAutoErrorHandlingEnabled(true);
554  connect(job, SIGNAL(result(KJob*)),
555  this, SLOT(_k_copyResultToTempFile(KJob*)));
556  return true;
557  }
558 
559  // Ask before running any executables...
560  if (KParts::BrowserRun::allowExecution(mimeType, replyUrl)) {
561  KService::Ptr offer = dlg.selectedService();
562  // HACK: The check below is necessary to break an infinite
563  // recursion that occurs whenever this function is called as a result
564  // of receiving content that can be rendered by the app using this engine.
565  // For example a text/html header that containing a content-disposition
566  // header is received by the app using this class.
567  if (isMimeTypeAssociatedWithSelf(offer)) {
568  reloadRequestWithoutDisposition(reply);
569  } else {
570  QList<QUrl> list;
571  list.append(replyUrl);
572  bool success = false;
573  // qDebug() << "Suggested file name:" << suggestedFileName;
574  if (offer) {
575  success = KRun::run(*offer, list, d->windowWidget(), false, suggestedFileName);
576  } else {
577  success = KRun::displayOpenWithDialog(list, d->windowWidget(), false, suggestedFileName);
578  if (!success) {
579  break;
580  }
581  }
582  // For non KIO apps and cancelled Open With dialog, remove slave on hold.
583  if (!success || (offer && !offer->categories().contains(QL1S("KDE")))) {
584  KIO::SimpleJob::removeOnHold(); // Remove any slave-on-hold...
585  }
586  }
587  return true;
588  }
589  // TODO: Instead of silently failing when allowExecution fails, notify
590  // the user why the requested action cannot be fulfilled...
591  return false;
592  case KParts::BrowserOpenOrSaveQuestion::Save:
593  // Do not download local files...
594  if (!replyUrl.isLocalFile()) {
595  QString downloadCmd(reply->property("DownloadManagerExe").toString());
596  if (!downloadCmd.isEmpty()) {
597  downloadCmd += QLatin1Char(' ');
598  downloadCmd += KShell::quoteArg(replyUrl.url());
599  if (!suggestedFileName.isEmpty()) {
600  downloadCmd += QLatin1Char(' ');
601  downloadCmd += KShell::quoteArg(suggestedFileName);
602  }
603  // qDebug() << "download command:" << downloadCmd;
604  if (KRun::runCommand(downloadCmd, view())) {
605  return true;
