161 KWebWallet::WebFormList KWebWallet::KWebWalletPrivate::parseFormData(QWebFrame *frame, bool fillform, bool ignorepasswd)
162 {
163  Q_ASSERT(frame);
164 
165  KWebWallet::WebFormList list;
166 
167  const QVariant result(frame->evaluateJavaScript(QL1S(FILLABLE_FORM_ELEMENT_EXTRACTOR_JS)));
168  const QVariantList results(result.toList());
169  Q_FOREACH (const QVariant &formVariant, results) {
170  QVariantMap map = formVariant.toMap();
171  KWebWallet::WebForm form;
172  form.url = urlForFrame(frame);
173  form.name = map[QL1S("name")].toString();
174  form.index = map[QL1S("index")].toString();
175  bool formHasPasswords = false;
176  const QVariantList elements = map[QL1S("elements")].toList();
177  QList<KWebWallet::WebForm::WebField> inputFields;
178  Q_FOREACH (const QVariant &element, elements) {
179  QVariantMap elementMap(element.toMap());
180  const QString name(elementMap[QL1S("name")].toString());
181  const QString value(ignorepasswd ? QString() : elementMap[QL1S("value")].toString());
182  if (name.isEmpty()) {
183  continue;
184  }
185  if (fillform && elementMap[QL1S("readonly")].toBool()) {
186  continue;
187  }
188  if (elementMap[QL1S("type")].toString().compare(QL1S("password"), Qt::CaseInsensitive) == 0) {
189  if (!fillform && value.isEmpty()) {
190  continue;
191  }
192  formHasPasswords = true;
193  }
194  inputFields.append(qMakePair(name, value));
195  }
196 
197  // Only add the input fields on form save requests...
198  if (formHasPasswords || fillform) {
199  form.fields = inputFields;
200  }
201 
202  // Add the form to the list if we are saving it or it has cached data.
203  if ((fillform && q->hasCachedFormData(form)) || (!fillform && !form.fields.isEmpty())) {
204  list << form;
205  }
206  }
207 
208  return list;
209 }

211 void KWebWallet::KWebWalletPrivate::fillDataFromCache(KWebWallet::WebFormList &formList)
212 {
213  if (!wallet) {
214  qWarning() << "Unable to retrieve form data from wallet";
215  return;
216  }
217 
218  QString lastKey;
219  QMap<QString, QString> cachedValues;
220  QMutableListIterator <WebForm> formIt(formList);
221 
222  while (formIt.hasNext()) {
223  KWebWallet::WebForm &form = formIt.next();
224  const QString key(walletKey(form));
225  if (key != lastKey && wallet->readMap(key, cachedValues) != 0) {
226  qWarning() << "Unable to read form data for key:" << key;
227  continue;
228  }
229 
230  for (int i = 0, count = form.fields.count(); i < count; ++i) {
231  form.fields[i].second = cachedValues.value(form.fields[i].first);
232  }
233  lastKey = key;
234  }
235 }

237 void KWebWallet::KWebWalletPrivate::saveDataToCache(const QString &key)
238 {
239  // Make sure the specified keys exists before acting on it. See BR# 270209.
240  if (!pendingSaveRequests.contains(key)) {
241  return;
242  }
243 
244  bool success = false;
245  const QUrl url = pendingSaveRequests.value(key).first().url;
246 
247  if (wallet) {
248  int count = 0;
249  const KWebWallet::WebFormList list = pendingSaveRequests.value(key);
250  QListIterator<KWebWallet::WebForm> formIt(list);
251 
252  while (formIt.hasNext()) {
253  QMap<QString, QString> values, storedValues;
254  const KWebWallet::WebForm form = formIt.next();
255  const QString accessKey = walletKey(form);
256  if (confirmSaveRequestOverwrites.contains(url)) {
257  confirmSaveRequestOverwrites.remove(url);
258  const int status = wallet->readMap(accessKey, storedValues);
259  if (status == 0 && storedValues.count()) {
260  QListIterator<KWebWallet::WebForm::WebField> fieldIt(form.fields);
261  while (fieldIt.hasNext()) {
262  const KWebWallet::WebForm::WebField field = fieldIt.next();
263  if (storedValues.contains(field.first) &&
264  storedValues.value(field.first) != field.second) {
265  emit q->saveFormDataRequested(key, url);
266  return;
267  }
268  }
269  // If we got here it means the new credential is exactly
270  // the same as the one already cached ; so skip the
271  // re-saving part...
272  success = true;
273  continue;
274  }
275  }
276  QListIterator<KWebWallet::WebForm::WebField> fieldIt(form.fields);
277  while (fieldIt.hasNext()) {
278  const KWebWallet::WebForm::WebField field = fieldIt.next();
279  values.insert(field.first, field.second);
280  }
281 
282  if (wallet->writeMap(accessKey, values) == 0) {
283  count++;
284  } else {
285  qWarning() << "Unable to write form data to wallet";
286  }
287  }
288 
289  if (list.isEmpty() || count > 0) {
290  success = true;
291  }
292 
293  pendingSaveRequests.remove(key);
294  } else {
295  qWarning() << "NULL KWallet instance!";
296  }
297 
298  emit q->saveFormDataCompleted(url, success);
299 }

331 void KWebWallet::KWebWalletPrivate::_k_openWalletDone(bool ok)
332 {
333  Q_ASSERT(wallet);
334 
335  if (ok &&
336  (wallet->hasFolder(KWallet::Wallet::FormDataFolder()) ||
337  wallet->createFolder(KWallet::Wallet::FormDataFolder())) &&
338  wallet->setFolder(KWallet::Wallet::FormDataFolder())) {
339 
340  // Do pending fill requests...
341  if (!pendingFillRequests.isEmpty()) {
342  QList<QUrl> urlList;
343  QMutableHashIterator<QUrl, FormsData> requestIt(pendingFillRequests);
344  while (requestIt.hasNext()) {
345  requestIt.next();
346  KWebWallet::WebFormList list = requestIt.value().forms;
347  fillDataFromCache(list);
348  q->fillWebForm(requestIt.key(), list);
349  }
350 
351  pendingFillRequests.clear();
352  }
353 
354  // Do pending save requests...
355  if (!pendingSaveRequests.isEmpty()) {
356  QListIterator<QString> keysIt(pendingSaveRequests.keys());
357  while (keysIt.hasNext()) {
358  saveDataToCache(keysIt.next());
359  }
360  }
361 
362  // Do pending remove requests...
363  if (!pendingRemoveRequests.isEmpty()) {
364  removeDataFromCache(pendingRemoveRequests);
365  pendingRemoveRequests.clear();
366  }
367  } else {
368  // Delete the wallet if opening the wallet failed or we were unable
369  // to change to the folder we wanted to change to.
370  delete wallet.take();
371  }
372 }

423 void KWebWallet::fillFormData(QWebFrame *frame, bool recursive)
424 {
425  if (!frame) {
426  return;
427  }
428 
429  QList<QUrl> urlList;
430  WebFormList formsList = d->parseFormData(frame);
431  if (!formsList.isEmpty()) {
432  const QUrl url(urlForFrame(frame));
433  if (d->pendingFillRequests.contains(url)) {
434  qWarning() << "Duplicate request rejected!";
435  } else {
436  KWebWalletPrivate::FormsData data;
437  data.frame = QPointer<QWebFrame>(frame);
438  data.forms << formsList;
439  d->pendingFillRequests.insert(url, data);
440  urlList << url;
441  }
442  }
443 
444  if (recursive) {
445  QList<QWebFrame *> childFrameList;
446  collectAllChildFrames(frame, childFrameList);
447  QListIterator<QWebFrame *> frameIt(childFrameList);
448  while (frameIt.hasNext()) {
449  QWebFrame *childFrame = frameIt.next();
450  formsList = d->parseFormData(childFrame);
451  if (formsList.isEmpty()) {
452  continue;
453  }
454  const QUrl url(childFrame->url());
455  if (d->pendingFillRequests.contains(url)) {
456  qWarning() << "Duplicate request rejected!!!";
457  } else {
458  KWebWalletPrivate::FormsData data;
459  data.frame = QPointer<QWebFrame>(childFrame);
460  data.forms << formsList;
461  d->pendingFillRequests.insert(url, data);
462  urlList << url;
463  }
464  }
465  }
466 
467  if (!urlList.isEmpty()) {
468  fillFormDataFromCache(urlList);
469  }
470 }

486 void KWebWallet::saveFormData(QWebFrame *frame, bool recursive, bool ignorePasswordFields)
487 {
488  if (!frame) {
489  return;
490  }
491 
492  QString key;
493  createSaveKeyFor(frame, &key);
494  if (d->pendingSaveRequests.contains(key)) {
495  return;
496  }
497 
498  WebFormList list = d->parseFormData(frame, false, ignorePasswordFields);
499  if (recursive) {
500  QList<QWebFrame *> childFrameList;
501  collectAllChildFrames(frame, childFrameList);
502  QListIterator<QWebFrame *> frameIt(childFrameList);
503  while (frameIt.hasNext()) {
504  list << d->parseFormData(frameIt.next(), false, ignorePasswordFields);
505  }
506  }
507 
508  if (list.isEmpty()) {
509  return;
510  }
511 
512  d->pendingSaveRequests.insert(key, list);
513 
514  QMutableListIterator<WebForm> it(list);
515  while (it.hasNext()) {
516  const WebForm form(it.next());
517  if (hasCachedFormData(form)) {
518  it.remove();
519  }
520  }
521 
522  if (list.isEmpty()) {
523  d->confirmSaveRequestOverwrites.insert(urlForFrame(frame));
524  saveFormDataToCache(key);
525  return;
526  }
527 
528  emit saveFormDataRequested(key, urlForFrame(frame));
529 }

554 void KWebWallet::fillWebForm(const QUrl &url, const KWebWallet::WebFormList &forms)
555 {
556  QPointer<QWebFrame> frame = d->pendingFillRequests.value(url).frame;
557  if (!frame) {
558  return;
559  }
560 
561  QString script;
562  bool wasFilled = false;
563 
564  Q_FOREACH (const KWebWallet::WebForm &form, forms) {
565  Q_FOREACH (const KWebWallet::WebForm::WebField &field, form.fields) {
566  QString value = field.second;
567  value.replace(QL1C('\\'), QL1S("\\\\"));
568  script += QString::fromLatin1("if (document.forms[\"%1\"].elements[\"%2\"]) document.forms[\"%1\"].elements[\"%2\"].value=\"%3\";\n")
569  .arg((form.name.isEmpty() ? form.index : form.name))
570  .arg(field.first).arg(value);
571  }
572  }
573 
574  if (!script.isEmpty()) {
575  wasFilled = true;
576  frame.data()->evaluateJavaScript(script);
577  }
578 
579  emit fillFormRequestCompleted(wasFilled);
