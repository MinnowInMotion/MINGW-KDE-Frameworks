122 int KCategorizedSortFilterProxyModel::compareCategories(const QModelIndex &left, const QModelIndex &right) const
123 {
124  QVariant l = (left.model() ? left.model()->data(left, CategorySortRole) : QVariant());
125  QVariant r = (right.model() ? right.model()->data(right, CategorySortRole) : QVariant());
126 
127  Q_ASSERT(l.isValid());
128  Q_ASSERT(r.isValid());
129  Q_ASSERT(l.type() == r.type());
130 
131  if (l.type() == QVariant::String) {
132  QString lstr = l.toString();
133  QString rstr = r.toString();
134 
135  if (d->sortCategoriesByNaturalComparison) {
136  return d->m_collator.compare(lstr, rstr);
137  } else {
138  if (lstr < rstr) {
139  return -1;
140  }
141 
142  if (lstr > rstr) {
143  return 1;
144  }
145 
146  return 0;
147  }
148  }
