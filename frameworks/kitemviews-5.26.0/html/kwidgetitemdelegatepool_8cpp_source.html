82 QList<QWidget *> KWidgetItemDelegatePool::findWidgets(const QPersistentModelIndex &idx,
83  const QStyleOptionViewItem &option,
84  UpdateWidgetsEnum updateWidgets) const
85 {
86  QList<QWidget *> result;
87 
88  if (!idx.isValid()) {
89  return result;
90  }
91 
92  QModelIndex index;
93  if (const QAbstractProxyModel *proxyModel = qobject_cast<const QAbstractProxyModel *>(idx.model())) {
94  index = proxyModel->mapToSource(idx);
95  } else {
96  index = idx;
97  }
98 
99  if (!index.isValid()) {
100  return result;
101  }
102 
103  if (d->usedWidgets.contains(index)) {
104  result = d->usedWidgets[index];
105  } else {
106  result = d->delegate->createItemWidgets(index);
107  d->allocatedWidgets << result;
108  d->usedWidgets[index] = result;
109  foreach (QWidget *widget, result) {
110  d->widgetInIndex[widget] = index;
111  widget->setParent(d->delegate->d->itemView->viewport());
112  widget->installEventFilter(d->eventListener);
113  widget->setVisible(true);
114  }
115  }
116 
117  if (updateWidgets == UpdateWidgets) {
118  foreach (QWidget *widget, result) {
119  widget->setVisible(true);
120  }
121 
122  d->delegate->updateItemWidgets(result, option, idx);
123 
124  foreach (QWidget *widget, result) {
125  widget->move(widget->x() + option.rect.left(), widget->y() + option.rect.top());
126  }
127  }
128 
129  return result;
130 }

160 bool KWidgetItemDelegateEventListener::eventFilter(QObject *watched, QEvent *event)
161 {
162  QWidget *widget = static_cast<QWidget *>(watched);
163 
164  if (event->type() == QEvent::Destroy && !poolPrivate->clearing) {
165  qWarning() << "User of KWidgetItemDelegate should not delete widgets created by createItemWidgets!";
166  // assume the application has kept a list of widgets and tries to delete them manually
167  // they have been reparented to the view in any case, so no leaking occurs
168  poolPrivate->widgetInIndex.remove(widget);
169  QWidget *viewport = poolPrivate->delegate->d->itemView->viewport();
170  QApplication::sendEvent(viewport, event);
171  }
172  if (dynamic_cast<QInputEvent *>(event) && !poolPrivate->delegate->blockedEventTypes(widget).contains(event->type())) {
173  QWidget *viewport = poolPrivate->delegate->d->itemView->viewport();
174  switch (event->type()) {
175  case QEvent::MouseMove:
176  case QEvent::MouseButtonPress:
177  case QEvent::MouseButtonRelease:
178  case QEvent::MouseButtonDblClick: {
179  QMouseEvent *mouseEvent = static_cast<QMouseEvent *>(event);
180  QMouseEvent evt(event->type(), viewport->mapFromGlobal(mouseEvent->globalPos()),
181  mouseEvent->button(), mouseEvent->buttons(), mouseEvent->modifiers());
182  QApplication::sendEvent(viewport, &evt);
183  }
184  break;
185  case QEvent::Wheel: {
186  QWheelEvent *wheelEvent = static_cast<QWheelEvent *>(event);
187  QWheelEvent evt(viewport->mapFromGlobal(wheelEvent->globalPos()),
188  wheelEvent->delta(), wheelEvent->buttons(), wheelEvent->modifiers(),
189  wheelEvent->orientation());
190  QApplication::sendEvent(viewport, &evt);
191  }
192  break;
193  case QEvent::TabletMove:
194  case QEvent::TabletPress:
195  case QEvent::TabletRelease:
196  case QEvent::TabletEnterProximity:
197  case QEvent::TabletLeaveProximity: {
198  QTabletEvent *tabletEvent = static_cast<QTabletEvent *>(event);
199  QTabletEvent evt(event->type(), QPointF(viewport->mapFromGlobal(tabletEvent->globalPos())),
200  tabletEvent->globalPosF(), tabletEvent->device(),
201  tabletEvent->pointerType(), tabletEvent->pressure(), tabletEvent->xTilt(),
202  tabletEvent->yTilt(), tabletEvent->tangentialPressure(), tabletEvent->rotation(),
203  tabletEvent->z(), tabletEvent->modifiers(), tabletEvent->uniqueId());
204  QApplication::sendEvent(viewport, &evt);
205  }
206  break;
207  default:
208  QApplication::sendEvent(viewport, event);
209  break;
210  }
211  }
212 
213  return QObject::eventFilter(watched, event);
214 }
