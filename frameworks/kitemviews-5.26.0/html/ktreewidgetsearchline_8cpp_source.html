79 void KTreeWidgetSearchLine::Private::_k_rowsInserted(const QModelIndex &parentIndex, int start, int end) const
80 {
81  QAbstractItemModel *model = qobject_cast<QAbstractItemModel *>(q->sender());
82  if (!model) {
83  return;
84  }
85 
86  QTreeWidget *widget = 0L;
87  foreach (QTreeWidget *tree, treeWidgets)
88  if (tree->model() == model) {
89  widget = tree;
90  break;
91  }
92 
93  if (!widget) {
94  return;
95  }
96 
97  QTreeWidgetWorkaround *widgetW = static_cast<QTreeWidgetWorkaround *>(widget);
98  for (int i = start; i <= end; ++i) {
99  if (QTreeWidgetItem *item = widgetW->itemFromIndex(model->index(i, 0, parentIndex))) {
100  bool newHidden = !q->itemMatches(item, q->text());
101  if (item->isHidden() != newHidden) {
102  item->setHidden(newHidden);
103  emit q->hiddenChanged(item, newHidden);
104  }
105  }
106  }
107 }

115 void KTreeWidgetSearchLine::Private::_k_slotColumnActivated(QAction *action)
116 {
117  if (!action) {
118  return;
119  }
120 
121  bool ok;
122  int column = action->data().toInt(&ok);
123 
124  if (!ok) {
125  return;
126  }
127 
128  if (action->isChecked()) {
129  if (!searchColumns.isEmpty()) {
130  if (!searchColumns.contains(column)) {
131  searchColumns.append(column);
132  }
133 
134  if (searchColumns.count() == treeWidgets.first()->header()->count() - treeWidgets.first()->header()->hiddenSectionCount()) {
135  searchColumns.clear();
136  }
137 
138  } else {
139  searchColumns.append(column);
140  }
141  } else {
142  if (searchColumns.isEmpty()) {
143  QHeaderView *const header = treeWidgets.first()->header();
144 
145  for (int i = 0; i < header->count(); i++) {
146  if (i != column && !header->isSectionHidden(i)) {
147  searchColumns.append(i);
148  }
149  }
150 
151  } else if (searchColumns.contains(column)) {
152  searchColumns.removeAll(column);
153  }
154  }
155 
156  q->updateSearch();
157 }

198 bool KTreeWidgetSearchLine::Private::checkItemParentsVisible(QTreeWidgetItem *item)
199 {
200  bool childMatch = false;
201  for (int i = 0; i < item->childCount(); ++i) {
202  childMatch |= checkItemParentsVisible(item->child(i));
203  }
204 
205  // Should this item be shown? It should if any children should be, or if it matches.
206  bool newHidden = !childMatch && !q->itemMatches(item, search);
207  if (item->isHidden() != newHidden) {
208  item->setHidden(newHidden);
209  emit q->hiddenChanged(item, newHidden);
210  }
211 
212  return !newHidden;
213 }

323 void KTreeWidgetSearchLine::updateSearch(QTreeWidget *treeWidget)
324 {
325  if (!treeWidget || !treeWidget->topLevelItemCount()) {
326  return;
327  }
328 
329  // If there's a selected item that is visible, make sure that it's visible
330  // when the search changes too (assuming that it still matches).
331 
332  QTreeWidgetItem *currentItem = treeWidget->currentItem();
333 
334  if (d->keepParentsVisible)
335  for (int i = 0; i < treeWidget->topLevelItemCount(); ++i) {
336  d->checkItemParentsVisible(treeWidget->topLevelItem(i));
337  }
338  else {
339  d->checkItemParentsNotVisible(treeWidget);
340  }
341 
342  if (currentItem) {
343  treeWidget->scrollToItem(currentItem);
344  }
345 
346  emit searchUpdated(d->search);
347 }

399 bool KTreeWidgetSearchLine::itemMatches(const QTreeWidgetItem *item, const QString &pattern) const
400 {
401  if (pattern.isEmpty()) {
402  return true;
403  }
404 
405  // If the search column list is populated, search just the columns
406  // specified. If it is empty default to searching all of the columns.
407 
408  if (!d->searchColumns.isEmpty()) {
409  QList<int>::ConstIterator it = d->searchColumns.constBegin();
410  for (; it != d->searchColumns.constEnd(); ++it) {
411  if (*it < item->treeWidget()->columnCount() &&
412  item->text(*it).indexOf(pattern, 0, d->caseSensitive) >= 0) {
413  return true;
414  }
415  }
416  } else {
417  for (int i = 0; i < item->treeWidget()->columnCount(); i++) {
418  if (item->treeWidget()->columnWidth(i) > 0 &&
419  item->text(i).indexOf(pattern, 0, d->caseSensitive) >= 0) {
420  return true;
421  }
422  }
423  }
424 
425  return false;
426 }

500 bool KTreeWidgetSearchLine::canChooseColumnsCheck()
501 {
502  // This is true if either of the following is true:
503 
504  // there are no listviews connected
505  if (d->treeWidgets.isEmpty()) {
506  return false;
507  }
508 
509  const QTreeWidget *first = d->treeWidgets.first();
510 
511  const unsigned int numcols = first->columnCount();
512  // the listviews have only one column,
513  if (numcols < 2) {
514  return false;
515  }
516 
517  QStringList headers;
518  for (unsigned int i = 0; i < numcols; ++i) {
519  headers.append(first->headerItem()->text(i));
520  }
521 
522  QList<QTreeWidget *>::ConstIterator it = d->treeWidgets.constBegin();
523  for (++it /* skip the first one */; it != d->treeWidgets.constEnd(); ++it) {
524  // the listviews have different numbers of columns,
525  if ((unsigned int)(*it)->columnCount() != numcols) {
526  return false;
527  }
528 
529  // the listviews differ in column labels.
530  QStringList::ConstIterator jt;
531  unsigned int i;
532  for (i = 0, jt = headers.constBegin(); i < numcols; ++i, ++jt) {
533  Q_ASSERT(jt != headers.constEnd());
534 
535  if ((*it)->headerItem()->text(i) != *jt) {
536  return false;
537  }
538  }
539  }
540 
541  return true;
542 }
