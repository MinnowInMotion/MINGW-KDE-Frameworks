254 QVariant ScriptableLiveConnectExtension::callFunctionReference(ScriptableExtension *,
255  quint64 o, const QString &f, const ScriptableExtension::ArgList &a)
256 {
257  QStringList qargs;
258  // Convert args to strings for LC use.
259  qargs.reserve(a.size());
260  for (int i = 0; i < a.size(); ++i) {
261  bool ok;
262  qargs.append(toLC(a[i], &ok));
263  if (!ok) {
264  return unimplemented();
265  }
266  }

268  LiveConnectExtension::Type retType;
269  unsigned long retObjId;
270  QString retVal;
271  if (wrapee->call((unsigned long)o, f, qargs, retType, retObjId, retVal)) {
272  return acquireValue(fromLC(QString(), retType, retObjId, retVal));
273  } else {
274  return unimplemented();

304 QVariant ScriptableLiveConnectExtension::fromLC(const QString &name,
305  LiveConnectExtension::Type type,
306  unsigned long objId,
307  const QString &value)
308 {
309  switch (type) {
310  case KParts::LiveConnectExtension::TypeBool: {
311  bool ok;
312  int i = value.toInt(&ok);
313  if (ok) {
314  return QVariant(bool(i));
315  }
316  return QVariant(value.toLower() == QLatin1String("true"));
317  }
318  case KParts::LiveConnectExtension::TypeObject:
319  case KParts::LiveConnectExtension::TypeFunction: {
320  if (!refCounts.contains(objId)) {
321  refCounts[objId] = 0;
322  }
323 
324  Object o = ScriptableExtension::Object(this, objId);
325  if (type == KParts::LiveConnectExtension::TypeObject) {
326  return QVariant::fromValue(o);
327  } else {
328  return QVariant::fromValue(FunctionRef(o, name));
329  }
330  }
331 
332  case KParts::LiveConnectExtension::TypeNumber:
333  return QVariant(value.toDouble());
334 
335  case KParts::LiveConnectExtension::TypeString:
336  return QVariant(value);
337 
338  case KParts::LiveConnectExtension::TypeVoid:
339  default:
340  return QVariant::fromValue(ScriptableExtension::Undefined());
341  }
342 }

344 QString ScriptableLiveConnectExtension::toLC(const QVariant &in, bool *ok)
345 {
346  *ok = true; // most of the time.
347 
348  // Objects (or exceptions) can't be converted
349  if (in.canConvert<ScriptableExtension::Object>() ||
350  in.canConvert<ScriptableExtension::Exception>() ||
351  in.canConvert<ScriptableExtension::FunctionRef>()) {
352 
353  *ok = false;
354  return QString();
355  }
356 
357  // Convert null and undefined to appropriate strings
358  // ### this matches old KHTML behavior, but is this sensible?
359  if (in.canConvert<ScriptableExtension::Null>()) {
360  return QStringLiteral("null");
361  }
362 
363  if (in.canConvert<ScriptableExtension::Undefined>()) {
364  return QStringLiteral("undefined");
365  }
366 
367  if (in.type() == QVariant::Bool) {
368  return in.toBool() ? QStringLiteral("true") : QStringLiteral("false");
369  }
370 
371  // Just stringify everything else, makes sense for nums as well.
372  if (in.canConvert<QString>()) {
373  return in.toString();
374  }
375 
376  // something really icky...
377  *ok = false;
378  return QString();
379 }

397 void ScriptableLiveConnectExtension::liveConnectEvent(const unsigned long, const QString &event,
398  const LiveConnectExtension::ArgList &args)
399 {
400  // We want to evaluate in the enclosure's context.
401  QVariant enclosure = enclosingObject();
402  if (!enclosure.canConvert<Object>()) {
403  releaseValue(enclosure);
404  // qDebug() << "No enclosure, can't evaluate";
405  return;
406  }
407 
408  Object enclosureObj = enclosure.value<Object>();
409 
410  if (!host()->isScriptLanguageSupported(ECMAScript)) {
411  releaseValue(enclosure);
412  // qDebug() << "Host can't evaluate ECMAScript";
413  }
414 
415  // Compute a string to evaluate. We need to escape a lot of stuff
416  // since we're composing a bunch of strings into one.
417  QString script;
418  script.sprintf("%s(", event.toLatin1().constData());
419 
420  LiveConnectExtension::ArgList::const_iterator i = args.begin();
421  const LiveConnectExtension::ArgList::const_iterator argsBegin = i;
422  const LiveConnectExtension::ArgList::const_iterator argsEnd = args.end();
423 
424  for (; i != argsEnd; ++i) {
425  if (i != argsBegin) {
426  script += QLatin1Char(',');
427  }
428  if ((*i).first == KParts::LiveConnectExtension::TypeString) {
429  script += QLatin1Char('"') +
430  QString((*i).second).replace(QLatin1Char('\\'), QStringLiteral("\\\\")).replace(QLatin1Char('"'), QStringLiteral("\\\"")) +
431  QLatin1Char('"');
432  } else {
433  script += (*i).second;
434  }
435  }
436  script += QLatin1Char(')');
437 
438  // qDebug() << script;
439 
440  // Ask host to evaluate.. (unfortunately, we can't do anything with the result,
441  // but anything that uses this interface isn't expective one in the first place)
442  QVariant result = host()->evaluateScript(this, enclosureObj.objId, script);
443 
444  releaseValue(result);
445  releaseValue(enclosure);
446 }
