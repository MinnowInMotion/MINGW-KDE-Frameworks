88 QList<Plugin::PluginInfo> Plugin::pluginInfos(const QString &componentName)
89 {
90  QList<PluginInfo> plugins;
91 
92  QMap<QString, QStringList> sortedPlugins;
93 
94  const QStringList dirs = QStandardPaths::locateAll(QStandardPaths::GenericDataLocation, componentName + QStringLiteral("/kpartplugins"), QStandardPaths::LocateDirectory);
95  Q_FOREACH (const QString &dir, dirs) {
96  Q_FOREACH (const QString &file, QDir(dir).entryList(QStringList("*.rc"))) {
97  const QFileInfo fInfo(dir + QLatin1Char('/') + file);
98  QMap<QString, QStringList>::Iterator mapIt = sortedPlugins.find(fInfo.fileName());
99  if (mapIt == sortedPlugins.end()) {
100  mapIt = sortedPlugins.insert(fInfo.fileName(), QStringList());
101  }
102  mapIt.value().append(fInfo.absoluteFilePath());
103  }
104  }
105 
106  QMap<QString, QStringList>::ConstIterator mapIt = sortedPlugins.constBegin();
107  QMap<QString, QStringList>::ConstIterator mapEnd = sortedPlugins.constEnd();
108  for (; mapIt != mapEnd; ++mapIt) {
109  PluginInfo info;
110  QString doc;
111  info.m_absXMLFileName = KXMLGUIClient::findMostRecentXMLFile(mapIt.value(), doc);
112  if (info.m_absXMLFileName.isEmpty()) {
113  continue;
114  }
115 
116  // qDebug() << "found KParts Plugin : " << info.m_absXMLFileName;
117  info.m_relXMLFileName = QStringLiteral("kpartplugins/") + mapIt.key();
118 
119  info.m_document.setContent(doc);
120  if (info.m_document.documentElement().isNull()) {
121  continue;
122  }
123 
124  plugins.append(info);
125  }
126 
127  return plugins;
128 }

135 void Plugin::loadPlugins(QObject *parent, const QList<PluginInfo> &pluginInfos, const QString &componentName)
136 {
137  QList<PluginInfo>::ConstIterator pIt = pluginInfos.begin();
138  QList<PluginInfo>::ConstIterator pEnd = pluginInfos.end();
139  for (; pIt != pEnd; ++pIt) {
140  QString library = (*pIt).m_document.documentElement().attribute(QStringLiteral("library"));
141 
142  if (library.isEmpty() || hasPlugin(parent, library)) {
143  continue;
144  }
145 
146  Plugin *plugin = loadPlugin(parent, library, (*pIt).m_document.documentElement().attribute(QStringLiteral("X-KDE-PluginKeyword")));
147 
148  if (plugin) {
149  plugin->d->m_parentInstance = componentName;
150  plugin->setXMLFile((*pIt).m_relXMLFileName, false, false);
151  plugin->setDOMDocument((*pIt).m_document);
152 
153  }
154  }
155 
156 }

164 Plugin *Plugin::loadPlugin(QObject *parent, const QString &libname, const QString &keyword)
165 {
166  KPluginLoader loader(libname);
167  KPluginFactory *factory = loader.factory();
168 
169  if (!factory) {
170  return 0;
171  }
172 
173  Plugin *plugin = factory->create<Plugin>(keyword, parent);
174  if (!plugin) {
175  return 0;
176  }
177  plugin->d->m_library = libname;
178  return plugin;

213 void Plugin::loadPlugins(QObject *parent, KXMLGUIClient *parentGUIClient,
214  const QString &componentName, bool enableNewPluginsByDefault,
215  int interfaceVersionRequired)
216 {
217  KConfigGroup cfgGroup(KSharedConfig::openConfig(componentName + QStringLiteral("rc")), "KParts Plugins");
218  const QList<PluginInfo> plugins = pluginInfos(componentName);
219  QList<PluginInfo>::ConstIterator pIt = plugins.begin();
220  const QList<PluginInfo>::ConstIterator pEnd = plugins.end();
221  for (; pIt != pEnd; ++pIt) {
222  QDomElement docElem = (*pIt).m_document.documentElement();
223  QString library = docElem.attribute(QStringLiteral("library"));
224  QString keyword;
225 
226  if (library.isEmpty()) {
227  continue;
228  }
229 
230  // Check configuration
231  const QString name = docElem.attribute(QStringLiteral("name"));
232 
233  bool pluginEnabled = enableNewPluginsByDefault;
234  if (cfgGroup.hasKey(name + QStringLiteral("Enabled"))) {
235  pluginEnabled = cfgGroup.readEntry(name + QStringLiteral("Enabled"), false);
236  } else { // no user-setting, load plugin default setting
237  QString relPath = componentName + QLatin1Char('/') + (*pIt).m_relXMLFileName;
238  relPath.truncate(relPath.lastIndexOf(QLatin1Char('.'))); // remove extension
239  relPath += QStringLiteral(".desktop");
240  //qDebug() << "looking for " << relPath;
241  const QString desktopfile = QStandardPaths::locate(QStandardPaths::GenericDataLocation, relPath);
242  if (!desktopfile.isEmpty()) {
243  //qDebug() << "loadPlugins found desktop file for " << name << ": " << desktopfile;
244  KDesktopFile _desktop(desktopfile);
245  const KConfigGroup desktop = _desktop.desktopGroup();
246  keyword = desktop.readEntry("X-KDE-PluginKeyword", "");
247  pluginEnabled = desktop.readEntry("X-KDE-PluginInfo-EnabledByDefault",
248  enableNewPluginsByDefault);
249  if (interfaceVersionRequired != 0) {
250  const int version = desktop.readEntry("X-KDE-InterfaceVersion", 1);
251  if (version != interfaceVersionRequired) {
252  // qDebug() << "Discarding plugin " << name << ", interface version " << version << ", expected " << interfaceVersionRequired;
253  pluginEnabled = false;
254  }
255  }
256  } else {
257  //qDebug() << "loadPlugins no desktop file found in " << relPath;
258  }
259  }
260 
261  // search through already present plugins
262  const QObjectList pluginList = parent->children();
263 
264  bool pluginFound = false;
265  for (QObjectList::ConstIterator it = pluginList.begin(); it != pluginList.end(); ++it) {
266  Plugin *plugin = qobject_cast<Plugin *>(*it);
267  if (plugin && plugin->d->m_library == library) {
268  // delete and unload disabled plugins
269  if (!pluginEnabled) {
270  // qDebug() << "remove plugin " << name;
271  KXMLGUIFactory *factory = plugin->factory();
272  if (factory) {
273  factory->removeClient(plugin);
274  }
275  delete plugin;
276  }
277 
278  pluginFound = true;
279  break;
280  }
281  }
282 
283  // if the plugin is already loaded or if it's disabled in the
284  // configuration do nothing
285  if (pluginFound || !pluginEnabled) {
286  continue;
287  }
288 
289  // qDebug() << "load plugin " << name << " " << library << " " << keyword;
290  Plugin *plugin = loadPlugin(parent, library, keyword);
291 
292  if (plugin) {
293  plugin->d->m_parentInstance = componentName;
294  plugin->setXMLFile((*pIt).m_relXMLFileName, false, false);
295  plugin->setDOMDocument((*pIt).m_document);
296  parentGUIClient->insertChildClient(plugin);
297  }
298  }
299 }
