41 // Internal class, use to store the status of the actions
42 class KBitArray
43 {
44 public:
45  int val;
46  KBitArray()
47  {
48  val = 0;
49  }
50  bool operator [](int index)
51  {
52  return (val & (1 << index)) ? true : false;
53  }
54  void setBit(int index, bool value)
55  {
56  if (value) {
57  val = val | (1 << index);
58  } else {
59  val = val & ~(1 << index);
60  }
61  }
62 };

96 {
97  s_actionSlotMap()->insert("cut", SLOT(cut()));

99  s_actionSlotMap()->insert("paste", SLOT(paste()));
100  s_actionSlotMap()->insert("print", SLOT(print()));
101  // Tricky. Those aren't actions in fact, but simply methods that a browserextension

104  //s_actionSlotMap()->insert( "refreshMimeTypes", SLOT(refreshMimeTypes()) );

106  // Create the action-number map
107  ActionSlotMap::ConstIterator it = s_actionSlotMap()->constBegin();
108  ActionSlotMap::ConstIterator itEnd = s_actionSlotMap()->constEnd();
109  for (int i = 0; it != itEnd; ++it, ++i) {

117 BrowserExtension::BrowserExtension(KParts::ReadOnlyPart *parent)
118  : QObject(parent), d(new BrowserExtensionPrivate(parent))
119 {
120  //qDebug() << "BrowserExtension::BrowserExtension() " << this;
121 
122  if (s_actionSlotMap()->isEmpty())
123  // Create the action-slot map
124  {
125  BrowserExtensionPrivate::createActionSlotMap();
126  }
127 
128  // Set the initial status of the actions depending on whether
129  // they're supported or not
130  const QMetaObject *metaobj = metaObject();
131  ActionSlotMap::ConstIterator it = s_actionSlotMap()->constBegin();
132  ActionSlotMap::ConstIterator itEnd = s_actionSlotMap()->constEnd();
133  for (int i = 0; it != itEnd; ++it, ++i) {
134  // Does the extension have a slot with the name of this action ?
135  QByteArray slotSig = it.key() + "()";
136  d->m_actionStatus.setBit(i, metaobj->indexOfMethod(slotSig.constData()) != -1);
137  }
138 
139  connect(d->m_part, static_cast<void (KParts::ReadOnlyPart::*)()>(&KParts::ReadOnlyPart::completed),
140  this, &BrowserExtension::slotCompleted);
141  connect(this, &BrowserExtension::openUrlRequest,
142  this, &BrowserExtension::slotOpenUrlRequest);
143  connect(this, &BrowserExtension::enableAction,
144  this, &BrowserExtension::slotEnableAction);
145  connect(this, &BrowserExtension::setActionText,
146  this, &BrowserExtension::slotSetActionText);
147 }

211 void BrowserExtension::pasteRequest()
212 {
213  QString plain(QStringLiteral("plain"));
214  QString url = QApplication::clipboard()->text(plain, QClipboard::Selection).trimmed();
215  // Remove linefeeds and any whitespace surrounding it.
216  url.remove(QRegExp(QStringLiteral("[\\ ]*\\n+[\\ ]*")));
217 
218  // Check if it's a URL
219  QStringList filters = KUriFilter::self()->pluginNames();
220  filters.removeAll(QStringLiteral("kuriikwsfilter"));
221  filters.removeAll(QStringLiteral("localdomainurifilter"));
222  KUriFilterData filterData;
223  filterData.setData(url);
224  filterData.setCheckForExecutables(false);
225  if (KUriFilter::self()->filterUri(filterData, filters)) {
226  switch (filterData.uriType()) {
227  case KUriFilterData::LocalFile:
228  case KUriFilterData::LocalDir:
229  case KUriFilterData::NetProtocol:
230  slotOpenUrlRequest(filterData.uri());
231  break;
232  case KUriFilterData::Error:
233  KMessageBox::sorry(d->m_part->widget(), filterData.errorMsg());
234  break;
235  default:
236  break;
237  }
238  } else if (KUriFilter::self()->filterUri(filterData, QStringList(QStringLiteral("kuriikwsfilter"))) &&
239  url.length() < 250) {
240  if (KMessageBox::questionYesNo(d->m_part->widget(),
241  i18n("<qt>Do you want to search the Internet for <b>%1</b>?</qt>", url.toHtmlEscaped()),
242  i18n("Internet Search"), KGuiItem(i18n("&Search"), QStringLiteral("edit-find")),
243  KStandardGuiItem::cancel(), QStringLiteral("MiddleClickSearch")) == KMessageBox::Yes) {
244  slotOpenUrlRequest(filterData.uri());
245  }
246  }
247 }

249 void BrowserExtension::slotOpenUrlRequest(const QUrl &url, const KParts::OpenUrlArguments &args, const KParts::BrowserArguments &browserArgs)
250 {
251  //qDebug() << this << " BrowserExtension::slotOpenURLRequest(): url=" << url.url();
252  BrowserExtensionPrivate::DelayedRequest req;
253  req.m_delayedURL = url;
254  req.m_delayedArgs = args;
255  req.m_delayedBrowserArgs = browserArgs;
256  d->m_requests.append(req);
257  QTimer::singleShot(0, this, SLOT(slotEmitOpenUrlRequestDelayed()));
258 }

260 void BrowserExtension::slotEmitOpenUrlRequestDelayed()
261 {
262  if (d->m_requests.isEmpty()) {
263  return;
264  }
265  BrowserExtensionPrivate::DelayedRequest req = d->m_requests.front();
266  d->m_requests.pop_front();
267  emit openUrlRequestDelayed(req.m_delayedURL, req.m_delayedArgs, req.m_delayedBrowserArgs);
268  // tricky: do not do anything here! (no access to member variables, etc.)
269 }
