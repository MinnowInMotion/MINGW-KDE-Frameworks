114 void BrowserRun::scanFile()
115 {
116  // qDebug() << KRun::url();
117 
118  // Let's check for well-known extensions
119  // Not when there is a query in the URL, in any case.
120  // Optimization for http/https, findByURL doesn't trust extensions over http.
121  QString protocol = KRun::url().scheme();
122 
123  if (!KProtocolInfo::proxiedBy(protocol).isEmpty()) {
124  QString dummy;
125  protocol = KProtocolManager::slaveProtocol(KRun::url(), dummy);
126  }
127 
128  if (!KRun::url().hasQuery() && !protocol.startsWith(QLatin1String("http"))) {
129  QMimeDatabase db;
130  QMimeType mime = db.mimeTypeForUrl(KRun::url());
131  if (!mime.isDefault() || isLocalFile()) {
132  // qDebug() << "MIME TYPE is" << mime.name();
133  mimeTypeDetermined(mime.name());
134  return;
135  }
136  }
137 
138  QMap<QString, QString> &metaData = d->m_args.metaData();
139  if (d->m_part) {
140  const QString proto = d->m_part->url().scheme();
141 
142  if (proto == QLatin1String("https") || proto == QLatin1String("webdavs")) {
143  metaData.insert(QStringLiteral("main_frame_request"), QStringLiteral("TRUE"));
144  metaData.insert(QStringLiteral("ssl_was_in_use"), QStringLiteral("TRUE"));
145  // metaData.insert(QStringLiteral("ssl_activate_warnings"), QStringLiteral("TRUE"));
146  } else if (proto == QLatin1String("http") || proto == QLatin1String("webdav")) {
147  // metaData.insert(QStringLiteral("ssl_activate_warnings"), QStringLiteral("TRUE"));
148  metaData.insert(QStringLiteral("ssl_was_in_use"), QStringLiteral("FALSE"));
149  }
150 
151  // Set the PropagateHttpHeader meta-data if it has not already been set...
152  if (!metaData.contains(QStringLiteral("PropagateHttpHeader"))) {
153  metaData.insert(QStringLiteral("PropagateHttpHeader"), QStringLiteral("TRUE"));
154  }
155  }
156 
157  KIO::TransferJob *job;
158  if (d->m_browserArgs.doPost() && KRun::url().scheme().startsWith(QLatin1String("http"))) {
159  job = KIO::http_post(KRun::url(), d->m_browserArgs.postData, KIO::HideProgressInfo);
160  job->addMetaData(QStringLiteral("content-type"), d->m_browserArgs.contentType());
161  } else {
162  job = KIO::get(KRun::url(),
163  d->m_args.reload() ? KIO::Reload : KIO::NoReload,
164  KIO::HideProgressInfo);
165  }
166 
167  if (d->m_bRemoveReferrer) {
168  metaData.remove(QStringLiteral("referrer"));
169  }
170 
171  job->addMetaData(metaData);
172  KJobWidgets::setWindow(job, d->m_window);
173  connect(job, &KIO::TransferJob::result,
174  this, &BrowserRun::slotBrowserScanFinished);
175  connect(job, static_cast<void (KIO::TransferJob::*)(KIO::Job*,const QString &)>(&KIO::TransferJob::mimetype),
176  this, &BrowserRun::slotBrowserMimetype);
177  setJob(job);
178 }

197 static QMimeType fixupMimeType(const QString &mimeType, const QString &fileName)
198 {
199  QMimeDatabase db;
200  QMimeType mime = db.mimeTypeForName(mimeType);

202  mime = db.mimeTypeForFile(fileName, QMimeDatabase::MatchExtension);
203  }
204  return mime;

207 void BrowserRun::slotBrowserMimetype(KIO::Job *_job, const QString &type)
208 {
209  Q_ASSERT(_job == KRun::job()); Q_UNUSED(_job)
210  KIO::TransferJob *job = static_cast<KIO::TransferJob *>(KRun::job());
211  // Update our URL in case of a redirection
212  //qDebug() << "old URL=" << KRun::url();
213  //qDebug() << "new URL=" << job->url();
214  setUrl(job->url());
215 
216  if (job->isErrorPage()) {
217  d->m_mimeType = type;
218  handleError(job);
219  setJob(0);
220  } else {
221  // qDebug() << "found" << type << "for" << KRun::url();
222 
223  // Suggested filename given by the server (e.g. HTTP content-disposition)
224  // When set, we should really be saving instead of embedding
225  const QString suggestedFileName = job->queryMetaData(QStringLiteral("content-disposition-filename"));
226  setSuggestedFileName(suggestedFileName); // store it (in KRun)
227  //qDebug() << "suggestedFileName=" << suggestedFileName;
228  d->m_contentDisposition = job->queryMetaData(QStringLiteral("content-disposition-type"));
229 
230  const QString modificationTime = job->queryMetaData(QStringLiteral("content-disposition-modification-date"));
231  if (!modificationTime.isEmpty()) {
232  d->m_args.metaData().insert(QStringLiteral("content-disposition-modification-date"), modificationTime);
233  }
234 
235  QMapIterator<QString, QString> it(job->metaData());
236  while (it.hasNext()) {
237  it.next();
238  if (it.key().startsWith(QLatin1String("ssl_"), Qt::CaseInsensitive)) {
239  d->m_args.metaData().insert(it.key(), it.value());
240  }
241  }
242 
243  // Make a copy to avoid a dead reference
244  QString _type = type;
245  job->putOnHold();
246  setJob(0);
247 
248  // If the current mime-type is the default mime-type, then attempt to
249  // determine the "real" mimetype from the file name.
250  QMimeType mime = fixupMimeType(_type, suggestedFileName.isEmpty() ? url().fileName() : suggestedFileName);
251  if (mime.isValid() && mime.name() != _type) {
252  _type = mime.name();
253  }
254 
255  mimeTypeDetermined(_type);
256  }
257 }

264 BrowserRun::NonEmbeddableResult BrowserRun::handleNonEmbeddable(const QString &_mimeType, KService::Ptr *selectedService)
265 {
266  QString mimeType(_mimeType);
267  Q_ASSERT(!hasFinished()); // only come here if the mimetype couldn't be embedded
268  // Support for saving remote files.
269  if (mimeType != QLatin1String("inode/directory") && // dirs can't be saved
270  !KRun::url().isLocalFile()) {
271  if (isTextExecutable(mimeType)) {
272  mimeType = QStringLiteral("text/plain"); // view, don't execute
273  }
274  // ... -> ask whether to save
275  BrowserOpenOrSaveQuestion question(d->m_window, KRun::url(), mimeType);
276  question.setSuggestedFileName(suggestedFileName());
277  if (selectedService) {
278  question.setFeatures(BrowserOpenOrSaveQuestion::ServiceSelection);
279  }
280  BrowserOpenOrSaveQuestion::Result res = question.askOpenOrSave();
281  if (res == BrowserOpenOrSaveQuestion::Save) {
282  save(KRun::url(), suggestedFileName());
283  // qDebug() << "Save: returning Handled";
284  setFinished(true);
285  return Handled;
286  } else if (res == BrowserOpenOrSaveQuestion::Cancel) {
287  // saving done or canceled
288  // qDebug() << "Cancel: returning Handled";
289  setFinished(true);
290  return Handled;
291  } else { // "Open" chosen (done by KRun::foundMimeType, called when returning NotHandled)
292  // If we were in a POST, we can't just pass a URL to an external application.
293  // We must save the data to a tempfile first.
294  if (d->m_browserArgs.doPost()) {
295  // qDebug() << "request comes from a POST, can't pass a URL to another app, need to save";
296  d->m_mimeType = mimeType;
297  QString extension;
298  QString fileName = suggestedFileName().isEmpty() ? KRun::url().fileName() : suggestedFileName();
299  int extensionPos = fileName.lastIndexOf(QLatin1Char('.'));
300  if (extensionPos != -1) {
301  extension = fileName.mid(extensionPos); // keep the '.'
302  }
303  QTemporaryFile tempFile(QDir::tempPath() + QLatin1Char('/') + QCoreApplication::applicationName() + QLatin1String("XXXXXX") + extension);
304  tempFile.setAutoRemove(false);
305  tempFile.open();
306  QUrl destURL = QUrl::fromLocalFile(tempFile.fileName());
307  KIO::Job *job = KIO::file_copy(KRun::url(), destURL, 0600, KIO::Overwrite);
308  KJobWidgets::setWindow(job, d->m_window);
309  connect(job, &KIO::Job::result,
310  this, &BrowserRun::slotCopyToTempFileResult);
311  return Delayed; // We'll continue after the job has finished
312  }
313  if (selectedService && question.selectedService()) {
314  *selectedService = question.selectedService();
315  // KRun will use this when starting an app
316  KRun::setPreferredService(question.selectedService()->desktopEntryName());
317  }
318  }
319  }
320 
321  // Check if running is allowed
322  if (!d->m_bTrustedSource && // ... and untrusted source...
323  !allowExecution(mimeType, KRun::url())) { // ...and the user said no (for executables etc.)
324  setFinished(true);
325  return Handled;
326  }
327 
328  KIO::Scheduler::publishSlaveOnHold(); // publish any slave on hold so it can be reused.
329  return NotHandled;
330 }

388 void KParts::BrowserRun::saveUrl(const QUrl &url, const QString &suggestedFileName,
389  QWidget *window, const KParts::OpenUrlArguments &args)
390 {
391  // DownloadManager <-> konqueror integration
392  // find if the integration is enabled
393  // the empty key means no integration
394  // only use the downloadmanager for non-local urls
395  if (!url.isLocalFile()) {
396  KConfigGroup cfg = KSharedConfig::openConfig(QStringLiteral("konquerorrc"), KConfig::NoGlobals)->group("HTML Settings");
397  QString downloadManger = cfg.readPathEntry("DownloadManager", QString());
398  if (!downloadManger.isEmpty()) {
399  // then find the download manager location
400  // qDebug() << "Using: "<<downloadManger <<" as Download Manager";
401  QString cmd = QStandardPaths::findExecutable(downloadManger);
402  if (cmd.isEmpty()) {
403  QString errMsg = i18n("The Download Manager (%1) could not be found in your $PATH ", downloadManger);
404  QString errMsgEx = i18n("Try to reinstall it \n\nThe integration with Konqueror will be disabled.");
405  KMessageBox::detailedSorry(0, errMsg, errMsgEx);
406  cfg.writePathEntry("DownloadManager", QString());
407  cfg.sync();
408  } else {
409  // ### suggestedFileName not taken into account. Fix this (and
410  // the duplicated code) with shiny new KDownload class for 3.2 (pfeiffer)
411  // Until the shiny new class comes about, send the suggestedFileName
412  // along with the actual URL to download. (DA)
413  cmd += QLatin1Char(' ') + KShell::quoteArg(url.toString());
414  if (!suggestedFileName.isEmpty()) {
415  cmd += QLatin1Char(' ') + KShell::quoteArg(suggestedFileName);
416  }
417 
418  // qDebug() << "Calling command" << cmd;
419  // slave is already on hold (slotBrowserMimetype())
420  KIO::Scheduler::publishSlaveOnHold();
421  KRun::runCommand(cmd, window);
422  return;
423  }
424  }
425  }
426 
427  // no download manager available, let's do it ourself
428  QFileDialog *dlg = new QFileDialog(window);
429  dlg->setAcceptMode(QFileDialog::AcceptSave);
430  dlg->setWindowTitle(i18n("Save As"));
431  dlg->setConfirmOverwrite(true);
432 
433  QString name;
434  if (!suggestedFileName.isEmpty()) {
435  name = suggestedFileName;
436  } else {
437  name = url.fileName(); // can be empty, e.g. in case http://www.kde.org/
438  }
439 
440  dlg->selectFile(name);
441  if (dlg->exec()) {
442  QUrl destURL(dlg->selectedUrls().at(0));
443  if (destURL.isValid()) {
444  saveUrlUsingKIO(url, destURL, window, args.metaData());
445  }
446  }
447  delete dlg;
448 }

467 void BrowserRun::handleError(KJob *job)
468 {
469  if (!job) { // Shouldn't happen
470  qWarning() << "handleError called with job=0! hideErrorDialog=" << d->m_bHideErrorDialog;
471  return;
472  }
473 
474  KIO::TransferJob *tjob = qobject_cast<KIO::TransferJob *>(job);
475  if (tjob && tjob->isErrorPage() && !job->error()) {
476  // The default handling of error pages is to show them like normal pages
477  // But this is done here in handleError so that KHTMLRun can reimplement it
478  tjob->putOnHold();
479  setJob(0);
480  if (!d->m_mimeType.isEmpty()) {
481  mimeTypeDetermined(d->m_mimeType);
482  }
483  return;
484  }
485 
486  if (d->m_bHideErrorDialog && job->error() != KIO::ERR_NO_CONTENT) {
487  redirectToError(job->error(), job->errorText());
488  return;
489  }
490 
491  // Reuse code in KRun, to benefit from d->m_showingError etc.
492  KRun::handleError(job);

495 // static
496 QUrl BrowserRun::makeErrorUrl(int error, const QString &errorText, const QUrl &initialUrl)
497 {
498  /*
499  * The format of the error:/ URL is error:/?query#url,
500  * where two variables are passed in the query:
501  * error = int kio error code, errText = QString error text from kio
502  * The sub-url is the URL that we were trying to open.
503  */
504  QUrl newURL(QStringLiteral("error:/?error=%1&errText=%2")
505  .arg(error)
506  .arg(QString::fromUtf8(QUrl::toPercentEncoding(errorText))));
507 
508  QString cleanedOrigUrl = initialUrl.toString();
509  QUrl runURL(cleanedOrigUrl);
510  if (runURL.isValid()) {
511  runURL.setPassword(QString()); // don't put the password in the error URL
512  cleanedOrigUrl = runURL.toString();
513  }
514 
515  newURL.setFragment(cleanedOrigUrl);
516  return newURL;
517 }
