47 Part::~Part()
48 {
49  Q_D(Part);
50 
51  //qDebug() << this;
52 
53  if (d->m_widget) {
54  // We need to disconnect first, to avoid calling it !
55  disconnect(d->m_widget.data(), &QWidget::destroyed,
56  this, &Part::slotWidgetDestroyed);
57  }
58 
59  if (d->m_manager) {
60  d->m_manager->removePart(this);
61  }
62 
63  if (d->m_widget && d->m_autoDeleteWidget) {
64  // qDebug() << "deleting widget" << d->m_widget << d->m_widget->objectName();
65  delete static_cast<QWidget *>(d->m_widget);
66  }
67 
68  delete d->m_iconLoader;
69 }

156 void Part::customEvent(QEvent *ev)
157 {
158  if (PartActivateEvent::test(ev)) {
159  partActivateEvent(static_cast<PartActivateEvent *>(ev));
160  return;
161  }
162 
163  if (PartSelectEvent::test(ev)) {
164  partSelectEvent(static_cast<PartSelectEvent *>(ev));
165  return;
166  }
167 
168  if (GUIActivateEvent::test(ev)) {
169  guiActivateEvent(static_cast<GUIActivateEvent *>(ev));
170  return;
171  }
172 
173  QObject::customEvent(ev);
174 }
