143  int executeDialog(const QString &dontShowAgainName)
144  {
145  KConfigGroup cg(dontAskConfig, "Notification Messages"); // group name comes from KMessageBox
146  const QString dontAsk = cg.readEntry(dontShowAgainName, QString()).toLower();
147  if (dontAsk == QLatin1String("yes") || dontAsk == QLatin1String("true")) {
148  return Save;
149  } else if (dontAsk == QLatin1String("no") || dontAsk == QLatin1String("false")) {
150  return OpenDefault;
151  }
152 
153  const int result = exec();
154 
155  if (dontAskAgainCheckBox->isChecked()) {
156  cg.writeEntry(dontShowAgainName, result == BrowserOpenOrSaveQuestion::Save);
157  cg.sync();
158  }
159  return result;
160  }

240 BrowserOpenOrSaveQuestion::Result BrowserOpenOrSaveQuestion::askOpenOrSave()
241 {
242  d->questionLabel->setText(i18nc("@info", "Open '%1'?", d->url.toDisplayString(QUrl::PreferLocalFile)));
243  d->questionLabel->setTextInteractionFlags(Qt::TextSelectableByMouse);
244  d->openWithButton->hide();
245 
246  KGuiItem openWithDialogItem(i18nc("@label:button", "&Open with..."), QStringLiteral("document-open"));
247 
248  // I thought about using KFileItemActions, but we don't want a submenu, nor the slots....
249  // and we want no menu at all if there's only one offer.
250  // TODO: we probably need a setTraderConstraint(), to exclude the current application?
251  const KService::List apps = KFileItemActions::associatedApplications(QStringList() << d->mimeType,
252  QString() /* TODO trader constraint */);
253  if (apps.isEmpty()) {
254  KGuiItem::assign(d->openDefaultButton, openWithDialogItem);
255  } else {
256  KService::Ptr offer = apps.first();
257  KGuiItem openItem(i18nc("@label:button", "&Open with %1", offer->name()), offer->icon());
258  KGuiItem::assign(d->openDefaultButton, openItem);
259  if (d->features & ServiceSelection) {
260  // OpenDefault shall use this service
261  d->selectedService = apps.first();
262  d->openWithButton->show();
263  QMenu *menu = new QMenu(d);
264  if (apps.count() > 1) {
265  // Provide an additional button with a menu of associated apps
266  KGuiItem openWithItem(i18nc("@label:button", "&Open with"), QStringLiteral("document-open"));
267  KGuiItem::assign(d->openWithButton, openWithItem);
268  d->openWithButton->setMenu(menu);
269  QObject::connect(menu, &QMenu::triggered, d, &BrowserOpenOrSaveQuestionPrivate::slotAppSelected);
270  for (KService::List::const_iterator it = apps.begin(); it != apps.end(); ++it) {
271  QAction *act = createAppAction(*it, d);
272  menu->addAction(act);
273  }
274  QAction *openWithDialogAction = new QAction(d);
275  openWithDialogAction->setIcon(QIcon::fromTheme(QStringLiteral("document-open")));
276  openWithDialogAction->setText(openWithDialogItem.text());
277  menu->addAction(openWithDialogAction);
278  } else {
279  // Only one associated app, already offered by the other menu -> add "Open With..." button
280  KGuiItem::assign(d->openWithButton, openWithDialogItem);
281  }
282  } else {
283  // qDebug() << "Not using new feature ServiceSelection; port the caller to BrowserOpenOrSaveQuestion::setFeature(ServiceSelection)";
284  }
285  }
286 
287  // KEEP IN SYNC with kdebase/runtime/keditfiletype/filetypedetails.cpp!!!
288  const QString dontAskAgain = QLatin1String("askSave") + d->mimeType;
289 
290  const int choice = d->executeDialog(dontAskAgain);
291  return choice == BrowserOpenOrSaveQuestionPrivate::Save ? Save
292  : (choice == BrowserOpenOrSaveQuestionPrivate::Cancel ? Cancel : Open);
293 }

300 bool BrowserOpenOrSaveQuestionPrivate::autoEmbedMimeType(int flags)
301 {
302  // SYNC SYNC SYNC SYNC SYNC SYNC SYNC SYNC SYNC SYNC SYNC SYNC SYNC SYNC
303  // NOTE: Keep this function in sync with
304  // kdebase/runtime/keditfiletype/filetypedetails.cpp
305  // FileTypeDetails::updateAskSave()
306 
307  // Don't ask for:
308  // - html (even new tabs would ask, due to about:blank!)
309  // - dirs obviously (though not common over HTTP :),
310  // - images (reasoning: no need to save, most of the time, because fast to see)
311  // e.g. postscript is different, because takes longer to read, so
312  // it's more likely that the user might want to save it.
313  // - multipart/* ("server push", see kmultipart)
314  // KEEP IN SYNC!!!
315  if (flags != (int)BrowserRun::AttachmentDisposition && mime.isValid() && (
316  mime.inherits(QStringLiteral("text/html")) ||
317  mime.inherits(QStringLiteral("application/xml")) ||
318  mime.inherits(QStringLiteral("inode/directory")) ||
319  mimeType.startsWith(QLatin1String("image")) ||
320  mime.inherits(QStringLiteral("multipart/x-mixed-replace")) ||
321  mime.inherits(QStringLiteral("multipart/replace")))) {
322  return true;
323  }
324  return false;
325 }
