78 void MainWindow::createGUI(Part *part)
79 {
80 #if 0
81  // qDebug() << "part=" << part
82  << (part ? part->metaObject()->className() : "")
83  << (part ? part->objectName() : "");
84 #endif
85  KXMLGUIFactory *factory = guiFactory();
86 
87  Q_ASSERT(factory);
88 
89  if (d->m_activePart) {
90 #if 0
91  // qDebug() << "deactivating GUI for" << d->m_activePart
92  << d->m_activePart->metaObject()->className()
93  << d->m_activePart->objectName();
94 #endif
95 
96  GUIActivateEvent ev(false);
97  QApplication::sendEvent(d->m_activePart, &ev);
98 
99  factory->removeClient(d->m_activePart);
100 
101  disconnect(d->m_activePart.data(), &Part::setWindowCaption,
102  this, static_cast<void (MainWindow::*)(const QString &)>(&MainWindow::setCaption));
103  disconnect(d->m_activePart.data(), &Part::setStatusBarText,
104  this, &MainWindow::slotSetStatusBarText);
105  }
106 
107  if (!d->m_bShellGUIActivated) {
108  loadPlugins(this, this, KAboutData::applicationData());
109  createShellGUI();
110  d->m_bShellGUIActivated = true;
111  }
112 
113  if (part) {
114  // do this before sending the activate event
115  if (d->m_manageWindowTitle) {
116  connect(part, &Part::setWindowCaption,
117  this, static_cast<void (MainWindow::*)(const QString &)>(&MainWindow::setCaption));
118  }
119  connect(part, &Part::setStatusBarText,
120  this, &MainWindow::slotSetStatusBarText);
121 
122  factory->addClient(part);
123 
124  GUIActivateEvent ev(true);
125  QApplication::sendEvent(part, &ev);
126  }
127 
128  d->m_activePart = part;
129 }

136 void MainWindow::createShellGUI(bool create)
137 {
138  Q_ASSERT(d->m_bShellGUIActivated != create);
139  d->m_bShellGUIActivated = create;
140  if (create) {
141  if (isHelpMenuEnabled() && !d->m_helpMenu) {
142  d->m_helpMenu = new KHelpMenu(this, KAboutData::applicationData(), true);
143 
144  KActionCollection *actions = actionCollection();
145  QAction *helpContentsAction = d->m_helpMenu->action(KHelpMenu::menuHelpContents);
146  QAction *whatsThisAction = d->m_helpMenu->action(KHelpMenu::menuWhatsThis);
147  QAction *reportBugAction = d->m_helpMenu->action(KHelpMenu::menuReportBug);
148  QAction *switchLanguageAction = d->m_helpMenu->action(KHelpMenu::menuSwitchLanguage);
149  QAction *aboutAppAction = d->m_helpMenu->action(KHelpMenu::menuAboutApp);
150  QAction *aboutKdeAction = d->m_helpMenu->action(KHelpMenu::menuAboutKDE);
151  QAction *donateAction = d->m_helpMenu->action(KHelpMenu::menuDonate);
152 
153  if (helpContentsAction) {
154  actions->addAction(helpContentsAction->objectName(), helpContentsAction);
155  }
156  if (whatsThisAction) {
157  actions->addAction(whatsThisAction->objectName(), whatsThisAction);
158  }
159  if (reportBugAction) {
160  actions->addAction(reportBugAction->objectName(), reportBugAction);
161  }
162  if (switchLanguageAction) {
163  actions->addAction(switchLanguageAction->objectName(), switchLanguageAction);
164  }
165  if (aboutAppAction) {
166  actions->addAction(aboutAppAction->objectName(), aboutAppAction);
167  }
168  if (aboutKdeAction) {
169  actions->addAction(aboutKdeAction->objectName(), aboutKdeAction);
170  }
171  if (donateAction) {
172  actions->addAction(donateAction->objectName(), donateAction);
173  }
174  }
175 
176  QString f = xmlFile();
177  setXMLFile(KXMLGUIClient::standardsXmlFileLocation());
178  if (!f.isEmpty()) {
179  setXMLFile(f, true);
180  } else {
181  QString auto_file(componentName() + QStringLiteral("ui.rc"));
182  setXMLFile(auto_file, true);
183  }
184 
185  GUIActivateEvent ev(true);
186  QApplication::sendEvent(this, &ev);
187 
188  guiFactory()->addClient(this);
189  } else {
190  GUIActivateEvent ev(false);
191  QApplication::sendEvent(this, &ev);
192 
193  guiFactory()->removeClient(this);
194  }
195 }
