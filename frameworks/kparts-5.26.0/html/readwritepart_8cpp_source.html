83 bool ReadWritePart::queryClose()
84 {
85  Q_D(ReadWritePart);
86 
87  if (!isReadWrite() || !isModified()) {
88  return true;
89  }
90 
91  QString docName = url().fileName();
92  if (docName.isEmpty()) {
93  docName = i18n("Untitled");
94  }
95 
96  QWidget *parentWidget = widget();
97  if (!parentWidget) {
98  parentWidget = QApplication::activeWindow();
99  }
100 
101  int res = KMessageBox::warningYesNoCancel(parentWidget,
102  i18n("The document \"%1\" has been modified.\n"
103  "Do you want to save your changes or discard them?", docName),
104  i18n("Close Document"), KStandardGuiItem::save(), KStandardGuiItem::discard());
105 
106  bool abortClose = false;
107  bool handled = false;
108 
109  switch (res) {
110  case KMessageBox::Yes :
111  sigQueryClose(&handled, &abortClose);
112  if (!handled) {
113  if (d->m_url.isEmpty()) {
114  QUrl url = QFileDialog::getSaveFileUrl(parentWidget);
115  if (url.isEmpty()) {
116  return false;
117  }
118 
119  saveAs(url);
120  } else {
121  save();
122  }
123  } else if (abortClose) {
124  return false;
125  }
126  return waitSaveComplete();
127  case KMessageBox::No :
128  return true;
129  default : // case KMessageBox::Cancel :
130  return false;
131  }
132 }

167 bool ReadWritePart::saveAs(const QUrl &url)
168 {
169  Q_D(ReadWritePart);
170 
171  if (!url.isValid()) {
172  qCritical() << "saveAs: Malformed URL " << url << endl;
173  return false;
174  }
175  d->m_duringSaveAs = true;
176  d->m_originalURL = d->m_url;
177  d->m_originalFilePath = d->m_file;
178  d->m_url = url; // Store where to upload in saveToURL
179  d->prepareSaving();
180  bool result = save(); // Save local file and upload local file
181  if (result) {
182  if (d->m_originalURL != d->m_url) {
183  emit urlChanged(d->m_url);
184  }
185 
186  emit setWindowCaption(d->m_url.toDisplayString());
187  } else {
188  d->m_url = d->m_originalURL;
189  d->m_file = d->m_originalFilePath;
190  d->m_duringSaveAs = false;
191  d->m_originalURL = QUrl();
192  d->m_originalFilePath.clear();
193  }
194 
195  return result;
196 }

198 // Set m_file correctly for m_url
199 void ReadWritePartPrivate::prepareSaving()
200 {
201  // Local file
202  if (m_url.isLocalFile()) {
203  if (m_bTemp) { // get rid of a possible temp file first
204  // (happens if previous url was remote)
205  QFile::remove(m_file);
206  m_bTemp = false;
207  }
208  m_file = m_url.toLocalFile();
209  } else {
210  // Remote file
211  // We haven't saved yet, or we did but locally - provide a temp file
212  if (m_file.isEmpty() || !m_bTemp) {
213  QTemporaryFile tempFile;
214  tempFile.setAutoRemove(false);
215  tempFile.open();
216  m_file = tempFile.fileName();
217  m_bTemp = true;
218  }
219  // otherwise, we already had a temp file
220  }
221 }

223 static inline bool makeHardLink(const QString& src, const QString& dest)
224 {
225 #ifndef Q_OS_WIN
226  return ::link(QFile::encodeName(src).constData(), QFile::encodeName(dest).constData()) == 0;

232 bool ReadWritePart::saveToUrl()
233 {
234  Q_D(ReadWritePart);
235 
236  if (d->m_url.isLocalFile()) {
237  setModified(false);
238  emit completed();
239  // if m_url is a local file there won't be a temp file -> nothing to remove
240  Q_ASSERT(!d->m_bTemp);
241  d->m_saveOk = true;
242  d->m_duringSaveAs = false;
243  d->m_originalURL = QUrl();
244  d->m_originalFilePath.clear();
245  return true; // Nothing to do
246  } else {
247  if (d->m_uploadJob) {
248  QFile::remove(d->m_uploadJob->srcUrl().toLocalFile());
249  d->m_uploadJob->kill();
250  d->m_uploadJob = 0;
251  }
252  QTemporaryFile *tempFile = new QTemporaryFile();
253  tempFile->open();
254  QString uploadFile = tempFile->fileName();
255  delete tempFile;
256  QUrl uploadUrl = QUrl::fromLocalFile(uploadFile);
257  // Create hardlink
258  if (!makeHardLink(d->m_file, uploadFile)) {
259  // Uh oh, some error happened.
260  return false;
261  }
262  d->m_uploadJob = KIO::file_move(uploadUrl, d->m_url, -1, KIO::Overwrite);
263  KJobWidgets::setWindow(d->m_uploadJob, widget());
264  connect(d->m_uploadJob, SIGNAL(result(KJob*)), this, SLOT(_k_slotUploadFinished(KJob*)));
265  return true;
266  }
267 }

269 void ReadWritePartPrivate::_k_slotUploadFinished(KJob *)
270 {
271  Q_Q(ReadWritePart);
272 
273  if (m_uploadJob->error()) {
274  QFile::remove(m_uploadJob->srcUrl().toLocalFile());
275  QString error = m_uploadJob->errorString();
276  m_uploadJob = 0;
277  if (m_duringSaveAs) {
278  q->setUrl(m_originalURL);
279  m_file = m_originalFilePath;
280  }
281  emit q->canceled(error);
282  } else {
283  ::org::kde::KDirNotify::emitFilesAdded(m_url.adjusted(QUrl::RemoveFilename));
284 
285  m_uploadJob = 0;
286  q->setModified(false);
287  emit q->completed();
288  m_saveOk = true;
289  }
290  m_duringSaveAs = false;
291  m_originalURL = QUrl();
292  m_originalFilePath.clear();
293  if (m_waitForSave) {
294  m_eventLoop.quit();
295  }
296 }
