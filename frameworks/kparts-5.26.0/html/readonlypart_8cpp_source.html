128 bool ReadOnlyPart::openUrl(const QUrl &url)
129 {
130  Q_D(ReadOnlyPart);
131 
132  if (!url.isValid()) {
133  return false;
134  }
135  if (d->m_bAutoDetectedMime) {
136  d->m_arguments.setMimeType(QString());
137  d->m_bAutoDetectedMime = false;
138  }
139  OpenUrlArguments args = d->m_arguments;
140  if (!closeUrl()) {
141  return false;
142  }
143  d->m_arguments = args;
144  setUrl(url);
145 
146  d->m_file.clear();
147 
148  if (d->m_url.isLocalFile()) {
149  d->m_file = d->m_url.toLocalFile();
150  return d->openLocalFile();
151  } else if (KProtocolInfo::protocolClass(url.scheme()) == QLatin1String(":local")) {
152  // Maybe we can use a "local path", to avoid a temp copy?
153  KIO::JobFlags flags = d->m_showProgressInfo ? KIO::DefaultFlags : KIO::HideProgressInfo;
154  d->m_statJob = KIO::mostLocalUrl(d->m_url, flags);
155  KJobWidgets::setWindow(d->m_statJob, widget());
156  connect(d->m_statJob, SIGNAL(result(KJob*)), this, SLOT(_k_slotStatJobFinished(KJob*)));
157  return true;
158  } else {
159  d->openRemoteFile();
160  return true;
161  }
162 }

171 bool ReadOnlyPartPrivate::openLocalFile()
172 {
173  Q_Q(ReadOnlyPart);
174  emit q->started(0);
175  m_bTemp = false;
176  // set the mimetype only if it was not already set (for example, by the host application)
177  if (m_arguments.mimeType().isEmpty()) {
178  // get the mimetype of the file
179  // using findByUrl() to avoid another string -> url conversion
180  QMimeDatabase db;
181  QMimeType mime = db.mimeTypeForUrl(m_url);
182  if (!mime.isDefault()) {
183  m_arguments.setMimeType(mime.name());
184  m_bAutoDetectedMime = true;
185  }
186  }
187  const bool ret = q->openFile();
188  if (ret) {
189  emit q->setWindowCaption(m_url.toDisplayString());
190  emit q->completed();
191  } else {
192  emit q->canceled(QString());
193  }
194  return ret;
195 }

197 void ReadOnlyPartPrivate::openRemoteFile()
198 {
199  Q_Q(ReadOnlyPart);
200  m_bTemp = true;
201  // Use same extension as remote file. This is important for mimetype-determination (e.g. koffice)
202  QString fileName = m_url.fileName();
203  QFileInfo fileInfo(fileName);
204  QString ext = fileInfo.completeSuffix();
205  QString extension;
206  if (!ext.isEmpty() && !m_url.hasQuery()) { // not if the URL has a query, e.g. cgi.pl?something
207  extension = QLatin1Char('.') + ext; // keep the '.'
208  }
209  QTemporaryFile tempFile(QDir::tempPath() + QLatin1Char('/') + q->componentData().componentName() + QLatin1String("XXXXXX") + extension);
210  tempFile.setAutoRemove(false);
211  tempFile.open();
212  m_file = tempFile.fileName();
213 
214  QUrl destURL = QUrl::fromLocalFile(m_file);
215  KIO::JobFlags flags = m_showProgressInfo ? KIO::DefaultFlags : KIO::HideProgressInfo;
216  flags |= KIO::Overwrite;
217  m_job = KIO::file_copy(m_url, destURL, 0600, flags);
218  KJobWidgets::setWindow(m_job, q->widget());
219  emit q->started(m_job);
220  QObject::connect(m_job, SIGNAL(result(KJob*)), q, SLOT(_k_slotJobFinished(KJob*)));
221  QObject::connect(m_job, SIGNAL(mimetype(KIO::Job*,QString)),
222  q, SLOT(_k_slotGotMimeType(KIO::Job*,QString)));
223 }

241 bool ReadOnlyPart::closeUrl()
242 {
243  Q_D(ReadOnlyPart);
244 
245  abortLoad(); //just in case
246 
247  d->m_arguments = KParts::OpenUrlArguments();
248 
249  if (d->m_bTemp) {
250  QFile::remove(d->m_file);
251  d->m_bTemp = false;
252  }
253  // It always succeeds for a read-only part,
254  // but the return value exists for reimplementations
255  // (e.g. pressing cancel for a modified read-write part)
256  return true;
257 }

259 void ReadOnlyPartPrivate::_k_slotStatJobFinished(KJob *job)
260 {
261  Q_ASSERT(job == m_statJob);
262  m_statJob = 0;
263 
264  // We could emit canceled on error, but we haven't even emitted started yet,
265  // this could maybe confuse some apps? So for now we'll just fallback to KIO::get
266  // and error again. Well, maybe this even helps with wrong stat results.
267  if (!job->error()) {
268  const QUrl localUrl = static_cast<KIO::StatJob *>(job)->mostLocalUrl();
269  if (localUrl.isLocalFile()) {
270  m_file = localUrl.toLocalFile();
271  (void)openLocalFile();
272  return;
273  }
274  }
275  openRemoteFile();
276 }

278 void ReadOnlyPartPrivate::_k_slotJobFinished(KJob *job)
279 {
280  Q_Q(ReadOnlyPart);
281 
282  Q_ASSERT(job == m_job);
283  m_job = 0;
284  if (job->error()) {
285  emit q->canceled(job->errorString());
286  } else {
287  if (q->openFile()) {
288  emit q->setWindowCaption(m_url.toDisplayString());
289  emit q->completed();
290  } else {
291  emit q->canceled(QString());
292  }
293  }
294 }
