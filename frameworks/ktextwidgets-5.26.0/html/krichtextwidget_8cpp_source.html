544 void KRichTextWidget::Private::_k_updateCharFormatActions(const QTextCharFormat &format)
545 {
546  QFont f = format.font();
547 
548  if (richTextSupport & SupportFontFamily) {
549  action_font_family->setFont(f.family());
550  }
551  if (richTextSupport & SupportFontSize) {
552  if (f.pointSize() > 0) {
553  action_font_size->setFontSize((int)f.pointSize());
554  }
555  }
556 
557  if (richTextSupport & SupportBold) {
558  action_text_bold->setChecked(f.bold());
559  }
560 
561  if (richTextSupport & SupportItalic) {
562  action_text_italic->setChecked(f.italic());
563  }
564 
565  if (richTextSupport & SupportUnderline) {
566  action_text_underline->setChecked(f.underline());
567  }
568 
569  if (richTextSupport & SupportStrikeOut) {
570  action_text_strikeout->setChecked(f.strikeOut());
571  }
