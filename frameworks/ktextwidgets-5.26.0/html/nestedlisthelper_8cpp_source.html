105 bool NestedListHelper::handleAfterKeyPressEvent(QKeyEvent *event)
106 {
107  // Only attempt to handle Backspace and Return
108  if ((event->key() != Qt::Key_Backspace)
109  && (event->key() != Qt::Key_Return)) {
110  return false;
111  }
112 
113  QTextCursor cursor = textEdit->textCursor();
114  bool handled = false;
115 
116  if (!cursor.hasSelection() && cursor.currentList()) {
117 
118  // Check if we're on the last list item.
119  // itemNumber is zero indexed
120  QTextBlock currentBlock = cursor.block();
121  if (cursor.currentList()->count() == cursor.currentList()->itemNumber(currentBlock) + 1) {
122  // Last block in this list, but may have just gained another list below.
123  if (currentBlock.next().textList()) {
124  reformatList();
125  }
126 
127  // No need to reformatList in this case. reformatList is slow.
128  if ((event->key() == Qt::Key_Return) || (event->key() == Qt::Key_Backspace)) {
129  reformatBoundingItemSpacing();
130  handled = true;
131  }
132  } else {
133  reformatList();
134  }
135  }
136  return handled;
137 }

171 void NestedListHelper::processList(QTextList *list)
172 {
173  QTextBlock block = list->item(0);
174  int thisListIndent = list->format().indent();
175 
176  QTextCursor cursor = QTextCursor(block);
177  list = cursor.createList(list->format());
178  bool processingSubList = false;
179  while (block.next().textList() != 0) {
180  block = block.next();
181 
182  QTextList *nextList = block.textList();
183  int nextItemIndent = nextList->format().indent();
184  if (nextItemIndent < thisListIndent) {
185  return;
186  } else if (nextItemIndent > thisListIndent) {
187  if (processingSubList) {
188  continue;
189  }
190  processingSubList = true;
191  processList(nextList);
192  } else {
193  processingSubList = false;
194  list->add(block);
195  }
196  }

198 // nextList = 0;

245 void NestedListHelper::handleOnIndentMore()
246 {
247  QTextCursor cursor = textEdit->textCursor();
248 
249  QTextListFormat listFmt;
250  if (!cursor.currentList()) {
251 
252  QTextListFormat::Style style;
253  cursor = topOfSelection();
254  cursor.movePosition(QTextCursor::PreviousBlock);
255  if (cursor.currentList()) {
256  style = cursor.currentList()->format().style();
257  } else {
258 
259  cursor = bottomOfSelection();
260  cursor.movePosition(QTextCursor::NextBlock);
261 
262  if (cursor.currentList()) {
263  style = cursor.currentList()->format().style();
264  } else {
265  style = QTextListFormat::ListDisc;
266  }
267  }
268  handleOnBulletType(style);
269  } else {
270  listFmt = cursor.currentList()->format();
271  listFmt.setIndent(listFmt.indent() + 1);
272 
273  cursor.createList(listFmt);
274  reformatList();
275  }
276 
277  reformatBoundingItemSpacing();
278 }

280 void NestedListHelper::handleOnIndentLess()
281 {
282  QTextCursor cursor = textEdit->textCursor();
283  QTextList *currentList = cursor.currentList();
284  if (!currentList) {
285  return;
286  }
287  QTextListFormat listFmt;
288  listFmt = currentList->format();
289  if (listFmt.indent() > 1) {
290  listFmt.setIndent(listFmt.indent() - 1);
291  cursor.createList(listFmt);
292  reformatList(cursor.block());
293  } else {
294  QTextBlockFormat bfmt;
295  bfmt.setObjectIndex(-1);
296  cursor.setBlockFormat(bfmt);
297  reformatList(cursor.block().next());
298  }
299  reformatBoundingItemSpacing();
300 }

333 void NestedListHelper::reformatBoundingItemSpacing(QTextBlock block)
334 {
335  // This is a workaround for qt bug 201228. Spacing between items is not kept
336  // consistent.
337  // Fixed scheduled for qt4.5
338  // -- Stephen Kelly, 8th June 2008
339 
340  int nextBlockTopMargin = listNoMargin;
341  int previousBlockBottomMargin = listNoMargin;
342  int thisBlockBottomMargin = listBottomMargin;
343  int thisBlockTopMargin = listTopMargin;
344  bool prevBlockValid = block.previous().isValid();
345  bool nextBlockValid = block.next().isValid();
346 
347  if (block.textList()) {
348  if (prevBlockValid && block.previous().textList()) {
349  thisBlockTopMargin = listNoMargin;
350  }
351 
352  if (nextBlockValid && block.next().textList()) {
353  thisBlockBottomMargin = listNoMargin;
354  }
355  } else {
356  if (prevBlockValid && !block.previous().textList()) {
357  thisBlockTopMargin = listNoMargin;
358  }
359  if (nextBlockValid && !block.next().textList()) {
360  thisBlockBottomMargin = listNoMargin;
361  }
362 
363  }
364  QTextBlockFormat fmt;
365  QTextCursor cursor;
366 
367  fmt = block.blockFormat();
368  fmt.setBottomMargin(thisBlockBottomMargin);
369  fmt.setTopMargin(thisBlockTopMargin);
370  cursor = QTextCursor(block);
371  cursor.setBlockFormat(fmt);
372 
373  if (nextBlockValid) {
374  block = block.next();
375  fmt = block.blockFormat();
376  fmt.setTopMargin(nextBlockTopMargin);
377  cursor = QTextCursor(block);
378  cursor.setBlockFormat(fmt);
379 
380  block = block.previous();
381  }
382  if (prevBlockValid) {
383  block = block.previous();
384  fmt = block.blockFormat();
385  fmt.setBottomMargin(previousBlockBottomMargin);
386  cursor = QTextCursor(block);
387  cursor.setBlockFormat(fmt);
388  }
389 }
