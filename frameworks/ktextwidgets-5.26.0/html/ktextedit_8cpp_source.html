163 void KTextEdit::Private::checkSpelling(bool force)
164 {
165  if (parent->document()->isEmpty()) {
166  KMessageBox::information(parent, i18n("Nothing to spell check."));
167  if (force) {
168  emit parent->spellCheckingFinished();
169  }
170  return;
171  }
172  Sonnet::BackgroundChecker *backgroundSpellCheck = new Sonnet::BackgroundChecker;
173  if (!spellCheckingLanguage.isEmpty()) {
174  backgroundSpellCheck->changeLanguage(spellCheckingLanguage);
175  }
176  Sonnet::Dialog *spellDialog = new Sonnet::Dialog(
177  backgroundSpellCheck, force ? parent : 0);
178  backgroundSpellCheck->setParent(spellDialog);
179  spellDialog->setAttribute(Qt::WA_DeleteOnClose, true);
180  spellDialog->activeAutoCorrect(showAutoCorrectionButton);
181  connect(spellDialog, SIGNAL(replace(QString,int,QString)),
182  parent, SLOT(spellCheckerCorrected(QString,int,QString)));
183  connect(spellDialog, SIGNAL(misspelling(QString,int)),
184  parent, SLOT(spellCheckerMisspelling(QString,int)));
185  connect(spellDialog, &Sonnet::Dialog::autoCorrect,
186  parent, &KTextEdit::spellCheckerAutoCorrect);
187  connect(spellDialog, SIGNAL(done(QString)),
188  parent, SLOT(spellCheckerFinished()));
189  connect(spellDialog, SIGNAL(cancel()),
190  parent, SLOT(spellCheckerCanceled()));
191  //Laurent in sonnet/dialog.cpp we emit done(QString) too => it calls here twice spellCheckerFinished not necessary
192  /*
193  connect(spellDialog, SIGNAL(stop()),
194  parent, SLOT(spellCheckerFinished()));
195  */
196  connect(spellDialog, &Sonnet::Dialog::spellCheckStatus,
197  parent, &KTextEdit::spellCheckStatus);
198  connect(spellDialog, &Sonnet::Dialog::languageChanged,
199  parent, &KTextEdit::languageChanged);
200  if (force) {
201  connect(spellDialog, SIGNAL(done(QString)), parent, SIGNAL(spellCheckingFinished()));
202  connect(spellDialog, &Sonnet::Dialog::cancel, parent, &KTextEdit::spellCheckingCanceled);
203  //Laurent in sonnet/dialog.cpp we emit done(QString) too => it calls here twice spellCheckerFinished not necessary
204  //connect(spellDialog, SIGNAL(stop()), parent, SIGNAL(spellCheckingFinished()));
205  }
206  originalDoc = QTextDocumentFragment(parent->document());
207  spellDialog->setBuffer(parent->toPlainText());
208  spellDialog->show();
209 }

294 void KTextEdit::Private::slotReplaceText(const QString &text, int replacementIndex, int replacedLength, int matchedLength)
295 {
296  //qDebug() << "Replace: [" << text << "] ri:" << replacementIndex << " rl:" << replacedLength << " ml:" << matchedLength;
297  QTextCursor tc = parent->textCursor();
298  tc.setPosition(replacementIndex);
299  tc.movePosition(QTextCursor::NextCharacter, QTextCursor::KeepAnchor, matchedLength);
300  tc.removeSelectedText();
301  tc.insertText(text.mid(replacementIndex, replacedLength));
302  if (replace->options() & KReplaceDialog::PromptOnReplace) {
303  parent->setTextCursor(tc);
304  parent->ensureCursorVisible();
305  }
306  lastReplacedPosition = replacementIndex;
307 }

388 bool KTextEdit::Private::handleShortcut(const QKeyEvent *event)
389 {
390  const int key = event->key() | event->modifiers();
391 
392  if (KStandardShortcut::copy().contains(key)) {
393  parent->copy();
394  return true;
395  } else if (KStandardShortcut::paste().contains(key)) {
396  parent->paste();
397  return true;
398  } else if (KStandardShortcut::cut().contains(key)) {
399  parent->cut();
400  return true;
401  } else if (KStandardShortcut::undo().contains(key)) {
402  if (!parent->isReadOnly()) {
403  parent->undo();
404  }
405  return true;
406  } else if (KStandardShortcut::redo().contains(key)) {
407  if (!parent->isReadOnly()) {
408  parent->redo();
409  }
410  return true;
411  } else if (KStandardShortcut::deleteWordBack().contains(key)) {
412  if (!parent->isReadOnly()) {
413  parent->deleteWordBack();
414  }
415  return true;
416  } else if (KStandardShortcut::deleteWordForward().contains(key)) {
417  if (!parent->isReadOnly()) {
418  parent->deleteWordForward();
419  }
420  return true;
421  } else if (KStandardShortcut::backwardWord().contains(key)) {
422  QTextCursor cursor = parent->textCursor();
423  cursor.movePosition(QTextCursor::PreviousWord);
424  parent->setTextCursor(cursor);
425  return true;
426  } else if (KStandardShortcut::forwardWord().contains(key)) {
427  QTextCursor cursor = parent->textCursor();
428  cursor.movePosition(QTextCursor::NextWord);
429  parent->setTextCursor(cursor);
430  return true;
431  } else if (KStandardShortcut::next().contains(key)) {
432  QTextCursor cursor = parent->textCursor();
433  bool moved = false;
434  qreal lastY = parent->cursorRect(cursor).bottom();
435  qreal distance = 0;
436  do {
437  qreal y = parent->cursorRect(cursor).bottom();
438  distance += qAbs(y - lastY);
439  lastY = y;
440  moved = cursor.movePosition(QTextCursor::Down);
441  } while (moved && distance < parent->viewport()->height());
442 
443  if (moved) {
444  cursor.movePosition(QTextCursor::Up);
445  parent->verticalScrollBar()->triggerAction(QAbstractSlider::SliderPageStepAdd);
446  }
447  parent->setTextCursor(cursor);
448  return true;
449  } else if (KStandardShortcut::prior().contains(key)) {
450  QTextCursor cursor = parent->textCursor();
451  bool moved = false;
452  qreal lastY = parent->cursorRect(cursor).bottom();
453  qreal distance = 0;
454  do {
455  qreal y = parent->cursorRect(cursor).bottom();
456  distance += qAbs(y - lastY);
457  lastY = y;
458  moved = cursor.movePosition(QTextCursor::Up);
459  } while (moved && distance < parent->viewport()->height());
460 
461  if (moved) {
462  cursor.movePosition(QTextCursor::Down);
463  parent->verticalScrollBar()->triggerAction(QAbstractSlider::SliderPageStepSub);
464  }
465  parent->setTextCursor(cursor);
466  return true;
467  } else if (KStandardShortcut::begin().contains(key)) {
468  QTextCursor cursor = parent->textCursor();
469  cursor.movePosition(QTextCursor::Start);
470  parent->setTextCursor(cursor);
471  return true;
472  } else if (KStandardShortcut::end().contains(key)) {
473  QTextCursor cursor = parent->textCursor();
474  cursor.movePosition(QTextCursor::End);
475  parent->setTextCursor(cursor);
476  return true;
477  } else if (KStandardShortcut::beginningOfLine().contains(key)) {
478  QTextCursor cursor = parent->textCursor();
479  cursor.movePosition(QTextCursor::StartOfLine);
480  parent->setTextCursor(cursor);
481  return true;
482  } else if (KStandardShortcut::endOfLine().contains(key)) {
483  QTextCursor cursor = parent->textCursor();
484  cursor.movePosition(QTextCursor::EndOfLine);
485  parent->setTextCursor(cursor);
486  return true;
487  } else if (findReplaceEnabled && KStandardShortcut::find().contains(key)) {
488  parent->slotFind();
489  return true;
490  } else if (findReplaceEnabled && KStandardShortcut::findNext().contains(key)) {
491  parent->slotFindNext();
492  return true;
493  } else if (findReplaceEnabled && KStandardShortcut::findPrev().contains(key)) {
494  parent->slotFindPrevious();
495  return true;
496  } else if (findReplaceEnabled && KStandardShortcut::replace().contains(key)) {
497  if (!parent->isReadOnly()) {
498  parent->slotReplace();
499  }
500  return true;
501  } else if (KStandardShortcut::pasteSelection().contains(key)) {
502  QString text = QApplication::clipboard()->text(QClipboard::Selection);
503  if (!text.isEmpty()) {
504  parent->insertPlainText(text); // TODO: check if this is html? (MiB)
505  }
506  return true;
507  }
508  return false;
509 }

528 QMenu *KTextEdit::mousePopupMenu()
529 {
530  QMenu *popup = createStandardContextMenu();
531  if (!popup) {
532  return 0;
533  }
534  connect(popup, SIGNAL(triggered(QAction*)),
535  this, SLOT(menuActivated(QAction*)));
536 
537  const bool emptyDocument = document()->isEmpty();
538  if (!isReadOnly()) {
539  QList<QAction *> actionList = popup->actions();
540  enum { UndoAct, RedoAct, CutAct, CopyAct, PasteAct, ClearAct, SelectAllAct, NCountActs };
541  QAction *separatorAction = 0L;
542  int idx = actionList.indexOf(actionList[SelectAllAct]) + 1;
543  if (idx < actionList.count()) {
544  separatorAction = actionList.at(idx);
545  }
546  if (separatorAction) {
547  QAction *clearAllAction = KStandardAction::clear(this, SLOT(undoableClear()), popup);
548  if (emptyDocument) {
549  clearAllAction->setEnabled(false);
550  }
551  popup->insertAction(separatorAction, clearAllAction);
552  }
553  }
554  KIconTheme::assignIconsToContextMenu(isReadOnly() ? KIconTheme::ReadOnlyText
555  : KIconTheme::TextEditor,
556  popup->actions());
557 
558  if (!isReadOnly()) {
559  popup->addSeparator();
560  if (!d->speller) {
561  d->speller = new Sonnet::Speller();
562  }
563  if (!d->speller->availableBackends().isEmpty()) {
564 
565  d->spellCheckAction = popup->addAction(QIcon::fromTheme(QStringLiteral("tools-check-spelling")),
566  i18n("Check Spelling..."));
567  if (emptyDocument) {
568  d->spellCheckAction->setEnabled(false);
569  }
570  if (checkSpellingEnabled()) {
571  d->languagesMenu = new QMenu(i18n("Spell Checking Language"), popup);
572  QActionGroup *languagesGroup = new QActionGroup(d->languagesMenu);
573  languagesGroup->setExclusive(true);
574 
575  QMapIterator<QString, QString> i(d->speller->availableDictionaries());
576  const QString language = spellCheckingLanguage();
577  while (i.hasNext()) {
578  i.next();
579 
580  QAction *languageAction = d->languagesMenu->addAction(i.key());
581  languageAction->setCheckable(true);
582  languageAction->setChecked(language == i.value() || (language.isEmpty()
583  && d->speller->defaultLanguage() == i.value()));
584  languageAction->setData(i.value());
585  languageAction->setActionGroup(languagesGroup);
586  connect(languageAction, &QAction::triggered,
587  [this, languageAction]() {
588  setSpellCheckingLanguage(languageAction->data().toString());
589  });
590  }
591  popup->addMenu(d->languagesMenu);
592  }
593 
594  d->autoSpellCheckAction = popup->addAction(i18n("Auto Spell Check"));
595  d->autoSpellCheckAction->setCheckable(true);
596  d->autoSpellCheckAction->setChecked(checkSpellingEnabled());
597  popup->addSeparator();
598  }
599  if (d->showTabAction) {
600  d->allowTab = popup->addAction(i18n("Allow Tabulations"));
601  d->allowTab->setCheckable(true);
602  d->allowTab->setChecked(!tabChangesFocus());
603  }
604  }
605 
606  if (d->findReplaceEnabled) {
607  QAction *findAction = KStandardAction::find(this, SLOT(slotFind()), popup);
608  QAction *findNextAction = KStandardAction::findNext(this, SLOT(slotFindNext()), popup);
609  QAction *findPrevAction = KStandardAction::findPrev(this, SLOT(slotFindPrevious()), popup);
610  if (emptyDocument) {
611  findAction->setEnabled(false);
612  findNextAction->setEnabled(false);
613  findPrevAction->setEnabled(false);
614  } else {
615  findNextAction->setEnabled(d->find != 0);
616  findPrevAction->setEnabled(d->find != 0);
617  }
618  popup->addSeparator();
619  popup->addAction(findAction);
620  popup->addAction(findNextAction);
621  popup->addAction(findPrevAction);
622 
623  if (!isReadOnly()) {
624  QAction *replaceAction = KStandardAction::replace(this, SLOT(slotReplace()), popup);
625  if (emptyDocument) {
626  replaceAction->setEnabled(false);
627  }
628  popup->addAction(replaceAction);
629  }
630  }
631 #ifdef HAVE_SPEECH
632  popup->addSeparator();
633  QAction *speakAction = popup->addAction(i18n("Speak Text"));
634  speakAction->setIcon(QIcon::fromTheme(QStringLiteral("preferences-desktop-text-to-speech")));
635  speakAction->setEnabled(!emptyDocument);
636  connect(speakAction, &QAction::triggered, this, &KTextEdit::slotSpeakText);
637 #endif
638  return popup;
639 }

815 void KTextEdit::slotDoReplace()
816 {

841  connect(d->replace, SIGNAL(replace(QString,int,int,int)),
842  this, SLOT(slotReplaceText(QString,int,int,int)));
843 
844  d->repDlg->close();
845  slotReplaceNext();

848 void KTextEdit::slotReplaceNext()
849 {
850  if (!d->replace) {
851  return;
852  }
853 
854  d->lastReplacedPosition = -1;
855  if (!(d->replace->options() & KReplaceDialog::PromptOnReplace)) {
856  textCursor().beginEditBlock(); // #48541
857  viewport()->setUpdatesEnabled(false);
858  }
859 
860  KFind::Result res = KFind::NoMatch;
861

877  }
