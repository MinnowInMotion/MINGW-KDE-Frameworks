577 void KFindDialog::KFindDialogPrivate::_k_slotOk()
578 {
579  // Nothing to find?
580  if (q->pattern().isEmpty()) {
581  KMessageBox::error(q, i18n("You must enter some text to search for."));
582  return;
583  }
584 
585  if (regExp->isChecked()) {
586  // Check for a valid regular expression.
587  QRegExp _regExp(q->pattern());
588 
589  if (!_regExp.isValid()) {
590  KMessageBox::error(q, i18n("Invalid regular expression."));
591  return;
592  }
593  }
594 
595  find->addToHistory(q->pattern());
596 
597  if (q->windowModality() != Qt::NonModal) {
598  q->accept();
599  }
600  emit q->okClicked();
601 }
