122 bool KFind::needData() const
123 {
124  // always true when d->text is empty.
125  if (d->options & KFind::FindBackwards)
126  // d->index==-1 and d->lastResult==Match means we haven't answered nomatch yet
127  // This is important in the "replace with a prompt" case.
128  {
129  return (d->index < 0 && d->lastResult != Match);

131  // "index over length" test removed: we want to get a nomatch before we set data again
132  // This is important in the "replace with a prompt" case.
133  {
134  return d->index == INDEX_NOMATCH;

143 void KFind::setData(int id, const QString &data, int startPos)
144 {
145  // cache the data for incremental find
146  if (d->options & KFind::FindIncremental) {
147  if (id != -1) {
148  d->customIds = true;
149  } else {
150  id = d->currentId + 1;
151  }
152 
153  Q_ASSERT(id <= d->data.size());
154 
155  if (id == d->data.size()) {
156  d->data.append(Private::Data(id, data, true));
157  } else {
158  d->data.replace(id, Private::Data(id, data, true));
159  }
160  Q_ASSERT(d->data.at(id).text == data);
161  }
162 
163  if (!(d->options & KFind::FindIncremental) || needData()) {
164  d->text = data;
165 
166  if (startPos != -1) {
167  d->index = startPos;
168  } else if (d->options & KFind::FindBackwards) {
169  d->index = d->text.length();
170  } else {
171  d->index = 0;
172  }
173 #ifdef DEBUG_FIND
174  //qDebug() << "setData: '" << d->text << "' d->index=" << d->index;
175 #endif
176  Q_ASSERT(d->index != INDEX_NOMATCH);
177  d->lastResult = NoMatch;
178 
179  d->currentId = id;

194 KFind::Result KFind::find()
195 {
196  Q_ASSERT(d->index != INDEX_NOMATCH || d->patternChanged);
197 
198  if (d->lastResult == Match && !d->patternChanged) {
199  // Move on before looking for the next match, _if_ we just found a match
200  if (d->options & KFind::FindBackwards) {
201  d->index--;
202  if (d->index == -1) { // don't call KFind::find with -1, it has a special meaning
203  d->lastResult = NoMatch;
204  return NoMatch;
205  }
206  } else {
207  d->index++;
208  }
209  }
210  d->patternChanged = false;
211 
212  if (d->options & KFind::FindIncremental) {
213  // if the current pattern is shorter than the matchedPattern we can
214  // probably look up the match in the incrementalPath
215  if (d->pattern.length() < d->matchedPattern.length()) {
216  Private::Match match;
217  if (!d->pattern.isEmpty()) {
218  match = d->incrementalPath.value(d->pattern);
219  } else if (d->emptyMatch) {
220  match = *d->emptyMatch;
221  }
222  QString previousPattern(d->matchedPattern);
223  d->matchedPattern = d->pattern;
224  if (!match.isNull()) {
225  bool clean = true;
226 
227  // find the first result backwards on the path that isn't dirty
228  while (d->data.at(match.dataId).dirty == true &&
229  !d->pattern.isEmpty()) {
230  d->pattern.truncate(d->pattern.length() - 1);
231 
232  match = d->incrementalPath.value(d->pattern);
233 
234  clean = false;
235  }
236 
237  // remove all matches that lie after the current match
238  while (d->pattern.length() < previousPattern.length()) {
239  d->incrementalPath.remove(previousPattern);
240  previousPattern.truncate(previousPattern.length() - 1);
241  }
242 
243  // set the current text, index, etc. to the found match
244  d->text = d->data.at(match.dataId).text;
245  d->index = match.index;
246  d->matchedLength = match.matchedLength;
247  d->currentId = match.dataId;
248 
249  // if the result is clean we can return it now
250  if (clean) {
251  if (d->customIds) {
252  emit highlight(d->currentId, d->index, d->matchedLength);
253  } else {
254  emit highlight(d->text, d->index, d->matchedLength);
255  }
256 
257  d->lastResult = Match;
258  d->matchedPattern = d->pattern;
259  return Match;
260  }
261  }
262  // if we couldn't look up the match, the new pattern isn't a
263  // substring of the matchedPattern, so we start a new search
264  else {
265  d->startNewIncrementalSearch();
266  }
267  }
268  // if the new pattern is longer than the matchedPattern we might be
269  // able to proceed from the last search
270  else if (d->pattern.length() > d->matchedPattern.length()) {
271  // continue from the previous pattern
272  if (d->pattern.startsWith(d->matchedPattern)) {
273  // we can't proceed from the previous position if the previous
274  // position already failed
275  if (d->index == INDEX_NOMATCH) {
276  return NoMatch;
277  }
278 
279  QString temp(d->pattern);
280  d->pattern.truncate(d->matchedPattern.length() + 1);
281  d->matchedPattern = temp;
282  }
283  // start a new search
284  else {
285  d->startNewIncrementalSearch();
286  }
287  }
288  // if the new pattern is as long as the matchedPattern, we reset if
289  // they are not equal
290  else if (d->pattern != d->matchedPattern) {
291  d->startNewIncrementalSearch();
292  }
293  }
294 
295 #ifdef DEBUG_FIND
296  //qDebug() << "d->index=" << d->index;
297 #endif
298  do {
299  // if we have multiple data blocks in our cache, walk through these
300  // blocks till we either searched all blocks or we find a match
301  do {
302  // Find the next candidate match.
303  if (d->options & KFind::RegularExpression) {
304  d->index = KFind::find(d->text, *d->regExp, d->index, d->options, &d->matchedLength);
305  } else {
306  d->index = KFind::find(d->text, d->pattern, d->index, d->options, &d->matchedLength);
307  }
308 
309  if (d->options & KFind::FindIncremental) {
310  d->data[d->currentId].dirty = false;
311  }
312 
313  if (d->index == -1 && d->currentId < d->data.count() - 1) {
314  d->text = d->data.at(++d->currentId).text;
315 
316  if (d->options & KFind::FindBackwards) {
317  d->index = d->text.length();
318  } else {
319  d->index = 0;
320  }
321  } else {
322  break;
323  }
324  } while (!(d->options & KFind::RegularExpression));
325 
326  if (d->index != -1) {
327  // Flexibility: the app can add more rules to validate a possible match
328  if (validateMatch(d->text, d->index, d->matchedLength)) {
329  bool done = true;
330 
331  if (d->options & KFind::FindIncremental) {
332  if (d->pattern.isEmpty()) {
333  delete d->emptyMatch;
334  d->emptyMatch = new Private::Match(d->currentId, d->index, d->matchedLength);
335  } else {
336  d->incrementalPath.insert(d->pattern, Private::Match(d->currentId, d->index, d->matchedLength));
337  }
338 
339  if (d->pattern.length() < d->matchedPattern.length()) {
340  d->pattern += d->matchedPattern.midRef(d->pattern.length(), 1);
341  done = false;
342  }
343  }
344 
345  if (done) {
346  d->matches++;
347  // Tell the world about the match we found, in case someone wants to
348  // highlight it.
349  if (d->customIds) {
350  emit highlight(d->currentId, d->index, d->matchedLength);
351  } else {
352  emit highlight(d->text, d->index, d->matchedLength);
353  }
354 
355  if (!d->dialogClosed) {
356  findNextDialog(true)->show();
357  }
358 
359 #ifdef DEBUG_FIND
360  //qDebug() << "Match. Next d->index=" << d->index;
361 #endif
362  d->lastResult = Match;
363  return Match;
364  }
365  } else { // Skip match
366  if (d->options & KFind::FindBackwards) {
367  d->index--;
368  } else {
369  d->index++;
370  }
371  }
372  } else {
373  if (d->options & KFind::FindIncremental) {
374  QString temp(d->pattern);
375  temp.truncate(temp.length() - 1);
376  d->pattern = d->matchedPattern;
377  d->matchedPattern = temp;
378  }
379 
380  d->index = INDEX_NOMATCH;
381  }
382  } while (d->index != INDEX_NOMATCH);
383 
384 #ifdef DEBUG_FIND
385  //qDebug() << "NoMatch. d->index=" << d->index;
386 #endif
387  d->lastResult = NoMatch;
388  return NoMatch;
389 }

391 void KFind::Private::startNewIncrementalSearch()
392 {
393  Private::Match *match = emptyMatch;
394  if (match == 0) {
395  text.clear();
396  index = 0;
397  currentId = 0;
398  } else {
399  text = data.at(match->dataId).text;
400  index = match->index;
401  currentId = match->dataId;

415 static bool isWholeWords(const QString &text, int starts, int matchedLength)
416 {
417  if (starts == 0 || !isInWord(text.at(starts - 1))) {
418  const int ends = starts + matchedLength;
419  if (ends == text.length() || !isInWord(text.at(ends))) {
420  return true;
421  }
422  }
423  return false;

427 {
428  if (options & KFind::WholeWordsOnly) {

440 // static
441 int KFind::find(const QString &text, const QString &pattern, int index, long options, int *matchedLength)
442 {
443  // Handle regular expressions in the appropriate way.
444  if (options & KFind::RegularExpression) {
445  Qt::CaseSensitivity caseSensitive = (options & KFind::CaseSensitive) ? Qt::CaseSensitive : Qt::CaseInsensitive;
446  QRegExp regExp(pattern, caseSensitive);
447 
448  return find(text, regExp, index, options, matchedLength);
449  }
450 
451  // In Qt4 QString("aaaaaa").lastIndexOf("a",6) returns -1; we need
452  // to start at text.length() - pattern.length() to give a valid index to QString.
453  if (options & KFind::FindBackwards) {
454  index = qMin(qMax(0, text.length() - pattern.length()), index);
455  }
456 
457  Qt::CaseSensitivity caseSensitive = (options & KFind::CaseSensitive) ? Qt::CaseSensitive : Qt::CaseInsensitive;
458 
459  if (options & KFind::FindBackwards) {
460  // Backward search, until the beginning of the line...
461  while (index >= 0) {
462  // ...find the next match.
463  index = text.lastIndexOf(pattern, index, caseSensitive);
464  if (index == -1) {
465  break;
466  }
467 
468  if (matchOk(text, index, pattern.length(), options)) {
469  break;
470  }
471  index--;
472  //qDebug() << "decrementing:" << index;
473  }
474  } else {
475  // Forward search, until the end of the line...
476  while (index <= text.length()) {
477  // ...find the next match.
478  index = text.indexOf(pattern, index, caseSensitive);
479  if (index == -1) {
480  break;
481  }
482 
483  if (matchOk(text, index, pattern.length(), options)) {
484  break;
485  }
486  index++;
487  }
488  if (index > text.length()) { // end of line
489  //qDebug() << "at" << index << "-> not found";
490  index = -1; // not found
491  }
492  }
493  if (index <= -1) {
494  *matchedLength = 0;
495  } else {
496  *matchedLength = pattern.length();
497  }
498  return index;
499 }

501 // Core method for the regexp-based find
502 static int doFind(const QString &text, const QRegExp &pattern, int index, long options, int *matchedLength)
503 {
504  if (options & KFind::FindBackwards) {
505  // Backward search, until the beginning of the line...
506  while (index >= 0) {
507  // ...find the next match.
508  index = text.lastIndexOf(pattern, index);
509  if (index == -1) {
510  break;
511  }
512 
513  /*int pos =*/ pattern.indexIn(text.mid(index));
514  *matchedLength = pattern.matchedLength();
515  if (matchOk(text, index, *matchedLength, options)) {
516  break;
517  }
518  index--;
519  }
520  } else {
521  // Forward search, until the end of the line...
522  while (index <= text.length()) {
523  // ...find the next match.
524  index = text.indexOf(pattern, index);
525  if (index == -1) {
526  break;
527  }
528 
529  /*int pos =*/ pattern.indexIn(text.mid(index));
530  *matchedLength = pattern.matchedLength();
531  if (matchOk(text, index, *matchedLength, options)) {
532  break;
533  }
534  index++;
535  }
536  if (index > text.length()) { // end of line
537  index = -1; // not found
538  }
539  }
540  if (index == -1) {
541  *matchedLength = 0;
542  }
543  return index;
544 }

546 // Since QRegExp doesn't support multiline searches (the equivalent of perl's /m)
547 // we have to cut the text into lines if the pattern starts with ^ or ends with $.
548 static int lineBasedFind(const QString &text, const QRegExp &pattern, int index, long options, int *matchedLength)
549 {
550  const QStringList lines = text.split(QLatin1Char('\n'));
551  int offset = 0;
552  // Use "index" to find the first line we should start from
553  int startLineNumber = 0;
554  for (; startLineNumber < lines.count(); ++startLineNumber) {
555  const QString line = lines.at(startLineNumber);
556  if (index < offset + line.length()) {
557  break;
558  }
559  offset += line.length() + 1 /*newline*/;
560  }
561 
562  if (options & KFind::FindBackwards) {
563 
564  if (startLineNumber == lines.count()) {
565  // We went too far, go back to the last line
566  --startLineNumber;
567  offset -= lines.at(startLineNumber).length() + 1;
568  }
569 
570  for (int lineNumber = startLineNumber; lineNumber >= 0; --lineNumber) {
571  const QString line = lines.at(lineNumber);
572  const int ret = doFind(line, pattern, lineNumber == startLineNumber ? index - offset : line.length(), options, matchedLength);
573  if (ret > -1) {
574  return ret + offset;
575  }
576  offset -= line.length() + 1 /*newline*/;
577  }
578 
579  } else {
580  for (int lineNumber = startLineNumber; lineNumber < lines.count(); ++lineNumber) {
581  const QString line = lines.at(lineNumber);
582  const int ret = doFind(line, pattern, lineNumber == startLineNumber ? (index - offset) : 0, options, matchedLength);
583  if (ret > -1) {
584  return ret + offset;
585  }
586  offset += line.length() + 1 /*newline*/;
587  }
588  }
589  return -1;
590 }

631 bool KFind::shouldRestart(bool forceAsking, bool showNumMatches) const
632 {
633  // Only ask if we did a "find from cursor", otherwise it's pointless.
634  // Well, unless the user can modify the document during a search operation,
635  // hence the force boolean.
636  if (!forceAsking && (d->options & KFind::FromCursor) == 0) {
637  displayFinalDialog();
638  return false;
639  }
640  QString message;
641  if (showNumMatches) {
642  if (numMatches()) {
643  message = i18np("1 match found.", "%1 matches found.", numMatches());
644  } else {
645  message = i18n("No matches found for '<b>%1</b>'.", d->pattern.toHtmlEscaped());
646  }
647  } else {
648  if (d->options & KFind::FindBackwards) {
649  message = i18n("Beginning of document reached.");
650  } else {
651  message = i18n("End of document reached.");
652  }
653  }
654 
655  message += QLatin1String("<br><br>"); // can't be in the i18n() of the first if() because of the plural form.
656  // Hope this word puzzle is ok, it's a different sentence
657  message +=
658  (d->options & KFind::FindBackwards) ?
659  i18n("Continue from the end?")
660  : i18n("Continue from the beginning?");
661 
662  int ret = KMessageBox::questionYesNo(dialogsParent(), QStringLiteral("<qt>%1</qt>").arg(message),
663  QString(), KStandardGuiItem::cont(), KStandardGuiItem::stop());
664  bool yes = (ret == KMessageBox::Yes);
665  if (yes) {
666  const_cast<KFind *>(this)->d->options &= ~KFind::FromCursor; // clear FromCursor option
667  }
668  return yes;
669 }
