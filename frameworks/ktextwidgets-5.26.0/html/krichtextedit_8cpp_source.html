99 void KRichTextEditPrivate::mergeFormatOnWordOrSelection(const QTextCharFormat &format)
100 {
101  QTextCursor cursor = q->textCursor();
102  QTextCursor wordStart(cursor);
103  QTextCursor wordEnd(cursor);
104 
105  wordStart.movePosition(QTextCursor::StartOfWord);
106  wordEnd.movePosition(QTextCursor::EndOfWord);
107 
108  cursor.beginEditBlock();
109  if (!cursor.hasSelection() && cursor.position() != wordStart.position() && cursor.position() != wordEnd.position()) {
110  cursor.select(QTextCursor::WordUnderCursor);
111  }
112  cursor.mergeCharFormat(format);
113  q->mergeCurrentCharFormat(format);
114  cursor.endEditBlock();
115 }

386 void KRichTextEdit::selectLinkText(QTextCursor *cursor) const
387 {
388  // If the cursor is on a link, select the text of the link.
389  if (cursor->charFormat().isAnchor()) {
390  QString aHref = cursor->charFormat().anchorHref();
391 
392  // Move cursor to start of link
393  while (cursor->charFormat().anchorHref() == aHref) {
394  if (cursor->atStart()) {
395  break;
396  }
397  cursor->setPosition(cursor->position() - 1);
398  }
399  if (cursor->charFormat().anchorHref() != aHref) {
400  cursor->setPosition(cursor->position() + 1, QTextCursor::KeepAnchor);
401  }
402 
403  // Move selection to the end of the link
404  while (cursor->charFormat().anchorHref() == aHref) {
405  if (cursor->atEnd()) {
406  break;
407  }
408  cursor->setPosition(cursor->position() + 1, QTextCursor::KeepAnchor);
409  }
410  if (cursor->charFormat().anchorHref() != aHref) {
411  cursor->setPosition(cursor->position() - 1, QTextCursor::KeepAnchor);
412  }
413  } else if (cursor->hasSelection()) {
414  // Nothing to to. Using the currently selected text as the link text.
415  } else {
416 
417  // Select current word
418  cursor->movePosition(QTextCursor::StartOfWord);
419  cursor->movePosition(QTextCursor::EndOfWord, QTextCursor::KeepAnchor);
420  }
421 }

428 void KRichTextEdit::updateLink(const QString &linkUrl, const QString &linkText)
429 {
430  selectLinkText();
431 
432  QTextCursor cursor = textCursor();
433  cursor.beginEditBlock();
434 
435  if (!cursor.hasSelection()) {
436  cursor.select(QTextCursor::WordUnderCursor);
437  }
438 
439  QTextCharFormat format = cursor.charFormat();
440  // Save original format to create an extra space with the existing char
441  // format for the block
442  const QTextCharFormat originalFormat = format;
443  if (!linkUrl.isEmpty()) {
444  // Add link details
445  format.setAnchor(true);
446  format.setAnchorHref(linkUrl);
447  // Workaround for QTBUG-1814:
448  // Link formatting does not get applied immediately when setAnchor(true)
449  // is called. So the formatting needs to be applied manually.
450  format.setUnderlineStyle(QTextCharFormat::SingleUnderline);
451  format.setUnderlineColor(KColorScheme(QPalette::Active, KColorScheme::View).foreground(KColorScheme::LinkText).color());
452  format.setForeground(KColorScheme(QPalette::Active, KColorScheme::View).foreground(KColorScheme::LinkText).color());
453  d->activateRichText();
454  } else {
455  // Remove link details
456  format.setAnchor(false);
457  format.setAnchorHref(QString());
458  // Workaround for QTBUG-1814:
459  // Link formatting does not get removed immediately when setAnchor(false)
460  // is called. So the formatting needs to be applied manually.
461  QTextDocument defaultTextDocument;
462  QTextCharFormat defaultCharFormat = defaultTextDocument.begin().charFormat();
463 
464  format.setUnderlineStyle(defaultCharFormat.underlineStyle());
465  format.setUnderlineColor(defaultCharFormat.underlineColor());
466  format.setForeground(defaultCharFormat.foreground());
467  }
468 
469  // Insert link text specified in dialog, otherwise write out url.
470  QString _linkText;
471  if (!linkText.isEmpty()) {
472  _linkText = linkText;
473  } else {
474  _linkText = linkUrl;
475  }
476  cursor.insertText(_linkText, format);
477 
478  // Insert a space after the link if at the end of the block so that
479  // typing some text after the link does not carry link formatting
480  if (!linkUrl.isEmpty() && cursor.atBlockEnd()) {
481  cursor.setPosition(cursor.selectionEnd());
482  cursor.setCharFormat(originalFormat);
483  cursor.insertText(QStringLiteral(" "));
484  }
485 
486  cursor.endEditBlock();
487 }

531 QString KRichTextEdit::toCleanHtml() const
532 {
533  QString result = toHtml();
534 
535  static const QString EMPTYLINEHTML = QLatin1String(
536  "<p style=\"-qt-paragraph-type:empty; margin-top:0px; margin-bottom:0px; "
537  "margin-left:0px; margin-right:0px; -qt-block-indent:0; text-indent:0px; \">&nbsp;</p>");
538 
539  // Qt inserts various style properties based on the current mode of the editor (underline,
540  // bold, etc), but only empty paragraphs *also* have qt-paragraph-type set to 'empty'.
541  static const QString EMPTYLINEREGEX = QStringLiteral(
542  "<p style=\"-qt-paragraph-type:empty;(.*)</p>");
543 
544  static const QString OLLISTPATTERNQT = QStringLiteral(
545  "<ol style=\"margin-top: 0px; margin-bottom: 0px; margin-left: 0px;");
546 
547  static const QString ULLISTPATTERNQT = QStringLiteral(
548  "<ul style=\"margin-top: 0px; margin-bottom: 0px; margin-left: 0px;");
549 
550  static const QString ORDEREDLISTHTML = QStringLiteral(
551  "<ol style=\"margin-top: 0px; margin-bottom: 0px;");
552 
553  static const QString UNORDEREDLISTHTML = QStringLiteral(
554  "<ul style=\"margin-top: 0px; margin-bottom: 0px;");
555 
556  // fix 1 - empty lines should show as empty lines - MS Outlook treats margin-top:0px; as
557  // a non-existing line.
558  // Although we can simply remove the margin-top style property, we still get unwanted results
559  // if you have three or more empty lines. It's best to replace empty <p> elements with <p>&nbsp;</p>.
560 
561  QRegExp emptyLineFinder(EMPTYLINEREGEX);
562  emptyLineFinder.setMinimal(true);
563 
564  // find the first occurrence
565  int offset = emptyLineFinder.indexIn(result, 0);
566  while (offset != -1) {
567  // replace all the matching text with the new line text
568  result.replace(offset, emptyLineFinder.matchedLength(), EMPTYLINEHTML);
569  // advance the search offset to just beyond the last replace
570  offset += EMPTYLINEHTML.length();
571  // find the next occurrence
572  offset = emptyLineFinder.indexIn(result, offset);
573  }
574 
575  // fix 2a - ordered lists - MS Outlook treats margin-left:0px; as
576  // a non-existing number; e.g: "1. First item" turns into "First Item"
577  result.replace(OLLISTPATTERNQT, ORDEREDLISTHTML);
578 
579  // fix 2b - unordered lists - MS Outlook treats margin-left:0px; as
580  // a non-existing bullet; e.g: "* First bullet" turns into "First Bullet"
581  result.replace(ULLISTPATTERNQT, UNORDEREDLISTHTML);
582 
583  return result;
584 }
