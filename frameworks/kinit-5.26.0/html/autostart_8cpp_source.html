62 static QString extractName(QString path) // krazy:exclude=passbyvalue
63 {
64  int i = path.lastIndexOf(QLatin1Char('/'));
65  if (i >= 0) {
66  path = path.mid(i + 1);
67  }
68  i = path.lastIndexOf(QLatin1Char('.'));
69  if (i >= 0) {
70  path = path.left(i);
71  }
72  return path;
73 }

75 void
76 AutoStart::loadAutoStartList()
77 {
78  // XDG autostart dirs
79 
80  // Make unique list of relative paths
81  QStringList files;
82  QStringList dirs = QStandardPaths::locateAll(QStandardPaths::GenericConfigLocation, QStringLiteral("autostart"), QStandardPaths::LocateDirectory);
83  Q_FOREACH (const QString &dir, dirs) {
84  const QStringList fileNames = QDir(dir).entryList(QStringList() << QStringLiteral("*.desktop"));
85  Q_FOREACH (const QString &file, fileNames) {
86  if (!files.contains(file)) {
87  files.append(file);
88  }
89  }
90  }
91 
92  for (QStringList::ConstIterator it = files.constBegin(); it != files.constEnd(); ++it) {
93  KAutostart config(*it);
94  if (!config.autostarts(QStringLiteral("KDE"), KAutostart::CheckAll)) {
95  continue;
96  }
97 
98  const QString file = QStandardPaths::locate(QStandardPaths::GenericConfigLocation, QStringLiteral("autostart/") + *it);
99  AutoStartItem *item = new AutoStartItem;
100  item->name = extractName(*it);
101  item->service = file;
102  item->startAfter = config.startAfter();
103  item->phase = config.startPhase();
104  if (item->phase < 0) {
105  item->phase = 0;
106  }
107  m_startList->append(item);
108  }
109 }

111 QString
112 AutoStart::startService()
113 {
114  if (m_startList->isEmpty()) {
115  return QString();
116  }
117 
118  while (!m_started.isEmpty()) {
119 
120  // Check for items that depend on previously started items
121  QString lastItem = m_started[0];
122  QMutableListIterator<AutoStartItem *> it(*m_startList);
123  while (it.hasNext()) {
124  AutoStartItem *item = it.next();
125  if (item->phase == m_phase
126  && item->startAfter == lastItem) {
127  m_started.prepend(item->name);
128  QString service = item->service;
129  it.remove();
130  delete item;
131  return service;
132  }
133  }
134  m_started.removeFirst();
135  }
136 
137  // Check for items that don't depend on anything
138  AutoStartItem *item;
139  QMutableListIterator<AutoStartItem *> it(*m_startList);
140  while (it.hasNext()) {
141  item = it.next();
142  if (item->phase == m_phase
143  && item->startAfter.isEmpty()) {
144  m_started.prepend(item->name);
145  QString service = item->service;
146  it.remove();
147  delete item;
148  return service;
149  }
150  }
151 
152  // Just start something in this phase
153  it = *m_startList;
154  while (it.hasNext()) {
155  item = it.next();
156  if (item->phase == m_phase) {
157  m_started.prepend(item->name);
158  QString service = item->service;
159  it.remove();
160  delete item;
161  return service;
162  }
163  }
164 
165  return QString();
166 }
