79 void
80 proctitle_init(int argc, char *argv[])
81 {
82 #if HAVE___PROGNAME
83  // progname may be a reference to argv[0]
84  __progname = strdup(__progname);
85 #else
86  if (argc == 0 || argv[0] == NULL) {
87  __progname = "unknown"; /* XXX */
88  } else {
89  char *p = strrchr(argv[0], '/');
90  if (p == NULL)
91  p = argv[0];
92  else
93  p++;
94 
95  __progname = strdup(p);
96  }
97 #endif
98 
99 #if PT_TYPE == PT_REUSEARGV
100  if (argc == 0 || argv[0] == NULL)
101  return;
102 
103  extern char **environ;
104  char *lastargv = NULL;
105  char **envp = environ;
106  int i;
107 
108  /*
109  * NB: This assumes that argv has already been copied out of the
110  * way. This is true for kdeinit, but may not be true for other
111  * programs. Beware.
112  */
113 
114  /* Fail if we can't allocate room for the new environment */
115  for (i = 0; envp[i] != NULL; i++)
116  ;
117  if ((environ = (char**)calloc(i + 1, sizeof(*environ))) == NULL) {
118  environ = envp; /* put it back */
119  return;
120  }
121 
122  /*
123  * Find the last argv string or environment variable within
124  * our process memory area.
125  */
126  for (i = 0; i < argc; i++) {
127  if (lastargv == NULL || lastargv + 1 == argv[i])
128  lastargv = argv[i] + strlen(argv[i]);
129  }
130  for (i = 0; envp[i] != NULL; i++) {
131  if (lastargv + 1 == envp[i])
132  lastargv = envp[i] + strlen(envp[i]);
133  }
134 
135  argv[1] = NULL;
136  argv_start = argv[0];
137  argv_env_len = lastargv - argv[0] - 1;
138 
139  /*
140  * Copy environment
141  * XXX - will truncate env on strdup fail
142  */
143  for (i = 0; envp[i] != NULL; i++)
144  environ[i] = strdup(envp[i]);
145  environ[i] = NULL;
146 #endif /* PT_REUSEARGV */
147 }

149 void
150 proctitle_set(const char *fmt, ...)
151 {
152 #if PT_TYPE != PT_NONE
153 #if PT_TYPE == PT_REUSEARGV
154  if (argv_env_len <= 0)
155  return;
156 #endif
157 
158  bool skip_proctitle = false;
159  if (fmt != NULL && fmt[0] == '-') {
160  skip_proctitle = true;
161  ++fmt;
162  }
163  char ptitle[1024];
164  memset(ptitle, '\0', sizeof(ptitle));
165  size_t len = 0;
166 
167 #if ADD_PROCTITLE
168  if (!skip_proctitle) {
169  strncpy(ptitle, __progname, sizeof(ptitle)-1);
170  len = strlen(ptitle);
171  if (fmt != NULL && sizeof(ptitle) - len > 2) {
172  strcpy(ptitle + len, ": ");
173  len += 2;
174  }
175  }
176 #endif
177 
178  if (fmt != NULL) {
179  int r = -1;
180  if (len < sizeof(ptitle) - 1) {
181  va_list ap;
182  va_start(ap, fmt);
183  r = vsnprintf(ptitle + len, sizeof(ptitle) - len , fmt, ap);
184  va_end(ap);
185  }
186  if (r == -1 || (size_t)r >= sizeof(ptitle) - len)
187  return;
188  }
189 
190 #if PT_TYPE == PT_PSTAT
191  union pstun pst;
192  pst.pst_command = ptitle;
193  pstat(PSTAT_SETCMD, pst, strlen(ptitle), 0, 0);
194 #elif PT_TYPE == PT_REUSEARGV
195  strncpy(argv_start, ptitle, argv_env_len);
196  argv_start[argv_env_len-1] = '\0';
197 #elif PT_TYPE == PT_SETPROCTITLE
198  if (fmt == NULL) {
199  setproctitle(NULL);
200 #if defined(__FreeBSD__)
201  } else if (skip_proctitle) {
202  // setproctitle on FreeBSD allows skipping the process title
203  setproctitle("-%s", ptitle);
204 #endif
205  } else {
206  setproctitle("%s", ptitle);
207  }
208 #endif
209 
210 #endif /* !PT_NONE */
211 }
