81 static int show_message_box( const QString &title, const QString &message )
82 {
83  int (*_CGSDefaultConnection_p)();
84  _CGSDefaultConnection_p = (int (*)()) dlsym(RTLD_DEFAULT, "_CGSDefaultConnection");
85  if (!_CGSDefaultConnection_p) {
86  NSLog(@"qt_fatal_message_box(%@,%@): couldn't load _CGSDefaultConnection: %s",
87  [NSString stringWithCString:message.toUtf8().constData() encoding:NSUTF8StringEncoding],
88  [NSString stringWithCString:title.toUtf8().constData() encoding:NSUTF8StringEncoding], dlerror());
89  }
90  if (_CGSDefaultConnection_p && (*_CGSDefaultConnection_p)()) {
91  @autoreleasepool {
92  NSAlert* alert = [[[NSAlert alloc] init] autorelease];
93  NSString *msg, *tit;
94  @synchronized([NSAlert class]){
95  [alert addButtonWithTitle:@"OK"];
96  [alert setAlertStyle:NSCriticalAlertStyle];
97  [alert setMessageText:@"" ];
98  if( !(msg = [NSString stringWithCString:message.toUtf8().constData() encoding:NSUTF8StringEncoding]) ){
99  msg = [NSString stringWithCString:message.toLatin1().constData() encoding:NSASCIIStringEncoding];
100  }
101  if( !(tit = [NSString stringWithCString:title.toUtf8().constData() encoding:NSUTF8StringEncoding]) ){
102  tit = [NSString stringWithCString:title.toLatin1().constData() encoding:NSASCIIStringEncoding];
103  }
104  if( msg ){
105  [alert setInformativeText:msg];
106  }
107  else{
108  NSLog( @"msg=%@ tit=%@", msg, tit );
109  }
110  [[alert window] setTitle:tit];
111  return NSAlertSecondButtonReturn == [alert runModal];
112  }
113  }
114  }
115  return 0;
116 }

118 static void exitWithErrorMsg(const QString &errorMsg, bool doExit)
119 {
120  fprintf(stderr, "%s\n", errorMsg.toLocal8Bit().data());
121  QByteArray utf8ErrorMsg = errorMsg.toUtf8();
122  d.result = 3; // Error with msg
123  write(d.fd[1], &d.result, 1);
124  int l = utf8ErrorMsg.length();
125  write(d.fd[1], &l, sizeof(int));
126  write(d.fd[1], utf8ErrorMsg.data(), l);
127  close(d.fd[1]);
128  show_message_box(QStringLiteral("kdeinit5"), errorMsg);
129  if (doExit) {
130  exit(255);
131  }
132 }

134 pid_t launch(int argc, const char *_name, const char *args,
135  const char *cwd, int envc, const char *envs,
136  bool reset_env, const char *tty, bool avoid_loops,
137  const char *startup_id_str) // krazy:exclude=doublequote_chars
138 {
139  QString bin, libpath;
140  QByteArray name;
141  QByteArray execpath;
142 
143  if (_name[0] != '/') {
144  name = _name;
145  bin = QFile::decodeName(name);
146  execpath = execpath_avoid_loops(name, envc, envs, avoid_loops);
147  } else {
148  name = _name;
149  bin = QFile::decodeName(name);
150  name = name.mid(name.lastIndexOf('/') + 1);
151 
152  // FIXME: this .so extension stuff is very Linux-specific
153  // a .so extension can occur on OS X, but the typical extension is .dylib
154  if (bin.endsWith(QStringLiteral(".so")) || bin.endsWith(QStringLiteral(".dylib"))) {
155  libpath = bin;
156  } else {
157  execpath = _name;
158  }
159  }
160 #ifndef NDEBUG
161  fprintf(stderr, "kdeinit5: preparing to launch '%s'\n", libpath.isEmpty()
162  ? execpath.constData() : libpath.toUtf8().constData());
163 #endif
164  if (!args) {
165  argc = 1;
166  }
167 
168  // do certain checks before forking
169  if (execpath.isEmpty()) {
170  QString errorMsg = i18n("Could not find '%1' executable.", QFile::decodeName(_name));
171  exitWithErrorMsg(errorMsg, false);
172  d.result = 3;
173  d.fork = 0;
174  return d.fork;
175  } else if (!libpath.isEmpty()) {
176  QString errorMsg = i18n("Launching library '%1' is not supported on OS X.", libpath);
177  exitWithErrorMsg(errorMsg, false);
178  d.result = 3;
179  d.fork = 0;
180  return d.fork;
181  }
182 
183  if (0 > pipe(d.fd)) {
184  perror("kdeinit5: pipe() failed");
185  d.result = 3;
186  d.errorMsg = i18n("Unable to start new process.\n"
187  "The system may have reached the maximum number of open files possible or the maximum number of open files that you are allowed to use has been reached.").toUtf8();
188  d.fork = 0;
189  return d.fork;
190  }
191 
192  // find out this path before forking, doing it afterwards
193  // crashes on some platforms, notably OSX
194  const QString bundlepath = QStandardPaths::findExecutable(QFile::decodeName(execpath));
195 
196  const QString argvexe = QStandardPaths::findExecutable(QLatin1String(_name));
197 
198  d.errorMsg = 0;
199  d.fork = fork();
200  switch (d.fork) {
201  case -1:
202  perror("kdeinit5: fork() failed");
203  d.result = 3;
204  d.errorMsg = i18n("Unable to create new process.\n"
205  "The system may have reached the maximum number of processes possible or the maximum number of processes that you are allowed to use has been reached.").toUtf8();
206  close(d.fd[0]);
207  close(d.fd[1]);
208  d.fork = 0;
209  break;
210  case 0: {
212  close(d.fd[0]);
213  close_fds();
214  reset_oom_protect();
215 
216  // Try to chdir, either to the requested directory or to the user's document path by default.
217  // We ignore errors - if you write a desktop file with Exec=foo and Path=/doesnotexist,
218  // we still want to execute `foo` even if the chdir() failed.
219  if (cwd && *cwd) {
220  (void)chdir(cwd);
221  }
222 
223  if (reset_env) { // KWRAPPER/SHELL
224 
225  QList<QByteArray> unset_envs;
226  for (int tmp_env_count = 0;
227  environ[tmp_env_count];
228  tmp_env_count++) {
229  unset_envs.append(environ[ tmp_env_count ]);
230  }
231  foreach (const QByteArray &tmp, unset_envs) {
232  int pos = tmp.indexOf('=');
233  if (pos >= 0) {
234  unsetenv(tmp.left(pos).constData());
235  }
236  }
237  }
238 
239  for (int i = 0; i < envc; i++) {
240  putenv((char *)envs);
241  while (*envs != 0) {
242  envs++;
243  }
244  envs++;
245  }
246 
247  {
248  QByteArray procTitle;
249  // launching an executable: can do an exec directly
250  d.argv = (char **) malloc(sizeof(char *) * (argc + 1));
251  if (!argvexe.isEmpty()) {
252  QByteArray cstr = argvexe.toLocal8Bit();
253  d.argv[0] = strdup(cstr.data());
254  } else {
255  d.argv[0] = (char *) _name;
256  }
257  for (int i = 1; i < argc; i++) {
258  d.argv[i] = (char *) args;
259  procTitle += ' ';
260  procTitle += (char *) args;
261  while (*args != 0) {
262  args++;
263  }
264  args++;
265  }
266  d.argv[argc] = 0;
267 
268 #ifndef SKIP_PROCTITLE
269 
270  proctitle_set("%s%s", name.data(), procTitle.data() ? procTitle.data() : "");
271 #endif
272  }
273 
274  if (!execpath.isEmpty()) {
275  // we're launching an executable; even after a fork and being exec'ed, that
276  // executable is allowed to use non-POSIX APIs.
277  d.result = 2; // Try execing
278  write(d.fd[1], &d.result, 1);
279 
280  // We set the close on exec flag.
281  // Closing of d.fd[1] indicates that the execvp succeeded!
282  fcntl(d.fd[1], F_SETFD, FD_CLOEXEC);
283 
284  setup_tty(tty);
285 
286  QByteArray executable = execpath;
287  if (!bundlepath.isEmpty()) {
288  executable = QFile::encodeName(bundlepath);
289  }
290  // TODO: we probably want to use [[NSProcessInfo processInfo] setProcessName:NSString*] here to
291  // make the new app show up under its own name in non-POSIX process listings.
292 
293  if (!executable.isEmpty()) {
294 #ifndef NDEBUG
295  qDebug() << "execvp" << executable;
296  for (int i = 0 ; i < argc ; ++i) {
297  qDebug() << "arg #" << i << "=" << d.argv[i];
298  }
299 #endif
300  // attempt to the correct application name
301  QFileInfo fi(QString::fromUtf8(executable));
302  [[NSProcessInfo processInfo] setProcessName:(NSString*)fi.baseName().toCFString()];
303  qApp->setApplicationName(fi.baseName());
304  execvp(executable.constData(), d.argv);
305  }
306 
307  d.result = 1; // Error
308  write(d.fd[1], &d.result, 1);
309  close(d.fd[1]);
310  exit(255);
311  }
312 
313  break;
314  }
315  default:
317  close(d.fd[1]);
318  bool exec = false;
319  for (;;) {
320  d.n = read(d.fd[0], &d.result, 1);
321  if (d.n == 1) {
322  if (d.result == 2) {
323 #ifndef NDEBUG
324  //fprintf(stderr, "kdeinit5: no kdeinit module, trying exec....\n");
325 #endif
326  exec = true;
327  continue;
328  }
329  if (d.result == 3) {
330  int l = 0;
331  d.n = read(d.fd[0], &l, sizeof(int));
332  if (d.n == sizeof(int)) {
333  QByteArray tmp;
334  tmp.resize(l + 1);
335  d.n = read(d.fd[0], tmp.data(), l);
336  tmp[l] = 0;
337  if (d.n == l) {
338  d.errorMsg = tmp;
339  }
340  }
341  }
342  // Finished
343  break;
344  }
345  if (d.n == -1) {
346  if (errno == ECHILD) { // a child died.
347  continue;
348  }
349  if (errno == EINTR || errno == EAGAIN) { // interrupted or more to read
350  continue;
351  }
352  }
353  if (d.n == 0) {
354  if (exec) {
355  d.result = 0;
356  } else {
357  fprintf(stderr, "kdeinit5: (%s %s) Pipe closed unexpectedly", name.constData(), execpath.constData());
358  perror("kdeinit5: Pipe closed unexpectedly");
359  d.result = 1; // Error
360  }
361  break;
362  }
363  perror("kdeinit5: Error reading from pipe");
364  d.result = 1; // Error
365  break;
366  }
367  close(d.fd[0]);
368  }
369  return d.fork;
370 }

384 {
385  int argc = *_NSGetArgc();
386  char **argv = *_NSGetArgv();

389  uint32_t buflen = PATH_MAX;

395 
396  newargv[argc] = (char*)"--nofork";
