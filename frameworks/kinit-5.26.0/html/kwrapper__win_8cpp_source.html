20 /*
21  kde application starter
22  - allows starting kde application without any additional path settings [1]
23  - supports multiple root installation which is often used by packagers
24  (adds bin/lib directories from KDEDIRS environment to PATH environment)
25  - designed to start kde application from windows start menu entries
26  - support for reading KDEDIRS setting from flat file
27  (located in <path-of-kwrapper.exe>/../../kdedirs.cache)
28  - planned: automatic KDEDIRS detection support
29 
30 [1] recent kde cmake buildsystem on win32 installs shared libraries
31  into lib instead of bin. This requires to have the lib directory
32  in the PATH environment variable too (required for all pathes in KDEDIRS)
33 
34  TODO: There is an prelimary concept of setting KDEDIRS environment variable
35  from a cache file located on a well known path relative from the requested
36  application.
37  The recent implementation expects a file name 'kdedirs.cache' two level
38  above this executable which will be <ProgramFiles> in case kwrapper5 lives
39  in <Programfiles>/kde5/bin.
40  This works not in any case especially when running application inside the
41  build directory.
42 
43 */

45 #include <stdio.h>
46 #include <assert.h>
47 #include <stdlib.h>
48 #include <process.h>
49 #include <windows.h>

51 #include <QString>
52 #include <QProcess>
53 #include <QtDebug>
54 #include <QFileInfo>
55 #include <QCoreApplication>
56 #include <QList>
57 
58 bool verbose = 0;
59 
60 int main(int argc, char **argv)
61 {
62  QCoreApplication app(argc, argv);
63 
64  QStringList envPath;
65  QStringList searchPath;
66  QString exeToStart;
67  QString myAppName = QStringLiteral("kwrapper5:");
68  QStringList exeParams;
69  int firstParam = 1;
70 
71  if (QCoreApplication::arguments().size() == 1) {
72  qDebug() << myAppName << "no application given";
73  return 1;
74  }
75 
76  if (QCoreApplication::arguments().at(1) == QStringLiteral("--verbose")) {
77  verbose = 1;
78  firstParam = 2;
79  }
80 
81  exeToStart = QCoreApplication::arguments().at(firstParam);
82 
83  for (int i = firstParam + 1; i < QCoreApplication::arguments().size(); i++) {
84  exeParams << QCoreApplication::arguments().at(i);
85  }
86 
87  QString path = QString::fromLocal8Bit(qgetenv("PATH")).toLower()
88  .replace(QLatin1Char('\\'), QLatin1Char('/'));
89 
94  foreach (const QString &a, path.split(QLatin1Char(';'))) {
95  if (!envPath.contains(a)) {
96  envPath << a;
97  }
98  if (!a.endsWith(QLatin1String("/lib")) && !a.endsWith(QLatin1String("/lib/")) && !searchPath.contains(a)) {
99  searchPath << a;
100  }
101  }
102 
103  // add current install path
104  path = QCoreApplication::applicationDirPath().toLower().replace(QLatin1Char('\\'), QLatin1Char('/'));
105  if (!envPath.contains(path)) {
106  envPath << path;
107  }
108 
109  // detect directory where kdedirs.cache lives
110  // this is not complete, KDEDIRS path should be used as base too
111  QFileInfo fi(path + QStringLiteral("/../.."));
112  QString rootPath = fi.canonicalPath();
113 
114  if (verbose) {
115  qDebug() << "try to find kdedirs.cache in" << rootPath;
116  }
117 
118  // add current lib path to client path environment
119  path = path.replace(QStringLiteral("bin"), QStringLiteral("lib"));
120  if (!envPath.contains(path)) {
121  envPath << path;
122  }
123 
129  path = QString::fromLocal8Bit(qgetenv("KDEDIRS")).toLower().replace(QLatin1Char('\\'), QLatin1Char('/'));
130  QStringList kdedirs;
131 
132  if (path.size() > 0) {
133  kdedirs = path.split(QLatin1Char(';'));
134  }
135 
136  bool changedKDEDIRS = 0;
137  // setup kdedirs if not present
138  if (kdedirs.size() == 0) {
139  QStringList kdedirsCacheList;
140 #ifdef Q_CC_MSVC
141  kdedirsCacheList << rootPath + QStringLiteral("/kdedirs.cache.msvc");
142 #endif
143  kdedirsCacheList << rootPath + QStringLiteral("/kdedirs.cache");
144 
145  bool found = false;
146  foreach (const QString &kdedirsCachePath, kdedirsCacheList) {
147  QFile f(kdedirsCachePath);
148  if (f.exists()) {
149  f.open(QIODevice::ReadOnly);
150  QByteArray data = f.readAll();
151  f.close();
152  kdedirs = QString::fromLocal8Bit(data).split(QLatin1Char(';'));
153  if (verbose) {
154  qDebug() << "load kdedirs cache from " << kdedirsCachePath << "values=" << kdedirs;
155  }
156  found = true;
157  break;
158  }
159  }
160  if (!found) {
161  /*
162  f.open(QIODevice::WriteOnly);
163  // search all pathes one level above for a directory share/apps
164  // write entries into a cache
165  f.write(kdedirs.join(";").toLatin1());
166  f.close();
167  */
168  }
169  changedKDEDIRS = 1;
170  }
171  if (verbose) {
172  qDebug() << "found KDEDIRS\n\t" << kdedirs.join(QStringLiteral("\n\t"));
173  }
174 
175  foreach (const QString &a, kdedirs) {
176  if (!envPath.contains(a + QStringLiteral("/bin"))) {
177  envPath << a + QStringLiteral("/bin");
178  }
179  if (!envPath.contains(a + QStringLiteral("/lib"))) {
180  envPath << a + QStringLiteral("/lib");
181  }
182  if (!searchPath.contains(a + QStringLiteral("/bin"))) {
183  searchPath << a + QStringLiteral("/bin");
184  }
185  }
186 
187  // find executable
188  WCHAR _appName[MAX_PATH + 1];
189 
190  if (verbose) {
191  qDebug() << "search " << exeToStart << "in";
192  }
193 
194  bool found = false;
195  foreach (const QString &a, searchPath) {
196  if (verbose) {
197  qDebug() << "\t" << a;
198  }
199 #ifndef _WIN32_WCE
200  if (SearchPathW((LPCWSTR)a.utf16(), (LPCWSTR)exeToStart.utf16(),
201  L".exe", MAX_PATH + 1, (LPWSTR)_appName, NULL)) {
202  found = true;
203  break;
204  }
205 #else
206  if (QFile::exists(a + "/" + exeToStart + ".exe")) {
207  found = true;
208  break;
209  }
210 #endif
211  }
212  QString appName = QString::fromUtf16((unsigned short *)_appName);
213 
214  if (!found) {
215  qWarning() << myAppName << "application not found";
216  return 3;
217  }
218 
219  if (verbose) {
220  qDebug() << "run" << exeToStart << "with params" << exeParams
221  << "and PATH environment\n\t" << envPath.join(QStringLiteral("\n\t"));
222  }
223 
224  // setup client process envirionment
225  QStringList env = QProcess::systemEnvironment();
226  env.replaceInStrings(QRegExp(QStringLiteral("^PATH=(.*)"), Qt::CaseInsensitive),
227  QLatin1String("PATH=") + envPath.join(QLatin1Char(';')));
228  if (changedKDEDIRS) {
229  env << QStringLiteral("KDEDIRS=") + kdedirs.join(QLatin1Char(';'));
230  }
231 
232  QProcess *process = new QProcess;
233  process->setEnvironment(env);
234  process->start(appName, exeParams);
235  if (process->state() == QProcess::NotRunning) {
236  qWarning() << myAppName << "process not running";
237  return 4;
238  }
239  process->waitForStarted();
240 
241  return 0;
242 }
