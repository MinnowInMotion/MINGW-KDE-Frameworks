141 static QString getProcessName(DWORD pid)
142 {
143  HANDLE hModuleSnap = INVALID_HANDLE_VALUE;
144  MODULEENTRY32 me32;
145 
146  hModuleSnap = CreateToolhelp32Snapshot(TH32CS_SNAPMODULE, pid);
147  if (hModuleSnap == INVALID_HANDLE_VALUE) {
148  return QString();
149  }
150 
151  me32.dwSize = sizeof(MODULEENTRY32);
152 
153  if (!Module32First(hModuleSnap, &me32)) {
154  CloseHandle(hModuleSnap); // clean the snapshot object
155  return QString();
156  }
157  QString name = QString::fromWCharArray(me32.szExePath);
158  CloseHandle(hModuleSnap);
159  return name;
160 }

167 static PSID getProcessOwner(HANDLE hProcess)
168 {
169 #ifndef _WIN32_WCE
170  HANDLE hToken = NULL;
171  PSID sid;
172 
173  OpenProcessToken(hProcess, TOKEN_READ, &hToken);
174  if (hToken) {
175  DWORD size;
176  PTOKEN_USER userStruct;
177 
178  // check how much space is needed
179  GetTokenInformation(hToken, TokenUser, NULL, 0, &size);
180  if (ERROR_INSUFFICIENT_BUFFER == GetLastError()) {
181  userStruct = reinterpret_cast<PTOKEN_USER>(new BYTE[size]);
182  GetTokenInformation(hToken, TokenUser, userStruct, size, &size);
183 
184  sid = copySid(userStruct->User.Sid);
185  CloseHandle(hToken);
186  delete [] userStruct;
187  return sid;
188  }
189  }
190 #endif
191  return 0;
192 }

304 void ProcessList::init()
305 {
306  HANDLE h;
307  PROCESSENTRY32 pe32;
308 
309  h = CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS, 0);
310  if (h == INVALID_HANDLE_VALUE) {
311  return;
312  }
313  pe32.dwSize = sizeof(PROCESSENTRY32);
314  if (!Process32First(h, &pe32)) {
315  return;
316  }
317 
318  do {
319  HANDLE hProcess = getProcessHandle(pe32.th32ProcessID);
320  if (!hProcess) {
321  continue;
322  }
323  QString name = getProcessName(pe32.th32ProcessID);
324 #ifndef _WIN32_WCE
325  PSID sid = getProcessOwner(hProcess);
326  if (!sid || m_userId && !EqualSid(m_userId, sid)) {
327  freeSid(sid);
328  continue;
329  }
330 #else
331  PSID sid = 0;
332 #endif
333  m_processes << new ProcessListEntry(hProcess, name, pe32.th32ProcessID, sid);
334  } while (Process32Next(h, &pe32));
335 #ifndef _WIN32_WCE
336  CloseHandle(h);
337 #else
338  CloseToolhelp32Snapshot(h);
339 #endif
340 }

342 ProcessListEntry *ProcessList::find(const QString &name)
343 {
344  ProcessListEntry *ple;
345  foreach (ple, m_processes) {
346  if (ple->pid < 0) {
347  qDebug() << "negative pid!";
348  continue;
349  }
350 
351  if (ple->name != name && ple->name != name + ".exe") {
352  continue;
353  }
354 
355  if (!ple->path.isEmpty() && !ple->path.toLower().startsWith(QString(QStringLiteral(CMAKE_INSTALL_PREFIX)).toLower())) {
356  // process is outside of installation directory
357  qDebug() << "path of the process" << name << "seems to be outside of the installPath:" << ple->path << QStringLiteral(CMAKE_INSTALL_PREFIX);
358  continue;
359  }
360  return ple;
361  }
362  return NULL;
363 }

365 bool ProcessList::terminateProcess(const QString &name)
366 {
367  qDebug() << "going to terminate process" << name;
368  ProcessListEntry *p = find(name);
369  if (!p) {
370  qDebug() << "could not find ProcessListEntry for process name" << name;
371  return false;
372  }
373 
374  bool ret = TerminateProcess(p->handle, 0);
375  if (ret) {
376  int i = m_processes.indexOf(p);
377  if (i != -1) {
378  m_processes.removeAt(i);
379  }
380  delete p;
381  return true;
382  } else {
383  return false;
384  }

387 // internal launch function
388 int launch(const QString &cmd)
389 {
390  QProcess *proc = new QProcess();
391  proc->start(cmd);
392  proc->waitForStarted();
393  startedProcesses << proc;
394  _PROCESS_INFORMATION *_pid = proc->pid();
395  int pid = _pid ? _pid->dwProcessId : 0;
396  if (verbose) {
397  fprintf(stderr, "%s", proc->readAllStandardError().constData());
398  fprintf(stderr, "%s", proc->readAllStandardOutput().constData());
399  }
400  if (pid) {
401  if (verbose) {
402  fprintf(stderr, "kdeinit5: Launched %s, pid = %ld\n", qPrintable(cmd), (long) pid);
403  }
404  } else {
405  if (verbose) {
406  fprintf(stderr, "kdeinit5: could not launch %s, exiting\n", qPrintable(cmd));
407  }
408  }
409  return pid;
410 }

413 bool checkIfRegisteredInDBus(const QString &name, int _timeout = 10)
414 {
415  int timeout = _timeout * 5;
416  while (timeout) {
417  if (QDBusConnection::sessionBus().interface()->isServiceRegistered(name)) {
418  break;
419  }
420  Sleep(200);
421  timeout--;
422  }
423  if (!timeout) {
424  if (verbose) {
425  fprintf(stderr, "not registered %s in dbus after %d secs\n", qPrintable(name), _timeout);
426  }
427  return false;
428  }
429  if (verbose) {
430  fprintf(stderr, "%s is registered in dbus\n", qPrintable(name));
431  }
432  return true;
433 }

473 void quitApplicationsOverDBus()
474 {
475  QDBusConnection connection = QDBusConnection::sessionBus();
476  QDBusConnectionInterface *bus = connection.interface();
477  const QStringList services = bus->registeredServiceNames();
478  foreach (const QString &service, services) {
479  if (service.startsWith(QLatin1String("org.freedesktop.DBus")) || service.startsWith(QLatin1Char(':'))) {
480  continue;
481  }
482  QDBusInterface *iface = new QDBusInterface(service,
483  QLatin1String("/MainApplication"),
484  QLatin1String("org.kde.KApplication"),
485  connection);
486  if (!iface->isValid()) {
487  if (verbose) {
488  fprintf(stderr, "invalid interface of service %s\n", service.toLatin1().data());
489  }
490  continue;
491  }
492  iface->call("quit");
493  if (iface->lastError().isValid()) {
494  if (verbose) {
495  fprintf(stderr, "killing %s with result\n", iface->lastError().message().toLatin1().data());
496  }
497  }
498  delete iface;
499  }

502 int main(int argc, char **argv, char **envp)
503 {
504  pid_t pid = 0;
505  bool launch_dbus = true;
506  bool launch_klauncher = true;
507  bool launch_kded = true;
508  bool suicide = false;
509  bool listProcesses = false;
510  bool killProcesses = false;
511  bool listAppsInDBus = false;
512  bool quitAppsOverDBus = false;
513  bool shutdown = false;
514 
516  char **safe_argv = (char **) malloc(sizeof(char *) * argc);
517  for (int i = 0; i < argc; i++) {
518  safe_argv[i] = strcpy((char *)malloc(strlen(argv[i]) + 1), argv[i]);
519  if (strcmp(safe_argv[i], "--no-dbus") == 0) {
520  launch_dbus = false;
521  }
522  if (strcmp(safe_argv[i], "--no-klauncher") == 0) {
523  launch_klauncher = false;
524  }
525  if (strcmp(safe_argv[i], "--no-kded") == 0) {
526  launch_kded = false;
527  }
528  if (strcmp(safe_argv[i], "--suicide") == 0) {
529  suicide = true;
530  }
531 #ifdef ENABLE_EXIT
532  if (strcmp(safe_argv[i], "--exit") == 0) {
533  keep_running = 0;
534  }
535 #endif
536  if (strcmp(safe_argv[i], "--verbose") == 0) {
537  verbose = 1;
538  }
539  if (strcmp(safe_argv[i], "--version") == 0) {
540  printf("Qt: %s\n", qVersion());
541  printf("KDE: %s\n", KINIT_VERSION_STRING);
542  exit(0);
543  }
544  if (strcmp(safe_argv[i], "--help") == 0) {
545  printf("Usage: kdeinit5 [options]\n");
546 #ifdef ENABLE_EXIT
547  printf(" --exit Terminate when kded has run\n");
548 #endif
549  printf(" --help this help page\n");
550  printf(" --list list kde processes\n");
551  printf(" --list-dbus-apps list all applications registered in dbus\n");
552  printf(" --quit-over-dbus quit all application registered in dbus\n");
553  printf(" --no-dbus do not start dbus-daemon\n");
554  printf(" --no-klauncher do not start klauncher\n");
555  printf(" --no-kded do not start kded\n");
556  printf(" --shutdown safe shutdown of all running kde processes\n");
557  printf(" first over dbus, then using hard kill\n");
558 #ifdef ENABLE_SUICIDE
559  printf(" --suicide terminate when no KDE applications are left running\n");
560 #endif
561  printf(" --terminate hard kill of *all* running kde processes\n");
562  printf(" --verbose print verbose messages\n");
563  printf(" --version Show version information\n");
564  exit(0);
565  }
566  if (strcmp(safe_argv[i], "--list") == 0) {
567  listProcesses = true;
568  }
569  if (strcmp(safe_argv[i], "--shutdown") == 0) {
570  shutdown = true;
571  }
572  if (strcmp(safe_argv[i], "--terminate") == 0 || strcmp(safe_argv[i], "--kill") == 0) {
573  killProcesses = true;
574  }
575  if (strcmp(safe_argv[i], "--list-dbus-apps") == 0) {
576  listAppsInDBus = true;
577  }
578  if (strcmp(safe_argv[i], "--quit-over-dbus") == 0) {
579  quitAppsOverDBus = true;
580  }
581  }
582 
583  PSID currentSid = getCurrentProcessOwner();
584  if (verbose) {
585  fprintf(stderr, "current user sid: %s\n", qPrintable(toString(currentSid)));
586  }
587  ProcessList processList(currentSid);
588  freeSid(currentSid);
589 
590  if (listProcesses) {
591  listAllRunningKDEProcesses(processList);
592  return 0;
593  } else if (killProcesses) {
594  terminateAllRunningKDEProcesses(processList);
595  return 0;
596  } else if (listAppsInDBus) {
597  listAllNamedAppsInDBus();
598  return 0;
599  } else if (quitAppsOverDBus) {
600  quitApplicationsOverDBus();
601  return 0;
602  } else if (shutdown) {
603  quitApplicationsOverDBus();
604  Sleep(2000);
605  terminateAllRunningKDEProcesses(processList);
606  }
607 
608 #ifdef _DEBUG
609  // first try to launch dbus-daemond in debug mode
610  if (launch_dbus && processList.find("dbus-daemond")) {
611  launch_dbus = false;
612  }
613  if (launch_dbus) {
614  pid = launch("dbus-launchd.exe");
615  if (!pid) {
616  pid = launch("dbus-launchd.bat");
617  }
618  launch_dbus = (pid == 0);
619  }
620 #endif
621  if (launch_dbus && !processList.find("dbus-daemon")) {
622  if (!pid) {
623  pid = launch("dbus-launch.exe");
624  }
625  if (!pid) {
626  pid = launch("dbus-launch.bat");
627  }
628  if (!pid) {
629  exit(1);
630  }
631  }
632 
633  if (launch_klauncher && !processList.find("klauncher")) {
634  pid = launch("klauncher");
635  if (!pid || !checkIfRegisteredInDBus("org.kde.klauncher5", 10)) {
636  exit(1);
637  }
638  }
639 
640  if (launch_kded && !processList.find(KDED_EXENAME)) {
641  pid = launch(KDED_EXENAME);
642  if (!pid || !checkIfRegisteredInDBus("org.kde.kded5", 10)) {
643  exit(1);
644  }
645  }
646 
647  for (int i = 1; i < argc; i++) {
648  if (safe_argv[i][0] == '+') {
649  pid = launch(safe_argv[i] + 1);
650  } else if (safe_argv[i][0] == '-') {
651  // Ignore
652  } else {
653  pid = launch(safe_argv[i]);
654  }
655  }
656 
658  for (int i = 0; i < argc; i++) {
659  free(safe_argv[i]);
660  }
661  free(safe_argv);
662 
664 #ifdef ENABLE_SUICIDE
665  if (suicide) {
666  QProcess *proc;
667  int can_exit = 1;
668  do {
669  foreach (proc, startedProcesses) {
670  if (proc->state() != QProcess::NotRunning) {
671  can_exit = 0;
672  }
673  }
674  if (!can_exit) {
675  Sleep(2000);
676  }
677  } while (!can_exit);
678  return 0;
679  }
680 #endif
681  return 0;
682 }
