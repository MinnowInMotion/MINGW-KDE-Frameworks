221 /*
222  * Read 'len' bytes from 'sock' into buffer.
223  * returns -1 on failure, 0 on no data.
224  */
225 static int
226 read_socket(int sock, char *buffer, int len)
227 {
228  ssize_t result;
229  int bytes_left = len;
230  while (bytes_left > 0) {
231  // in case we get a request to start an application and data arrive
232  // to kdeinitSocket at the same time, requestStart() will already
233  // call slotKDEInitData(), so we must check there's still something
234  // to read, otherwise this would block
235 
236  // Same thing if kdeinit dies without warning.
237 
238  fd_set in;
239  timeval tm = { 30, 0 }; // 30 seconds timeout, so we're not stuck in case kdeinit dies on us
240  FD_ZERO(&in);
241  FD_SET(sock, &in);
242  select(sock + 1, &in, 0, 0, &tm);
243  if (!FD_ISSET(sock, &in)) {
244  qCDebug(KLAUNCHER) << "read_socket" << sock << "nothing to read, kdeinit5 must be dead";
245  return -1;
246  }
247 
248  result = read(sock, buffer, bytes_left);
249  if (result > 0) {
250  buffer += result;
251  bytes_left -= result;
252  } else if (result == 0) {
253  return -1;
254  } else if ((result == -1) && (errno != EINTR)) {
255  return -1;
256  }
257  }
258  return 0;
259 }
260 #endif

285 void KLauncher::processRequestReturn(int status, const QByteArray &requestData)
286 {
287  if (status == LAUNCHER_CHILD_DIED) {
288  long *request_data;
289  request_data = (long *) requestData.data();
290  processDied(request_data[0], request_data[1]);
291  return;
292  }
293  if (lastRequest && (status == LAUNCHER_OK)) {
294  long *request_data;
295  request_data = (long *) requestData.data();
296  lastRequest->pid = (pid_t)(*request_data);
297  qCDebug(KLAUNCHER).nospace() << lastRequest->name << " (pid " << lastRequest->pid << ") up and running.";
298  switch (lastRequest->dbus_startup_type) {
299  case KService::DBusNone:
300  if (lastRequest->wait) {
301  lastRequest->status = KLaunchRequest::Launching;
302  } else {
303  lastRequest->status = KLaunchRequest::Running;
304  }
305  break;
306  case KService::DBusUnique:
307  case KService::DBusMulti:
308  lastRequest->status = KLaunchRequest::Launching;
309  break;
310  }
311  lastRequest = 0;
312  return;
313  }
314  if (lastRequest && (status == LAUNCHER_ERROR)) {
315  lastRequest->status = KLaunchRequest::Error;
316  qCDebug(KLAUNCHER) << lastRequest->name << " failed.";
317  if (!requestData.isEmpty()) {
318  lastRequest->errorMsg = QString::fromUtf8((char *) requestData.data());
319  }
320  lastRequest = 0;
321  return;
322  }
323 
324  qWarning() << "Unexpected request return" << (unsigned int) status;
325 }

327 void
328 KLauncher::processDied(pid_t pid, long exitStatus)
329 {
330 #ifdef KLAUNCHER_VERBOSE_OUTPUT
331  qCDebug(KLAUNCHER) << pid << "exitStatus=" << exitStatus;
332 #else
333  Q_UNUSED(exitStatus);
334  // We should probably check the exitStatus for the uniqueapp case?
335 #endif
336  foreach (KLaunchRequest *request, requestList) {
337 #ifdef KLAUNCHER_VERBOSE_OUTPUT
338  qCDebug(KLAUNCHER) << " had pending request" << request->pid;
339 #endif
340  if (request->pid == pid) {
341  if ((request->dbus_startup_type == KService::DBusUnique)
342  && QDBusConnection::sessionBus().interface()->isServiceRegistered(request->dbus_name)) {
343  request->status = KLaunchRequest::Running;
344 #ifdef KLAUNCHER_VERBOSE_OUTPUT
345  qCDebug(KLAUNCHER) << pid << "running as a unique app";
346 #endif
347  } else if(request->dbus_startup_type == KService::DBusNone && request->wait) {
348  request->status = KLaunchRequest::Running;
349 #ifdef KLAUNCHER_VERBOSE_OUTPUT
350  qCDebug(KLAUNCHER) << pid << "running as DBusNone with wait to true";
351 #endif
352  } else {
353  request->status = KLaunchRequest::Error;
354 #ifdef KLAUNCHER_VERBOSE_OUTPUT
355  qCDebug(KLAUNCHER) << pid << "died, requestDone. status=" << request->status;
356 #endif
357  }
358  requestDone(request);
359  return;
360  }
361  }
362 #ifdef KLAUNCHER_VERBOSE_OUTPUT
363  qCDebug(KLAUNCHER) << "found no pending requests for PID" << pid;
364 #endif
365 }

367 static bool matchesPendingRequest(const QString &appId, const QString &pendingAppId)
368 {
369  // appId just registered, e.g. org.koffice.kword-12345
370  // Let's see if this is what pendingAppId (e.g. org.koffice.kword or *.kword) was waiting for.
371 
372  const QString newAppId = appId.left(appId.lastIndexOf(QLatin1Char('-'))); // strip out the -12345 if present.
373 
374  qCDebug(KLAUNCHER) << "appId=" << appId << "newAppId=" << newAppId << "pendingAppId=" << pendingAppId;
375 
376  if (pendingAppId.startsWith(QLatin1String("*."))) {
377  const QString pendingName = pendingAppId.mid(2);
378  const QString appName = newAppId.mid(newAppId.lastIndexOf(QLatin1Char('.')) + 1);
379  qCDebug(KLAUNCHER) << "appName=" << appName;
380  return appName == pendingName;
381  }
382 
383  return newAppId == pendingAppId;
384 }

386 void
387 KLauncher::slotNameOwnerChanged(const QString &appId, const QString &oldOwner,
388  const QString &newOwner)
389 {
390  Q_UNUSED(oldOwner);
391  if (appId.isEmpty() || newOwner.isEmpty()) {
392  return;
393  }
394 
395 #ifdef KLAUNCHER_VERBOSE_OUTPUT
396  qCDebug(KLAUNCHER) << "new app" << appId;
397 #endif
398  foreach (KLaunchRequest *request, requestList) {
399  if (request->status != KLaunchRequest::Launching) {
400  continue;
401  }
402 
403 #ifdef KLAUNCHER_VERBOSE_OUTPUT
404  qCDebug(KLAUNCHER) << "had pending request" << request->name << s_DBusStartupTypeToString[request->dbus_startup_type] << "dbus_name" << request->dbus_name << request->tolerant_dbus_name;
405 #endif
406  // For unique services check the requested service name first
407  if (request->dbus_startup_type == KService::DBusUnique) {
408  if ((appId == request->dbus_name) || // just started
409  QDBusConnection::sessionBus().interface()->isServiceRegistered(request->dbus_name)) { // was already running
410  request->status = KLaunchRequest::Running;
411 #ifdef KLAUNCHER_VERBOSE_OUTPUT
412  qCDebug(KLAUNCHER) << "OK, unique app" << request->dbus_name << "is running";
413 #endif
414  requestDone(request);
415  continue;
416  } else {
417 #ifdef KLAUNCHER_VERBOSE_OUTPUT
418  qCDebug(KLAUNCHER) << "unique app" << request->dbus_name << "not running yet";
419 #endif
420  }
421  }
422 
423  const QString rAppId = !request->tolerant_dbus_name.isEmpty() ? request->tolerant_dbus_name : request->dbus_name;
424 #ifdef KLAUNCHER_VERBOSE_OUTPUT
425  qCDebug(KLAUNCHER) << "using" << rAppId << "for matching";
426 #endif
427  if (rAppId.isEmpty()) {
428  continue;
429  }
430 
431  if (matchesPendingRequest(appId, rAppId)) {
432 #ifdef KLAUNCHER_VERBOSE_OUTPUT
433  qCDebug(KLAUNCHER) << "ok, request done";
434 #endif
435  request->dbus_name = appId;
436  request->status = KLaunchRequest::Running;
437  requestDone(request);
438  continue;
439  }
440  }

486 void
487 KLauncher::requestDone(KLaunchRequest *request)
488 {
489  if ((request->status == KLaunchRequest::Running) ||
490  (request->status == KLaunchRequest::Done)) {
491  requestResult.result = 0;
492  requestResult.dbusName = request->dbus_name;
493  requestResult.error = QStringLiteral(""); // not null, cf assert further down
494  requestResult.pid = request->pid;
495  } else {
496  requestResult.result = 1;
497  requestResult.dbusName.clear();
498  requestResult.error = i18n("KDEInit could not launch '%1'", request->name);
499  if (!request->errorMsg.isEmpty()) {
500  requestResult.error += QStringLiteral(":\n") + request->errorMsg;
501  }
502  requestResult.pid = 0;
503 
504 #if HAVE_XCB
505  if (!request->startup_dpy.isEmpty() && mIsX11) {
506  XCBConnection conn = getXCBConnection(request->startup_dpy);
507  if (conn) {
508  KStartupInfoId id;
509  id.initId(request->startup_id);
510  KStartupInfo::sendFinishXcb(conn.conn, conn.screen, id);
511  }
512  }
513 #endif
514  }
515 
516  if (request->autoStart) {
517  mAutoTimer.start(0);
518  }
519 
520  if (request->transaction.type() != QDBusMessage::InvalidMessage) {
521  if (requestResult.dbusName.isNull()) { // null strings can't be sent
522  requestResult.dbusName.clear();
523  }
524  Q_ASSERT(!requestResult.error.isNull());
525  quintptr stream_pid = requestResult.pid;
526  QDBusConnection::sessionBus().send(request->transaction.createReply(QVariantList() << requestResult.result
527  << requestResult.dbusName
528  << requestResult.error
529  << stream_pid));
530  }
531 #ifdef KLAUNCHER_VERBOSE_OUTPUT
532  qCDebug(KLAUNCHER) << "removing done request" << request->name << "PID" << request->pid;
533 #endif
534 
535  requestList.removeAll(request);
536  delete request;
537 }

546 void
547 KLauncher::requestStart(KLaunchRequest *request)
548 {
549 #ifdef USE_KPROCESS_FOR_KIOSLAVES
550  requestList.append(request);
551  lastRequest = request;
552 
553  QProcess *process = new QProcess;
554  process->setProcessChannelMode(QProcess::MergedChannels);
555  connect(process, SIGNAL(readyReadStandardOutput()), this, SLOT(slotGotOutput()));
556  connect(process, SIGNAL(finished(int,QProcess::ExitStatus)), this, SLOT(slotFinished(int,QProcess::ExitStatus)));
557  request->process = process;
558 
559 // process.setEnvironment(envlist);
560  QStringList args;
561  foreach (const QString &arg, request->arg_list) {
562  args << arg;
563  }
564 
565  QString executable = request->name;
566 #ifdef Q_OS_OSX
567  const QString bundlepath = QStandardPaths::findExecutable(executable);
568  if (!bundlepath.isEmpty()) {
569  executable = bundlepath;
570  }
571 #endif
572  process->start(executable, args);
573 
574  if (!process->waitForStarted()) {
575  processRequestReturn(LAUNCHER_ERROR, "");
576  } else {
577 #ifndef Q_OS_WIN
578  request->pid = process->pid();
579 #else
580  request->pid = process->pid()->dwProcessId;
581 #endif
582  QByteArray data((char *)&request->pid, sizeof(int));
583  processRequestReturn(LAUNCHER_OK, data);
584  }
585  return;
586 
587 #else
588  requestList.append(request);
589  // Send request to kdeinit.
590  klauncher_header request_header;
591  QByteArray requestData;
592  requestData.reserve(1024);
593 
594  appendLong(requestData, request->arg_list.count() + 1);
595  requestData.append(request->name.toLocal8Bit());
596  requestData.append('\0');
597  foreach (const QString &arg, request->arg_list) {
598  requestData.append(arg.toLocal8Bit()).append('\0');
599  }
600  appendLong(requestData, request->envs.count());
601  foreach (const QString &env, request->envs) {
602  requestData.append(env.toLocal8Bit()).append('\0');
603  }
604  appendLong(requestData, 0); // avoid_loops, always false here
605 #if HAVE_X11
606  bool startup_notify = mIsX11 && !request->startup_id.isNull() && request->startup_id != "0";
607  if (startup_notify) {
608  requestData.append(request->startup_id).append('\0');
609  }
610 #endif
611  if (!request->cwd.isEmpty()) {
612  requestData.append(QFile::encodeName(request->cwd)).append('\0');
613  }
614 
615 #if HAVE_X11
616  request_header.cmd = startup_notify ? LAUNCHER_EXT_EXEC : LAUNCHER_EXEC_NEW;
617 #else
618  request_header.cmd = LAUNCHER_EXEC_NEW;
619 #endif
620  request_header.arg_length = requestData.length();
621 
622 #ifdef KLAUNCHER_VERBOSE_OUTPUT
623  qCDebug(KLAUNCHER) << "Asking kdeinit to start" << request->name << request->arg_list
624  << "cmd=" << commandToString(request_header.cmd);
625 #endif
626 
627  kde_safe_write(kdeinitSocket, &request_header, sizeof(request_header));

701 bool
702 KLauncher::start_service(KService::Ptr service, const QStringList &_urls,
703  const QStringList &envs, const QByteArray &startup_id,
704  bool blind, bool autoStart, const QDBusMessage &msg)
705 {
706  QStringList urls = _urls;
707  bool runPermitted = KDesktopFile::isAuthorizedDesktopFile(service->entryPath());
708 
709  if (!service->isValid() || !runPermitted) {
710  requestResult.result = ENOEXEC;
711  if (service->isValid()) {
712  requestResult.error = i18n("Service '%1' must be executable to run.", service->entryPath());
713  } else {
714  requestResult.error = i18n("Service '%1' is malformatted.", service->entryPath());
715  }
716  cancel_service_startup_info(NULL, startup_id, envs); // cancel it if any
717  return false;
718  }
719  KLaunchRequest *request = new KLaunchRequest;
720  request->autoStart = autoStart;
721 
722  if ((urls.count() > 1) && !service->allowMultipleFiles()) {
723  // We need to launch the application N times. That sucks.
724  // We ignore the result for application 2 to N.
725  // For the first file we launch the application in the
726  // usual way. The reported result is based on this
727  // application.
728  QStringList::ConstIterator it = urls.constBegin();
729  for (++it;
730  it != urls.constEnd();
731  ++it) {
732  QStringList singleUrl;
733  singleUrl.append(*it);
734  QByteArray startup_id2 = startup_id;
735  if (!startup_id2.isEmpty() && startup_id2 != "0") {
736  startup_id2 = "0"; // can't use the same startup_id several times // krazy:exclude=doublequote_chars
737  }
738  start_service(service, singleUrl, envs, startup_id2, true, false, msg);
739  }
740  const QString firstURL = urls.at(0);
741  urls.clear();
742  urls.append(firstURL);
743  }
744  const QList<QUrl> qurls = QUrl::fromStringList(urls);
745 
746  createArgs(request, service, qurls);
747 
748  // We must have one argument at least!
749  if (request->arg_list.isEmpty()) {
750  requestResult.result = ENOEXEC;
751  requestResult.error = i18n("Service '%1' is malformatted.", service->entryPath());
752  delete request;
753  cancel_service_startup_info(NULL, startup_id, envs);
754  return false;
755  }
756 
757  request->name = request->arg_list.takeFirst();
758 
759  if (request->name.endsWith(QLatin1String("/kioexec"))) {
760  // Special case for kioexec; if createArgs said we were going to use it,
761  // then we have to expect a kioexec-PID, not a org.kde.finalapp...
762  // Testcase: konqueror www.kde.org, RMB on link, open with, kruler.
763 
764  request->dbus_startup_type = KService::DBusMulti;
765  request->dbus_name = QStringLiteral("org.kde.kioexec");
766  } else {
767  request->dbus_startup_type = service->dbusStartupType();
768 
769  if ((request->dbus_startup_type == KService::DBusUnique) ||
770  (request->dbus_startup_type == KService::DBusMulti)) {
771  const QVariant v = service->property(QStringLiteral("X-DBUS-ServiceName"));
772  if (v.isValid()) {
773  request->dbus_name = v.toString();
774  }
775  if (request->dbus_name.isEmpty()) {
776  const QString binName = KIO::DesktopExecParser::executableName(service->exec());
777  request->dbus_name = QStringLiteral("org.kde.") + binName;
778  request->tolerant_dbus_name = QStringLiteral("*.") + binName;
779  }
780  }
781  }
782 
783 #ifdef KLAUNCHER_VERBOSE_OUTPUT
784  qCDebug(KLAUNCHER) << "name=" << request->name << "dbus_name=" << request->dbus_name
785  << "startup type=" << s_DBusStartupTypeToString[request->dbus_startup_type];
786 #endif
787 
788  request->pid = 0;
789  request->wait = false;
790  request->envs = envs;
791  send_service_startup_info(request, service, startup_id, envs);
792 
793  // Request will be handled later.
794  if (!blind && !autoStart) {
795  msg.setDelayedReply(true);
796  request->transaction = msg;
797  }
798  queueRequest(request);
799  return true;
800 }

802 void
803 KLauncher::send_service_startup_info(KLaunchRequest *request, KService::Ptr service, const QByteArray &startup_id,
804  const QStringList &envs)

807  if (!mIsX11) {
808  return;

846  }

1241 #if HAVE_XCB
1242 KLauncher::XCBConnection KLauncher::getXCBConnection(const QByteArray &_displayName)
1243 {
1244  const auto displayName = !_displayName.isEmpty() ? _displayName : qgetenv("DISPLAY");
1245 
1246  // If cached connection is same as request
1247  if (mCached && displayName == mCached.displayName) {
1248  // Check error, if so close it, otherwise it's still valid, reuse the cached one
1249  if (xcb_connection_has_error(mCached.conn)) {
1250  close();
1251  } else {
1252  return mCached;
1253  }
1254  }
1255 
1256  // At this point, the cached connection can't be reused, make a new connection
1257  XCBConnection conn;
1258  conn.conn = xcb_connect(displayName.constData(), &conn.screen);
1259  if (conn) {
1260  // check error first, if so return an empty one
1261  if (xcb_connection_has_error(conn.conn)) {
1262  xcb_disconnect(conn.conn);
1263  return XCBConnection();
1264  }
1265 
1266  // if it's valid, replace mCached with new connection
