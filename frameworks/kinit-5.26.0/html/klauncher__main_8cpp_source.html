44 static void sig_handler(int sig_num)
45 {
46  // No recursion
47  signal(SIGHUP, SIG_IGN);
48  signal(SIGTERM, SIG_IGN);
49  fprintf(stderr, "klauncher: Exiting on signal %d\n", sig_num);

63 bool mac_set_dbus_address(QString value)
64 {
65  if (!value.isEmpty() && QFile::exists(value) && (QFile::permissions(value) & QFile::WriteUser)) {
66  value = QLatin1String("unix:path=") + value;
67  qputenv("DBUS_SESSION_BUS_ADDRESS", value.toLocal8Bit());
68  // qDebug() << "set session bus address to" << value;
69  return true;
70  }
71  return false;
72 }

78 void mac_initialize_dbus()
79 {
80  enum
81  {
82  timeout = 3000 // msec. Copied from old kdecore/kernel/kkernel_mac.cpp
83  };
84 
85  if (dbus_initialized) {
86  return;
87  }
88 
89  QString dbusVar = QString::fromLocal8Bit(qgetenv("DBUS_SESSION_BUS_ADDRESS"));
90  if (!dbusVar.isEmpty()) {
91  dbus_initialized = true;
92  return;
93  }
94 
95  dbusVar = QFile::decodeName(qgetenv("DBUS_LAUNCHD_SESSION_BUS_SOCKET"));
96  if (mac_set_dbus_address(dbusVar)) {
97  dbus_initialized = true;
98  return;
99  }
100 
101  QString externalProc;
102  QStringList path = QFile::decodeName(qgetenv("KDEDIRS")).split(QLatin1Char(':')).replaceInStrings(QRegExp(QLatin1String("$")), QLatin1String("/bin"));
103  path << QFile::decodeName(qgetenv("PATH")).split(QLatin1Char(':')) << QLatin1String("/usr/local/bin");
104 
105  for (int i = 0; i < path.size(); ++i) {
106  QString testLaunchctl = QString(path.at(i)).append(QLatin1String("/launchctl"));
107  if (QFile(testLaunchctl).exists()) {
108  externalProc = testLaunchctl;
109  break;
110  }
111  }
112 
113  if (!externalProc.isEmpty()) {
114  QProcess qp;
115  qp.setTextModeEnabled(true);
116 
117  qp.start(externalProc, QStringList() << QLatin1String("getenv") << QLatin1String("DBUS_LAUNCHD_SESSION_BUS_SOCKET"));
118  if (!qp.waitForFinished(timeout)) {
119  // qDebug() << "error running" << externalProc << qp.errorString();
120  return;
121  }
122  if (qp.exitCode() != 0) {
123  // qDebug() << externalProc << "unsuccessful:" << qp.readAllStandardError();
124  return;
125  }
126 
127  QString line = QString::fromLatin1(qp.readLine()).trimmed(); // read the first line
128  if (mac_set_dbus_address(line)) {
129  dbus_initialized = true; // hooray
130  }
131  }
132 
133  if (dbus_initialized == false) {
134  // qDebug() << "warning: unable to initialize D-Bus environment!";
135  }
136 
137 }

140 extern "C" Q_DECL_EXPORT int kdemain(int argc, char **argv)
141 {
142 #ifndef Q_OS_WIN
143  // Started via kdeinit.
144  int launcherFd;
145  if (argc != 2 || memcmp(argv[1], "--fd=", 5) || !(launcherFd = atoi(argv[1] + 5))) {
146  fprintf(stderr, "%s", i18n("klauncher: This program is not supposed to be started manually.\n"
147  "klauncher: It is started automatically by kdeinit5.\n").toLocal8Bit().data());
148  return 1;
149  }
150 #endif
151 
152 #if defined(Q_OS_DARWIN)
153  CFBundleRef mainBundle = CFBundleGetMainBundle();
154  if (mainBundle) {
155  // get the application's Info Dictionary. For app bundles this would live in the bundle's Info.plist,
156  // for regular executables it is obtained in another way.
157  CFMutableDictionaryRef infoDict = (CFMutableDictionaryRef) CFBundleGetInfoDictionary(mainBundle);
158  if (infoDict) {
159  // Add or set the "LSUIElement" key with/to value "1". This can simply be a CFString.
160  CFDictionarySetValue(infoDict, CFSTR("LSUIElement"), CFSTR("1"));
161  // That's it. We're now considered as an "agent" by the window server, and thus will have
162  // neither menubar nor presence in the Dock or App Switcher.
163  }
164  }
165  mac_initialize_dbus();
166 #endif
167 
168  // WABA: Make sure not to enable session management.
169  qunsetenv("SESSION_MANAGER");
170 
171  // Disable the GLib event loop (rh#983110)
172  const bool wasQtNoGlibSet = !qEnvironmentVariableIsEmpty("QT_NO_GLIB");
173  if (!wasQtNoGlibSet) {
174  qputenv("QT_NO_GLIB", "1");
175  }
176 
177  // We need a QGuiApplication as we use X11
178  QGuiApplication app(argc, argv);
179  app.setApplicationName(QStringLiteral("klauncher"));
180 
181  // Now get rid of QT_NO_GLIB again so launched processes don't inherit it
182  if (!wasQtNoGlibSet) {
183  qunsetenv("QT_NO_GLIB");
184  }
185 
186  int maxTry = 3;
187  while (true) {
188  QString service(QStringLiteral("org.kde.klauncher5")); // same as ktoolinvocation.cpp
189  if (!QDBusConnection::sessionBus().isConnected()) {
190  qWarning() << "No DBUS session-bus found. Check if you have started the DBUS server.";
191  return 1;
192  }
193  QDBusReply<QDBusConnectionInterface::RegisterServiceReply> reply =
194  QDBusConnection::sessionBus().interface()->registerService(service);
195  if (!reply.isValid()) {
196  qWarning() << "DBUS communication problem!";
197  return 1;
198  }
199  if (reply == QDBusConnectionInterface::ServiceRegistered) {
200  break;
201  }
202 
203  if (--maxTry == 0) {
204  qWarning() << "Another instance of klauncher is already running!";
205  return 1;
206  }
207 
208  // Wait a bit...
209  qWarning() << "Waiting for already running klauncher to exit.";
210  QThread::sleep(1);
211 
212  // Try again...
213  }
214 
215 #ifndef USE_KPROCESS_FOR_KIOSLAVES
216  KLauncher *launcher = new KLauncher(launcherFd);
217 #else
218  KLauncher *launcher = new KLauncher();
219 #endif
220  QDBusConnection::sessionBus().registerObject(QStringLiteral("/"), launcher);
221 
222 #ifndef USE_KPROCESS_FOR_KIOSLAVES
223  if (pipe(sigpipe) != 0) {
224  perror("klauncher: pipe failed.");
225  return 1;
226  }
227  QSocketNotifier *signotif = new QSocketNotifier(sigpipe[ 0 ], QSocketNotifier::Read, launcher);
228  QObject::connect(signotif, SIGNAL(activated(int)), launcher, SLOT(destruct()));
229  KCrash::setEmergencySaveFunction(sig_handler);
230  signal(SIGHUP, sig_handler);
231  signal(SIGPIPE, SIG_IGN);
232  signal(SIGTERM, sig_handler);
233 #endif
234 
235  return app.exec();
236 }
