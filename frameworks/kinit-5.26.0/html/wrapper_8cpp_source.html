60 static const QString generate_socket_file_name()
61 {
62 
63 #if HAVE_X11 || HAVE_XCB // qt5: see displayEnvVarName_c()
64  QByteArray display = qgetenv(displayEnvVarName_c());
65  if (display.isEmpty()) {
66  fprintf(stderr, "Error: could not determine $%s.\n", displayEnvVarName_c());
67  return QString();
68  }
69  int i;
70  if ((i = display.lastIndexOf('.')) > display.lastIndexOf(':') && i >= 0) {
71  display.truncate(i);
72  }
73 
74  display.replace(':', '_');
75 #ifdef __APPLE__
76  // not entirely impossible, so let's leave it
77  display.replace('/', '_');
78 #endif
79 #else
80  // not using a DISPLAY variable; use an empty string instead
81  QByteArray display = "";
82 #endif
83  // WARNING, if you change the socket name, adjust kwrapper too
84  const QString socketFileName = QStringLiteral("kdeinit5_%1").arg(QLatin1String(display));
85  return socketFileName;
86 }

88 /*
89  * Write 'len' bytes from 'buffer' into 'sock'.
90  * returns 0 on success, -1 on failure.
91  */
92 static int write_socket(int sock, char *buffer, int len)
93 {
94  ssize_t result;
95  int bytes_left = len;
96  while (bytes_left > 0) {
97  result = write(sock, buffer, bytes_left);
98  if (result > 0) {
99  buffer += result;
100  bytes_left -= result;
101  } else if (result == 0) {
102  return -1;
103  } else if ((result == -1) && (errno != EINTR) && (errno != EAGAIN)) {
104  return -1;
105  }
106  }
107  return 0;
108 }

132 static int openSocket()
133 {
134  const QString socketFileName = generate_socket_file_name();
135  if (socketFileName.isEmpty()) {
136  return -1;
137  }
138  QByteArray socketName = QFile::encodeName(QStandardPaths::writableLocation(QStandardPaths::RuntimeLocation) +
139  QLatin1Char('/') + socketFileName);
140  const char *sock_file = socketName.constData();
141 
142  struct sockaddr_un server;
143  if (strlen(sock_file) >= sizeof(server.sun_path)) {
144  fprintf(stderr, "Warning: Path of socketfile exceeds UNIX_PATH_MAX.\n");
145  return -1;
146  }
147 
148  /*
149  * create the socket stream
150  */
151  int s = socket(PF_UNIX, SOCK_STREAM, 0);
152  if (s < 0) {
153  perror("Warning: socket() failed: ");
154  return -1;
155  }
156 
157  server.sun_family = AF_UNIX;
158  strcpy(server.sun_path, sock_file);
159  kde_socklen_t socklen = sizeof(server);
160  if (connect(s, (struct sockaddr *)&server, socklen) == -1) {
161  fprintf(stderr, "kdeinit5_wrapper: Warning: connect(%s) failed:", sock_file);
162  perror(" ");
163  close(s);
164  return -1;
165  }
166  return s;
167 }

174 static void setup_signal_handler(int signo, int clean)
175 {
176  struct sigaction sa;
177  if (clean) {
178  sa.sa_handler = SIG_DFL;
179  } else {
180  sa.sa_handler = sig_pass_handler;
181  }
182  sigemptyset(&sa.sa_mask);
183  sigaddset(&sa.sa_mask, signo);
184  sa.sa_flags = 0; /* don't use SA_RESTART */
185  sigaction(signo, &sa, 0);
186 }

188 static void sig_pass_handler(int signo)
189 {
190  int save_errno = errno;
191  if (signo == SIGTSTP) {
192  kill(kwrapper_pid, SIGSTOP); /* pass the signal to the real process */
193  } else { /* SIGTSTP wouldn't work ... I don't think is much */
194  kill(kwrapper_pid, signo); /* of a problem */
195  }
196 
197  if (signo == SIGCONT) {
198  setup_signals(); /* restore signals */
199  } else if (signo == SIGCHLD)
200  ; /* nothing, ignore */
201  else { /* do the default action ( most of them quit the app ) */
202  setup_signal_handler(signo, 1);
203  raise(signo); /* handle the signal again */
204  }
205 
206  errno = save_errno;
207 }

233 static int kwrapper_run(pid_t wrapped, int sock)
234 {
235  klauncher_header header;
236  char *buffer;
237  long pid, status;
238 
239  kwrapper_pid = wrapped;
240  setup_signals();
241 
242  read_socket(sock, (char *)&header, sizeof(header));
243 
244  if (header.cmd != LAUNCHER_CHILD_DIED) {
245  fprintf(stderr, "Unexpected response from KInit (response = %ld).\n", header.cmd);
246  exit(255);
247  }
248 
249  buffer = (char *) malloc(header.arg_length);
250  if (buffer == NULL) {
251  perror("Error: malloc() failed\n");
252  exit(255);
253  }
254 
255  read_socket(sock, buffer, header.arg_length);
256  pid = ((long *) buffer)[0];
257  if (pid != kwrapper_pid) {
258  fprintf(stderr, "Unexpected LAUNCHER_CHILD_DIED from KInit - pid = %ld\n", pid);
259  exit(255);
260  }
261 
262  status = ((long *) buffer)[1];
263  free(buffer);
264  return (int) status;
265 }

267 int main(int argc, char **argv)
268 {
269  int i;
270  int wrapper = 0;
271  int ext_wrapper = 0;
272  int kwrapper = 0;
273  long arg_count;
274  long env_count;
275  klauncher_header header;
276  char *start, *p, *buffer;
277  char cwd[8192];
278  const char *tty = NULL;
279  long avoid_loops = 0;
280  const char *startup_id = NULL;
281  int sock;
282 
283  long size = 0;
284 
285  start = argv[0];
286  p = start + strlen(argv[0]);
287  while (--p > start) {
288  if (*p == '/') {
289  break;
290  }
291  }
292  if (p > start) {
293  p++;
294  }
295  start = p;
296 
297  if (strcmp(start, "kdeinit5_wrapper") == 0) {
298  wrapper = 1;
299  } else if (strcmp(start, "kshell5") == 0) {
300  ext_wrapper = 1;
301  } else if (strcmp(start, "kwrapper5") == 0) {
302  kwrapper = 1;
303  } else if (strcmp(start, "kdeinit5_shutdown") == 0) {
304  if (argc > 1) {
305  fprintf(stderr, "Usage: %s\n\n", start);
306  fprintf(stderr, "Shuts down kdeinit5 master process and terminates all processes spawned from it.\n");
307  exit(255);
308  }
309  sock = openSocket();
310  if (sock < 0) {
311  fprintf(stderr, "Error: Can not contact kdeinit5!\n");
312  exit(255);
313  }
314  header.cmd = LAUNCHER_TERMINATE_KDE;
315  header.arg_length = 0;
316  write_socket(sock, (char *) &header, sizeof(header));
317  read_socket(sock, (char *) &header, 1); /* wait for the socket to close */
318  return 0;
319  }
320 
321  if (wrapper || ext_wrapper || kwrapper) {
322  argv++;
323  argc--;
324  if (argc < 1) {
325  fprintf(stderr, "Usage: %s <application> [<args>]\n", start);
326  exit(255); /* usage should be documented somewhere ... */
327  }
328  start = argv[0];
329  }
330 
331  sock = openSocket();
332  if (sock < 0) { /* couldn't contact kdeinit5, start argv[ 0 ] directly */
333  execvp(argv[ 0 ], argv);
334  fprintf(stderr, "Error: Can not run %s !\n", argv[ 0 ]);
335  exit(255);
336  }
337 
338  if (!wrapper && !ext_wrapper && !kwrapper) {
339  /* was called as a symlink */
340  avoid_loops = 1;
341 #if defined(WE_ARE_KWRAPPER)
342  kwrapper = 1;
343 #elif defined(WE_ARE_KSHELL)
344  ext_wrapper = 1;
345 #else
346  wrapper = 1;
347 #endif
348  }
349 
350  arg_count = argc;
351  env_count = 0;
352 
353  size += sizeof(long); /* Number of arguments*/
354 
355  size += strlen(start) + 1; /* Size of first argument. */
356 
357  for (i = 1; i < argc; i++) {
358  size += strlen(argv[i]) + 1;
359  }
360  if (wrapper) {
361  size += sizeof(long); /* empty envs */
362  }
363  if (ext_wrapper || kwrapper) {
364  if (!getcwd(cwd, 8192)) {
365  cwd[0] = '\0';
366  }
367  size += strlen(cwd) + 1;
368 
369  size += sizeof(long); /* Number of env.vars. */
370 
371  for (; environ[env_count]; env_count++) {
372  int l = strlen(environ[env_count]) + 1;
373  size += l;
374  }
375 
376  if (kwrapper) {
377  tty = ttyname(1);
378  if (!tty || !isatty(2)) {
379  tty = "";
380  }
381  size += strlen(tty) + 1;
382  }
383  }
384 
385  size += sizeof(avoid_loops);
386 
387  if (!wrapper) {
388  startup_id = getenv("DESKTOP_STARTUP_ID");
389  if (startup_id == NULL) {
390  startup_id = "";
391  }
392  size += strlen(startup_id) + 1;
393  }
394 
395  if (wrapper) {
396  header.cmd = LAUNCHER_EXEC_NEW;
397  } else if (kwrapper) {
398  header.cmd = LAUNCHER_KWRAPPER;
399  } else {
400  header.cmd = LAUNCHER_SHELL;
401  }
402  header.arg_length = size;
403  write_socket(sock, (char *) &header, sizeof(header));
404 
405  buffer = (char *) malloc(size);
406  if (buffer == NULL) {
407  perror("Error: malloc() failed.");
408  exit(255);
409  }
410  p = buffer;
411 
412  memcpy(p, &arg_count, sizeof(arg_count));
413  p += sizeof(arg_count);
414 
415  memcpy(p, start, strlen(start) + 1);
416  p += strlen(start) + 1;
417 
418  for (i = 1; i < argc; i++) {
419  memcpy(p, argv[i], strlen(argv[i]) + 1);
420  p += strlen(argv[i]) + 1;
421  }
422 
423  if (wrapper) {
424  long dummy = 0;
425  memcpy(p, &dummy, sizeof(dummy)); /* empty envc */
426  p += sizeof(dummy);
427  }
428  if (ext_wrapper || kwrapper) {
429  memcpy(p, cwd, strlen(cwd) + 1);
430  p += strlen(cwd) + 1;
431 
432  memcpy(p, &env_count, sizeof(env_count));
433  p += sizeof(env_count);
434 
435  for (i = 0; i < env_count; i++) {
436  int l = strlen(environ[i]) + 1;
437  memcpy(p, environ[i], l);
438  p += l;
439  }
440 
441  if (kwrapper) {
442  memcpy(p, tty, strlen(tty) + 1);
443  p += strlen(tty) + 1;
444  }
445  }
446 
447  memcpy(p, &avoid_loops, sizeof(avoid_loops));
448  p += sizeof(avoid_loops);
449 
450  if (!wrapper) {
451  memcpy(p, startup_id, strlen(startup_id) + 1);
452  p += strlen(startup_id) + 1;
453  }
454 
455  if (p - buffer != size) /* should fail only if you change this source and do */
456  /* a stupid mistake, it should be assert() actually */
457  {
458  fprintf(stderr, "Oops. Invalid format.\n");
459  exit(255);
460  }
461 
462  write_socket(sock, buffer, size);
463  free(buffer);
464 
465  if (read_socket(sock, (char *) &header, sizeof(header)) == -1) {
466  fprintf(stderr, "Communication error with KInit.\n");
467  exit(255);
468  }
469 
470  if (header.cmd == LAUNCHER_OK) {
471  long pid;
472  buffer = (char *) malloc(header.arg_length);
473  if (buffer == NULL) {
474  perror("Error: malloc() failed\n");
475  exit(255);
476  }
477  read_socket(sock, buffer, header.arg_length);
478  pid = *((long *) buffer);
479  if (!kwrapper) { /* kwrapper shouldn't print any output */
480  printf("Launched ok, pid = %ld\n", pid);
481  } else {
482  exit(kwrapper_run(pid, sock));
483  }
484  } else if (header.cmd == LAUNCHER_ERROR) {
485  fprintf(stderr, "KInit could not launch '%s'.\n", start);
486  exit(255);
487  } else {
488  fprintf(stderr, "Unexpected response from KInit (response = %ld).\n", header.cmd);
489  exit(255);
490  }
491  exit(0);
492 }
