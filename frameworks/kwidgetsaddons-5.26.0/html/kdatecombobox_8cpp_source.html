113 QString KDateComboBoxPrivate::dateFormat(QLocale::FormatType format)
114 {
115  // Clearly a workaround for QLocale using "yy" way too often for years
116  // and so you get no way to distinguish between 1913 and 2013 anymore...
117  // bummer.
118  QString res = q->locale().dateFormat(format);
119  res.replace(QStringLiteral("yy"), QStringLiteral("yyyy"));
120  res.replace(QStringLiteral("yyyyyyyy"), QStringLiteral("yyyy"));
121  return res;
122 }

129 void KDateComboBoxPrivate::initDateWidget()
130 {
131  q->blockSignals(true);
132  q->clear();
133 
134  // If EditTime then set the line edit
135  q->lineEdit()->setReadOnly((m_options & KDateComboBox::EditDate) != KDateComboBox::EditDate);
136 
137  // If SelectTime then make list items visible
138  if ((m_options & KDateComboBox::SelectDate) == KDateComboBox::SelectDate ||
139  (m_options & KDateComboBox::DatePicker) == KDateComboBox::DatePicker ||
140  (m_options & KDateComboBox::DatePicker) == KDateComboBox::DateKeywords) {
141  q->setMaxVisibleItems(1);
142  } else {
143  q->setMaxVisibleItems(0);
144  }
145 
146  q->setSizeAdjustPolicy(QComboBox::AdjustToContents);
147  q->addItem(formatDate(m_date));
148  q->setCurrentIndex(0);
149  q->setSizeAdjustPolicy(QComboBox::AdjustToContentsOnFirstShow);
150  q->blockSignals(false);
151 
152  m_dateMenu->clear();
153  m_actions.clear();
154 
155  if ((m_options & KDateComboBox::SelectDate) == KDateComboBox::SelectDate) {
156 
157  if ((m_options & KDateComboBox::DatePicker) == KDateComboBox::DatePicker) {
158  m_dateMenu->addAction(m_datePickerAction);
159  m_dateMenu->addSeparator();
160  }
161 
162  if ((m_options & KDateComboBox::DateKeywords) == KDateComboBox::DateKeywords) {
163  if (m_dateMap.isEmpty()) {
164  addMenuAction(KDateComboBox::tr("Next Year", "@option next year"), m_date.addYears(1));
165  addMenuAction(KDateComboBox::tr("Next Month", "@option next month"), m_date.addMonths(1));
166  addMenuAction(KDateComboBox::tr("Next Week", "@option next week"), m_date.addDays(7));
167  addMenuAction(KDateComboBox::tr("Tomorrow", "@option tomorrow"), m_date.addDays(1));
168  addMenuAction(KDateComboBox::tr("Today", "@option today"), m_date);
169  addMenuAction(KDateComboBox::tr("Yesterday", "@option yesterday"), m_date.addDays(-1));
170  addMenuAction(KDateComboBox::tr("Last Week", "@option last week"), m_date.addDays(-7));
171  addMenuAction(KDateComboBox::tr("Last Month", "@option last month"), m_date.addMonths(-1));
172  addMenuAction(KDateComboBox::tr("Last Year", "@option last year"), m_date.addYears(-1));
173  m_dateMenu->addSeparator();
174  addMenuAction(KDateComboBox::tr("No Date", "@option do not specify a date"), QDate());
175 
176  } else {
177  QMapIterator<QDate, QString> i(m_dateMap);
178  while (i.hasNext()) {
179  i.next();
180  if (i.value().isEmpty()) {
181  addMenuAction(formatDate(i.key()), i.key());
182  } else if (i.value().toLower() == QLatin1String("separator")) {
183  m_dateMenu->addSeparator();
184  } else {
185  addMenuAction(i.value(), i.key());
186  }
187  }
188  }
189  enableMenuDates();
190  }
191  }
192 }

203 void KDateComboBoxPrivate::enableMenuDates()
204 {
205  // Hide menu dates if they are outside the date range
206  for (int i = 0; i < m_actions.count(); ++i) {
207  QDate date = m_actions[i]->data().toDate();
208  m_actions[i]->setVisible(!date.isValid() || (date >= m_minDate && date <= m_maxDate));
209  }

266 void KDateComboBoxPrivate::warnDate()
267 {
268  if (!m_warningShown && !q->isValid() &&
269  (m_options & KDateComboBox::WarnOnInvalid) == KDateComboBox::WarnOnInvalid) {
270  QString warnMsg;
271  if (!m_date.isValid()) {
272  warnMsg = KDateComboBox::tr("The date you entered is invalid", "@info");
273  } else if (m_minDate.isValid() && m_date < m_minDate) {
274  if (m_minWarnMsg.isEmpty()) {
275  warnMsg = KDateComboBox::tr("Date cannot be earlier than %1", "@info").arg(formatDate(m_minDate));
276  } else {
277  warnMsg = m_minWarnMsg;
278  warnMsg.replace(QStringLiteral("%1"), formatDate(m_minDate));
279  }
280  } else if (m_maxDate.isValid() && m_date > m_maxDate) {
281  if (m_maxWarnMsg.isEmpty()) {
282  warnMsg = KDateComboBox::tr("Date cannot be later than %1", "@info").arg(formatDate(m_maxDate));
283  } else {
284  warnMsg = m_maxWarnMsg;
285  warnMsg.replace(QStringLiteral("%1"), formatDate(m_maxDate));
286  }
287  }
288  m_warningShown = true;
289  KMessageBox::sorry(q, warnMsg);
290  }

458 void KDateComboBox::keyPressEvent(QKeyEvent *keyEvent)
459 {
460  QDate temp;
461  switch (keyEvent->key()) {
462  case Qt::Key_Down:
463  temp = d->m_date.addDays(-1);
464  break;
465  case Qt::Key_Up:
466  temp = d->m_date.addDays(1);
467  break;
468  case Qt::Key_PageDown:
469  temp = d->m_date.addMonths(-1);
470  break;
471  case Qt::Key_PageUp:
472  temp = d->m_date.addMonths(1);
473  break;
474  default:
475  QComboBox::keyPressEvent(keyEvent);
476  return;
477  }
478  if (temp.isValid() && temp >= d->m_minDate && temp <= d->m_maxDate) {
479  d->enterDate(temp);
480  }
481 }

490 void KDateComboBox::showPopup()
491 {
492  if (!isEditable() ||
493  !d->m_dateMenu ||
494  (d->m_options & KDateComboBox::SelectDate) != KDateComboBox::SelectDate) {
495  return;
496  }
497 
498  d->m_datePicker->blockSignals(true);
499  d->m_datePicker->setDate(d->m_date);
500  d->m_datePicker->blockSignals(false);
501 
502  const QRect desk = QApplication::desktop()->screenGeometry(this);
503 
504  QPoint popupPoint = mapToGlobal(QPoint(0, 0));
505 
506  const int dateFrameHeight = d->m_dateMenu->sizeHint().height();
507  if (popupPoint.y() + height() + dateFrameHeight > desk.bottom()) {
508  popupPoint.setY(popupPoint.y() - dateFrameHeight);
509  } else {
510  popupPoint.setY(popupPoint.y() + height());
511  }
512 
513  const int dateFrameWidth = d->m_dateMenu->sizeHint().width();
514  if (popupPoint.x() + dateFrameWidth > desk.right()) {
515  popupPoint.setX(desk.right() - dateFrameWidth);
516  }
517 
518  if (popupPoint.x() < desk.left()) {
519  popupPoint.setX(desk.left());
520  }
521 
522  if (popupPoint.y() < desk.top()) {
523  popupPoint.setY(desk.top());
524  }
525 
526  d->m_dateMenu->popup(popupPoint);
527 }
