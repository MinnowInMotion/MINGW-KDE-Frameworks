36 void KPixmapSequence::Private::loadSequence(const QPixmap &bigPixmap, const QSize &frameSize)
37 {
38  if (bigPixmap.isNull()) {
39  qWarning() << "Invalid pixmap specified.";
40  return;
41  }
42 
43  QSize size(frameSize);
44  if (!size.isValid()) {
45  size = QSize(bigPixmap.width(), bigPixmap.width());
46  }
47  if (bigPixmap.width() % size.width() ||
48  bigPixmap.height() % size.height()) {
49  qWarning() << "Invalid framesize.";
50  return;
51  }
52 
53  const int rowCount = bigPixmap.height() / size.height();
54  const int colCount = bigPixmap.width() / size.width();
55  mFrames.resize(rowCount * colCount);
56 
57  int pos = 0;
58  for (int row = 0; row < rowCount; ++row) {
59  for (int col = 0; col < colCount; ++col) {
60  QPixmap pix = bigPixmap.copy(col * size.width(), row * size.height(), size.width(), size.height());
61  mFrames[pos++] = pix;
62  }
63  }
64 }
