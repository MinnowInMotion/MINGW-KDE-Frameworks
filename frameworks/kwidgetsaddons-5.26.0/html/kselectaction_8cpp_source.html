43 // QAction::setText("Hi") and then KPopupAccelManager exec'ing, causes
44 // QAction::text() to return "&Hi" :( Comboboxes don't have accels and
45 // display ampersands literally.
46 static QString DropAmpersands(const QString &text)
47 {
48  QString label = text;
49 
50  int p = label.indexOf(QLatin1Char('&'));
51  while (p >= 0 && p < label.length() - 1) {
52  if (label[p + 1].isLetterOrNumber() // Valid accelerator.
53  || label[p + 1] == QLatin1Char('&')) { // Escaped accelerator marker.
54  label = label.left(p) + label.mid(p + 1);
55  }
56 
57  p = label.indexOf(QLatin1Char('&'), p + 1);
58  }
59 
60  return label;
61 }

185 QAction *KSelectAction::action(const QString &text, Qt::CaseSensitivity cs) const
186 {
187  QString compare;
188  if (cs == Qt::CaseSensitive) {
189  compare = text;
190  } else {
191  compare = text.toLower();
192  }
193 
194  foreach (QAction *action, selectableActionGroup()->actions()) {
195  const QString text = ::DropAmpersands(action->text());
196  if (cs == Qt::CaseSensitive) {
197  if (text == compare) {
198  return action;
199  }
200 
201  } else if (cs == Qt::CaseInsensitive) {
202  if (text.toLower() == compare) {
203  return action;

398 void KSelectAction::clear()
399 {
400  Q_D(KSelectAction);
401  //qDebug () << "KSelectAction::clear()";
402 
403  // we need to delete the actions later since we may get a call to clear()
404  // from a method called due to a triggered(...) signal
405  const QList<QAction *> actions = d->m_actionGroup->actions();
406  for (int i = 0; i < actions.count(); ++i) {
407  // deleteLater() only removes us from the actions() list (among
408  // other things) on the next entry into the event loop. Until then,
409  // e.g. action() and setCurrentItem() will be working on items
410  // that are supposed to have been deleted. So detach the action to
411  // prevent this from happening.
412  removeAction(actions[i]);
413 
414  actions[i]->deleteLater();
415  }
416 }

535 QWidget *KSelectAction::createWidget(QWidget *parent)
536 {
537  Q_D(KSelectAction);
538  QMenu *menu = qobject_cast<QMenu *>(parent);
539  if (menu) { // If used in a menu want to return 0 and use only the text, not a widget
540  return 0;
541  }
542  ToolBarMode mode = toolBarMode();
543  QToolBar *toolBar = qobject_cast<QToolBar *>(parent);
544  if (!toolBar && mode != ComboBoxMode) { // we can return a combobox just fine.
545  return 0;
546  }
547  switch (mode) {
548  case MenuMode: {
549  QToolButton *button = new QToolButton(toolBar);
550  button->setToolTip(toolTip());
551  button->setWhatsThis(whatsThis());
552  button->setStatusTip(statusTip());
553  button->setAutoRaise(true);
554  button->setFocusPolicy(Qt::NoFocus);
555  button->setIconSize(toolBar->iconSize());
556  button->setToolButtonStyle(toolBar->toolButtonStyle());
557  QObject::connect(toolBar, &QToolBar::iconSizeChanged,
558  button, &QAbstractButton::setIconSize);
559  QObject::connect(toolBar, &QToolBar::toolButtonStyleChanged,
560  button, &QToolButton::setToolButtonStyle);
561  button->setDefaultAction(this);
562  QObject::connect(button, &QToolButton::triggered, toolBar, &QToolBar::actionTriggered);
563 
564  button->setPopupMode(toolButtonPopupMode());
565 
566  button->addActions(selectableActionGroup()->actions());
567 
568  d->m_buttons.append(button);
569  return button;
570  }
571 
572  case ComboBoxMode: {
573  QComboBox *comboBox = new QComboBox(parent);
574  comboBox->installEventFilter(this);
575 
576  if (d->m_maxComboViewCount != -1) {
577  comboBox->setMaxVisibleItems(d->m_maxComboViewCount);
578  }
579 
580  if (d->m_comboWidth > 0) {
581  comboBox->setMaximumWidth(d->m_comboWidth);
582  }
583 
584  comboBox->setEditable(isEditable());
585  comboBox->setToolTip(toolTip());
586  comboBox->setWhatsThis(whatsThis());
587  comboBox->setStatusTip(statusTip());
588 
589  foreach (QAction *action, selectableActionGroup()->actions()) {
590  comboBox->addAction(action);
591  }
592 
593  if (selectableActionGroup()->actions().isEmpty()) {
594  comboBox->setEnabled(false);
595  }
596 
597  connect(comboBox, SIGNAL(destroyed(QObject*)), SLOT(_k_comboBoxDeleted(QObject*)));
598  connect(comboBox, SIGNAL(currentIndexChanged(int)), SLOT(_k_comboBoxCurrentIndexChanged(int)));
599  d->m_comboBoxes.append(comboBox);
600 
601  return comboBox;
602  }
603  }
604 
605  return 0L;
606 }

637 // KSelectAction::eventFilter() is called before action->setChecked()
638 // invokes the signal to update QActionGroup so KSelectAction::currentItem()
639 // returns an old value. There are 3 possibilities, where n actions will
640 // report QAction::isChecked() where n is:
641 //
642 // 0: the checked action was unchecked
643 // 1: the checked action did not change
644 // 2: another action was checked but QActionGroup has not been invoked yet
645 // to uncheck the one that was checked before
646 //
647 // TODO: we might want to cache this since QEvent::ActionChanged is fired
648 // often.
649 static int TrueCurrentItem(KSelectAction *sa)
650 {
651  QAction *curAction = sa->currentAction();
652  //qDebug () << "\tTrueCurrentItem(" << sa << ") curAction=" << curAction;
653 
654  foreach (QAction *action, sa->actions()) {
655  if (action->isChecked()) {
656  //qDebug () << "\t\taction " << action << " (text=" << action->text () << ") isChecked";
657 
658  // 2 actions checked case?
659  if (action != curAction) {
660  //qDebug () << "\t\t\tmust be newly selected one";
661  return sa->actions().indexOf(action);
662  }
663  }
664  }
665 
666  //qDebug () << "\t\tcurrent action still selected? " << (curAction && curAction->isChecked ());
667  // 1 or 0 actions checked case (in that order)?
668  return (curAction && curAction->isChecked()) ? sa->actions().indexOf(curAction) : -1;
669 }

671 bool KSelectAction::eventFilter(QObject *watched, QEvent *event)
672 {
673  QComboBox *comboBox = qobject_cast <QComboBox *> (watched);
674  if (!comboBox) {
675  return false/*propagate event*/;
676  }
677 
678  // If focus is lost, replace any edited text with the currently selected
679  // item.
680  if (event->type() == QEvent::FocusOut) {
681  QFocusEvent *const e = static_cast <QFocusEvent *>(event);
682  //qDebug () << "KSelectAction::eventFilter(FocusOut)"
683  // << " comboBox: ptr=" << comboBox
684  // << " reason=" << e->reason ()
685  // << endl;
686 
687  if (e->reason() != Qt::ActiveWindowFocusReason/*switch window*/ &&
688  e->reason() != Qt::PopupFocusReason/*menu*/ &&
689  e->reason() != Qt::OtherFocusReason/*inconsistently reproduceable actions...*/) {
