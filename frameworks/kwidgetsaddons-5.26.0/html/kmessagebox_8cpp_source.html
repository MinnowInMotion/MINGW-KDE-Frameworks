88 static QIcon themedMessageBoxIcon(QMessageBox::Icon icon)
89 {
90  QString icon_name;
91 
92  switch (icon) {
93  case QMessageBox::NoIcon:
94  return QIcon();
95  break;
96  case QMessageBox::Information:
97  icon_name = QStringLiteral("dialog-information");
98  break;
99  case QMessageBox::Warning:
100  icon_name = QStringLiteral("dialog-warning");
101  break;
102  case QMessageBox::Critical:
103  icon_name = QStringLiteral("dialog-error");
104  break;
105  default:
106  break;
107  }
108 
109  QIcon ret = QIcon::fromTheme(icon_name);
110 
111  if (ret.isNull()) {
112  return QMessageBox::standardIcon(icon);
113  } else {
114  return ret;
115  }
116 }

127 static void setMainWindow(QWidget *subWidget, WId mainWindowId)
128 {
129  // Set the WA_NativeWindow attribute to force the creation of the QWindow.
130  // Without this QWidget::windowHandle() returns 0.
131  subWidget->setAttribute(Qt::WA_NativeWindow, true);
132  QWindow *subWindow = subWidget->windowHandle();
133  Q_ASSERT(subWindow);
134 
135  QWindow *mainWindow = QWindow::fromWinId(mainWindowId);
136  // mainWindow is not the child of any object, so make sure it gets deleted at some point
137  QObject::connect(subWidget, &QObject::destroyed, mainWindow, &QObject::deleteLater);
138  subWindow->setTransientParent(mainWindow);
139 }

200 QDialogButtonBox::StandardButton createKMessageBox(QDialog *dialog, QDialogButtonBox *buttons, const QIcon &icon,
201  const QString &text, const QStringList &strlist,
202  const QString &ask, bool *checkboxReturn, Options options,
203  const QString &details, QMessageBox::Icon notifyType)
204 {
205  DialogButtonsHelper *buttonsHelper = new DialogButtonsHelper(dialog, buttons);
206 
207  QWidget *mainWidget = new QWidget(dialog);
208  QVBoxLayout *mainLayout = new QVBoxLayout(mainWidget);
209  const int spacingHint = mainWidget->style()->pixelMetric(QStyle::PM_DefaultLayoutSpacing);
210  mainLayout->setSpacing(spacingHint * 2); // provide extra spacing
211  mainLayout->setMargin(0);
212 
213  QHBoxLayout *hLayout = new QHBoxLayout();
214  hLayout->setMargin(0);
215  hLayout->setSpacing(-1); // use default spacing
216  mainLayout->addLayout(hLayout, 5);
217 
218  QLabel *iconLabel = new QLabel(mainWidget);
219 
220  if (!icon.isNull()) {
221  QStyleOption option;
222  option.initFrom(mainWidget);
223  iconLabel->setPixmap(icon.pixmap(mainWidget->style()->pixelMetric(QStyle::PM_MessageBoxIconSize, &option, mainWidget)));
224  }
225 
226  QVBoxLayout *iconLayout = new QVBoxLayout();
227  iconLayout->addStretch(1);
228  iconLayout->addWidget(iconLabel);
229  iconLayout->addStretch(5);
230 
231  hLayout->addLayout(iconLayout, 0);
232  hLayout->addSpacing(spacingHint);
233 
234  QLabel *messageLabel = new QLabel(text, mainWidget);
235  messageLabel->setOpenExternalLinks(options & KMessageBox::AllowLink);
236  Qt::TextInteractionFlags flags = Qt::TextSelectableByMouse | Qt::TextSelectableByKeyboard;
237  if (options & KMessageBox::AllowLink) {
238  flags |= Qt::LinksAccessibleByMouse | Qt::LinksAccessibleByKeyboard;
239  }
240  messageLabel->setTextInteractionFlags(flags);
241 
242  QRect desktop = QApplication::desktop()->screenGeometry(dialog);
243  bool usingSqueezedTextLabel = false;
244  if (messageLabel->sizeHint().width() > desktop.width() * 0.5) {
245  // enable automatic wrapping of messages which are longer than 50% of screen width
246  messageLabel->setWordWrap(true);
247  // use a squeezed label if text is still too wide
248  usingSqueezedTextLabel = messageLabel->sizeHint().width() > desktop.width() * 0.85;
249  if (usingSqueezedTextLabel) {
250  delete messageLabel;
251  messageLabel = new KSqueezedTextLabel(text, mainWidget);
252  messageLabel->setOpenExternalLinks(options & KMessageBox::AllowLink);
253  messageLabel->setTextInteractionFlags(flags);
254  }
255  }
256 
257  QPalette messagePal(messageLabel->palette());
258  messagePal.setColor(QPalette::Window, Qt::transparent);
259  messageLabel->setPalette(messagePal);
260 
261  bool usingScrollArea = desktop.height() / 3 < messageLabel->sizeHint().height();
262  if (usingScrollArea) {
263  QScrollArea *messageScrollArea = new QScrollArea(mainWidget);
264  messageScrollArea->setWidget(messageLabel);
265  messageScrollArea->setFrameShape(QFrame::NoFrame);
266  messageScrollArea->setWidgetResizable(true);
267  QPalette scrollPal(messageScrollArea->palette());
268  scrollPal.setColor(QPalette::Window, Qt::transparent);
269  messageScrollArea->viewport()->setPalette(scrollPal);
270  hLayout->addWidget(messageScrollArea, 5);
271  } else {
272  hLayout->addWidget(messageLabel, 5);
273  }
274 
275  const bool usingListWidget = !strlist.isEmpty();
276  if (usingListWidget) {
277  // enable automatic wrapping since the listwidget has already a good initial width
278  messageLabel->setWordWrap(true);
279  QListWidget *listWidget = new QListWidget(mainWidget);
280  listWidget->addItems(strlist);
281 
282  QStyleOptionViewItem styleOption;
283  styleOption.initFrom(listWidget);
284  QFontMetrics fm(styleOption.font);
285  int w = listWidget->width();
286  Q_FOREACH (const QString &str, strlist) {
287  w = qMax(w, fm.width(str));
288  }
289  const int borderWidth = listWidget->width() - listWidget->viewport()->width() + listWidget->verticalScrollBar()->height();
290  w += borderWidth;
291  if (w > desktop.width() * 0.85) { // limit listWidget size to 85% of screen width
292  w = qRound(desktop.width() * 0.85);
293  }
294  listWidget->setMinimumWidth(w);
295 
296  mainLayout->addWidget(listWidget, usingScrollArea ? 10 : 50);
297  listWidget->setSelectionMode(QListWidget::NoSelection);
298  messageLabel->setSizePolicy(QSizePolicy::Preferred, QSizePolicy::Minimum);
299  } else if (!usingScrollArea) {
300  mainLayout->addStretch(15);
301  }
302 
303  QPointer<QCheckBox> checkbox = 0;
304  if (!ask.isEmpty()) {
305  checkbox = new QCheckBox(ask, mainWidget);
306  mainLayout->addWidget(checkbox);
307  if (checkboxReturn) {
308  checkbox->setChecked(*checkboxReturn);
309  }
310  }
311 
312  QVBoxLayout *topLayout = new QVBoxLayout;
313  dialog->setLayout(topLayout);
314  topLayout->addWidget(mainWidget);
315 
316  if (!details.isEmpty()) {
317  QGroupBox *detailsGroup = new QGroupBox(QApplication::translate("KMessageBox", "Details"));
318  QVBoxLayout *detailsLayout = new QVBoxLayout(detailsGroup);
319  if (details.length() < 512) {
320  QLabel *detailsLabel = new QLabel(details);
321  detailsLabel->setOpenExternalLinks(options & KMessageBox::AllowLink);
322  Qt::TextInteractionFlags flags = Qt::TextSelectableByMouse | Qt::TextSelectableByKeyboard;
323  if (options & KMessageBox::AllowLink) {
324  flags |= Qt::LinksAccessibleByMouse | Qt::LinksAccessibleByKeyboard;
325  };
326  detailsLabel->setTextInteractionFlags(flags);
327  detailsLabel->setWordWrap(true);
328  detailsLayout->addWidget(detailsLabel, 50);
329  } else {
330  QTextBrowser *detailTextEdit = new QTextBrowser();
331  detailTextEdit->setText(details);
332  detailTextEdit->setReadOnly(true);
333  detailTextEdit->setMinimumHeight(detailTextEdit->fontMetrics().lineSpacing() * 11);
334  detailsLayout->addWidget(detailTextEdit, 50);
335  }
336  if (!usingListWidget) {
337  mainLayout->setStretchFactor(hLayout, 10);
338  }
339  topLayout->addWidget(detailsGroup);
340  buttonsHelper->setDetailsWidget(detailsGroup);
341  detailsGroup->hide();
342  }
343 
344  buttons->setParent(dialog);
345  topLayout->addWidget(buttons);
346 
347  if (!usingListWidget && !usingScrollArea && !usingSqueezedTextLabel && details.isEmpty()) {
348  dialog->setFixedSize(dialog->sizeHint() + QSize(10, 10));
349  } else if (!details.isEmpty() && dialog->minimumHeight() < iconLabel->sizeHint().height() * 2) { //strange bug...
350  if (!usingScrollArea) {
351  dialog->setMinimumSize(300, qMax(150, qMax(iconLabel->sizeHint().height(), messageLabel->sizeHint().height())));
352  } else {
353  dialog->setMinimumSize(300, qMax(150, iconLabel->sizeHint().height()));
354  }
355  }
356 
357  if ((options & KMessageBox::Dangerous)) {
358  QPushButton *cancelButton = buttons->button(QDialogButtonBox::Cancel);
359  QPushButton *noButton = buttons->button(QDialogButtonBox::No);
360 
361  if (cancelButton && cancelButton->isEnabled()) {
362  cancelButton->setDefault(true);
363  cancelButton->setFocus();
364  } else if (noButton && noButton->isEnabled()) {
365  noButton->setDefault(true);
366  noButton->setFocus();
367  }
368  }
369 
370 #ifndef Q_OS_WIN // FIXME problems with KNotify on Windows
371  if ((options & KMessageBox::Notify)) {
372  QString message = text;
373  if (!strlist.isEmpty()) {
374  message += QLatin1Char('\n') + strlist.join(QLatin1Char('\n'));
375  }
376  notifyInterface()->sendNotification(notifyType, message, dialog->topLevelWidget());
377  }
378 #endif
379 
380  if (KMessageBox_exec_hook) {
381  return KMessageBox_exec_hook(dialog);
382  }
383 
384  if ((options & KMessageBox::NoExec)) {
385  return QDialogButtonBox::NoButton; // We have to return something.
386  }
387 
388  // We use a QPointer because the dialog may get deleted
389  // during exec() if the parent of the dialog gets deleted.
390  // In that case the QPointer will reset to 0.
391  QPointer<QDialog> guardedDialog = dialog;
392 
393  const QDialogButtonBox::StandardButton result = QDialogButtonBox::StandardButton(guardedDialog->exec());
394  if (checkbox && checkboxReturn) {
395  *checkboxReturn = checkbox->isChecked();
396  }
397 
398  delete guardedDialog;
399  return result;
400 }

462 static ButtonCode questionYesNoListInternal(QDialog *dialog, const QString &text,
463  const QStringList &strlist,
464  const QString &caption,
465  const KGuiItem &buttonYes_,
466  const KGuiItem &buttonNo_,
467  const QString &dontAskAgainName,
468  Options options)

470  ButtonCode res;
471  if (!shouldBeShownYesNo(dontAskAgainName, res)) {
472  delete dialog;
473  return res;
474  }

477  I18N_FILTER_BUTTON_NO(buttonNo_, buttonNo)
478  I18N_POST_BUTTON_FILTER

480  dialog->setWindowTitle(caption.isEmpty() ? QApplication::translate("KMessageBox", "Question") : caption);
481  dialog->setObjectName(QStringLiteral("questionYesNo"));

483  QDialogButtonBox *buttonBox = new QDialogButtonBox(dialog);
484  buttonBox->setStandardButtons(QDialogButtonBox::Yes | QDialogButtonBox::No);

486  KGuiItem::assign(buttonBox->button(QDialogButtonBox::No), buttonNo);
487

489 
490  bool checkboxResult = false;
491  const int result = createKMessageBox(dialog, buttonBox, QMessageBox::Information, text, strlist,
492  dontAskAgainName.isEmpty() ? QString() : QApplication::translate("KMessageBox", "Do not ask again"),
493  &checkboxResult, options);
494  res = (result == QDialogButtonBox::Yes ? Yes : No);
495 
496  if (checkboxResult) {
497  saveDontShowAgainYesNo(dontAskAgainName, res);
498  }
