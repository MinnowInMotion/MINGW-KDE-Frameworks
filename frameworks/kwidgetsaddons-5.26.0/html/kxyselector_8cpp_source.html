118 void KXYSelector::Private::setValues(int _xPos, int _yPos)
119 {
120  int w = q->style()->pixelMetric(QStyle::PM_DefaultFrameWidth);
121 
122  xPos = _xPos;
123  yPos = _yPos;
124 
125  if (xPos > maxX) {
126  xPos = maxX;
127  } else if (xPos < minX) {
128  xPos = minX;
129  }
130 
131  if (yPos > maxY) {
132  yPos = maxY;
133  } else if (yPos < minY) {
134  yPos = minY;
135  }
136 
137  Q_ASSERT(maxX != minX);
138  int xp = w + (q->width() - 2 * w) * xPos / (maxX - minX);
139 
140  Q_ASSERT(maxY != minY);
141  int yp = q->height() - w - (q->height() - 2 * w) * yPos / (maxY - minY);
142 
143  q->setPosition(xp, yp);
144 }

205 void KXYSelector::valuesFromPosition(int x, int y, int &xVal, int &yVal) const
206 {
207  int w = style()->pixelMetric(QStyle::PM_DefaultFrameWidth);
208 
209  xVal = ((d->maxX - d->minX) * (x - w)) / (width() - 2 * w);
210  yVal = d->maxY - (((d->maxY - d->minY) * (y - w)) / (height() - 2 * w));
211 
212  if (xVal > d->maxX) {
213  xVal = d->maxX;
214  } else if (xVal < d->minX) {
215  xVal = d->minX;
216  }
217 
218  if (yVal > d->maxY) {
219  yVal = d->maxY;
220  } else if (yVal < d->minY) {
221  yVal = d->minY;
222  }
223 }

225 void KXYSelector::setPosition(int xp, int yp)
226 {
227  int w = style()->pixelMetric(QStyle::PM_DefaultFrameWidth);
228 
229  if (xp < w) {
230  xp = w;
231  } else if (xp > width() - w) {
232  xp = width() - w;
233  }
234 
235  if (yp < w) {
236  yp = w;
237  } else if (yp > height() - w) {
238  yp = height() - w;
239  }
240 
241  d->px = xp;
242  d->py = yp;
243 
244  update();
245 }
