38 void KDualActionPrivate::updateFromCurrentState()
39 {
40  KGuiItem &currentItem = item(isActive);
41  QAction *qq = static_cast<QAction *>(q);
42  qq->setIcon(currentItem.icon());
43  qq->setText(currentItem.text());
44  qq->setToolTip(currentItem.toolTip());
45 }
46 
47 void KDualActionPrivate::slotTriggered()
48 {
49  if (!autoToggle) {
50  return;
51  }
52  q->setActive(!isActive);
53  q->activeChangedByUser(isActive);
54 }
