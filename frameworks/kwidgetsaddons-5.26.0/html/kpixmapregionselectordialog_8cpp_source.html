41  void init()
42  {
43  //When the image is rotated we need to enforce the maximum width&height into the
44  //KPixmapRegionSelectorWidget; in order to avoid the dialog to get out of the screen
45  q->connect(pixmapSelectorWidget, SIGNAL(pixmapRotated()), q, SLOT(_k_adjustPixmapSize()));
46  }
47 
48  void _k_adjustPixmapSize()
49  {
50  if (pixmapSelectorWidget) {
51  //Set maximum size for picture
52  QDesktopWidget desktopWidget;
53  QRect screen = desktopWidget.availableGeometry();
54  pixmapSelectorWidget->setMaximumWidgetSize(
55  (int)(screen.width() * 4.0 / 5), (int)(screen.height() * 4.0 / 5));
56  }
57  }
58 };

60 KPixmapRegionSelectorDialog::KPixmapRegionSelectorDialog(QWidget *parent)

69  d->pixmapSelectorWidget = new KPixmapRegionSelectorWidget(this);

71  QDialogButtonBox *buttonBox = new QDialogButtonBox(this);
72  buttonBox->setStandardButtons(QDialogButtonBox::Ok | QDialogButtonBox::Cancel);
73  connect(buttonBox, &QDialogButtonBox::accepted, this, &QDialog::accept);
74  connect(buttonBox, &QDialogButtonBox::rejected, this, &QDialog::reject);
75 
76  boxLayout->addWidget(label);
77  boxLayout->addWidget(d->pixmapSelectorWidget);
