100 void KEditListWidgetPrivate::setEditor(QLineEdit *newLineEdit, QWidget *representationWidget)
101 {
102  if (editingWidget != lineEdit &&
103  editingWidget != representationWidget) {
104  delete editingWidget;
105  }
106  if (lineEdit != newLineEdit) {
107  delete lineEdit;
108  }
109  lineEdit = newLineEdit ? newLineEdit : new QLineEdit(q);
110  editingWidget = representationWidget ?
111  representationWidget : lineEdit;
112 
113  if (representationWidget) {
114  representationWidget->setParent(q);
115  }
116 
117  mainLayout->insertWidget(0, editingWidget);
118 
119  lineEdit->installEventFilter(q);
120 
121  q->connect(lineEdit, &QLineEdit::textChanged, q, &KEditListWidget::typedSomething);
122  q->connect(lineEdit, &QLineEdit::returnPressed, q, &KEditListWidget::addItem);
123 
124  // maybe supplied lineedit has some text already
125  q->typedSomething(lineEdit->text());
126 
127  // fix tab ordering
128  q->setTabOrder(editingWidget, listView);
129  QWidget *w = listView;
130  if (servNewButton) {
131  q->setTabOrder(w, servNewButton);
132  w = servNewButton;

292 void KEditListWidget::setButtons(Buttons buttons)
293 {
294  if (d->buttons == buttons) {
295  return;
296  }
297 
298  if ((buttons & Add) && !d->servNewButton) {
299  d->servNewButton = new QPushButton(QIcon::fromTheme(QStringLiteral("list-add")), tr("&Add"), this);
300  d->servNewButton->setEnabled(false);
301  d->servNewButton->show();
302  connect(d->servNewButton, &QAbstractButton::clicked, this, &KEditListWidget::addItem);
303 
304  d->btnsLayout->insertWidget(0, d->servNewButton);
305  } else if ((buttons & Add) == 0 && d->servNewButton) {
306  delete d->servNewButton;
307  d->servNewButton = 0;
308  }
309 
310  if ((buttons & Remove) && !d->servRemoveButton) {
311  d->servRemoveButton = new QPushButton(QIcon::fromTheme(QStringLiteral("list-remove")), tr("&Remove"), this);
312  d->servRemoveButton->setEnabled(false);
313  d->servRemoveButton->show();
314  connect(d->servRemoveButton, &QAbstractButton::clicked, this, &KEditListWidget::removeItem);
315 
316  d->btnsLayout->insertWidget(1, d->servRemoveButton);
317  } else if ((buttons & Remove) == 0 && d->servRemoveButton) {
318  delete d->servRemoveButton;
319  d->servRemoveButton = 0;
320  }
321 
322  if ((buttons & UpDown) && !d->servUpButton) {
323  d->servUpButton = new QPushButton(QIcon::fromTheme(QStringLiteral("arrow-up")), tr("Move &Up"), this);
324  d->servUpButton->setEnabled(false);
325  d->servUpButton->show();
326  connect(d->servUpButton, &QAbstractButton::clicked, this, &KEditListWidget::moveItemUp);
327 
328  d->servDownButton = new QPushButton(QIcon::fromTheme(QStringLiteral("arrow-down")), tr("Move &Down"), this);
329  d->servDownButton->setEnabled(false);
330  d->servDownButton->show();
331  connect(d->servDownButton, &QAbstractButton::clicked, this, &KEditListWidget::moveItemDown);
332 
333  d->btnsLayout->insertWidget(2, d->servUpButton);
334  d->btnsLayout->insertWidget(3, d->servDownButton);
335  } else if ((buttons & UpDown) == 0 && d->servUpButton) {
336  delete d->servUpButton; d->servUpButton = 0;
337  delete d->servDownButton; d->servDownButton = 0;
338  }

353 void KEditListWidget::typedSomething(const QString &text)
354 {
355  if (currentItem() >= 0) {
356  if (currentText() != d->lineEdit->text()) {
357  // IMHO changeItem() shouldn't do anything with the value
358  // of currentItem() ... like changing it or emitting signals ...
359  // but TT disagree with me on this one (it's been that way since ages ... grrr)
360  bool block = d->listView->signalsBlocked();
361  d->listView->blockSignals(true);
362  QModelIndex currentIndex = d->selectedIndex();
363  if (currentIndex.isValid()) {
364  d->model->setData(currentIndex, text);
365  }
366  d->listView->blockSignals(block);
367  emit changed();
368  }
369  }
370 
371  if (!d->servNewButton) {
372  return;
373  }
374 
375  if (!d->lineEdit->hasAcceptableInput()) {
376  d->servNewButton->setEnabled(false);
377  return;
378  }
379 
380  if (!d->checkAtEntering) {
381  d->servNewButton->setEnabled(!text.isEmpty());
382  } else {
383  if (text.isEmpty()) {
384  d->servNewButton->setEnabled(false);
385  } else {
386  QStringList list = d->model->stringList();
387  bool enable = !list.contains(text, Qt::CaseSensitive);
388  d->servNewButton->setEnabled(enable);
389  }
390  }
391 }

393 void KEditListWidget::moveItemUp()
394 {
395  if (!d->listView->isEnabled()) {
396  QApplication::beep();
397  return;
398  }
399 
400  QModelIndex index = d->selectedIndex();
401  if (index.isValid()) {
402  if (index.row() == 0) {
403  QApplication::beep();
404  return;
405  }
406 
407  QModelIndex aboveIndex = d->model->index(index.row() - 1, index.column());
408 
409  QString tmp = d->model->data(aboveIndex, Qt::DisplayRole).toString();
410  d->model->setData(aboveIndex, d->model->data(index, Qt::DisplayRole));
411  d->model->setData(index, tmp);
412 
413  d->listView->selectionModel()->select(index, QItemSelectionModel::Deselect);
414  d->listView->selectionModel()->select(aboveIndex, QItemSelectionModel::Select);
415  }
416 
417  emit changed();
418 }

447 void KEditListWidget::addItem()
448 {
449  // when checkAtEntering is true, the add-button is disabled, but this
450  // slot can still be called through Key_Return/Key_Enter. So we guard
451  // against this.
452  if (!d->servNewButton || !d->servNewButton->isEnabled()) {
453  return;
454  }
455 
456  QModelIndex currentIndex = d->selectedIndex();
457 
458  const QString &currentTextLE = d->lineEdit->text();
459  bool alreadyInList(false);
460  //if we didn't check for dupes at the inserting we have to do it now
461  if (!d->checkAtEntering) {
462  // first check current item instead of dumb iterating the entire list
463  if (currentIndex.isValid()) {
464  if (d->model->data(currentIndex, Qt::DisplayRole).toString() == currentTextLE) {
465  alreadyInList = true;
466  }
467  } else {
468  alreadyInList = d->model->stringList().contains(currentTextLE, Qt::CaseSensitive);
469  }
470  }
471  if (d->servNewButton) {
472  d->servNewButton->setEnabled(false);
473  }
474 
475  bool block = d->lineEdit->signalsBlocked();
476  d->lineEdit->blockSignals(true);
477  d->lineEdit->clear();
478  d->lineEdit->blockSignals(block);
479 
480  d->listView->selectionModel()->setCurrentIndex(currentIndex, QItemSelectionModel::Deselect);
481 
482  if (!alreadyInList) {
483  block = d->listView->signalsBlocked();
484 
485  if (currentIndex.isValid()) {
486  d->model->setData(currentIndex, currentTextLE);
487  } else {
488  QStringList lst;
489  lst << currentTextLE;
490  lst << d->model->stringList();
491  d->model->setStringList(lst);
492  }
493  emit changed();
494  emit added(currentTextLE); // TODO: pass the index too
495  }
496 
497  d->updateButtonState();
498 }

532 void KEditListWidget::enableMoveButtons(const QModelIndex &newIndex, const QModelIndex &)
533 {
534  int index = newIndex.row();
535 
536  // Update the lineEdit when we select a different line.
537  if (currentText() != d->lineEdit->text()) {
538  d->lineEdit->setText(currentText());
539  }
540 
541  bool moveEnabled = d->servUpButton && d->servDownButton;
542 
543  if (moveEnabled) {
544  if (d->model->rowCount() <= 1) {
545  d->servUpButton->setEnabled(false);
546  d->servDownButton->setEnabled(false);
547  } else if (index == (d->model->rowCount() - 1)) {
548  d->servUpButton->setEnabled(true);
549  d->servDownButton->setEnabled(false);
550  } else if (index == 0) {
551  d->servUpButton->setEnabled(false);
552  d->servDownButton->setEnabled(true);
553  } else {
554  d->servUpButton->setEnabled(true);
555  d->servDownButton->setEnabled(true);
556  }
557  }
558 
559  if (d->servRemoveButton) {
560  d->servRemoveButton->setEnabled(true);
561  }
562 }
