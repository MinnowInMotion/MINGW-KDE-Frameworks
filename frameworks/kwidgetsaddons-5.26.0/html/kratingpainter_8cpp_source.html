57 QPixmap KRatingPainter::Private::getPixmap(int size, QIcon::State state)
58 {
59  bool toGray = (state == QIcon::Off);
60  QPixmap p;
61 
62  if (!customPixmap.isNull()) {
63  p = customPixmap.scaled(QSize(size, size));
64  } else {
65  QIcon _icon(icon);
66  if (_icon.isNull()) {
67  if (state == QIcon::Off
68  && QIcon::hasThemeIcon(QStringLiteral("rating-unrated"))) {
69  _icon = QIcon::fromTheme(QStringLiteral("rating-unrated"));
70  // our theme provided a separate icon, no need to desaturate
71  toGray = false;
72  } else {
73  _icon = QIcon::fromTheme(QStringLiteral("rating"));
74  }
75  }
76  p = _icon.pixmap(size);
77  }
78 
79  if (toGray) {
80  QImage img = p.toImage().convertToFormat(QImage::Format_ARGB32);
81  _k_imageToGrayScale(img, 1.0);
82  return QPixmap::fromImage(img);
83  }
84  return p;
85 }

177 static void _k_imageToGrayScale(QImage &img, float value)
178 {
179  QRgb *data = (QRgb *) img.bits();
180  QRgb *end = data + img.width() * img.height();
181 
182  unsigned char gray;
183  unsigned char val = (unsigned char)(255.0 * value);
184  while (data != end) {
185  gray = qGray(*data);
186  *data = qRgba((val * gray + (255 - val) * qRed(*data)) >> 8,
187  (val * gray + (255 - val) * qGreen(*data)) >> 8,
188  (val * gray + (255 - val) * qBlue(*data)) >> 8,
189  qAlpha(*data));
190  ++data;
191  }
192 }
193 
194 static void _k_imageToSemiTransparent(QImage &img)
195 {
196  QRgb *data = (QRgb *) img.bits();
197  QRgb *end = data + img.width() * img.height();
198 
199  while (data != end) {
200  *data = qRgba(qRed(*data),
201  qGreen(*data),
202  qBlue(*data),
203  qAlpha(*data) >> 1);
204  ++data;
205  }
206 }

208 void KRatingPainter::paint(QPainter *painter, const QRect &rect, int rating, int hoverRating) const
209 {
210  rating = qMin(rating, d->maxRating);
211  hoverRating = qMin(hoverRating, d->maxRating);
212 
213  int numUsedStars = d->bHalfSteps ? d->maxRating / 2 : d->maxRating;
214 
215  if (hoverRating >= 0 && hoverRating < rating) {
216  int tmp = hoverRating;
217  hoverRating = rating;
218  rating = tmp;
219  }
220 
221  int usedSpacing = d->spacing;
222 
223  // get the rating pixmaps
224  int maxHSizeOnePix = (rect.width() - (numUsedStars - 1) * usedSpacing) / numUsedStars;
225  QPixmap ratingPix = d->getPixmap(qMin(rect.height(), maxHSizeOnePix), QIcon::On);
226  QSize ratingPixSize = ratingPix.size() / ratingPix.devicePixelRatio();
227 
228  QPixmap disabledRatingPix = d->getPixmap(qMin(rect.height(), maxHSizeOnePix), QIcon::Off);
229  QImage disabledRatingImage = disabledRatingPix.toImage().convertToFormat(QImage::Format_ARGB32);
230  QPixmap hoverPix;
231 
232  // if we are disabled we become gray and more transparent
233  if (!d->isEnabled) {
234  ratingPix = disabledRatingPix;
235 
236  _k_imageToSemiTransparent(disabledRatingImage);
237  disabledRatingPix = QPixmap::fromImage(disabledRatingImage);
238  }
239 
240  bool half = d->bHalfSteps && rating % 2;
241  int numRatingStars = d->bHalfSteps ? rating / 2 : rating;
242 
243  int numHoverStars = 0;
244  bool halfHover = false;
245  if (hoverRating >= 0 && rating != hoverRating && d->isEnabled) {
246  numHoverStars = d->bHalfSteps ? hoverRating / 2 : hoverRating;
247  halfHover = d->bHalfSteps && hoverRating % 2;
248 
249  disabledRatingImage = ratingPix.toImage().convertToFormat(QImage::Format_ARGB32);
250  _k_imageToGrayScale(disabledRatingImage, 0.5);
251 
252  hoverPix = QPixmap::fromImage(disabledRatingImage);
253  }
254 
255  if (d->alignment & Qt::AlignJustify && numUsedStars > 1) {
256  int w = rect.width();
257  w -= numUsedStars * ratingPixSize.width();
258  usedSpacing = w / (numUsedStars - 1);
259  }
260 
261  int ratingAreaWidth = ratingPixSize.width() * numUsedStars + usedSpacing * (numUsedStars - 1);
262 
263 
264  int i = 0;
265  int x = rect.x();
266  if (d->alignment & Qt::AlignRight) {
267  x += (rect.width() - ratingAreaWidth);
268  } else if (d->alignment & Qt::AlignHCenter) {
269  x += (rect.width() - ratingAreaWidth) / 2;
270  }
271 
272  int xInc = ratingPixSize.width() + usedSpacing;
273  if (d->direction == Qt::RightToLeft) {
274  x = rect.width() - ratingPixSize.width() - x;
275  xInc = -xInc;
276  }
277 
278  int y = rect.y();
279  if (d->alignment & Qt::AlignVCenter) {
280  y += (rect.height() / 2 - ratingPixSize.height() / 2);
281  } else if (d->alignment & Qt::AlignBottom) {
282  y += (rect.height() - ratingPixSize.height());
283  }
284  for (; i < numRatingStars; ++i) {
285  painter->drawPixmap(x, y, ratingPix);
286  x += xInc;
287  }
288  if (half) {
289  painter->drawPixmap(x, y, ratingPixSize.width() / 2, ratingPixSize.height(),
290  d->direction == Qt::RightToLeft ? (numHoverStars > 0 ? hoverPix : disabledRatingPix) : ratingPix,
291  0, 0, ratingPix.width() / 2, ratingPix.height()); //source sizes are deliberately not device independent
292  painter->drawPixmap(x + ratingPixSize.width() / 2, y, ratingPixSize.width() / 2, ratingPixSize.height(),
293  d->direction == Qt::RightToLeft ? ratingPix : (numHoverStars > 0 ? hoverPix : disabledRatingPix),
294  ratingPix.width() / 2, 0, ratingPix.width() / 2, ratingPix.height());
295  x += xInc;
296  ++i;
297  }
298  for (; i < numHoverStars; ++i) {
299  painter->drawPixmap(x, y, hoverPix);
300  x += xInc;
301  }
302  if (halfHover) {
303  painter->drawPixmap(x, y, ratingPixSize.width() / 2, ratingPixSize.height(),
304  d->direction == Qt::RightToLeft ? disabledRatingPix : hoverPix,
305  0, 0, ratingPix.width() / 2, ratingPix.height());
306  painter->drawPixmap(x + ratingPixSize.width() / 2, y, ratingPixSize.width() / 2, ratingPixSize.height(),
307  d->direction == Qt::RightToLeft ? hoverPix : disabledRatingPix,
308  ratingPix.width() / 2, 0, ratingPix.width() / 2, ratingPix.height());
309  x += xInc;
310  ++i;
311  }
312  for (; i < numUsedStars; ++i) {
313  painter->drawPixmap(x, y, disabledRatingPix);
314  x += xInc;
315  }
316 }

318 int KRatingPainter::ratingFromPosition(const QRect &rect, const QPoint &pos) const
319 {
320  int usedSpacing = d->spacing;
321  int numUsedStars = d->bHalfSteps ? d->maxRating / 2 : d->maxRating;
322  int maxHSizeOnePix = (rect.width() - (numUsedStars - 1) * usedSpacing) / numUsedStars;
323  QPixmap ratingPix = d->getPixmap(qMin(rect.height(), maxHSizeOnePix));
324  QSize ratingPixSize = ratingPix.size() / ratingPix.devicePixelRatio();
325 
326  int ratingAreaWidth = ratingPixSize.width() * numUsedStars + usedSpacing * (numUsedStars - 1);
327 
328  QRect usedRect(rect);
329  if (d->alignment & Qt::AlignRight) {
330  usedRect.setLeft(rect.right() - ratingAreaWidth);
331  } else if (d->alignment & Qt::AlignHCenter) {
332  int x = (rect.width() - ratingAreaWidth) / 2;
333  usedRect.setLeft(rect.left() + x);
334  usedRect.setRight(rect.right() - x);
335  } else { // d->alignment & Qt::AlignLeft
336  usedRect.setRight(rect.left() + ratingAreaWidth - 1);
337  }
338 
339  if (d->alignment & Qt::AlignBottom) {
340  usedRect.setTop(rect.bottom() - ratingPixSize.height() + 1);
341  } else if (d->alignment & Qt::AlignVCenter) {
342  int x = (rect.height() - ratingPixSize.height()) / 2;
343  usedRect.setTop(rect.top() + x);
344  usedRect.setBottom(rect.bottom() - x);
345  } else { // d->alignment & Qt::AlignTop
346  usedRect.setBottom(rect.top() + ratingPixSize.height() - 1);
347  }
348 
349  if (usedRect.contains(pos)) {
350  int x = 0;
351  if (d->direction == Qt::RightToLeft) {
352  x = usedRect.right() - pos.x();
353  } else {
354  x = pos.x() - usedRect.left();
355  }
356 
357  double one = (double)usedRect.width() / (double)d->maxRating;
358 
359 // qDebug() << "rating:" << ( int )( ( double )x/one + 0.5 );
360 
361  return (int)((double)x / one + 0.5);
362  } else {
363  return -1;
364  }
365 }
