409 void
410 KRuler::setRulerMetricStyle(KRuler::MetricStyle style)
411 {
412  switch (style) {
413  default: /* fall through */
414  case Custom:
415  return;
416  case Pixel:
417  setLittleMarkDistance(1);
418  setMediumMarkDistance(5);
419  setBigMarkDistance(10);
420 
421  setShowTinyMarks(false);
422  setShowLittleMarks(true);
423  setShowMediumMarks(true);
424  setShowBigMarks(true);
425  setShowEndMarks(true);
426 
427  update(contentsRect());
428  setPixelPerMark(10.0);
429 
430  break;
431  case Inch:
432  setTinyMarkDistance(1);
433  setLittleMarkDistance(2);
434  setMediumMarkDistance(4);
435  setBigMarkDistance(8);
436 
437  setShowTinyMarks(true);
438  setShowLittleMarks(true);
439  setShowMediumMarks(true);
440  setShowBigMarks(true);
441  setShowEndMarks(true);
442 
443  update(contentsRect());
444  setPixelPerMark(9.0);
445 
446  break;
447  case Millimetres: /* fall through */
448  case Centimetres: /* fall through */
449  case Metres:
450  setLittleMarkDistance(1);
451  setMediumMarkDistance(5);
452  setBigMarkDistance(10);
453 
454  setShowTinyMarks(false);
455  setShowLittleMarks(true);
456  setShowMediumMarks(true);
457  setShowBigMarks(true);
458  setShowEndMarks(true);
459 
460  update(contentsRect());
461  setPixelPerMark(3.0);
462  }
463  switch (style) {
464  case Pixel:
465  setEndLabel(QStringLiteral("pixel"));
466  break;
467  case Inch:
468  setEndLabel(QStringLiteral("inch"));
469  break;
470  case Millimetres:
471  setEndLabel(QStringLiteral("mm"));
472  break;
473  case Centimetres:
474  setEndLabel(QStringLiteral("cm"));
475  break;
476  case Metres:
477  setEndLabel(QStringLiteral("m"));
478  default: /* never reached, see above switch */
479  /* empty command */;
480  }
481  // if the style changes one of the values,
482  // update would have been called inside the methods
483  // -> no update() call needed here !
484 }

579 void
580 KRuler::slotNewValue(int _value)
581 {
582  int oldvalue = value();
583  if (oldvalue == _value) {
584  return;
585  }
586  // setValue(_value);
587  setValue(_value);
588  if (value() == oldvalue) {
589  return;
590  }
591  // get the rectangular of the old and the new ruler pointer
592  // and repaint only him
593  if (d->dir == Qt::Horizontal) {
594  QRect oldrec(-5 + oldvalue, 10, 11, 6);
595  QRect newrec(-5 + _value, 10, 11, 6);
596  repaint(oldrec.united(newrec));
597  } else {
598  QRect oldrec(10, -5 + oldvalue, 6, 11);
599  QRect newrec(10, -5 + _value, 6, 11);
600  repaint(oldrec.united(newrec));
601  }
602 }

629 void
630 KRuler::paintEvent(QPaintEvent * /*e*/)
631 {
632  // debug ("KRuler::drawContents, %s",(horizontal==dir)?"horizontal":"vertical");
633 
634  QStylePainter p(this);
635 #ifdef PROFILING
636  QTime time;
637  time.start();
638  for (int profile = 0; profile < 10; profile++) {
639 #endif
640 
641  int value = this->value(),
642  minval = minimum(),
643  maxval;
644  if (d->dir == Qt::Horizontal) {
645  maxval = maximum()
646  + d->offset
647  - (d->lengthFix ? (height() - d->endOffset_length) : d->endOffset_length);
648  } else {
649  maxval = maximum()
650  + d->offset
651  - (d->lengthFix ? (width() - d->endOffset_length) : d->endOffset_length);
652  }
653  //ioffsetval = value-offset;
654  // pixelpm = (int)ppm;
655  // left = clip.left(),
656  // right = clip.right();
657  double f, fend,
658  offsetmin = (double)(minval - d->offset),
659  offsetmax = (double)(maxval - d->offset),
660  fontOffset = (((double)minval) > offsetmin) ? (double)minval : offsetmin;
661 
662  // draw labels
663  QFont font = p.font();
664  font.setPointSize(LABEL_SIZE);
665  p.setFont(font);
666  // draw littlemarklabel
667 
668  // draw mediummarklabel
669 
670  // draw bigmarklabel
671 
672  // draw endlabel
673  if (d->showEndL) {
674  if (d->dir == Qt::Horizontal) {
675  p.translate(fontOffset, 0);
676  p.drawText(END_LABEL_X, END_LABEL_Y, d->endlabel);
677  } else { // rotate text +pi/2 and move down a bit
678  //QFontMetrics fm(font);
679 #ifdef KRULER_ROTATE_TEST
680  p.rotate(-90.0 + rotate);
681  p.translate(-8.0 - fontOffset - d->fontWidth + xtrans,
682  ytrans);
683 #else
684  p.rotate(-90.0);
685  p.translate(-8.0 - fontOffset - d->fontWidth, 0.0);
686 #endif
687  p.drawText(END_LABEL_X, END_LABEL_Y, d->endlabel);
688  }
689  p.resetMatrix();
690  }
691 
692  // draw the tiny marks
693  if (d->showtm) {
694  fend = d->ppm * d->tmDist;
695  for (f = offsetmin; f < offsetmax; f += fend) {
696  if (d->dir == Qt::Horizontal) {
697  p.drawLine((int)f, BASE_MARK_X1, (int)f, BASE_MARK_X2);
698  } else {
699  p.drawLine(BASE_MARK_X1, (int)f, BASE_MARK_X2, (int)f);
700  }
701  }
702  }
703  if (d->showlm) {
704  // draw the little marks
705  fend = d->ppm * d->lmDist;
706  for (f = offsetmin; f < offsetmax; f += fend) {
707  if (d->dir == Qt::Horizontal) {
708  p.drawLine((int)f, LITTLE_MARK_X1, (int)f, LITTLE_MARK_X2);
709  } else {
710  p.drawLine(LITTLE_MARK_X1, (int)f, LITTLE_MARK_X2, (int)f);
711  }
712  }
713  }
714  if (d->showmm) {
715  // draw medium marks
716  fend = d->ppm * d->mmDist;
717  for (f = offsetmin; f < offsetmax; f += fend) {
718  if (d->dir == Qt::Horizontal) {
719  p.drawLine((int)f, MIDDLE_MARK_X1, (int)f, MIDDLE_MARK_X2);
720  } else {
721  p.drawLine(MIDDLE_MARK_X1, (int)f, MIDDLE_MARK_X2, (int)f);
722  }
723  }
724  }
725  if (d->showbm) {
726  // draw big marks
727  fend = d->ppm * d->bmDist;
728  for (f = offsetmin; f < offsetmax; f += fend) {
729  if (d->dir == Qt::Horizontal) {
730  p.drawLine((int)f, BIG_MARK_X1, (int)f, BIG_MARK_X2);
731  } else {
732  p.drawLine(BIG_MARK_X1, (int)f, BIG_MARK_X2, (int)f);
733  }
734  }
735  }
736  if (d->showem) {
737  // draw end marks
738  if (d->dir == Qt::Horizontal) {
739  p.drawLine(minval - d->offset, END_MARK_X1, minval - d->offset, END_MARK_X2);
740  p.drawLine(maxval - d->offset, END_MARK_X1, maxval - d->offset, END_MARK_X2);
741  } else {
742  p.drawLine(END_MARK_X1, minval - d->offset, END_MARK_X2, minval - d->offset);
743  p.drawLine(END_MARK_X1, maxval - d->offset, END_MARK_X2, maxval - d->offset);
744  }
745  }
746 
747  // draw pointer
748  if (d->showpointer) {
749  QPolygon pa(4);
750  if (d->dir == Qt::Horizontal) {
751  pa.setPoints(3, value - 5, 10, value + 5, 10, value/*+0*/, 15);
752  } else {
753  pa.setPoints(3, 10, value - 5, 10, value + 5, 15, value/*+0*/);
754  }
755  p.setBrush(p.background().color());
756  p.drawPolygon(pa);
757  }
758 
759 #ifdef PROFILING
760  }
761  int elapsed = time.elapsed();
