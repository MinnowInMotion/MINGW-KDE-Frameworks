241 void KCharSelectTablePrivate::_k_resizeCells()
242 {
243  KCharSelectItemModel *model = static_cast<KCharSelectItemModel*>(q->model());
244  if (!model) return;
245 
246  const int viewportWidth = q->viewport()->size().width();
247 
248  QFontMetrics fontMetrics(font);
249 
250  // Determine the max width of the displayed characters
251  // fontMetrics.maxWidth() doesn't help because of font fallbacks
252  // (testcase: Malayalam characters)
253  int maxCharWidth = 0;
254  const QVector<uint> chars = model->chars();
255  for (int i = 0; i < chars.size(); ++i) {
256  uint thisChar = chars.at(i);
257  if(s_data()->isPrint(thisChar)) {
258  maxCharWidth = qMax(maxCharWidth, fontMetrics.boundingRect(QString::fromUcs4(&thisChar, 1)).width());
259  }
260  }
261  // Avoid too narrow cells
262  maxCharWidth = qMax(maxCharWidth, 2 * fontMetrics.xHeight());
263  maxCharWidth = qMax(maxCharWidth, fontMetrics.height());
264  // Add the necessary padding, trying to match the delegate
265  const int textMargin = q->style()->pixelMetric(QStyle::PM_FocusFrameHMargin, 0, q) + 1;
266  maxCharWidth += 2 * textMargin;
267 
268  const int columns = qMax(1, viewportWidth / maxCharWidth);
269  model->setColumnCount(columns);
270 
271  const uint oldChar = q->chr();
272 
273  const int new_w = viewportWidth / columns;
274  const int rows = model->rowCount();
275  q->setUpdatesEnabled(false);
276  QHeaderView *hHeader = q->horizontalHeader();
277  const int spaceLeft = viewportWidth - new_w * columns;
278  for (int i = 0; i <= columns; ++i) {
279  if (i < spaceLeft) {
280  hHeader->resizeSection(i, new_w + 1);
281  } else {
282  hHeader->resizeSection(i, new_w);
283  }
284  }
285 
286  QHeaderView *vHeader = q->verticalHeader();
287 #ifdef Q_OS_WIN
288  int new_h = fontMetrics.lineSpacing() + 1;
289 #else
290  int new_h = fontMetrics.xHeight() * 3;
291 #endif
292  const int fontHeight = fontMetrics.height();
293  if (new_h < 5 || new_h < 4 + fontHeight) {
294  new_h = qMax(5, 4 + fontHeight);
295  }
296  for (int i = 0; i < rows; ++i) {
297  vHeader->resizeSection(i, new_h);
298  }
299 
300  q->setUpdatesEnabled(true);
301  q->setChar(oldChar);
302 }

607 void KCharSelect::setCurrentCodePoint(uint c)
608 {
609  if (!d->allPlanesEnabled && QChar::requiresSurrogates(c)) {
610  qCritical("You must setAllPlanesEnabled(true) to use non-BMP characters");
611  c = QChar::ReplacementCharacter;
612  }
613  if (c > QChar::LastValidCodePoint) {
614  qWarning("Code point outside Unicode range");
615  c = QChar::LastValidCodePoint;
616  }
617  bool oldHistoryEnabled = d->historyEnabled;
618  d->historyEnabled = false;
619  int block = s_data()->blockIndex(c);
620  int section = s_data()->sectionIndex(block);
621  d->sectionCombo->setCurrentIndex(section);
622  int index = d->blockCombo->findData(block);
623  if (index != -1) {
624  d->blockCombo->setCurrentIndex(index);
625  }
626  d->historyEnabled = oldHistoryEnabled;
627  d->charTable->setChar(c);
628 }

630 void KCharSelect::KCharSelectPrivate::historyAdd(uint c, bool fromSearch, const QString &searchString)
631 {
632  //qDebug() << "about to add char" << c << "fromSearch" << fromSearch << "searchString" << searchString;
633 
634  if (!historyEnabled) {
635  return;
636  }
637 
638  if (!history.isEmpty() && c == history.last().c) {
639  //avoid duplicates
640  return;
641  }
642 
643  //behave like a web browser, i.e. if user goes back from B to A then clicks C, B is forgotten
644  while (!history.isEmpty() && inHistory != history.count() - 1) {
645  history.removeLast();
646  }
647 
648  while (history.size() >= MaxHistoryItems) {
649  history.removeFirst();
650  }
651 
652  HistoryItem item;
653  item.c = c;
654  item.fromSearch = fromSearch;
655  item.searchString = searchString;
656  history.append(item);
657 
658  inHistory = history.count() - 1;
659  updateBackForwardButtons();
660 }

901 QString KCharSelect::KCharSelectPrivate::createLinks(QString s)
902 {
903  QRegExp rx(QStringLiteral("\\b([\\dABCDEF]{4,5})\\b"));
904 
905  QStringList chars;
906  int pos = 0;
907 
908  while ((pos = rx.indexIn(s, pos)) != -1) {
909  chars << rx.cap(1);
910  pos += rx.matchedLength();
911  }
912 
913  QSet<QString> chars2 = QSet<QString>::fromList(chars);
914  foreach (const QString &c, chars2) {
915  int unicode = c.toInt(0, 16);
916  if (!allPlanesEnabled && QChar::requiresSurrogates(unicode)) {
917  continue;
918  }
919  QString link = QStringLiteral("<a href=\"") + c + QStringLiteral("\">");
920  if (s_data()->isPrint(unicode)) {
921  link += QStringLiteral("&#8206;&#") + QString::number(unicode) + QStringLiteral(";&nbsp;");
922  }
923  link += QStringLiteral("U+") + c + QLatin1Char(' ');
924  link += s_data()->name(unicode).toHtmlEscaped() + QStringLiteral("</a>");
925  s.replace(c, link);
926  }
927  return s;
928 }
