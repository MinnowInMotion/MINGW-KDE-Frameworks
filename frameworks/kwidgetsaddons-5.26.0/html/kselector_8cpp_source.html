93 QRect KSelector::contentsRect() const
94 {
95  int w = indent() ? style()->pixelMetric(QStyle::PM_DefaultFrameWidth) : 0;
96  //TODO: is the height:width ratio of an indicator arrow always 2:1? hm.
97  int iw = (w < ARROWSIZE) ? ARROWSIZE : w;
98 
99  if (orientation() == Qt::Vertical) {
100  if (arrowDirection() == Qt::RightArrow) {
101  return QRect(w + ARROWSIZE, iw,
102  width() - w * 2 - ARROWSIZE,
103  height() - iw * 2);
104  } else {
105  return QRect(w, iw,
106  width() - w * 2 - ARROWSIZE,
107  height() - iw * 2);
108  }
109  } else { // Qt::Horizontal
110  if (arrowDirection() == Qt::UpArrow) {
111  return QRect(iw, w,
112  width() - 2 * iw,
113  height() - w * 2 - ARROWSIZE);
114  } else {
115  return QRect(iw, w + ARROWSIZE,
116  width() - 2 * iw,
117  height() - w * 2 - ARROWSIZE);
118  }
119  }
120 }

122 void KSelector::paintEvent(QPaintEvent *)
123 {
124  QPainter painter;
125  int w = style()->pixelMetric(QStyle::PM_DefaultFrameWidth);
126  int iw = (w < ARROWSIZE) ? ARROWSIZE : w;
127 
128  painter.begin(this);
129 
130  drawContents(&painter);
131 
132  QBrush brush;
133 
134  QPoint pos = calcArrowPos(value());
135  drawArrow(&painter, pos);
136 
137  if (indent()) {
138  QStyleOptionFrame opt;
139  opt.initFrom(this);
140  opt.state = QStyle::State_Sunken;
141  if (orientation() == Qt::Vertical) {
142  opt.rect.adjust(0, iw - w, -5, w - iw);
143  } else {
144  opt.rect.adjust(iw - w, 0, w - iw, -5);
145  }
146  QBrush oldBrush = painter.brush();
147  painter.setBrush(Qt::NoBrush);
148  style()->drawPrimitive(QStyle::PE_Frame, &opt, &painter, this);
149  painter.setBrush(oldBrush);
150  }
151 
152  painter.end();
153 }

180 void KSelector::moveArrow(const QPoint &pos)
181 {
182  int val;
183  int w = style()->pixelMetric(QStyle::PM_DefaultFrameWidth);
184  int iw = (w < ARROWSIZE) ? ARROWSIZE : w;
185 
186  if (orientation() == Qt::Vertical)
187  val = (maximum() - minimum()) * (height() - pos.y() - iw)
188  / (height() - iw * 2) + minimum();
189  else
190  val = (maximum() - minimum()) * (pos.x() - iw)
191  / (width() - iw * 2) + minimum();
192 
193  setValue(val);
194  update();
195 }

197 QPoint KSelector::calcArrowPos(int val)
198 {
199  QPoint p;
200  int w = style()->pixelMetric(QStyle::PM_DefaultFrameWidth);
201  int iw = (w < ARROWSIZE) ? ARROWSIZE : w;
202 
203  if (orientation() == Qt::Vertical) {
204  p.setY(height() - iw - 1 - (height() - 2 * iw - 1) * val / (maximum() - minimum()));
205 
206  if (d->arrowPE == QStyle::PE_IndicatorArrowRight) {
207  p.setX(0);
208  } else {
209  p.setX(width() - 5);
210  }
211  } else {
212  p.setX(iw + (width() - 2 * iw - 1) * val / (maximum() - minimum()));
213 
214  if (d->arrowPE == QStyle::PE_IndicatorArrowDown) {
215  p.setY(0);
216  } else {
217  p.setY(height() - 5);
218  }
219  }
220 
221  return p;
222 }

224 void KSelector::setArrowDirection(Qt::ArrowType direction)
225 {
226  switch (direction) {
227  case Qt::UpArrow:
228  if (orientation() == Qt::Horizontal) {
229  d->arrowPE = QStyle::PE_IndicatorArrowUp;
230  } else {
231  d->arrowPE = QStyle::PE_IndicatorArrowLeft;
232  }
233  break;
234  case Qt::DownArrow:
235  if (orientation() == Qt::Horizontal) {
236  d->arrowPE = QStyle::PE_IndicatorArrowDown;
237  } else {
238  d->arrowPE = QStyle::PE_IndicatorArrowRight;
239  }
240  break;
241  case Qt::LeftArrow:
242  if (orientation() == Qt::Vertical) {
243  d->arrowPE = QStyle::PE_IndicatorArrowLeft;
244  } else {
245  d->arrowPE = QStyle::PE_IndicatorArrowDown;
246  }
247  break;
248  case Qt::RightArrow:
249  if (orientation() == Qt::Vertical) {
250  d->arrowPE = QStyle::PE_IndicatorArrowRight;
251  } else {
252  d->arrowPE = QStyle::PE_IndicatorArrowUp;
253  }
254  break;
255 
256  case Qt::NoArrow:
257  break;
258  }
259 }

261 Qt::ArrowType KSelector::arrowDirection() const
262 {
263  switch (d->arrowPE) {
264  case QStyle::PE_IndicatorArrowUp:
265  return Qt::UpArrow;
266  break;
267  case QStyle::PE_IndicatorArrowDown:
268  return Qt::DownArrow;
269  break;
270  case QStyle::PE_IndicatorArrowRight:
271  return Qt::RightArrow;
272  break;
273  case QStyle::PE_IndicatorArrowLeft:
274  default:
275  return Qt::LeftArrow;
276  break;
277  }

283 void KSelector::drawArrow(QPainter *painter, const QPoint &pos)
284 {
285  painter->setPen(QPen());
286  painter->setBrush(QBrush(palette().color(QPalette::ButtonText)));
287 
288  QStyleOption o;
289 
290  if (orientation() == Qt::Vertical) {
291  o.rect = QRect(pos.x(), pos.y() - ARROWSIZE / 2,
292  ARROWSIZE, ARROWSIZE);
293  } else {
294  o.rect = QRect(pos.x() - ARROWSIZE / 2, pos.y(),
295  ARROWSIZE, ARROWSIZE);
296 
297  }
298  style()->drawPrimitive(d->arrowPE, &o, painter, this);
299 }

318 void KGradientSelector::drawContents(QPainter *painter)
319 {
320  d->gradient.setStart(contentsRect().topLeft());
321  if (orientation() == Qt::Vertical) {
322  d->gradient.setFinalStop(contentsRect().bottomLeft());
323  } else {
324  d->gradient.setFinalStop(contentsRect().topRight());
325  }
326  QBrush gradientBrush(d->gradient);
327 
328  if (!gradientBrush.isOpaque()) {
329  QPixmap chessboardPattern(16, 16);
330  QPainter patternPainter(&chessboardPattern);
331  patternPainter.fillRect(0, 0, 8, 8, Qt::black);
332  patternPainter.fillRect(8, 8, 8, 8, Qt::black);
333  patternPainter.fillRect(0, 8, 8, 8, Qt::white);
334  patternPainter.fillRect(8, 0, 8, 8, Qt::white);
335  patternPainter.end();
336  painter->fillRect(contentsRect(), QBrush(chessboardPattern));
337  }
338  painter->fillRect(contentsRect(), gradientBrush);
339 
340  if (orientation() == Qt::Vertical) {
341  int yPos = contentsRect().top() + painter->fontMetrics().ascent() + 2;
342  int xPos = contentsRect().left() + (contentsRect().width() -
343  painter->fontMetrics().width(d->text2)) / 2;
344  QPen pen(qGray(firstColor().rgb()) > 180 ? Qt::black : Qt::white);
345  painter->setPen(pen);
346  painter->drawText(xPos, yPos, d->text2);
347 
348  yPos = contentsRect().bottom() - painter->fontMetrics().descent() - 2;
349  xPos = contentsRect().left() + (contentsRect().width() -
350  painter->fontMetrics().width(d->text1)) / 2;
351  pen.setColor(qGray(secondColor().rgb()) > 180 ? Qt::black : Qt::white);
352  painter->setPen(pen);
353  painter->drawText(xPos, yPos, d->text1);
354  } else {
355  int yPos = contentsRect().bottom() - painter->fontMetrics().descent() - 2;
356 
357  QPen pen(qGray(firstColor().rgb()) > 180 ? Qt::black : Qt::white);
358  painter->setPen(pen);
359  painter->drawText(contentsRect().left() + 2, yPos, d->text1);
360 
361  pen.setColor(qGray(secondColor().rgb()) > 180 ? Qt::black : Qt::white);
362  painter->setPen(pen);
363  painter->drawText(contentsRect().right() -
364  painter->fontMetrics().width(d->text2) - 2, yPos, d->text2);
365  }
366 }
