56 void KToolBarLabelAction::setBuddy(QAction *buddy)
57 {
58  d->buddy = buddy;
59 
60  QList<QLabel *> labels;
61  Q_FOREACH (QWidget *widget, associatedWidgets())
62  if (QToolBar *toolBar = qobject_cast<QToolBar *>(widget))
63  if (QLabel *label = qobject_cast<QLabel *>(toolBar->widgetForAction(this))) {
64  labels.append(label);
65  }
66 
67  Q_FOREACH (QWidget *widget, buddy->associatedWidgets())
68  if (QToolBar *toolBar = qobject_cast<QToolBar *>(widget)) {
69  QWidget *newBuddy = toolBar->widgetForAction(buddy);
70  Q_FOREACH (QLabel *label, labels) {
71  label->setBuddy(newBuddy);
72  }
73  return;
74  }
75 }

94 bool KToolBarLabelAction::eventFilter(QObject *watched, QEvent *event)
95 {
96  if (d->label && d->buddy && event->type() == QEvent::PolishRequest && watched == d->label) {
97  Q_FOREACH (QWidget *widget, d->buddy->associatedWidgets()) {
98  if (QToolBar *toolBar = qobject_cast<QToolBar *>(widget)) {
99  QWidget *newBuddy = toolBar->widgetForAction(d->buddy);
100  d->label->setBuddy(newBuddy);
101  }
102  }
103  }
104 
105  return QWidgetAction::eventFilter(watched, event);
106 }

108 QWidget *KToolBarLabelAction::createWidget(QWidget *_parent)
109 {
110  QToolBar *parent = qobject_cast<QToolBar *>(_parent);
111  if (!parent) {
112  return QWidgetAction::createWidget(_parent);
113  }
114  if (!d->label) {

121  d->label->setBackgroundRole(QPalette::Button);
122  d->label->setAlignment((QApplication::isRightToLeft() ? Qt::AlignRight : Qt::AlignLeft) |
123  Qt::AlignVCenter);
124  d->label->adjustSize();
125  d->label->setText(text());
