139 void KPixmapRegionSelectorWidget::Private::updatePixmap()
140 {
141  Q_ASSERT(!m_originalPixmap.isNull());
142  if (m_originalPixmap.isNull()) {
143  m_label->setPixmap(m_originalPixmap);
144  return;
145  }
146  if (m_selectedRegion.width() > m_originalPixmap.width()) {
147  m_selectedRegion.setWidth(m_originalPixmap.width());
148  }
149  if (m_selectedRegion.height() > m_originalPixmap.height()) {
150  m_selectedRegion.setHeight(m_originalPixmap.height());
151  }
152 
153  QPainter painter;
154  if (m_linedPixmap.isNull()) {
155  m_linedPixmap = m_originalPixmap;
156  QPainter p(&m_linedPixmap);
157  p.setCompositionMode(QPainter::CompositionMode_SourceAtop);
158  p.fillRect(m_linedPixmap.rect(), QColor(0, 0, 0, 100));
159  }
160 
161  QPixmap pixmap = m_linedPixmap;
162  painter.begin(&pixmap);
163  painter.drawPixmap(m_selectedRegion.topLeft(),
164  m_originalPixmap, m_selectedRegion);
165 
166  painter.end();
167 
168  m_label->setPixmap(pixmap);
169 
170  qApp->sendPostedEvents(0, QEvent::LayoutRequest);
171 
172  if (m_selectedRegion == m_originalPixmap.rect()) {//d->m_label->rect()) //### CHECK!
173  m_rubberBand->hide();
174  } else {
175  m_rubberBand->setGeometry(QRect(m_selectedRegion.topLeft(),
176  m_selectedRegion.size()));
177 
178  /* m_rubberBand->setGeometry(QRect(m_label -> mapToGlobal(m_selectedRegion.topLeft()),
179  m_selectedRegion.size()));
180  */
181  if (m_state != None) {
182  m_rubberBand->show();
183  }
184  }
185 }

205 void KPixmapRegionSelectorWidget::rotate(RotateDirection direction)
206 {
207  int w = d->m_originalPixmap.width();
208  int h = d->m_originalPixmap.height();
209  QImage img = d->m_unzoomedPixmap.toImage();
210  if (direction == Rotate90) {
211  img = img.transformed(QTransform().rotate(90.0));
212  } else if (direction == Rotate180) {
213  img = img.transformed(QTransform().rotate(180.0));
214  } else {
215  img = img.transformed(QTransform().rotate(270.0));
216  }
217 
218  d->m_unzoomedPixmap = QPixmap::fromImage(img);
219 
220  img = d->m_originalPixmap.toImage();
221  if (direction == Rotate90) {
222  img = img.transformed(QTransform().rotate(90.0));
223  } else if (direction == Rotate180) {
224  img = img.transformed(QTransform().rotate(180.0));
225  } else {
226  img = img.transformed(QTransform().rotate(270.0));
227  }
228 
229  d->m_originalPixmap = QPixmap::fromImage(img);
230 
231  d->m_linedPixmap = QPixmap();
232 
233  if (d->m_forcedAspectRatio > 0 && d->m_forcedAspectRatio != 1) {
234  resetSelection();
235  } else {
236  switch (direction) {
237  case (Rotate90): {
238  int x = h - d->m_selectedRegion.y() - d->m_selectedRegion.height();
239  int y = d->m_selectedRegion.x();
240  d->m_selectedRegion.setRect(x, y, d->m_selectedRegion.height(), d->m_selectedRegion.width());
241  d->updatePixmap();
242  // qApp->sendPostedEvents(0,QEvent::LayoutRequest);
243  // updatePixmap();
244 
245  } break;
246  case (Rotate270): {
247  int x = d->m_selectedRegion.y();
248  int y = w - d->m_selectedRegion.x() - d->m_selectedRegion.width();
249  d->m_selectedRegion.setRect(x, y, d->m_selectedRegion.height(), d->m_selectedRegion.width());
250  d->updatePixmap();
251  // qApp->sendPostedEvents(0,QEvent::LayoutRequest);
252  // updatePixmap();
253  } break;
254  default: resetSelection();
255  }
256  }
257 
258  emit pixmapRotated();
259 }

271 bool KPixmapRegionSelectorWidget::eventFilter(QObject *obj, QEvent *ev)
272 {
273  if (ev->type() == QEvent::MouseButtonPress) {
274  QMouseEvent *mev = (QMouseEvent *)(ev);
275  //qDebug() << QString("click at %1,%2").arg( mev->x() ).arg( mev->y() );
276 
277  if (mev->button() == Qt::RightButton) {
278  QMenu *popup = createPopupMenu();
279  popup->exec(mev->globalPos());
280  delete popup;
281  return true;
282  }
283 
284  QCursor cursor;
285 
286  if (d->m_selectedRegion.contains(mev->pos())
287  && d->m_selectedRegion != d->m_originalPixmap.rect()) {
288  d->m_state = Private::Moving;
289  cursor.setShape(Qt::SizeAllCursor);
290  d->m_rubberBand->show();
291  } else {
292  d->m_state = Private::Resizing;
293  cursor.setShape(Qt::CrossCursor);
294  }
295  QApplication::setOverrideCursor(cursor);
296 
297  d->m_tempFirstClick = mev->pos();
298 
299  return true;
300  }
301 
302  if (ev->type() == QEvent::MouseMove) {
303  QMouseEvent *mev = (QMouseEvent *)(ev);
304 
305  //qDebug() << QString("move to %1,%2").arg( mev->x() ).arg( mev->y() );
306 
307  if (d->m_state == Private::Resizing) {
308  setSelectedRegion(
309  d->calcSelectionRectangle(d->m_tempFirstClick, mev->pos()));
310  } else if (d->m_state == Private::Moving) {
311  int mevx = mev->x();
312  int mevy = mev->y();
313  bool mouseOutside = false;
314  if (mevx < 0) {
315  d->m_selectedRegion.translate(-d->m_selectedRegion.x(), 0);
316  mouseOutside = true;
317  } else if (mevx > d->m_originalPixmap.width()) {
318  d->m_selectedRegion.translate(d->m_originalPixmap.width() - d->m_selectedRegion.width() - d->m_selectedRegion.x(), 0);
319  mouseOutside = true;
320  }
321  if (mevy < 0) {
322  d->m_selectedRegion.translate(0, -d->m_selectedRegion.y());
323  mouseOutside = true;
324  } else if (mevy > d->m_originalPixmap.height()) {
325  d->m_selectedRegion.translate(0, d->m_originalPixmap.height() - d->m_selectedRegion.height() - d->m_selectedRegion.y());
326  mouseOutside = true;
327  }
328  if (mouseOutside) {
329  d->updatePixmap();
330  return true;
331  };
332 
333  d->m_selectedRegion.translate(mev->x() - d->m_tempFirstClick.x(),
334  mev->y() - d->m_tempFirstClick.y());
335 
336  // Check that the region has not fallen outside the image
337  if (d->m_selectedRegion.x() < 0) {
338  d->m_selectedRegion.translate(-d->m_selectedRegion.x(), 0);
339  } else if (d->m_selectedRegion.right() > d->m_originalPixmap.width()) {
340  d->m_selectedRegion.translate(-(d->m_selectedRegion.right() - d->m_originalPixmap.width()), 0);
341  }
342 
343  if (d->m_selectedRegion.y() < 0) {
344  d->m_selectedRegion.translate(0, -d->m_selectedRegion.y());
345  } else if (d->m_selectedRegion.bottom() > d->m_originalPixmap.height()) {
346  d->m_selectedRegion.translate(0, -(d->m_selectedRegion.bottom() - d->m_originalPixmap.height()));
347  }
348 
349  d->m_tempFirstClick = mev->pos();
350  d->updatePixmap();
351  }
352  return true;
353  }
354 
355  if (ev->type() == QEvent::MouseButtonRelease) {
356  QMouseEvent *mev = (QMouseEvent *)(ev);
357 
358  if (d->m_state == Private::Resizing && mev->pos() == d->m_tempFirstClick) {
359  resetSelection();
360  }
361 
362  d->m_state = Private::None;
363  QApplication::restoreOverrideCursor();
364  d->m_rubberBand->hide();
365  return true;
366  }
367 
368  QWidget::eventFilter(obj, ev);
369  return false;
370 }

372 QRect KPixmapRegionSelectorWidget::Private::calcSelectionRectangle(const QPoint &startPoint, const QPoint &_endPoint)
373 {
374  QPoint endPoint = _endPoint;
375  if (endPoint.x() < 0) {
376  endPoint.setX(0);
377  } else if (endPoint.x() > m_originalPixmap.width()) {
378  endPoint.setX(m_originalPixmap.width());
379  }
380  if (endPoint.y() < 0) {
381  endPoint.setY(0);
382  } else if (endPoint.y() > m_originalPixmap.height()) {
383  endPoint.setY(m_originalPixmap.height());
384  }
385  int w = abs(startPoint.x() - endPoint.x());
386  int h = abs(startPoint.y() - endPoint.y());
387 
388  if (m_forcedAspectRatio > 0) {
389  double aspectRatio = w / double(h);
390 
391  if (aspectRatio > m_forcedAspectRatio) {
392  h = (int)(w / m_forcedAspectRatio);
393  } else {
394  w = (int)(h * m_forcedAspectRatio);
395  }
396  }
397 
398  int x, y;
399  if (startPoint.x() < endPoint.x()) {
400  x = startPoint.x();
401  } else {
402  x = startPoint.x() - w;
403  }
404 
405  if (startPoint.y() < endPoint.y()) {
406  y = startPoint.y();
407  } else {
408  y = startPoint.y() - h;
409  }
410 
411  if (x < 0) {
412  w += x;
413  x = 0;
414  h = (int)(w / m_forcedAspectRatio);
415 
416  if (startPoint.y() > endPoint.y()) {
417  y = startPoint.y() - h;
418  }
419  } else if (x + w > m_originalPixmap.width()) {
420  w = m_originalPixmap.width() - x;
421  h = (int)(w / m_forcedAspectRatio);
422 
423  if (startPoint.y() > endPoint.y()) {
424  y = startPoint.y() - h;
425  }
426  }
427 
428  if (y < 0) {
429  h += y;
430  y = 0;
431  w = (int)(h * m_forcedAspectRatio);
432 
433  if (startPoint.x() > endPoint.x()) {
434  x = startPoint.x() - w;
435  }
436  } else if (y + h > m_originalPixmap.height()) {
437  h = m_originalPixmap.height() - y;
438  w = (int)(h * m_forcedAspectRatio);
439 
440  if (startPoint.x() > endPoint.x()) {
441  x = startPoint.x() - w;
442  }
443  }
444 
445  return QRect(x, y, w, h);
446 }

472 void KPixmapRegionSelectorWidget::setMaximumWidgetSize(int width, int height)
473 {
474  d->m_maxWidth = width;
475  d->m_maxHeight = height;
476 
477  if (d->m_selectedRegion == d->m_originalPixmap.rect()) {
478  d->m_selectedRegion = QRect();
479  }
480  d->m_originalPixmap = d->m_unzoomedPixmap;
481 
482  // qDebug() << QString(" original Pixmap :") << d->m_originalPixmap.rect();
483  // qDebug() << QString(" unzoomed Pixmap : %1 x %2 ").arg(d->m_unzoomedPixmap.width()).arg(d->m_unzoomedPixmap.height());
484 
485  if (!d->m_originalPixmap.isNull() &&
486  (d->m_originalPixmap.width() > d->m_maxWidth ||
487  d->m_originalPixmap.height() > d->m_maxHeight)) {
488  /* We have to resize the pixmap to get it complete on the screen */
489  QImage image = d->m_originalPixmap.toImage();
490  d->m_originalPixmap = QPixmap::fromImage(image.scaled(width, height, Qt::KeepAspectRatio, Qt::SmoothTransformation));
491  double oldZoomFactor = d->m_zoomFactor;
492  d->m_zoomFactor = d->m_originalPixmap.width() / (double)d->m_unzoomedPixmap.width();
493 
494  if (d->m_selectedRegion.isValid()) {
495  d->m_selectedRegion =
496  QRect((int)(d->m_selectedRegion.x() * d->m_zoomFactor / oldZoomFactor),
497  (int)(d->m_selectedRegion.y() * d->m_zoomFactor / oldZoomFactor),
498  (int)(d->m_selectedRegion.width() * d->m_zoomFactor / oldZoomFactor),
499  (int)(d->m_selectedRegion.height() * d->m_zoomFactor / oldZoomFactor));
500  }
501  }
502 
503  if (!d->m_selectedRegion.isValid()) {
504  d->m_selectedRegion = d->m_originalPixmap.rect();
505  }
506 
507  d->m_linedPixmap = QPixmap();
508  d->updatePixmap();
509  resize(d->m_label->width(), d->m_label->height());
510 }
