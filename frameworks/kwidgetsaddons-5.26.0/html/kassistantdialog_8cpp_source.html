47  QModelIndex getNext(QModelIndex nextIndex)
48  {
49  QModelIndex currentIndex;
50  do {
51  currentIndex = nextIndex;
52  nextIndex = currentIndex.child(0, 0);
53  if (!nextIndex.isValid()) {
54  nextIndex = currentIndex.sibling(currentIndex.row() + 1, 0);
55  }
56  } while (nextIndex.isValid() && !appropriate.value(pageModel->item(nextIndex), true));
57  return nextIndex;
58  }
59 
60  QModelIndex getPrevious(QModelIndex nextIndex)
61  {
62  QModelIndex currentIndex;
63  do {
64  currentIndex = nextIndex;
65  nextIndex = currentIndex.sibling(currentIndex.row() - 1, 0);
66  if (!nextIndex.isValid()) {
67  nextIndex = currentIndex.parent();
68  }
69  } while (nextIndex.isValid() && !appropriate.value(pageModel->item(nextIndex), true));
70  return nextIndex;
71  }
72 };

96 void KAssistantDialog::Private::init()
97 {
98  QDialogButtonBox *buttonBox = q->buttonBox();
99 
100  buttonBox->setStandardButtons(QDialogButtonBox::Cancel | QDialogButtonBox::Help);
101  backButton = new QPushButton;
102 
103  const QString iconBack = QApplication::isRightToLeft() ? QStringLiteral("go-next") : QStringLiteral("go-previous");
104  const QString iconNext = QApplication::isRightToLeft() ? QStringLiteral("go-previous") : QStringLiteral("go-next");
105  backButton->setText(tr("&Back", "go back"));
106  backButton->setIcon(QIcon::fromTheme(iconBack));
107  backButton->setToolTip(tr("Go back one step"));
108  q->connect(backButton, &QAbstractButton::clicked, q, &KAssistantDialog::back);
109  buttonBox->addButton(backButton, QDialogButtonBox::ActionRole);
110 
111  nextButton = new QPushButton;
112  nextButton->setText(tr("Next", "Opposite to Back"));
113  nextButton->setIcon(QIcon::fromTheme(iconNext));
114  nextButton->setDefault(true);
115  q->connect(nextButton, &QAbstractButton::clicked, q, &KAssistantDialog::next);
116  buttonBox->addButton(nextButton, QDialogButtonBox::ActionRole);
117 
118  finishButton = new QPushButton;
119  finishButton->setText(tr("Finish"));
120  finishButton->setIcon(QIcon::fromTheme(QStringLiteral("dialog-ok-apply")));
121  buttonBox->addButton(finishButton, QDialogButtonBox::AcceptRole);
122 
123  q->setFaceType(KPageDialog::Plain);
124 
125  q->connect(q, SIGNAL(currentPageChanged(KPageWidgetItem*,KPageWidgetItem*)), q, SLOT(_k_slotUpdateButtons()));
126 }

159 void KAssistantDialog::Private::_k_slotUpdateButtons()
160 {
161  QModelIndex currentIndex = pageModel->index(q->currentPage());
162  //change the caption of the next/finish button
163  QModelIndex nextIndex = getNext(currentIndex);
164  finishButton->setEnabled(!nextIndex.isValid() && q->isValid(q->currentPage()));
165  nextButton->setEnabled(nextIndex.isValid() && q->isValid(q->currentPage()));
166  finishButton->setDefault(!nextIndex.isValid());
167  nextButton->setDefault(nextIndex.isValid());
168  //enable or disable the back button;
169  nextIndex = getPrevious(currentIndex);
170  backButton->setEnabled(nextIndex.isValid());
171 }
