69 void KNewPasswordDialog::KNewPasswordDialogPrivate::_k_passwordStatusChanged()
70 {
71  switch (ui.pwdWidget->passwordStatus()) {
72  case KNewPasswordWidget::PasswordTooShort:
73  ui.buttonBox->button(QDialogButtonBox::Ok)->setEnabled(false);
74  //~ singular Password must be at least %n character long
75  //~ plural Password must be at least %n characters long
76  showMessageWidget(tr("Password must be at least %n character(s) long", "", ui.pwdWidget->minimumPasswordLength()), KMessageWidget::Error);
77  break;
78  case KNewPasswordWidget::EmptyPasswordNotAllowed:
79  ui.buttonBox->button(QDialogButtonBox::Ok)->setEnabled(false);
80  showMessageWidget(tr("Password is empty"), KMessageWidget::Warning);
81  break;
82  case KNewPasswordWidget::PasswordNotVerified:
83  ui.buttonBox->button(QDialogButtonBox::Ok)->setEnabled(false);
84  showMessageWidget(tr("Passwords do not match"), KMessageWidget::Warning);
85  break;
86  case KNewPasswordWidget::WeakPassword:
87  case KNewPasswordWidget::StrongPassword:
88  ui.buttonBox->button(QDialogButtonBox::Ok)->setEnabled(true);
89  showMessageWidget(tr("Passwords match"), KMessageWidget::Positive);
90  break;
91  }
92 }

138 bool KNewPasswordDialog::checkAndGetPassword(QString *pwd)
139 {
140  pwd->clear();
141 
142  if (d->ui.pwdWidget->passwordStatus() == KNewPasswordWidget::WeakPassword) {
143  QMessageBox::StandardButton selectedButton = QMessageBox::warning(this,
144  tr("Low Password Strength"),
145  tr("The password you have entered has a low strength. "
146  "To improve the strength of "
147  "the password, try:\n"
148  " - using a longer password;\n"
149  " - using a mixture of upper- and lower-case letters;\n"
150  " - using numbers or symbols as well as letters.\n"
151  "\n"
152  "Would you like to use this password anyway?"),
153  QMessageBox::Yes | QMessageBox::No);
154 
155  if (selectedButton == QMessageBox::No) {
156  return false;
157  }
158  }
159  if (!checkPassword(d->ui.pwdWidget->password())) {
160  return false;
161  }
162 
163  *pwd = d->ui.pwdWidget->password();
164  return true;
165 }

167 void KNewPasswordDialog::accept()
168 {
169  QString pwd;
170  if (!checkAndGetPassword(&pwd)) {
171  return;
172  }
173  d->pass = pwd;
174  emit newPassword(d->pass);
175  QDialog::accept();
176 }
