75 void KCursorPrivateAutoHideEventFilter::hideCursor()
76 {
77  m_autoHideTimer.stop();
78 
79  if (m_isCursorHidden) {
80  return;
81  }
82 
83  m_isCursorHidden = true;
84 
85  QWidget *w = mouseWidget();
86 
87  m_isOwnCursor = w->testAttribute(Qt::WA_SetCursor);
88  if (m_isOwnCursor) {
89  m_oldCursor = w->cursor();
90  }
91 
92  w->setCursor(QCursor(Qt::BlankCursor));
93 }
94 
95 void KCursorPrivateAutoHideEventFilter::unhideCursor()
96 {
97  m_autoHideTimer.stop();
98 
99  if (!m_isCursorHidden) {
100  return;
101  }
102 
103  m_isCursorHidden = false;
104 
105  QWidget *w = mouseWidget();
106 
107  if (w->cursor().shape() != Qt::BlankCursor) { // someone messed with the cursor already
108  return;
109  }
110 
111  if (m_isOwnCursor) {
112  w->setCursor(m_oldCursor);
113  } else {
114  w->unsetCursor();
115  }
116 }

133 bool KCursorPrivateAutoHideEventFilter::eventFilter(QObject *o, QEvent *e)
134 {
135  Q_UNUSED(o);
136  // o is m_widget or its viewport
137  //Q_ASSERT( o == m_widget );
138 
139  switch (e->type()) {
140  case QEvent::Leave:
141  case QEvent::FocusOut:
142  case QEvent::WindowDeactivate:
143  unhideCursor();
144  break;
145  case QEvent::KeyPress:
146  case QEvent::ShortcutOverride:
147  hideCursor();
148  break;
149  case QEvent::Enter:
150  case QEvent::FocusIn:
151  case QEvent::MouseButtonPress:
152  case QEvent::MouseButtonRelease:
153  case QEvent::MouseButtonDblClick:
154  case QEvent::MouseMove:
155  case QEvent::Show:
156  case QEvent::Hide:
157  case QEvent::Wheel:
158  unhideCursor();
159  if (m_widget->hasFocus()) {
160  m_autoHideTimer.setSingleShot(true);
161  m_autoHideTimer.start(KCursorPrivate::self()->hideCursorDelay);
162  }
163  break;
164  default:
165  break;
166  }
167 
168  return false;
169 }

196 void KCursorPrivate::setAutoHideCursor(QWidget *w, bool enable, bool customEventFilter)
197 {
198  if (!w || !enabled) {
199  return;
200  }
201 
202  QWidget *viewport = 0;
203  QAbstractScrollArea *sv = qobject_cast<QAbstractScrollArea *>(w);
204  if (sv) {
205  viewport = sv->viewport();
206  }
207 
208  if (enable) {
209  if (m_eventFilters.contains(w)) {
210  return;
211  }
212  KCursorPrivateAutoHideEventFilter *filter = new KCursorPrivateAutoHideEventFilter(w);
213  m_eventFilters.insert(w, filter);
214  if (viewport) {
215  m_eventFilters.insert(viewport, filter);
216  connect(viewport, &QObject::destroyed, this, &KCursorPrivate::slotViewportDestroyed);
217  }
218  if (!customEventFilter) {
219  w->installEventFilter(filter); // for key events
220  if (viewport) {
221  viewport->installEventFilter(filter); // for mouse events
222  }
223  }
224  connect(w, &QObject::destroyed,
225  this, &KCursorPrivate::slotWidgetDestroyed);
226  } else {
227  KCursorPrivateAutoHideEventFilter *filter = m_eventFilters.take(w);
228  if (filter == 0) {
229  return;
230  }
231  w->removeEventFilter(filter);
232  if (viewport) {
233  m_eventFilters.remove(viewport);
234  disconnect(viewport, &QObject::destroyed, this, &KCursorPrivate::slotViewportDestroyed);
235  viewport->removeEventFilter(filter);
236  }
237  delete filter;
238  disconnect(w, &QObject::destroyed,
239  this, &KCursorPrivate::slotWidgetDestroyed);
240  }
241 }

243 bool KCursorPrivate::eventFilter(QObject *o, QEvent *e)
244 {
245  if (!enabled || e->type() == QEvent::ChildAdded) {
246  return false;
247  }
248 
249  KCursorPrivateAutoHideEventFilter *filter = m_eventFilters.value(o);
250 
251  Q_ASSERT(filter != 0);
252  if (filter == 0) {
253  return false;
254  }
255 
256  return filter->eventFilter(o, e);
257 }

259 void KCursorPrivate::slotViewportDestroyed(QObject *o)
260 {
261  m_eventFilters.remove(o);
262 }

264 void KCursorPrivate::slotWidgetDestroyed(QObject *o)
265 {
266  KCursorPrivateAutoHideEventFilter *filter = m_eventFilters.take(o);
267 
268  Q_ASSERT(filter != 0);
269 
270  filter->resetWidget(); // so that dtor doesn't access it
271  delete filter;
272 }
