32 void KPageWidgetPrivate::_k_slotCurrentPageChanged(const QModelIndex &current, const QModelIndex &before)
33 {
34  KPageWidgetItem *currentItem = 0;
35  if (current.isValid()) {
36  currentItem = model()->item(current);
37  }
38 
39  KPageWidgetItem *beforeItem = 0;
40  if (before.isValid()) {
41  beforeItem = model()->item(before);
42  }
43 
44  Q_Q(KPageWidget);
45  emit q->currentPageChanged(currentItem, beforeItem);
46 }

48 KPageWidget::KPageWidget(KPageWidgetPrivate &dd, QWidget *parent)
49  : KPageView(dd, parent)
50 {
51  Q_D(KPageWidget);
52  connect(this, SIGNAL(currentPageChanged(QModelIndex,QModelIndex)),
53  this, SLOT(_k_slotCurrentPageChanged(QModelIndex,QModelIndex)));
54 
55  if (!d->KPageViewPrivate::model) {
56  setModel(new KPageWidgetModel(this));
57  } else {
58  Q_ASSERT(qobject_cast<KPageWidgetModel *>(d->KPageViewPrivate::model));
59  }
60 
61  connect(d->model(), &KPageWidgetModel::toggled,
62  this, &KPageWidget::pageToggled);
63 }
