68  bool eventFilter(QObject *object, QEvent *event) Q_DECL_OVERRIDE
69  {
70  Q_UNUSED(object);
71 
72  // To catch outside clicks, it is sufficient to check for
73  // hide events on Qt::Popup type widgets
74  if (event->type() == QEvent::Hide && m_popup) {
75  // do not set d->result here, because the popup
76  // hides itself after leaving the event loop.
77  emit m_popup->leaveModality();
78  }
79  return false;
80  }
81 };

148 void KPopupFrame::popup(const QPoint &pos)
149 {
150  // Make sure the whole popup is visible.
151  QRect desktopGeometry = QApplication::desktop()->screenGeometry(pos);
152 
153  int x = pos.x();
154  int y = pos.y();
155  int w = width();
156  int h = height();
157  if (x + w > desktopGeometry.x() + desktopGeometry.width()) {
158  x = desktopGeometry.width() - w;
159  }
160  if (y + h > desktopGeometry.y() + desktopGeometry.height()) {
161  y = desktopGeometry.height() - h;
162  }
163  if (x < desktopGeometry.x()) {
164  x = 0;
165  }
166  if (y < desktopGeometry.y()) {
167  y = 0;
168  }
169 
170  // Pop the thingy up.
171  move(x, y);
172  show();
173  d->main->setFocus();
174 }
