151 void KViewStateSerializerPrivate::listenToPendingChanges()
152 {
153  Q_Q(KViewStateSerializer);
154  // watch the model for stuff coming in delayed
155  if (hasPendingChanges()) {
156  const QAbstractItemModel *model = getModel();
157  if (model) {
158  q->disconnect(model, SIGNAL(rowsInserted(QModelIndex,int,int)),
159  q, SLOT(rowsInserted(QModelIndex,int,int)));
160  q->connect(model, SIGNAL(rowsInserted(QModelIndex,int,int)),
161  SLOT(rowsInserted(QModelIndex,int,int)));
162  return;
163  } else {
164  q->deleteLater();
165  }
166  } else {
167  q->deleteLater();
168  }
169 }

181 QStringList KViewStateSerializerPrivate::getExpandedItems(const QModelIndex &index) const
182 {
183  Q_Q(const KViewStateSerializer);
184 
185  QStringList expansion;
186  for (int i = 0; i < m_treeView->model()->rowCount(index); ++i) {
187  const QModelIndex child = m_treeView->model()->index(i, 0, index);
188 
189  // http://bugreports.qt.nokia.com/browse/QTBUG-18039
190  if (m_treeView->model()->hasChildren(child)) {
191  if (m_treeView->isExpanded(child)) {
192  expansion << q->indexToConfigString(child);
193  }
194  expansion << getExpandedItems(child);
195  }
196  }
197  return expansion;
198 }

200 void KViewStateSerializerPrivate::restoreCurrentItem()
201 {
202  Q_Q(KViewStateSerializer);
203 
204  QModelIndex currentIndex = q->indexFromConfigString(m_selectionModel->model(), m_pendingCurrent);
205  if (currentIndex.isValid()) {
206  if (m_treeView) {
207  m_treeView->setCurrentIndex(currentIndex);
208  } else {
209  m_selectionModel->setCurrentIndex(currentIndex, QItemSelectionModel::NoUpdate);
210  }
211  m_pendingCurrent.clear();
212  }
213 }

233 void KViewStateSerializerPrivate::restoreExpanded()
234 {
235  Q_Q(KViewStateSerializer);
236 
237  QSet<QString>::iterator it = m_pendingExpansions.begin();
238  for (; it != m_pendingExpansions.end();) {
239  QModelIndex idx = q->indexFromConfigString(m_treeView->model(), *it);
240  if (idx.isValid()) {
241  m_treeView->expand(idx);
242  it = m_pendingExpansions.erase(it);
243  } else {
244  ++it;
245  }
246  }
