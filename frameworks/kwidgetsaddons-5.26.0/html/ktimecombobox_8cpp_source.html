93 QString KTimeComboBoxPrivate::timeFormatToInputMask(const QString &format, bool nullMask)
94 {
95  const QLocale locale = q->locale();
96 
97  //TODO not sure this will always work, does it support DigitSets, am/pm is dodgy?
98  QString mask = formatTime(QTime(12, 34, 56, 789));
99  QString null = mask;
100  mask.replace(locale.toString(12), QLatin1String("09"));
101  null.replace(locale.toString(12), QLatin1String(""));
102  mask.replace(locale.toString(34), QLatin1String("99"));
103  null.replace(locale.toString(34), QLatin1String(""));
104  mask.replace(locale.toString(56), QLatin1String("99"));
105  null.replace(locale.toString(56), QLatin1String(""));
106  mask.replace(locale.toString(789), QLatin1String("900"));
107  null.replace(locale.toString(789), QLatin1String(""));
108  if (format.contains(QLatin1String("ap")) ||
109  format.contains(QLatin1String("AP"))) {
110  QString am = locale.amText();
111  QString pm = locale.pmText();
112  int ampmLen = qMax(am.length(), pm.length());
113  QString ampmMask;
114  for (int i = 0; i < ampmLen; ++i) {
115  ampmMask.append(QLatin1Char('a'));
116  }
117  mask.replace(pm, ampmMask);
118  null.replace(pm, QLatin1String(""));
119  }
120 
121  if (nullMask) {
122  return null;
123  } else {
124  return mask;
125  }
126 }

128 QTime KTimeComboBoxPrivate::nearestIntervalTime(const QTime &time)
129 {
130  int i = 0;
131  while (q->itemData(i).toTime() < time) {
132  ++i;
133  }
134  QTime before = q->itemData(i).toTime();
135  QTime after = q->itemData(i + 1).toTime();
136  if (before.secsTo(time) <= time.secsTo(after)) {
137  return before;
138  } else {
139  return after;
140  }
141 }

148 void KTimeComboBoxPrivate::initTimeWidget()
149 {
150  q->blockSignals(true);
151  q->clear();
152 
153  // Set the input mask from the current format
154  const QLocale locale = q->locale();
155  q->lineEdit()->setInputMask(timeFormatToInputMask(locale.timeFormat(m_displayFormat)));
156  m_nullString = timeFormatToInputMask(locale.timeFormat(m_displayFormat), true);
157 
158  // If EditTime then set the line edit
159  q->lineEdit()->setReadOnly((m_options & KTimeComboBox::EditTime) != KTimeComboBox::EditTime);
160 
161  // If SelectTime then make list items visible
162  if ((m_options & KTimeComboBox::SelectTime) == KTimeComboBox::SelectTime) {
163  q->setMaxVisibleItems(10);
164  } else {
165  q->setMaxVisibleItems(0);
166  }
167 
168  // Populate the drop-down time list
169  // If no time list set the use the time interval
170  if (m_timeList.isEmpty()) {
171  QTime startTime = m_minTime;
172  QTime thisTime(startTime.hour(), 0, 0, 0);
173  while (thisTime.isValid() && thisTime <= startTime) {
174  thisTime = thisTime.addSecs(m_timeListInterval * 60);
175  }
176  QTime endTime = m_maxTime;
177  q->addItem(formatTime(startTime), startTime);
178  while (thisTime.isValid() && thisTime < endTime) {
179  q->addItem(formatTime(thisTime), thisTime);
180  QTime newTime = thisTime.addSecs(m_timeListInterval * 60);
181  if (newTime.isValid() && newTime > thisTime) {
182  thisTime = newTime;
183  } else {
184  thisTime = QTime();
185  }
186  }
187  q->addItem(formatTime(endTime), endTime);
188  } else {
189  foreach (const QTime &thisTime, m_timeList) {
190  if (thisTime.isValid() && thisTime >= m_minTime && thisTime <= m_maxTime) {
191  q->addItem(formatTime(thisTime), thisTime);
192  }
193  }
194  }
195  q->blockSignals(false);
196 }

198 void KTimeComboBoxPrivate::updateTimeWidget()
199 {
200  q->blockSignals(true);
201  int pos = q->lineEdit()->cursorPosition();
202  //Set index before setting text otherwise it overwrites
203  int i = 0;
204  if (!m_time.isValid() || m_time < m_minTime) {
205  i = 0;
206  } else if (m_time > m_maxTime) {
207  i = q->count() - 1;
208  } else {
209  while (q->itemData(i).toTime() < m_time && i < q->count() - 1) {
210  ++i;
211  }
212  }
213  q->setCurrentIndex(i);
214  if (m_time.isValid()) {
215  q->lineEdit()->setText(formatTime(m_time));
216  } else {
217  q->lineEdit()->setText(QString());
218  }
219  q->lineEdit()->setCursorPosition(pos);
220  q->blockSignals(false);
221 }

246 void KTimeComboBoxPrivate::warnTime()
247 {
248  if (!m_warningShown && !q->isValid() &&
249  (m_options & KTimeComboBox::WarnOnInvalid) == KTimeComboBox::WarnOnInvalid) {
250  QString warnMsg;
251  if (!m_time.isValid()) {
252  warnMsg = KTimeComboBox::tr("The time you entered is invalid", "@info");
253  } else if (m_time < m_minTime) {
254  if (m_minWarnMsg.isEmpty()) {
255  warnMsg = KTimeComboBox::tr("Time cannot be earlier than %1", "@info").arg(formatTime(m_minTime));
256  } else {
257  warnMsg = m_minWarnMsg;
258  warnMsg.replace(QStringLiteral("%1"), formatTime(m_minTime));
259  }
260  } else if (m_time > m_maxTime) {
261  if (m_maxWarnMsg.isEmpty()) {
262  warnMsg = KTimeComboBox::tr("Time cannot be later than %1", "@info").arg(formatTime(m_maxTime));
263  } else {
264  warnMsg = m_maxWarnMsg;
265  warnMsg.replace(QStringLiteral("%1"), formatTime(m_maxTime));
266  }
267  }
268  m_warningShown = true;
269  KMessageBox::sorry(q, warnMsg);
270  }

420 void KTimeComboBox::setTimeListInterval(int minutes)
421 {
422  if (minutes != d->m_timeListInterval) {
423  //Must be able to exactly divide the valid time period
424  int lowMins = (d->m_minTime.hour() * 60) + d->m_minTime.minute();
425  int hiMins = (d->m_maxTime.hour() * 60) + d->m_maxTime.minute();
426  if (d->m_minTime.minute() == 0 && d->m_maxTime.minute() == 59) {
427  ++hiMins;
428  }
429  if ((hiMins - lowMins) % minutes == 0) {
430  d->m_timeListInterval = minutes;
431  d->m_timeList.clear();
432  } else {
433  return;
434  }
435  d->initTimeWidget();
436  }
437 }

472 void KTimeComboBox::keyPressEvent(QKeyEvent *keyEvent)
473 {
474  QTime temp;
475  switch (keyEvent->key()) {
476  case Qt::Key_Down:
477  temp = d->m_time.addSecs(-60);
478  break;
479  case Qt::Key_Up:
480  temp = d->m_time.addSecs(60);
481  break;
482  case Qt::Key_PageDown:
483  temp = d->m_time.addSecs(-3600);
484  break;
485  case Qt::Key_PageUp:
486  temp = d->m_time.addSecs(3600);
487  break;
488  default:
489  QComboBox::keyPressEvent(keyEvent);
490  return;
491  }
492  if (temp.isValid() && temp >= d->m_minTime && temp <= d->m_maxTime) {
493  d->enterTime(temp);
494  }
495 }
