88 void KFontSizeAction::setFontSize(int size)
89 {
90  if (size == fontSize()) {
91  const QString test = QString::number(size);
92  Q_FOREACH (QAction *action, actions()) {
93  if (action->text() == test) {
94  setCurrentAction(action);
95  return;
96  }
97  }
98  }
99 
100  if (size < 1) {
101  qWarning() << "KFontSizeAction: Size " << size << " is out of range";
102  return;
103  }
104 
105  QAction *a = action(QString::number(size));
106  if (!a) {
107  // Insert at the correct position in the list (to keep sorting)
108  QList<int> lst;
109  // Convert to list of ints
110  QStringListIterator itemsIt(items());
111  while (itemsIt.hasNext()) {
112  lst.append(itemsIt.next().toInt());
113  }
114  // New size
115  lst.append(size);
116  // Sort the list
117  qSort(lst);
118  Q_FOREACH (int it, lst) {
119  QAction *const action = addAction(QString::number(it));
120  if (it == size) {
121  setCurrentAction(action);
122  }
123  }
124 
125  } else {
126  setCurrentAction(a);
127  }
128 }
