43 static int minimumListWidth(const QListWidget *list)
44 {
45  int w = 0;
46  for (int i = 0; i < list->count(); i++) {
47  int itemWidth = list->visualItemRect(list->item(i)).width();
48  // ...and add a space on both sides for not too tight look.
49  itemWidth += list->fontMetrics().width(QLatin1Char(' ')) * 2;
50  w = qMax(w, itemWidth);
51  }
52  if (w == 0) {
53  w = 40;
54  }
55  w += list->frameWidth() * 2;
56  w += list->verticalScrollBar()->sizeHint().width();
57  return w;
58 }
59 
60 static int minimumListHeight(const QListWidget *list, int numVisibleEntry)
61 {
62  int w = list->count() > 0 ? list->visualItemRect(list->item(0)).height() :
63  list->fontMetrics().lineSpacing();
64 
65  if (w < 0) {
66  w = 10;
67  }
68  if (numVisibleEntry <= 0) {
69  numVisibleEntry = 4;
70  }
71  return (w * numVisibleEntry + 2 * list->frameWidth());
72 }

546 void KFontChooser::Private::_k_family_chosen_slot(const QString &family)
547 {
548  if (!signalsAllowed) {
549  return;
550  }
551  signalsAllowed = false;
552 
553  QString currentFamily;
554  if (family.isEmpty()) {
555  Q_ASSERT(familyListBox->currentItem());
556  if (familyListBox->currentItem()) {
557  currentFamily = qtFamilies[familyListBox->currentItem()->text()];
558  }
559  } else {
560  currentFamily = qtFamilies[family];
561  }
562 
563  // Get the list of styles available in this family.
564  QFontDatabase dbase;
565  QStringList styles = dbase.styles(currentFamily);
566  if (styles.isEmpty()) {
567  // Avoid extraction, it is in kdeqt.po
568  styles.append(TR_NOX("Normal", "QFontDatabase"));
569  }
570 
571  // Filter style strings and add to the listbox.
572  QString pureFamily;
573  splitFontString(family, &pureFamily);
574  QStringList filteredStyles;
575  qtStyles.clear();
576  styleIDs.clear();
577  Q_FOREACH (const QString &style, styles) {
578  // Sometimes the font database will report an invalid style,
579  // that falls back back to another when set.
580  // Remove such styles, by checking set/get round-trip.
581  QFont testFont = dbase.font(currentFamily, style, 10);
582  if (dbase.styleString(testFont) != style) {
583  styles.removeAll(style);
584  continue;
585  }
586 
587  QString fstyle = tr("%1", "@item Font style").arg(style);
588  if (!filteredStyles.contains(fstyle)) {
589  filteredStyles.append(fstyle);
590  qtStyles.insert(fstyle, style);
591  styleIDs.insert(fstyle, styleIdentifier(testFont));
592  }
593  }
594  styleListBox->clear();
595  styleListBox->addItems(filteredStyles);
596 
597  // Try to set the current style in the listbox to that previous.
598  int listPos = filteredStyles.indexOf(selectedStyle.isEmpty() ? TR_NOX("Normal", "QFontDatabase") : selectedStyle);
599  if (listPos < 0) {
600  // Make extra effort to have Italic selected when Oblique was chosen,
601  // and vice versa, as that is what the user would probably want.
602  QString styleIt = tr("Italic", "@item font");
603  QString styleOb = tr("Oblique", "@item font");
604  for (int i = 0; i < 2; ++i) {
605  int pos = selectedStyle.indexOf(styleIt);
606  if (pos >= 0) {
607  QString style = selectedStyle;
608  style.replace(pos, styleIt.length(), styleOb);
609  listPos = filteredStyles.indexOf(style);
610  if (listPos >= 0) {
611  break;
612  }
613  }
614  qSwap(styleIt, styleOb);
615  }
616  }
617  styleListBox->setCurrentRow(listPos >= 0 ? listPos : 0);
618  QString currentStyle = qtStyles[styleListBox->currentItem()->text()];
619 
620  // Recompute the size listbox for this family/style.
621  qreal currentSize = setupSizeListBox(currentFamily, currentStyle);
622  sizeOfFont->setValue(currentSize);
623 
624  selFont = dbase.font(currentFamily, currentStyle, int(currentSize));
625  if (dbase.isSmoothlyScalable(currentFamily, currentStyle) && selFont.pointSize() == floor(currentSize)) {
626  selFont.setPointSizeF(currentSize);
627  }
628  emit q->fontSelected(selFont);
629 
630  signalsAllowed = true;

698 void KFontChooser::Private::_k_size_value_slot(double dval)
699 {
700  if (!signalsAllowed) {
701  return;
702  }
703  signalsAllowed = false;
704 
705  // We compare with qreal, so convert for platforms where qreal != double.
706  qreal val = qreal(dval);
707 
708  QFontDatabase dbase;
709  QString family = qtFamilies[familyListBox->currentItem()->text()];
710  QString style = qtStyles[styleListBox->currentItem()->text()];
711 
712  // Reset current size slot in list if it was customized.
713  if (customSizeRow >= 0 && sizeListBox->currentRow() == customSizeRow) {
714  sizeListBox->item(customSizeRow)->setText(standardSizeAtCustom);
715  customSizeRow = -1;
716  }
717 
718  bool canCustomize = true;
719 
720  // For Qt-bad-sizes workaround: skip this block unconditionally
721  if (!dbase.isSmoothlyScalable(family, style)) {
722  // Bitmap font, allow only discrete sizes.
723  // Determine the nearest in the direction of change.
724  canCustomize = false;
725  int nrows = sizeListBox->count();
726  int row = sizeListBox->currentRow();
727  int nrow;
728  if (val - selFont.pointSizeF() > 0) {
729  for (nrow = row + 1; nrow < nrows; ++nrow)
730  if (QLocale::system().toDouble(sizeListBox->item(nrow)->text()) >= val) {
731  break;
732  }
733  } else {
734  for (nrow = row - 1; nrow >= 0; --nrow)
735  if (QLocale::system().toDouble(sizeListBox->item(nrow)->text()) <= val) {
736  break;
737  }
738  }
739  // Make sure the new row is not out of bounds.
740  nrow = nrow < 0 ? 0 : nrow >= nrows ? nrows - 1 : nrow;
741  // Get the size from the new row and set the spinbox to that size.
742  val = QLocale::system().toDouble(sizeListBox->item(nrow)->text());
743  sizeOfFont->setValue(val);
744  }
745 
746  // Set the current size in the size listbox.
747  int row = nearestSizeRow(val, canCustomize);
748  sizeListBox->setCurrentRow(row);
749 
750  selectedSize = val;
751  selFont.setPointSizeF(val);
752  emit q->fontSelected(selFont);
753 
754  signalsAllowed = true;
755 }

767 int KFontChooser::Private::nearestSizeRow(qreal val, bool customize)
768 {
769  qreal diff = 1000;
770  int row = 0;
771  for (int r = 0; r < sizeListBox->count(); ++r) {
772  qreal cval = QLocale::system().toDouble(sizeListBox->item(r)->text());
773  if (qAbs(cval - val) < diff) {
774  diff = qAbs(cval - val);
775  row = r;
776  }
777  }
778  // For Qt-bad-sizes workaround: ignore value of customize, use true
779  if (customize && diff > 0) {
780  customSizeRow = row;
781  standardSizeAtCustom = sizeListBox->item(row)->text();
782  sizeListBox->item(row)->setText(formatFontSize(val));
783  }
784  return row;
785 }

787 qreal KFontChooser::Private::fillSizeList(const QList<qreal> &sizes_)
788 {
789  if (!sizeListBox) {
790  return 0; //assertion.
791  }
792 
793  QList<qreal> sizes = sizes_;
794  bool canCustomize = false;
795  if (sizes.count() == 0) {
796  static const int c[] = {
797  4, 5, 6, 7,
798  8, 9, 10, 11,
799  12, 13, 14, 15,
800  16, 17, 18, 19,
801  20, 22, 24, 26,
802  28, 32, 48, 64,
803  72, 80, 96, 128,
804  0
805  };
806  for (int i = 0; c[i]; ++i) {
807  sizes.append(c[i]);
808  }
809  // Since sizes were not supplied, this is a vector font,
810  // and size slot customization is allowed.
811  canCustomize = true;
812  }
813 
814  // Insert sizes into the listbox.
815  sizeListBox->clear();
816  qSort(sizes);
817  Q_FOREACH (qreal size, sizes) {
818  sizeListBox->addItem(formatFontSize(size));
819  }
820 
821  // Return the nearest to selected size.
822  // If the font is vector, the nearest size is always same as selected,
823  // thus size slot customization is allowed.
824  // If the font is bitmap, the nearest size need not be same as selected,
825  // thus size slot customization is not allowed.
826  customSizeRow = -1;
827  int row = nearestSizeRow(selectedSize, canCustomize);
828  return QLocale::system().toDouble(sizeListBox->item(row)->text());
829 }

831 qreal KFontChooser::Private::setupSizeListBox(const QString &family, const QString &style)
832 {
833  QFontDatabase dbase;
834  QList<qreal> sizes;
835  if (dbase.isSmoothlyScalable(family, style)) {
836  // A vector font.
837  //>sampleEdit->setPaletteBackgroundPixmap( VectorPixmap ); // TODO
838  } else {
839  // A bitmap font.

842  Q_FOREACH (int size, smoothSizes) {
843  sizes.append(size);

845  }
846

857  }

862 void KFontChooser::Private::setupDisplay()
863 {
864  QFontDatabase dbase;
865  QString family = selFont.family().toLower();
866  QString styleID = styleIdentifier(selFont);
867  qreal size = selFont.pointSizeF();
868  if (size == -1) {
869  size = QFontInfo(selFont).pointSizeF();
870  }
871 
872  int numEntries, i;
873 
874  // Direct family match.
875  numEntries = familyListBox->count();
876  for (i = 0; i < numEntries; ++i) {
877  if (family == qtFamilies[familyListBox->item(i)->text()].toLower()) {
878  familyListBox->setCurrentRow(i);
879  break;
880  }
881  }
882 
883  // 1st family fallback.
884  if (i == numEntries) {
885  if (family.contains(QLatin1Char('['))) {
886  family = family.left(family.indexOf(QLatin1Char('['))).trimmed();
887  for (i = 0; i < numEntries; ++i) {
888  if (family == qtFamilies[familyListBox->item(i)->text()].toLower()) {
889  familyListBox->setCurrentRow(i);
890  break;
891  }
892  }
893  }
894  }
895 
896  // 2nd family fallback.
897  if (i == numEntries) {
898  QString fallback = family + QStringLiteral(" [");
899  for (i = 0; i < numEntries; ++i) {
900  if (qtFamilies[familyListBox->item(i)->text()].toLower().startsWith(fallback)) {
901  familyListBox->setCurrentRow(i);
902  break;
903  }
904  }
905  }
906 
907  // 3rd family fallback.
908  if (i == numEntries) {
909  for (i = 0; i < numEntries; ++i) {
910  if (qtFamilies[familyListBox->item(i)->text()].toLower().startsWith(family)) {
911  familyListBox->setCurrentRow(i);
912  break;
913  }
914  }
915  }
916 
917  // Family fallback in case nothing matched. Otherwise, diff doesn't work
918  if (i == numEntries) {
919  familyListBox->setCurrentRow(0);
920  }
921 
922  // By setting the current item in the family box, the available
923  // styles and sizes for that family have been collected.
924  // Try now to set the current items in the style and size boxes.
925 
926  // Set current style in the listbox.
927  numEntries = styleListBox->count();
928  for (i = 0; i < numEntries; ++i) {
929  if (styleID == styleIDs[styleListBox->item(i)->text()]) {
930  styleListBox->setCurrentRow(i);
931  break;
932  }
933  }
934  if (i == numEntries) {
935  // Style not found, fallback.
936  styleListBox->setCurrentRow(0);
937  }
938 
939  // Set current size in the listbox.
940  // If smoothly scalable, allow customizing one of the standard size slots,
941  // otherwise just select the nearest available size.
942  QString currentFamily = qtFamilies[familyListBox->currentItem()->text()];
943  QString currentStyle = qtStyles[styleListBox->currentItem()->text()];
944  bool canCustomize = dbase.isSmoothlyScalable(currentFamily, currentStyle);
945  sizeListBox->setCurrentRow(nearestSizeRow(size, canCustomize));
946 
947  // Set current size in the spinbox.
948  sizeOfFont->setValue(QLocale::system().toDouble(sizeListBox->currentItem()->text()));
949 }

951 void KFontChooser::getFontList(QStringList &list, uint fontListCriteria)
952 {
953  QFontDatabase dbase;
954  QStringList lstSys(dbase.families());
955 
956  // if we have criteria; then check fonts before adding
957  if (fontListCriteria) {
958  QStringList lstFonts;
959  for (QStringList::const_iterator it = lstSys.constBegin(); it != lstSys.constEnd(); ++it) {
960  if ((fontListCriteria & FixedWidthFonts) > 0 && !dbase.isFixedPitch(*it)) {
961  continue;
962  }
963  if (((fontListCriteria & (SmoothScalableFonts | ScalableFonts)) == ScalableFonts) &&
964  !dbase.isBitmapScalable(*it)) {
965  continue;
966  }
967  if ((fontListCriteria & SmoothScalableFonts) > 0 && !dbase.isSmoothlyScalable(*it)) {
968  continue;
969  }
970  lstFonts.append(*it);
971  }
972 
973  if ((fontListCriteria & FixedWidthFonts) > 0) {
974  // Fallback.. if there are no fixed fonts found, it's probably a
975  // bug in the font server or Qt. In this case, just use 'fixed'
976  if (lstFonts.count() == 0) {
977  lstFonts.append(QStringLiteral("fixed"));
978  }
979  }
980 
981  lstSys = lstFonts;
982  }
983 
984  lstSys.sort();
985 
986  list = lstSys;
987 }
