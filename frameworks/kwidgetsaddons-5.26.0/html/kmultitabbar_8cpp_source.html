317 void KMultiTabBarTab::computeMargins(int *hMargin, int *vMargin) const
318 {
319  // Unfortunately, QStyle does not give us enough information to figure out
320  // where to place things, so we try to reverse-engineer it
321  QStyleOptionToolButton opt;
322  initStyleOption(&opt);
323 
324  QPixmap iconPix = iconPixmap();
325  QSize trialSize = iconPix.size() / iconPix.devicePixelRatio();
326  QSize expandSize = style()->sizeFromContents(QStyle::CT_ToolButton, &opt, trialSize, this);
327 
328  *hMargin = (expandSize.width() - trialSize.width()) / 2;
329  *vMargin = (expandSize.height() - trialSize.height()) / 2;
330 }

332 QSize KMultiTabBarTab::computeSizeHint(bool withText) const
333 {
334  // Compute as horizontal first, then flip around if need be.
335  QStyleOptionToolButton opt;
336  initStyleOption(&opt);
337 
338  int hMargin, vMargin;
339  computeMargins(&hMargin, &vMargin);
340 
341  // Compute interior size, starting from pixmap..
342  QPixmap iconPix = iconPixmap();
343  QSize size = iconPix.size() / iconPix.devicePixelRatio();
344 
345  // Always include text height in computation, to avoid resizing the minor direction
346  // when expanding text..
347  QSize textSize = fontMetrics().size(0, text());
348  size.setHeight(qMax(size.height(), textSize.height()));
349 
350  // Pick margins for major/minor direction, depending on orientation
351  int majorMargin = isVertical() ? vMargin : hMargin;
352  int minorMargin = isVertical() ? hMargin : vMargin;
353 
354  size.setWidth(size.width() + 2 * majorMargin);
355  size.setHeight(size.height() + 2 * minorMargin);
356 
357  if (withText)
358  // Add enough room for the text, and an extra major margin.
359  {
360  size.setWidth(size.width() + textSize.width() + majorMargin);
361  }
362 
363  // flip time?
364  if (isVertical()) {
365  return QSize(size.height(), size.width());
366  } else {
367  return size;
368  }
369 }

399 void KMultiTabBarTab::paintEvent(QPaintEvent *)
400 {
401  QPainter painter(this);
402 
403  QStyleOptionToolButton opt;
404  initStyleOption(&opt);
405 
406  // Paint bevel..
407  if (underMouse() || isChecked()) {
408  opt.text.clear();
409  opt.icon = QIcon();
410  style()->drawComplexControl(QStyle::CC_ToolButton, &opt, &painter, this);
411  }
412 
413  int hMargin, vMargin;
414  computeMargins(&hMargin, &vMargin);
415 
416  // We first figure out how much room we have for the text, based on
417  // icon size and margin, try to fit in by eliding, and perhaps
418  // give up on drawing the text entirely if we're too short on room
419  QPixmap icon = iconPixmap();
420  int textRoom = 0;
421  int iconRoom = 0;
422 
423  QString t;
424  if (shouldDrawText()) {
425  if (isVertical()) {
426  iconRoom = icon.height() / icon.devicePixelRatio() + 2 * vMargin;
427  textRoom = height() - iconRoom - vMargin;
428  } else {
429  iconRoom = icon.width() / icon.devicePixelRatio() + 2 * hMargin;
430  textRoom = width() - iconRoom - hMargin;
431  }
432 
433  t = painter.fontMetrics().elidedText(text(), Qt::ElideRight, textRoom);
434 
435  // See whether anything is left. Qt will return either
436  // ... or the ellipsis unicode character, 0x2026
437  if (t == QLatin1String("...") || t == QChar(0x2026)) {
438  t.clear();
439  }
440  }
441 
442  // Label time.... Simple case: no text, so just plop down the icon right in the center
443  // We only do this when the button never draws the text, to avoid jumps in icon position
444  // when resizing
445  if (!shouldDrawText()) {
446  style()->drawItemPixmap(&painter, rect(), Qt::AlignCenter | Qt::AlignVCenter, icon);
447  return;
448  }
449 
450  // Now where the icon/text goes depends on text direction and tab position
451  QRect iconArea;
452  QRect labelArea;
453 
454  bool bottomIcon = false;
455  bool rtl = layoutDirection() == Qt::RightToLeft;
456  if (isVertical()) {
457  if (m_position == KMultiTabBar::Left && !rtl) {
458  bottomIcon = true;
459  }
460  if (m_position == KMultiTabBar::Right && rtl) {
461  bottomIcon = true;
462  }
463  }
464  //alignFlags = Qt::AlignLeading | Qt::AlignVCenter;
465 
466  if (isVertical()) {
467  if (bottomIcon) {
468  labelArea = QRect(0, vMargin, width(), textRoom);
469  iconArea = QRect(0, vMargin + textRoom, width(), iconRoom);
470  } else {
471  labelArea = QRect(0, iconRoom, width(), textRoom);
472  iconArea = QRect(0, 0, width(), iconRoom);
473  }
474  } else {
475  // Pretty simple --- depends only on RTL/LTR
476  if (rtl) {
477  labelArea = QRect(hMargin, 0, textRoom, height());
478  iconArea = QRect(hMargin + textRoom, 0, iconRoom, height());
479  } else {
480  labelArea = QRect(iconRoom, 0, textRoom, height());
481  iconArea = QRect(0, 0, iconRoom, height());
482  }
483  }
484 
485  style()->drawItemPixmap(&painter, iconArea, Qt::AlignCenter | Qt::AlignVCenter, icon);
486 
487  if (t.isEmpty()) {
488  return;
489  }
490 
491  QRect labelPaintArea = labelArea;
492 
493  if (isVertical()) {
494  // If we're vertical, we paint to a simple 0,0 origin rect,
495  // and get the transformations to get us in the right place
496  labelPaintArea = QRect(0, 0, labelArea.height(), labelArea.width());
497 
498  QTransform tr;
499 
500  if (bottomIcon) {
501  tr.translate(labelArea.x(), labelPaintArea.width() + labelArea.y());
502  tr.rotate(-90);
503  } else {
504  tr.translate(labelPaintArea.height() + labelArea.x(), labelArea.y());
505  tr.rotate(90);
506  }
507  painter.setTransform(tr);
508  }
509 
510  opt.text = t;
511  opt.icon = QIcon();
512  opt.rect = labelPaintArea;
513  style()->drawControl(QStyle::CE_ToolButtonLabel, &opt, &painter, this);
514 }
