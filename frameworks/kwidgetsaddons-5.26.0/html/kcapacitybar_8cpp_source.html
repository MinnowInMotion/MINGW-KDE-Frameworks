180 void KCapacityBar::drawCapacityBar(QPainter *p, const QRect &rect) const
181 {
182  if (d->ce_capacityBar) {
183  QStyleOptionProgressBar opt;
184  opt.initFrom(this);
185  opt.rect = rect;
186  opt.minimum = 0;
187  opt.maximum = 100;
188  opt.progress = d->value;
189  opt.text = d->text;
190  opt.textAlignment = Qt::AlignCenter;
191  opt.textVisible = true;
192  style()->drawControl(d->ce_capacityBar, &opt, p, this);
193 
194  return;
195  }
196 
197  p->setRenderHints(QPainter::Antialiasing | QPainter::TextAntialiasing);
198 
199  p->save();
200 
201  QRect drawRect(rect);
202 
203  if (d->drawTextMode == DrawTextOutline) {
204  drawRect.setHeight(d->barHeight);
205  }
206 
207  QPainterPath outline;
208  outline.moveTo(rect.left() + ROUND_MARGIN / 4 + 1, rect.top());
209  outline.lineTo(rect.left() + drawRect.width() - ROUND_MARGIN / 4 - 1, rect.top());
210  outline.quadTo(rect.left() + drawRect.width() + ROUND_MARGIN / 2, drawRect.height() / 2 + rect.top(), rect.left() + drawRect.width() - ROUND_MARGIN / 4 - 1, drawRect.height() + rect.top());
211  outline.lineTo(rect.left() + ROUND_MARGIN / 4 + 1, drawRect.height() + rect.top());
212  outline.quadTo(-ROUND_MARGIN / 2 + rect.left(), drawRect.height() / 2 + rect.top(), rect.left() + ROUND_MARGIN / 4 + 1, rect.top());
213  const QColor fillColor = palette().window().color().darker(DarkShade);
214  p->fillPath(outline, QColor(fillColor.red(), fillColor.green(), fillColor.blue(), 50));
215 
216  QRadialGradient bottomGradient(QPointF(rect.width() / 2, drawRect.bottom() + 1), rect.width() / 2);
217  bottomGradient.setColorAt(0, palette().window().color().darker(LightShade));
218  bottomGradient.setColorAt(1, Qt::transparent);
219  p->fillRect(QRect(rect.left(), drawRect.bottom() + rect.top(), rect.width(), 1), bottomGradient);
220 
221  p->translate(rect.left() + 2, rect.top() + 1);
222 
223  drawRect.setWidth(drawRect.width() - 4);
224  drawRect.setHeight(drawRect.height() - 2);
225 
226  QPainterPath path;
227  path.moveTo(ROUND_MARGIN / 4, 0);
228  path.lineTo(drawRect.width() - ROUND_MARGIN / 4, 0);
229  path.quadTo(drawRect.width() + ROUND_MARGIN / 2, drawRect.height() / 2, drawRect.width() - ROUND_MARGIN / 4, drawRect.height());
230  path.lineTo(ROUND_MARGIN / 4, drawRect.height());
231  path.quadTo(-ROUND_MARGIN / 2, drawRect.height() / 2, ROUND_MARGIN / 4, 0);
232 
233  QLinearGradient linearGradient(0, 0, 0, drawRect.height());
234  linearGradient.setColorAt(0.5, palette().window().color().darker(MidShade));
235  linearGradient.setColorAt(1, palette().window().color().darker(LightShade));
236  p->fillPath(path, linearGradient);
237 
238  p->setBrush(Qt::NoBrush);
239  p->setPen(Qt::NoPen);
240 
241  if (d->continuous || !d->fillFullBlocks) {
242  int start = (layoutDirection() == Qt::LeftToRight) ? -1
243  : (drawRect.width() + 2) - (drawRect.width() + 2) * (d->value / 100.0);
244 
245  p->setClipRect(QRect(start, 0, (drawRect.width() + 2) * (d->value / 100.0), drawRect.height()), Qt::IntersectClip);
246  }
247 
248  int left = (layoutDirection() == Qt::LeftToRight) ? 0
249  : drawRect.width();
250 
251  int right = (layoutDirection() == Qt::LeftToRight) ? drawRect.width()
252  : 0;
253 
254  int roundMargin = (layoutDirection() == Qt::LeftToRight) ? ROUND_MARGIN
255  : -ROUND_MARGIN;
256 
257  int spacing = 2;
258  int verticalSpacing = VERTICAL_SPACING;
259  int slotWidth = 6;
260  int start = roundMargin / 4;
261 
262  QPainterPath internalBar;
263  internalBar.moveTo(left + roundMargin / 4, 0);
264  internalBar.lineTo(right - roundMargin / 4, 0);
265  internalBar.quadTo(right + roundMargin / 2, drawRect.height() / 2, right - roundMargin / 4, drawRect.height());
266  internalBar.lineTo(left + roundMargin / 4, drawRect.height());
267  internalBar.quadTo(left - roundMargin / 2, drawRect.height() / 2, left + roundMargin / 4, 0);
268 
269  QLinearGradient fillInternalBar(left, 0, right, 0);
270  fillInternalBar.setColorAt(0, palette().window().color().darker(MidShade));
271  fillInternalBar.setColorAt(0.5, palette().window().color().darker(LightShade));
272  fillInternalBar.setColorAt(1, palette().window().color().darker(MidShade));
273 
274  if (d->drawTextMode == KCapacityBar::DrawTextInline) {
275  p->save();
276  p->setOpacity(p->opacity() * 0.7);
277  }
278 
279  if (!d->continuous) {
280  int numSlots = (drawRect.width() - ROUND_MARGIN - ((slotWidth + spacing) * 2)) / (slotWidth + spacing);
281  int stopSlot = floor((numSlots + 2) * (d->value / 100.0));
282 
283  int plusOffset = d->fillFullBlocks ? ((drawRect.width() - ROUND_MARGIN - ((slotWidth + spacing) * 2)) - (numSlots * (slotWidth + spacing))) / 2.0
284  : 0;
285 
286  if (!d->fillFullBlocks || stopSlot) {
287  QPainterPath firstSlot;
288  firstSlot.moveTo(left + roundMargin / 4, verticalSpacing);
289  firstSlot.lineTo(left + slotWidth + roundMargin / 4 + plusOffset, verticalSpacing);
290  firstSlot.lineTo(left + slotWidth + roundMargin / 4 + plusOffset, drawRect.height() - verticalSpacing);
291  firstSlot.lineTo(left + roundMargin / 4, drawRect.height() - verticalSpacing);
292  firstSlot.quadTo(left, drawRect.height() / 2, left + roundMargin / 4, verticalSpacing);
293  p->fillPath(firstSlot, fillInternalBar);
294  start += slotWidth + spacing + plusOffset;
295 
296  bool stopped = false;
297  for (int i = 0; i < numSlots + 1; i++) {
298  if (d->fillFullBlocks && (i == (stopSlot + 1))) {
299  stopped = true;
300  break;
301  }
302  p->fillRect(QRect(rect.left() + start, rect.top() + verticalSpacing, slotWidth, drawRect.height() - verticalSpacing * 2), fillInternalBar);
303  start += slotWidth + spacing;
304  }
305 
306  if (!d->fillFullBlocks || (!stopped && (stopSlot != (numSlots + 1)) && (stopSlot != numSlots))) {
307  QPainterPath lastSlot;
308  lastSlot.moveTo(start, verticalSpacing);
309  lastSlot.lineTo(start, drawRect.height() - verticalSpacing);
310  lastSlot.lineTo(start + slotWidth + plusOffset, drawRect.height() - verticalSpacing);
311  lastSlot.quadTo(start + roundMargin, drawRect.height() / 2, start + slotWidth + plusOffset, verticalSpacing);
312  lastSlot.lineTo(start, verticalSpacing);
313  p->fillPath(lastSlot, fillInternalBar);
314  }
315  }
316  } else {
317  p->fillPath(internalBar, fillInternalBar);
318  }
319 
320  if (d->drawTextMode == KCapacityBar::DrawTextInline) {
321  p->restore();
322  }
323 
324  p->save();
325  p->setClipping(false);
326  QRadialGradient topGradient(QPointF(rect.width() / 2, drawRect.top()), rect.width() / 2);
327  const QColor fillTopColor = palette().window().color().darker(LightShade);
328  topGradient.setColorAt(0, QColor(fillTopColor.red(), fillTopColor.green(), fillTopColor.blue(), 127));
329  topGradient.setColorAt(1, Qt::transparent);
330  p->fillRect(QRect(rect.left(), rect.top() + drawRect.top(), rect.width(), 2), topGradient);
331  p->restore();
332 
333  p->save();
334  p->setClipRect(QRect(-1, 0, rect.width(), drawRect.height() / 2), Qt::ReplaceClip);
335  QLinearGradient glassGradient(0, -5, 0, drawRect.height());
336  const QColor fillGlassColor = palette().base().color();
337  glassGradient.setColorAt(0, QColor(fillGlassColor.red(), fillGlassColor.green(), fillGlassColor.blue(), 255));
338  glassGradient.setColorAt(1, Qt::transparent);
339  p->fillPath(internalBar, glassGradient);
340  p->restore();
341 
342  p->restore();
343 
344  if (d->drawTextMode == KCapacityBar::DrawTextInline) {
345  QRect rect(drawRect);
346  rect.setHeight(rect.height() + 4);
347  p->drawText(rect, Qt::AlignCenter, fontMetrics().elidedText(d->text, Qt::ElideRight, drawRect.width() - 2 * ROUND_MARGIN));
348  } else {
349  p->drawText(rect, Qt::AlignBottom | d->horizontalTextAlignment, fontMetrics().elidedText(d->text, Qt::ElideRight, drawRect.width()));
350  }
351 }

353 QSize KCapacityBar::minimumSizeHint() const
354 {
355  int width = (d->drawTextMode == KCapacityBar::DrawTextInline) ?
356  fontMetrics().width(d->text) + ROUND_MARGIN * 2 :
357  fontMetrics().width(d->text);
358 
359  int height = (d->drawTextMode == KCapacityBar::DrawTextInline) ?
360  qMax(fontMetrics().height(), d->barHeight) :
361  (d->text.isEmpty() ? 0 : fontMetrics().height() + VERTICAL_SPACING * 2) + d->barHeight;
362 
363  if (height % 2) {
364  height++;
365  }
366 
367  return QSize(width, height);
368 }
