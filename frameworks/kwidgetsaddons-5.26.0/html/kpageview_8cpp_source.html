37 void KPageViewPrivate::_k_rebuildGui()
38 {
39  // clean up old view
40  Q_Q(KPageView);
41 
42  QModelIndex currentLastIndex;
43  if (view && view->selectionModel()) {
44  QObject::disconnect(view->selectionModel(), SIGNAL(selectionChanged(QItemSelection,QItemSelection)),
45  q, SLOT(_k_pageSelected(QItemSelection,QItemSelection)));
46  currentLastIndex = view->selectionModel()->currentIndex();
47  }
48 
49  delete view;
50  view = q->createView();
51 
52  Q_ASSERT(view);
53 
54  view->setSelectionBehavior(QAbstractItemView::SelectItems);
55  view->setSelectionMode(QAbstractItemView::SingleSelection);
56 
57  if (model) {
58  view->setModel(model);
59  }
60 
61  // setup new view
62  if (view->selectionModel()) {
63  QObject::connect(view->selectionModel(), SIGNAL(selectionChanged(QItemSelection,QItemSelection)), q, SLOT(_k_pageSelected(QItemSelection,QItemSelection)));
64 
65  if (currentLastIndex.isValid()) {
66  view->selectionModel()->setCurrentIndex(currentLastIndex, QItemSelectionModel::Select);
67  } else if (model) {
68  view->selectionModel()->setCurrentIndex(model->index(0, 0), QItemSelectionModel::Select);
69  }
70  }
71 
72  if (faceType == KPageView::Tabbed) {
73  stack->setVisible(false);
74  layout->removeWidget(stack);
75  } else {
76  layout->addWidget(stack, 2, 1);
77  stack->setVisible(true);
78  }
79 
80  titleWidget->setVisible(q->showPageHeader());
81 
82  Qt::Alignment alignment = q->viewPosition();
83  if (alignment & Qt::AlignTop) {
84  layout->addWidget(view, 2, 1);
85  } else if (alignment & Qt::AlignRight) {
86  layout->addWidget(view, 1, 2, 2, 1);
87  } else if (alignment & Qt::AlignBottom) {
88  layout->addWidget(view, 4, 1);
89  } else if (alignment & Qt::AlignLeft) {
90  layout->addWidget(view, 1, 0, 2, 1);
91  }
92 }

114 void KPageViewPrivate::cleanupPages()
115 {
120  const QList<QWidget *> widgets = collectPages();
121 
122  for (int i = 0; i < stack->count(); ++i) {
123  QWidget *page = stack->widget(i);
124 
125  bool found = false;
126  for (int j = 0; j < widgets.count(); ++j) {
127  if (widgets[ j ] == page) {
128  found = true;
129  }
130  }
131 
132  if (!found) {
133  stack->removeWidget(page);
134  }
135  }
136 }

138 QList<QWidget *> KPageViewPrivate::collectPages(const QModelIndex &parentIndex)
139 {
144  QList<QWidget *> retval;
145 
146  int rows = model->rowCount(parentIndex);
147  for (int j = 0; j < rows; ++j) {
148  const QModelIndex index = model->index(j, 0, parentIndex);
149  retval.append(qvariant_cast<QWidget *>(model->data(index, KPageModel::WidgetRole)));
150 
151  if (model->rowCount(index) > 0) {
152  retval += collectPages(index);
153  }
154  }
155 
156  return retval;
157 }

159 KPageView::FaceType KPageViewPrivate::detectAutoFace() const
160 {
161  if (!model) {
162  return KPageView::Plain;
163  }
164 
168  bool hasSubPages = false;
169  const int count = model->rowCount();
170  for (int i = 0; i < count; ++i) {
171  if (model->rowCount(model->index(i, 0)) > 0) {
172  hasSubPages = true;
173  break;
174  }
175  }
176 
177  if (hasSubPages) {
178  return KPageView::Tree;
179  }
180 
181  if (model->rowCount() > 1) {
182  return KPageView::List;
183  }
184 
185  return KPageView::Plain;
186 }

188 void KPageViewPrivate::_k_modelChanged()
189 {
190  if (!model) {
191  return;
192  }
193 
198  if (faceType == KPageView::Auto) {
199  _k_rebuildGui();
200  // If you discover some crashes use the line below instead...
201  //QTimer::singleShot(0, q, SLOT(_k_rebuildGui()));
202  }
203 
207  QSize size = stack->size();
208  const QList<QWidget *> widgets = collectPages();
209  for (int i = 0; i < widgets.count(); ++i) {
210  const QWidget *widget = widgets[ i ];
211  if (widget) {
212  size = size.expandedTo(widget->minimumSizeHint());
213  }
214  }
215  stack->setMinimumSize(size);
216 
217  updateSelection();
218 }

220 void KPageViewPrivate::_k_pageSelected(const QItemSelection &index, const QItemSelection &previous)
221 {
222  if (!model) {
223  return;
224  }
225 
226  // Return if the current Index is not valid
227  if (index.indexes().size() != 1) {
228  return;
229  }
230  QModelIndex currentIndex = index.indexes().first();
231 
232  QModelIndex previousIndex;
233  // The previous index can be invalid
234  if (previous.indexes().size() == 1) {
235  previousIndex = previous.indexes().first();
236  }
237 
238  if (faceType != KPageView::Tabbed) {
239  QWidget *widget = qvariant_cast<QWidget *>(model->data(currentIndex, KPageModel::WidgetRole));
240 
241  if (widget) {
242  if (stack->indexOf(widget) == -1) { // not included yet
243  stack->addWidget(widget);
244  }
245 
246  stack->setCurrentWidget(widget);
247  } else {
248  stack->setCurrentWidget(defaultWidget);
249  }
250 
251  updateTitleWidget(currentIndex);
252  }
253 
254  Q_Q(KPageView);
255  emit q->currentPageChanged(currentIndex, previousIndex);
256 }

342  disconnect(d->model, SIGNAL(dataChanged(QModelIndex,QModelIndex)),
343  this, SLOT(_k_dataChanged(QModelIndex,QModelIndex)));

345 
346  d->model = model;
347 
348  if (d->model) {

356  }

439 QAbstractItemView *KPageView::createView()
440 {
441  Q_D(KPageView);
442  if (d->faceType == Auto) {
443  const FaceType faceType = d->detectAutoFace();
444 
445  if (faceType == Plain) {
446  return new KDEPrivate::KPagePlainView(this);
447  } else if (faceType == List) {
448  return new KDEPrivate::KPageListView(this);
449  } else if (faceType == Tree) {
450  return new KDEPrivate::KPageTreeView(this);
451  } else { // should never happen
452  return 0;
453  }
454  } else if (d->faceType == Plain) {
455  return new KDEPrivate::KPagePlainView(this);
456  } else if (d->faceType == List) {
457  return new KDEPrivate::KPageListView(this);
458  } else if (d->faceType == Tree) {
459  return new KDEPrivate::KPageTreeView(this);
460  } else if (d->faceType == Tabbed) {
461  return new KDEPrivate::KPageTabbedView(this);
462  } else {
463  return 0;
464  }
