275 QVariant KPageWidgetModel::data(const QModelIndex &index, int role) const
276 {
277  if (!index.isValid()) {
278  return QVariant();
279  }
280 
281  PageItem *item = static_cast<PageItem *>(index.internalPointer());
282 
283  if (role == Qt::DisplayRole) {
284  return QVariant(item->pageWidgetItem()->name());
285  } else if (role == Qt::DecorationRole) {
286  return QVariant(item->pageWidgetItem()->icon());
287  } else if (role == HeaderRole) {
288  return QVariant(item->pageWidgetItem()->header());
289  } else if (role == WidgetRole) {
290  return QVariant::fromValue(item->pageWidgetItem()->widget());
291  } else if (role == Qt::CheckStateRole) {
292  if (item->pageWidgetItem()->isCheckable()) {
293  return (item->pageWidgetItem()->isChecked() ? Qt::Checked : Qt::Unchecked);
294  } else {
295  return QVariant();
296  }
297  } else {
298  return QVariant();
299  }
300 }

435 void KPageWidgetModel::insertPage(KPageWidgetItem *before, KPageWidgetItem *item)
436 {
437  PageItem *beforePageItem = d_func()->rootItem->findChild(before);
438  if (!beforePageItem) {
439  qDebug("Invalid KPageWidgetItem passed!");
440  return;
441  }
442 
443  emit layoutAboutToBeChanged();
444 
445  connect(item, SIGNAL(changed()), this, SLOT(_k_itemChanged()));
446  connect(item, SIGNAL(toggled(bool)), this, SLOT(_k_itemToggled(bool)));
447 
448  PageItem *parent = beforePageItem->parent();
449  // The row to be inserted
450  int row = beforePageItem->row();
451 
452  QModelIndex index;
453  if (parent != d_func()->rootItem) {
454  index = createIndex(parent->row(), 0, parent);
455  }
456 
457  beginInsertRows(index, row, row);
458 
459  PageItem *newPageItem = new PageItem(item, parent);
460  parent->insertChild(row, newPageItem);
461 
462  endInsertRows();
463 
464  emit layoutChanged();
465 }

507 void KPageWidgetModel::removePage(KPageWidgetItem *item)
508 {
509  if (!item) {
510  return;
511  }
512 
513  Q_D(KPageWidgetModel);
514 
515  PageItem *pageItem = d->rootItem->findChild(item);
516  if (!pageItem) {
517  qDebug("Invalid KPageWidgetItem passed!");
518  return;
519  }
520 
521  emit layoutAboutToBeChanged();
522 
523  disconnect(item, SIGNAL(changed()), this, SLOT(_k_itemChanged()));
524  disconnect(item, SIGNAL(toggled(bool)), this, SLOT(_k_itemToggled(bool)));
525 
526  PageItem *parentPageItem = pageItem->parent();
527  int row = parentPageItem->row();
528 
529  QModelIndex index;
530  if (parentPageItem != d->rootItem) {
531  index = createIndex(row, 0, parentPageItem);
532  }
533 
534  beginRemoveRows(index, pageItem->row(), pageItem->row());
535 
536  parentPageItem->removeChild(pageItem->row());
537  delete pageItem;
538 
539  endRemoveRows();
540 
541  emit layoutChanged();
542 }
