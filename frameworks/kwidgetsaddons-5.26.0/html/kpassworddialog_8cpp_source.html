242 void KPasswordDialog::addCommentLine(const QString &label,
243  const QString &comment)
244 {
245  int gridMarginLeft, gridMarginTop, gridMarginRight, gridMarginBottom;
246  d->ui.formLayout->getContentsMargins(&gridMarginLeft, &gridMarginTop, &gridMarginRight, &gridMarginBottom);
247 
248  int spacing = d->ui.formLayout->horizontalSpacing();
249  if (spacing < 0) {
250  // same inter-column spacing for all rows, see comment in qformlayout.cpp
251  spacing = style()->combinedLayoutSpacing(QSizePolicy::Label, QSizePolicy::LineEdit, Qt::Horizontal, 0, this);
252  }
253 
254  QLabel *c = new QLabel(comment, this);
255  c->setWordWrap(true);
256 
257  d->ui.formLayout->insertRow(d->commentRow, label, c);
258  ++d->commentRow;
259 
260  // cycle through column 0 widgets and see the max width so we can set the minimum height of
261  // column 2 wordwrapable labels
262  int firstColumnWidth = 0;
263  for (int i = 0; i < d->ui.formLayout->rowCount(); ++i) {
264  QLayoutItem *li = d->ui.formLayout->itemAt(i, QFormLayout::LabelRole);
265  if (li) {
266  QWidget *w = li->widget();
267  if (w && !w->isHidden()) {
268  firstColumnWidth = qMax(firstColumnWidth, w->sizeHint().width());
269  }
270  }
271  }
272  for (int i = 0; i < d->ui.formLayout->rowCount(); ++i) {
273  QLayoutItem *li = d->ui.formLayout->itemAt(i, QFormLayout::FieldRole);
274  if (li) {
275  QLabel *l = qobject_cast<QLabel *>(li->widget());
276  if (l && l->wordWrap()) {
277  const int marginHint = style()->pixelMetric(QStyle::PM_DefaultChildMargin);
278  int w = sizeHint().width() - firstColumnWidth - (2 * marginHint) - gridMarginLeft - gridMarginRight - spacing;
279  l->setMinimumSize(w, l->heightForWidth(w));
280  }
281  }
282  }
283 }

285 void KPasswordDialog::showErrorMessage(const QString &message, const ErrorType type)
286 {
287  d->ui.errorMessage->setText(message, KTitleWidget::ErrorMessage);
288 
289  QFont bold = font();
290  bold.setBold(true);
291  switch (type) {
292  case PasswordError:
293  d->ui.passwordLabel->setFont(bold);
294  d->ui.passEdit->clear();
295  d->ui.passEdit->setFocus();
296  break;
297  case UsernameError:
298  if (d->ui.userEdit->isVisibleTo(this)) {
299  d->ui.userNameLabel->setFont(bold);
300  d->ui.userEdit->setFocus();
301  }
302  break;
303  case DomainError:
304  if (d->ui.domainEdit->isVisibleTo(this)) {
305  d->ui.domainLabel->setFont(bold);
306  d->ui.domainEdit->setFocus();
307  }
308  break;
309  case FatalError:
310  d->ui.userNameLabel->setEnabled(false);
311  d->ui.userEdit->setEnabled(false);
312  d->ui.passwordLabel->setEnabled(false);
313  d->ui.passEdit->setEnabled(false);
314  d->ui.keepCheckBox->setEnabled(false);
315  d->ui.buttonBox->button(QDialogButtonBox::Ok)->setEnabled(false);
316  break;
317  default:
318  break;
319  }
320  adjustSize();
321 }

351 void KPasswordDialog::setKnownLogins(const QMap<QString, QString> &knownLogins)
352 {
353  const int nr = knownLogins.count();
354  if (nr == 0) {
355  return;
356  }
357 
358  if (nr == 1) {
359  d->ui.userEdit->setText(knownLogins.begin().key());
360  setPassword(knownLogins.begin().value());
361  return;
362  }
363 
364  Q_ASSERT(!d->ui.userEdit->isReadOnly());
365  if (!d->userEditCombo) {
366  int row = -1;
367  QFormLayout::ItemRole userEditRole = QFormLayout::FieldRole;
368 
369  d->ui.formLayout->getWidgetPosition(d->ui.userEdit, &row, &userEditRole);
370  d->ui.formLayout->removeWidget(d->ui.userEdit);
371  delete d->ui.userEdit;
372  d->userEditCombo = new QComboBox(d->ui.credentialsGroup);
373  d->userEditCombo->setEditable(true);
374  d->ui.userEdit = d->userEditCombo->lineEdit();
375  d->ui.userNameLabel->setBuddy(d->userEditCombo);
376  d->ui.formLayout->setWidget(row > -1 ? row : 0, userEditRole, d->userEditCombo);
377 
378  setTabOrder(d->ui.userEdit, d->ui.anonymousRadioButton);
379  setTabOrder(d->ui.anonymousRadioButton, d->ui.domainEdit);
380  setTabOrder(d->ui.domainEdit, d->ui.passEdit);
381  setTabOrder(d->ui.passEdit, d->ui.keepCheckBox);
382  connect(d->ui.userEdit, SIGNAL(returnPressed()), d->ui.passEdit, SLOT(setFocus()));
383  }
384 
385  d->knownLogins = knownLogins;
386  d->userEditCombo->addItems(knownLogins.keys());
387  d->userEditCombo->setFocus();
388 
389  connect(d->userEditCombo, SIGNAL(activated(QString)),
390  this, SLOT(activated(QString)));
391 }

401 void KPasswordDialog::accept()
402 {
403  if (!d->ui.errorMessage->isHidden()) {
404  d->ui.errorMessage->setText(QString());
405  }
406 
407  // reset the font in case we had an error previously
408  if (!d->ui.passwordLabel->isHidden()) {
409  d->ui.passwordLabel->setFont(font());
410  d->ui.userNameLabel->setFont(font());
411  }
412 
413  // we do this to allow the error message, if any, to go away
414  // checkPassword() may block for a period of time
415  QTimer::singleShot(0, this, SLOT(actuallyAccept()));
416 }

418 void KPasswordDialog::KPasswordDialogPrivate::actuallyAccept()
419 {
420  if (!q->checkPassword()) {
421  return;
422  }
423 
424  bool keep = ui.keepCheckBox->isVisibleTo(q) && ui.keepCheckBox->isChecked();
425  emit q->gotPassword(q->password(), keep);
426 
427  if (ui.userEdit->isVisibleTo(q)) {
428  emit q->gotUsernameAndPassword(q->username(), q->password(), keep);
429  }
430 
431  q->QDialog::accept();
432 }
