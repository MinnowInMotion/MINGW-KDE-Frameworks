181 static inline int adjustedHoverRating(bool halfStep, int hoverRating, int rating)
182 {
183  // intentionally skip zero, or half step is disabled.
184  if (!halfStep || hoverRating == 0) {
185  return hoverRating;
186  }
187 
188  // See bug 171343, if we click on a star we want it to be full star, click again
189  // make it half, click third time make it clear.
190 
191  // round up hoverRating to next full star rating
192  const int hoveredFullStarRating = hoverRating + (hoverRating % 2);
193  // Check if the star under mouse is the last half or whole star of the rating.
194  if (hoveredFullStarRating == rating || hoveredFullStarRating == rating + 1) {
195  // If current pointed star is not empty, next rating will be rating - 1
196  // if we point at 4th star and rating is 8 (4 star), next click will make it 7
197  // if we point at 4th star and rating is 7 (3.5 star), next click will make it 6
198  hoverRating = rating - 1;
199  } else {
200  // otherwise make it a full star rating
201  hoverRating = hoveredFullStarRating;
202  }
203  return hoverRating;
204 }

206 void KRatingWidget::mousePressEvent(QMouseEvent *e)
207 {
208  if (e->button() == Qt::LeftButton) {
209  d->hoverRating = adjustedHoverRating(halfStepsEnabled(),
210  d->ratingPainter.ratingFromPosition(contentsRect(), e->pos()),
211  d->rating);
212  // avoid set a rating to something less than zero, it may happen if widget is scaled and
213  // mouse is clicked outside the star region.
214  if (d->hoverRating >= 0) {
215  setRating(d->hoverRating);
216  }
217  }
218 }

220 void KRatingWidget::mouseMoveEvent(QMouseEvent *e)
221 {
222  // when moving the mouse we show the user what the result of clicking will be
223  const int prevHoverRating = d->hoverRating;
224  d->hoverRating = adjustedHoverRating(halfStepsEnabled(),
225  d->ratingPainter.ratingFromPosition(contentsRect(), e->pos()),
226  d->rating);
227  if (d->hoverRating != prevHoverRating) {
228  update();
229  }
230 }

246 QSize KRatingWidget::sizeHint() const
247 {
248  int numPix = d->ratingPainter.maxRating();
249  if (d->ratingPainter.halfStepsEnabled()) {
250  numPix /= 2;
251  }
252 
253  QSize pixSize(d->pixSize, d->pixSize);
254  if (!d->ratingPainter.customPixmap().isNull()) {
255  pixSize = d->ratingPainter.customPixmap().size() / d->ratingPainter.customPixmap().devicePixelRatio();
256  }
257 
258  return QSize(pixSize.width() * numPix + spacing() * (numPix - 1) + frameWidth() * 2,
259  pixSize.height() + frameWidth() * 2);
260 }
