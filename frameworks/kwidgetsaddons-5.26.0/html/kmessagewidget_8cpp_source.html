64 void KMessageWidgetPrivate::init(KMessageWidget *q_ptr)
65 {
66  q = q_ptr;
67 
68  q->setSizePolicy(QSizePolicy::Minimum, QSizePolicy::Fixed);
69 
70  timeLine = new QTimeLine(500, q);
71  QObject::connect(timeLine, SIGNAL(valueChanged(qreal)), q, SLOT(slotTimeLineChanged(qreal)));
72  QObject::connect(timeLine, SIGNAL(finished()), q, SLOT(slotTimeLineFinished()));
73 
74  content = new QFrame(q);
75  content->setSizePolicy(QSizePolicy::Expanding, QSizePolicy::Fixed);
76 
77  wordWrap = false;
78 
79  iconLabel = new QLabel(content);
80  iconLabel->setSizePolicy(QSizePolicy::Fixed, QSizePolicy::Fixed);
81  iconLabel->hide();
82 
83  textLabel = new QLabel(content);
84  textLabel->setSizePolicy(QSizePolicy::Expanding, QSizePolicy::Fixed);
85  textLabel->setTextInteractionFlags(Qt::TextBrowserInteraction);
86  QObject::connect(textLabel, &QLabel::linkActivated, q, &KMessageWidget::linkActivated);
87  QObject::connect(textLabel, &QLabel::linkHovered, q, &KMessageWidget::linkHovered);
88 
89  QAction *closeAction = new QAction(q);
90  closeAction->setText(KMessageWidget::tr("&Close"));
91  closeAction->setToolTip(KMessageWidget::tr("Close message"));
92  closeAction->setIcon(q->style()->standardIcon(QStyle::SP_DialogCloseButton));
93 
94  QObject::connect(closeAction, &QAction::triggered, q, &KMessageWidget::animatedHide);
95 
96  closeButton = new QToolButton(content);
97  closeButton->setAutoRaise(true);
98  closeButton->setDefaultAction(closeAction);
99 
100  q->setMessageType(KMessageWidget::Information);
101 }

103 void KMessageWidgetPrivate::createLayout()
104 {
105  delete content->layout();
106 
107  content->resize(q->size());
108 
109  qDeleteAll(buttons);
110  buttons.clear();
111 
112  Q_FOREACH (QAction *action, q->actions()) {
113  QToolButton *button = new QToolButton(content);
114  button->setDefaultAction(action);
115  button->setToolButtonStyle(Qt::ToolButtonTextBesideIcon);
116  buttons.append(button);
117  }
118 
119  // AutoRaise reduces visual clutter, but we don't want to turn it on if
120  // there are other buttons, otherwise the close button will look different
121  // from the others.
122  closeButton->setAutoRaise(buttons.isEmpty());
123 
124  if (wordWrap) {
125  QGridLayout *layout = new QGridLayout(content);
126  // Set alignment to make sure icon does not move down if text wraps
127  layout->addWidget(iconLabel, 0, 0, 1, 1, Qt::AlignHCenter | Qt::AlignTop);
128  layout->addWidget(textLabel, 0, 1);
129 
130  QHBoxLayout *buttonLayout = new QHBoxLayout;
131  buttonLayout->addStretch();
132  Q_FOREACH (QToolButton *button, buttons) {
133  // For some reason, calling show() is necessary if wordwrap is true,
134  // otherwise the buttons do not show up. It is not needed if
135  // wordwrap is false.
136  button->show();
137  buttonLayout->addWidget(button);
138  }
139  buttonLayout->addWidget(closeButton);
140  layout->addItem(buttonLayout, 1, 0, 1, 2);
141  } else {
142  QHBoxLayout *layout = new QHBoxLayout(content);
143  layout->addWidget(iconLabel);
144  layout->addWidget(textLabel);
145 
146  Q_FOREACH (QToolButton *button, buttons) {
147  layout->addWidget(button);
148  }
149 
150  layout->addWidget(closeButton);
151  };
152 
153  if (q->isVisible()) {
154  q->setFixedHeight(content->sizeHint().height());
155  }
156  q->updateGeometry();
157 }

166 void KMessageWidgetPrivate::updateSnapShot()
167 {
168  // Attention: updateSnapShot calls QWidget::render(), which causes the whole
169  // window layouts to be activated. Calling this method from resizeEvent()
170  // can lead to infinite recursion, see:
171  // https://bugs.kde.org/show_bug.cgi?id=311336
172  contentSnapShot = QPixmap(content->size() * q->devicePixelRatio());
173  contentSnapShot.setDevicePixelRatio(q->devicePixelRatio());
174  contentSnapShot.fill(Qt::transparent);
175  content->render(&contentSnapShot, QPoint(), QRegion(), QWidget::DrawChildren);
176 }

184 void KMessageWidgetPrivate::slotTimeLineFinished()
185 {
186  if (timeLine->direction() == QTimeLine::Forward) {
187  // Show
188  // We set the whole geometry here, because it may be wrong if a
189  // KMessageWidget is shown right when the toplevel window is created.
190  content->setGeometry(0, 0, q->width(), bestContentHeight());
191 
192  // notify about finished animation
193  emit q->showAnimationFinished();
194  } else {
195  // hide and notify about finished animation
196  q->hide();
197  emit q->hideAnimationFinished();
198  }
199 }

249 static QColor darkShade(QColor c)
250 {
251  qreal contrast = 0.7; // taken from kcolorscheme for the dark shade
252 
253  qreal darkAmount;
254  if (c.lightnessF() < 0.006) { /* too dark */
255  darkAmount = 0.02 + 0.40 * contrast;
256  } else if (c.lightnessF() > 0.93) { /* too bright */
257  darkAmount = -0.06 - 0.60 * contrast;
258  } else {
259  darkAmount = (-c.lightnessF()) * (0.55 + contrast * 0.35);
260  }
261 
262  qreal v = c.lightnessF() + darkAmount;
263  v = v > 0.0 ? (v < 1.0 ? v : 1.0) : 0.0;
264  c.setHsvF(c.hslHueF(), c.hslSaturationF(), v);
265  return c;
266 }

268 void KMessageWidget::setMessageType(KMessageWidget::MessageType type)
269 {
270  d->messageType = type;
271  QColor bg0, bg1, bg2, border, fg;
272  switch (type) {
273  case Positive:
274  bg1.setRgb(39, 174, 96); // values taken from kcolorscheme.cpp (Positive)
275  fg.setRgb(239, 240, 241);
276  break;
277  case Information:
278  bg1 = palette().highlight().color();
279  fg = palette().highlightedText().color();
280  break;
281  case Warning:
282  bg1.setRgb(246, 116, 0); // values taken from kcolorscheme.cpp (Neutral)
283  fg.setRgb(239, 240, 241);
284  break;
285  case Error:
286  bg1.setRgb(218, 68, 83); // values taken from kcolorscheme.cpp (Negative)
287  fg.setRgb(239, 240, 241);
288  break;
289  }
290 
291  // Colors
292  bg0 = bg1.lighter(110);
293  bg2 = bg1.darker(110);
294  border = darkShade(bg1);
295 
296  d->content->setStyleSheet(
297  QString::fromLatin1(".QFrame {"
298  "background-color: qlineargradient(x1: 0, y1: 0, x2: 0, y2: 1,"
299  " stop: 0 %1,"
300  " stop: 0.1 %2,"
301  " stop: 1.0 %3);"
302  "border-radius: 5px;"
303  "border: 1px solid %4;"
304  "margin: %5px;"
305  "}"
306  ".QLabel { color: %6; }"
307  )
308  .arg(bg0.name())
309  .arg(bg1.name())
310  .arg(bg2.name())
311  .arg(border.name())
312  // DefaultFrameWidth returns the size of the external margin + border width. We know our border is 1px, so we subtract this from the frame normal QStyle FrameWidth to get our margin
313  .arg(style()->pixelMetric(QStyle::PM_DefaultFrameWidth, 0, this) - 1)
314  .arg(fg.name())
315  );
316 }

368 void KMessageWidget::setWordWrap(bool wordWrap)
369 {
370  d->wordWrap = wordWrap;
371  d->textLabel->setWordWrap(wordWrap);
372  QSizePolicy policy = sizePolicy();
373  policy.setHeightForWidth(wordWrap);
374  setSizePolicy(policy);
375  d->updateLayout();

407 void KMessageWidget::animatedShow()
408 {
409  if (!style()->styleHint(QStyle::SH_Widget_Animate, 0, this)) {
410  show();
411  emit showAnimationFinished();
412  return;
413  }
414 
415  if (isVisible()) {
416  return;
417  }
418 
419  QFrame::show();
420  setFixedHeight(0);
421  int wantedHeight = d->bestContentHeight();
422  d->content->setGeometry(0, -wantedHeight, width(), wantedHeight);
423 
424  d->updateSnapShot();
425 
426  d->timeLine->setDirection(QTimeLine::Forward);
427  if (d->timeLine->state() == QTimeLine::NotRunning) {
428  d->timeLine->start();
429  }
430 }

432 void KMessageWidget::animatedHide()
433 {
434  if (!style()->styleHint(QStyle::SH_Widget_Animate, 0, this)) {
435  hide();
436  emit hideAnimationFinished();
437  return;
438  }
439 
440  if (!isVisible()) {
441  return;
442  }
443 
444  d->content->move(0, -d->content->height());
445  d->updateSnapShot();
446 
447  d->timeLine->setDirection(QTimeLine::Backward);
448  if (d->timeLine->state() == QTimeLine::NotRunning) {
449  d->timeLine->start();
450  }
451 }
