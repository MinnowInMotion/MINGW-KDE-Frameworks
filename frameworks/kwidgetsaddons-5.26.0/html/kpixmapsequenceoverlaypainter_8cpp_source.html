67 void KPixmapSequenceOverlayPainter::Private::_k_timeout()
68 {
69  if (sequence().isEmpty()) {
70  return;
71  }
72  ++m_counter;
73  m_counter %= sequence().frameCount();
74  if (m_widget) {
75  m_widget->update(pixmapRect());
76  }
77 }

79 void KPixmapSequenceOverlayPainter::Private::paintFrame()
80 {
81  if (m_counter >= sequence().frameCount()) {
82  return;
83  }
84  QPainter p(m_widget);
85  p.drawPixmap(pixmapRect(), sequence().frameAt(m_counter), QRect(QPoint(0, 0), sequence().frameSize()));
86 }

93 QRect KPixmapSequenceOverlayPainter::Private::pixmapRect()
94 {
95  QRect rect(m_rect);
96  if (!rect.isValid()) {
97  rect = m_widget->rect();
98  }
99 
100  QPoint pos(rect.topLeft());
101  if (m_alignment & Qt::AlignHCenter) {
102  pos.setX(rect.center().x() - (sequence().frameSize().width() / 2));
103  } else if (m_alignment & Qt::AlignRight) {
104  pos.setX(rect.right() - sequence().frameSize().width());
105  }
106 
107  if (m_alignment & Qt::AlignVCenter) {
108  pos.setY(rect.center().y() - (sequence().frameSize().height() / 2));
109  } else if (m_alignment & Qt::AlignBottom) {
110  pos.setY(rect.bottom() - sequence().frameSize().height());
111  }
112 
113  pos += m_offset;
114 
115  return QRect(pos, sequence().frameSize());
116 }

246 bool KPixmapSequenceOverlayPainter::eventFilter(QObject *obj, QEvent *event)
247 {
248  if (obj == d->m_widget) {
249  switch (event->type()) {
250  case QEvent::Paint:
251  // make sure we paint after everyone else including other event filters
252  obj->removeEventFilter(this); // don't recurse...
253  QCoreApplication::sendEvent(obj, event);
254  d->paintFrame();
255  obj->installEventFilter(this); // catch on...
256  return true;
257  break;
258  case QEvent::Hide:
259  d->m_timer.stop();
260  break;
261  case QEvent::Show:
262  if (d->m_started) {
263  d->m_timer.start();
264  d->m_widget->update(d->pixmapRect());
265  }
266  break;
267  default:
268  break;
269  }
270  }
271 
272  return false;
273 }
