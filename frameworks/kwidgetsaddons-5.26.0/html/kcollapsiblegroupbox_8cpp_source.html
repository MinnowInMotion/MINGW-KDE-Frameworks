57 KCollapsibleGroupBox::KCollapsibleGroupBox(QWidget* parent):
58  QWidget(parent),
59  d(new KCollapsibleGroupBoxPrivate(this))
60 {
61  d->recalculateHeaderSize();
62 
63  d->animation = new QTimeLine(500, this); //duration matches kmessagewidget
64  connect(d->animation, &QTimeLine::valueChanged, this, [this](qreal value) {
65  setFixedHeight((d->contentSize().height() * value) + d->headerSize.height());
66  });
67  connect(d->animation, &QTimeLine::stateChanged, this, [this](QTimeLine::State state) {
68  if (state == QTimeLine::NotRunning) {
69  //when collapsed hide contents to save resources and more importantly get it out the focus chain
70  d->updateChildrenVisibility(d->isExpanded);
71  }
72  });
73 
74  setSizePolicy(QSizePolicy::MinimumExpanding, QSizePolicy::Fixed);
75  setFocusPolicy(Qt::TabFocus);
76  setMouseTracking(true);
77 }

111 void KCollapsibleGroupBox::setExpanded(bool expanded)
112 {
113  if (expanded == d->isExpanded) {
114  return;
115  }
116 
117  d->isExpanded = expanded;
118  emit expandedChanged();
119 
120  d->updateChildrenVisibility(true);
121 
122  d->animation->setDirection(expanded ? QTimeLine::Forward : QTimeLine::Backward);
123  d->animation->start();
124 
125  //when going from collapsed to expanded changing the child visibility calls an updateGeometry
126  //which calls sizeHint with expanded true before the first frame of the animation kicks in
127  //trigger an effective frame 0
128  if (expanded) {
129  setFixedHeight(d->headerSize.height());
130  }
131 }

153 void KCollapsibleGroupBox::paintEvent(QPaintEvent *event)
154 {
155  QPainter p(this);
156 
157  QStyleOptionButton baseOption;
158  baseOption.initFrom(this);
159  baseOption.rect = QRect(0, 0, width(), d->headerSize.height());
160  baseOption.text = d->title;
161 
162  if (d->headerContainsMouse) {
163  baseOption.state |= QStyle::State_MouseOver;
164  }
165 
166  QStyle::PrimitiveElement element;
167  if (d->isExpanded) {
168  element = QStyle::PE_IndicatorArrowDown;
169  } else {
170  element = isLeftToRight() ? QStyle::PE_IndicatorArrowRight : QStyle::PE_IndicatorArrowLeft;
171  }
172 
173  QStyleOptionButton indicatorOption = baseOption;
174  indicatorOption.rect = style()->subElementRect(QStyle::SE_CheckBoxIndicator, &indicatorOption, this);
175  style()->drawPrimitive(element, &indicatorOption, &p, this);
176 
177  QStyleOptionButton labelOption = baseOption;
178  labelOption.rect = style()->subElementRect(QStyle::SE_CheckBoxContents, &labelOption, this);
179  style()->drawControl(QStyle::CE_CheckBoxLabel, &labelOption, &p, this);
180 
181  Q_UNUSED(event)
182 }

199 bool KCollapsibleGroupBox::event(QEvent *event)
200 {
201  switch (event->type()) {
202  case QEvent::StyleChange:
203  /*fall through*/
204  case QEvent::FontChange:
205  d->recalculateHeaderSize();
206  break;
207  case QEvent::Shortcut:
208  {
209  QShortcutEvent *se = static_cast<QShortcutEvent*>(event);
210  if(d->shortcutId == se->shortcutId()) {
211  toggle();
212  return true;
213  }
214  break;
215  }
216  case QEvent::ChildAdded:
217  {
218  QChildEvent *ce = static_cast<QChildEvent*>(event);
219  if (ce->child()->isWidgetType()) {
220  static_cast<QWidget*>(ce->child())->setVisible(d->isExpanded);
221  }
222  break;
223  }
224  case QEvent::LayoutRequest:
225  if (d->animation->state() == QTimeLine::NotRunning) {
226  setFixedHeight(sizeHint().height());
227  }
228  break;
229  default:
230  break;
231  }
232 
233  return QWidget::event(event);
234 }

266 void KCollapsibleGroupBox::keyPressEvent(QKeyEvent *event)
267 {

270  return;

279 void KCollapsibleGroupBox::resizeEvent(QResizeEvent *event)
280 {
281  int top, left, right, bottom;
282  getContentsMargins(&left, &top, &right, &bottom);
283 
284  if (layout()) {
285  //we don't want the layout trying to fit the current frame of the animation so always set it to the target height
286  layout()->setGeometry(QRect(left, top, width() - left - right, layout()->sizeHint().height()));
287  }
288 
289  Q_UNUSED(bottom)
290  QWidget::resizeEvent(event);
291 }

293 void KCollapsibleGroupBoxPrivate::recalculateHeaderSize()
294 {
295  QStyleOption option;
296  option.initFrom(q);
297 
298  QSize textSize = q->style()->itemTextRect(option.fontMetrics, QRect(), Qt::TextShowMnemonic, false,
299  title).size();
300 
301  headerSize = q->style()->sizeFromContents(QStyle::CT_CheckBox, &option, textSize, q);
302  q->setContentsMargins(q->style()->pixelMetric(QStyle::PM_IndicatorWidth), headerSize.height(), 0, 0);
303 }
