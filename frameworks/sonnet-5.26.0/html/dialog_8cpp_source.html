214 void Dialog::setProgressDialogVisible(bool b)
215 {
216  if (!b) {
217  d->deleteProgressDialog(true);
218  } else if (d->progressDialogTimeout >= 0) {
219  if (d->progressDialog) {
220  return;
221  }
222  d->progressDialog = new QProgressDialog(this);
223  d->progressDialog->setLabelText(tr("Spell checking in progress...", "progress label"));
224  d->progressDialog->setWindowTitle(tr("Check Spelling", "@title:window"));
225  d->progressDialog->setModal(true);
226  d->progressDialog->setAutoClose(false);
227  d->progressDialog->setAutoReset(false);
228  // create an 'indefinite' progress box as we currently cannot get progress feedback from
229  // the speller
230  d->progressDialog->reset();
231  d->progressDialog->setRange(0, 0);
232  d->progressDialog->setValue(0);
233  connect(d->progressDialog, SIGNAL(canceled()), this, SLOT(slotCancel()));
234  d->progressDialog->setMinimumDuration(d->progressDialogTimeout);
235  }

397 void Dialog::slotMisspelling(const QString &word, int start)
398 {
399  setGuiEnabled(true);
400  setProgressDialogVisible(false);
401  emit misspelling(word, start);
402  //NOTE this is HACK I had to introduce because BackgroundChecker lacks 'virtual' marks on methods
403  //this dramatically reduces spellchecking time in Lokalize
404  //as this doesn't fetch suggestions for words that are present in msgid
405  if (!updatesEnabled()) {
406  return;
407  }
408 
409  d->currentWord = word;
410  d->currentPosition = start;
411  if (d->replaceAllMap.contains(word)) {
412  d->ui.m_replacement->setText(d->replaceAllMap[ word ]);
413  slotReplaceWord();
414  } else {
415  updateDialog(word);
416  }
417  QDialog::show();
418 }

420 void Dialog::slotDone()
421 {
422  d->restart = false;
423  emit done(d->checker->text());
424  if (d->restart) {
425  updateDictionaryComboBox();
426  d->checker->setText(d->originalBuffer);
427  d->restart = false;
428  } else {
429  setProgressDialogVisible(false);
430  emit spellCheckStatus(tr("Spell check complete."));
431  accept();
432  if (!d->canceled && d->showCompletionMessageBox) {
433  QMessageBox::information(this, tr("Spell check complete."), tr("Check Spelling", "@title:window"));
434  }
435  }
