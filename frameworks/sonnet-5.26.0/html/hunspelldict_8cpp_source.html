35 HunspellDict::HunspellDict(const QString &lang, QString path)
36  : SpellerPlugin(lang)
37  , m_speller(0)
38  , m_codec(0)
39 {
40  qCDebug(SONNET_HUNSPELL) << "Loading dictionary for" << lang << "from" << path;
41 
42  if (!path.endsWith(QLatin1Char('/'))) {
43  path += QLatin1Char('/');
44  }
45  path += lang;
46  QString dictionary = path + QStringLiteral(".dic");
47  QString aff = path + QStringLiteral(".aff");
48 
49  if (QFileInfo::exists(dictionary) && QFileInfo::exists(aff)) {
50  m_speller = new Hunspell(aff.toLocal8Bit().constData(), dictionary.toLocal8Bit().constData());
51  m_codec = QTextCodec::codecForName(m_speller->get_dic_encoding());
52  if (!m_codec) {
53  qWarning() << "Failed to find a text codec for name" << m_speller->get_dic_encoding() << "defaulting to locale text codec";
54  m_codec = QTextCodec::codecForLocale();
55  Q_ASSERT(m_codec);
56  }
57  } else {
58  qCWarning(SONNET_HUNSPELL) << "Unable to find dictionary for" << lang << "in path" << path;
59  }
60 
61  QString userDic = QDir::home().filePath(QLatin1String(".hunspell_") % lang);
62  QFile userDicFile(userDic);
63  if (userDicFile.open(QIODevice::ReadOnly | QIODevice::Text)) {
64  qCDebug(SONNET_HUNSPELL) << "Load a user dictionary" << userDic;
65  QTextStream userDicIn(&userDicFile);
66  while (!userDicIn.atEnd()) {
67  QString word = userDicIn.readLine();
68  if (word.contains(QLatin1Char('/'))) {
69  QStringList wordParts = word.split(QLatin1Char('/'));
70  m_speller->add_with_affix(toDictEncoding(wordParts.at(0)).constData(), toDictEncoding(wordParts.at(1)).constData());
71  } if (word.at(0) == QLatin1Char('*')) {
72  m_speller->remove(toDictEncoding(word.mid(1)).constData());
73  } else {
74  m_speller->add(toDictEncoding(word).constData());
75  }
76  }
77  userDicFile.close();
78  }
79  qCDebug(SONNET_HUNSPELL) << "Created " << m_speller;
80 }

103 QStringList HunspellDict::suggest(const QString &word) const
104 {
105  if (!m_speller) {
106  return QStringList();
107  }
108  char **selection;
109  QStringList lst;
110  int nbWord = m_speller->suggest(&selection, toDictEncoding(word).constData());
111  for (int i = 0; i < nbWord; ++i) {
112  lst << m_codec->toUnicode(selection[i]);
113  }
114  m_speller->free_list(&selection, nbWord);
115  return lst;
