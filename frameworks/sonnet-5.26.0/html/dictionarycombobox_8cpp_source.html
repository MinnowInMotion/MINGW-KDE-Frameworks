100 void DictionaryComboBox::reloadCombo()
101 {
102  clear();
103  Sonnet::Speller *speller = new Sonnet::Speller();
104  QMap<QString, QString> dictionaries = speller->availableDictionaries();
105  QMapIterator<QString, QString> i(dictionaries);
106  while (i.hasNext()) {
107  i.next();
108  addItem(i.key(), i.value());
109  }
110  delete speller;
111 }
