74 bool SpellCheckDecorator::Private::onContextMenuEvent(QContextMenuEvent *event)
75 {
76  if (!m_highlighter) {
77  createDefaultHighlighter();
78  }
79 
80  // Obtain the cursor at the mouse position and the current cursor
81  QTextCursor cursorAtMouse;
82  if (m_textEdit) {
83  cursorAtMouse = m_textEdit->cursorForPosition(event->pos());
84  } else {
85  cursorAtMouse = m_plainTextEdit->cursorForPosition(event->pos());
86  }
87  const int mousePos = cursorAtMouse.position();
88  QTextCursor cursor;
89  if (m_textEdit) {
90  cursor = m_textEdit->textCursor();
91  } else {
92  cursor = m_plainTextEdit->textCursor();
93  }
94 
95  // Check if the user clicked a selected word
96  const bool selectedWordClicked = cursor.hasSelection() &&
97  mousePos >= cursor.selectionStart() &&
98  mousePos <= cursor.selectionEnd();
99 
100  // Get the word under the (mouse-)cursor and see if it is misspelled.
101  // Don't include apostrophes at the start/end of the word in the selection.
102  QTextCursor wordSelectCursor(cursorAtMouse);
103  wordSelectCursor.clearSelection();
104  wordSelectCursor.select(QTextCursor::WordUnderCursor);
105  QString selectedWord = wordSelectCursor.selectedText();
106 
107  bool isMouseCursorInsideWord = true;
108  if ((mousePos < wordSelectCursor.selectionStart() ||
109  mousePos >= wordSelectCursor.selectionEnd())
110  && (selectedWord.length() > 1)) {
111  isMouseCursorInsideWord = false;
112  }
113 
114  // Clear the selection again, we re-select it below (without the apostrophes).
115  wordSelectCursor.setPosition(wordSelectCursor.position() - selectedWord.size());
116  if (selectedWord.startsWith(QLatin1Char('\'')) || selectedWord.startsWith(QLatin1Char('\"'))) {
117  selectedWord = selectedWord.right(selectedWord.size() - 1);
118  wordSelectCursor.movePosition(QTextCursor::NextCharacter, QTextCursor::MoveAnchor);
119  }
120  if (selectedWord.endsWith(QLatin1Char('\'')) || selectedWord.endsWith(QLatin1Char('\"'))) {
121  selectedWord.chop(1);
122  }
123 
124  wordSelectCursor.movePosition(QTextCursor::NextCharacter,
125  QTextCursor::KeepAnchor, selectedWord.size());
126 
127  const bool wordIsMisspelled = isMouseCursorInsideWord &&
128  m_highlighter &&
129  m_highlighter->isActive() &&
130  !selectedWord.isEmpty() &&
131  m_highlighter->isWordMisspelled(selectedWord);
132 
133  // If the user clicked a selected word, do nothing.
134  // If the user clicked somewhere else, move the cursor there.
135  // If the user clicked on a misspelled word, select that word.
136  // Same behavior as in OpenOffice Writer.
137  bool checkBlock = q->isSpellCheckingEnabledForBlock(cursorAtMouse.block().text());
138  if (!selectedWordClicked) {
139  if (wordIsMisspelled && checkBlock) {
140  if (m_textEdit) {
141  m_textEdit->setTextCursor(wordSelectCursor);
142  } else {
143  m_plainTextEdit->setTextCursor(wordSelectCursor);
144  }
145  } else {
146  if (m_textEdit) {
147  m_textEdit->setTextCursor(cursorAtMouse);
148  } else {
149  m_plainTextEdit->setTextCursor(cursorAtMouse);
150  }
151  }
152  if (m_textEdit) {
153  cursor = m_textEdit->textCursor();
154  } else {
155  cursor = m_plainTextEdit->textCursor();
156  }
157  }
158 
159  // Use standard context menu for already selected words, correctly spelled
160  // words and words inside quotes.
161  if (!wordIsMisspelled || selectedWordClicked || !checkBlock) {
162  return false;
163  }
164  execSuggestionMenu(event->globalPos(), selectedWord, cursor);
165  return true;
166 }

168 void SpellCheckDecorator::Private::execSuggestionMenu(const QPoint &pos, const QString &selectedWord, const QTextCursor &_cursor)
169 {
170  QTextCursor cursor = _cursor;
171  QMenu menu; //don't use KMenu here we don't want auto management accelerator
172 
173  //Add the suggestions to the menu
174  const QStringList reps = m_highlighter->suggestionsForWord(selectedWord);
175  if (reps.isEmpty()) {
176  QAction *suggestionsAction = menu.addAction(tr("No suggestions for %1").arg(selectedWord));
177  suggestionsAction->setEnabled(false);
178  } else {
179  QStringList::const_iterator end(reps.constEnd());
180  for (QStringList::const_iterator it = reps.constBegin(); it != end; ++it) {
181  menu.addAction(*it);
182  }
183  }
184 
185  menu.addSeparator();
186 
187  QAction *ignoreAction = menu.addAction(tr("Ignore"));
188  QAction *addToDictAction = menu.addAction(tr("Add to Dictionary"));
189  //Execute the popup inline
190  const QAction *selectedAction = menu.exec(pos);
191 
192  if (selectedAction) {
193  Q_ASSERT(cursor.selectedText() == selectedWord);
194 
195  if (selectedAction == ignoreAction) {
196  m_highlighter->ignoreWord(selectedWord);
197  m_highlighter->rehighlight();
198  } else if (selectedAction == addToDictAction) {
199  m_highlighter->addWordToDictionary(selectedWord);
200  m_highlighter->rehighlight();
201  }
202 
203  // Other actions can only be one of the suggested words
204  else {
205  const QString replacement = selectedAction->text();
206  Q_ASSERT(reps.contains(replacement));
207  cursor.insertText(replacement);
208  if (m_textEdit) {
209  m_textEdit->setTextCursor(cursor);
210  } else {
211  m_plainTextEdit->setTextCursor(cursor);
212  }
213  }
214  }
215 }
