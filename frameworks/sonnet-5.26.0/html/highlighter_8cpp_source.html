211 void Highlighter::slotAutoDetection()
212 {
213  bool savedActive = d->active;
214 
215  //don't disable just because 1 of 4 is misspelled.
216  if (d->automatic && d->wordCount >= 10) {
217  // tme = Too many errors
218  bool tme = (d->errorCount >= d->disableWordCount) && (
219  d->errorCount * 100 >= d->disablePercentage * d->wordCount);
220  if (d->active && tme) {
221  d->active = false;
222  } else if (!d->active && !tme) {
223  d->active = true;
224  }
225  }
226 
227  if (d->active != savedActive) {
228  if (d->active) {
229  emit activeChanged(tr("As-you-type spell checking enabled."));
230  } else {
231  qCDebug(SONNET_LOG_UI) << "Sonnet: Disabling spell checking, too many errors";
232  emit activeChanged(tr("Too many misspelled words. "
233  "As-you-type spell checking disabled."));
234  }
235 
236  d->completeRehighlightRequired = true;
237  d->rehighlightRequest->setInterval(100);
238  d->rehighlightRequest->setSingleShot(true);
239  }
240 }

262 void Highlighter::contentsChange(int pos, int add, int rem)
263 {
264  // Invalidate the cache where the text has changed
265  const QTextBlock &lastBlock = document()->findBlock(pos + add - rem);
266  QTextBlock block = document()->findBlock(pos);
267  do {
268  LanguageCache* cache=dynamic_cast<LanguageCache*>(block.userData());
269  if (cache) cache->invalidate(pos-block.position());
270  block = block.next();
271  } while (block.isValid() && block < lastBlock);
272 }

274 void Highlighter::highlightBlock(const QString &text)
275 {
276  if (text.isEmpty() || !d->active || !d->spellCheckerFound) {
277  return;
278  }
279 
280  if (!d->connected) {
281  connect(document(), SIGNAL(contentsChange(int,int,int)),
282  SLOT(contentsChange(int,int,int)));
283  d->connected = true;
284  }
285  QTextCursor cursor;
286  if (d->textEdit) {
287  cursor = d->textEdit->textCursor();
288  } else {
289  cursor = d->plainTextEdit->textCursor();
290  }
291  int index = cursor.position();
292 
293  const int lengthPosition = text.length() - 1;
294 
295  if ( index != lengthPosition ||
296  ( lengthPosition > 0 && !text[lengthPosition-1].isLetter() ) ) {
297  d->languageFilter->setBuffer(text);
298 
299  LanguageCache* cache=dynamic_cast<LanguageCache*>(currentBlockUserData());
300  if (!cache) {
301  cache = new LanguageCache;
302  setCurrentBlockUserData(cache);
303  }
304 
305  while (d->languageFilter->hasNext()) {
306  QStringRef sentence=d->languageFilter->next();
307  if (d->spellchecker->testAttribute(Speller::AutoDetectLanguage)) {
308 
309  QString lang;
310  QPair<int,int> spos=QPair<int,int>(sentence.position(),sentence.length());
311  // try cache first
312  if (cache->languages.contains(spos)) {
313  lang=cache->languages.value(spos);
314  } else {
315  lang=d->languageFilter->language();
316  if (!d->languageFilter->isSpellcheckable()) lang.clear();
317  cache->languages[spos]=lang;
318  }
319  if (lang.isEmpty()) continue;
320  d->spellchecker->setLanguage(lang);
321  }
322 
323 
324  d->tokenizer->setBuffer(sentence.toString());
325  int offset=sentence.position();
326  while (d->tokenizer->hasNext()) {
327  QStringRef word=d->tokenizer->next();
328  if (!d->tokenizer->isSpellcheckable()) continue;
329  ++d->wordCount;
330  if (d->spellchecker->isMisspelled(word.toString())) {
331  ++d->errorCount;
332  setMisspelled(word.position()+offset, word.length());
333  } else {
334  unsetMisspelled(word.position()+offset, word.length());
335  }
336  }
337  }
338  }
339  //QTimer::singleShot( 0, this, SLOT(checkWords()) );
340  setCurrentBlockState(0);
341 }

381 bool Highlighter::eventFilter(QObject *o, QEvent *e)
382 {
383  if (!d->spellCheckerFound) {
384  return false;
385  }
386  if ((o == d->textEdit || o == d->plainTextEdit) && (e->type() == QEvent::KeyPress)) {
387  QKeyEvent *k = static_cast<QKeyEvent *>(e);
388  //d->autoReady = true;
389  if (d->rehighlightRequest->isActive()) { // try to stay out of the users way
390  d->rehighlightRequest->start(500);
391  }
392  if (k->key() == Qt::Key_Enter ||
393  k->key() == Qt::Key_Return ||
394  k->key() == Qt::Key_Up ||
395  k->key() == Qt::Key_Down ||
396  k->key() == Qt::Key_Left ||
397  k->key() == Qt::Key_Right ||
398  k->key() == Qt::Key_PageUp ||
399  k->key() == Qt::Key_PageDown ||
400  k->key() == Qt::Key_Home ||
401  k->key() == Qt::Key_End ||
402  ((k->modifiers() == Qt::ControlModifier) &&
403  ((k->key() == Qt::Key_A) ||
404  (k->key() == Qt::Key_B) ||
405  (k->key() == Qt::Key_E) ||
406  (k->key() == Qt::Key_N) ||
407  (k->key() == Qt::Key_P)))) {
408  if (intraWordEditing()) {
409  setIntraWordEditing(false);
410  d->completeRehighlightRequired = true;
411  d->rehighlightRequest->setInterval(500);
412  d->rehighlightRequest->setSingleShot(true);
413  d->rehighlightRequest->start();
414  }
415  } else {
416  setIntraWordEditing(true);
417  }
418  if (k->key() == Qt::Key_Space ||
419  k->key() == Qt::Key_Enter ||
420  k->key() == Qt::Key_Return) {
421  QTimer::singleShot(0, this, SLOT(slotAutoDetection()));
422  }
423  }
424 
425  else if ((( d->textEdit && ( o == d->textEdit->viewport())) || (d->plainTextEdit && (o == d->plainTextEdit->viewport()))) &&
426  (e->type() == QEvent::MouseButtonPress)) {
427  //d->autoReady = true;
428  if (intraWordEditing()) {
429  setIntraWordEditing(false);
430  d->completeRehighlightRequired = true;
431  d->rehighlightRequest->setInterval(0);
432  d->rehighlightRequest->setSingleShot(true);
433  d->rehighlightRequest->start();
434  }
435  }
436  return false;
437 }
