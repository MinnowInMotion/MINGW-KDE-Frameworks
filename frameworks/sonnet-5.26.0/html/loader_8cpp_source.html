85 SpellerPlugin *Loader::createSpeller(const QString &language,
86  const QString &clientName) const
87 {
88  QString pclient = clientName;
89  QString plang = language;
90 
91  if (plang.isEmpty()) {
92  plang = d->settings->defaultLanguage();
93  }
94 
95  const QVector<Client *> lClients = d->languageClients[plang];
96 
97  if (lClients.isEmpty()) {
98  qCWarning(SONNET_LOG_CORE) << "No language dictionaries for the language:" << plang;
99  return 0;
100  }
101 
102  QVectorIterator<Client *> itr(lClients);
103  while (itr.hasNext()) {
104  Client *item = itr.next();
105  if (!pclient.isEmpty()) {
106  if (pclient == item->name()) {
107  SpellerPlugin *dict = item->createSpeller(plang);
108  return dict;
109  }
110  } else {
111  //the first one is the one with the highest
112  //reliability
113  SpellerPlugin *dict = item->createSpeller(plang);
114  return dict;
115  }
116  }
117 
118  return 0;

147  QString currentDictionary = langCode, // e.g. en_GB-ize-wo_accents
148  isoCode, // locale ISO name
149  variantName, // dictionary variant name e.g. w_accents
150  localizedLang, // localized language
151  localizedCountry, // localized country

155  int minusPos, // position of "-" char
156  variantCount = 0; // used to iterate over variantList

158  struct variantListType {
159  const char *variantShortName;
160  const char *variantEnglishName;
161  };
162 
163  /*
164  * This redefines the QT_TRANSLATE_NOOP3 macro provided by Qt to indicate that
165  * statically initialised text should be translated so that it expands to just
166  * the string that should be translated, making it possible to use it in the
167  * single string construct below.
168  */
169 #undef QT_TRANSLATE_NOOP3
170 #define QT_TRANSLATE_NOOP3(a, b, c) b
171 
172  const variantListType variantList[] = {
173  { "40", QT_TRANSLATE_NOOP3("Sonnet::Loader", "40", "dictionary variant") }, // what does 40 mean?
174  { "60", QT_TRANSLATE_NOOP3("Sonnet::Loader", "60", "dictionary variant") }, // what does 60 mean?
175  { "80", QT_TRANSLATE_NOOP3("Sonnet::Loader", "80", "dictionary variant") }, // what does 80 mean?

177  { "ize", QT_TRANSLATE_NOOP3("Sonnet::Loader", "-ize suffixes", "dictionary variant") },
178  { "ise-w_accents", QT_TRANSLATE_NOOP3("Sonnet::Loader", "-ise suffixes and with accents", "dictionary variant") },
179  { "ise-wo_accents", QT_TRANSLATE_NOOP3("Sonnet::Loader", "-ise suffixes and without accents", "dictionary variant") },
180  { "ize-w_accents", QT_TRANSLATE_NOOP3("Sonnet::Loader", "-ize suffixes and with accents", "dictionary variant") },
181  { "ize-wo_accents", QT_TRANSLATE_NOOP3("Sonnet::Loader", "-ize suffixes and without accents", "dictionary variant") },
182  { "lrg", QT_TRANSLATE_NOOP3("Sonnet::Loader", "large", "dictionary variant") },

184  { "sml", QT_TRANSLATE_NOOP3("Sonnet::Loader", "small", "dictionary variant") },
185  { "variant_0", QT_TRANSLATE_NOOP3("Sonnet::Loader", "variant 0", "dictionary variant") },

197  minusPos = currentDictionary.indexOf(QLatin1Char('-'));
198  if (minusPos != -1) {
199  variantName = currentDictionary.right(currentDictionary.length() - minusPos - 1);
200  while (variantList[variantCount].variantShortName != 0)
201  if (QLatin1String(variantList[variantCount].variantShortName) == variantName) {
202  break;
203  } else {
204  variantCount++;
205  }
206  if (variantList[variantCount].variantShortName != 0) {
207  variantEnglish = variantList[variantCount].variantEnglishName;
208  } else {
209  variantEnglish = variantName.toLatin1();
210  }
211 
212  localizedVariant = tr(variantEnglish.constData(), "dictionary variant");
213  isoCode = currentDictionary.left(minusPos);
214  } else {
215  isoCode = currentDictionary;
216  }
217 
218  QLocale locale(isoCode);
219  localizedCountry = locale.nativeCountryName();
220  localizedLang = locale.nativeLanguageName();
221 
222  if (localizedLang.isEmpty() && localizedCountry.isEmpty()) {
223  return isoCode; // We have nothing
224  }
225 
226  if (!localizedCountry.isEmpty() && !localizedVariant.isEmpty()) { // We have both a country name and a variant
227  return tr("%1 (%2) [%3]", "dictionary name; %1 = language name, %2 = country name and %3 = language variant name"
228  ).arg(localizedLang, localizedCountry, localizedVariant);
229  } else if (!localizedCountry.isEmpty()) { // We have a country name
230  return tr("%1 (%2)", "dictionary name; %1 = language name, %2 = country name"
231  ).arg(localizedLang, localizedCountry);
232  } else { // We only have a language name
233  return localizedLang;

260 void Loader::loadPlugins()
261 {
262 #ifndef SONNET_STATIC
263  const QStringList libPaths = QCoreApplication::libraryPaths() << QStringLiteral(INSTALLATION_PLUGIN_PATH);
264  const QLatin1String pathSuffix("/kf5/sonnet/");
265  int plugins = 0;
266  Q_FOREACH (const QString &libPath, libPaths) {
267  QDir dir(libPath + pathSuffix);
268  if (!dir.exists()) {
269  continue;
270  }
271  Q_FOREACH (const QString &fileName, dir.entryList(QDir::Files)) {
272  loadPlugin(dir.absoluteFilePath(fileName));
273  plugins++;
274  }
275  }
276  if (plugins == 0) {
277  qCWarning(SONNET_LOG_CORE) << "Sonnet: No speller backends available!";
278  }
279 #else
280 #ifdef Q_OS_MAC
281  loadPlugin(QString());
282 #endif
283  loadPlugin(QStringLiteral("Hunspell"));
284 #endif
285 }

287 void Loader::loadPlugin(const QString &pluginPath)
288 {
289 #ifndef SONNET_STATIC
290  QPluginLoader plugin(pluginPath);
291  if (!plugin.load()) { // We do this separately for better error handling
292  qCWarning(SONNET_LOG_CORE) << "Sonnet: Unable to load plugin" << pluginPath << "Error:" << plugin.errorString();
293  return;
294  }
295 
296  Client *client = qobject_cast<Client *>(plugin.instance());
297  if (!client) {
298  qCWarning(SONNET_LOG_CORE) << "Sonnet: Invalid plugin loaded" << pluginPath;
299  plugin.unload(); // don't leave it in memory
300  return;
301  }
302 #else
303  Client *client = 0;
304  if (pluginPath == QLatin1String("Hunspell"))
305  client = new HunspellClient(this);
306 #ifdef Q_OS_MAC
307  else
308  client = new NSSpellCheckerClient(this);
309 #endif
310 #endif
311 
312  const QStringList languages = client->languages();
313  d->clients.append(client->name());
314 
315  Q_FOREACH (const QString &language, languages) {
316  QVector<Client *> &languageClients = d->languageClients[language];
317 
318  if (languageClients.isEmpty() || client->reliability() < languageClients.first()->reliability()) {
319  languageClients.append(client); // less reliable, to the end
320  } else {
321  languageClients.prepend(client); // more reliable, to the front
322  }
323  }
324 }
