170  bool storePersonal(const QString &personalWord,
171  const QString &bad = QString(),
172  const QString &good = QString()) const Q_DECL_NOEXCEPT
173  {
174  QFile userDictFile(m_userDictionaryFilepath);
175 
176  if (!userDictFile.open(QIODevice::ReadWrite)) {
177  qCWarning(SONNET_VOIKKO) << "Could not save personal dictionary. Failed to open file:"
178  << m_userDictionaryFilepath;
179  qCWarning(SONNET_VOIKKO) << "Reason:" << userDictFile.errorString();
180  return false;
181  }
182 
183  QJsonDocument dictDoc = QJsonDocument::fromJson(userDictFile.readAll());
184  auto root = readJsonRootObject(m_userDictionaryFilepath);
185  auto languageNode = root[q->language()].toObject();
186 
187  // Empty value means we are storing a bad:good pair
188  if (personalWord.isEmpty()) {
189  addReplacementToNode(languageNode, bad, good);
190  } else {
191  addPersonalWordToNode(languageNode, personalWord);
192  }
193 
194  root[q->language()] = languageNode;
195  dictDoc.setObject(root);
196 
197  userDictFile.reset();
198  userDictFile.write(dictDoc.toJson());
199  userDictFile.close();
200  qCDebug(SONNET_VOIKKO) << "Changes to user dictionary saved to file: "
201  << m_userDictionaryFilepath;
202 
203  return true;
204  };

210  void loadUserDictionary() Q_DECL_NOEXCEPT
211  {
212  // If root is empty we will fail later on when checking if
213  // languageNode is empty.
214  auto root = readJsonRootObject(m_userDictionaryFilepath);
215  auto languageNode = root[q->language()].toObject();
216 
217  if (languageNode.isEmpty()) {

219  }
220 
221  loadUserWords(languageNode);
222  loadUserReplacements(languageNode);
223  };

255  inline void loadUserReplacements(const QJsonObject &languageNode) Q_DECL_NOEXCEPT
256  {
257  auto words = languageNode[replacements_str()].toArray();
258  Q_FOREACH(auto pair, words) {
259  m_replacements[pair.toObject()[replacement_bad_str()].toString()] =
260  pair.toObject()[replacement_good_str()].toString();
261  }
262  qCDebug(SONNET_VOIKKO)
263  << QStringLiteral("Loaded %1 replacements from the user dictionary.").arg(words.size());
264  }

279 bool VoikkoDict::isCorrect(const QString &word) const
280 {
281  // Check the session word list and personal word list first
282  if (d->m_sessionWords.contains(word) || d->m_personalWords.contains(word)) {
283  return true;
284  }
285 
286  return (voikkoSpellUcs4(d->m_handle, d->QStringToWchar(word)) == VOIKKO_SPELL_OK);
287 }

289 QStringList VoikkoDict::suggest(const QString &word) const
290 {
291  QStringList suggestions;
292 
293  auto userDictPos = d->m_replacements.constFind(word);
294  if (userDictPos != d->m_replacements.constEnd()) {
295  suggestions.append(*userDictPos);
296  }
297 
298  auto voikkoSuggestions = voikkoSuggestUcs4(d->m_handle, d->QStringToWchar(word));
299 
300  if (!voikkoSuggestions) {
301  return suggestions;
302  }
303 
304  for (int i = 0; voikkoSuggestions[i] != Q_NULLPTR; ++i) {
305  QString suggestion = QString::fromWCharArray(voikkoSuggestions[i]);
306  suggestions.append(suggestion);
307  }
308  qCDebug(SONNET_VOIKKO) << "Misspelled:" << word
309  << "|Suggestons:" << suggestions.join(QStringLiteral(", "));
310 
311 
312  voikko_free_suggest_ucs4(voikkoSuggestions);
313 
314  return suggestions;
315 }
