230 void Settings::save()
231 {
232  QSettings settings(QStringLiteral("KDE"), QStringLiteral("Sonnet"));
233  settings.setValue(QStringLiteral("defaultClient"), d->defaultClient);
234  settings.setValue(QStringLiteral("defaultLanguage"), d->defaultLanguage);
235  settings.setValue(QStringLiteral("checkUppercase"), d->checkUppercase);
236  settings.setValue(QStringLiteral("skipRunTogether"), d->skipRunTogether);
237  settings.setValue(QStringLiteral("backgroundCheckerEnabled"), d->backgroundCheckerEnabled);
238  settings.setValue(QStringLiteral("checkerEnabledByDefault"), d->checkerEnabledByDefault);
239  settings.setValue(QStringLiteral("autodetectLanguage"), d->autodetectLanguage);
240  QString defaultLanguage = QStringLiteral("ignore_%1").arg(d->defaultLanguage);
241  if (settings.contains(defaultLanguage) && d->ignore.isEmpty()) {
242  settings.remove(defaultLanguage);
243  } else if (!d->ignore.isEmpty()) {
244  settings.setValue(defaultLanguage, QStringList(d->ignore.keys()));
245  }
246 }

272 void Settings::restore()
273 {
274  QSettings settings(QStringLiteral("KDE"), QStringLiteral("Sonnet"));
275  d->defaultClient = settings.value(QStringLiteral("defaultClient"), QString()).toString();
276  d->defaultLanguage = settings.value(QStringLiteral("defaultLanguage"), QLocale::system().bcp47Name()).toString();
277 
278  //same defaults are in the default filter (filter.cpp)
279  d->checkUppercase = settings.value(QStringLiteral("checkUppercase"), true).toBool();
280  d->skipRunTogether = settings.value(QStringLiteral("skipRunTogether"), true).toBool();
281  d->backgroundCheckerEnabled = settings.value(QStringLiteral("backgroundCheckerEnabled"), true).toBool();
282  d->checkerEnabledByDefault = settings.value(QStringLiteral("checkerEnabledByDefault"), false).toBool();
283  d->disablePercentage = settings.value(QStringLiteral("Sonnet_AsYouTypeDisablePercentage"), 90).toInt();
284  d->disableWordCount = settings.value(QStringLiteral("Sonnet_AsYouTypeDisableWordCount"), 100).toInt();
285  d->autodetectLanguage = settings.value(QStringLiteral("autodetectLanguage"), true).toBool();
286 
287  const QString ignoreEntry = QStringLiteral("ignore_%1").arg(d->defaultLanguage);
288  const QStringList ignores = settings.value(ignoreEntry, kdeWords()).toStringList();
289  setQuietIgnoreList(ignores);
290 }
