44 static void transformToMsDos(const QDateTime &_dt, char *buffer)
45 {
46  const QDateTime dt = _dt.isValid() ? _dt : QDateTime::currentDateTime();
47  const quint16 time =
48  (dt.time().hour() << 11) // 5 bit hour
49  | (dt.time().minute() << 5) // 6 bit minute
50  | (dt.time().second() >> 1); // 5 bit double seconds
51 
52  buffer[0] = char(time);
53  buffer[1] = char(time >> 8);
54 
55  const quint16 date =
56  ((dt.date().year() - 1980) << 9) // 7 bit year 1980-based
57  | (dt.date().month() << 5) // 4 bit month
58  | (dt.date().day()); // 5 bit day
59 
60  buffer[2] = char(date);
61  buffer[3] = char(date >> 8);
62 }

64 static uint transformFromMsDos(const char *buffer)
65 {
66  quint16 time = (uchar)buffer[0] | ((uchar)buffer[1] << 8);
67  int h = time >> 11;
68  int m = (time & 0x7ff) >> 5;
69  int s = (time & 0x1f) * 2;
70  QTime qt(h, m, s);
71 
72  quint16 date = (uchar)buffer[2] | ((uchar)buffer[3] << 8);
73  int y = (date >> 9) + 1980;
74  int o = (date & 0x1ff) >> 5;
75  int d = (date & 0x1f);
76  QDate qd(y, o, d);
77 
78  QDateTime dt(qd, qt);
79  return dt.toTime_t();
80 }

124 static bool parseExtTimestamp(const char *buffer, int size, bool islocal,
125  ParseFileInfo &pfi)
126 {
127  if (size < 1) {
128  //qDebug() << "premature end of extended timestamp (#1)";
129  return false;
130  }/*end if*/
131  int flags = *buffer; // read flags
132  buffer += 1;
133  size -= 1;
134 
135  if (flags & 1) { // contains modification time
136  if (size < 4) {
137  //qDebug() << "premature end of extended timestamp (#2)";
138  return false;
139  }/*end if*/
140  pfi.mtime = uint((uchar)buffer[0] | (uchar)buffer[1] << 8
141  | (uchar)buffer[2] << 16 | (uchar)buffer[3] << 24);
142  buffer += 4;
143  size -= 4;
144  }/*end if*/
145  // central extended field cannot contain more than the modification time
146  // even if other flags are set
147  if (!islocal) {
148  pfi.exttimestamp_seen = true;
149  return true;
150  }/*end if*/
151 
152  if (flags & 2) { // contains last access time
153  if (size < 4) {
154  //qDebug() << "premature end of extended timestamp (#3)";
155  return true;
156  }/*end if*/
157  pfi.atime = uint((uchar)buffer[0] | (uchar)buffer[1] << 8
158  | (uchar)buffer[2] << 16 | (uchar)buffer[3] << 24);
159  buffer += 4;
160  size -= 4;
161  }/*end if*/
162 
163  if (flags & 4) { // contains creation time
164  if (size < 4) {
165  //qDebug() << "premature end of extended timestamp (#4)";
166  return true;
167  }/*end if*/
168  pfi.ctime = uint((uchar)buffer[0] | (uchar)buffer[1] << 8
169  | (uchar)buffer[2] << 16 | (uchar)buffer[3] << 24);
170  buffer += 4;
171  }/*end if*/
172 
173  pfi.exttimestamp_seen = true;
174  return true;
175 }

185 static bool parseInfoZipUnixOld(const char *buffer, int size, bool islocal,
186  ParseFileInfo &pfi)
187 {
188  // spec mandates to omit this field if one of the newer fields are available
189  if (pfi.exttimestamp_seen || pfi.newinfounix_seen) {
190  return true;
191  }
192 
193  if (size < 8) {
194  //qDebug() << "premature end of Info-ZIP unix extra field old";
195  return false;
196  }
197 
198  pfi.atime = uint((uchar)buffer[0] | (uchar)buffer[1] << 8
199  | (uchar)buffer[2] << 16 | (uchar)buffer[3] << 24);
200  buffer += 4;
201  pfi.mtime = uint((uchar)buffer[0] | (uchar)buffer[1] << 8

253 static bool parseExtraField(const char *buffer, int size, bool islocal,
254  ParseFileInfo &pfi)
255 {
256  // extra field in central directory doesn't contain useful data, so we
257  // don't bother parsing it
258  if (!islocal) {
259  return true;
260  }
261 
262  while (size >= 4) { // as long as a potential extra field can be read
263  int magic = (uchar)buffer[0] | (uchar)buffer[1] << 8;
264  buffer += 2;
265  int fieldsize = (uchar)buffer[0] | (uchar)buffer[1] << 8;
266  buffer += 2;
267  size -= 4;
268 
269  if (fieldsize > size) {
270  //qDebug() << "fieldsize: " << fieldsize << " size: " << size;
271  //qDebug() << "premature end of extra fields reached";
272  break;
273  }
274 
275  switch (magic) {
276  case 0x5455: // extended timestamp
277  if (!parseExtTimestamp(buffer, fieldsize, islocal, pfi)) {
278  return false;
279  }
280  break;
281  case 0x5855: // old Info-ZIP unix extra field
282  if (!parseInfoZipUnixOld(buffer, fieldsize, islocal, pfi)) {
283  return false;
284  }
285  break;
286 #if 0 // not needed yet
287  case 0x7855: // new Info-ZIP unix extra field
288  if (!parseInfoZipUnixNew(buffer, fieldsize, islocal, pfi)) {
289  return false;
290  }
291  break;
292 #endif
293  default:
294  /* ignore everything else */
295  ;
296  }/*end switch*/
297 
298  buffer += fieldsize;
299  size -= fieldsize;
300  }/*wend*/
301  return true;
302 }

315 static bool handlePossibleHeaderBegin(const char *buffer, QIODevice *dev)
316 {
317  // we have to detect three magic tokens here:
318  // PK34 for the next local header in case there is no data descriptor
319  // PK12 for the central header in case there is no data descriptor
320  // PK78 for the data descriptor in case it is following the compressed data
321  // TODO: optimize using 32bit const data for comparison instead of byte-wise,
322  // given we run at least on 32bit CPUs
323 
324  if (buffer[0] == 'K') {
325  if (buffer[1] == 7 && buffer[2] == 8) {
326  // data descriptor token found
327  dev->seek(dev->pos() + 12); // skip the 'data_descriptor'
328  return true;
329  }
330 
331  if ((buffer[1] == 1 && buffer[2] == 2)
332  || (buffer[1] == 3 && buffer[2] == 4)) {
333  // central/local header token found
334  dev->seek(dev->pos() - 4);
335  // go back 4 bytes, so that the magic bytes can be found
336  // in the next cycle...
337  return true;
338  }
339  }
340  return false;
341 }

349 static bool seekToNextHeaderToken(QIODevice *dev)
350 {
351  bool headerTokenFound = false;
352  char buffer[3];
353 
354  while (!headerTokenFound) {
355  int n = dev->read(buffer, 1);
356  if (n < 1) {
357  //qWarning() << "Invalid ZIP file. Unexpected end of file. (#2)";
358  return false;
359  }
360 
361  if (buffer[0] != 'P') {
362  continue;
363  }
364 
365  n = dev->read(buffer, 3);
366  if (n < 3) {
367  //qWarning() << "Invalid ZIP file. Unexpected end of file. (#3)";
368  return false;
369  }
370 
371  if (handlePossibleHeaderBegin(buffer, dev)) {
372  headerTokenFound = true;
373  } else {
374  for (int i = 0; i < 3; ++i) {
375  if (buffer[i] == 'P') {
376  // We have another P character so we must go back a little to check if it is a magic
377  dev->seek(dev->pos() - 3 + i);
378  break;
379  }
380  }
381  }
382  }
383  return true;
384 }

437 bool KZip::openArchive(QIODevice::OpenMode mode)
438 {
439  //qDebug();
440  d->m_fileList.clear();
441 
442  if (mode == QIODevice::WriteOnly) {
443  return true;
444  }
445 
446  char buffer[47];
447 
448  // Check that it's a valid ZIP file
449  // KArchive::open() opened the underlying device already.
450 
451  quint64 offset = 0; // holds offset, where we read
452  int n;
453 
454  // contains information gathered from the local file headers
455  QHash<QByteArray, ParseFileInfo> pfi_map;
456 
457  QIODevice *dev = device();
458 
459  // We set a bool for knowing if we are allowed to skip the start of the file
460  bool startOfFile = true;
461 
462  for (;;) { // repeat until 'end of entries' signature is reached
463  //qDebug() << "loop starts";
464  //qDebug() << "dev->pos() now : " << dev->pos();
465  n = dev->read(buffer, 4);
466 
467  if (n < 4) {
468  //qWarning() << "Invalid ZIP file. Unexpected end of file. (#1)";
469  return false;
470  }
471 
472  if (!memcmp(buffer, "PK\5\6", 4)) { // 'end of entries'
473  //qDebug() << "PK56 found end of archive";
474  startOfFile = false;
475  break;
476  }
477 
478  if (!memcmp(buffer, "PK\3\4", 4)) { // local file header
479  //qDebug() << "PK34 found local file header";
480  startOfFile = false;
481  // can this fail ???
482  dev->seek(dev->pos() + 2); // skip 'version needed to extract'
483 
484  // read static header stuff
485  n = dev->read(buffer, 24);
486  if (n < 24) {
487  //qWarning() << "Invalid ZIP file. Unexpected end of file. (#4)";
488  return false;
489  }
490 
491  int gpf = (uchar)buffer[0]; // "general purpose flag" not "general protection fault" ;-)
492  int compression_mode = (uchar)buffer[2] | (uchar)buffer[3] << 8;
493  uint mtime = transformFromMsDos(buffer + 4);
494 
495  const qint64 compr_size = uint(uchar(buffer[12])) | uint(uchar(buffer[13])) << 8 |
496  uint(uchar(buffer[14])) << 16 | uint(uchar(buffer[15])) << 24;
497  const qint64 uncomp_size = uint(uchar(buffer[16])) | uint(uchar(buffer[17])) << 8 |
498  uint(uchar(buffer[18])) << 16 | uint(uchar(buffer[19])) << 24;
499  const int namelen = uint(uchar(buffer[20])) | uint(uchar(buffer[21])) << 8;
500  const int extralen = uint(uchar(buffer[22])) | uint(uchar(buffer[23])) << 8;
501 
502  /*
503  qDebug() << "general purpose bit flag: " << gpf;
504  qDebug() << "compressed size: " << compr_size;
505  qDebug() << "uncompressed size: " << uncomp_size;
506  qDebug() << "namelen: " << namelen;
507  qDebug() << "extralen: " << extralen;
508  qDebug() << "archive size: " << dev->size();
509  */
510 
511  // read fileName
512  Q_ASSERT(namelen > 0);
513  QByteArray fileName = dev->read(namelen);
514  if (fileName.size() < namelen) {
515  //qWarning() << "Invalid ZIP file. Name not completely read (#2)";
516  return false;
517  }
518 
519  ParseFileInfo pfi;
520  pfi.mtime = mtime;
521 
522  // read and parse the beginning of the extra field,
523  // skip rest of extra field in case it is too long
524  unsigned int extraFieldEnd = dev->pos() + extralen;
525  pfi.extralen = extralen;
526  int handledextralen = qMin(extralen, (int)sizeof buffer);
527 
528  //if (handledextralen)
529  // qDebug() << "handledextralen: " << handledextralen;
530 
531  n = dev->read(buffer, handledextralen);
532  // no error msg necessary as we deliberately truncate the extra field
533  if (!parseExtraField(buffer, handledextralen, true, pfi)) {
534  //qWarning() << "Invalid ZIP File. Broken ExtraField.";
535  return false;
536  }
537 
538  // jump to end of extra field
539  dev->seek(extraFieldEnd);
540 
541  // we have to take care of the 'general purpose bit flag'.
542  // if bit 3 is set, the header doesn't contain the length of
543  // the file and we look for the signature 'PK\7\8'.
544  if (gpf & 8) {
545  // here we have to read through the compressed data to find
546  // the next PKxx
547  if (!seekToNextHeaderToken(dev)) {
548  return false;
549  }
550  } else {
551  // here we skip the compressed data and jump to the next header
552  //qDebug() << "general purpose bit flag indicates, that local file header contains valid size";
553  bool foundSignature = false;
554  // check if this could be a symbolic link
555  if (compression_mode == NoCompression
556  && uncomp_size <= max_path_len
557  && uncomp_size > 0) {
558  // read content and store it
559  // If it's not a symlink, then we'll just discard the data for now.
560  pfi.guessed_symlink = dev->read(uncomp_size);
561  if (pfi.guessed_symlink.size() < uncomp_size) {
562  //qWarning() << "Invalid ZIP file. Unexpected end of file. (#5)";
563  return false;
564  }
565  } else {
566  if (compr_size > dev->size()) {
567  // here we cannot trust the compressed size, so scan through the compressed
568  // data to find the next header
569  if (!seekToNextHeaderToken(dev)) {
570  return false;
571  }
572  foundSignature = true;
573  } else {
574 // qDebug() << "before interesting dev->pos(): " << dev->pos();
575  bool success = dev->seek(dev->pos() + compr_size); // can this fail ???
576  Q_UNUSED(success); // prevent warning in release builds.
577  Q_ASSERT(success); // let's see...
578  /* qDebug() << "after interesting dev->pos(): " << dev->pos();
579  if (success)
580  qDebug() << "dev->at was successful... ";
581  else
582  qDebug() << "dev->at failed... ";*/
583  }
584 
585  }
586  // test for optional data descriptor
587  if (!foundSignature) {
588 // qDebug() << "Testing for optional data descriptor";
589  // read static data descriptor
590  n = dev->read(buffer, 4);
591  if (n < 4) {
592 // qWarning() << "Invalid ZIP file. Unexpected end of file. (#1)";
593  return false;
594  }
595 
596  if (buffer[0] != 'P' || !handlePossibleHeaderBegin(buffer + 1, dev)) {
597  // assume data descriptor without signature
598  dev->seek(dev->pos() + 8); // skip rest of the 'data_descriptor'
599  }
600  }
601 
602 // not needed any more
603  /* // here we calculate the length of the file in the zip
604  // with headers and jump to the next header.
605  uint skip = compr_size + namelen + extralen;
606  offset += 30 + skip;*/
607  }
608  pfi_map.insert(fileName, pfi);
609  } else if (!memcmp(buffer, "PK\1\2", 4)) { // central block
610  //qDebug() << "PK12 found central block";
611  startOfFile = false;
612 
613  // so we reached the central header at the end of the zip file
614  // here we get all interesting data out of the central header
615  // of a file
616  offset = dev->pos() - 4;
617 
618  //set offset for appending new files
619  if (d->m_offset == 0) {
620  d->m_offset = offset;
621  }
622 
623  n = dev->read(buffer + 4, 42);
624  if (n < 42) {
625  //qWarning() << "Invalid ZIP file, central entry too short"; // not long enough for valid entry
626  return false;
627  }
628 
629  //int gpf = (uchar)buffer[9] << 8 | (uchar)buffer[10];
630  //qDebug() << "general purpose flag=" << gpf;
631  // length of the fileName (well, pathname indeed)
632  int namelen = (uchar)buffer[29] << 8 | (uchar)buffer[28];
633  Q_ASSERT(namelen > 0);
634  QByteArray bufferName = dev->read(namelen);
635  if (bufferName.size() < namelen) {
636  //qWarning() << "Invalid ZIP file. Name not completely read";
637  }
638 
639  ParseFileInfo pfi = pfi_map.value(bufferName, ParseFileInfo());
640 
641  QString name(QFile::decodeName(bufferName));
642 
643  //qDebug() << "name: " << name;
644  // only in central header ! see below.
645  // length of extra attributes
646  int extralen = (uchar)buffer[31] << 8 | (uchar)buffer[30];
647  // length of comment for this file
648  int commlen = (uchar)buffer[33] << 8 | (uchar)buffer[32];
649  // compression method of this file
650  int cmethod = (uchar)buffer[11] << 8 | (uchar)buffer[10];
651 
652  //qDebug() << "cmethod: " << cmethod;
653  //qDebug() << "extralen: " << extralen;
654 
655  // crc32 of the file
656  uint crc32 = (uchar)buffer[19] << 24 | (uchar)buffer[18] << 16 |
657  (uchar)buffer[17] << 8 | (uchar)buffer[16];
658 
659  // uncompressed file size
660  uint ucsize = (uchar)buffer[27] << 24 | (uchar)buffer[26] << 16 |
661  (uchar)buffer[25] << 8 | (uchar)buffer[24];
662  // compressed file size
663  uint csize = (uchar)buffer[23] << 24 | (uchar)buffer[22] << 16 |
664  (uchar)buffer[21] << 8 | (uchar)buffer[20];
665 
666  // offset of local header
667  uint localheaderoffset = (uchar)buffer[45] << 24 | (uchar)buffer[44] << 16 |
668  (uchar)buffer[43] << 8 | (uchar)buffer[42];
669 
670  // some clever people use different extra field lengths
671  // in the central header and in the local header... funny.
672  // so we need to get the localextralen to calculate the offset
673  // from localheaderstart to dataoffset
674  int localextralen = pfi.extralen; // FIXME: this will not work if
675  // no local header exists
676 
677  //qDebug() << "localextralen: " << localextralen;
678 
679  // offset, where the real data for uncompression starts
680  uint dataoffset = localheaderoffset + 30 + localextralen + namelen; //comment only in central header
681 
682  //qDebug() << "esize: " << esize;
683  //qDebug() << "eoffset: " << eoffset;
684  //qDebug() << "csize: " << csize;
685 
686  int os_madeby = (uchar)buffer[5];
687  bool isdir = false;
688  int access = 0100644;
689 
690  if (os_madeby == 3) { // good ole unix
691  access = (uchar)buffer[40] | (uchar)buffer[41] << 8;
692  }
693 
694  QString entryName;
695 
696  if (name.endsWith(QLatin1Char('/'))) { // Entries with a trailing slash are directories
697  isdir = true;
698  name = name.left(name.length() - 1);
699  if (os_madeby != 3) {
700  access = S_IFDIR | 0755;
701  } else {
702  access |= S_IFDIR | 0700;
703  }
704  }
705 
706  int pos = name.lastIndexOf(QLatin1Char('/'));
707  if (pos == -1) {
708  entryName = name;
709  } else {
710  entryName = name.mid(pos + 1);
711  }
712  Q_ASSERT(!entryName.isEmpty());
713 
714  KArchiveEntry *entry;
715  if (isdir) {
716  QString path = QDir::cleanPath(name);
717  const KArchiveEntry *ent = rootDir()->entry(path);
718  if (ent && ent->isDirectory()) {
719  //qDebug() << "Directory already exists, NOT going to add it again";
720  entry = 0;
721  } else {
722  QDateTime mtime = KArchivePrivate::time_tToDateTime(pfi.mtime);
723  entry = new KArchiveDirectory(this, entryName, access, mtime, rootDir()->user(), rootDir()->group(), QString());
724  //qDebug() << "KArchiveDirectory created, entryName= " << entryName << ", name=" << name;
725  }
726  } else {
727  QString symlink;
728  if ((access & QT_STAT_MASK) == QT_STAT_LNK) {
729  symlink = QFile::decodeName(pfi.guessed_symlink);
730  }
731  QDateTime mtime = KArchivePrivate::time_tToDateTime(pfi.mtime);
732  entry = new KZipFileEntry(this, entryName, access, mtime,
733  rootDir()->user(), rootDir()->group(),
734  symlink, name, dataoffset,
735  ucsize, cmethod, csize);
736  static_cast<KZipFileEntry *>(entry)->setHeaderStart(localheaderoffset);
737  static_cast<KZipFileEntry *>(entry)->setCRC32(crc32);
738  //qDebug() << "KZipFileEntry created, entryName= " << entryName << ", name=" << name;
739  d->m_fileList.append(static_cast<KZipFileEntry *>(entry));
740  }
741 
742  if (entry) {
743  if (pos == -1) {
744  rootDir()->addEntry(entry);
745  } else {
746  // In some tar files we can find dir/./file => call cleanPath
747  QString path = QDir::cleanPath(name.left(pos));
748  // Ensure container directory exists, create otherwise
749  KArchiveDirectory *tdir = findOrCreate(path);
750  tdir->addEntry(entry);
751  }
752  }
753 
754  //calculate offset to next entry
755  offset += 46 + commlen + extralen + namelen;
756  bool b = dev->seek(offset);
757  Q_ASSERT(b);
758  if (!b) {
759  return false;
760  }
761  } else if (startOfFile) {
762  // The file does not start with any ZIP header (e.g. self-extractable ZIP files)
763  // Therefore we need to find the first PK\003\004 (local header)
764  //qDebug() << "Try to skip start of file";
765  startOfFile = false;
766  bool foundSignature = false;
767 
768  while (!foundSignature) {
769  n = dev->read(buffer, 1);
770  if (n < 1) {
771  //qWarning() << "Invalid ZIP file. Unexpected end of file. " ;
772  return false;
773  }
774 
775  if (buffer[0] != 'P') {
776  continue;
777  }
778 
779  n = dev->read(buffer, 3);
780  if (n < 3) {
781  //qWarning() << "Invalid ZIP file. Unexpected end of file. " ;
782  return false;
783  }
784 
785  // We have to detect the magic token for a local header: PK\003\004
786  /*
787  * Note: we do not need to check the other magics, if the ZIP file has no
788  * local header, then it has not any files!
789  */
790  if (buffer[0] == 'K' && buffer[1] == 3 && buffer[2] == 4) {
791  foundSignature = true;
792  dev->seek(dev->pos() - 4); // go back 4 bytes, so that the magic bytes can be found...
793  } else {
794  for (int i = 0; i < 3; ++i) {
795  if (buffer[i] == 'P') {
796  // We have another P character so we must go back a little to check if it is a magic
797  dev->seek(dev->pos() - 3 + i);
798  break;
799  }
800  }
801  }
802  }
803  } else {
804  //qWarning() << "Invalid ZIP file. Unrecognized header at offset " << offset;
805 
806  return false;
807  }
808  }
809  //qDebug() << "*** done *** ";
810  return true;
811 }

933  if (d->m_extraField == ModificationTime) {

965  buffer[3] = 6;

968  buffer[5] = 0;
969 
970  buffer[6] = 0; // number of disk with start of central dir
971  buffer[7] = 0;
972

974  //qDebug() << "number of files (count): " << count;
975 
976  buffer[8] = char(count); // total number of entries in central dir of
977  buffer[9] = char(count >> 8); // this disk
978 
979  buffer[10] = buffer[8]; // total number of entries in the central dir
980  buffer[11] = buffer[9];
981 
982  int cdsize = centraldirendoffset - centraldiroffset;
983  buffer[12] = char(cdsize); // size of the central dir
984  buffer[13] = char(cdsize >> 8);
985  buffer[14] = char(cdsize >> 16);
986  buffer[15] = char(cdsize >> 24);
987

990

992  buffer[17] = char(centraldiroffset >> 8);
993  buffer[18] = char(centraldiroffset >> 16);
994  buffer[19] = char(centraldiroffset >> 24);
995 
996  buffer[20] = 0; //zipfile comment length
997  buffer[21] = 0;
998 
999  if (device()->write(buffer, 22) != 22) {
1000  return false;
1001  }
1002 
1003  return true;
