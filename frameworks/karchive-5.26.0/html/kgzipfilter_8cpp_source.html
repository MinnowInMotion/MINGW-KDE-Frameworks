81 bool KGzipFilter::init(int mode, Flag flag)
82 {
83  if (d->isInitialized) {
84  terminate();
85  }
86  d->zStream.next_in = Z_NULL;
87  d->zStream.avail_in = 0;
88  if (mode == QIODevice::ReadOnly) {
89  const int windowBits = (flag == RawDeflate)
90  ? -MAX_WBITS /*no zlib header*/
91  : (flag == GZipHeader) ?
92  MAX_WBITS + 32 /* auto-detect and eat gzip header */
93  : MAX_WBITS /*zlib header*/;
94  const int result = inflateInit2(&d->zStream, windowBits);
95  if (result != Z_OK) {
96  //qDebug() << "inflateInit2 returned " << result;
97  return false;
98  }
99  } else if (mode == QIODevice::WriteOnly) {
100  int result = deflateInit2(&d->zStream, Z_DEFAULT_COMPRESSION, Z_DEFLATED, -MAX_WBITS, 8, Z_DEFAULT_STRATEGY); // same here
101  if (result != Z_OK) {
102  //qDebug() << "deflateInit returned " << result;
103  return false;
104  }
105  } else {
106  //qWarning() << "KGzipFilter: Unsupported mode " << mode << ". Only QIODevice::ReadOnly and QIODevice::WriteOnly supported";
107  return false;
108  }
109  d->mode = mode;
110  d->compressed = true;
111  d->headerWritten = false;
112  d->footerWritten = false;
113  d->isInitialized = true;
114  return true;
115 }

160 bool KGzipFilter::readHeader()
161 {
162  // We now rely on zlib to read the full header (see the MAX_WBITS + 32 in init).
163  // We just use this method to check if the data is actually compressed.
164 
165 #ifdef DEBUG_GZIP
166  qDebug() << "avail=" << d->zStream.avail_in;
167 #endif
168  // Assume not compressed until we see a gzip header
169  d->compressed = false;
170  const Bytef *p = d->zStream.next_in;
171  int i = d->zStream.avail_in;
172  if ((i -= 10) < 0) {
173  return false; // Need at least 10 bytes
174  }
175 #ifdef DEBUG_GZIP
176  qDebug() << "first byte is " << QString::number(*p, 16);
177 #endif
178  if (*p++ != 0x1f) {
179  return false; // GZip magic
180  }
181 #ifdef DEBUG_GZIP
182  qDebug() << "second byte is " << QString::number(*p, 16);
183 #endif
184  if (*p++ != 0x8b) {
185  return false;
186  }
187 
188  d->compressed = true;
189 #ifdef DEBUG_GZIP
190  qDebug() << "header OK";
191 #endif
192  return true;
193 }
194 
195 /* Output a 16 bit value, lsb first */
196 #define put_short(w) \
197  *p++ = uchar((w) & 0xff); \
198  *p++ = uchar(ushort(w) >> 8);
199 
200 /* Output a 32 bit value to the bit stream, lsb first */
201 #define put_long(n) \
202  put_short((n) & 0xffff); \
203  put_short((ulong(n)) >> 16);

205 bool KGzipFilter::writeHeader(const QByteArray &fileName)
206 {
207  Bytef *p = d->zStream.next_out;
208  int i = d->zStream.avail_out;
209  *p++ = 0x1f;
210  *p++ = 0x8b;
211  *p++ = Z_DEFLATED;
212  *p++ = ORIG_NAME;
213  put_long(time(0L)); // Modification time (in unix format)
214  *p++ = 0; // Extra flags (2=max compress, 4=fastest compress)
215  *p++ = 3; // Unix
216 
217  uint len = fileName.length();
218  for (uint j = 0; j < len; ++j) {
219  *p++ = fileName[j];
220  }
221  *p++ = 0;
222  int headerSize = p - d->zStream.next_out;
223  i -= headerSize;
224  Q_ASSERT(i > 0);
225  d->crc = crc32(0L, Z_NULL, 0);
226  d->zStream.next_out = p;
227  d->zStream.avail_out = i;
228  d->headerWritten = true;
229  return true;
230 }

232 void KGzipFilter::writeFooter()
233 {
234  Q_ASSERT(d->headerWritten);
235  Q_ASSERT(!d->footerWritten);
236  Bytef *p = d->zStream.next_out;
237  int i = d->zStream.avail_out;
238  //qDebug() << "avail_out=" << i << "writing CRC=" << QString::number(d->crc, 16) << "at p=" << p;
239  put_long(d->crc);
240  //qDebug() << "writing totalin=" << d->zStream.total_in << "at p=" << p;
241  put_long(d->zStream.total_in);
242  i -= p - d->zStream.next_out;
243  d->zStream.next_out = p;
244  d->zStream.avail_out = i;
245  d->footerWritten = true;
246 }

270 KGzipFilter::Result KGzipFilter::uncompress_noop()
271 {
272  // I'm not sure we really need support for that (uncompressed streams),
273  // but why not, it can't hurt to have it. One case I can think of is someone
274  // naming a tar file "blah.tar.gz" :-)
275  if (d->zStream.avail_in > 0) {
276  int n = (d->zStream.avail_in < d->zStream.avail_out) ? d->zStream.avail_in : d->zStream.avail_out;
277  memcpy(d->zStream.next_out, d->zStream.next_in, n);
278  d->zStream.avail_out -= n;
279  d->zStream.next_in += n;
280  d->zStream.avail_in -= n;
281  return KFilterBase::Ok;
282  } else {
283  return KFilterBase::End;
284  }
285 }

287 KGzipFilter::Result KGzipFilter::uncompress()
288 {
289 #ifndef NDEBUG
290  if (d->mode == 0) {
291  //qWarning() << "mode==0; KGzipFilter::init was not called!";
292  return KFilterBase::Error;
293  } else if (d->mode == QIODevice::WriteOnly) {
294  //qWarning() << "uncompress called but the filter was opened for writing!";
295  return KFilterBase::Error;
296  }
297  Q_ASSERT(d->mode == QIODevice::ReadOnly);
298 #endif
299 
300  if (!d->compressed) {
301  return uncompress_noop();
302  }
303 
304 #ifdef DEBUG_GZIP
305  qDebug() << "Calling inflate with avail_in=" << inBufferAvailable() << " avail_out=" << outBufferAvailable();
306  qDebug() << " next_in=" << d->zStream.next_in;
307 #endif
308 
309  while (d->zStream.avail_in > 0) {
310  int result = inflate(&d->zStream, Z_SYNC_FLUSH);
311 
312 #ifdef DEBUG_GZIP
313  qDebug() << " -> inflate returned " << result;
314  qDebug() << " now avail_in=" << inBufferAvailable() << " avail_out=" << outBufferAvailable();
315  qDebug() << " next_in=" << d->zStream.next_in;
316 #endif
317 
318  if (result == Z_OK) {
319  return KFilterBase::Ok;
320  }
321 
322  // We can't handle any other results
323  if (result != Z_STREAM_END) {
324  return KFilterBase::Error;
325  }
326 
327  // It really was the end
328  if (d->zStream.avail_in == 0) {
329  return KFilterBase::End;
330  }
331 
332  // Store before resetting
333  Bytef *data = d->zStream.next_in; // This is increased appropriately by zlib beforehand
334  uInt size = d->zStream.avail_in;
335 
336  // Reset the stream, if that fails we assume we're at the end
337  if (!init(d->mode)) {
338  return KFilterBase::End;
339  }
340 
341  // Reset the data to where we left off
342  d->zStream.next_in = data;
343  d->zStream.avail_in = size;
344  }
345 
346  return KFilterBase::End;
347 }

349 KGzipFilter::Result KGzipFilter::compress(bool finish)
350 {
351  Q_ASSERT(d->compressed);
352  Q_ASSERT(d->mode == QIODevice::WriteOnly);
353 
354  const Bytef *p = d->zStream.next_in;
355  ulong len = d->zStream.avail_in;
356 #ifdef DEBUG_GZIP
357  qDebug() << " calling deflate with avail_in=" << inBufferAvailable() << " avail_out=" << outBufferAvailable();
358 #endif
359  const int result = deflate(&d->zStream, finish ? Z_FINISH : Z_NO_FLUSH);
360  if (result != Z_OK && result != Z_STREAM_END) {
361  //qDebug() << " deflate returned " << result;
362  }
363  if (d->headerWritten) {
364  //qDebug() << "Computing CRC for the next " << len - d->zStream.avail_in << " bytes";
365  d->crc = crc32(d->crc, p, len - d->zStream.avail_in);
366  }
367  KGzipFilter::Result callerResult = result == Z_OK ? KFilterBase::Ok : (Z_STREAM_END ? KFilterBase::End : KFilterBase::Error);
368 
369  if (result == Z_STREAM_END && d->headerWritten && !d->footerWritten) {
370  if (d->zStream.avail_out >= 8 /*footer size*/) {
371  //qDebug() << "finished, write footer";
372  writeFooter();
373  } else {
374  // No room to write the footer (#157706/#188415), we'll have to do it on the next pass.
375  //qDebug() << "finished, but no room for footer yet";
376  callerResult = KFilterBase::Ok;
377  }
378  }
379  return callerResult;
380 }
