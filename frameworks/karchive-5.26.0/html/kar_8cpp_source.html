83 bool KAr::openArchive(QIODevice::OpenMode mode)
84 {
85  // Open archive
86 
87  if (mode == QIODevice::WriteOnly) {
88  return true;
89  }
90  if (mode != QIODevice::ReadOnly && mode != QIODevice::ReadWrite) {
91  //qWarning() << "Unsupported mode " << mode;
92  return false;
93  }
94 
95  QIODevice *dev = device();
96  if (!dev) {
97  return false;
98  }
99 
100  QByteArray magic = dev->read(7);
101  if (magic != "!<arch>") {
102  //qWarning() << "Invalid main magic";
103  return false;
104  }
105 
106  char *ar_longnames = 0;
107  while (! dev->atEnd()) {
108  QByteArray ar_header;
109  ar_header.resize(61);
110 
111  dev->seek(dev->pos() + (2 - (dev->pos() % 2)) % 2); // Ar headers are padded to byte boundary
112 
113  if (dev->read(ar_header.data(), 60) != 60) { // Read ar header
114  //qWarning() << "Couldn't read header";
115  delete[] ar_longnames;
116  //return false;
117  return true; // Probably EOF / trailing junk
118  }
119 
120  if (!ar_header.endsWith("`\n")) { // Check header magic // krazy:exclude=strings
121  //qWarning() << "Invalid magic";
122  delete[] ar_longnames;
123  return false;
124  }
125 
126  QByteArray name = ar_header.mid(0, 16); // Process header
127  const int date = ar_header.mid(16, 12).toInt();
128  //const int uid = ar_header.mid( 28, 6 ).toInt();
129  //const int gid = ar_header.mid( 34, 6 ).toInt();
130  const int mode = ar_header.mid(40, 8).toInt();
131  const qint64 size = ar_header.mid(48, 10).toInt();
132 
133  bool skip_entry = false; // Deal with special entries
134  if (name.mid(0, 1) == "/") {
135  if (name.mid(1, 1) == "/") { // Longfilename table entry
136  delete[] ar_longnames;
137  ar_longnames = new char[size + 1];
138  ar_longnames[size] = '\0';
139  dev->read(ar_longnames, size);
140  skip_entry = true;
141  //qDebug() << "Read in longnames entry";
142  } else if (name.mid(1, 1) == " ") { // Symbol table entry
143  //qDebug() << "Skipped symbol entry";
144  dev->seek(dev->pos() + size);
145  skip_entry = true;
146  } else { // Longfilename
147  //qDebug() << "Longfilename #" << name.mid(1, 15).toInt();
148  if (! ar_longnames) {
149  //qWarning() << "Invalid longfilename reference";
150  delete[] ar_longnames;
151  return false;
152  }
153  name = &ar_longnames[name.mid(1, 15).toInt()];
154  name = name.left(name.indexOf("/"));
155  }
156  }
157  if (skip_entry) {
158  continue;
159  }
160 
161  name = name.trimmed(); // Process filename
162  name.replace('/', QByteArray());
163  //qDebug() << "Filename: " << name << " Size: " << size;
164 
165  KArchiveEntry *entry = new KArchiveFile(this, QString::fromLocal8Bit(name.constData()), mode, KArchivePrivate::time_tToDateTime(date),
166  rootDir()->user(), rootDir()->group(), /*symlink*/ QString(),
167  dev->pos(), size);
168  rootDir()->addEntry(entry); // Ar files don't support directories, so everything in root
169 
170  dev->seek(dev->pos() + size); // Skip contents
171  }
172  delete[] ar_longnames;
173 
174  return true;
175 }
