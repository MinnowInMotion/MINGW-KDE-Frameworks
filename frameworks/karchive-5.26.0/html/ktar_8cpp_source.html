86 // Only called when a filename was given
87 bool KTar::createDevice(QIODevice::OpenMode mode)
88 {
89  if (d->mimetype.isEmpty()) {
90  // Find out mimetype manually
91 
92  QMimeDatabase db;
93  QMimeType mime;
94  if (mode != QIODevice::WriteOnly && QFile::exists(fileName())) {
95  // Give priority to file contents: if someone renames a .tar.bz2 to .tar.gz,
96  // we can still do the right thing here.
97  QFile f(fileName());
98  if (f.open(QIODevice::ReadOnly)) {
99  mime = db.mimeTypeForData(&f);
100  }
101  if (!mime.isValid()) {
102  // Unable to determine mimetype from contents, get it from file name
103  mime = db.mimeTypeForFile(fileName(), QMimeDatabase::MatchExtension);
104  }
105  } else {
106  mime = db.mimeTypeForFile(fileName(), QMimeDatabase::MatchExtension);
107  }
108 
109  //qDebug() << mode << mime->name();
110 
111  if (mime.inherits(QStringLiteral("application/x-compressed-tar")) || mime.inherits(QString::fromLatin1(application_gzip))) {
112  // gzipped tar file (with possibly invalid file name), ask for gzip filter
113  d->mimetype = QString::fromLatin1(application_gzip);
114  } else if (mime.inherits(QStringLiteral("application/x-bzip-compressed-tar")) || mime.inherits(QString::fromLatin1(application_bzip))) {
115  // bzipped2 tar file (with possibly invalid file name), ask for bz2 filter
116  d->mimetype = QString::fromLatin1(application_bzip);
117  } else if (mime.inherits(QStringLiteral("application/x-lzma-compressed-tar")) || mime.inherits(QString::fromLatin1(application_lzma))) {
118  // lzma compressed tar file (with possibly invalid file name), ask for xz filter
119  d->mimetype = QString::fromLatin1(application_lzma);
120  } else if (mime.inherits(QStringLiteral("application/x-xz-compressed-tar")) || mime.inherits(QString::fromLatin1(application_xz))) {
121  // xz compressed tar file (with possibly invalid name), ask for xz filter
122  d->mimetype = QString::fromLatin1(application_xz);
123  }
124  }
125 
126  if (d->mimetype == QLatin1String("application/x-tar")) {
127  return KArchive::createDevice(mode);
128  } else if (mode == QIODevice::WriteOnly) {
129  if (!KArchive::createDevice(mode)) {
130  return false;
131  }
132  if (!d->mimetype.isEmpty()) {
133  // Create a compression filter on top of the QSaveFile device that KArchive created.
134  //qDebug() << "creating KFilterDev for" << d->mimetype;
135  KCompressionDevice::CompressionType type = KFilterDev::compressionTypeForMimeType(d->mimetype);
136  KCompressionDevice *compressionDevice = new KCompressionDevice(device(), true, type);
137  setDevice(compressionDevice);
138  }
139  return true;
140  } else {
141  // The compression filters are very slow with random access.
142  // So instead of applying the filter to the device,
143  // the file is completely extracted instead,
144  // and we work on the extracted tar file.
145  // This improves the extraction speed by the tar ioslave dramatically,
146  // if the archive file contains many files.
147  // This is because the tar ioslave extracts one file after the other and normally
148  // has to walk through the decompression filter each time.
149  // Which is in fact nearly as slow as a complete decompression for each file.
150 
151  Q_ASSERT(!d->tmpFile);
152  d->tmpFile = new QTemporaryFile();
153  d->tmpFile->setFileTemplate(QDir::tempPath() + QStringLiteral("/") + QLatin1String("ktar-XXXXXX.tar"));
154  d->tmpFile->open();
155  //qDebug() << "creating tempfile:" << d->tmpFile->fileName();
156 
157  setDevice(d->tmpFile);
158  return true;
159  }
160 }

182 qint64 KTar::KTarPrivate::readRawHeader(char *buffer)
183 {
184  // Read header
185  qint64 n = q->device()->read(buffer, 0x200);
186  // we need to test if there is a prefix value because the file name can be null
187  // and the prefix can have a value and in this case we don't reset n.
188  if (n == 0x200 && (buffer[0] != 0 || buffer[0x159] != 0)) {
189  // Make sure this is actually a tar header
190  if (strncmp(buffer + 257, "ustar", 5)) {
191  // The magic isn't there (broken/old tars), but maybe a correct checksum?
192 
193  int check = 0;
194  for (uint j = 0; j < 0x200; ++j) {
195  check += buffer[j];
196  }
197 
198  // adjust checksum to count the checksum fields as blanks
199  for (uint j = 0; j < 8 /*size of the checksum field including the \0 and the space*/; j++) {
200  check -= buffer[148 + j];
201  }
202  check += 8 * ' ';
203 
204  QByteArray s = QByteArray::number(check, 8); // octal
205 
206  // only compare those of the 6 checksum digits that mean something,
207  // because the other digits are filled with all sorts of different chars by different tars ...
208  // Some tars right-justify the checksum so it could start in one of three places - we have to check each.
209  if (strncmp(buffer + 148 + 6 - s.length(), s.data(), s.length())
210  && strncmp(buffer + 148 + 7 - s.length(), s.data(), s.length())
211  && strncmp(buffer + 148 + 8 - s.length(), s.data(), s.length())) {
212  /*qWarning() << "KTar: invalid TAR file. Header is:" << QByteArray( buffer+257, 5 )
213  << "instead of ustar. Reading from wrong pos in file?"
214  << "checksum=" << QByteArray( buffer + 148 + 6 - s.length(), s.length() );*/
215  return -1;
216  }
217  }/*end if*/
218  } else {
219  // reset to 0 if 0x200 because logical end of archive has been reached
220  if (n == 0x200) {
221  n = 0;
222  }
223  }/*end if*/
224  return n;
225 }

227 bool KTar::KTarPrivate::readLonglink(char *buffer, QByteArray &longlink)
228 {
229  qint64 n = 0;
230  //qDebug() << "reading longlink from pos " << q->device()->pos();
231  QIODevice *dev = q->device();
232  // read size of longlink from size field in header
233  // size is in bytes including the trailing null (which we ignore)
234  qint64 size = QByteArray(buffer + 0x7c, 12).trimmed().toLongLong(0, 8 /*octal*/);
235 
236  size--; // ignore trailing null
237  longlink.resize(size);
238  qint64 offset = 0;
239  while (size > 0) {
240  int chunksize = qMin(size, 0x200LL);
241  n = dev->read(longlink.data() + offset, chunksize);
242  if (n == -1) {
243  return false;
244  }
245  size -= chunksize;
246  offset += 0x200;
247  }/*wend*/
248  // jump over the rest
249  const int skip = 0x200 - (n % 0x200);
250  if (skip <= 0x200) {
251  if (dev->read(buffer, skip) != skip) {
252  return false;
253  }
254  }
255  longlink.truncate(qstrlen(longlink.constData()));
256  return true;
257 }

259 qint64 KTar::KTarPrivate::readHeader(char *buffer, QString &name, QString &symlink)
260 {
261  name.truncate(0);
262  symlink.truncate(0);
263  while (true) {
264  qint64 n = readRawHeader(buffer);
265  if (n != 0x200) {
266  return n;
267  }
268 
269  // is it a longlink?
270  if (strcmp(buffer, "././@LongLink") == 0) {
271  char typeflag = buffer[0x9c];
272  QByteArray longlink;
273  readLonglink(buffer, longlink);
274  switch (typeflag) {
275  case 'L':
276  name = QFile::decodeName(longlink.constData());
277  break;
278  case 'K':
279  symlink = QFile::decodeName(longlink.constData());
280  break;
281  }/*end switch*/
282  } else {
283  break;
284  }/*end if*/
285  }/*wend*/
286 
287  // if not result of longlink, read names directly from the header
288  if (name.isEmpty())
289  // there are names that are exactly 100 bytes long
290  // and neither longlink nor \0 terminated (bug:101472)
291  {
292  name = QFile::decodeName(QByteArray(buffer, qstrnlen(buffer, 100)));
293  }
294  if (symlink.isEmpty()) {
295  char *symlinkBuffer = buffer + 0x9d /*?*/;
296  symlink = QFile::decodeName(QByteArray(symlinkBuffer, qstrnlen(symlinkBuffer, 100)));
297  }
298 
299  return 0x200;
300 }

307 bool KTar::KTarPrivate::fillTempFile(const QString &fileName)
308 {
309  if (! tmpFile) {
310  return true;
311  }
312 
313  //qDebug() << "filling tmpFile of mimetype" << mimetype;
314 
315  KCompressionDevice::CompressionType compressionType = KFilterDev::compressionTypeForMimeType(mimetype);
316  KCompressionDevice filterDev(fileName, compressionType);
317 
318  QFile *file = tmpFile;
319  Q_ASSERT(file->isOpen());
320  Q_ASSERT(file->openMode() & QIODevice::WriteOnly);
321  file->seek(0);
322  QByteArray buffer;
323  buffer.resize(8 * 1024);
324  if (! filterDev.open(QIODevice::ReadOnly)) {
325  return false;
326  }
327  qint64 len = -1;
328  while (!filterDev.atEnd() && len != 0) {
329  len = filterDev.read(buffer.data(), buffer.size());
330  if (len < 0) { // corrupted archive
331  return false;
332  }
333  if (file->write(buffer.data(), len) != len) { // disk full
334  return false;
335  }
336  }
337  filterDev.close();
338 
339  file->flush();
340  file->seek(0);
341  Q_ASSERT(file->isOpen());
342  Q_ASSERT(file->openMode() & QIODevice::ReadOnly);
343 
344  //qDebug() << "filling tmpFile finished.";
345  return true;
346 }

348 bool KTar::openArchive(QIODevice::OpenMode mode)
349 {
350 
351  if (!(mode & QIODevice::ReadOnly)) {
352  return true;
353  }
354 
355  if (!d->fillTempFile(fileName())) {
356  return false;
357  }
358 
359  // We'll use the permission and user/group of d->rootDir
360  // for any directory we emulate (see findOrCreate)
361  //struct stat buf;
362  //stat( fileName(), &buf );
363 
364  d->dirList.clear();
365  QIODevice *dev = device();
366 
367  if (!dev) {
368  return false;
369  }
370 
371  // read dir information
372  char buffer[0x200];
373  bool ende = false;
374  do {
375  QString name;
376  QString symlink;
377 
378  // Read header
379  qint64 n = d->readHeader(buffer, name, symlink);
380  if (n < 0) {
381  return false;
382  }
383  if (n == 0x200) {
384  bool isdir = false;
385 
386  if (name.endsWith(QLatin1Char('/'))) {
387  isdir = true;
388  name.truncate(name.length() - 1);
389  }
390 
391  QByteArray prefix = QByteArray(buffer + 0x159, 155);
392  if (prefix[0] != '\0') {
393  name = (QString::fromLatin1(prefix.constData()) + QLatin1Char('/') + name);
394  }
395 
396  int pos = name.lastIndexOf(QLatin1Char('/'));
397  QString nm = (pos == -1) ? name : name.mid(pos + 1);
398 
399  // read access
400  buffer[0x6b] = 0;
401  char *dummy;
402  const char *p = buffer + 0x64;
403  while (*p == ' ') {
404  ++p;
405  }
406  int access = strtol(p, &dummy, 8);
407 
408  // read user and group
409  QString user = QString::fromLocal8Bit(buffer + 0x109);
410  QString group = QString::fromLocal8Bit(buffer + 0x129);
411 
412  // read time
413  buffer[0x93] = 0;
414  p = buffer + 0x88;
415  while (*p == ' ') {
416  ++p;
417  }
418  uint time = strtol(p, &dummy, 8);
419 
420  // read type flag
421  char typeflag = buffer[0x9c];
422  // '0' for files, '1' hard link, '2' symlink, '5' for directory
423  // (and 'L' for longlink fileNames, 'K' for longlink symlink targets)
424  // 'D' for GNU tar extension DUMPDIR, 'x' for Extended header referring
425  // to the next file in the archive and 'g' for Global extended header
426 
427  if (typeflag == '5') {
428  isdir = true;
429  }
430 
431  bool isDumpDir = false;
432  if (typeflag == 'D') {
433  isdir = false;
434  isDumpDir = true;
435  }
436  //qDebug() << nm << "isdir=" << isdir << "pos=" << dev->pos() << "typeflag=" << typeflag << " islink=" << ( typeflag == '1' || typeflag == '2' );
437 
438  if (typeflag == 'x' || typeflag == 'g') { // pax extended header, or pax global extended header
439  // Skip it for now. TODO: implement reading of extended header, as per http://pubs.opengroup.org/onlinepubs/009695399/utilities/pax.html
440  (void)dev->read(buffer, 0x200);
441  continue;
442  }
443 
444  if (isdir) {
445  access |= S_IFDIR; // f*cking broken tar files
446  }
447 
448  KArchiveEntry *e;
449  if (isdir) {
450  //qDebug() << "directory" << nm;
451  e = new KArchiveDirectory(this, nm, access, KArchivePrivate::time_tToDateTime(time), user, group, symlink);
452  } else {
453  // read size
454  QByteArray sizeBuffer(buffer + 0x7c, 12);
455  qint64 size = sizeBuffer.trimmed().toLongLong(0, 8 /*octal*/);
456  //qDebug() << "sizeBuffer='" << sizeBuffer << "' -> size=" << size;
457 
458  // for isDumpDir we will skip the additional info about that dirs contents
459  if (isDumpDir) {
460  //qDebug() << nm << "isDumpDir";
461  e = new KArchiveDirectory(this, nm, access, KArchivePrivate::time_tToDateTime(time), user, group, symlink);
462  } else {
463 
464  // Let's hack around hard links. Our classes don't support that, so make them symlinks
465  if (typeflag == '1') {
466  //qDebug() << "Hard link, setting size to 0 instead of" << size;
467  size = 0; // no contents
468  }
469 
470  //qDebug() << "file" << nm << "size=" << size;
471  e = new KArchiveFile(this, nm, access, KArchivePrivate::time_tToDateTime(time), user, group, symlink,
472  dev->pos(), size);
473  }
474 
475  // Skip contents + align bytes
476  qint64 rest = size % 0x200;
477  qint64 skip = size + (rest ? 0x200 - rest : 0);
478  //qDebug() << "pos()=" << dev->pos() << "rest=" << rest << "skipping" << skip;
479  if (! dev->seek(dev->pos() + skip)) {
480  //qWarning() << "skipping" << skip << "failed";
481  }
482  }
483 
484  if (pos == -1) {
485  if (nm == QLatin1String(".")) { // special case
486  Q_ASSERT(isdir);
487  if (isdir) {
488  setRootDir(static_cast<KArchiveDirectory *>(e));
489  }
490  } else {
491  rootDir()->addEntry(e);
492  }
493  } else {
494  // In some tar files we can find dir/./file => call cleanPath
495  QString path = QDir::cleanPath(name.left(pos));
496  // Ensure container directory exists, create otherwise
497  KArchiveDirectory *d = findOrCreate(path);
498  d->addEntry(e);
499  }
500  } else {
501  //qDebug("Terminating. Read %d bytes, first one is %d", n, buffer[0]);
502  d->tarEnd = dev->pos() - n; // Remember end of archive
503  ende = true;
504  }
505  } while (!ende);
506  return true;
507 }

514 bool KTar::KTarPrivate::writeBackTempFile(const QString &fileName)
515 {
516  if (!tmpFile) {
517  return true;
518  }
519 
520  //qDebug() << "Write temporary file to compressed file" << fileName << mimetype;
521 
522  bool forced = false;
523  if (QLatin1String(application_gzip) == mimetype || QLatin1String(application_bzip) == mimetype ||
524  QLatin1String(application_lzma) == mimetype || QLatin1String(application_xz) == mimetype) {

526  }

539  dev.setOrigFileName(origFileName);
540  }
541  file->seek(0);
542  QByteArray buffer;
543  buffer.resize(8 * 1024);
544  qint64 len;
545  while (!file->atEnd()) {
546  len = file->read(buffer.data(), buffer.size());
547  dev.write(buffer.data(), len); // TODO error checking
548  }
549  file->close();
550  dev.close();
551 
552  //qDebug() << "Write temporary file to compressed file done.";
553  return true;

575 bool KTar::doFinishWriting(qint64 size)
576 {
577  // Write alignment
578  int rest = size % 0x200;
579  if ((mode() & QIODevice::ReadWrite) == QIODevice::ReadWrite) {
580  d->tarEnd = device()->pos() + (rest ? 0x200 - rest : 0); // Record our new end of archive
581  }
582  if (rest) {
583  char buffer[0x201];
584  for (uint i = 0; i < 0x200; ++i) {
585  buffer[i] = 0;
586  }
587  qint64 nwritten = device()->write(buffer, 0x200 - rest);
588  return nwritten == 0x200 - rest;
589  }
590  return true;

616 void KTar::KTarPrivate::fillBuffer(char *buffer,
617  const char *mode, qint64 size, const QDateTime &mtime, char typeflag,
618  const char *uname, const char *gname)
619 {
620  // mode (as in stpos())
621  assert(strlen(mode) == 6);
622  memcpy(buffer + 0x64, mode, 6);
623  buffer[0x6a] = ' ';
624  buffer[0x6b] = '\0';
625 
626  // dummy uid
627  strcpy(buffer + 0x6c, " 765 "); // 501 in decimal
628  // dummy gid
629  strcpy(buffer + 0x74, " 144 "); // 100 in decimal
630 
631  // size
632  QByteArray s = QByteArray::number(size, 8); // octal
633  s = s.rightJustified(11, '0');
634  memcpy(buffer + 0x7c, s.data(), 11);
635  buffer[0x87] = ' '; // space-terminate (no null after)
636 
637  // modification time
638  const QDateTime modificationTime = mtime.isValid() ? mtime : QDateTime::currentDateTime();
639  s = QByteArray::number(static_cast<qulonglong>(modificationTime.toMSecsSinceEpoch() / 1000), 8); // octal
640  s = s.rightJustified(11, '0');
641  memcpy(buffer + 0x88, s.data(), 11);
642  buffer[0x93] = ' '; // space-terminate (no null after) -- well current tar writes a null byte
643 
644  // spaces, replaced by the check sum later
645  buffer[0x94] = 0x20;
646  buffer[0x95] = 0x20;
647  buffer[0x96] = 0x20;
648  buffer[0x97] = 0x20;
649  buffer[0x98] = 0x20;
650  buffer[0x99] = 0x20;
651 
652  /* From the tar sources :
653  Fill in the checksum field. It's formatted differently from the
654  other fields: it has [6] digits, a null, then a space -- rather than
655  digits, a space, then a null. */
656 
657  buffer[0x9a] = '\0';
658  buffer[0x9b] = ' ';
659 
660  // type flag (dir, file, link)
661  buffer[0x9c] = typeflag;
662 
663  // magic + version
664  strcpy(buffer + 0x101, "ustar");
665  strcpy(buffer + 0x107, "00");
666 
667  // user
668  strcpy(buffer + 0x109, uname);
669  // group
670  strcpy(buffer + 0x129, gname);
671 
672  // Header check sum
673  int check = 32;
674  for (uint j = 0; j < 0x200; ++j) {
675  check += buffer[j];
676  }
677  s = QByteArray::number(check, 8); // octal
678  s = s.rightJustified(6, '0');
679  memcpy(buffer + 0x94, s.constData(), 6);
680 }

682 void KTar::KTarPrivate::writeLonglink(char *buffer, const QByteArray &name, char typeflag,
683  const char *uname, const char *gname)
684 {
685  strcpy(buffer, "././@LongLink");
686  qint64 namelen = name.length() + 1;
687  fillBuffer(buffer, " 0", namelen, QDateTime(), typeflag, uname, gname);
688  q->device()->write(buffer, 0x200); // TODO error checking
689  qint64 offset = 0;
690  while (namelen > 0) {
691  int chunksize = qMin(namelen, 0x200LL);
692  memcpy(buffer, name.data() + offset, chunksize);
693  // write long name
694  q->device()->write(buffer, 0x200); // TODO error checking
695  // not even needed to reclear the buffer, tar doesn't do it
696  namelen -= chunksize;
697  offset += 0x200;
698  }/*wend*/
699 }

701 bool KTar::doPrepareWriting(const QString &name, const QString &user,
702  const QString &group, qint64 size, mode_t perm,
703  const QDateTime & /*atime*/, const QDateTime &mtime, const QDateTime & /*ctime*/)
704 {
705  if (!isOpen()) {
706  //qWarning() << "You must open the tar file before writing to it\n";
707  return false;
708  }
709 
710  if (!(mode() & QIODevice::WriteOnly)) {
711  //qWarning() << "You must open the tar file for writing\n";
712  return false;
713  }
714 
715  // In some tar files we can find dir/./file => call cleanPath
716  QString fileName(QDir::cleanPath(name));
717 
718  /*
719  // Create toplevel dirs
720  // Commented out by David since it's not necessary, and if anybody thinks it is,
721  // he needs to implement a findOrCreate equivalent in writeDir.
722  // But as KTar and the "tar" program both handle tar files without
723  // dir entries, there's really no need for that
724  QString tmp ( fileName );
725  int i = tmp.lastIndexOf( '/' );
726  if ( i != -1 )
727  {
728  QString d = tmp.left( i + 1 ); // contains trailing slash
729  if ( !m_dirList.contains( d ) )
730  {
731  tmp = tmp.mid( i + 1 );
732  writeDir( d, user, group ); // WARNING : this one doesn't create its toplevel dirs
733  }
734  }
735  */
736 
737  char buffer[0x201];
738  memset(buffer, 0, 0x200);
739  if ((mode() & QIODevice::ReadWrite) == QIODevice::ReadWrite) {
740  device()->seek(d->tarEnd); // Go to end of archive as might have moved with a read
741  }
742 
743  // provide converted stuff we need later on
744  const QByteArray encodedFileName = QFile::encodeName(fileName);
745  const QByteArray uname = user.toLocal8Bit();
746  const QByteArray gname = group.toLocal8Bit();
747 
748  // If more than 100 chars, we need to use the LongLink trick
749  if (fileName.length() > 99) {
750  d->writeLonglink(buffer, encodedFileName, 'L', uname.constData(), gname.constData());
751  }
752 
753  // Write (potentially truncated) name
754  strncpy(buffer, encodedFileName.constData(), 99);
755  buffer[99] = 0;
756  // zero out the rest (except for what gets filled anyways)
757  memset(buffer + 0x9d, 0, 0x200 - 0x9d);
758 
759  QByteArray permstr = QByteArray::number(static_cast<unsigned int>(perm), 8);
760  permstr = permstr.rightJustified(6, '0');
761  d->fillBuffer(buffer, permstr.constData(), size, mtime, 0x30, uname.constData(), gname.constData());
762 
763  // Write header
764  return device()->write(buffer, 0x200) == 0x200;
765 }
