113 bool KArchive::createDevice(QIODevice::OpenMode mode)
114 {
115  switch (mode) {
116  case QIODevice::WriteOnly:
117  if (!d->fileName.isEmpty()) {
118  // The use of QSaveFile can't be done in the ctor (no mode known yet)
119  //qDebug() << "Writing to a file using QSaveFile";
120  d->saveFile = new QSaveFile(d->fileName);
121  if (!d->saveFile->open(QIODevice::WriteOnly)) {
122  //qWarning() << "QSaveFile creation for " << d->fileName << " failed, " << d->saveFile->errorString();
123  delete d->saveFile;
124  d->saveFile = 0;
125  return false;
126  }
127  d->dev = d->saveFile;
128  Q_ASSERT(d->dev);
129  }
130  break;
131  case QIODevice::ReadOnly:
132  case QIODevice::ReadWrite:
133  // ReadWrite mode still uses QFile for now; we'd need to copy to the tempfile, in fact.
134  if (!d->fileName.isEmpty()) {
135  d->dev = new QFile(d->fileName);
136  d->deviceOwned = true;
137  }
138  break; // continued below
139  default:
140  //qWarning() << "Unsupported mode " << d->mode;
141  return false;
142  }
143  return true;
144 }

146 bool KArchive::close()
147 {
148  if (!isOpen()) {
149  return false; // already closed (return false or true? arguable...)
150  }
151 
152  // moved by holger to allow kzip to write the zip central dir
153  // to the file in closeArchive()
154  // DF: added d->dev so that we skip closeArchive if saving aborted.
155  bool closeSucceeded = true;
156  if (d->dev) {
157  closeSucceeded = closeArchive();
158  if (d->mode == QIODevice::WriteOnly && !closeSucceeded) {
159  d->abortWriting();
160  }
161  }
162 
163  if (d->dev && d->dev != d->saveFile) {
164  d->dev->close();
165  }
166 
167  // if d->saveFile is not null then it is equal to d->dev.
168  if (d->saveFile) {
169  closeSucceeded = d->saveFile->commit();
170  delete d->saveFile;
171  d->saveFile = 0;
172  }
173  if (d->deviceOwned) {
174  delete d->dev; // we created it ourselves in open()
175  }
176 
177  delete d->rootDir;
178  d->rootDir = 0;
179  d->mode = QIODevice::NotOpen;
180  d->dev = 0;
181  return closeSucceeded;
182 }

190 bool KArchive::addLocalFile(const QString &fileName, const QString &destName)
191 {
192  QFileInfo fileInfo(fileName);
193  if (!fileInfo.isFile() && !fileInfo.isSymLink()) {
194  //qWarning() << fileName << "doesn't exist or is not a regular file.";
195  return false;
196  }
197 
198 #if defined(Q_OS_UNIX)
199 #define STAT_METHOD QT_LSTAT
200 #else
201 #define STAT_METHOD QT_STAT
202 #endif
203  QT_STATBUF fi;
204  if (STAT_METHOD(QFile::encodeName(fileName).constData(), &fi) == -1) {
205  /*qWarning() << "stat'ing" << fileName
206  << "failed:" << strerror(errno);*/
207  return false;
208  }
209 
210  if (fileInfo.isSymLink()) {
211  QString symLinkTarget;
212  // Do NOT use fileInfo.readLink() for unix symlinks!
213  // It returns the -full- path to the target, while we want the target string "as is".
214 #if defined(Q_OS_UNIX) && !defined(Q_OS_OS2EMX)
215  const QByteArray encodedFileName = QFile::encodeName(fileName);
216  QByteArray s;
217 #if defined(PATH_MAX)
218  s.resize(PATH_MAX + 1);
219 #else
220  int path_max = pathconf(encodedFileName.data(), _PC_PATH_MAX);
221  if (path_max <= 0) {
222  path_max = 4096;
223  }
224  s.resize(path_max);
225 #endif
226  int len = readlink(encodedFileName.data(), s.data(), s.size() - 1);
227  if (len >= 0) {
228  s[len] = '\0';
229  symLinkTarget = QFile::decodeName(s.constData());
230  }
231 #endif
232  if (symLinkTarget.isEmpty()) { // Mac or Windows
233  symLinkTarget = fileInfo.symLinkTarget();
234  }
235  return writeSymLink(destName, symLinkTarget, fileInfo.owner(),
236  fileInfo.group(), fi.st_mode, fileInfo.lastRead(), fileInfo.lastModified(),
237  fileInfo.created());
238  }/*end if*/
239 
240  qint64 size = fileInfo.size();
241 
242  // the file must be opened before prepareWriting is called, otherwise
243  // if the opening fails, no content will follow the already written
244  // header and the tar file is effectively f*cked up
245  QFile file(fileName);
246  if (!file.open(QIODevice::ReadOnly)) {
247  //qWarning() << "couldn't open file " << fileName;
248  return false;
249  }
250 
251  if (!prepareWriting(destName, fileInfo.owner(), fileInfo.group(), size,
252  fi.st_mode, fileInfo.lastRead(), fileInfo.lastModified(), fileInfo.created())) {
253  //qWarning() << " prepareWriting" << destName << "failed";
254  return false;
255  }
256 
257  // Read and write data in chunks to minimize memory usage
258  QByteArray array;
259  array.resize(int(qMin(qint64(1024 * 1024), size)));
260  qint64 n;
261  qint64 total = 0;
262  while ((n = file.read(array.data(), array.size())) > 0) {
263  if (!writeData(array.data(), n)) {
264  //qWarning() << "writeData failed";
265  return false;
266  }
267  total += n;
268  }
269  Q_ASSERT(total == size);
270 
271  if (!finishWriting(size)) {
272  //qWarning() << "finishWriting failed";
273  return false;
274  }
275  return true;
276 }

278 bool KArchive::addLocalDirectory(const QString &path, const QString &destName)
279 {
280  QDir dir(path);
281  if (!dir.exists()) {
282  return false;
283  }
284  dir.setFilter(dir.filter() | QDir::Hidden);
285  const QStringList files = dir.entryList();
286  for (QStringList::ConstIterator it = files.begin(); it != files.end(); ++it) {
287  if (*it != QLatin1String(".") && *it != QLatin1String("..")) {
288  QString fileName = path + QLatin1Char('/') + *it;
289 // qDebug() << "storing " << fileName;
290  QString dest = destName.isEmpty() ? *it : (destName + QLatin1Char('/') + *it);
291  QFileInfo fileInfo(fileName);
292 
293  if (fileInfo.isFile() || fileInfo.isSymLink()) {
294  addLocalFile(fileName, dest);
295  } else if (fileInfo.isDir()) {
296  addLocalDirectory(fileName, dest);
297  }
298  // We omit sockets
299  }
300  }
301  return true;
302 }

418 KArchiveDirectory *KArchive::findOrCreate(const QString &path)
419 {
420  //qDebug() << path;
421  if (path.isEmpty() || path == QLatin1String("/") || path == QLatin1String(".")) { // root dir => found
422  //qDebug() << "returning rootdir";
423  return rootDir();
424  }
425  // Important note : for tar files containing absolute paths
426  // (i.e. beginning with "/"), this means the leading "/" will
427  // be removed (no KDirectory for it), which is exactly the way
428  // the "tar" program works (though it displays a warning about it)
429  // See also KArchiveDirectory::entry().
430 
431  // Already created ? => found
432  const KArchiveEntry *ent = rootDir()->entry(path);
433  if (ent) {
434  if (ent->isDirectory())
435  //qDebug() << "found it";
436  {
437  const KArchiveDirectory *dir = static_cast<const KArchiveDirectory *>(ent);
438  return const_cast<KArchiveDirectory *>(dir);
439  } else {
440  //qWarning() << "Found" << path << "but it's not a directory";
441  }
442  }
443 
444  // Otherwise go up and try again
445  int pos = path.lastIndexOf(QLatin1Char('/'));
446  KArchiveDirectory *parent;
447  QString dirname;
448  if (pos == -1) { // no more slash => create in root dir
449  parent = rootDir();
450  dirname = path;
451  } else {
452  QString left = path.left(pos);
453  dirname = path.mid(pos + 1);
454  parent = findOrCreate(left); // recursive call... until we find an existing dir.
455  }
456 
457  //qDebug() << "found parent " << parent->name() << " adding " << dirname << " to ensure " << path;
458  // Found -> add the missing piece
459  KArchiveDirectory *e = new KArchiveDirectory(this, dirname, d->rootDir->permissions(),
460  d->rootDir->date(), d->rootDir->user(),
461  d->rootDir->group(), QString());
462  parent->addEntry(e);
463  return e; // now a directory to <path> exists
464 }

692 bool KArchiveFile::copyTo(const QString &dest) const
693 {
694  QFile f(dest + QLatin1Char('/') + name());
695  if (f.open(QIODevice::ReadWrite | QIODevice::Truncate)) {
696  QIODevice *inputDev = createDevice();
697 
698  // Read and write data in chunks to minimize memory usage
699  const qint64 chunkSize = 1024 * 1024;
700  qint64 remainingSize = d->size;
701  QByteArray array;
702  array.resize(int(qMin(chunkSize, remainingSize)));
703 
704  while (remainingSize > 0) {
705  const qint64 currentChunkSize = qMin(chunkSize, remainingSize);
706  const qint64 n = inputDev->read(array.data(), currentChunkSize);
707  Q_UNUSED(n) // except in Q_ASSERT
708  Q_ASSERT(n == currentChunkSize);
709  f.write(array.data(), currentChunkSize);
710  remainingSize -= currentChunkSize;
711  }
712  f.setPermissions(withExecutablePerms(f.permissions(), permissions()));
713  f.close();
714 
715  delete inputDev;
716  return true;
717  }
718  return false;
719 }

754 const KArchiveEntry *KArchiveDirectory::entry(const QString &_name) const
755 {
756  QString name = QDir::cleanPath(_name);
757  int pos = name.indexOf(QLatin1Char('/'));
758  if (pos == 0) { // ouch absolute path (see also KArchive::findOrCreate)
759  if (name.length() > 1) {
760  name = name.mid(1); // remove leading slash
761  pos = name.indexOf(QLatin1Char('/')); // look again
762  } else { // "/"
763  return this;
764  }
765  }
766  // trailing slash ? -> remove
767  if (pos != -1 && pos == name.length() - 1) {
768  name = name.left(pos);
769  pos = name.indexOf(QLatin1Char('/')); // look again
770  }
771  if (pos != -1) {
772  const QString left = name.left(pos);
773  const QString right = name.mid(pos + 1);
774 
775  //qDebug() << "left=" << left << "right=" << right;
776 
777  const KArchiveEntry *e = d->entries.value(left);
778  if (!e || !e->isDirectory()) {
779  return 0;
780  }
781  return static_cast<const KArchiveDirectory *>(e)->entry(right);
782  }
783 
784  return d->entries.value(name);
785 }

841 bool KArchiveDirectory::copyTo(const QString &dest, bool recursiveCopy) const
842 {
843  QDir root;
844  const QString destDir(QDir(dest).absolutePath()); // get directory path without any "." or ".."
845 
846  QList<const KArchiveFile *> fileList;
847  QMap<qint64, QString> fileToDir;
848 
849  // placeholders for iterated items
850  QStack<const KArchiveDirectory *> dirStack;
851  QStack<QString> dirNameStack;
852 
853  dirStack.push(this); // init stack at current directory
854  dirNameStack.push(destDir); // ... with given path
855  do {
856  const KArchiveDirectory *curDir = dirStack.pop();
857 
858  // extract only to specified folder if it is located within archive's extraction folder
859  // otherwise put file under root position in extraction folder
860  QString curDirName = dirNameStack.pop();
861  if (!QDir(curDirName).absolutePath().startsWith(destDir)) {
862  qWarning() << "Attempted export into folder" << curDirName
863  << "which is outside of the extraction root folder" << destDir << "."
864  << "Changing export of contained files to extraction root folder.";
865  curDirName = destDir;
866  }
867 
868  if (!root.mkpath(curDirName)) {
869  return false;
870  }
871 
872  const QStringList dirEntries = curDir->entries();
873  for (QStringList::const_iterator it = dirEntries.begin(); it != dirEntries.end(); ++it) {
874  const KArchiveEntry *curEntry = curDir->entry(*it);
875  if (!curEntry->symLinkTarget().isEmpty()) {
876  QString linkName = curDirName + QLatin1Char('/') + curEntry->name();
877  // To create a valid link on Windows, linkName must have a .lnk file extension.
878 #ifdef Q_OS_WIN
879  if (!linkName.endsWith(QLatin1String(".lnk"))) {
880  linkName += QLatin1String(".lnk");
881  }
882 #endif
883  QFile symLinkTarget(curEntry->symLinkTarget());
884  if (!symLinkTarget.link(linkName)) {
885  //qDebug() << "symlink(" << curEntry->symLinkTarget() << ',' << linkName << ") failed:" << strerror(errno);
886  }
887  } else {
888  if (curEntry->isFile()) {
889  const KArchiveFile *curFile = dynamic_cast<const KArchiveFile *>(curEntry);
890  if (curFile) {
891  fileList.append(curFile);
892  fileToDir.insert(curFile->position(), curDirName);
893  }
894  }
895 
896  if (curEntry->isDirectory() && recursiveCopy) {
897  const KArchiveDirectory *ad = dynamic_cast<const KArchiveDirectory *>(curEntry);
898  if (ad) {
899  dirStack.push(ad);
900  dirNameStack.push(curDirName + QLatin1Char('/') + curEntry->name());
901  }
902  }
903  }
904  }
905  } while (!dirStack.isEmpty());
906 
907  qSort(fileList.begin(), fileList.end(), sortByPosition); // sort on d->pos, so we have a linear access
908 
909  for (QList<const KArchiveFile *>::const_iterator it = fileList.constBegin(), end = fileList.constEnd();
910  it != end; ++it) {
911  const KArchiveFile *f = *it;
912  qint64 pos = f->position();
913  if (!f->copyTo(fileToDir[pos])) {
914  return false;
915  }
916  }
917  return true;
918 }
