110 bool KRcc::openArchive(QIODevice::OpenMode mode)
111 {
112  // Open archive
113 
114  if (mode == QIODevice::WriteOnly) {
115  return true;
116  }
117  if (mode != QIODevice::ReadOnly && mode != QIODevice::ReadWrite) {
118  //qWarning() << "Unsupported mode " << mode;
119  return false;
120  }
121 
122  QUuid uuid = QUuid::createUuid();
123  d->m_prefix = QLatin1Char('/') + uuid.toString();
124  if (!QResource::registerResource(fileName(), d->m_prefix)) {
125  return false;
126  }
127 
128  QDir dir(QLatin1Char(':') + d->m_prefix);
129  d->createEntries(dir, rootDir(), this);
130  return true;
131 }

133 void KRcc::KRccPrivate::createEntries(const QDir &dir, KArchiveDirectory *parentDir, KRcc *q)
134 {
135  Q_FOREACH (const QString &fileName, dir.entryList()) {
136  const QString entryPath = dir.path() + QLatin1Char('/') + fileName;
137  const QFileInfo info(entryPath);
138  if (info.isFile()) {
139  KArchiveEntry *entry = new KRccFileEntry(q, fileName, 0444, info.lastModified(),
140  parentDir->user(), parentDir->group(), info.size(), entryPath);
141  parentDir->addEntry(entry);
142  } else {
143  KArchiveDirectory *entry = new KArchiveDirectory(q, fileName, 0555, info.lastModified(),
144  parentDir->user(), parentDir->group(), /*symlink*/ QString());
145  parentDir->addEntry(entry);
146  createEntries(QDir(entryPath), entry, q);
147  }
148  }
149 }
