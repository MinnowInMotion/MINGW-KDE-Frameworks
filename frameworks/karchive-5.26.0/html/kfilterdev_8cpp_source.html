25 static KCompressionDevice::CompressionType findCompressionByFileName(const QString &fileName)
26 {
27  if (fileName.endsWith(QLatin1String(".gz"), Qt::CaseInsensitive)) {
28  return KCompressionDevice::GZip;
29  }
30 #if HAVE_BZIP2_SUPPORT
31  if (fileName.endsWith(QLatin1String(".bz2"), Qt::CaseInsensitive)) {
32  return KCompressionDevice::BZip2;
33  }
34 #endif
35 #if HAVE_XZ_SUPPORT
36  if (fileName.endsWith(QLatin1String(".lzma"), Qt::CaseInsensitive) || fileName.endsWith(QLatin1String(".xz"), Qt::CaseInsensitive)) {
37  return KCompressionDevice::Xz;
38  }
39 #endif
40  else {
41  // not a warning, since this is called often with other mimetypes (see #88574)...
42  // maybe we can avoid that though?
43  //qDebug() << "findCompressionByFileName : no compression found for " << fileName;
44  }
45 
46  return KCompressionDevice::None;
47 }

54 KCompressionDevice::CompressionType KFilterDev::compressionTypeForMimeType(const QString &mimeType)
55 {
56  if (mimeType == QLatin1String("application/x-gzip")) {
57  return KCompressionDevice::GZip;
58  }
59 #if HAVE_BZIP2_SUPPORT
60  if (mimeType == QLatin1String("application/x-bzip")
61  || mimeType == QLatin1String("application/x-bzip2") // old name, kept for compatibility
62  ) {
63  return KCompressionDevice::BZip2;
64  }
65 #endif
66 #if HAVE_XZ_SUPPORT
67  if (mimeType == QLatin1String("application/x-lzma") // legacy name, still used
68  || mimeType == QLatin1String("application/x-xz") // current naming
69  ) {
70  return KCompressionDevice::Xz;
71  }
72 #endif
73  QMimeDatabase db;
74  const QMimeType mime = db.mimeTypeForName(mimeType);
75  if (mime.isValid()) {
76  if (mime.inherits(QStringLiteral("application/x-gzip"))) {
77  return KCompressionDevice::GZip;
78  }
79 #if HAVE_BZIP2_SUPPORT
80  if (mime.inherits(QStringLiteral("application/x-bzip"))) {
81  return KCompressionDevice::BZip2;
82  }
83 #endif
84 #if HAVE_XZ_SUPPORT
85  if (mime.inherits(QStringLiteral("application/x-lzma"))) {
86  return KCompressionDevice::Xz;
87  }
88 
89  if (mime.inherits(QStringLiteral("application/x-xz"))) {
90  return KCompressionDevice::Xz;
91  }
92 #endif
93  }
94 
95  // not a warning, since this is called often with other mimetypes (see #88574)...
96  // maybe we can avoid that though?

98  return KCompressionDevice::None;
99 }
