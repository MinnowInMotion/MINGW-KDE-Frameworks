125 bool KCompressionDevice::open(QIODevice::OpenMode mode)
126 {
127  if (isOpen()) {
128  //qWarning() << "KCompressionDevice::open: device is already open";
129  return true; // QFile returns false, but well, the device -is- open...
130  }
131  //qDebug() << mode;
132  if (mode == QIODevice::ReadOnly) {
133  d->buffer.resize(0);
134  } else {
135  d->buffer.resize(BUFFER_SIZE);
136  d->filter->setOutBuffer(d->buffer.data(), d->buffer.size());
137  }
138  d->bNeedHeader = !d->bSkipHeaders;
139  d->filter->setFilterFlags(d->bSkipHeaders ? KFilterBase::NoHeaders : KFilterBase::WithHeaders);
140  if (!d->filter->init(mode)) {
141  return false;
142  }
143  d->bOpenedUnderlyingDevice = !d->filter->device()->isOpen();
144  bool ret = d->bOpenedUnderlyingDevice ? d->filter->device()->open(mode) : true;
145  d->result = KFilterBase::Ok;
146 
147  if (!ret) {
148  //qWarning() << "KCompressionDevice::open: Couldn't open underlying device";
149  } else {
150  setOpenMode(mode);
151  }
152 
153  return ret;
154 }

156 void KCompressionDevice::close()
157 {
158  if (!isOpen()) {
159  return;
160  }
161  if (d->filter->mode() == QIODevice::WriteOnly) {
162  write(0L, 0); // finish writing
163  }
164  //qDebug() << "Calling terminate().";
165 
166  if (!d->filter->terminate()) {
167  //qWarning() << "KCompressionDevice::close: terminate returned an error";
168  }
169  if (d->bOpenedUnderlyingDevice) {
170  d->filter->device()->close();
171  }
172  setOpenMode(QIODevice::NotOpen);
173 }

175 bool KCompressionDevice::seek(qint64 pos)
176 {
177  qint64 ioIndex = this->pos(); // current position
178  if (ioIndex == pos) {
179  return true;
180  }
181 
182  //qDebug() << "seek(" << pos << ") called, current pos=" << ioIndex;
183 
184  Q_ASSERT(d->filter->mode() == QIODevice::ReadOnly);
185 
186  if (pos == 0) {
187  // We can forget about the cached data
188  d->bNeedHeader = !d->bSkipHeaders;
189  d->result = KFilterBase::Ok;
190  d->filter->setInBuffer(0L, 0);
191  d->filter->reset();
192  QIODevice::seek(pos);
193  return d->filter->device()->reset();
194  }
195 
196  qint64 bytesToRead;
197  if (ioIndex < pos) { // we can start from here
198  bytesToRead = pos - ioIndex;
199  } else {
200  // we have to start from 0 ! Ugly and slow, but better than the previous
201  // solution (KTarGz was allocating everything into memory)
202  if (!seek(0)) { // recursive
203  return false;
204  }
205  bytesToRead = pos;
206  }
207 
208  //qDebug() << "reading " << bytesToRead << " dummy bytes";
209  QByteArray dummy(qMin(bytesToRead, qint64(3 * BUFFER_SIZE)), 0);
210  d->bIgnoreData = true;
211  const bool result = (read(dummy.data(), bytesToRead) == bytesToRead);
212  d->bIgnoreData = false;
213  QIODevice::seek(pos);
214  return result;
215 }

224 qint64 KCompressionDevice::readData(char *data, qint64 maxlen)
225 {
226  Q_ASSERT(d->filter->mode() == QIODevice::ReadOnly);
227  //qDebug() << "maxlen=" << maxlen;
228  KFilterBase *filter = d->filter;
229 
230  uint dataReceived = 0;
231 
232  // We came to the end of the stream
233  if (d->result == KFilterBase::End) {
234  return dataReceived;
235  }
236 
237  // If we had an error, return -1.
238  if (d->result != KFilterBase::Ok) {
239  return -1;
240  }
241 
242  qint64 outBufferSize;
243  if (d->bIgnoreData) {
244  outBufferSize = qMin(maxlen, static_cast<qint64>(3 * BUFFER_SIZE));
245  } else {
246  outBufferSize = maxlen;
247  }
248  outBufferSize -= dataReceived;
249  qint64 availOut = outBufferSize;
250  filter->setOutBuffer(data, outBufferSize);
251 
252  while (dataReceived < maxlen) {
253  if (filter->inBufferEmpty()) {
254  // Not sure about the best size to set there.
255  // For sure, it should be bigger than the header size (see comment in readHeader)
256  d->buffer.resize(BUFFER_SIZE);
257  // Request data from underlying device
258  int size = filter->device()->read(d->buffer.data(),
259  d->buffer.size());
260  //qDebug() << "got" << size << "bytes from device";
261  if (size) {
262  filter->setInBuffer(d->buffer.data(), size);
263  } else {
264  // Not enough data available in underlying device for now
265  break;
266  }
267  }
268  if (d->bNeedHeader) {
269  (void) filter->readHeader();
270  d->bNeedHeader = false;
271  }
272 
273  d->result = filter->uncompress();
274 
275  if (d->result == KFilterBase::Error) {
276  //qWarning() << "KCompressionDevice: Error when uncompressing data";
277  break;
278  }
279 
280  // We got that much data since the last time we went here
281  uint outReceived = availOut - filter->outBufferAvailable();
282  //qDebug() << "avail_out = " << filter->outBufferAvailable() << " result=" << d->result << " outReceived=" << outReceived;
283  if (availOut < uint(filter->outBufferAvailable())) {
284  //qWarning() << " last availOut " << availOut << " smaller than new avail_out=" << filter->outBufferAvailable() << " !";
285  }
286 
287  dataReceived += outReceived;
288  if (!d->bIgnoreData) { // Move on in the output buffer
289  data += outReceived;
290  availOut = maxlen - dataReceived;
291  } else if (maxlen - dataReceived < outBufferSize) {
292  availOut = maxlen - dataReceived;
293  }
294  if (d->result == KFilterBase::End) {
295  // We're actually at the end, no more data to check
296  if (filter->device()->atEnd()) {
297  break;
298  }
299 
300  // Still not done, re-init and try again
301  filter->init(filter->mode());
302  }
303  filter->setOutBuffer(data, availOut);
304  }
305 
306  return dataReceived;
307 }

309 qint64 KCompressionDevice::writeData(const char *data /*0 to finish*/, qint64 len)
310 {
311  KFilterBase *filter = d->filter;
312  Q_ASSERT(filter->mode() == QIODevice::WriteOnly);
313  // If we had an error, return 0.
314  if (d->result != KFilterBase::Ok) {
315  return 0;
316  }
317 
318  bool finish = (data == 0L);
319  if (!finish) {
320  filter->setInBuffer(data, len);
321  if (d->bNeedHeader) {
322  (void)filter->writeHeader(d->origFileName);
323  d->bNeedHeader = false;
324  }
325  }
326 
327  uint dataWritten = 0;
328  uint availIn = len;
329  while (dataWritten < len || finish) {
330 
331  d->result = filter->compress(finish);
332 
333  if (d->result == KFilterBase::Error) {
334  //qWarning() << "KCompressionDevice: Error when compressing data";
335  // What to do ?
336  break;
337  }
338 
339  // Wrote everything ?
340  if (filter->inBufferEmpty() || (d->result == KFilterBase::End)) {
341  // We got that much data since the last time we went here
342  uint wrote = availIn - filter->inBufferAvailable();
343 
344  //qDebug() << " Wrote everything for now. avail_in=" << filter->inBufferAvailable() << "result=" << d->result << "wrote=" << wrote;
345 
346  // Move on in the input buffer
347  data += wrote;
348  dataWritten += wrote;
349 
350  availIn = len - dataWritten;
351  //qDebug() << " availIn=" << availIn << "dataWritten=" << dataWritten << "pos=" << pos();
352  if (availIn > 0) {
353  filter->setInBuffer(data, availIn);
354  }
355  }
356 
357  if (filter->outBufferFull() || (d->result == KFilterBase::End) || finish) {
358  //qDebug() << " writing to underlying. avail_out=" << filter->outBufferAvailable();
359  int towrite = d->buffer.size() - filter->outBufferAvailable();
360  if (towrite > 0) {
361  // Write compressed data to underlying device
362  int size = filter->device()->write(d->buffer.data(), towrite);
363  if (size != towrite) {
364  //qWarning() << "KCompressionDevice::write. Could only write " << size << " out of " << towrite << " bytes";
365  return 0; // indicate an error (happens on disk full)
366  }
367  //qDebug() << " wrote " << size << " bytes";
368  }
369  if (d->result == KFilterBase::End) {
370  Q_ASSERT(finish); // hopefully we don't get end before finishing
371  break;
372  }
373  d->buffer.resize(BUFFER_SIZE);
374  filter->setOutBuffer(d->buffer.data(), d->buffer.size());
375  }
376  }
377 
378  return dataWritten;
379 }
