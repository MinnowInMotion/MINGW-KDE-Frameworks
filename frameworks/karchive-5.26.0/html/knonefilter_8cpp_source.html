126 KNoneFilter::Result KNoneFilter::copyData()
127 {
128  Q_ASSERT(d->avail_out > 0);
129  if (d->avail_in > 0) {
130  const int n = qMin(d->avail_in, d->avail_out);
131  memcpy(d->next_out, d->next_in, n);
132  d->avail_out -= n;
133  d->next_in += n;
134  d->next_out += n;
135  d->avail_in -= n;
136  return KFilterBase::Ok;
137  } else {
138  return KFilterBase::End;
139  }
140 }
