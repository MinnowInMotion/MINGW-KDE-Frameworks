72 bool KBzip2Filter::init(int mode)
73 {
74  if (d->isInitialized) {
75  terminate();
76  }
77 
78  d->zStream.next_in = 0;
79  d->zStream.avail_in = 0;
80  if (mode == QIODevice::ReadOnly) {
81  const int result = bzDecompressInit(&d->zStream, 0, 0);
82  if (result != BZ_OK) {
83  //qDebug() << "bzDecompressInit returned " << result;
84  return false;
85  }
86  } else if (mode == QIODevice::WriteOnly) {
87  const int result = bzCompressInit(&d->zStream, 5, 0, 0);
88  if (result != BZ_OK) {
89  //qDebug() << "bzDecompressInit returned " << result;
90  return false;
91  }
92  } else {
93  //qWarning() << "Unsupported mode " << mode << ". Only QIODevice::ReadOnly and QIODevice::WriteOnly supported";
94  return false;
95  }
96  d->mode = mode;
97  d->isInitialized = true;
98  return true;
99 }

106 bool KBzip2Filter::terminate()
107 {
108  if (d->mode == QIODevice::ReadOnly) {
109  const int result = bzDecompressEnd(&d->zStream);
110  if (result != BZ_OK) {
111  //qDebug() << "bzDecompressEnd returned " << result;
112  return false;
113  }
114  } else if (d->mode == QIODevice::WriteOnly) {
115  const int result = bzCompressEnd(&d->zStream);
116  if (result != BZ_OK) {
117  //qDebug() << "bzCompressEnd returned " << result;
118  return false;
119  }
120  } else {
121  //qWarning() << "Unsupported mode " << d->mode << ". Only QIODevice::ReadOnly and QIODevice::WriteOnly supported";
122  return false;
123  }
124  d->isInitialized = false;
125  return true;
126 }

157 KBzip2Filter::Result KBzip2Filter::uncompress()
158 {
159  //qDebug() << "Calling bzDecompress with avail_in=" << inBufferAvailable() << " avail_out=" << outBufferAvailable();
160  int result = bzDecompress(&d->zStream);
161  if (result < BZ_OK) {
162  qWarning() << "bzDecompress returned" << result;
163  }
164 
165  switch (result) {
166  case BZ_OK:
167  return KFilterBase::Ok;
168  case BZ_STREAM_END:
169  return KFilterBase::End;
170  default:
171  return KFilterBase::Error;
172  }
173 }

175 KBzip2Filter::Result KBzip2Filter::compress(bool finish)
176 {
177  //qDebug() << "Calling bzCompress with avail_in=" << inBufferAvailable() << " avail_out=" << outBufferAvailable();
178  int result = bzCompress(&d->zStream, finish ? BZ_FINISH : BZ_RUN);
179 
180  switch (result) {
181  case BZ_OK:
182  case BZ_FLUSH_OK:
183  case BZ_RUN_OK:
184  case BZ_FINISH_OK:
185  return KFilterBase::Ok;
186  break;
187  case BZ_STREAM_END:
188  //qDebug() << " bzCompress returned " << result;
189  return KFilterBase::End;
190  break;
191  default:
192  //qDebug() << " bzCompress returned " << result;
193  return KFilterBase::Error;
194  break;
195  }
