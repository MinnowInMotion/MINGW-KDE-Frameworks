591 {

594  if (!project.name().isEmpty()) {
595  postParameters.insert(QLatin1String("name"), project.name());
596  }
597  if (!project.summary().isEmpty()) {
598  postParameters.insert(QLatin1String("summary"), project.summary());
599  }
600  if (!project.description().isEmpty()) {
601  postParameters.insert(QLatin1String("description"), project.description());
602  }
603  if (!project.url().isEmpty()) {
604  postParameters.insert(QLatin1String("url"), project.url());
605  }
606  if (project.developers().count()) {
607  postParameters.insert(QLatin1String("developers"), project.developers().join(QLatin1String("\n")));
608  }
609  if (!project.version().isEmpty()) {
610  postParameters.insert(QLatin1String("version"), project.version());
611  }
612  if (!project.license().isEmpty()) {
613  postParameters.insert(QLatin1String("license"), project.license());
614  }
615  if (!project.requirements().isEmpty()) {
616  postParameters.insert(QLatin1String("requirements"), project.requirements());
617  }
618  // The specfile generator expects an empty string parameter if it is supposed to regenerate the spec file
619  // So we need to check for nullity here as opposed to an empty string.
620  if (!project.specFile().isNull()) {
621  postParameters.insert(QLatin1String("specfile"), project.specFile());
622  }
623  return postParameters;
624 }
