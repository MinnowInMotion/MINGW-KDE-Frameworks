189 QList<Attica::DownloadDescription> Attica::Content::downloadUrlDescriptions() const
190 {
191  QList<Attica::DownloadDescription> descriptions;
192  QMap<QString, QString>::const_iterator iter = d->m_extendedAttributes.constBegin();
193  while (iter != d->m_extendedAttributes.constEnd()) {
194  QString key = iter.key();
195  if (key.startsWith(QLatin1String("downloadname"))) {
196  bool ok;
197  // remove "downloadlink", get the rest as number
198  int num = key.rightRef(key.size() - 12).toInt(&ok);
199  if (ok) {
200  // check if the download actually has a name
201  if (!iter.value().isEmpty()) {
202  descriptions.append(downloadUrlDescription(num));
203  }
204  }
205  }
206  ++iter;
207  }
208  return descriptions;
209 }

211 Attica::DownloadDescription Attica::Content::downloadUrlDescription(int number) const
212 {
213  QString num(QString::number(number));
214  DownloadDescription desc;
215 
216  Attica::DownloadDescription::Type downloadType = Attica::DownloadDescription::LinkDownload;
217  if (attribute(QLatin1String("downloadway") + num) == QLatin1String("0")) {
218  downloadType = Attica::DownloadDescription::FileDownload;
219  } else if (attribute(QLatin1String("downloadway") + num) == QLatin1String("1")) {
220  downloadType = Attica::DownloadDescription::LinkDownload;
221  } else if (attribute(QLatin1String("downloadway") + num) == QLatin1String("2")) {
222  downloadType = Attica::DownloadDescription::PackageDownload;
223  }
224  desc.setType(downloadType);
225  desc.setId(number);
226  desc.setName(attribute(QLatin1String("downloadname") + num));
227  desc.setDistributionType(attribute(QLatin1String("downloadtype") + num));
228  desc.setHasPrice(attribute(QLatin1String("downloadbuy") + num) == QLatin1String("1"));
229  desc.setLink(attribute(QLatin1String("downloadlink") + num));
230  desc.setPriceReason(attribute(QLatin1String("downloadreason") + num));
231  desc.setPriceAmount(attribute(QLatin1String("downloadprice") + num));
232  desc.setSize(attribute(QLatin1String("downloadsize") + num).toUInt());
233  desc.setGpgFingerprint(attribute(QLatin1String("downloadgpgfingerprint") + num));
234  desc.setGpgSignature(attribute(QLatin1String("downloadgpgsignature") + num));
235  desc.setPackageName(attribute(QLatin1String("downloadpackagename") + num));
236  desc.setRepository(attribute(QLatin1String("downloadrepository") + num));
237  return desc;
238 }

242  QList<Attica::HomePageEntry> homepages;
243  QMap<QString, QString>::const_iterator iter = d->m_extendedAttributes.constBegin();
244  while (iter != d->m_extendedAttributes.constEnd()) {
245  QString key = iter.key();
246  if (key.startsWith(QLatin1String("homepagetype"))) {
247  bool ok;
248  // remove "homepage", get the rest as number
249  int num = key.rightRef(key.size() - 12).toInt(&ok);
250  if (ok) {
251  // check if the homepage actually has a valid type
252  if (!iter.value().isEmpty()) {
253  homepages.append(homePageEntry(num));
254  }
255  }
256  }
257  ++iter;
258  }

263 Attica::HomePageEntry Attica::Content::homePageEntry(int number) const
264 {
265  QString num(QString::number(number));
266  HomePageEntry homepage;
267 
268  if (number == 1 && attribute(QLatin1String("homepage1")).isEmpty()) {
269  num.clear();
270  }
271  homepage.setType(attribute(QLatin1String("homepagetype") + num));
272  homepage.setUrl(QUrl(attribute(QLatin1String("homepage") + num)));
273  return homepage;
