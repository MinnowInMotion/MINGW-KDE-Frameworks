149 template <class T>
150 void Parser<T>::parseMetadataXml(QXmlStreamReader &xml)
151 {
152  while (!xml.atEnd()) {
153  xml.readNext();
154  if (xml.isEndElement() && xml.name() == QLatin1String("meta")) {
155  break;
156  } else if (xml.isStartElement()) {
157  if (xml.name() == QLatin1String("status")) {
158  m_metadata.setStatusString(xml.readElementText());
159  } else if (xml.name() == QLatin1String("statuscode")) {
160  m_metadata.setStatusCode(xml.readElementText().toInt());
161  } else if (xml.name() == QLatin1String("message")) {
162  m_metadata.setMessage(xml.readElementText());
163  } else if (xml.name() == QLatin1String("totalitems")) {
164  m_metadata.setTotalItems(xml.readElementText().toInt());
165  } else if (xml.name() == QLatin1String("itemsperpage")) {
166  m_metadata.setItemsPerPage(xml.readElementText().toInt());
167  }
168  }
169  }
170  if (xml.hasError()) {
171  // TODO: error handling in metadata?
172  qWarning() << "XML Error: " << xml.errorString();
173  }
174 
175 }
