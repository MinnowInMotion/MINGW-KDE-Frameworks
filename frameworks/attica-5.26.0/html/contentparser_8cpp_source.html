31 Content Content::Parser::parseXml(QXmlStreamReader &xml)
32 {
33  Content content;
34 
35  while (!xml.atEnd()) {
36  xml.readNext();
37 
38  if (xml.isStartElement()) {
39  if (xml.name() == QLatin1String("id")) {
40  content.setId(xml.readElementText());
41  } else if (xml.name() == QLatin1String("name")) {
42  content.setName(xml.readElementText());
43  } else if (xml.name() == QLatin1String("score")) {
44  content.setRating(xml.readElementText().toInt());
45  } else if (xml.name() == QLatin1String("downloads")) {
46  content.setDownloads(xml.readElementText().toInt());
47  } else if (xml.name() == QLatin1String("comments")) {
48  content.setNumberOfComments(xml.readElementText().toInt());
49  } else if (xml.name() == QLatin1String("created")) {
50  // Qt doesn't accept +-Timezone modifiers, truncate if the string contains them
51  QString dateString = xml.readElementText().left(19);
52  content.setCreated(QDateTime::fromString(dateString, Qt::ISODate));
53  } else if (xml.name() == QLatin1String("changed")) {
54  // Qt doesn't accept +-Timezone modifiers, truncate if the string contains them
55  QString dateString = xml.readElementText().left(19);
56  content.setUpdated(QDateTime::fromString(dateString, Qt::ISODate));
57  } else if (xml.name() == QLatin1String("icon")) {
58  Icon icon;
59  icon.setUrl(QUrl(xml.readElementText()));
60  QXmlStreamAttributes attributes = xml.attributes();
61  if (attributes.hasAttribute(QLatin1String("width"))) {
62  icon.setWidth(attributes.value(QLatin1String("width")).toString().toInt());
63  }
64  if (attributes.hasAttribute(QLatin1String("height"))) {
65  icon.setHeight(attributes.value(QLatin1String("height")).toString().toInt());
66  }
67  // append the icon to the current list of icons
68  QList<Icon> icons;
69  icons = content.icons();
70  icons.append(icon);
71  content.setIcons(icons);
72  } else if (xml.name() == QLatin1String("video")) {
73  QUrl video(xml.readElementText());
74  // append the video to the current list of videos
75  QList<QUrl> videos;
76  videos = content.videos();
77  videos.append(video);
78  content.setVideos(videos);
79  } else {
80  content.addAttribute(xml.name().toString(), xml.readElementText());
81  }
82  }
83 
84  if (xml.isEndElement() && xml.name() == QLatin1String("content")) {
85  break;
86  }
87  }
88 
89  // in case the server only sets creation date, use that as updated also
90  if (content.updated().isNull()) {
91  content.setUpdated(content.created());
92  }
93 
94  return content;
95 }
