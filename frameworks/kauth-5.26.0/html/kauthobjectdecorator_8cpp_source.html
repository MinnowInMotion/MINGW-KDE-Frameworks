55 void ObjectDecoratorPrivate::connectDecorated()
56 {
57  if (qobject_cast<QAbstractButton *>(decoratedObject)) {
58  q->connect(decoratedObject, SIGNAL(clicked()),
59  q, SLOT(slotActivated()));
60  return;
61  }
62 
63  if (qobject_cast<QAction *>(decoratedObject)) {
64  q->connect(decoratedObject, SIGNAL(triggered(bool)),
65  q, SLOT(slotActivated()));
66  return;
67  }
68 
69  qCWarning(KAUTH) << Q_FUNC_INFO << "We're not decorating an action or a button";
70 }

72 void ObjectDecoratorPrivate::linkActionToWidget()
73 {
74  QWidget *widget = qobject_cast<QWidget *>(decoratedObject);
75  if (widget) {
76  authAction.setParentWidget(widget);
77  return;
78  }
79 
80  QAction *action = qobject_cast<QAction *>(decoratedObject);
81  if (action) {
82  authAction.setParentWidget(action->parentWidget());
83  return;
84  }
85 
86  qCWarning(KAUTH) << Q_FUNC_INFO << "We're not decorating an action or a widget";
87 }

89 void ObjectDecoratorPrivate::slotActivated()
90 {
91  if (authAction.isValid()) {
92  KAuth::ExecuteJob *job = authAction.execute(KAuth::Action::AuthorizeOnlyMode);
93  q->connect(job, SIGNAL(statusChanged(KAuth::Action::AuthStatus)),
94  q, SLOT(authStatusChanged(KAuth::Action::AuthStatus)));
95  if (job->exec()) {
96  emit q->authorized(authAction);
97  } else {
98  decoratedObject->setProperty("enabled", false);
99  }
100  }
101 }

103 void ObjectDecoratorPrivate::authStatusChanged(KAuth::Action::AuthStatus status)
104 {
105  switch (status) {
106  case KAuth::Action::AuthorizedStatus:
107  decoratedObject->setProperty("enabled", true);
108  if (!oldIcon.isNull()) {
109  decoratedObject->setProperty("icon", QVariant::fromValue(oldIcon));
110  oldIcon = QIcon();
111  }
112  break;
113  case KAuth::Action::AuthRequiredStatus:
114  decoratedObject->setProperty("enabled", true);
115  oldIcon = decoratedObject->property("icon").value<QIcon>();
116  decoratedObject->setProperty("icon", QIcon::fromTheme(QLatin1String("dialog-password")));
117  break;
118  default:
119  decoratedObject->setProperty("enabled", false);
120  if (!oldIcon.isNull()) {
121  decoratedObject->setProperty("icon", QVariant::fromValue(oldIcon));
122  oldIcon = QIcon();
123  }
124  }
125 }

152 void ObjectDecorator::setAuthAction(const KAuth::Action &action)
153 {
154  if (d->authAction == action) {
155  return;
156  }
157 
158  if (d->authAction.isValid()) {
159  if (!d->oldIcon.isNull()) {
160  d->decoratedObject->setProperty("icon", QVariant::fromValue(d->oldIcon));
161  d->oldIcon = QIcon();
162  }
163  }
164 
165  if (action.isValid()) {
166  d->authAction = action;
167 
168  // Set the parent widget
169  d->linkActionToWidget();
170 
171  d->authStatusChanged(d->authAction.status());
172  }
173 }
