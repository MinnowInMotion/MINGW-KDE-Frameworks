74 void DBusHelperProxy::executeAction(const QString &action, const QString &helperID, const QVariantMap &arguments)
75 {
76  QByteArray blob;
77  QDataStream stream(&blob, QIODevice::WriteOnly);
78 
79  stream << arguments;
80 
81  m_busConnection.interface()->startService(helperID);
82 
83  if (!m_busConnection.connect(helperID, QLatin1String("/"), QLatin1String("org.kde.kf5auth"), QLatin1String("remoteSignal"), this, SLOT(remoteSignalReceived(int,QString,QByteArray)))) {
84  ActionReply errorReply = ActionReply::DBusErrorReply();
85  errorReply.setErrorDescription(tr("DBus Backend error: connection to helper failed. ")
86  + m_busConnection.lastError().message());
87  emit actionPerformed(action, errorReply);
88  }
89 
90  QDBusMessage message;
91  message = QDBusMessage::createMethodCall(helperID, QLatin1String("/"), QLatin1String("org.kde.kf5auth"), QLatin1String("performAction"));
92 
93  QList<QVariant> args;
94  args << action << BackendsManager::authBackend()->callerID() << blob;
95  message.setArguments(args);
96 
97  m_actionsInProgress.push_back(action);
98 
99  QDBusPendingCall pendingCall = m_busConnection.asyncCall(message);
100 
101  auto watcher = new QDBusPendingCallWatcher(pendingCall, this);
102 
103  connect(watcher, &QDBusPendingCallWatcher::finished, this, [this, action, watcher]() {
104  watcher->deleteLater();
105 
106  const QDBusMessage reply = watcher->reply();
107 
108  if (reply.type() == QDBusMessage::ErrorMessage) {
109  ActionReply r = ActionReply::DBusErrorReply();
110  r.setErrorDescription(tr("DBus Backend error: could not contact the helper. "
111  "Connection error: ") + m_busConnection.lastError().message() + tr(". Message error: ") + reply.errorMessage());
112  qCDebug(KAUTH) << reply.errorMessage();
113 
114  emit actionPerformed(action, r);
115  }
116  });
117 }

119 Action::AuthStatus DBusHelperProxy::authorizeAction(const QString &action, const QString &helperID)
120 {
121  if (!m_actionsInProgress.isEmpty()) {
122  return Action::ErrorStatus;
123  }
124 
125  m_busConnection.interface()->startService(helperID);
126 
127  QDBusMessage message;
128  message = QDBusMessage::createMethodCall(helperID, QLatin1String("/"), QLatin1String("org.kde.kf5auth"), QLatin1String("authorizeAction"));
129 
130  QList<QVariant> args;
131  args << action << BackendsManager::authBackend()->callerID();
132  message.setArguments(args);
133 
134  m_actionsInProgress.push_back(action);
135 
136  QEventLoop e;
137  QDBusPendingCall pendingCall = m_busConnection.asyncCall(message);
138  QDBusPendingCallWatcher watcher(pendingCall, this);
139  connect(&watcher, SIGNAL(finished(QDBusPendingCallWatcher*)), &e, SLOT(quit()));
140  e.exec();
141 
142  m_actionsInProgress.removeOne(action);
143 
144  QDBusMessage reply = pendingCall.reply();
145 
146  if (reply.type() == QDBusMessage::ErrorMessage || reply.arguments().size() != 1) {
147  return Action::ErrorStatus;
148  }
149 
150  return static_cast<Action::AuthStatus>(reply.arguments().first().toUInt());
151 }

175 void DBusHelperProxy::remoteSignalReceived(int t, const QString &action, QByteArray blob)
176 {
177  SignalType type = (SignalType)t;
178  QDataStream stream(&blob, QIODevice::ReadOnly);
179 
180  if (type == ActionStarted) {
181  emit actionStarted(action);
182  } else if (type == ActionPerformed) {
183  ActionReply reply = ActionReply::deserialize(blob);
184 
185  m_actionsInProgress.removeOne(action);
186  emit actionPerformed(action, reply);
187  } else if (type == DebugMessage) {
188  int level;
189  QString message;
190 
191  stream >> level >> message;
192 
193  debugMessageReceived(level, message);
194  } else if (type == ProgressStepIndicator) {
195  int step;
196  stream >> step;
197 
198  emit progressStep(action, step);
199  } else if (type == ProgressStepData) {
200  QVariantMap data;
201  stream >> data;
202 
203  emit progressStep(action, data);
204  }
205 }

222 QByteArray DBusHelperProxy::performAction(const QString &action, const QByteArray &callerID, QByteArray arguments)
223 {
224  if (!responder) {
225  return ActionReply::NoResponderReply().serialized();
226  }
227 
228  if (!m_currentAction.isEmpty()) {
229  return ActionReply::HelperBusyReply().serialized();
230  }
231 
232  QVariantMap args;
233  QDataStream s(&arguments, QIODevice::ReadOnly);
234  s >> args;
235 
236  m_currentAction = action;
237  emit remoteSignal(ActionStarted, action, QByteArray());
238  QEventLoop e;
239  e.processEvents(QEventLoop::AllEvents);
240 
241  ActionReply retVal;
242 
243  QTimer *timer = responder->property("__KAuth_Helper_Shutdown_Timer").value<QTimer *>();
244  timer->stop();
245 
246  if (BackendsManager::authBackend()->isCallerAuthorized(action, callerID)) {
247  QString slotname = action;
248  if (slotname.startsWith(m_name + QLatin1Char('.'))) {
249  slotname = slotname.right(slotname.length() - m_name.length() - 1);
250  }
251 
252  slotname.replace(QLatin1Char('.'), QLatin1Char('_'));
253 
254  bool success = QMetaObject::invokeMethod(responder, slotname.toLatin1().data(), Qt::DirectConnection,
255  Q_RETURN_ARG(ActionReply, retVal), Q_ARG(QVariantMap, args));
256 
257  if (!success) {
258  retVal = ActionReply::NoSuchActionReply();
259  }
260 
261  } else {
262  retVal = ActionReply::AuthorizationDeniedReply();
263  }
264 
265  timer->start();
266 
267  emit remoteSignal(ActionPerformed, action, retVal.serialized());
268  e.processEvents(QEventLoop::AllEvents);
269  m_currentAction.clear();
270  m_stopRequest = false;
271 
272  return retVal.serialized();
273 }

275 uint DBusHelperProxy::authorizeAction(const QString &action, const QByteArray &callerID)
276 {
277  if (!m_currentAction.isEmpty()) {
278  return static_cast<uint>(Action::ErrorStatus);
279  }
280 
281  m_currentAction = action;
282 
283  uint retVal;
284 
285  QTimer *timer = responder->property("__KAuth_Helper_Shutdown_Timer").value<QTimer *>();
286  timer->stop();
287 
288  if (BackendsManager::authBackend()->isCallerAuthorized(action, callerID)) {
289  retVal = static_cast<uint>(Action::AuthorizedStatus);
290  } else {
291  retVal = static_cast<uint>(Action::DeniedStatus);
292  }
293 
294  timer->start();
295  m_currentAction.clear();
296 
297  return retVal;
298 }

330 void debugMessageReceived(int t, const QString &message)
331 {
332  QtMsgType type = (QtMsgType)t;
333  switch (type) {
334  case QtDebugMsg:
335  qDebug("Debug message from helper: %s", message.toLatin1().data());
336  break;
337 #if QT_VERSION >= QT_VERSION_CHECK(5, 5, 0)
338  case QtInfoMsg:
339  qInfo("Info message from helper: %s", message.toLatin1().data());
340  break;
341 #endif
342  case QtWarningMsg:
343  qWarning("Warning from helper: %s", message.toLatin1().data());
344  break;
345  case QtCriticalMsg:
346  qCritical("Critical warning from helper: %s", message.toLatin1().data());
347  break;
348  case QtFatalMsg:
349  qFatal("Fatal error from helper: %s", message.toLatin1().data());
350  break;
351  }
352 }
