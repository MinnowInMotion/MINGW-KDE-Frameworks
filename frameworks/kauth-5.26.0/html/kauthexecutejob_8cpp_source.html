83 void ExecuteJob::start()
84 {
85  if (!d->action.isValid()) {
86  qCDebug(KAUTH) << "Tried to start an invalid action";
87  ActionReply reply(ActionReply::InvalidActionError);
88  reply.setErrorDescription(tr("Tried to start an invalid action"));
89  d->actionPerformedSlot(d->action.name(), reply);
90  return;
91  }
92 
93  switch (d->mode) {
94  case Action::ExecuteMode:
95  QTimer::singleShot(0, this, SLOT(doExecuteAction()));
96  break;
97  case Action::AuthorizeOnlyMode:
98  QTimer::singleShot(0, this, SLOT(doAuthorizeAction()));
99  break;
100  default: {
101  ActionReply reply(ActionReply::InvalidActionError);
102  reply.setErrorDescription(tr("Unknown execution mode chosen"));
103  d->actionPerformedSlot(d->action.name(), reply);
104  }
105  break;
106  }
107 }

109 void ExecuteJob::Private::doExecuteAction()
110 {
111  // If this action authorizes from the client, let's do it now
112  if (BackendsManager::authBackend()->capabilities() & KAuth::AuthBackend::AuthorizeFromClientCapability) {
113  if (BackendsManager::authBackend()->capabilities() & KAuth::AuthBackend::PreAuthActionCapability) {
114  BackendsManager::authBackend()->preAuthAction(action.name(), action.parentWidget());
115  }
116 
117  Action::AuthStatus s = BackendsManager::authBackend()->authorizeAction(action.name());
118 
119  if (s == Action::AuthorizedStatus) {
120  if (action.hasHelper()) {
121  BackendsManager::helperProxy()->executeAction(action.name(), action.helperId(), action.arguments());
122  } else {
123  // Done
124  actionPerformedSlot(action.name(), ActionReply::SuccessReply());
125  }
126  } else {
127  // Abort if authorization fails
128  switch (s) {
129  case Action::DeniedStatus:
130  actionPerformedSlot(action.name(), ActionReply::AuthorizationDeniedReply());
131  break;
132  case Action::InvalidStatus:
133  actionPerformedSlot(action.name(), ActionReply::InvalidActionReply());
134  break;
135  case Action::UserCancelledStatus:
136  actionPerformedSlot(action.name(), ActionReply::UserCancelledReply());
137  break;
138  default: {
139  ActionReply r(ActionReply::BackendError);
140  r.setErrorDescription(tr("Unknown status for the authentication procedure"));
141  actionPerformedSlot(action.name(), r);
142  }
143  break;
144  }
145  }
146  } else if (BackendsManager::authBackend()->capabilities() & KAuth::AuthBackend::AuthorizeFromHelperCapability) {
147  if (BackendsManager::authBackend()->capabilities() & KAuth::AuthBackend::PreAuthActionCapability) {
148  BackendsManager::authBackend()->preAuthAction(action.name(), action.parentWidget());
149  }
150  if (!action.hasHelper()) {
151  ActionReply r(ActionReply::InvalidActionReply());
152  r.setErrorDescription(tr("The current backend only allows helper authorization, but this action does not have a helper."));
153  actionPerformedSlot(action.name(), r);
154  return;
155  }
156  BackendsManager::helperProxy()->executeAction(action.name(), action.helperId(), action.arguments());
157  } else {
158  // There's something totally wrong here
159  ActionReply r(ActionReply::BackendError);
160  r.setErrorDescription(tr("The backend does not specify how to authorize"));
161  actionPerformedSlot(action.name(), r);
162  }
163 }

165 void ExecuteJob::Private::doAuthorizeAction()
166 {
167  // Check the status first
168  Action::AuthStatus s = action.status();
169  if (s == Action::AuthRequiredStatus) {
170  // Let's check what to do
171  if (BackendsManager::authBackend()->capabilities() & KAuth::AuthBackend::AuthorizeFromClientCapability) {
172  // In this case we can actually try an authorization
173  if (BackendsManager::authBackend()->capabilities() & KAuth::AuthBackend::PreAuthActionCapability) {
174  BackendsManager::authBackend()->preAuthAction(action.name(), action.parentWidget());
175  }
176 
177  s = BackendsManager::authBackend()->authorizeAction(action.name());
178  } else if (BackendsManager::authBackend()->capabilities() & KAuth::AuthBackend::AuthorizeFromHelperCapability) {
179  // In this case, just throw out success, as the auth will take place later
180  s = Action::AuthorizedStatus;
181  } else {
182  // This should never, never happen
183  ActionReply r(ActionReply::BackendError);
184  r.setErrorDescription(tr("The backend does not specify how to authorize"));
185  actionPerformedSlot(action.name(), r);
186  }
187  }
188 
189  // Return based on the current status
190  if (s == Action::AuthorizedStatus) {
191  actionPerformedSlot(action.name(), ActionReply::SuccessReply());
192  } else {
193  actionPerformedSlot(action.name(), ActionReply::AuthorizationDeniedReply());
194  }
195 }

197 void ExecuteJob::Private::actionPerformedSlot(const QString &taction, const ActionReply &reply)
198 {
199  if (taction == action.name()) {
200  if (reply.failed()) {
201  q->setError(reply.errorCode());
202  q->setErrorText(reply.errorDescription());
203  } else {

205  }

207  q->emitResult();
208  }
