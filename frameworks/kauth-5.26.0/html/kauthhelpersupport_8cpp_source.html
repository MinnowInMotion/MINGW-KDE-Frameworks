75 int HelperSupport::helperMain(int argc, char **argv, const char *id, QObject *responder)
76 {
77 #ifdef Q_OS_UNIX
78  fixEnvironment();
79 #endif
80 
81  openlog(id, 0, LOG_USER);
82  qInstallMessageHandler(&HelperSupport::helperDebugHandler);
83 
84  // NOTE: The helper proxy might use dbus, and we should have the qapp
85  // before using dbus.
86  QCoreApplication app(argc, argv);
87 
88  if (!BackendsManager::helperProxy()->initHelper(QString::fromLatin1(id))) {
89  syslog(LOG_DEBUG, "Helper initialization failed");
90  return -1;
91  }
92 
93  //closelog();
94  remote_dbg = true;
95 
96  BackendsManager::helperProxy()->setHelperResponder(responder);
97 
98  // Attach the timer
99  QTimer *timer = new QTimer(0);
100  responder->setProperty("__KAuth_Helper_Shutdown_Timer", QVariant::fromValue(timer));
101  timer->setInterval(10000);
102  timer->start();
103  QObject::connect(timer, SIGNAL(timeout()), &app, SLOT(quit()));
104  app.exec(); //krazy:exclude=crashy
105 
106  return 0;
107 }

109 void HelperSupport::helperDebugHandler(QtMsgType type, const QMessageLogContext &context, const QString &msgStr)
110 {
111  Q_UNUSED(context); // can be used to find out about file, line, function name
112  QByteArray msg = msgStr.toLocal8Bit();
113  if (!remote_dbg) {
114  int level = LOG_DEBUG;
115  switch (type) {
116  case QtDebugMsg:
117  level = LOG_DEBUG;
118  break;
119  case QtWarningMsg:
120  level = LOG_WARNING;
121  break;
122  case QtCriticalMsg:
123  case QtFatalMsg:
124  level = LOG_ERR;
125  break;
126 #if QT_VERSION >= QT_VERSION_CHECK(5, 5, 0)
127  case QtInfoMsg:
128  level = LOG_INFO;
129  break;
130 #endif
131  }
132  syslog(level, "%s", msg.constData());
133  } else {
134  BackendsManager::helperProxy()->sendDebugMessage(type, msg.constData());
135  }
136 
137  // Anyway I should follow the rule:
138  if (type == QtFatalMsg) {
139  exit(-1);
140  }
141 }
