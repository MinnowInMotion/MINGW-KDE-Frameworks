151 Package PackageLoader::loadPackage(const QString &packageFormat, const QString &packagePath)
152 {
153  if (!d->isDefaultLoader) {
154  Package p = internalLoadPackage(packageFormat);
155  if (p.hasValidStructure()) {
156  if (!packagePath.isEmpty()) {
157  p.setPath(packagePath);
158  }
159  return p;
160  }
161  }
162 
163  if (packageFormat.isEmpty()) {
164  return Package();
165  }
166 
167  PackageStructure *structure = loadPackageStructure(packageFormat);
168 
169  if (structure) {
170  Package p(structure);
171  if (!packagePath.isEmpty()) {
172  p.setPath(packagePath);
173  }
174  return p;
175  }
176 
177 #ifndef NDEBUG
178  // qDebug() << "Couldn't load Package for" << packageFormat << "! reason given: " << error;
179 #endif
180 
181  return Package();
182 }

184 QList<KPluginMetaData> PackageLoader::listPackages(const QString &packageFormat, const QString &packageRoot)
185 {
186  QList<KPluginMetaData> lst;
187 
188  //has been a root specified?
189  QString actualRoot = packageRoot;
190 
191  //try to take it from the ackage structure
192  if (actualRoot.isEmpty()) {
193  PackageStructure *structure = d->structures.value(packageFormat).data();
194  if (!structure) {
195  if (packageFormat == QStringLiteral("KPackage/Generic")) {
196  structure = new GenericPackage();
197  } else if (packageFormat == QStringLiteral("KPackage/GenericQML")) {
198  structure = new GenericQMLPackage();
199  }
200  }
201 
202  if (!structure) {
203  structure = loadPackageStructure(packageFormat);
204  }
205 
206  if (structure) {
207  d->structures.insert(packageFormat, structure);
208  Package p(structure);
209  actualRoot = p.defaultPackageRoot();
210 
211  }
212  }
213 
214  if (actualRoot.isEmpty()) {
215  actualRoot = packageFormat;
216  }
217 
218  QSet<QString> uniqueIds;
219 
220  //TODO: case in which defaultpackageroot is absolute
221  Q_FOREACH(auto const &datadir, QStandardPaths::standardLocations(QStandardPaths::GenericDataLocation)) {
222  const QString plugindir = datadir + '/' + actualRoot;
223  const QString &_ixfile = plugindir + QStringLiteral("kpluginindex.json");
224  QFile indexFile(_ixfile);
225  //qDebug() << "indexfile: " << _ixfile << indexFile.exists();
226  if (indexFile.exists()) {
227  indexFile.open(QIODevice::ReadOnly);
228  QJsonDocument jdoc = QJsonDocument::fromBinaryData(indexFile.readAll());
229  indexFile.close();
230 
231 
232  QJsonArray plugins = jdoc.array();
233 
234  for (QJsonArray::const_iterator iter = plugins.constBegin(); iter != plugins.constEnd(); ++iter) {
235  const QJsonObject &obj = QJsonValue(*iter).toObject();
236  const QString &pluginFileName = obj.value(QStringLiteral("FileName")).toString();
237  const KPluginMetaData m(obj, QString(), pluginFileName);
238  if (m.isValid() && !uniqueIds.contains(m.pluginId())) {
239  uniqueIds << m.pluginId();
240  lst << m;
241  }
242  }
243 
244  } else {
245  //qDebug() << "Not cached";
246  // If there's no cache file, fall back to listing the directory
247  const QDirIterator::IteratorFlags flags = QDirIterator::Subdirectories;
248  const QStringList nameFilters = QStringList(QStringLiteral("metadata.desktop")) << QStringLiteral("metadata.json");
249 
250  QDirIterator it(plugindir, nameFilters, QDir::Files, flags);
251  while (it.hasNext()) {
252  it.next();
253  const QString metadataPath = it.fileInfo().absoluteFilePath();
254 
255  const KPluginMetaData info(metadataPath);
256 
257  if (!info.isValid() || uniqueIds.contains(info.pluginId())) {
258  continue;
259  }
260 
261  if (packageFormat.isEmpty() || info.serviceTypes().contains(packageFormat)) {
262  uniqueIds << info.pluginId();
263  lst << info;
264  }
265  }
266  }
267  }
268 
269  return lst;
270 }

285 KPackage::PackageStructure *PackageLoader::loadPackageStructure(const QString &packageFormat)
286 {
287  PackageStructure *structure = d->structures.value(packageFormat).data();
288  if (!structure) {
289  if (packageFormat == QStringLiteral("KPackage/Generic")) {
290  structure = new GenericPackage();
291  d->structures.insert(packageFormat, structure);
292  } else if (packageFormat == QStringLiteral("KPackage/GenericQML")) {
293  structure = new GenericQMLPackage();
294  d->structures.insert(packageFormat, structure);
295  }
296  }
297 
298  if (structure) {
299  return structure;
300  }
301 
302  QStringList libraryPaths;
303 
304  const QString subDirectory = QStringLiteral("kpackage/packagestructure");
305 
306  Q_FOREACH (const QString &dir, QCoreApplication::libraryPaths()) {
307  QString d = dir + QDir::separator() + subDirectory;
308  if (!d.endsWith(QDir::separator())) {
309  d += QDir::separator();
310  }
311  libraryPaths << d;
312  }
313 
314 
315  QString pluginFileName;
316 
317  Q_FOREACH (const QString &plugindir, libraryPaths) {
318  const QString &_ixfile = plugindir + QStringLiteral("kpluginindex.json");
319  QFile indexFile(_ixfile);
320  if (indexFile.exists()) {
321  indexFile.open(QIODevice::ReadOnly);
322  QJsonDocument jdoc = QJsonDocument::fromBinaryData(indexFile.readAll());
323  indexFile.close();
324 
325 
326  QJsonArray plugins = jdoc.array();
327 
328  for (QJsonArray::const_iterator iter = plugins.constBegin(); iter != plugins.constEnd(); ++iter) {
329  const QJsonObject &obj = QJsonValue(*iter).toObject();
330  const QString &candidate = obj.value(QStringLiteral("FileName")).toString();
331  const KPluginMetaData m(obj, candidate);
332  if (m.pluginId() == packageFormat) {
333  pluginFileName = candidate;
334  break;
335  }
336  }
337  } else {
338  QVector<KPluginMetaData> plugins = KPluginLoader::findPlugins(plugindir);
339  QVectorIterator<KPluginMetaData> iter(plugins);
340  while (iter.hasNext()) {
341  auto md = iter.next();
342  if (md.pluginId() == packageFormat) {
343  pluginFileName = md.fileName();
344  break;
345  }
346  }
347  }
348  }
349 
350  QString error;
351  if (!pluginFileName.isEmpty()) {
352  KPluginLoader loader(pluginFileName);
353  const QVariantList argsWithMetaData = QVariantList() << loader.metaData().toVariantMap();
354  KPluginFactory *factory = loader.factory();
355  if (factory) {
356  structure = factory->create<PackageStructure>(0, argsWithMetaData);
357  if (!structure) {
358  error = QCoreApplication::translate("", "No service matching the requirements was found");
359  }
360  }
361  }
362 
363  if (structure && !error.isEmpty()) {
364  qWarning() << i18n("Could not load installer for package of type %1. Error reported was: %2",
365  packageFormat, error);
366  }
367 
368  if (structure)
369  d->structures.insert(packageFormat, structure);
370 
371  return structure;
372 }
