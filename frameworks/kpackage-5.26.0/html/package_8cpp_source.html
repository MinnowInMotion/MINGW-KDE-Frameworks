82 bool Package::isValid() const
83 {
84  if (!d->structure) {
85  return false;
86  }
87 
88  //Minimal packages with no metadata *are* supposed to be possible
89  //so if !metadata().isValid() go ahead
90  if (metadata().isValid() && metadata().value(QStringLiteral("isHidden"), QStringLiteral("false")) == QLatin1String("true")) {
91  return false;
92  }
93 
94  if (d->checkedValid) {
95  return d->valid;
96  }
97 
98  d->valid = true;
99 
100  //search for the file in all prefixes and in all possible paths for each prefix
101  //even if it's a big nested loop, usually there is one prefix and one location
102  //so shouldn't cause too much disk access
103  QHashIterator<QByteArray, ContentStructure> it(d->contents);
104  const QString rootPath = d->tempRoot.isEmpty() ? d->path : d->tempRoot;
105 
106  while (it.hasNext()) {
107  it.next();
108  if (!it.value().required) {
109  continue;
110  }
111 
112  bool failed = true;
113  foreach (const QString &path, it.value().paths) {
114  foreach (const QString &prefix, d->contentsPrefixPaths) {
115  if (QFile::exists(rootPath + prefix + path)) {
116  failed = false;
117  break;
118  }
119  }
120 
121  if (!failed) {
122  break;
123  }
124  }
125 
126  if (failed) {
127  //qWarning() << "Could not find required" << (it.value().directory ? "directory" : "file") << it.key() << "for package" << path();
128  d->valid = false;
129  break;
130  }
131  }
132 
133  return d->valid;
134 }

248 QString PackagePrivate::unpack(const QString &filePath)
249 {
250  KArchive *archive = 0;
251  QMimeDatabase db;
252  QMimeType mimeType = db.mimeTypeForFile(filePath);
253 
254  if (mimeType.inherits(QStringLiteral("application/zip"))) {
255  archive = new KZip(filePath);
256  } else if (mimeType.inherits(QStringLiteral("application/x-compressed-tar")) ||
257  mimeType.inherits(QStringLiteral("application/x-gzip")) ||
258  mimeType.inherits(QStringLiteral("application/x-tar")) ||
259  mimeType.inherits(QStringLiteral("application/x-bzip-compressed-tar")) ||
260  mimeType.inherits(QStringLiteral("application/x-xz")) ||
261  mimeType.inherits(QStringLiteral("application/x-lzma"))) {
262  archive = new KTar(filePath);
263  } else {
264  //qWarning() << "Could not open package file, unsupported archive format:" << filePath << mimeType.name();
265  }
266  QString tempRoot;
267  if (archive && archive->open(QIODevice::ReadOnly)) {
268  const KArchiveDirectory *source = archive->directory();
269  QTemporaryDir tempdir;
270  tempdir.setAutoRemove(false);
271  tempRoot = tempdir.path() + '/';
272  source->copyTo(tempRoot);
273 
274  if (!QFile::exists(tempdir.path() + "/metadata.desktop")) {
275  // search metadata.desktop, the zip file might have the package contents in a subdirectory
276  QDir unpackedPath(tempdir.path());
277  const QStringList &entries = unpackedPath.entryList(QDir::Dirs);
278  foreach (const QString &pack, entries) {
279  if ((pack != QLatin1String(".") && pack != QLatin1String("..")) &&
280  (QFile::exists(unpackedPath.absolutePath() + '/' + pack + "/metadata.desktop"))) {
281  tempRoot = unpackedPath.absolutePath() + '/' + pack + '/';
282  }
283  }
284  }
285 
286  createPackageMetadata(tempRoot);
287  } else {
288  //qWarning() << "Could not open package file:" << path;
289  }
290 
291  delete archive;
292  return tempRoot;
293 }

295 bool PackagePrivate::isInsidePackageDir(const QString &canonicalPath) const
296 {
297  // make sure that the target file is actually inside the package dir to prevent
298  // path traversal using symlinks or "../" path segments
299 
300  // make sure we got passed a valid path
301  Q_ASSERT(QFileInfo(canonicalPath).exists());
302  Q_ASSERT(QFileInfo(canonicalPath).canonicalFilePath() == canonicalPath);
303  // make sure that the base path is also canonical
304  // this was not the case until 5.8, making this check fail e.g. if /home is a symlink
305  // which in turn would make plasmashell not find the .qml files
306  Q_ASSERT(QDir(path).exists());
307  Q_ASSERT(path == QStringLiteral("/") || QDir(path).canonicalPath() + '/' == path);
308  if (canonicalPath.startsWith(path)) {
309  return true;
310  }
311  qWarning() << "Path traversal attempt detected:" << canonicalPath << "is not inside" << path;
312  return false;
313 }

316 QString Package::filePath(const QByteArray &fileType, const QString &filename) const
317 {
318  if (!d->valid) {
319  QString result = d->fallbackFilePath(fileType, filename);
320  if (result.isEmpty()) {
321  // qDebug() << fileType << "file with name" << filename
322  // << "was not found in package with path" << d->path;
323  }
324  return result;
325  }
326 
327  const QString discoveryKey(fileType + filename);
328  if (d->discoveries.contains(discoveryKey)) {
329  //qDebug() << "looking for" << discoveryKey << d->discoveries.value(discoveryKey);
330  return d->discoveries[discoveryKey];
331  }
332 
333  QStringList paths;
334 
335  if (qstrlen(fileType) != 0) {
336  //qDebug()<<d->contents.keys();
337  if (!d->contents.contains(fileType)) {
338  //qDebug() << "package does not contain" << fileType << filename;
339  return d->fallbackFilePath(fileType, filename);
340  }
341 
342  paths = d->contents[fileType].paths;
343 
344  if (paths.isEmpty()) {
345  //qDebug() << "no matching path came of it, while looking for" << fileType << filename;
346  d->discoveries.insert(discoveryKey, QString());
347  return d->fallbackFilePath(fileType, filename);
348  }
349  } else {
350  //when filetype is empty paths is always empty, so try with an empty string
351  paths << QString();
352  }
353 
354  //Nested loop, but in the medium case resolves to just one iteration
355  //qDebug() << "prefixes:" << prefixes.count() << prefixes;
356  foreach (const QString &contentsPrefix, d->contentsPrefixPaths) {
357  const QString prefix = fileType == "metadata" ? d->path : (d->path + contentsPrefix);
358 
359  foreach (const QString &path, paths) {
360  QString file = prefix + path;
361 
362  if (!filename.isEmpty()) {
363  file.append("/").append(filename);
364  }
365 
366  QFileInfo fi(file);
367  if (fi.exists()) {
368  if (d->externalPaths) {
369  //qDebug() << "found" << file;
370  d->discoveries.insert(discoveryKey, file);
371  return file;
372  }
373 
374  // ensure that we don't return files outside of our base path
375  // due to symlink or ../ games
376  if (d->isInsidePackageDir(fi.canonicalFilePath())) {
377  //qDebug() << "found" << file;
378  d->discoveries.insert(discoveryKey, file);
379  return file;
380  }
381  }
382  }
383  }
384 
385  //qDebug() << fileType << filename << "does not exist in" << prefixes << "at root" << d->path;
386  return d->fallbackFilePath(fileType, filename);
387 }

389 QStringList Package::entryList(const QByteArray &key) const
390 {
391  if (!d->valid) {
392  return QStringList();
393  }
394 
395  QHash<QByteArray, ContentStructure>::const_iterator it = d->contents.constFind(key);
396  if (it == d->contents.constEnd()) {
397  //qDebug() << "couldn't find" << key;
398  return QStringList();
399  }
400 
401  //qDebug() << "going to list" << key;
402  QStringList list;
403  foreach (const QString &prefix, d->contentsPrefixPaths) {
404  //qDebug() << " looking in" << prefix;
405  foreach (const QString &path, it.value().paths) {
406  //qDebug() << " looking in" << path;
407  if (it.value().directory) {
408  //qDebug() << "it's a directory, so trying out" << d->path + prefix + path;
409  QDir dir(d->path + prefix + path);
410 
411  if (d->externalPaths) {
412  list += dir.entryList(QDir::Files | QDir::Readable);
413  } else {
414  // ensure that we don't return files outside of our base path
415  // due to symlink or ../ games
416  QString canonicalized = dir.canonicalPath();
417  if (canonicalized.startsWith(d->path)) {
418  list += dir.entryList(QDir::Files | QDir::Readable);
419  }
420  }
421  } else {
422  const QString fullPath = d->path + prefix + path;
423  //qDebug() << "it's a file at" << fullPath << QFile::exists(fullPath);
424  if (!QFile::exists(fullPath)) {
425  continue;
426  }
427 
428  if (d->externalPaths) {
429  list += fullPath;
430  } else {
431  QDir dir(fullPath);
432  QString canonicalized = dir.canonicalPath() + QDir::separator();
433 
434  //qDebug() << "testing that" << canonicalized << "is in" << d->path;
435  if (canonicalized.startsWith(d->path)) {
436  list += fullPath;
437  }
438  }
439  }
440  }
441  }
442 
443  return list;
444 }

446 void Package::setPath(const QString &path)
447 {
448  // if the path is already what we have, don't bother
449  if (path == d->path) {
450  return;
451  }
452 
453  // our dptr is shared, and it is almost certainly going to change.
454  // hold onto the old pointer just in case it does not, however!
455  QExplicitlySharedDataPointer<PackagePrivate> oldD(d);
456  d.detach();
457 
458  // without structure we're doomed
459  if (!d->structure) {
460  d->path.clear();
461  d->discoveries.clear();
462  d->valid = false;
463  d->checkedValid = true;
464  return;
465  }
466 
467  // empty path => nothing to do
468  if (path.isEmpty()) {
469  d->path.clear();
470  d->discoveries.clear();
471  d->valid = false;
472  d->structure.data()->pathChanged(this);
473  return;
474  }
475 
476  // now we look for all possible paths, including resolving
477  // relative paths against the system search paths
478  QStringList paths;
479  if (QDir::isRelativePath(path)) {
480  QString p;
481 
482  if (d->defaultPackageRoot.isEmpty()) {
483  p = path % "/";
484  } else {
485  p = d->defaultPackageRoot % path % "/";
486  }
487 
488  if (QDir::isRelativePath(p)) {
489  paths << QStandardPaths::locateAll(QStandardPaths::GenericDataLocation, p, QStandardPaths::LocateDirectory);
490  } else {
491  const QDir dir(p);
492  if (QFile::exists(dir.canonicalPath())) {
493  paths << p;
494  }
495  }
496 
497  //qDebug() << "paths:" << p << paths << d->defaultPackageRoot;
498  } else {
499  const QDir dir(path);
500  if (QFile::exists(dir.canonicalPath())) {
501  paths << path;
502  }
503  }
504 
505  QFileInfo fileInfo(path);
506  if (fileInfo.isFile() && d->tempRoot.isEmpty()) {
507  d->path = fileInfo.canonicalFilePath();
508  d->tempRoot = d->unpack(path);
509  }
510 
511  // now we search each path found, caching our previous path to know if
512  // anything actually really changed
513  const QString previousPath = d->path;
514  foreach (const QString &p, paths) {
515  d->checkedValid = false;
516  QDir dir(p);
517  Q_ASSERT(QFile::exists(dir.canonicalPath()));
518  d->path = dir.canonicalPath();
519  // canonicalPath() does not include a trailing slash (unless it is the root dir)
520  if (!d->path.endsWith(QLatin1Char('/'))) {
521  d->path.append('/');
522  }
523 
524  // we need to tell the structure we're changing paths ...
525  d->structure.data()->pathChanged(this);
526  // ... and then testing the results for validity
527  if (isValid()) {
528  break;
529  }
530  }
531 
532 
533  // if nothing did change, then we go back to the old dptr
534  if (d->path == previousPath) {
535  d = oldD;
536  return;
537  }
538 
539  // .. but something did change, so we get rid of our discovery cache
540  d->discoveries.clear();
541  delete d->metadata;
542  d->metadata = 0;
543 
544  // uh-oh, but we didn't end up with anything valid, so we sadly reset ourselves
545  // to futility.
546  if (!d->valid) {
547  d->path.clear();
548  d->structure.data()->pathChanged(this);
549  }
550 }

589 QByteArray Package::cryptographicHash(QCryptographicHash::Algorithm algorithm) const
590 {
591  if (!d->valid) {
592  qWarning() << "can not create hash due to Package being invalid";
593  return QByteArray();
594  }
595 
596  QCryptographicHash hash(algorithm);
597  const QString metadataPath = d->path + "metadata.desktop";
598  if (QFile::exists(metadataPath)) {
599  QFile f(metadataPath);
600  if (f.open(QIODevice::ReadOnly)) {
601  while (!f.atEnd()) {
602  hash.addData(f.read(1024));
603  }
604  } else {
605  qWarning() << "could not add" << f.fileName() << "to the hash; file could not be opened for reading.";
606  }
607  } else {
608  qWarning() << "no metadata at" << metadataPath;
609  }
610 
611  foreach (const QString &prefix, d->contentsPrefixPaths) {
612  const QString basePath = d->path + prefix;
613  QDir dir(basePath);
614 
615  if (!dir.exists()) {
616  return QByteArray();
617  }
618 
619  d->updateHash(basePath, QString(), dir, hash);
620  }
621 
622  return hash.result().toHex();
623 }

866 void PackagePrivate::updateHash(const QString &basePath, const QString &subPath, const QDir &dir, QCryptographicHash &hash)
867 {
868  // hash is calculated as a function of:
869  // * files ordered alphabetically by name, with each file's:
870  // * path relative to the content root
871  // * file data
872  // * directories ordered alphabetically by name, with each dir's:
873  // * path relative to the content root
874  // * file listing (recursing)
875  // symlinks (in both the file and dir case) are handled by adding
876  // the name of the symlink itself and the abs path of what it points to
877 
878  const QDir::SortFlags sorting = QDir::Name | QDir::IgnoreCase;
879  const QDir::Filters filters = QDir::Hidden | QDir::System | QDir::NoDotAndDotDot;
880  foreach (const QString &file, dir.entryList(QDir::Files | filters, sorting)) {
881  if (!subPath.isEmpty()) {
882  hash.addData(subPath.toUtf8());
883  }
884 
885  hash.addData(file.toUtf8());
886 
887  QFileInfo info(dir.path() + '/' + file);
888  if (info.isSymLink()) {
889  hash.addData(info.symLinkTarget().toUtf8());
890  } else {
891  QFile f(info.filePath());
892  if (f.open(QIODevice::ReadOnly)) {
893  while (!f.atEnd()) {
894  hash.addData(f.read(1024));
895  }
896  } else {
897  qWarning() << "could not add" << f.fileName() << "to the hash; file could not be opened for reading. "
898  << "permissions fail?" << info.permissions() << info.isFile();
899  }
900  }
901  }
902 
903  foreach (const QString &subDirPath, dir.entryList(QDir::Dirs | filters, sorting)) {
904  const QString relativePath = subPath + subDirPath + '/';
905  hash.addData(relativePath.toUtf8());
906 
907  QDir subDir(dir.path());
908  subDir.cd(subDirPath);
909 
910  if (subDir.path() != subDir.canonicalPath()) {
911  hash.addData(subDir.canonicalPath().toUtf8());
912  } else {
913  updateHash(basePath, relativePath, subDir, hash);
914  }
915  }
916 }

944 bool PackagePrivate::hasCycle(const KPackage::Package &package)
945 {
946  if (!package.d->fallbackPackage) {
947  return false;
948  }
949 
950  //This is the Floyd cycle detection algorithm
951  //http://en.wikipedia.org/wiki/Cycle_detection#Tortoise_and_hare
952  KPackage::Package *slowPackage = const_cast<KPackage::Package *>(&package);
953  KPackage::Package *fastPackage = const_cast<KPackage::Package *>(&package);
954 
955  while (fastPackage && fastPackage->d->fallbackPackage) {
956  //consider two packages the same if they have the same metadata
957  if ((fastPackage->d->fallbackPackage->metadata().isValid() && fastPackage->d->fallbackPackage->metadata() == slowPackage->metadata()) ||
958  (fastPackage->d->fallbackPackage->d->fallbackPackage && fastPackage->d->fallbackPackage->d->fallbackPackage->metadata().isValid() && fastPackage->d->fallbackPackage->d->fallbackPackage->metadata() == slowPackage->metadata())) {
959  qWarning() << "Warning: the fallback chain of " << package.metadata().pluginId() << "contains a cyclical dependency.";
960  return true;
961  }
962  fastPackage = fastPackage->d->fallbackPackage->d->fallbackPackage;
963  slowPackage = slowPackage->d->fallbackPackage;
964  }
965  return false;
966 }
