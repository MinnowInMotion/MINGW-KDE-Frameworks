57 static QVector<KPluginMetaData> listPackageTypes()
58 {
59  QStringList libraryPaths;
60 
61  const QString subDirectory = QStringLiteral("kpackage/packagestructure");
62 
63  Q_FOREACH (const QString &dir, QCoreApplication::libraryPaths()) {
64  QString d = dir + QDir::separator() + subDirectory;
65  if (!d.endsWith(QDir::separator())) {
66  d += QDir::separator();
67  }
68  libraryPaths << d;
69  }
70 
71  QVector<KPluginMetaData> offers;
72  Q_FOREACH (const QString &plugindir, libraryPaths) {
73  const QString &_ixfile = plugindir + QStringLiteral("kpluginindex.json");
74  QFile indexFile(_ixfile);
75  if (indexFile.exists()) {
76  indexFile.open(QIODevice::ReadOnly);
77  QJsonDocument jdoc = QJsonDocument::fromBinaryData(indexFile.readAll());
78  indexFile.close();
79 
80  QJsonArray plugins = jdoc.array();
81  for (QJsonArray::const_iterator iter = plugins.constBegin(); iter != plugins.constEnd(); ++iter) {
82  const QJsonObject obj = iter->toObject();
83  const QString candidate = obj.value(QStringLiteral("FileName")).toString();
84  offers << KPluginMetaData(obj, candidate);
85  }
86  } else {
87  QVector<KPluginMetaData> plugins = KPluginLoader::findPlugins(plugindir);
88  QVectorIterator<KPluginMetaData> iter(plugins);
89  while (iter.hasNext()) {
90  auto md = iter.next();
91  offers << md;
92  }
93  }
94  }
95  return offers;
96 }

131 void PackageTool::runMain()
132 {
133  KPackage::PackageStructure *structure = new KPackage::PackageStructure;
134  if (d->parser->isSet(QStringLiteral("hash"))) {
135  const QString path = d->parser->value(QStringLiteral("hash"));
136  KPackage::Package package(structure);
137  package.setPath(path);
138  const QString hash =
139  QString::fromLocal8Bit(package.cryptographicHash(QCryptographicHash::Sha1));
140  if (hash.isEmpty()) {
141  d->coutput(i18n("Failed to generate a Package hash for %1", path));
142  exit(9);
143  } else {
144  d->coutput(i18n("SHA1 hash for Package at %1: '%2'", package.path(), hash));
145  exit(0);
146  }
147  return;
148  }
149 
150  if (d->parser->isSet(QStringLiteral("list-types"))) {
151  d->listTypes();
152  exit(0);
153  return;
154  }
155 
156  QString type = d->parser->value(QStringLiteral("type"));
157  d->pluginTypes.clear();
158  d->installer = Package();
159 
160  if (d->parser->isSet(QStringLiteral("remove"))) {
161  d->package = d->parser->value(QStringLiteral("remove"));
162  } else if (d->parser->isSet(QStringLiteral("upgrade"))) {
163  d->package = d->parser->value(QStringLiteral("upgrade"));
164  } else if (d->parser->isSet(QStringLiteral("install"))) {
165  d->package = d->parser->value(QStringLiteral("install"));
166  } else if (d->parser->isSet(QStringLiteral("show"))) {
167  d->package = d->parser->value(QStringLiteral("show"));
168  } else if (d->parser->isSet(QStringLiteral("appstream-metainfo"))) {
169  d->package = d->parser->value(QStringLiteral("appstream-metainfo"));
170  }
171 
172  if (!QDir::isAbsolutePath(d->package)) {
173  d->packageFile = QDir(QDir::currentPath() + '/' + d->package).absolutePath();
174  d->packageFile = QFileInfo(d->packageFile).canonicalFilePath();
175  if (d->parser->isSet(QStringLiteral("upgrade"))) {
176  d->package = d->packageFile;
177  }
178  } else {
179  d->packageFile = d->package;
180  }
181 
182  if (!d->packageFile.isEmpty() && (!d->parser->isSet(QStringLiteral("type")) ||
183  type.compare(i18nc("package type", "wallpaper"), Qt::CaseInsensitive) == 0 ||
184  type.compare(QLatin1String("wallpaper"), Qt::CaseInsensitive) == 0)) {
185  // Check type for common plasma packages
186  KPackage::Package package(structure);
187  QString serviceType;
188  package.setPath(d->packageFile);
189 
190  if (package.isValid() && package.metadata().isValid()) {
191  serviceType = package.metadata().value(QStringLiteral("X-Plasma-ServiceType"));
192  if (serviceType.isEmpty()) {
193  serviceType = package.metadata().serviceTypes().first();
194  }
195  }
196 
197  if (!serviceType.isEmpty()) {
198  if (serviceType == QLatin1String("KPackage/Generic")) {
199  type = QStringLiteral("KPackage/Generic");
200  } else {
201  type = serviceType;
202  //qDebug() << "fallthrough type is" << serviceType;
203  }
204  }
205  }
206 
207  {
208  PackageStructure *structure = PackageLoader::self()->loadPackageStructure(type);
209 
210  if (structure) {
211  d->installer = Package(structure);
212  }
213 
214  if (!d->installer.hasValidStructure()) {
215  qWarning() << "Package type" << type << "not found";
216  }
217 
218  d->packageRoot = d->installer.defaultPackageRoot();
219  d->pluginTypes << type;
220  }
221  if (d->parser->isSet(QStringLiteral("show"))) {
222  const QString pluginName = d->package;
223  showPackageInfo(pluginName);
224  return;
225  } else if (d->parser->isSet(QStringLiteral("appstream-metainfo"))) {
226  const QString pluginName = d->package;
227  showAppstreamInfo(pluginName);
228  return;
229  }
230 
231  if (d->parser->isSet(QStringLiteral("list"))) {
232  d->packageRoot = findPackageRoot(d->package, d->packageRoot);
233  d->coutput(i18n("Listing service types: %1 in %2", d->pluginTypes.join(QByteArray(", ")), d->packageRoot));
234  listPackages(d->pluginTypes, d->packageRoot);
235  exit(0);
236 
237  } else if (d->parser->isSet(QStringLiteral("generate-index"))) {
238  recreateIndex();
239  exit(0);
240 
241  } else {
242  // install, remove or upgrade
243  if (!d->installer.isValid()) {
244 
245  d->installer = KPackage::Package(new KPackage::PackageStructure());
246  }
247 
248  d->packageRoot = findPackageRoot(d->package, d->packageRoot);
249 
250  if (d->parser->isSet(QStringLiteral("remove")) || d->parser->isSet(QStringLiteral("upgrade"))) {
251  QString pkgPath;
252  foreach (const QString &t, d->pluginTypes) {
253  KPackage::Package pkg = KPackage::PackageLoader::self()->loadPackage(t);
254  pkg.setPath(d->package);
255  if (pkg.isValid()) {
256  pkgPath = pkg.path();
257  if (pkgPath.isEmpty() && !d->packageFile.isEmpty()) {
258  pkgPath = d->packageFile;
259  }
260  continue;
261  }
262  }
263  if (pkgPath.isEmpty()) {
264  pkgPath = d->package;
265  }
266 
267  if (d->parser->isSet(QStringLiteral("upgrade"))) {
268  d->installer.setPath(d->package);
269  }
270  QString _p = d->packageRoot;
271  if (!_p.endsWith('/')) {
272  _p.append('/');
273  }
274  _p.append(d->package);
275  d->installer.setDefaultPackageRoot(d->packageRoot);
276  d->installer.setPath(pkgPath);
277 
278  if (!d->parser->isSet(QStringLiteral("type"))) {
279  foreach (const QString &st, d->installer.metadata().serviceTypes()) {
280  if (!d->pluginTypes.contains(st)) {
281  d->pluginTypes << st;
282  }
283  }
284  }
285 
286  QString pluginName;
287  if (d->installer.isValid()) {
288  d->metadata = d->installer.metadata();
289  if (!d->metadata.isValid()) {
290  pluginName = d->package;
291  } else if (!d->metadata.isValid() && d->metadata.pluginId().isEmpty()) {
292  // plugin name given in command line
293  pluginName = d->package;
294  } else {
295  // Parameter was a plasma package, get plugin name from the package
296  pluginName = d->metadata.pluginId();
297  }
298  }
299  QStringList installed = d->packages(d->pluginTypes);
300 
301  if (QFile::exists(d->packageFile)) {
302  d->installer.setPath(d->packageFile);
303  if (d->installer.isValid()) {
304  if (d->installer.metadata().isValid()) {
305  pluginName = d->installer.metadata().pluginId();
306  }
307  }
308  }
309  // Uninstalling ...
310  if (installed.contains(pluginName)) { // Assume it's a plugin name
311  d->installer.setPath(pluginName);
312  KJob *uninstallJob = d->installer.uninstall(pluginName, d->packageRoot);
313  connect(uninstallJob, SIGNAL(result(KJob*)), SLOT(packageUninstalled(KJob*)));
314  return;
315  } else {
316  d->coutput(i18n("Error: Plugin %1 is not installed.", pluginName));
317  exit(2);
318  }
319  }
320  if (d->parser->isSet(QStringLiteral("install"))) {
321  KJob *installJob = d->installer.install(d->packageFile, d->packageRoot);
322  connect(installJob, SIGNAL(result(KJob*)), SLOT(packageInstalled(KJob*)));
323  return;
324  }
325  if (d->package.isEmpty()) {
326  qWarning() << i18nc("No option was given, this is the error message telling the user he needs at least one, do not translate install, remove, upgrade nor list", "One of install, remove, upgrade or list is required.");
327  exit(6);
328  }
329  }

387 void translateKPluginToAppstream(const QString &tagName, const QString &configField, const QJsonObject &configObject, QXmlStreamWriter& writer)
388 {
389  const QRegularExpression rx(QStringLiteral("%1\\[(.*)\\]").arg(configField));
390  const QJsonValue native = configObject.value(configField);
391  if (native.isUndefined()) {
392  return;
393  }
394  writer.writeTextElement(tagName, native.toString());
395  for (auto it = configObject.begin(), itEnd = configObject.end(); it != itEnd; ++it) {
396  const auto match = rx.match(it.key());
397  if (match.hasMatch()) {
398  writer.writeStartElement(tagName);
399  writer.writeAttribute("xml:lang", match.captured(1));
400  writer.writeCharacters(it->toString());
401  writer.writeEndElement();
402  }
403  }

485 QString PackageTool::findPackageRoot(const QString &pluginName, const QString &prefix)
