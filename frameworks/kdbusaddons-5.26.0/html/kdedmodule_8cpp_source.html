48 void KDEDModule::setModuleName(const QString &name)
49 {
50  d->moduleName = name;
51  QDBusObjectPath realPath(QLatin1String("/modules/") + d->moduleName);
52 
53  if (realPath.path().isEmpty()) {
54  qWarning() << "The kded module name" << name << "is invalid!";
55  return;
56  }
57 
58  QDBusConnection::RegisterOptions regOptions;
59 
60  if (this->metaObject()->indexOfClassInfo("D-Bus Interface") != -1) {
61  // 1. There are kded modules that don't have a D-Bus interface.
62  // 2. qt 4.4.3 crashes when trying to emit signals on class without
63  // Q_CLASSINFO("D-Bus Interface", "<your interface>") but
64  // ExportSignal set.
65  // We try to solve that for now with just registering Properties and
66  // Adaptors. But we should investigate where the sense is in registering
67  // the module at all. Just for autoload? Is there a better solution?
68  regOptions = QDBusConnection::ExportScriptableContents | QDBusConnection::ExportAdaptors;
69  } else {
70  // Full functional module. Register everything.
71  regOptions = QDBusConnection::ExportScriptableSlots
72  | QDBusConnection::ExportScriptableProperties
73  | QDBusConnection::ExportAdaptors;
74  qDebug() << "Registration of kded module" << d->moduleName << "without D-Bus interface.";
75  }
76 
77  if (!QDBusConnection::sessionBus().registerObject(realPath.path(), this, regOptions)) {
78  // Happens for khotkeys but the module works. Need some time to investigate.
79  qDebug() << "registerObject() returned false for" << d->moduleName;
80  } else {
81  //qDebug() << "registerObject() successful for" << d->moduleName;
82  // Fix deadlock with Qt 5.6: this has to be delayed until the dbus thread is unlocked
83  QMetaObject::invokeMethod(this, "moduleRegistered", Qt::QueuedConnection, Q_ARG(QDBusObjectPath, realPath));
84  }
85 
86 }

95 QString KDEDModule::moduleForMessage(const QDBusMessage &message)
96 {
97  if (message.type() != QDBusMessage::MethodCallMessage) {
98  return QString();
99  }
100 
101  QString obj = message.path();
102  if (!obj.startsWith(QLatin1String(s_modules_path))) {
103  return QString();
104  }
105 
106  // Remove the <s_modules_path> part
107  obj = obj.mid(strlen(s_modules_path));
108 
109  // Remove the part after the modules name
110  const int index = obj.indexOf(QLatin1Char('/'));
111  if (index != -1) {
112  obj = obj.left(index);
113  }
114 
115  return obj;
116 }
