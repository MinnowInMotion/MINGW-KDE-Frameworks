52  QString generateServiceName()
53  {
54  const QCoreApplication *app = QCoreApplication::instance();
55  const QString domain = app->organizationDomain();
56  const QStringList parts = domain.split(QLatin1Char('.'), QString::SkipEmptyParts);
57 
58  QString reversedDomain;
59  if (parts.isEmpty()) {
60  reversedDomain = QStringLiteral("local.");
61  } else {
62  Q_FOREACH (const QString &part, parts) {
63  reversedDomain.prepend(QLatin1Char('.'));
64  reversedDomain.prepend(part);
65  }
66  }
67 
68  return reversedDomain + app->applicationName();
69  }

73  QString errorMessage;

77 KDBusService::KDBusService(StartupOptions options, QObject *parent)
78  : QObject(parent), d(new KDBusServicePrivate)
79 {
80  new KDBusServiceAdaptor(this);
81  new KDBusServiceExtensionsAdaptor(this);
82  QDBusConnectionInterface *bus = 0;
83 
84  if (!QDBusConnection::sessionBus().isConnected() || !(bus = QDBusConnection::sessionBus().interface())) {
85  d->errorMessage = QLatin1String("Session bus not found\n"
86  "To circumvent this problem try the following command (with Linux and bash)\n"
87  "export $(dbus-launch)");
88  }
89 
90  if (bus) {
91  d->serviceName = d->generateServiceName();
92  QString objectPath = QLatin1Char('/') + d->serviceName;
93  objectPath.replace(QLatin1Char('.'), QLatin1Char('/'));
94  objectPath.replace(QLatin1Char('-'), QLatin1Char('_')); // see spec change at https://bugs.freedesktop.org/show_bug.cgi?id=95129
95 
96  if (options & Multiple) {
97  const QString pid = QString::number(QCoreApplication::applicationPid());
98  d->serviceName += QLatin1Char('-') + pid;
99  }
100 
101  QDBusConnection::sessionBus().registerObject(QStringLiteral("/MainApplication"), QCoreApplication::instance(),
102  QDBusConnection::ExportAllSlots |
103  QDBusConnection::ExportScriptableProperties |
104  QDBusConnection::ExportAdaptors);
105  QDBusConnection::sessionBus().registerObject(objectPath, this,
106  QDBusConnection::ExportAdaptors);
107 
108  d->registered = bus->registerService(d->serviceName) == QDBusConnectionInterface::ServiceRegistered;
109 
110  if (!d->registered) {
111  if (options & Unique) {
112  // Already running so it's ok!
113  QVariantMap platform_data;
114  platform_data.insert(QStringLiteral("desktop-startup-id"), QString::fromUtf8(qgetenv("DESKTOP_STARTUP_ID")));
115  if (QCoreApplication::arguments().count() > 1) {
116  OrgKdeKDBusServiceInterface iface(d->serviceName, objectPath, QDBusConnection::sessionBus());
117  iface.setTimeout(5 * 60 * 1000); // Application can take time to answer
118  QDBusReply<int> reply = iface.CommandLine(QCoreApplication::arguments(), QDir::currentPath(), platform_data);
119  if (reply.isValid()) {
120  exit(reply.value());
121  } else {
122  d->errorMessage = reply.error().message();
123  }
124  } else {
125  OrgFreedesktopApplicationInterface iface(d->serviceName, objectPath, QDBusConnection::sessionBus());
126  iface.setTimeout(5 * 60 * 1000); // Application can take time to answer
127  QDBusReply<void> reply = iface.Activate(platform_data);
128  if (reply.isValid()) {
129  exit(0);
130  } else {
131  d->errorMessage = reply.error().message();
132  }
133  }
134  } else {
135  d->errorMessage = QLatin1String("Couldn't register name '")
136  + d->serviceName
137  + QLatin1String("' with DBUS - another process owns it already!");
138  }
139 
140  } else {
141  if (QCoreApplication *app = QCoreApplication::instance()) {
142  connect(app, SIGNAL(aboutToQuit()), this, SLOT(unregister()));
143  }
144  }
145  }
146 
147  if (!d->registered && ((options & NoExitOnFailure) == 0)) {
148  qCritical() << d->errorMessage;
149  exit(1);
150  }
151 }

173 void KDBusService::unregister()
174 {
175  QDBusConnectionInterface *bus = 0;
176  if (!d->registered || !QDBusConnection::sessionBus().isConnected() || !(bus = QDBusConnection::sessionBus().interface())) {
177  return;
178  }
179  bus->unregisterService(d->serviceName);
180 }

207 void KDBusService::ActivateAction(const QString &action_name, const QVariantList &maybeParameter, const QVariantMap &platform_data)
208 {
209  Q_UNUSED(platform_data);
210  // This is a workaround for DBus not supporting null variants.
211  const QVariant param = maybeParameter.count() == 1 ? maybeParameter.first() : QVariant();
212  emit activateActionRequested(action_name, param);
213  // TODO (via hook) KStartupInfo::appStarted(platform_data.value("desktop-startup-id"))
214  // if desktop-startup-id is set, the action is supposed to show a window (since it could be
215  // called when the app is not running)
