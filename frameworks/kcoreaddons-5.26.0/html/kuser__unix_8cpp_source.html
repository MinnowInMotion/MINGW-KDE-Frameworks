65  void fillPasswd(const passwd *p)
66  {
67  if (p) {
68 #ifndef __BIONIC__
69  QString gecos = QString::fromLocal8Bit(p->pw_gecos);
70 #else
71  QString gecos = QString();
72 #endif
73  QStringList gecosList = gecos.split(QLatin1Char(','));
74  // fill up the list, should be at least 4 entries
75  while (gecosList.size() < 4) {
76  gecosList << QString();
77  }
78 
79  uid = p->pw_uid;
80  gid = p->pw_gid;
81  loginName = QString::fromLocal8Bit(p->pw_name);
82  properties[KUser::FullName] = QVariant(gecosList[0]);
83  properties[KUser::RoomNumber] = QVariant(gecosList[1]);
84  properties[KUser::WorkPhone] = QVariant(gecosList[2]);
85  properties[KUser::HomePhone] = QVariant(gecosList[3]);
86  if (uid == ::getuid() && uid == ::geteuid()) {
87  homeDir = QFile::decodeName(qgetenv("HOME"));
88  }
89  if (homeDir.isEmpty()) {
90  homeDir = QFile::decodeName(p->pw_dir);
91  }
92  shell = QString::fromLocal8Bit(p->pw_shell);
93  }
94  }
95 };

200 template<class Func>
201 static void listGroupsForUser(const char *name, gid_t gid, uint maxCount, Func handleNextGroup)
202 {
203  if (Q_UNLIKELY(maxCount == 0)) {
204  return;
205  }
206  uint found = 0;
207 #if HAVE_GETGROUPLIST
208  QVarLengthArray<gid_t, 100> gid_buffer;
209  gid_buffer.resize(100);
210  int numGroups = gid_buffer.size();
211  int result = getgrouplist(name, gid, gid_buffer.data(), &numGroups);
212  if (result < 0 && uint(numGroups) < maxCount) {
213  // getgrouplist returns -1 if the buffer was too small to store all entries, the required size is in numGroups
214  qDebug("Buffer was too small: %d, need %d", gid_buffer.size(), numGroups);
215  gid_buffer.resize(numGroups);
216  numGroups = gid_buffer.size();
217  getgrouplist(name, gid, gid_buffer.data(), &numGroups);
218  }
219  for (int i = 0; i < numGroups && found < maxCount; ++i) {
220  struct group *g = getgrgid(gid_buffer[i]);
221  // should never be null, but better be safe than crash
222  if (g) {
223  found++;
224  handleNextGroup(g);
225  }
226  }
227 #else
228  // fall back to getgrent() and reading gr->gr_mem
229  // This is slower than getgrouplist, but works as well
230  // add the current gid, this is often not part of g->gr_mem (e.g. build.kde.org or my openSuSE 13.1 system)
231  struct group *g = getgrgid(gid);
232  if (g) {
233  handleNextGroup(g);
234  found++;
235  if (found >= maxCount) {
236  return;
237  }
238  }
239 
240  static const auto groupContainsUser = [](struct group * g, const char *name) -> bool {
241  for (char **user = g->gr_mem; *user; user++)
242  {
243  if (strcmp(name, *user) == 0) {
244  return true;
245  }
246  }
247  return false;
248  };
249 
250  setgrent();
251  while ((g = getgrent())) {
252  // don't add the current gid again
253  if (g->gr_gid != gid && groupContainsUser(g, name)) {
254  handleNextGroup(g);
255  found++;
256  if (found >= maxCount) {
257  break;
258  }
259  }
260  }
261  endgrent();
262 #endif
263 }

415 static void listGroupMembers(gid_t gid, uint maxCount, std::function<void(passwd *)> handleNextGroupUser)
416 {
417  if (maxCount == 0) {
418  return;
419  }
420  struct group *g = getgrgid(gid);
421  if (!g) {
422  return;
423  }
424  uint found = 0;
425  QVarLengthArray<uid_t> addedUsers;
426  struct passwd *p = 0;
427  for (char **user = g->gr_mem; *user; user++) {
428  if ((p = getpwnam(*user))) {
429  addedUsers.append(p->pw_uid);
430  handleNextGroupUser(p);
431  found++;
432  if (found >= maxCount) {
433  break;
434  }
435  }
436  }
437 
438  //gr_mem doesn't contain users where the primary group == gid -> we have to iterate over all users
439  setpwent();
440  while ((p = getpwent()) && found < maxCount) {
441  if (p->pw_gid != gid) {
442  continue; // only need primary gid since otherwise gr_mem already contains this user
443  }
444  // make sure we don't list a user twice
445  if (std::find(addedUsers.cbegin(), addedUsers.cend(), p->pw_uid) == addedUsers.cend()) {
446  handleNextGroupUser(p);
447  found++;
448  }
449  }
450  endpwent();
451 }
