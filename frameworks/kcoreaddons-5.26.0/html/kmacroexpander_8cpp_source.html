49 void KMacroExpanderBase::expandMacros(QString &str)
50 {
51  int pos;
52  int len;
53  ushort ec = d->escapechar.unicode();
54  QStringList rst;
55  QString rsts;
56 
57  for (pos = 0; pos < str.length();) {
58  if (ec != 0) {
59  if (str.unicode()[pos].unicode() != ec) {
60  goto nohit;
61  }
62  if (!(len = expandEscapedMacro(str, pos, rst))) {
63  goto nohit;
64  }
65  } else {
66  if (!(len = expandPlainMacro(str, pos, rst))) {
67  goto nohit;
68  }
69  }
70  if (len < 0) {
71  pos -= len;
72  continue;
73  }
74  rsts = rst.join(QLatin1Char(' '));
75  rst.clear();
76  str.replace(pos, len, rsts);
77  pos += rsts.length();
78  continue;
79  nohit:
80  pos++;
81  }
82 }

204 template <typename VT>
205 int
206 KMacroMapExpander<QString, VT>::expandPlainMacro(const QString &str, int pos, QStringList &ret)
207 {
208  if (pos && isIdentifier(str.unicode()[pos - 1].unicode())) {
209  return 0;
210  }
211  int sl;
212  for (sl = 0; isIdentifier(str.unicode()[pos + sl].unicode()); sl++)
213  ;
214  if (!sl) {
215  return 0;
216  }
217  typename QHash<QString, VT>::const_iterator it =
218  macromap.constFind(str.mid(pos, sl));
219  if (it != macromap.constEnd()) {
220  ret += it.value();
221  return sl;
222  }
223  return 0;
224 }

226 template <typename VT>
227 int
228 KMacroMapExpander<QString, VT>::expandEscapedMacro(const QString &str, int pos, QStringList &ret)
229 {
230  if (str.length() <= pos + 1) {
231  return 0;
232  }
233 
234  if (str.unicode()[pos + 1] == escapeChar()) {
235  ret += QString(escapeChar());
236  return 2;
237  }
238  int sl, rsl, rpos;
239  if (str.unicode()[pos + 1].unicode() == '{') {
240  rpos = pos + 2;
241  if ((sl = str.indexOf(QLatin1Char('}'), rpos)) < 0) {
242  return 0;
243  }
244  sl -= rpos;
245  rsl = sl + 3;
246  } else {
247  rpos = pos + 1;
248  for (sl = 0; isIdentifier(str.unicode()[rpos + sl].unicode()); ++sl)
249  ;
250  rsl = sl + 1;
251  }
252  if (!sl) {
253  return 0;
254  }
255  typename QHash<QString, VT>::const_iterator it =
256  macromap.constFind(str.mid(rpos, sl));
257  if (it != macromap.constEnd()) {
258  ret += it.value();
259  return rsl;
260  }
261  return 0;
262 }
