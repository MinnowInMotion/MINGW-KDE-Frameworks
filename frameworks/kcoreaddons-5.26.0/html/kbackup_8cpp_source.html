74 bool rcsBackupFile(const QString &qFilename,
75  const QString &backupDir,
76  const QString &backupMessage)
77 {
78  QFileInfo fileInfo(qFilename);
79 
80  QString qBackupFilename;
81  if (backupDir.isEmpty()) {
82  qBackupFilename = qFilename;
83  } else {
84  qBackupFilename = backupDir + fileInfo.fileName();
85  }
86  qBackupFilename += QString::fromLatin1(",v");
87 
88  // If backupDir is specified, copy qFilename to the
89  // backupDir and perform the commit there, unlinking
90  // backupDir/qFilename when finished.
91  if (!backupDir.isEmpty()) {
92  if (!QFile::copy(qFilename, backupDir + fileInfo.fileName())) {
93  return false;
94  }
95  fileInfo.setFile(backupDir + QLatin1Char('/') + fileInfo.fileName());
96  }
97 
98  const QString cipath = QStandardPaths::findExecutable(QStringLiteral("ci"));
99  const QString copath = QStandardPaths::findExecutable(QStringLiteral("co"));
100  const QString rcspath = QStandardPaths::findExecutable(QStringLiteral("rcs"));
101  if (cipath.isEmpty() || copath.isEmpty() || rcspath.isEmpty()) {
102  return false;
103  }
104 
105  // Check in the file unlocked with 'ci'
106  QProcess ci;
107  if (!backupDir.isEmpty()) {
108  ci.setWorkingDirectory(backupDir);
109  }
110  ci.start(cipath, QStringList() << QStringLiteral("-u") << fileInfo.filePath());
111  if (!ci.waitForStarted()) {
112  return false;
113  }
114  ci.write(backupMessage.toLocal8Bit());
115  ci.write(".");
116  ci.closeWriteChannel();
117  if (!ci.waitForFinished()) {
118  return false;
119  }
120 
121  // Use 'rcs' to unset strict locking
122  QProcess rcs;
123  if (!backupDir.isEmpty()) {
124  rcs.setWorkingDirectory(backupDir);
125  }
126  rcs.start(rcspath, QStringList() << QStringLiteral("-U") << qBackupFilename);
127  if (!rcs.waitForFinished()) {
128  return false;
129  }
130 
131  // Use 'co' to checkout the current revision and restore permissions
132  QProcess co;
133  if (!backupDir.isEmpty()) {
134  co.setWorkingDirectory(backupDir);
135  }
136  co.start(copath, QStringList() << qBackupFilename);
137  if (!co.waitForFinished()) {
138  return false;
139  }
140 
141  if (!backupDir.isEmpty()) {
142  return QFile::remove(fileInfo.filePath());
143  } else {
144  return true;
145  }

148 bool numberedBackupFile(const QString &qFilename,
149  const QString &backupDir,
150  const QString &backupExtension,
151  const uint maxBackups)
152 {
153  QFileInfo fileInfo(qFilename);
154 
155  // The backup file name template.
156  QString sTemplate;
157  if (backupDir.isEmpty()) {
158  sTemplate = qFilename + QLatin1String(".%1") + backupExtension;
159  } else {
160  sTemplate = backupDir + QLatin1Char('/') + fileInfo.fileName() + QLatin1String(".%1") + backupExtension;
161  }
162 
163  // First, search backupDir for numbered backup files to remove.
164  // Remove all with number 'maxBackups' and greater.
165  QDir d = backupDir.isEmpty() ? fileInfo.dir() : backupDir;
166  d.setFilter(QDir::Files | QDir::Hidden | QDir::NoSymLinks);
167  const QStringList nameFilters = QStringList(fileInfo.fileName() + QLatin1String(".*") + backupExtension);
168  d.setNameFilters(nameFilters);
169  d.setSorting(QDir::Name);
170 
171  uint maxBackupFound = 0;
172  Q_FOREACH (const QFileInfo &fi, d.entryInfoList()) {
173  if (fi.fileName().endsWith(backupExtension)) {
174  // sTemp holds the file name, without the ending backupExtension
175  QString sTemp = fi.fileName();
176  sTemp.truncate(fi.fileName().length() - backupExtension.length());
177  // compute the backup number
178  int idex = sTemp.lastIndexOf(QLatin1Char('.'));
179  if (idex > 0) {
180  bool ok;
181  uint num = sTemp.midRef(idex + 1).toUInt(&ok);
182  if (ok) {
183  if (num >= maxBackups) {
184  QFile::remove(fi.filePath());
185  } else {
186  maxBackupFound = qMax(maxBackupFound, num);
187  }
188  }
189  }
190  }
191  }
192 
193  // Next, rename max-1 to max, max-2 to max-1, etc.
194  QString to = sTemplate.arg(maxBackupFound + 1);
195  for (int i = maxBackupFound; i > 0; i--) {
196  QString from = sTemplate.arg(i);
197 // qCDebug(KCOREADDONS_DEBUG) << "KBackup renaming " << from << " to " << to;
198  QFile::rename(from, to);
199  to = from;
200  }
201 
202  // Finally create most recent backup by copying the file to backup number 1.
203 // qCDebug(KCOREADDONS_DEBUG) << "KBackup copying " << qFilename << " to " << sTemplate.arg(1);
204  return QFile::copy(qFilename, sTemplate.arg(1));
205 }
