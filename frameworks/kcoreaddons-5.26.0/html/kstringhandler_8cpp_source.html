33 QString KStringHandler::capwords(const QString &text)
34 {
35  if (text.isEmpty()) {
36  return text;
37  }
38 
39  const QString strippedText = text.trimmed();
40  const QString space = QString(QLatin1Char(' '));
41  const QStringList words = capwords(strippedText.split(space));
42 
43  QString result = text;
44  result.replace(strippedText, words.join(space));
45  return result;
46 }

88 {
89  bool ignoreMax = 0 == max;
90 
91  QStringList l;
92 
93  int searchStart = 0;
94 
95  int tokenStart = s.indexOf(sep, searchStart);
96 
97  while (-1 != tokenStart && (ignoreMax || l.count() < max - 1)) {
98  if (!s.midRef(searchStart, tokenStart - searchStart).isEmpty()) {
99  l << s.mid(searchStart, tokenStart - searchStart);
100  }
101 
102  searchStart = tokenStart + sep.length();
103  tokenStart = s.indexOf(sep, searchStart);
104  }
105 
106  if (!s.midRef(searchStart, s.length() - searchStart).isEmpty()) {
107  l << s.mid(searchStart, s.length() - searchStart);
108  }
109

166 QString KStringHandler::tagUrls(const QString &text)
167 {
168  /*static*/ QRegExp urlEx(QStringLiteral("(www\\.(?!\\.)|(fish|(f|ht)tp(|s))://)[\\d\\w\\./,:_~\\?=&;#@\\-\\+\\%\\$]+[\\d\\w/]"));
169 
170  QString richText(text);
171  int urlPos = 0, urlLen;
172  while ((urlPos = urlEx.indexIn(richText, urlPos)) >= 0) {
173  urlLen = urlEx.matchedLength();
174  QString href = richText.mid(urlPos, urlLen);
175  // Qt doesn't support (?<=pattern) so we do it here
176  if ((urlPos > 0) && richText[urlPos - 1].isLetterOrNumber()) {
177  urlPos++;
178  continue;
179  }
180  // Don't use QString::arg since %01, %20, etc could be in the string
181  QString anchor = QStringLiteral("<a href=\"") + href + QStringLiteral("\">") + href + QStringLiteral("</a>");
182  richText.replace(urlPos, urlLen, anchor);
183 
184  urlPos += anchor.length();
185  }
186  return richText;
187 }

189 QString KStringHandler::obscure(const QString &str)
190 {

192  const QChar *unicode = str.unicode();
193  for (int i = 0; i < str.length(); ++i)
194  // yes, no typo. can't encode ' ' or '!' because
195  // they're the unicode BOM. stupid scrambling. stupid.

202 bool KStringHandler::isUtf8(const char *buf)
203 {
204  int i, n;
205  unsigned char c;
206  bool gotone = false;
207 
208  if (!buf) {
209  return true; // whatever, just don't crash
210  }
211 
212 #define F 0 /* character never appears in text */
213 #define T 1 /* character appears in plain ASCII text */
214 #define I 2 /* character appears in ISO-8859 text */
215 #define X 3 /* character appears in non-ISO extended ASCII (Mac, IBM PC) */
216 
217  static const unsigned char text_chars[256] = {
218  /* BEL BS HT LF FF CR */
219  F, F, F, F, F, F, F, T, T, T, T, F, T, T, F, F, /* 0x0X */
220  /* ESC */
221  F, F, F, F, F, F, F, F, F, F, F, T, F, F, F, F, /* 0x1X */
222  T, T, T, T, T, T, T, T, T, T, T, T, T, T, T, T, /* 0x2X */
223  T, T, T, T, T, T, T, T, T, T, T, T, T, T, T, T, /* 0x3X */
224  T, T, T, T, T, T, T, T, T, T, T, T, T, T, T, T, /* 0x4X */
225  T, T, T, T, T, T, T, T, T, T, T, T, T, T, T, T, /* 0x5X */
226  T, T, T, T, T, T, T, T, T, T, T, T, T, T, T, T, /* 0x6X */
227  T, T, T, T, T, T, T, T, T, T, T, T, T, T, T, F, /* 0x7X */
228  /* NEL */
229  X, X, X, X, X, T, X, X, X, X, X, X, X, X, X, X, /* 0x8X */
230  X, X, X, X, X, X, X, X, X, X, X, X, X, X, X, X, /* 0x9X */
231  I, I, I, I, I, I, I, I, I, I, I, I, I, I, I, I, /* 0xaX */
232  I, I, I, I, I, I, I, I, I, I, I, I, I, I, I, I, /* 0xbX */
233  I, I, I, I, I, I, I, I, I, I, I, I, I, I, I, I, /* 0xcX */
234  I, I, I, I, I, I, I, I, I, I, I, I, I, I, I, I, /* 0xdX */
235  I, I, I, I, I, I, I, I, I, I, I, I, I, I, I, I, /* 0xeX */
236  I, I, I, I, I, I, I, I, I, I, I, I, I, I, I, I /* 0xfX */
237  };
238 
239  /* *ulen = 0; */
240  for (i = 0; (c = buf[i]); ++i) {
241  if ((c & 0x80) == 0) { /* 0xxxxxxx is plain ASCII */
242  /*
243  * Even if the whole file is valid UTF-8 sequences,
244  * still reject it if it uses weird control characters.
245  */
246 
247  if (text_chars[c] != T) {
248  return false;
249  }
250 
251  } else if ((c & 0x40) == 0) { /* 10xxxxxx never 1st byte */
252  return false;
253  } else { /* 11xxxxxx begins UTF-8 */
254  int following;
255 
256  if ((c & 0x20) == 0) { /* 110xxxxx */
257  following = 1;
258  } else if ((c & 0x10) == 0) { /* 1110xxxx */
259  following = 2;
260  } else if ((c & 0x08) == 0) { /* 11110xxx */
261  following = 3;
262  } else if ((c & 0x04) == 0) { /* 111110xx */
263  following = 4;
264  } else if ((c & 0x02) == 0) { /* 1111110x */
265  following = 5;
266  } else {
267  return false;
268  }
269 
270  for (n = 0; n < following; ++n) {
271  i++;
272  if (!(c = buf[i])) {
273  goto done;
274  }
275 
276  if ((c & 0x80) == 0 || (c & 0x40)) {
277  return false;
278  }
279  }
280  gotone = true;
281  }
282  }
283 done:
284  return gotone; /* don't claim it's UTF-8 if it's all 7-bit */
285 }

306 QString KStringHandler::preProcessWrap(const QString &text)
307 {
308  const QChar zwsp(0x200b);
309 
310  QString result;
311  result.reserve(text.length());
312 
313  for (int i = 0; i < text.length(); i++) {
314  const QChar c = text[i];
315  bool openingParens = (c == QLatin1Char('(') || c == QLatin1Char('{') || c == QLatin1Char('['));
316  bool singleQuote = (c == QLatin1Char('\''));
317  bool closingParens = (c == QLatin1Char(')') || c == QLatin1Char('}') || c == QLatin1Char(']'));
318  bool breakAfter = (closingParens || c.isPunct() || c.isSymbol());
319  bool nextIsSpace = (i == (text.length() - 1) || text[i + 1].isSpace());
320  bool prevIsSpace = (i == 0 || text[i - 1].isSpace() || result[result.length() - 1] == zwsp);
321 
322  // Provide a breaking opportunity before opening parenthesis
323  if (openingParens && !prevIsSpace) {
324  result += zwsp;
325  }
326 
327  // Provide a word joiner before the single quote
328  if (singleQuote && !prevIsSpace) {
329  result += QChar(0x2060);
330  }
331 
332  result += c;
333 
334  if (breakAfter && !openingParens && !nextIsSpace && !singleQuote) {
335  result += zwsp;
336  }
337  }
338 
339  return result;
340 }
