30 static int fromHex(QChar cUnicode)
31 {
32  char c = cUnicode.toLatin1();
33 
34  if (c >= '0' && c <= '9') {
35  return c - '0';
36  } else if (c >= 'A' && c <= 'F') {
37  return c - 'A' + 10;
38  } else if (c >= 'a' && c <= 'f') {
39  return c - 'a' + 10;
40  }
41  return -1;
42 }

71 QStringList KShell::splitArgs(const QString &args, Options flags, Errors *err)
72 {
73  QStringList ret;
74  bool firstword = flags & AbortOnMeta;
75 
76  for (int pos = 0;;) {
77  QChar c;
78  do {
79  if (pos >= args.length()) {
80  goto okret;
81  }
82  c = args.unicode()[pos++];
83  } while (c == QLatin1Char(' '));
84  QString cret;
85  if ((flags & TildeExpand) && c == QLatin1Char('~')) {
86  int opos = pos;
87  for (;; pos++) {
88  if (pos >= args.length()) {
89  break;
90  }
91  c = args.unicode()[pos];
92  if (c == QLatin1Char('/') || c == QLatin1Char(' ')) {
93  break;
94  }
95  if (isQuoteMeta(c)) {
96  pos = opos;
97  c = QLatin1Char('~');
98  goto notilde;
99  }
100  if ((flags & AbortOnMeta) && isMeta(c)) {
101  goto metaerr;
102  }
103  }
104  QString ccret = homeDir(args.mid(opos, pos - opos));
105  if (ccret.isEmpty()) {
106  pos = opos;
107  c = QLatin1Char('~');
108  goto notilde;
109  }
110  if (pos >= args.length()) {
111  ret += ccret;
112  goto okret;
113  }
114  pos++;
115  if (c == QLatin1Char(' ')) {
116  ret += ccret;
117  firstword = false;
118  continue;
119  }
120  cret = ccret;
121  }
122  // before the notilde label, as a tilde does not match anyway
123  if (firstword) {
124  if (c == QLatin1Char('_') ||
125  (c >= QLatin1Char('A') && c <= QLatin1Char('Z')) ||
126  (c >= QLatin1Char('a') && c <= QLatin1Char('z'))) {
127  int pos2 = pos;
128  QChar cc;
129  do {
130  if (pos2 >= args.length()) {
131  // Exactly one word
132  ret += args.mid(pos - 1);
133  goto okret;
134  }
135  cc = args.unicode()[pos2++];
136  } while (cc == QLatin1Char('_') ||
137  (cc >= QLatin1Char('A') && cc <= QLatin1Char('Z')) ||
138  (cc >= QLatin1Char('a') && cc <= QLatin1Char('z')) ||
139  (cc >= QLatin1Char('0') && cc <= QLatin1Char('9')));
140  if (cc == QLatin1Char('=')) {
141  goto metaerr;
142  }
143  }
144  }
145  notilde:
146  do {
147  if (c == QLatin1Char('\'')) {
148  int spos = pos;
149  do {
150  if (pos >= args.length()) {
151  goto quoteerr;
152  }
153  c = args.unicode()[pos++];
154  } while (c != QLatin1Char('\''));
155  cret += args.midRef(spos, pos - spos - 1);
156  } else if (c == QLatin1Char('"')) {
157  for (;;) {
158  if (pos >= args.length()) {
159  goto quoteerr;
160  }
161  c = args.unicode()[pos++];
162  if (c == QLatin1Char('"')) {
163  break;
164  }
165  if (c == QLatin1Char('\\')) {
166  if (pos >= args.length()) {
167  goto quoteerr;
168  }
169  c = args.unicode()[pos++];
170  if (c != QLatin1Char('"') &&
171  c != QLatin1Char('\\') &&
172  !((flags & AbortOnMeta) &&
173  (c == QLatin1Char('$') ||
174  c == QLatin1Char('`')))) {
175  cret += QLatin1Char('\\');
176  }
177  } else if ((flags & AbortOnMeta) &&
178  (c == QLatin1Char('$') ||
179  c == QLatin1Char('`'))) {
180  goto metaerr;
181  }
182  cret += c;
183  }
184  } else if (c == QLatin1Char('$') && pos < args.length() &&
185  args.unicode()[pos] == QLatin1Char('\'')) {
186  pos++;
187  for (;;) {
188  if (pos >= args.length()) {
189  goto quoteerr;
190  }
191  c = args.unicode()[pos++];
192  if (c == QLatin1Char('\'')) {
193  break;
194  }
195  if (c == QLatin1Char('\\')) {
196  if (pos >= args.length()) {
197  goto quoteerr;
198  }
199  c = args.unicode()[pos++];
200  switch (c.toLatin1()) {
201  case 'a': cret += QLatin1Char('\a'); break;
202  case 'b': cret += QLatin1Char('\b'); break;
203  case 'e': cret += QLatin1Char('\033'); break;
204  case 'f': cret += QLatin1Char('\f'); break;
205  case 'n': cret += QLatin1Char('\n'); break;
206  case 'r': cret += QLatin1Char('\r'); break;
207  case 't': cret += QLatin1Char('\t'); break;
208  case '\\': cret += QLatin1Char('\\'); break;
209  case '\'': cret += QLatin1Char('\''); break;
210  case 'c':
211  if (pos >= args.length()) {
212  goto quoteerr;
213  }
214  cret += args.unicode()[pos++].toLatin1() & 31;
215  break;
216  case 'x': {
217  if (pos >= args.length()) {
218  goto quoteerr;
219  }
220  int hv = fromHex(args.unicode()[pos++]);
221  if (hv < 0) {
222  goto quoteerr;
223  }
224  if (pos < args.length()) {
225  int hhv = fromHex(args.unicode()[pos]);
226  if (hhv > 0) {
227  hv = hv * 16 + hhv;
228  pos++;
229  }
230  cret += QChar(hv);
231  }
232  break;
233  }
234  default:
235  if (c.toLatin1() >= '0' && c.toLatin1() <= '7') {
236  char cAscii = c.toLatin1();
237  int hv = cAscii - '0';
238  for (int i = 0; i < 2; i++) {
239  if (pos >= args.length()) {
240  break;
241  }
242  c = args.unicode()[pos];
243  if (c.toLatin1() < '0' || c.toLatin1() > '7') {
244  break;
245  }
246  hv = hv * 8 + (c.toLatin1() - '0');
247  pos++;
248  }
249  cret += QChar(hv);
250  } else {
251  cret += QLatin1Char('\\');
252  cret += c;
253  }
254  break;
255  }
256  } else {
257  cret += c;
258  }
259  }
260  } else {
261  if (c == QLatin1Char('\\')) {
262  if (pos >= args.length()) {
263  goto quoteerr;
264  }
265  c = args.unicode()[pos++];
266  } else if ((flags & AbortOnMeta) && isMeta(c)) {
267  goto metaerr;
268  }
269  cret += c;
270  }
271  if (pos >= args.length()) {
272  break;
273  }
274  c = args.unicode()[pos++];
275  } while (c != QLatin1Char(' '));
276  ret += cret;
277  firstword = false;
278  }
279 
280 okret:
281  if (err) {
282  *err = NoError;
283  }
284  return ret;
285 
286 quoteerr:
287  if (err) {
288  *err = BadQuoting;
289  }
290  return QStringList();
291 
292 metaerr:
293  if (err) {
294  *err = FoundMeta;
295  }
296  return QStringList();
297 }

310 QString KShell::quoteArg(const QString &arg)
311 {
312  if (!arg.length()) {
313  return QString::fromLatin1("''");
314  }
315  for (int i = 0; i < arg.length(); i++)
316  if (isSpecial(arg.unicode()[i])) {
317  QChar q(QLatin1Char('\''));
318  return QString(arg).replace(q, QLatin1String("'\\''")).prepend(q).append(q);
319  }
320  return arg;
321 }
