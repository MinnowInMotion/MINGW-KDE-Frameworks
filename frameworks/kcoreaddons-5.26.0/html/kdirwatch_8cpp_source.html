147 /* All entries (files/directories) to be watched in the
148  * application (coming from multiple KDirWatch instances)
149  * are registered in a single KDirWatchPrivate instance.
150  *
151  * At the moment, the following methods for file watching
152  * are supported:
153  * - Polling: All files to be watched are polled regularly
154  * using stat (more precise: QFileInfo.lastModified()).
155  * The polling frequency is determined from global kconfig
156  * settings, defaulting to 500 ms for local directories
157  * and 5000 ms for remote mounts
158  * - FAM (File Alternation Monitor): first used on IRIX, SGI
159  * has ported this method to LINUX. It uses a kernel part
160  * (IMON, sending change events to /dev/imon) and a user
161  * level damon (fam), to which applications connect for
162  * notification of file changes. For NFS, the fam damon
163  * on the NFS server machine is used; if IMON is not built
164  * into the kernel, fam uses polling for local files.
165  * - INOTIFY: In LINUX 2.6.13, inode change notification was
166  * introduced. You're now able to watch arbitrary inode's
167  * for changes, and even get notification when they're
168  * unmounted.
169  */

265 void KDirWatchPrivate::inotifyEventReceived()
266 {
267  //qCDebug(KDIRWATCH);
268 #if HAVE_SYS_INOTIFY_H
269  if (!supports_inotify) {
270  return;
271  }
272 
273  int pending = -1;
274  int offsetStartRead = 0; // where we read into buffer
275  char buf[8192];
276  assert(m_inotify_fd > -1);
277  ioctl(m_inotify_fd, FIONREAD, &pending);
278 
279  while (pending > 0) {
280 
281  const int bytesToRead = qMin<int>(pending, sizeof(buf) - offsetStartRead);
282 
283  int bytesAvailable = read(m_inotify_fd, &buf[offsetStartRead], bytesToRead);
284  pending -= bytesAvailable;
285  bytesAvailable += offsetStartRead;
286  offsetStartRead = 0;
287 
288  int offsetCurrent = 0;
289  while (bytesAvailable >= int(sizeof(struct inotify_event))) {
290  const struct inotify_event *const event = reinterpret_cast<inotify_event *>(&buf[offsetCurrent]);
291  const int eventSize = sizeof(struct inotify_event) + event->len;
292  if (bytesAvailable < eventSize) {
293  break;
294  }
295 
296  bytesAvailable -= eventSize;
297  offsetCurrent += eventSize;
298 
299  QString path;
300  // strip trailing null chars, see inotify_event documentation
301  // these must not end up in the final QString version of path
302  int len = event->len;
303  while (len > 1 && !event->name[len - 1]) {
304  --len;
305  }
306  QByteArray cpath(event->name, len);
307  if (len) {
308  path = QFile::decodeName(cpath);
309  }
310 
311  if (!path.isEmpty() && isNoisyFile(cpath.data())) {
312  continue;
313  }
314 
315  // Is set to true if the new event is a directory, false otherwise. This prevents a stat call in clientsForFileOrDir
316  const bool isDir = (event->mask & (IN_ISDIR));
317 
318  // now we're in deep trouble of finding the
319  // associated entries
320  // for now, we suck and iterate
321  for (EntryMap::Iterator it = m_mapEntries.begin();
322  it != m_mapEntries.end();) {
323 
324  Entry *e = &(*it);
325  ++it;
326  if (e->wd == event->wd) {
327  const bool wasDirty = e->dirty;
328  e->dirty = true;
329 
330  const QString tpath = e->path + QLatin1Char('/') + path;
331 
332  if (s_verboseDebug) {
333  qCDebug(KDIRWATCH).nospace() << "got event 0x" << qPrintable(QString::number(event->mask, 16)) << " for " << e->path;
334  }
335 
336  if (event->mask & IN_DELETE_SELF) {
337  if (s_verboseDebug) {
338  qCDebug(KDIRWATCH) << "-->got deleteself signal for" << e->path;
339  }
340  e->m_status = NonExistent;
341  e->wd = -1;
342  e->m_ctime = invalid_ctime;
343  emitEvent(e, Deleted, e->path);
344  // If the parent dir was already watched, tell it something changed
345  Entry *parentEntry = entry(e->parentDirectory());
346  if (parentEntry) {
347  parentEntry->dirty = true;
348  }
349  // Add entry to parent dir to notice if the entry gets recreated
350  addEntry(0, e->parentDirectory(), e, true /*isDir*/);
351  }
352  if (event->mask & IN_IGNORED) {
353  // Causes bug #207361 with kernels 2.6.31 and 2.6.32!
354  //e->wd = -1;
355  }
356  if (event->mask & (IN_CREATE | IN_MOVED_TO)) {
357  Entry *sub_entry = e->findSubEntry(tpath);
358 
359  if (s_verboseDebug) {
360  qCDebug(KDIRWATCH) << "-->got CREATE signal for" << (tpath) << "sub_entry=" << sub_entry;
361  qCDebug(KDIRWATCH) << *e;
362  }
363 
364  // The code below is very similar to the one in checkFAMEvent...
365  if (sub_entry) {
366  // We were waiting for this new file/dir to be created
367  sub_entry->dirty = true;
368  rescan_timer.start(0); // process this asap, to start watching that dir
369  } else if (e->isDir && !e->m_clients.empty()) {
370  const QList<Client *> clients = e->inotifyClientsForFileOrDir(isDir);
371  Q_FOREACH (Client *client, clients) {
372  // See discussion in addEntry for why we don't addEntry for individual
373  // files in WatchFiles mode with inotify.
374  if (isDir) {
375  addEntry(client->instance, tpath, 0, isDir,
376  isDir ? client->m_watchModes : KDirWatch::WatchDirOnly);
377  }
378  }
379  if (!clients.isEmpty()) {
380  emitEvent(e, Created, tpath);
381  qCDebug(KDIRWATCH).nospace() << clients.count() << " instance(s) monitoring the new "
382  << (isDir ? "dir " : "file ") << tpath;
383  }
384  e->m_pendingFileChanges.append(e->path);
385  if (!rescan_timer.isActive()) {
386  rescan_timer.start(m_PollInterval); // singleshot
387  }
388  }
389  }
390  if (event->mask & (IN_DELETE | IN_MOVED_FROM)) {
391  if (s_verboseDebug) {
392  qCDebug(KDIRWATCH) << "-->got DELETE signal for" << tpath;
393  }
394  if ((e->isDir) && (!e->m_clients.empty())) {
395  Client *client = 0;
396  // A file in this directory has been removed. It wasn't an explicitly
397  // watched file as it would have its own watch descriptor, so
398  // no addEntry/ removeEntry bookkeeping should be required. Emit
399  // the event immediately if any clients are interested.
400  KDirWatch::WatchModes flag = isDir ? KDirWatch::WatchSubDirs : KDirWatch::WatchFiles;
401  int counter = 0;
402  Q_FOREACH (client, e->m_clients) { // krazy:exclude=Q_FOREACH
403  if (client->m_watchModes & flag) {
404  counter++;
405  }
406  }
407  if (counter != 0) {
408  emitEvent(e, Deleted, tpath);
409  }
410  }
411  }
412  if (event->mask & (IN_MODIFY | IN_ATTRIB)) {
413  if ((e->isDir) && (!e->m_clients.empty())) {
414  if (s_verboseDebug) {
415  qCDebug(KDIRWATCH) << "-->got MODIFY signal for" << (tpath);
416  }
417  // A file in this directory has been changed. No
418  // addEntry/ removeEntry bookkeeping should be required.
419  // Add the path to the list of pending file changes if
420  // there are any interested clients.
421  //QT_STATBUF stat_buf;
422  //QByteArray tpath = QFile::encodeName(e->path+'/'+path);
423  //QT_STAT(tpath, &stat_buf);
424  //bool isDir = S_ISDIR(stat_buf.st_mode);
425 
426  // The API doc is somewhat vague as to whether we should emit
427  // dirty() for implicitly watched files when WatchFiles has
428  // not been specified - we'll assume they are always interested,
429  // regardless.
430  // Don't worry about duplicates for the time
431  // being; this is handled in slotRescan.
432  e->m_pendingFileChanges.append(tpath);
433  // Avoid stat'ing the directory if only an entry inside it changed.
434  e->dirty = (wasDirty || (path.isEmpty() && (event->mask & IN_ATTRIB)));
435  }
436  }
437 
438  if (!rescan_timer.isActive()) {
439  rescan_timer.start(m_PollInterval); // singleshot
440  }
441 
442  break;
443  }
444  }
445  }
446  if (bytesAvailable > 0) {
447  // copy partial event to beginning of buffer
448  memmove(buf, &buf[offsetCurrent], bytesAvailable);
449  offsetStartRead = bytesAvailable;
450  }
451  }
452 #endif
453 }

592 KDirWatchPrivate::Entry *KDirWatchPrivate::entry(const QString &_path)
593 {
594 // we only support absolute paths
595  if (_path.isEmpty() || QDir::isRelativePath(_path)) {
596  return 0;
597  }
598 
599  QString path(_path);
600 
601  if (path.length() > 1 && path.endsWith(QLatin1Char('/'))) {
602  path.truncate(path.length() - 1);
603  }
604 
605  EntryMap::Iterator it = m_mapEntries.find(path);
606  if (it == m_mapEntries.end()) {
607  return 0;
608  } else {

805  || (path.startsWith(QLatin1String("/dev/")) && !path.startsWith(QLatin1String("/dev/."))
806  && !path.startsWith(QLatin1String("/dev/shm")))
807 #endif
808  ) {
809  return; // Don't even go there.
810  }
811 
812  if (path.length() > 1 && path.endsWith(QLatin1Char('/'))) {
813  path.truncate(path.length() - 1);
814  }
815 
816  EntryMap::Iterator it = m_mapEntries.find(path);
817  if (it != m_mapEntries.end()) {
818  if (sub_entry) {
819  (*it).m_entries.append(sub_entry);
820  if (s_verboseDebug) {
821  qCDebug(KDIRWATCH) << "Added already watched Entry" << path
822  << "(for" << sub_entry->path << ")";
823  }
824  } else {
825  (*it).addClient(instance, watchModes);
826  if (s_verboseDebug) {
827  qCDebug(KDIRWATCH) << "Added already watched Entry" << path
828  << "(now" << (*it).clientCount() << "clients)"
829  << QStringLiteral("[%1]").arg(instance->objectName());
830  }
831  }
832  return;
833  }
834 
835  // we have a new path to watch
836 
837  QT_STATBUF stat_buf;
838  bool exists = (QT_STAT(QFile::encodeName(path).constData(), &stat_buf) == 0);
839 
840  EntryMap::iterator newIt = m_mapEntries.insert(path, Entry());
841  // the insert does a copy, so we have to use <e> now
842  Entry *e = &(*newIt);
843 
844  if (exists) {
845  e->isDir = (stat_buf.st_mode & QT_STAT_MASK) == QT_STAT_DIR;
846 
847 #ifndef Q_OS_WIN
848  if (e->isDir && !isDir) {
849  if (QT_LSTAT(QFile::encodeName(path).constData(), &stat_buf) == 0) {
850  if ((stat_buf.st_mode & QT_STAT_MASK) == QT_STAT_LNK) {
851  // if it's a symlink, don't follow it
852  e->isDir = false;
853  }
854  }
855  }
856 #endif
857 
858  if (e->isDir && !isDir) {
859  qCWarning(KCOREADDONS_DEBUG) << "KDirWatch:" << path << "is a directory. Use addDir!";
860  } else if (!e->isDir && isDir) {
861  qWarning("KDirWatch: %s is a file. Use addFile!", qPrintable(path));
862  }
863 
864  if (!e->isDir && (watchModes != KDirWatch::WatchDirOnly)) {
865  qCWarning(KCOREADDONS_DEBUG) << "KDirWatch:" << path << "is a file. You can't use recursive or "
866  "watchFiles options";
867  watchModes = KDirWatch::WatchDirOnly;
868  }
869

871  // ctime is the 'creation time' on windows - use mtime instead
872  e->m_ctime = stat_buf.st_mtime;
873 #else
874  e->m_ctime = stat_buf.st_ctime;
875 #endif
876  e->m_status = Normal;
877  e->m_nlink = stat_buf.st_nlink;
878  e->m_ino = stat_buf.st_ino;
879  } else {
880  e->isDir = isDir;
881  e->m_ctime = invalid_ctime;
882  e->m_status = NonExistent;
883  e->m_nlink = 0;
884  e->m_ino = 0;
885  }
886 
887  e->path = path;
888  if (sub_entry) {
889  e->m_entries.append(sub_entry);
890  } else {
891  e->addClient(instance, watchModes);
892  }
893 
894  if (s_verboseDebug) {
895  qCDebug(KDIRWATCH).nospace() << "Added " << (e->isDir ? "Dir " : "File ") << path
896  << (e->m_status == NonExistent ? " NotExisting" : "")
897  << " for " << (sub_entry ? sub_entry->path : QString())
898  << " [" << (instance ? instance->objectName() : QString()) << "]";
899  }
900 
901  // now setup the notification method
902  e->m_mode = UnknownMode;
903  e->msecLeft = 0;
904 
905  if (isNoisyFile(QFile::encodeName(path).data())) {
906  return;
907  }
908 
909  if (exists && e->isDir && (watchModes != KDirWatch::WatchDirOnly)) {
910  QFlags<QDir::Filter> filters = QDir::NoDotAndDotDot;
911 
912  if ((watchModes & KDirWatch::WatchSubDirs) &&
913  (watchModes & KDirWatch::WatchFiles)) {
914  filters |= (QDir::Dirs | QDir::Files);
915  } else if (watchModes & KDirWatch::WatchSubDirs) {
916  filters |= QDir::Dirs;
917  } else if (watchModes & KDirWatch::WatchFiles) {
918  filters |= QDir::Files;
919  }
920 
921 #if HAVE_SYS_INOTIFY_H
922  if (e->m_mode == INotifyMode || (e->m_mode == UnknownMode && m_preferredMethod == KDirWatch::INotify)) {
923  //qCDebug(KDIRWATCH) << "Ignoring WatchFiles directive - this is implicit with inotify";
924  // Placing a watch on individual files is redundant with inotify
925  // (inotify gives us WatchFiles functionality "for free") and indeed
926  // actively harmful, so prevent it. WatchSubDirs is necessary, though.
927  filters &= ~QDir::Files;
928  }
929 #endif
930 
931  QDir basedir(e->path);
932  const QFileInfoList contents = basedir.entryInfoList(filters);
933  for (QFileInfoList::const_iterator iter = contents.constBegin();
934  iter != contents.constEnd(); ++iter) {

936  // treat symlinks as files--don't follow them.
937  bool isDir = fileInfo.isDir() && !fileInfo.isSymLink();
938 
939  addEntry(instance, fileInfo.absoluteFilePath(), 0, isDir,
940  isDir ? watchModes : KDirWatch::WatchDirOnly);
941  }
942  }
943 
944  addWatch(e);
945 }

947 void KDirWatchPrivate::addWatch(Entry *e)
948 {
949  // If the watch is on a network filesystem use the nfsPreferredMethod as the
950  // default, otherwise use preferredMethod as the default, if the methods are
951  // the same we can skip the mountpoint check
952 
953  // This allows to configure a different method for NFS mounts, since inotify
954  // cannot detect changes made by other machines. However as a default inotify
955  // is fine, since the most common case is a NFS-mounted home, where all changes
956  // are made locally. #177892.
957  KDirWatch::Method preferredMethod = m_preferredMethod;
958  if (m_nfsPreferredMethod != m_preferredMethod) {
959  if (KFileSystemType::fileSystemType(e->path) == KFileSystemType::Nfs) {
960  preferredMethod = m_nfsPreferredMethod;
961  }
962  }
963 
964  // Try the appropriate preferred method from the config first
965  bool entryAdded = false;
966  switch (preferredMethod) {
967 #if HAVE_FAM
968  case KDirWatch::FAM: entryAdded = useFAM(e); break;
969 #endif
970 #if HAVE_SYS_INOTIFY_H
971  case KDirWatch::INotify: entryAdded = useINotify(e); break;
972 #endif
973 #if HAVE_QFILESYSTEMWATCHER
974  case KDirWatch::QFSWatch: entryAdded = useQFSWatch(e); break;
975 #endif
976  case KDirWatch::Stat: entryAdded = useStat(e); break;
977  }
978 
979  // Failing that try in order INotify, FAM, QFSWatch, Stat
980  if (!entryAdded) {
981 #if HAVE_SYS_INOTIFY_H
982  if (useINotify(e)) {
983  return;
984  }
985 #endif
986 #if HAVE_FAM
987  if (useFAM(e)) {
988  return;
989  }
990 #endif
991 #if HAVE_QFILESYSTEMWATCHER
992  if (useQFSWatch(e)) {
993  return;
994  }
995 #endif
996  useStat(e);
997  }
