45 bool KCompositeJob::addSubjob(KJob *job)
46 {
47  Q_D(KCompositeJob);
48  if (job == 0 || d->subjobs.contains(job)) {
49  return false;
50  }
51 
52  job->setParent(this);
53  d->subjobs.append(job);
54  connect(job, SIGNAL(result(KJob*)),
55  SLOT(slotResult(KJob*)));
56 
57  // Forward information from that subjob.
58  connect(job, SIGNAL(infoMessage(KJob*,QString,QString)),
59  SLOT(slotInfoMessage(KJob*,QString,QString)));
60 
61  return true;
62 }
63 
64 bool KCompositeJob::removeSubjob(KJob *job)
65 {
66  Q_D(KCompositeJob);
67  if (job == 0) {
68  return false;
69  }
70 
71  job->setParent(0);
72  d->subjobs.removeAll(job);
73 
74  return true;
75 }

96 void KCompositeJob::slotResult(KJob *job)
97 {
98  // Did job have an error ?
99  if (job->error() && !error()) {
100  // Store it in the parent only if first error
101  setError(job->error());
102  setErrorText(job->errorText());
103  emitResult();
104  }
105 
106  removeSubjob(job);
107 }
