30 /*
31  * A short introduction into cmd semantics:
32  * - Variable expansion is done first, without regard to *any* escaping -
33  * if something looks like an existing variable, it is replaced.
34  * - Then follows regular tokenization by the shell. &, &&, | and || are
35  * command delimiters. ( and ) are command grouping operators; they are
36  * recognized only a the start resp. end of a command; mismatched )s are
37  * an error if any (s are present. <, > are just like under UNIX - they can
38  * appear *anywhere* in a command, perform their function and are cut out.
39  * @ at the start of a command is eaten (local echo off - no function as
40  * far as cmd /c is concerned). : at the start of a command declares a label,
41  * which effectively means the remainder of the line is a comment - note that
42  * command separators are not recognized past that point.
43  * ^ is the escape char for everything including itself.
44  * cmd ignores *all* special chars between double quotes, so there is no
45  * way to escape the closing quote. Note that the quotes are *not* removed
46  * from the resulting command line.
47  * - Then follows delayed variable expansion if it is enabled and at least
48  * one exclamation mark is present. This involves another layer of ^
49  * escaping, regardless of quotes. (Win2k+)
50  * - Then follows argument splitting as described in
51  * http://msdn2.microsoft.com/en-us/library/ms880421.aspx .
52  * Note that this is done by the called application and therefore might
53  * be subject to completely different semantics, in fact.
54  */

56 inline static bool isMetaChar(ushort c)
57 {
58  static const uchar iqm[] = {
59  0x00, 0x00, 0x00, 0x00, 0x40, 0x03, 0x00, 0x50,
60  0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x10
61  }; // &()<>|
62 
63  return (c < sizeof(iqm) * 8) && (iqm[c / 8] & (1 << (c & 7)));
64 }

66 inline static bool isSpecialChar(ushort c)
67 {
68  // Chars that should be quoted (TM). This includes:
69  // - control chars & space
70  // - the shell meta chars &()<>^|
71  // - the potential separators ,;=
72  static const uchar iqm[] = {
73  0xff, 0xff, 0xff, 0xff, 0x41, 0x13, 0x00, 0x78,
74  0x00, 0x00, 0x00, 0x40, 0x00, 0x00, 0x00, 0x10
75  };
76 
77  return (c < sizeof(iqm) * 8) && (iqm[c / 8] & (1 << (c & 7)));
78 }

86 {
87  QString args(_args);
88  QStringList ret;
89 
90  const QLatin1Char bs('\\'), dq('\"');
91 
92  if (flags & AbortOnMeta) {
93  args.remove(PERCENT_ESCAPE);
94  if (args.indexOf(QLatin1Char('%')) >= 0) {
95  if (err) {
96  *err = FoundMeta;
97  }
98  return QStringList();
99  }
100 
101  args = _args;
102  args.replace(PERCENT_ESCAPE, QLatin1String("%"));
103 
104  if (!args.isEmpty() && args[0].unicode() == '@') {
105  args.remove(0, 1);
106  }
107 
108  for (int p = 0; p < args.length(); p++) {
109  ushort c = args[p].unicode();
110  if (c == '^') {
111  args.remove(p, 1);
112  } else if (c == '"') {
113  while (++p < args.length() && args[p].unicode() != '"')
114  ;
115  } else if (isMetaChar(c)) {
116  if (err) {
117  *err = FoundMeta;
118  }
119  return QStringList();
120  }
121  }
122  }
123 
124  if (err) {
125  *err = NoError;
126  }
127 
128  int p = 0;
129  const int length = args.length();
130  forever {
131  while (p < length && isWhiteSpace(args[p].unicode()))
132  {
133  ++p;
134  }
135  if (p == length)
136  {
137  return ret;
138  }
139 
140  QString arg;
141  bool inquote = false;
142  forever {
143  bool copy = true; // copy this char
144  int bslashes = 0; // number of preceding backslashes to insert
145  while (p < length && args[p] == bs)
146  {
147  ++p;
148  ++bslashes;
149  }
150  if (p < length && args[p] == dq)
151  {
152  if (bslashes % 2 == 0) {
153  // Even number of backslashes, so the quote is not escaped.
154  if (inquote) {
155  if (p + 1 < length && args[p + 1] == dq) {
156  // Two consecutive quotes make a literal quote.
157  // This is not documented on MSDN.
158  ++p;
159  } else {
160  // Closing quote
161  copy = false;
162  inquote = !inquote;
163  }
164  } else {
165  // Opening quote
166  copy = false;
167  inquote = !inquote;
168  }
169  }
170  bslashes /= 2;
171  }
172 
173  while (--bslashes >= 0)
174  {
175  arg.append(bs);
176  }
177 
178  if (p == length || (!inquote && isWhiteSpace(args[p].unicode())))
179  {
180  ret.append(arg);
181  if (inquote) {
182  if (err) {
183  *err = BadQuoting;
184  }
185  return QStringList();
186  }
187  break;
188  }
189 
190  if (copy)
191  {
192  arg.append(args[p]);
193  }
194  ++p;
195  }
196  }

200 QString KShell::quoteArgInternal(const QString &arg, bool _inquote)
201 {
202  // Escape quotes, preceding backslashes are doubled. Surround with quotes.
203  // Note that cmd does not understand quote escapes in quoted strings,
204  // so the quoting needs to be "suspended".
205  const QLatin1Char bs('\\'), dq('\"');
206  QString ret;
207  bool inquote = _inquote;
208  int bslashes = 0;
209  for (int p = 0; p < arg.length(); p++) {
210  if (arg[p] == bs) {
211  bslashes++;
212  } else if (arg[p] == dq) {
213  if (inquote) {
214  ret.append(dq);
215  inquote = false;
216  }
217  for (; bslashes; bslashes--) {
218  ret.append(QLatin1String("\\\\"));
219  }
220  ret.append(QLatin1String("\\^\""));
221  } else {
222  if (!inquote) {
223  ret.append(dq);
224  inquote = true;
225  }
226  for (; bslashes; bslashes--) {
227  ret.append(bs);
228  }
229  ret.append(arg[p]);
230  }
231  }
232  ret.replace(QLatin1Char('%'), PERCENT_ESCAPE);
233  if (bslashes) {
234  // Ensure that we don't have directly trailing backslashes,
235  // so concatenating with another string won't cause surprises.
236  if (!inquote && !_inquote) {
237  ret.append(dq);
238  }
239  for (; bslashes; bslashes--) {
240  ret.append(QLatin1String("\\\\"));
241  }
242  ret.append(dq);
243  if (inquote && _inquote) {
244  ret.append(dq);
245  }
246  } else if (inquote != _inquote) {
247  ret.append(dq);
248  }
249  return ret;
250 }

252 QString KShell::quoteArg(const QString &arg)
253 {
254  if (arg.isEmpty()) {
255  return QString::fromLatin1("\"\"");
256  }
257 
258  // Ensure that we don't have directly trailing backslashes,
259  // so concatenating with another string won't cause surprises.
260  if (arg.endsWith(QLatin1Char('\\'))) {
261  return quoteArgInternal(arg, false);
262  }
263 
264  for (int x = arg.length() - 1; x >= 0; --x)
265  if (isSpecialChar(arg[x].unicode())) {
266  return quoteArgInternal(arg, false);
267  }
268 
269  // Escape quotes. Preceding backslashes are doubled.
270  // Note that the remaining string is not quoted.
271  QString ret(arg);
272  ret.replace(QRegExp(QLatin1String("(\\\\*)\"")), QLatin1String("\\1\\1\\^\""));
273  ret.replace(QLatin1Char('%'), PERCENT_ESCAPE);
274  return ret;
275 }
