26 static QByteArray uriListData(const QList<QUrl> &urls)
27 {
28  // compatible with qmimedata.cpp encoding of QUrls
29  QByteArray result;
30  for (int i = 0; i < urls.size(); ++i) {
31  result += urls.at(i).toEncoded();
32  result += "\r\n";
33  }
34  return result;
35 }

64 QList<QUrl> KUrlMimeData::urlsFromMimeData(const QMimeData *mimeData,
65  DecodeOptions decodeOptions,
66  MetaDataMap *metaData)
67 {
68  QList<QUrl> uris;
69  const char *firstMimeType = s_kdeUriListMime;
70  const char *secondMimeType = "text/uri-list";
71  if (decodeOptions == PreferLocalUrls) {
72  qSwap(firstMimeType, secondMimeType);
73  }
74  QByteArray ba = mimeData->data(QString::fromLatin1(firstMimeType));
75  if (ba.isEmpty()) {
76  ba = mimeData->data(QString::fromLatin1(secondMimeType));
77  }
78  if (!ba.isEmpty()) {
79  // Code from qmimedata.cpp
80  QList<QByteArray> urls = ba.split('\n');
81  for (int i = 0; i < urls.size(); ++i) {
82  QByteArray data = urls.at(i).trimmed();
83  if (!data.isEmpty()) {
84  uris.append(QUrl::fromEncoded(data));
85  }
86  }
87  }
88  if (metaData) {
89  const QByteArray metaDataPayload = mimeData->data(QStringLiteral("application/x-kio-metadata"));
90  if (!metaDataPayload.isEmpty()) {
91  QString str = QString::fromUtf8(metaDataPayload.constData());
92  Q_ASSERT(str.endsWith(QLatin1String("$@@$")));
93  str.truncate(str.length() - 4);
94  const QStringList lst = str.split(QStringLiteral("$@@$"));
95  bool readingKey = true; // true, then false, then true, etc.
96  QString key;
97  for (QStringList::const_iterator it = lst.begin(); it != lst.end(); ++it) {
98  if (readingKey) {
99  key = *it;
100  } else {
101  metaData->insert(key, *it);
102  }
103  readingKey = !readingKey;
104  }
105  Q_ASSERT(readingKey); // an odd number of items would be, well, odd ;-)
106  }
107  }
108  return uris;
109 }
