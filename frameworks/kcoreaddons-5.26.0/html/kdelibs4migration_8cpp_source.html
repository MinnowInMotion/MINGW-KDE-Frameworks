38 Kdelibs4Migration::Kdelibs4Migration()
39  : d(new Kdelibs4MigrationPrivate)
40 {
41  if (qEnvironmentVariableIsSet("KDEHOME")) {
42  //qCDebug(KCOREADDONS_DEBUG) << "Using KDEHOME as the location of the old config file";
43  d->m_kdeHome = QString::fromLocal8Bit(qgetenv("KDEHOME"));
44  } else {
45  QDir homeDir = QDir::home();
46  QVector<QString> testSubdirs;
47  testSubdirs << QStringLiteral(KDE4_DEFAULT_HOME) << QStringLiteral(".kde4") << QStringLiteral(".kde");
48 #ifdef Q_OS_WIN
49  WCHAR wPath[MAX_PATH + 1];
50  if (SHGetFolderPathW(NULL, CSIDL_APPDATA, NULL, SHGFP_TYPE_CURRENT, wPath) == S_OK) {
51  testSubdirs << QDir::fromNativeSeparators(QString::fromUtf16((const ushort *) wPath)) +
52  QLatin1String("/" KDE4_DEFAULT_HOME);
53  }
54 #endif
55  Q_FOREACH (const QString &testSubdir, testSubdirs) {
56  if (homeDir.exists(testSubdir)) {
57  //qCDebug(KCOREADDONS_DEBUG) << "Using" << testSubdir << "as the location of the old config file";
58  d->m_kdeHome = homeDir.filePath(testSubdir);
59  break;
60  }
61  }
62  if (d->m_kdeHome.isEmpty()) {
63  d->m_kdeHome = homeDir.filePath(QStringLiteral(KDE4_DEFAULT_HOME));
64  }
65  }
66 
67  if (!d->m_kdeHome.isEmpty() && !d->m_kdeHome.endsWith(QLatin1Char('/'))) {
68  d->m_kdeHome.append(QLatin1Char('/'));
69  }
70 }

87 QString Kdelibs4Migration::locateLocal(const char *type, const QString &filename) const
88 {
89  if (d->m_kdeHome.isEmpty()) {
90  return QString();
91  }
92  const QString dir = saveLocation(type);
93  if (dir.isEmpty()) {
94  return QString();
95  }
96  const QString file = dir + filename;
97  if (QFile::exists(file)) {
98  return file;
99  }
100  return QString();
101 }

116 QString Kdelibs4Migration::saveLocation(const char *type, const QString &suffix) const
117 {
118  if (d->m_kdeHome.isEmpty()) {
119  return QString();
120  }
121  static const int numResources = sizeof(s_subdirs) / sizeof(*s_subdirs);
122  for (uint i = 0; i < numResources; ++i) {
123  if (qstrcmp(s_subdirs[i].type, type) == 0) {
124  QString dir = d->m_kdeHome + QString::fromLatin1(s_subdirs[i].subdir) + suffix;
125  if (!dir.endsWith(QLatin1Char('/'))) {
126  dir += QLatin1Char('/');
127  }
128  return dir;
129  }
130  }
131  qCWarning(KCOREADDONS_DEBUG) << "No such resource" << type;
132  return QString();
133 }
