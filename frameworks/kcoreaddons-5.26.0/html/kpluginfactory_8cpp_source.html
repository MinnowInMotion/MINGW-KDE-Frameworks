52 void KPluginFactory::registerPlugin(const QString &keyword, const QMetaObject *metaObject, CreateInstanceFunction instanceFunction)
53 {
54  Q_D(KPluginFactory);
55 
56  Q_ASSERT(metaObject);
57 
58  // we allow different interfaces to be registered without keyword
59  if (!keyword.isEmpty()) {
60  if (d->createInstanceHash.contains(keyword)) {
61  qCWarning(KCOREADDONS_DEBUG) << "A plugin with the keyword" << keyword << "was already registered. A keyword must be unique!";
62  }
63  d->createInstanceHash.insert(keyword, KPluginFactoryPrivate::Plugin(metaObject, instanceFunction));
64  } else {
65  QList<KPluginFactoryPrivate::Plugin> clashes(d->createInstanceHash.values(keyword));
66  const QMetaObject *superClass = metaObject->superClass();
67  if (superClass) {
68  foreach (const KPluginFactoryPrivate::Plugin &plugin, clashes) {
69  for (const QMetaObject *otherSuper = plugin.first->superClass(); otherSuper;
70  otherSuper = otherSuper->superClass()) {
71  if (superClass == otherSuper) {
72  qCWarning(KCOREADDONS_DEBUG) << "Two plugins with the same interface(" << superClass->className() << ") were registered. Use keywords to identify the plugins.";
73  }
74  }
75  }
76  }
77  foreach (const KPluginFactoryPrivate::Plugin &plugin, clashes) {
78  superClass = plugin.first->superClass();
79  if (superClass) {
80  for (const QMetaObject *otherSuper = metaObject->superClass(); otherSuper;
81  otherSuper = otherSuper->superClass()) {
82  if (superClass == otherSuper) {
83  qCWarning(KCOREADDONS_DEBUG) << "Two plugins with the same interface(" << superClass->className() << ") were registered. Use keywords to identify the plugins.";
84  }
85  }
86  }
87  }
88  d->createInstanceHash.insertMulti(keyword, KPluginFactoryPrivate::Plugin(metaObject, instanceFunction));
89  }
90 }

113 QObject *KPluginFactory::create(const char *iface, QWidget *parentWidget, QObject *parent, const QVariantList &args, const QString &keyword)
114 {
115  Q_D(KPluginFactory);
116 
117  QObject *obj = 0;
118 
119 #ifndef KCOREADDONS_NO_DEPRECATED
120  if (keyword.isEmpty()) {
121 
122  const QStringList argsStringList = variantListToStringList(args);
123 
124  if ((obj = reinterpret_cast<QObject *>(createPartObject(parentWidget, parent, iface, argsStringList)))) {
125  objectCreated(obj);
126  return obj;
127  }
128 
129  if ((obj = createObject(parent, iface, argsStringList))) {
130  objectCreated(obj);
131  return obj;
132  }
133  }
134 #endif
135 
136  const QList<KPluginFactoryPrivate::Plugin> candidates(d->createInstanceHash.values(keyword));
137  // for !keyword.isEmpty() candidates.count() is 0 or 1
138 
139  foreach (const KPluginFactoryPrivate::Plugin &plugin, candidates) {
140  for (const QMetaObject *current = plugin.first; current; current = current->superClass()) {
141  if (0 == qstrcmp(iface, current->className())) {
142  if (obj) {
143  qCWarning(KCOREADDONS_DEBUG) << "ambiguous interface requested from a DSO containing more than one plugin";
144  }
145  obj = plugin.second(parentWidget, parent, args);
146  break;
147  }
148  }
149  }
150 
151  if (obj) {
152  emit objectCreated(obj);
153  }
154  return obj;
155 }
