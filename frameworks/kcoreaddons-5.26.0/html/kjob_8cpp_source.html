96 void KJob::finishJob(bool emitResult)
97 {
98  Q_D(KJob);
99  d->isFinished = true;
100 
101  if (d->eventLoop) {
102  d->eventLoop->quit();
103  }
104 
105  // If we are displaying a progress dialog, remove it first.
106  emit finished(this, QPrivateSignal());
107 
108  if (emitResult) {
109  emit result(this, QPrivateSignal());
110  }
111 
112  if (isAutoDelete()) {
113  deleteLater();
114  }
115 }

180 bool KJob::exec()
181 {
182  Q_D(KJob);
183  // Usually this job would delete itself, via deleteLater() just after
184  // emitting result() (unless configured otherwise). Since we use an event
185  // loop below, that event loop will process the deletion event and we'll
186  // have been deleted when exec() returns. This crashes, so temporarily
187  // suspend autodeletion and manually do it afterwards.
188  const bool wasAutoDelete = isAutoDelete();
189  setAutoDelete(false);
190 
191  Q_ASSERT(! d->eventLoop);
192 
193  QEventLoop loop(this);
194  d->eventLoop = &loop;
195 
196  start();
197  if (!d->isFinished) {
198  d->eventLoop->exec(QEventLoop::ExcludeUserInputEvents);
199  }
200  d->eventLoop = 0;
201 
202  if (wasAutoDelete) {
203  deleteLater();
204  }
205  return (d->error == NoError);
206 }

250 void KJob::setProcessedAmount(Unit unit, qulonglong amount)
251 {
252  Q_D(KJob);
253  bool should_emit = (d->processedAmount[unit] != amount);
254 
255  d->processedAmount[unit] = amount;
256 
257  if (should_emit) {
258  emit processedAmount(this, unit, amount);
259  if (unit == d->progressUnit) {
260  emit processedSize(this, amount);
261  emitPercent(d->processedAmount[unit], d->totalAmount[unit]);
262  }
263  }
264 }

296 void KJob::emitPercent(qulonglong processedAmount, qulonglong totalAmount)
297 {
298  Q_D(KJob);
299  // calculate percents
300  if (totalAmount) {
301  unsigned long oldPercentage = d->percentage;
302  d->percentage = 100.0 * processedAmount / totalAmount;
303  if (d->percentage != oldPercentage) {
304  emit percent(this, d->percentage);
305  }
306  }
307 }

309 void KJob::emitSpeed(unsigned long value)
310 {
311  Q_D(KJob);
312  if (!d->speedTimer) {
313  d->speedTimer = new QTimer(this);
314  connect(d->speedTimer, SIGNAL(timeout()), SLOT(_k_speedTimeout()));
315  }
316 
317  emit speed(this, value);
318  d->speedTimer->start(5000); // 5 seconds interval should be enough
319 }
