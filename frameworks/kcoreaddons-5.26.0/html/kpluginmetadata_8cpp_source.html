61 KPluginMetaData::KPluginMetaData(const QString &file)
62 {
63  if (file.endsWith(QStringLiteral(".desktop"))) {
64  loadFromDesktopFile(file, QStringList());
65  } else if (file.endsWith(QStringLiteral(".json"))) {
66  d = new KPluginMetaDataPrivate;
67  QFile f(file);
68  bool b = f.open(QIODevice::ReadOnly);
69  if (!b) {
70  qCWarning(KCOREADDONS_DEBUG) << "Couldn't open" << file;
71  return;
72  }
73  m_metaData = QJsonDocument::fromJson(f.readAll()).object();
74  m_fileName = file;
75  d->metaDataFileName = file;
76  } else {
77  QPluginLoader loader(file);
78  m_fileName = QFileInfo(loader.fileName()).absoluteFilePath();
79  m_metaData = loader.metaData().value(QStringLiteral("MetaData")).toObject();
80  }
81 }

118 void KPluginMetaData::loadFromDesktopFile(const QString &file, const QStringList &serviceTypes)
119 {
120  QString libraryPath;
121  if (!DesktopFileParser::convert(file, serviceTypes, m_metaData, &libraryPath)) {
122  Q_ASSERT(!isValid());
123  return; // file could not be parsed for some reason, leave this object invalid
124  }
125  d = new KPluginMetaDataPrivate;
126  d->metaDataFileName = QFileInfo(file).absoluteFilePath();
127  if (!libraryPath.isEmpty()) {
128  // this was a plugin with a shared library
129  m_fileName = libraryPath;
130  } else {
131  // no library, make filename point to the .desktop file
132  m_fileName = d->metaDataFileName;
133  }
134 }

168 QStringList KPluginMetaData::readStringList(const QJsonObject &obj, const QString &key)
169 {
170  const QJsonValue value = obj.value(key);
171  if (value.isUndefined() || value.isObject() || value.isNull()) {
172  return QStringList();
173  } else if (value.isArray()) {
174  return value.toVariant().toStringList();
175  } else {
176  QString asString = value.isString() ? value.toString() : value.toVariant().toString();
177  const QString id = obj.value(QStringLiteral("KPlugin")).toObject().value(QStringLiteral("Id")).toString();
178  qCWarning(KCOREADDONS_DEBUG) << "Expected JSON property" << key << "to be a string list."
179  " Treating it as a list with a single entry:" << asString << id.toLatin1().constData();
180  return QStringList(asString);
181  }
182 }

184 QJsonValue KPluginMetaData::readTranslatedValue(const QJsonObject &jo, const QString &key, const QJsonValue &defaultValue)
185 {
186  QString languageWithCountry = QLocale().name();
187  auto it = jo.constFind(key + QLatin1Char('[') + languageWithCountry + QLatin1Char(']'));
188  if (it != jo.constEnd()) {
189  return it.value();
190  }
191  const QStringRef language = languageWithCountry.midRef(0, languageWithCountry.indexOf(QLatin1Char('_')));
192  it = jo.constFind(key + QLatin1Char('[') + language + QLatin1Char(']'));
193  if (it != jo.constEnd()) {
194  return it.value();
195  }
196  // no translated value found -> check key
197  it = jo.constFind(key);
198  if (it != jo.constEnd()) {
199  return jo.value(key);
200  }
201  return defaultValue;
202 }

209 static inline void addPersonFromJson(const QJsonObject &obj, QList<KAboutPerson>* out) {
210  KAboutPerson person = KAboutPerson::fromJSON(obj);
211  if (person.name().isEmpty()) {
212  qCWarning(KCOREADDONS_DEBUG) << "Invalid plugin metadata: Attempting to create a KAboutPerson from json without 'Name' property:" << obj;
213  return;

215  out->append(person);

218 static QList<KAboutPerson> aboutPersonFromJSON(const QJsonValue &people)
219 {
220  QList<KAboutPerson> ret;
221  if (people.isObject()) {
222  // single author
223  addPersonFromJson(people.toObject(), &ret);
224  } else if (people.isArray()) {
225  foreach (const QJsonValue &val, people.toArray()) {
226  if (val.isObject()) {
227  addPersonFromJson(val.toObject(), &ret);
228  }
229  }
230  }
231  return ret;
232 }

284 QString KPluginMetaData::pluginId() const
285 {
286  QJsonObject root = rootObject();
287  auto nameFromMetaData = root.constFind(QStringLiteral("Id"));
288  if (nameFromMetaData != root.constEnd()) {
289  const QString id = nameFromMetaData.value().toString();
290  if (!id.isEmpty()) {
291  return id;
292  }
293  }
294  // passing QFileInfo an empty string gives the CWD, which is not what we want
295  if (m_fileName.isEmpty()) {
296  return QString();
297  }
298  return QFileInfo(m_fileName).baseName();
299 }
