180 bool KPluginLoader::load()
181 {
182  Q_D(KPluginLoader);
183 
184  if (!d->loader->load()) {
185  return false;
186  }
187 
188  if (d->pluginVersionResolved) {
189  return true;
190  }
191 
192  Q_ASSERT(!fileName().isEmpty());
193  QLibrary lib(fileName());
194  Q_ASSERT(lib.isLoaded()); // already loaded by QPluginLoader::load()
195 
196  // TODO: this messes up KPluginLoader::errorString(): it will change from unknown error to could not resolve kde_plugin_version
197  quint32 *version = reinterpret_cast<quint32 *>(lib.resolve("kde_plugin_version"));
198  if (version) {
199  d->pluginVersion = *version;
200  } else {
201  d->pluginVersion = ~0U;
202  }
203  d->pluginVersionResolved = true;
204 
205  return true;
206 }

241 void KPluginLoader::forEachPlugin(const QString &directory, std::function<void(const QString &)> callback)
242 {
243  QStringList dirsToCheck;
244  if (QDir::isAbsolutePath(directory)) {
245  dirsToCheck << directory;
246  } else {
247  foreach (const QString &libDir, QCoreApplication::libraryPaths()) {
248  dirsToCheck << libDir + QDir::separator() + directory;
249  }
250  }
251 
252  foreach (const QString &dir, dirsToCheck) {
253  QDirIterator it(dir, QDir::Files);
254  while (it.hasNext()) {
255  it.next();
256  if (QLibrary::isLibrary(it.fileName())) {
257  callback(it.fileInfo().absoluteFilePath());
258  }
259  }
260  }
261 }

263 QVector<KPluginMetaData> KPluginLoader::findPlugins(const QString &directory, std::function<bool(const KPluginMetaData &)> filter)
264 {
265  QVector<KPluginMetaData> ret;
266  forEachPlugin(directory, [&](const QString &pluginPath) {
267  KPluginMetaData metadata(pluginPath);
268  if (!metadata.isValid()) {
269  return;
270  }
271  if (filter && !filter(metadata)) {
272  return;
273  }
274  ret.append(metadata);
275  });
276  return ret;
277 }

289  std::function<bool(const KPluginMetaData &)> filter, QObject* parent)
290 {
291  QList<QObject *> ret;
292  QPluginLoader loader;
293  foreach (const KPluginMetaData &metadata, findPlugins(directory, filter)) {
294  loader.setFileName(metadata.fileName());
