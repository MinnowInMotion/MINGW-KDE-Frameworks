82 {

103 template<class T, class Callback, class EnumFunction>
104 static void netApiEnumerate(uint maxCount, Callback callback, EnumFunction enumFunc)
105 {
106  NET_API_STATUS nStatus = NERR_Success;
107  DWORD_PTR resumeHandle = 0;
108  uint total = 0;
109  int level = NetApiTypeInfo<T>::level;
110  do {
111  LPBYTE buffer = nullptr;
112  DWORD entriesRead = 0;
113  DWORD totalEntries = 0;
114  nStatus = enumFunc(level, &buffer, &entriesRead, &totalEntries, &resumeHandle);
115  //qDebug("Net*Enum(level = %d) returned %d entries, total was (%d), status = %d, resume handle = %llx",
116  // level, entriesRead, totalEntries, nStatus, resumeHandle);
117 
118  // buffer must always be freed, even if Net*Enum fails
119  ScopedNetApiBuffer<T> groupInfo((T*)buffer);
120  if (nStatus == NERR_Success || nStatus == ERROR_MORE_DATA) {
121  for (DWORD i = 0; total < maxCount && i < entriesRead; i++, total++) {
122  callback(groupInfo.get()[i]);
123  }
124  } else {
125  qWarning("NetApi enumerate function failed: status = %d", (int)nStatus);
126  }
127  } while (nStatus == ERROR_MORE_DATA);
128 }

187  static QString guessHomeDir(const QString &username, KUserId uid)
188  {
189  // usri11_home_dir/usri4_home_dir is often empty
190  // check whether it is the homedir for the current user and if not then fall back to "<user profiles dir><user name>"
191  if (uid == KUserId::currentUserId()) {
192  return QDir::homePath();
193  }
194  QString homeDir;
195  WCHAR profileDirPath[MAX_PATH];
196  DWORD bufSize = MAX_PATH;
197  BOOL result = GetProfilesDirectoryW(profileDirPath, &bufSize);
198  if (result) {
199  // This might not be correct: e.g. with local user and domain user with same
200  // In that case it could be C:\Users\Foo (local user) vs C:\Users\Foo.DOMAIN (domain user)
201  // However it is still much better than the previous code which just returned the current users home dir
202  homeDir = QString::fromWCharArray(profileDirPath) + QLatin1Char('\\') + username;
203  }
204  return homeDir;
205  }

217  static Ptr create(KUserId uid)
218  {
219  if (!uid.isValid()) {
220  return sharedNull;
221  }
222  // now find the fully qualified name for the user
223  DWORD nameBufferLen = UNLEN + 1;
224  WCHAR nameBuffer[UNLEN + 1];
225  DWORD domainBufferLen = UNLEN + 1;
226  WCHAR domainBuffer[UNLEN + 1];
227  SID_NAME_USE use;
228  if (!LookupAccountSidW(nullptr, uid.nativeId(), nameBuffer, &nameBufferLen, domainBuffer, &domainBufferLen, &use)) {
229  qCWarning(KCOREADDONS_DEBUG) << "Could not lookup user " << uid.toString() << "error =" << GetLastError();
230  return sharedNull;
231  }
232  QString loginName = QString::fromWCharArray(nameBuffer);
233  QString domainName = QString::fromWCharArray(domainBuffer);
234  if (use != SidTypeUser && use != SidTypeDeletedAccount) {
235  qCWarning(KCOREADDONS_DEBUG).nospace()
236  << "SID for " << domainName << "\\" << loginName << " (" << uid.toString()
237  << ") is not of type user (" << SidTypeUser << " or " << SidTypeDeletedAccount
238  << "). Got type " << use << " instead.";
239  return sharedNull;
240  }
241  // now get the server name to query (could be null for local machine)
242  LPWSTR servernameTmp = nullptr;
243  NET_API_STATUS status = NetGetAnyDCName(nullptr, 0, (LPBYTE *)&servernameTmp);
244  if (status != NERR_Success) {
245  // this always fails on my desktop system, don't spam the output
246  // qDebug("NetGetAnyDCName failed with error %d", status);
247  }
248  ScopedNetApiBuffer<WCHAR> servername(servernameTmp);
249 
250  QString fullName;
251  QString homeDir;
252  KGroupId group;
253  bool isAdmin = false;
254  // must NOT pass the qualified name ("domain\user") here or lookup fails -> just the name
255  // try USER_INFO_4 first, MSDN says it is valid only on servers (whatever that means), it works on my desktop system
256  // If it fails fall back to USER_INFO11, which has all the needed information except primary group
257  if (auto userInfo4 = getUserInfo<USER_INFO_4>(servername.get(), loginName, &status)) {
258  Q_ASSERT(KUserId(userInfo4->usri4_user_sid) == uid); // if this is not the same we have a logic error
259  fullName = QString::fromWCharArray(userInfo4->usri4_full_name);
260  homeDir = QString::fromWCharArray(userInfo4->usri4_home_dir);
261  isAdmin = userInfo4->usri4_priv == USER_PRIV_ADMIN;
262  // now determine the primary group:
263  const DWORD primaryGroup = userInfo4->usri4_primary_group_id;
264  // primary group is a relative identifier, i.e. in order to get the SID for that group
265  // we have to take the user SID and replace the last subauthority value with the relative identifier
266  group = KGroupId(uid.nativeId()); // constructor does not check whether the sid refers to a group
267  Q_ASSERT(group.isValid());
268  UCHAR numSubauthorities = *GetSidSubAuthorityCount(group.nativeId());
269  PDWORD lastSubAutority = GetSidSubAuthority(group.nativeId(), numSubauthorities - 1);
270  *lastSubAutority = primaryGroup;
271  } else if (auto userInfo11 = getUserInfo<USER_INFO_11>(servername.get(), loginName, &status)) {
272  fullName = QString::fromWCharArray(userInfo11->usri11_full_name);
273  homeDir = QString::fromWCharArray(userInfo11->usri11_home_dir);
274  isAdmin = userInfo11->usri11_priv == USER_PRIV_ADMIN;
275  } else {
276  qCWarning(KCOREADDONS_DEBUG).nospace() << "Could not get information for user " << domainName << "\\" << loginName
277  << ": error code = " << status;
278  return sharedNull;
279  }
280  if (homeDir.isEmpty()) {
281  homeDir = guessHomeDir(loginName, uid);
282  }
283  //if we couldn't find a primary group just take the first group found for this user
284  if (!group.isValid()) {
285  enumerateGroupsForUser<GROUP_USERS_INFO_0>(1, loginName, [&](const GROUP_USERS_INFO_0 &info) {
286  group = KGroupId::fromName(QString::fromWCharArray(info.grui0_name));
287  });
288  }
289  return Ptr(new Private(uid, group, loginName, fullName, domainName, homeDir, isAdmin));
290  }

419 template <typename Platform>
420 static QString faceIconPathImpl(LPCWSTR username)
421 {
422  static COMInitializer COMinit;
423 
424  static W32Library shellMod = LoadLibraryA("shell32.dll");
425  if (!shellMod) {
426  return QString();
427  }
428  static typename Platform::funcptr_t sgupp_ptr = reinterpret_cast<typename Platform::funcptr_t>(
429  GetProcAddress(shellMod, MAKEINTRESOURCEA(Platform::ordinal)));
430  if (!sgupp_ptr) {
431  return QString();
432  }
433 
434  WCHAR pathBuf[MAX_PATH];
435 
436  HRESULT res = Platform::getPicturePath(sgupp_ptr, username, pathBuf, MAX_PATH);
437  if (res != S_OK) {
438  return QString();
439  }
440  return QString::fromWCharArray(pathBuf);

800 template<class T, class Callback>
801 static T sidFromName(const QString &name, Callback callback)
802 {
803  if (name.isEmpty()) {
804  // for some reason empty string will always return S-1-5-32 which is of type domain
805  // we only want users or groups -> return invalid
806  return T();
807  }
808  char buffer[SECURITY_MAX_SID_SIZE];
809  DWORD sidLength = SECURITY_MAX_SID_SIZE;
810  // ReferencedDomainName must be passed or LookupAccountNameW fails
811  // Documentation says it is optional, however if not passed the function fails and returns the required size
812  WCHAR domainBuffer[1024];
813  DWORD domainBufferSize = 1024;
814  SID_NAME_USE sidType;
815  bool ok = LookupAccountNameW(nullptr, (LPCWSTR)name.utf16(), buffer, &sidLength, domainBuffer, &domainBufferSize, &sidType);
816  if (!ok) {
817  qCWarning(KCOREADDONS_DEBUG) << "Failed to lookup account" << name << "error code =" << GetLastError();
818  return T();
819  }
820  return callback(buffer, sidType);
821 }

851 static std::unique_ptr<char[]> queryProcessInformation(TOKEN_INFORMATION_CLASS type)
852 {
853  HANDLE _token;
854  if (!OpenProcessToken(GetCurrentProcess(), TOKEN_QUERY, &_token)) {
855  qWarning("Failed to get the token for the current process: %d", (int)GetLastError());
856  return nullptr;
857  }
858  ScopedHANDLE token(_token, handleCloser);
859  // query required size
860  DWORD requiredSize;
861  if (!GetTokenInformation(token.get(), type, nullptr, 0, &requiredSize)) {
862  if (GetLastError() != ERROR_INSUFFICIENT_BUFFER) {
863  qWarning("Failed to get the required size for the token information %d: %d",
864  type, (int)GetLastError());
865  return nullptr;
866  }
867  }
868  std::unique_ptr<char[]> buffer(new char[requiredSize]);
869  if (!GetTokenInformation(token.get(), type, buffer.get(), requiredSize, &requiredSize)) {
870  qWarning("Failed to get token information %d from current process: %d",
871  type, (int)GetLastError());
872  return nullptr;
873  }
874  return buffer;
875 }
