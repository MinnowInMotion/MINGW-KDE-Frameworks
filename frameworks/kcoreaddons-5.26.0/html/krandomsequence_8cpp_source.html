79 void KRandomSequence::setSeed(int lngSeed1)
80 {
81  // Convert the positive seed number to a negative one so that the draw()
82  // function can intialise itself the first time it is called. We just have
83  // to make sure that the seed used != 0 as zero perpetuates itself in a
84  // sequence of random numbers.
85  if (lngSeed1 < 0) {
86  d->lngSeed1 = -1;
87  } else if (lngSeed1 == 0) {
88  d->lngSeed1 = -((KRandom::random() & ~1) + 1);
89  } else {
90  d->lngSeed1 = -lngSeed1;
91  }
92 }

94 static const int sMod1 = 2147483563;
95 static const int sMod2 = 2147483399;
96 
97 void KRandomSequence::Private::draw()
98 {
99  static const int sMM1 = sMod1 - 1;
100  static const int sA1 = 40014;
101  static const int sA2 = 40692;
102  static const int sQ1 = 53668;
103  static const int sQ2 = 52774;
104  static const int sR1 = 12211;
105  static const int sR2 = 3791;
106  static const int sDiv = 1 + sMM1 / SHUFFLE_TABLE_SIZE;
107 
108  // Long period (>2 * 10^18) random number generator of L'Ecuyer with
109  // Bayes-Durham shuffle and added safeguards. Returns a uniform random
110  // deviate between 0.0 and 1.0 (exclusive of the endpoint values). Call
111  // with a negative number to initialize; thereafter, do not alter idum
112  // between successive deviates in a sequence. RNMX should approximate
113  // the largest floating point value that is less than 1.
114 
115  int j; // Index for the shuffle table
116  int k;
117 
118  // Initialise
119  if (lngSeed1 <= 0) {
120  lngSeed2 = lngSeed1;
121 
122  // Load the shuffle table after 8 warm-ups
123  for (j = SHUFFLE_TABLE_SIZE + 7; j >= 0; --j) {
124  k = lngSeed1 / sQ1;
125  lngSeed1 = sA1 * (lngSeed1 - k * sQ1) - k * sR1;
126  if (lngSeed1 < 0) {
127  lngSeed1 += sMod1;
128  }
129 
130  if (j < SHUFFLE_TABLE_SIZE) {
131  shuffleArray[j] = lngSeed1;
132  }
133  }
134 
135  lngShufflePos = shuffleArray[0];
136  }
137 
138  // Start here when not initializing
139 
140  // Compute lngSeed1 = ( lngIA1*lngSeed1 ) % lngIM1 without overflows
141  // by Schrage's method
142  k = lngSeed1 / sQ1;
143  lngSeed1 = sA1 * (lngSeed1 - k * sQ1) - k * sR1;
144  if (lngSeed1 < 0) {
145  lngSeed1 += sMod1;
146  }
147 
148  // Compute lngSeed2 = ( lngIA2*lngSeed2 ) % lngIM2 without overflows
149  // by Schrage's method
150  k = lngSeed2 / sQ2;
151  lngSeed2 = sA2 * (lngSeed2 - k * sQ2) - k * sR2;
152  if (lngSeed2 < 0) {
153  lngSeed2 += sMod2;
154  }
155 
156  j = lngShufflePos / sDiv;
157  lngShufflePos = shuffleArray[j] - lngSeed2;
158  shuffleArray[j] = lngSeed1;
159 
160  if (lngShufflePos < 1) {
161  lngShufflePos += sMM1;
162  }
163 }

165 void
166 KRandomSequence::modulate(int i)
167 {
168  d->lngSeed2 -= i;
169  if (d->lngSeed2 < 0) {
170  d->lngShufflePos += sMod2;
171  }
172  d->draw();
173  d->lngSeed1 -= i;
174  if (d->lngSeed1 < 0) {
175  d->lngSeed1 += sMod1;
176  }
177  d->draw();
178 }

180 double
181 KRandomSequence::getDouble()
182 {
183  static const double finalAmp = 1.0 / double(sMod1);
184  static const double epsilon = 1.2E-7;
185  static const double maxRand = 1.0 - epsilon;
186  double temp;
187  d->draw();
188  // Return a value that is not one of the endpoints
189  if ((temp = finalAmp * d->lngShufflePos) > maxRand) {
190  // We don't want to return 1.0
191  return maxRand;
192  } else {
193  return temp;
194  }
195 }
