199 QString KAboutLicense::text() const
200 {
201  QString result;
202 
203  const QString lineFeed = QStringLiteral("\n\n");
204 
205  if (d->_aboutData && !d->_aboutData->copyrightStatement().isEmpty()) {
206  result = d->_aboutData->copyrightStatement() + lineFeed;
207  }
208 
209  bool knownLicense = false;
210  QString pathToFile; // rel path if known license
211  switch (d->_licenseKey) {
212  case KAboutLicense::File:
213  pathToFile = d->_pathToLicenseTextFile;
214  break;
215  case KAboutLicense::GPL_V2:
216  knownLicense = true;
217  pathToFile = QStringLiteral("GPL_V2");
218  break;
219  case KAboutLicense::LGPL_V2:
220  knownLicense = true;
221  pathToFile = QStringLiteral("LGPL_V2");
222  break;
223  case KAboutLicense::BSDL:
224  knownLicense = true;
225  pathToFile = QStringLiteral("BSD");
226  break;
227  case KAboutLicense::Artistic:
228  knownLicense = true;
229  pathToFile = QStringLiteral("ARTISTIC");
230  break;
231  case KAboutLicense::QPL_V1_0:
232  knownLicense = true;
233  pathToFile = QStringLiteral("QPL_V1.0");
234  break;
235  case KAboutLicense::GPL_V3:
236  knownLicense = true;
237  pathToFile = QStringLiteral("GPL_V3");
238  break;
239  case KAboutLicense::LGPL_V3:
240  knownLicense = true;
241  pathToFile = QStringLiteral("LGPL_V3");
242  break;
243  case KAboutLicense::LGPL_V2_1:
244  knownLicense = true;
245  pathToFile = QStringLiteral("LGPL_V21");
246  break;
247  case KAboutLicense::Custom:
248  if (!d->_licenseText.isEmpty()) {
249  result = d->_licenseText;
250  break;
251  }
252  // fall through
253  default:
254  result += QCoreApplication::translate(
255  "KAboutLicense",
256  "No licensing terms for this program have been specified.\n"
257  "Please check the documentation or the source for any\n"
258  "licensing terms.\n");
259  }
260 
261  if (knownLicense) {
262  pathToFile = QStandardPaths::locate(QStandardPaths::GenericDataLocation,
263  QString::fromLatin1("kf5/licenses/") + pathToFile);
264  result += QCoreApplication::translate(
265  "KAboutLicense",
266  "This program is distributed under the terms of the %1.").arg(name(KAboutLicense::ShortName));
267  if (!pathToFile.isEmpty()) {
268  result += lineFeed;
269  }
270  }
271 
272  if (!pathToFile.isEmpty()) {
273  QFile file(pathToFile);
274  if (file.open(QIODevice::ReadOnly)) {
275  QTextStream str(&file);
276  result += str.readAll();
277  }
278  }
279 
280  return result;
281 }

283 QString KAboutLicense::name(KAboutLicense::NameFormat formatName) const
284 {
285  QString licenseShort;
286  QString licenseFull;
287 
288  switch (d->_licenseKey) {
289  case KAboutLicense::GPL_V2:
290  licenseShort = QCoreApplication::translate("KAboutLicense", "GPL v2", "@item license (short name)");
291  licenseFull = QCoreApplication::translate("KAboutLicense", "GNU General Public License Version 2", "@item license");
292  break;
293  case KAboutLicense::LGPL_V2:
294  licenseShort = QCoreApplication::translate("KAboutLicense", "LGPL v2", "@item license (short name)");
295  licenseFull = QCoreApplication::translate("KAboutLicense", "GNU Lesser General Public License Version 2", "@item license");
296  break;
297  case KAboutLicense::BSDL:
298  licenseShort = QCoreApplication::translate("KAboutLicense", "BSD License", "@item license (short name)");
299  licenseFull = QCoreApplication::translate("KAboutLicense", "BSD License", "@item license");
300  break;
301  case KAboutLicense::Artistic:
302  licenseShort = QCoreApplication::translate("KAboutLicense", "Artistic License", "@item license (short name)");
303  licenseFull = QCoreApplication::translate("KAboutLicense", "Artistic License", "@item license");
304  break;
305  case KAboutLicense::QPL_V1_0:
306  licenseShort = QCoreApplication::translate("KAboutLicense", "QPL v1.0", "@item license (short name)");
307  licenseFull = QCoreApplication::translate("KAboutLicense", "Q Public License", "@item license");
308  break;
309  case KAboutLicense::GPL_V3:
310  licenseShort = QCoreApplication::translate("KAboutLicense", "GPL v3", "@item license (short name)");
311  licenseFull = QCoreApplication::translate("KAboutLicense", "GNU General Public License Version 3", "@item license");
312  break;
313  case KAboutLicense::LGPL_V3:
314  licenseShort = QCoreApplication::translate("KAboutLicense", "LGPL v3", "@item license (short name)");
315  licenseFull = QCoreApplication::translate("KAboutLicense", "GNU Lesser General Public License Version 3", "@item license");
316  break;
317  case KAboutLicense::LGPL_V2_1:
318  licenseShort = QCoreApplication::translate("KAboutLicense", "LGPL v2.1", "@item license (short name)");
319  licenseFull = QCoreApplication::translate("KAboutLicense", "GNU Lesser General Public License Version 2.1", "@item license");
320  break;
321  case KAboutLicense::Custom:
322  case KAboutLicense::File:
323  licenseShort = licenseFull = QCoreApplication::translate("KAboutLicense", "Custom", "@item license");
324  break;
325  default:
326  licenseShort = licenseFull = QCoreApplication::translate("KAboutLicense", "Not specified", "@item license");
327  }
328 
329  const QString result =
330  (formatName == KAboutLicense::ShortName) ? licenseShort :
331  (formatName == KAboutLicense::FullName) ? licenseFull :
332  QString();
333 
334  return result;
335 }

348 KAboutLicense KAboutLicense::byKeyword(const QString &rawKeyword)
349 {
350  // Setup keyword->enum dictionary on first call.
351  // Use normalized keywords, by the algorithm below.
352  static QHash<QByteArray, KAboutLicense::LicenseKey> ldict;
353  if (ldict.isEmpty()) {
354  ldict.insert("gpl", KAboutLicense::GPL);
355  ldict.insert("gplv2", KAboutLicense::GPL_V2);
356  ldict.insert("gplv2+", KAboutLicense::GPL_V2);
357  ldict.insert("lgpl", KAboutLicense::LGPL);
358  ldict.insert("lgplv2", KAboutLicense::LGPL_V2);
359  ldict.insert("lgplv2+", KAboutLicense::LGPL_V2);
360  ldict.insert("bsd", KAboutLicense::BSDL);
361  ldict.insert("artistic", KAboutLicense::Artistic);
362  ldict.insert("qpl", KAboutLicense::QPL);
363  ldict.insert("qplv1", KAboutLicense::QPL_V1_0);
364  ldict.insert("qplv10", KAboutLicense::QPL_V1_0);
365  ldict.insert("gplv3", KAboutLicense::GPL_V3);
366  ldict.insert("gplv3+", KAboutLicense::GPL_V3);
367  ldict.insert("lgplv3", KAboutLicense::LGPL_V3);
368  ldict.insert("lgplv3+", KAboutLicense::LGPL_V3);
369  ldict.insert("lgplv21", KAboutLicense::LGPL_V2_1);
370  ldict.insert("lgplv21+", KAboutLicense::LGPL_V2_1);
371  }
372 
373  // Normalize keyword.
374  QString keyword = rawKeyword;
375  keyword = keyword.toLower();
376  keyword.remove(QLatin1Char(' '));
377  keyword.remove(QLatin1Char('.'));
378 
379  LicenseKey license = ldict.value(keyword.toLatin1(),
380  KAboutLicense::Custom);
381  return KAboutLicense(license, 0);
382 }

420 KAboutData::KAboutData(const QString &_componentName,
421  const QString &_displayName,
422  const QString &_version,
423  const QString &_shortDescription,
424  enum KAboutLicense::LicenseKey licenseType,
425  const QString &_copyrightStatement,
426  const QString &text,
427  const QString &homePageAddress,
428  const QString &bugsEmailAddress
429  )
430  : d(new Private)
431 {
432  d->_componentName = _componentName;
433  int p = d->_componentName.indexOf(QLatin1Char('/'));
434  if (p >= 0) {
435  d->_componentName = d->_componentName.mid(p + 1);
436  }
437 
438  d->_displayName = _displayName;
439  if (!d->_displayName.isEmpty()) { // KComponentData("klauncher") gives empty program name
440  d->_internalProgramName = _displayName.toUtf8();
441  }
442  d->_version = _version.toUtf8();
443  d->_shortDescription = _shortDescription;
444  d->_licenseList.append(KAboutLicense(licenseType, this));
445  d->_copyrightStatement = _copyrightStatement;
446  d->_otherText = text;
447  d->_homepageAddress = homePageAddress;
448  d->_bugEmailAddress = bugsEmailAddress.toUtf8();
449 
450  QUrl homePageUrl(homePageAddress);
451  if (!homePageUrl.isValid() || homePageUrl.scheme().isEmpty()) {
452  // Default domain if nothing else is better
453  homePageUrl.setUrl(QStringLiteral("https://kde.org/"));
454  }
455 
456  const QChar dotChar(QLatin1Char('.'));
457  QStringList hostComponents = homePageUrl.host().split(dotChar);
458 
459  // Remove leading component unless 2 (or less) components are present
460  if (hostComponents.size() > 2) {
461  hostComponents.removeFirst();
462  }
463 
464  d->organizationDomain = hostComponents.join(dotChar);
465 
466  // desktop file name is reverse domain name
467  std::reverse(hostComponents.begin(), hostComponents.end());
468  hostComponents.append(_componentName);
469 
470  d->desktopFileName = hostComponents.join(dotChar);
471 }

806 QList<KAboutPerson> KAboutData::Private::parseTranslators(const QString &translatorName, const QString &translatorEmail)
807 {
808  QList<KAboutPerson> personList;
809  if (translatorName.isEmpty() || translatorName == QStringLiteral("Your names")) {
810  return personList;
811  }
812 
813  const QStringList nameList(translatorName.split(QLatin1Char(',')));
814 
815  QStringList emailList;
816  if (!translatorEmail.isEmpty() && translatorEmail != QStringLiteral("Your emails")) {
817  emailList = translatorEmail.split(QLatin1Char(','), QString::KeepEmptyParts);
818  }
819 
820  QStringList::const_iterator nit;
821  QStringList::const_iterator eit = emailList.constBegin();
822 
823  for (nit = nameList.constBegin(); nit != nameList.constEnd(); ++nit) {
824  QString email;
825  if (eit != emailList.constEnd()) {
826  email = *eit;
827  ++eit;
828  }
829 
830  personList.append(KAboutPerson((*nit).trimmed(), email.trimmed(), true));
831  }
832 
833  return personList;

944 KAboutData KAboutData::applicationData()
945 {
946  QCoreApplication *app = QCoreApplication::instance();
947 
948  KAboutData *aboutData = s_registry->m_appData;
949 
950  // not yet existing
951  if (!aboutData) {
952  // init from current Q*Application data
953  aboutData = new KAboutData(QCoreApplication::applicationName(),
954  QString(),
955  QString());
956  // For applicationDisplayName & desktopFileName, which are only properties of QGuiApplication,
957  // we have to try to get them via the property system, as the static getter methods are
958  // part of QtGui only. Disadvantage: requires an app instance.
959  // Either get all or none of the properties & warn about it
960  if (app) {
961  aboutData->setOrganizationDomain(QCoreApplication::organizationDomain().toUtf8());
962  aboutData->setVersion(QCoreApplication::applicationVersion().toUtf8());
963  aboutData->setDisplayName(app->property("applicationDisplayName").toString());
964  aboutData->setDesktopFileName(app->property("desktopFileName").toString());
965  } else {
966  qCWarning(KABOUTDATA) << "Could not initialize the properties of KAboutData::applicationData by the equivalent properties from Q*Application: no app instance (yet) existing.";
967  }
968 
969  s_registry->m_appData = aboutData;
970  } else {
971  // check if in-sync with Q*Application metadata, as their setters could have been called
972  // after the last KAboutData::setApplicationData, with different values
973  warnIfOutOfSync("KAboutData::applicationData().componentName", aboutData->componentName(),
974  "QCoreApplication::applicationName", QCoreApplication::applicationName());
975  warnIfOutOfSync("KAboutData::applicationData().version", aboutData->version(),
976  "QCoreApplication::applicationVersion", QCoreApplication::applicationVersion());
977  warnIfOutOfSync("KAboutData::applicationData().organizationDomain", aboutData->organizationDomain(),
978  "QCoreApplication::organizationDomain", QCoreApplication::organizationDomain());
979  if (app) {
980  warnIfOutOfSync("KAboutData::applicationData().displayName", aboutData->displayName(),
981  "QGuiApplication::applicationDisplayName", app->property("applicationDisplayName").toString());
982  warnIfOutOfSync("KAboutData::applicationData().desktopFileName", aboutData->desktopFileName(),
983  "QGuiApplication::desktopFileName", app->property("desktopFileName").toString());
984  }
985  }
986

1061 void KAboutData::processCommandLine(QCommandLineParser *parser)
1062 {
1063  bool foundArgument = false;
1064  if (parser->isSet(QStringLiteral("author"))) {
1065  foundArgument = true;
1066  if (d->_authorList.isEmpty()) {
1067  printf("%s\n", qPrintable(QCoreApplication::translate("KAboutData CLI", "This application was written by somebody who wants to remain anonymous.")));
1068  } else {
1069  printf("%s\n", qPrintable(QCoreApplication::translate("KAboutData CLI", "%1 was written by:").arg(qAppName())));
1070  Q_FOREACH (const KAboutPerson &person, d->_authorList) {
1071  QString authorData = QStringLiteral(" ") + person.name();
1072  if (!person.emailAddress().isEmpty()) {
1073  authorData.append(QStringLiteral(" <") + person.emailAddress() + QStringLiteral(">"));
1074  }
1075  printf("%s\n", qPrintable(authorData));;
1076  }
1077  }
1078  if (!customAuthorTextEnabled()) {
1079  if (bugAddress() == QLatin1String("submit@bugs.kde.org") ) {
1080  printf("%s\n", qPrintable(QCoreApplication::translate("KAboutData CLI", "Please use http://bugs.kde.org to report bugs.")));
1081  } else if (!bugAddress().isEmpty()) {
1082  printf("%s\n", qPrintable(QCoreApplication::translate("KAboutData CLI", "Please report bugs to %1.").arg(bugAddress())));
1083  }
1084  } else {
1085  printf("%s\n", qPrintable(customAuthorPlainText()));
1086  }
1087  } else if (parser->isSet(QStringLiteral("license"))) {
1088  foundArgument = true;
1089  Q_FOREACH (const KAboutLicense &license, d->_licenseList) {
1090  printf("%s\n", qPrintable(license.text()));
1091  }
1092  }
1093 
1094  const QString desktopFileName = parser->value(QStringLiteral("desktopfile"));
1095  if (!desktopFileName.isEmpty()) {
1096  d->desktopFileName = desktopFileName;
1097  }
1098 
1099  if (foundArgument) {
1100  ::exit(EXIT_SUCCESS);
1101  }
