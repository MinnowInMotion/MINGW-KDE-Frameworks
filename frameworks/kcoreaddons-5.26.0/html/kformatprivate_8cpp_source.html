40 QString KFormatPrivate::formatByteSize(double size, int precision,
41  KFormat::BinaryUnitDialect dialect, KFormat::BinarySizeUnits units) const
42 {
43  // Current KDE default is IECBinaryDialect
44  if (dialect <= KFormat::DefaultBinaryDialect || dialect > KFormat::LastBinaryDialect) {
45  dialect = KFormat::IECBinaryDialect;
46  }
47 
48  // Current KDE default is to auto-adjust so the size falls in the range 0 to 1000/1024
49  if (units < KFormat::DefaultBinaryUnits || units > KFormat::UnitLastUnit) {
50  units = KFormat::DefaultBinaryUnits;
51  }
52 
53  int unit = 0; // Selects what unit to use
54  double multiplier = 1024.0;
55 
56  if (dialect == KFormat::MetricBinaryDialect) {
57  multiplier = 1000.0;
58  }
59 
60  // If a specific unit conversion is given, use it directly. Otherwise
61  // search until the result is in [0, multiplier] (or out of our range).
62  if (units == KFormat::DefaultBinaryUnits) {
63  while (qAbs(size) >= multiplier && unit < int(KFormat::UnitYottaByte)) {
64  size /= multiplier;
65  ++unit;
66  }
67  } else {
68  // A specific unit is in use
69  unit = static_cast<int>(units);
70  if (unit > 0) {
71  size /= pow(multiplier, unit);
72  }
73  }
74 
75  // Bytes, no rounding
76  if (unit == 0) {
77  precision = 0;
78  }
79 
80  QString numString = m_locale.toString(size, 'f', precision);
81 
82  // Do not remove "//:" comments below, they are used by the translators.
83  // NB: we cannot pass pluralization arguments, as the size may be negative
84  if (dialect == KFormat::MetricBinaryDialect) {
85  switch (unit) {
86  case KFormat::UnitByte:
87  //: MetricBinaryDialect size in bytes
88  return tr("%1 B", "MetricBinaryDialect").arg(numString);
89  case KFormat::UnitKiloByte:
90  //: MetricBinaryDialect size in 1000 bytes
91  return tr("%1 kB", "MetricBinaryDialect").arg(numString);
92  case KFormat::UnitMegaByte:
93  //: MetricBinaryDialect size in 10^6 bytes
94  return tr("%1 MB", "MetricBinaryDialect").arg(numString);
95  case KFormat::UnitGigaByte:
96  //: MetricBinaryDialect size in 10^9 bytes
97  return tr("%1 GB", "MetricBinaryDialect").arg(numString);
98  case KFormat::UnitTeraByte:
99  //: MetricBinaryDialect size in 10^12 bytes
100  return tr("%1 TB", "MetricBinaryDialect").arg(numString);
101  case KFormat::UnitPetaByte:
102  //: MetricBinaryDialect size in 10^15 bytes
103  return tr("%1 PB", "MetricBinaryDialect").arg(numString);
104  case KFormat::UnitExaByte:
105  //: MetricBinaryDialect size in 10^18 byte
106  return tr("%1 EB", "MetricBinaryDialect").arg(numString);
107  case KFormat::UnitZettaByte:
108  //: MetricBinaryDialect size in 10^21 bytes
109  return tr("%1 ZB", "MetricBinaryDialect").arg(numString);
110  case KFormat::UnitYottaByte:
111  //: MetricBinaryDialect size in 10^24 bytes
112  return tr("%1 YB", "MetricBinaryDialect").arg(numString);
113  }
114  } else if (dialect == KFormat::JEDECBinaryDialect) {
115  switch (unit) {
116  case KFormat::UnitByte:
117  //: JEDECBinaryDialect memory size in bytes
118  return tr("%1 B", "JEDECBinaryDialect").arg(numString);
119  case KFormat::UnitKiloByte:
120  //: JEDECBinaryDialect memory size in 1024 bytes
121  return tr("%1 KB", "JEDECBinaryDialect").arg(numString);
122  case KFormat::UnitMegaByte:
123  //: JEDECBinaryDialect memory size in 10^20 bytes
124  return tr("%1 MB", "JEDECBinaryDialect").arg(numString);
125  case KFormat::UnitGigaByte:
126  //: JEDECBinaryDialect memory size in 10^30 bytes
127  return tr("%1 GB", "JEDECBinaryDialect").arg(numString);
128  case KFormat::UnitTeraByte:
129  //: JEDECBinaryDialect memory size in 10^40 bytes
130  return tr("%1 TB", "JEDECBinaryDialect").arg(numString);
131  case KFormat::UnitPetaByte:
132  //: JEDECBinaryDialect memory size in 10^50 bytes
133  return tr("%1 PB", "JEDECBinaryDialect").arg(numString);
134  case KFormat::UnitExaByte:
135  //: JEDECBinaryDialect memory size in 10^60 bytes
136  return tr("%1 EB", "JEDECBinaryDialect").arg(numString);
137  case KFormat::UnitZettaByte:
138  //: JEDECBinaryDialect memory size in 10^70 bytes
139  return tr("%1 ZB", "JEDECBinaryDialect").arg(numString);
140  case KFormat::UnitYottaByte:
141  //: JEDECBinaryDialect memory size in 10^80 bytes
142  return tr("%1 YB", "JEDECBinaryDialect").arg(numString);
143  }
144  } else { // KFormat::IECBinaryDialect, KFormat::DefaultBinaryDialect
145  switch (unit) {

188 QString KFormatPrivate::formatDuration(quint64 msecs, KFormat::DurationFormatOptions options) const
189 {
190  quint64 ms = msecs;
191  int hours = ms / MSecsInHour;
192  ms = ms % MSecsInHour;
193  int minutes = ms / MSecsInMinute;
194  int roundMinutes = qRound(ms / 60000.0);
195  ms = ms % MSecsInMinute;
196  int seconds = ms / MSecsInSecond;
197  int roundSeconds = qRound(ms / 1000.0);
198  ms = ms % MSecsInSecond;
199 
200  if ((options & KFormat::InitialDuration) == KFormat::InitialDuration) {
201 
202  if ((options & KFormat::FoldHours) == KFormat::FoldHours
203  && (options & KFormat::ShowMilliseconds) == KFormat::ShowMilliseconds) {
204  //: @item:intext Duration format minutes, seconds and milliseconds
205  return tr("%1m%2.%3s").arg(hours * 60 + minutes, 1, 10, QLatin1Char('0'))
206  .arg(seconds, 2, 10, QLatin1Char('0'))
207  .arg(ms, 3, 10, QLatin1Char('0'));
208  } else if ((options & KFormat::FoldHours) == KFormat::FoldHours) {
209  //: @item:intext Duration format minutes and seconds
210  return tr("%1m%2s").arg(hours * 60 + minutes, 1, 10, QLatin1Char('0'))
211  .arg(roundSeconds, 2, 10, QLatin1Char('0'));
212  } else if ((options & KFormat::HideSeconds) == KFormat::HideSeconds) {
213  //: @item:intext Duration format hours and minutes
214  return tr("%1h%2m").arg(hours, 1, 10, QLatin1Char('0'))
215  .arg(roundMinutes, 2, 10, QLatin1Char('0'));
216  } else if ((options & KFormat::ShowMilliseconds) == KFormat::ShowMilliseconds) {
217  //: @item:intext Duration format hours, minutes, seconds, milliseconds
218  return tr("%1h%2m%3.%4s").arg(hours, 1, 10, QLatin1Char('0'))
219  .arg(minutes, 2, 10, QLatin1Char('0'))
220  .arg(seconds, 2, 10, QLatin1Char('0'))
221  .arg(ms, 3, 10, QLatin1Char('0'));
222  } else { // Default
223  //: @item:intext Duration format hours, minutes, seconds
224  return tr("%1h%2m%3s").arg(hours, 1, 10, QLatin1Char('0'))
225  .arg(minutes, 2, 10, QLatin1Char('0'))
226  .arg(roundSeconds, 2, 10, QLatin1Char('0'));
227  }
228 
229  } else {
230 
231  if ((options & KFormat::FoldHours) == KFormat::FoldHours
232  && (options & KFormat::ShowMilliseconds) == KFormat::ShowMilliseconds) {
233  //: @item:intext Duration format minutes, seconds and milliseconds
234  return tr("%1:%2.%3").arg(hours * 60 + minutes, 1, 10, QLatin1Char('0'))
235  .arg(seconds, 2, 10, QLatin1Char('0'))
236  .arg(ms, 3, 10, QLatin1Char('0'));
237  } else if ((options & KFormat::FoldHours) == KFormat::FoldHours) {
238  //: @item:intext Duration format minutes and seconds
239  return tr("%1:%2").arg(hours * 60 + minutes, 1, 10, QLatin1Char('0'))
240  .arg(roundSeconds, 2, 10, QLatin1Char('0'));
241  } else if ((options & KFormat::HideSeconds) == KFormat::HideSeconds) {
242  //: @item:intext Duration format hours and minutes
243  return tr("%1:%2").arg(hours, 1, 10, QLatin1Char('0'))
244  .arg(roundMinutes, 2, 10, QLatin1Char('0'));
245  } else if ((options & KFormat::ShowMilliseconds) == KFormat::ShowMilliseconds) {
246  //: @item:intext Duration format hours, minutes, seconds, milliseconds
247  return tr("%1:%2:%3.%4").arg(hours, 1, 10, QLatin1Char('0'))
248  .arg(minutes, 2, 10, QLatin1Char('0'))
249  .arg(seconds, 2, 10, QLatin1Char('0'))
250  .arg(ms, 3, 10, QLatin1Char('0'));
251  } else { // Default
252  //: @item:intext Duration format hours, minutes, seconds
253  return tr("%1:%2:%3").arg(hours, 1, 10, QLatin1Char('0'))
254  .arg(minutes, 2, 10, QLatin1Char('0'))
255  .arg(roundSeconds, 2, 10, QLatin1Char('0'));
256  }
257 
258  }
259 
260  Q_UNREACHABLE();
261  return QString();
262 }

264 QString KFormatPrivate::formatDecimalDuration(quint64 msecs, int decimalPlaces) const
265 {
266  if (msecs >= MSecsInDay) {
267  //: @item:intext %1 is a real number, e.g. 1.23 days
268  return tr("%1 days").arg(m_locale.toString(msecs / (MSecsInDay * 1.0), 'f', decimalPlaces));
269  } else if (msecs >= MSecsInHour) {
270  //: @item:intext %1 is a real number, e.g. 1.23 hours
271  return tr("%1 hours").arg(m_locale.toString(msecs / (MSecsInHour * 1.0), 'f', decimalPlaces));
272  } else if (msecs >= MSecsInMinute) {
273  //: @item:intext %1 is a real number, e.g. 1.23 minutes
274  return tr("%1 minutes").arg(m_locale.toString(msecs / (MSecsInMinute * 1.0), 'f', decimalPlaces));
275  } else if (msecs >= MSecsInSecond) {
276  //: @item:intext %1 is a real number, e.g. 1.23 seconds
277  return tr("%1 seconds").arg(m_locale.toString(msecs / (MSecsInSecond * 1.0), 'f', decimalPlaces));
278  }
279  //: @item:intext %1 is a whole number
280  //~ singular %n millisecond
281  //~ plural %n milliseconds
282  return tr("%n millisecond(s)", 0, msecs);
283 }

292 static QString formatSingleDuration(DurationUnits units, int n)
293 {
294  // NB: n is guaranteed to be non-negative
295  switch (units) {
296  case Days:
297  //: @item:intext %n is a whole number
298  //~ singular %n day
299  //~ plural %n days
300  return KFormatPrivate::tr("%n day(s)", 0, n);
301  case Hours:
302  //: @item:intext %n is a whole number
303  //~ singular %n hour
304  //~ plural %n hours
305  return KFormatPrivate::tr("%n hour(s)", 0, n);
306  case Minutes:
307  //: @item:intext %n is a whole number
308  //~ singular %n minute
309  //~ plural %n minutes
310  return KFormatPrivate::tr("%n minute(s)", 0, n);
311  case Seconds:
312  //: @item:intext %n is a whole number
313  //~ singular %n second
314  //~ plural %n seconds
315  return KFormatPrivate::tr("%n second(s)", 0, n);
316  }
317  Q_ASSERT(false);
318  return QString();
319 }

321 QString KFormatPrivate::formatSpelloutDuration(quint64 msecs) const
322 {
323  quint64 ms = msecs;
324  int days = ms / MSecsInDay;
325  ms = ms % (MSecsInDay);
326  int hours = ms / MSecsInHour;
327  ms = ms % MSecsInHour;
328  int minutes = ms / MSecsInMinute;
329  ms = ms % MSecsInMinute;
330  int seconds = qRound(ms / 1000.0);
331 
332  // Handle correctly problematic case #1 (look at KFormatTest::prettyFormatDuration())
333  if (seconds == 60) {
334  return formatSpelloutDuration(msecs - ms + MSecsInMinute);
335  }
336 
337  if (days && hours) {
338  /*: @item:intext days and hours. This uses the previous item:intext messages.
339  If this does not fit the grammar of your language please contact the i18n team to solve the problem */
340  return tr("%1 and %2").arg(formatSingleDuration(Days, days))
341  .arg(formatSingleDuration(Hours, hours));
342  } else if (days) {
343  return formatSingleDuration(Days, days);
344  } else if (hours && minutes) {
345  /*: @item:intext hours and minutes. This uses the previous item:intext messages.
346  If this does not fit the grammar of your language please contact the i18n team to solve the problem */
347  return tr("%1 and %2").arg(formatSingleDuration(Hours, hours))
348  .arg(formatSingleDuration(Minutes, minutes));
349  } else if (hours) {
350  return formatSingleDuration(Hours, hours);
351  } else if (minutes && seconds) {
352  /*: @item:intext minutes and seconds. This uses the previous item:intext messages.
353  If this does not fit the grammar of your language please contact the i18n team to solve the problem */
354  return tr("%1 and %2").arg(formatSingleDuration(Minutes, minutes))
355  .arg(formatSingleDuration(Seconds, seconds));
356  } else if (minutes) {
357  return formatSingleDuration(Minutes, minutes);
358  } else {
359  return formatSingleDuration(Seconds, seconds);
360  }
361 }

363 QString KFormatPrivate::formatRelativeDate(const QDate &date, QLocale::FormatType format) const
364 {
365  const int daysTo = QDate::currentDate().daysTo(date);
366  if (daysTo > 7 || daysTo < -7) {
367  return m_locale.toString(date, format);
368  }
369 
370  switch (daysTo) {
371  case 1:
372  return tr("Tomorrow");
373  case 0:
374  return tr("Today");
375  case -1:
376  return tr("Yesterday");
377  }
378 
379  if (daysTo < -1)
380  /*: a day of the week, eg "Monday" (but translated). Refers to the most recent such day that
381  has already happened. */
382  {
383  return tr("Last %1").arg(m_locale.dayName(date.dayOfWeek(), format));
384  } else if (daysTo > 1)
385  /*: a day of the week, eg "Monday" (but translated). Refers to the soonest such day that
386  has not already happened. */
387  {
388  return tr("Next %1").arg(m_locale.dayName(date.dayOfWeek(), format));
389  }
390  Q_UNREACHABLE();
391 }

393 QString KFormatPrivate::formatRelativeDateTime(const QDateTime &dateTime, QLocale::FormatType format) const
394 {

396  if (daysTo > 7 || daysTo < -7) {

398  }
399 
400  /*: relative datetime with %1 result of formatReleativeDate() and %2 the formatted time
401  If this does not fit the grammar of your language please contact the i18n team to solve the problem */
