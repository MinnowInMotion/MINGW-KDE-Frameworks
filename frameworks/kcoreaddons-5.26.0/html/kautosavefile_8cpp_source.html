51 static QStringList findAllStales(const QString &appName)
52 {
53  const QStringList dirs = QStandardPaths::standardLocations(QStandardPaths::GenericDataLocation);
54  QStringList files;
55 
56  Q_FOREACH (const QString &dir, dirs) {
57  QDir appDir(dir + QStringLiteral("/stalefiles/") + appName);
58  //qCDebug(KCOREADDONS_DEBUG) << "Looking in" << appDir.absolutePath();
59  Q_FOREACH (const QString &file, appDir.entryList(QDir::Files)) {
60  files << (appDir.absolutePath() + QLatin1Char('/') + file);
61  }
62  }
63  return files;
64 }

66 QString KAutoSaveFilePrivate::tempFileName()
67 {
68  // Note: we drop any query string and user/pass info
69  const QString protocol(managedFile.scheme());
70  const QString path(managedFile.adjusted(QUrl::RemoveFilename | QUrl::StripTrailingSlash).path());
71  QString name(managedFile.fileName());
72 
73  // Remove any part of the path to the right if it is longer than the max file size and
74  // ensure that the max filesize takes into account the other parts of the tempFileName
75  // Subtract 1 for the _ char, 3 for the padding separator, 5 is for the .lock
76  int pathLengthLimit = FILENAME_MAX - NamePadding - name.size() - protocol.size() - 9;
77 
78  QString junk = KRandom::randomString(NamePadding);
79  // tempName = fileName + junk.truncated + protocol + _ + path.truncated + junk
80  // This is done so that the separation between the filename and path can be determined
81  name += junk.rightRef(3) + protocol + QLatin1Char('_') + path.leftRef(pathLengthLimit) + junk;
82 
83  return QString::fromLatin1(QUrl::toPercentEncoding(name).constData());
84 }

131 bool KAutoSaveFile::open(OpenMode openmode)
132 {
133  if (d->managedFile.isEmpty()) {
134  return false;
135  }
136 
137  QString tempFile;
138  if (d->managedFileNameChanged) {
139  QString staleFilesDir = QStandardPaths::writableLocation(QStandardPaths::GenericDataLocation) +
140  QStringLiteral("/stalefiles/") + QCoreApplication::instance()->applicationName();
141  if (!QDir().mkpath(staleFilesDir)) {
142  return false;
143  }
144  tempFile = staleFilesDir + QChar::fromLatin1('/') + d->tempFileName();
145  } else {
146  tempFile = fileName();
147  }
148 
149  d->managedFileNameChanged = false;
150 
151  setFileName(tempFile);
152 
153  if (QFile::open(openmode)) {
154 
155  if (!d->lock)
156  {
157  d->lock = new QLockFile(tempFile + QStringLiteral(".lock"));
158  d->lock->setStaleLockTime(60 * 1000); // HARDCODE, 1 minute
159  }
160 
161  if (d->lock->isLocked() || d->lock->tryLock()) {
162  return true;
163  } else {
164  qCWarning(KCOREADDONS_DEBUG)<<"Could not lock file:"<<tempFile;
165  close();
166  }
167  }
168 
169  return false;
170 }

172 static QUrl extractManagedFilePath(const QString& staleFileName)
173 {
174  const QStringRef sep = staleFileName.rightRef(3);
175  int sepPos = staleFileName.indexOf(sep);
176  int pathPos = staleFileName.indexOf(QChar::fromLatin1('_'), sepPos);
177  QUrl managedFileName;
178  //name.setScheme(file.mid(sepPos + 3, pathPos - sep.size() - 3));
179  QByteArray encodedPath = staleFileName.midRef(pathPos+1, staleFileName.length()-pathPos-1-KAutoSaveFilePrivate::NamePadding).toLatin1();
180  managedFileName.setPath(QUrl::fromPercentEncoding(encodedPath) + QLatin1Char('/') + QFileInfo(staleFileName.left(sepPos)).fileName());
181  return managedFileName;
182 }

184 QList<KAutoSaveFile *> KAutoSaveFile::staleFiles(const QUrl &filename, const QString &applicationName)
185 {
186  QString appName(applicationName);
187  if (appName.isEmpty()) {
188  appName = QCoreApplication::instance()->applicationName();
189  }
190 
191  // get stale files
192  const QStringList files = findAllStales(appName);
193 
194  QList<KAutoSaveFile *> list;
195 
196  // contruct a KAutoSaveFile for stale files corresponding given filename
197  Q_FOREACH (const QString &file, files) {
198  if (file.endsWith(QLatin1String(".lock")) || (!filename.isEmpty() && extractManagedFilePath(file).path()!=filename.path())) {
199  continue;
200  }
201 
202  // sets managedFile
203  KAutoSaveFile *asFile = new KAutoSaveFile(filename.isEmpty()?extractManagedFilePath(file):filename);
204  asFile->setFileName(file);
205  asFile->d->managedFileNameChanged = false; // do not regenerate tempfile name
206  list.append(asFile);
207  }
208 
209  return list;
210 }
