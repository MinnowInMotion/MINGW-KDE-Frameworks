26 inline KFileSystemType::Type kde_typeFromName(const char *name)
27 {
28  if (qstrncmp(name, "nfs", 3) == 0
29  || qstrncmp(name, "autofs", 6) == 0
30  || qstrncmp(name, "cachefs", 7) == 0
31  || qstrncmp(name, "fuse.sshfs", 10) == 0
32  || qstrncmp(name, "xtreemfs@", 9) == 0) { // #178678
33  return KFileSystemType::Nfs;
34  }
35  if (qstrncmp(name, "fat", 3) == 0
36  || qstrncmp(name, "vfat", 4) == 0
37  || qstrncmp(name, "msdos", 5) == 0) {
38  return KFileSystemType::Fat;
39  }
40  if (qstrncmp(name, "cifs", 4) == 0
41  || qstrncmp(name, "smbfs", 5) == 0) {
42  return KFileSystemType::Smb;
43  }
44  if (qstrncmp(name, "ramfs", 5) == 0) {
45  return KFileSystemType::Ramfs;
46  }
47 
48  return KFileSystemType::Other;
49 }

95 static KFileSystemType::Type determineFileSystemTypeImpl(const QByteArray &path)
96 {
97  struct statfs buf;
98  if (statfs(path.constData(), &buf) != 0) {
99  //qCDebug(KCOREADDONS_DEBUG) << path << errno << strerror(errno);
100  return KFileSystemType::Unknown;
101  }
102  switch (static_cast<unsigned long>(buf.f_type)) {
103  case NFS_SUPER_MAGIC:
104  case AUTOFS_SUPER_MAGIC:
105  case AUTOFSNG_SUPER_MAGIC:
106  case FUSE_SUPER_MAGIC: // TODO could be anything. Need to use statfs() to find out more.
107  return KFileSystemType::Nfs;
108  case SMB_SUPER_MAGIC:
109  return KFileSystemType::Smb;
110  case MSDOS_SUPER_MAGIC:
111  return KFileSystemType::Fat;
112  case RAMFS_MAGIC:
113  return KFileSystemType::Ramfs;
114  default:
115  return KFileSystemType::Other;
116  }
117 }
