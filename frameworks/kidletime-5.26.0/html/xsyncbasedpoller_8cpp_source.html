187 void XSyncBasedPoller::addTimeout(int nextTimeout)
188 {
189  /* We need to set the counter to the idle time + the value
190  * requested for next timeout
191  */
192 
193  // If there's already an alarm for the requested timeout, skip
194  if (m_timeoutAlarm.contains(nextTimeout)) {
195  return;
196  }
197 
198  XSyncValue timeout;
199  XSyncAlarm newalarm = None;
200 
201  XSyncIntToValue(&timeout, nextTimeout);
202 
203  setAlarm(m_display, &newalarm, m_idleCounter,
204  XSyncPositiveComparison, timeout);
205 
206  m_timeoutAlarm.insert(nextTimeout, newalarm);
207 }

244 void XSyncBasedPoller::catchIdleEvent()
245 {
246  XSyncValue idleTime;
247 
248  XSyncQueryCounter(m_display, m_idleCounter, &idleTime);
249 
250  /* Set the reset alarm to fire the next time idleCounter < the
251  * current counter value. XSyncNegativeComparison means <= so
252  * we have to subtract 1 from the counter value
253  */
254 
255  //NOTE: this must be a int, else compilation might fail
256  int overflow;
257  XSyncValue add;
258  XSyncValue plusone;
259  XSyncIntToValue(&add, -1);
260  XSyncValueAdd(&plusone, idleTime, add, &overflow);
261  setAlarm(m_display, &m_resetAlarm, m_idleCounter,
262  XSyncNegativeComparison, plusone);
263 }

265 void XSyncBasedPoller::reloadAlarms()
266 {
267  XSyncValue timeout;
268 
269  for (QHash<int, XSyncAlarm>::iterator i = m_timeoutAlarm.begin(); i != m_timeoutAlarm.end(); ++i) {
270  XSyncIntToValue(&timeout, i.key());
271 
272  setAlarm(m_display, &(i.value()), m_idleCounter,
273  XSyncPositiveComparison, timeout);

277 bool XSyncBasedPoller::xcbEvent(xcb_generic_event_t *event)
278 {
279  //qDebug() << event->response_type << "waiting for" << m_sync_event+XCB_SYNC_ALARM_NOTIFY;
280  if (event->response_type != m_sync_event + XCB_SYNC_ALARM_NOTIFY) {
281  return false;
282  }
283 
284  xcb_sync_alarm_notify_event_t *alarmEvent = reinterpret_cast<xcb_sync_alarm_notify_event_t *>(event);
285 
286  if (alarmEvent->state == XCB_SYNC_ALARMSTATE_DESTROYED) {
287  return false;
288  }
289 
290  for (QHash<int, XSyncAlarm>::const_iterator i = m_timeoutAlarm.constBegin(); i != m_timeoutAlarm.constEnd(); ++i) {
291  if (alarmEvent->alarm == i.value()) {
292  /* Bling! Caught! */
293  emit timeoutReached(i.key());
294  // Update the alarm to fire back if the system gets inactive for the same time
295  catchIdleEvent();
296  return false;
297  }
298  }
299 
300  if (alarmEvent->alarm == m_resetAlarm) {
301  /* Resuming from idle here! */
302  stopCatchingIdleEvents();
303  reloadAlarms();
304  emit resumingFromIdle();
305  }
306 
307  return false;
308 }

310 void XSyncBasedPoller::setAlarm(Display *dpy, XSyncAlarm *alarm, XSyncCounter counter,
311  XSyncTestType test, XSyncValue value)
312 {
313  XSyncAlarmAttributes attr;
314  XSyncValue delta;
315  unsigned int flags;
316 
317  XSyncIntToValue(&delta, 0);
318 
319  attr.trigger.counter = counter;
320  attr.trigger.value_type = XSyncAbsolute;
321  attr.trigger.test_type = test;
322  attr.trigger.wait_value = value;
323  attr.delta = delta;
324 
325  flags = XSyncCACounter | XSyncCAValueType | XSyncCATestType |
326  XSyncCAValue | XSyncCADelta;
327 
328  if (*alarm) {
329  //xcb_sync_change_alarm_checked(m_xcb_connection, alarmId, ...
330  XSyncChangeAlarm(dpy, *alarm, flags, &attr);
331  } else {
332  *alarm = XSyncCreateAlarm(dpy, flags, &attr);
333  qDebug() << "Created alarm" << *alarm;
334  }
335 
336  XFlush(m_display);
337 }
