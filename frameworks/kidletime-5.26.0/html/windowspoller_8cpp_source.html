34 int WindowsPoller::getIdleTime()
35 {
36 #ifndef _WIN32_WCE
37  int idle = 0;
38 
39  LASTINPUTINFO lii;
40  memset(&lii, 0, sizeof(lii));
41 
42  lii.cbSize = sizeof(lii);
43 
44  BOOL ok = GetLastInputInfo(&lii);
45  if (ok) {
46  idle = GetTickCount() - lii.dwTime;
47  }
48 
49  return idle;
50 #else
51  return GetIdleTime();
52 #endif
53 }

67  int x = (int)100 * 65536 / width;
68  int y = (int)100 * 65536 / height;

70  mouse_event(MOUSEEVENTF_ABSOLUTE | MOUSEEVENTF_MOVE, x, y, NULL, NULL);
