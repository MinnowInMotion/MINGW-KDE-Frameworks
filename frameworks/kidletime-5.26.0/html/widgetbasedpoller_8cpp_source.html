39 bool WidgetBasedPoller::setUpPoller()

41  m_pollTimer = new QTimer(this);

43  //setup idle timer, with some smart polling

73 bool WidgetBasedPoller::eventFilter(QObject *object, QEvent *event)
74 {
75  if (object == m_grabber

77  detectedActivity();

112 int WidgetBasedPoller::poll()
113 {
114  int idle = getIdleTime();
115 
116  // Check if we reached a timeout..
117  Q_FOREACH(int timeOut, m_timeouts) {
118  if ( ( timeOut - idle < 300 && timeOut >= idle ) || ( idle - timeOut < 300 && idle > timeOut ) ) {
119  // Bingo!
120  emit timeoutReached( timeOut );
121  }
122  }
123 
124  // Let's check the timer now!
125  int mintime = 0;
126 
127  Q_FOREACH (int i, m_timeouts) {
128  if (i > idle && (i < mintime || mintime == 0)) {
129  mintime = i;
130  }
131  }
132 
133  //qDebug() << "mintime " << mintime << "idle " << idle;
134 
135  if (mintime != 0) {
136  m_pollTimer->start(mintime - idle);
137  } else {
138  m_pollTimer->stop();
139  }
140 
141  return idle;
142 }
