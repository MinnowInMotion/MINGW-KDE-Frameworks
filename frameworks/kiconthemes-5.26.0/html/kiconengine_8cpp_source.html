40 static inline int qIconModeToKIconState(QIcon::Mode mode)
41 {
42  int kstate;
43  switch (mode) {
44  default:
45  case QIcon::Normal:
46  kstate = KIconLoader::DefaultState;
47  break;
48  case QIcon::Active:
49  kstate = KIconLoader::ActiveState;
50  break;
51  case QIcon::Disabled:
52  kstate = KIconLoader::DisabledState;
53  break;
54  case QIcon::Selected:
55  kstate = KIconLoader::SelectedState;
56  break;
57  }
58  return kstate;
59 }

69 void KIconEngine::paint(QPainter *painter, const QRect &rect, QIcon::Mode mode, QIcon::State state)
70 {
71  if (!mIconLoader) {
72  return;
73  }
74 
75  Q_UNUSED(state)
76 
77  qreal dpr = painter->device()->devicePixelRatio();
78 
79  const int kstate = qIconModeToKIconState(mode);
80  const int iconSize = qMin(rect.width(), rect.height()) * dpr;
81  const QPixmap pix = mIconLoader.data()->loadIcon(mIconName, KIconLoader::Desktop, iconSize, kstate, mOverlays);
82  painter->drawPixmap(rect, pix);
83 }

85 QPixmap KIconEngine::pixmap(const QSize &size, QIcon::Mode mode, QIcon::State state)
86 {
87  Q_UNUSED(state)
88 
89  if (!size.isValid()) {
90  return QPixmap();
91  }
92 
93  if (!mIconLoader) {
94  QPixmap pm(size);
95  pm.fill(Qt::transparent);
96  return pm;
97  }
98 
99 
100  const int kstate = qIconModeToKIconState(mode);
101  const int iconSize = qMin(size.width(), size.height());
102  QPixmap pix = mIconLoader.data()->loadIcon(mIconName, KIconLoader::Desktop, iconSize, kstate, mOverlays);
103 
104  if (pix.size() == size) {
105  return pix;
106  }
107 
108  QPixmap pix2(size);
109  pix2.fill(QColor(0, 0, 0, 0));
110 
111  QPainter painter(&pix2);
112  painter.drawPixmap(QPoint(), pix);
113 
114  return pix2;
115 }
