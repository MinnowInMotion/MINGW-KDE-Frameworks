24 /* This tool converts an svg to a Mac OS X icns file.
25  * Note: Requires the 'iconutil' Mac OS X binary
26  */

42  do { \
43  if (!(isOk)) { \
44  fprintf(stderr, __VA_ARGS__); \
45  return 1; \
46  } \

49 static bool writeImage(QSvgRenderer &svg, int size, const QString &outFile1, const QString &outFile2 = QString())
50 {
51  QImage out(size, size, QImage::Format_ARGB32);
52  out.fill(Qt::transparent);
53 
54  QPainter painter(&out);
55  svg.render(&painter);
56  painter.end();
57 
58  if (!out.save(outFile1)) {
59  fprintf(stderr, "Unable to write %s\n", qPrintable(outFile1));
60  return false;
61  }
62  if (!outFile2.isEmpty() && !out.save(outFile2)) {
63  fprintf(stderr, "Unable to write %s\n", qPrintable(outFile2));
64  return false;
65  }
66 
67  return true;
68 }

70 int main(int argc, char *argv[])
71 {
72  QGuiApplication app(argc, argv);
73 
74  app.setApplicationName("ksvg2icns");
75  app.setApplicationVersion(KICONTHEMES_VERSION_STRING);
76  QCommandLineParser parser;
77  parser.setApplicationDescription(app.translate("main", "Creates an icns file from an svg image"));
78  parser.addPositionalArgument("iconname", app.translate("main", "The svg icon to convert"));
79  parser.addHelpOption();
80 
81  parser.process(app);
82  if (parser.positionalArguments().isEmpty()) {
83  parser.showHelp();
84  return 1;
85  }
86 
87  bool isOk;
88 
89  // create a temporary dir to create an iconset
90  QTemporaryDir tmpDir("ksvg2icns");
91  tmpDir.setAutoRemove(true);
92 
93  isOk = tmpDir.isValid();
94  EXIT_ON_ERROR(isOk, "Unable to create temporary directory\n");
95 
96  isOk = QDir(tmpDir.path()).mkdir("out.iconset");
97  EXIT_ON_ERROR(isOk, "Unable to create out.iconset directory\n");
98 
99  const QString outPath = tmpDir.path() + "/out.iconset";
100 
101  const QStringList &args = app.arguments();
102  EXIT_ON_ERROR(args.size() == 2,
103  "Usage: %s svg-image\n",
104  qPrintable(args.value(0)));
105 
106  const QString &svgFileName = args.at(1);
107 
108  // open the actual svg file
109  QSvgRenderer svg;
110  isOk = svg.load(svgFileName);
111  EXIT_ON_ERROR(isOk, "Unable to load %s\n", qPrintable(svgFileName));
112 
113  // The sizes are from:
114  // https://developer.apple.com/library/mac/documentation/GraphicsAnimation/Conceptual/HighResolutionOSX/Optimizing/Optimizing.html
115 
116  struct OutFiles
117  {
118  int size;
119  QString out1;
120  QString out2;
121  };
122 
123  // create the pngs in various resolutions
124  const OutFiles outFiles[] = {
125  { 1024, outPath + "/icon_512x512@2x.png", QString() },
126  { 512, outPath + "/icon_512x512.png", outPath + "/icon_256x256@2x.png" },
127  { 256, outPath + "/icon_256x256.png", outPath + "/icon_128x128@2x.png" },
128  { 128, outPath + "/icon_128x128.png", QString() },
129  { 64, outPath + "/icon_32x32@2x.png", QString() },
130  { 32, outPath + "/icon_32x32.png", outPath + "/icon_16x16@2x.png" },
131  { 16, outPath + "/icon_16x16.png", QString() }
132  };
133 
134  for (size_t i = 0; i < sizeof(outFiles) / sizeof(OutFiles); ++i) {
135  isOk = writeImage(svg, outFiles[i].size, outFiles[i].out1, outFiles[i].out2);
136  if (!isOk) {
137  return 1;
138  }
139  }
140 
141  // convert the iconset to icns using the "iconutil" command
142 
143  const QString outIcns = QFileInfo(svgFileName).baseName() + ".icns";
144 
145  const QStringList utilArgs = QStringList()
146  << "-c" << "icns"
147  << "-o" << outIcns
148  << outPath;
149 
150  QProcess iconUtil;
151  iconUtil.setProgram("iconUtil");
152  iconUtil.setArguments(utilArgs);
153 
154  iconUtil.start("iconutil", utilArgs, QIODevice::ReadOnly);
155  isOk = iconUtil.waitForFinished(-1);
156  EXIT_ON_ERROR(isOk, "Unable to launch iconutil: %s\n", qPrintable(iconUtil.errorString()));
157 
158  EXIT_ON_ERROR(iconUtil.exitStatus() == QProcess::NormalExit, "iconutil crashed!\n");
159  EXIT_ON_ERROR(iconUtil.exitCode() == 0, "iconutil returned %d\n", iconUtil.exitCode());
160 
161  return 0;
162 }
