45 // Support for icon themes in RCC files.
46 // The intended use case is standalone apps on Windows / MacOS / etc.
47 // For this reason we use AppDataLocation: BINDIR/data on Windows, Resources on OS X
48 void initRCCIconTheme()
49 {
50  const QString iconThemeRcc = QStandardPaths::locate(QStandardPaths::AppDataLocation, QStringLiteral("icontheme.rcc"));
51  if (!iconThemeRcc.isEmpty()) {
52  const QString iconThemeName = QStringLiteral("kf5_rcc_theme");
53  const QString iconSubdir = QStringLiteral("/icons/") + iconThemeName;
54  if (QResource::registerResource(iconThemeRcc, iconSubdir)) {
55  if (QFileInfo::exists(QLatin1Char(':') + iconSubdir + QStringLiteral("/index.theme"))) {
56  // Tell Qt about the theme
57  // Note that since qtbase commit a8621a3f8, this means the QPA (i.e. KIconLoader) will NOT be used.
58  QIcon::setThemeName(iconThemeName); // Qt looks under :/icons automatically
59  // Tell KIconTheme about the theme, in case KIconLoader is used directly
60  *_themeOverride() = iconThemeName;
61  } else {
62  qWarning() << "No index.theme found in" << iconThemeRcc;
63  QResource::unregisterResource(iconThemeRcc, iconSubdir);
64  }
65  } else {
66  qWarning() << "Invalid rcc file" << iconThemeRcc;
67  }
68  }
69 }

152 {
153  d->mInternalName = name;
154 
155  QStringList themeDirs;
156  QSet<QString> addedDirs; // Used for avoiding duplicates.
157 
158  // Applications can have local additions to the global "locolor" and
159  // "hicolor" icon themes. For these, the _global_ theme description
160  // files are used..
161 
162  if (!appName.isEmpty() &&
163  (name == defaultThemeName() || name == QLatin1String("hicolor") || name == QLatin1String("locolor"))) {
164  const QStringList icnlibs = QStandardPaths::standardLocations(QStandardPaths::GenericDataLocation);
165  for (QStringList::ConstIterator it = icnlibs.constBegin(); it != icnlibs.constEnd(); ++it) {
166  const QString cDir = *it + '/' + appName + "/icons/" + name + '/';
167  if (QFileInfo::exists(cDir)) {
168  themeDirs += cDir;
169  }
170  }
171 
172  if (!basePathHint.isEmpty()) {
173  // Checks for dir existing are done below
174  themeDirs += basePathHint + '/' + name + '/';
175  }
176  }
177 
178  // Find the theme description file. These are either locally in the :/icons resource path or global.
179  QStringList icnlibs;
180 
181  // local embedded icons have preference
182  icnlibs << QStringLiteral(":/icons");
183 
184  // global icons
185  icnlibs += QStandardPaths::locateAll(QStandardPaths::GenericDataLocation, QStringLiteral("icons"), QStandardPaths::LocateDirectory);
186 
187  // These are not in the icon spec, but e.g. GNOME puts some icons there anyway.
188  icnlibs += QStandardPaths::locateAll(QStandardPaths::GenericDataLocation, QStringLiteral("pixmaps"), QStandardPaths::LocateDirectory);
189

191  for (QStringList::ConstIterator it = icnlibs.constBegin(); it != icnlibs.constEnd(); ++it) {
192  const QString cDir = *it + '/' + name + '/';
193  if (QDir(cDir).exists()) {
194  themeDirs += cDir;
195  if (d->mDir.isEmpty()) {
196  if (QFileInfo::exists(cDir + "index.theme")) {
197  d->mDir = cDir;
198  fileName = d->mDir + "index.theme";
199  mainSection = QStringLiteral("Icon Theme");
200  } else if (QFileInfo::exists(cDir + "index.desktop")) {
201  d->mDir = cDir;
202  fileName = d->mDir + "index.desktop";

214  // Use KSharedConfig to avoid parsing the file many times, from each component.

226  *it = defaultThemeName();

228  }
229  }
230 
231  d->hidden = cfg.readEntry("Hidden", false);
232  d->followsColorScheme = cfg.readEntry("FollowsColorScheme", false);
233  d->example = cfg.readPathEntry("Example", QString());
234  d->screenshot = cfg.readPathEntry("ScreenShot", QString());
235  d->mExtensions = cfg.readEntry("KDE-Extensions", QStringList{ ".png", ".svgz", ".svg", ".xpm" });
236 
237  const QStringList dirs = cfg.readPathEntry("Directories", QStringList());
238  for (QStringList::ConstIterator it = dirs.begin(); it != dirs.end(); ++it) {
239  KConfigGroup cg(d->sharedConfig, *it);
240  for (QStringList::ConstIterator itDir = themeDirs.constBegin(); itDir != themeDirs.constEnd(); ++itDir) {
241  const QString currentDir(*itDir + *it + '/');
242  if (!addedDirs.contains(currentDir) && QDir(currentDir).exists()) {
243  addedDirs.insert(currentDir);
244  KIconThemeDir *dir = new KIconThemeDir(*itDir, *it, cg);
245  if (!dir->isValid()) {
246  delete dir;
247  } else {
248  d->mDirs.append(dir);
249  }
250  }
251  }
252  }

258  groups += QStringLiteral("Small");
259  groups += QStringLiteral("Panel");

393 QStringList KIconTheme::queryIconsByContext(int size, KIconLoader::Context context) const
394 {
395  int dw;
396 
397  // We want all the icons for a given context, but we prefer icons
398  // of size size . Note that this may (will) include duplicate icons
399  //QStringList iconlist[34]; // 33 == 48-16+1
400  QStringList iconlist[128]; // 33 == 48-16+1
401  // Usually, only the 0, 6 (22-16), 10 (32-22), 16 (48-32 or 32-16),
402  // 26 (48-22) and 32 (48-16) will be used, but who knows if someone
403  // will make icon themes with different icon sizes.
404 
405  foreach (KIconThemeDir *dir, d->mDirs) {
406  if ((context != KIconLoader::Any) && (context != dir->context())) {
407  continue;
408  }
409  dw = abs(dir->size() - size);
410  iconlist[(dw < 127) ? dw : 127] += dir->iconList();
411  }
412 
413  QStringList iconlistResult;
414  for (int i = 0; i < 128; i++) {
415  iconlistResult += iconlist[i];
416  }
417 
418  return iconlistResult;

446 QString KIconTheme::iconPath(const QString &name, int size, KIconLoader::MatchType match) const
447 {
448  QString path;
449  QString tempPath; // used to cache icon path if it exists
450  int delta = -INT_MAX; // current icon size delta of 'icon'
451  int dw = INT_MAX; // icon size delta of current directory
452 
453  // Search the directory that contains the icon which matches best to the requested
454  // size. If there is no directory which matches exactly to the requested size, the
455  // following criterias get applied:
456  // - Take a directory having icons with a minimum difference to the requested size.
457  // - Prefer directories that allow a downscaling even if the difference to
458  // the requested size is bigger than a directory where an upscaling is required.
459  foreach (KIconThemeDir *dir, d->mDirs) {
460  if (match == KIconLoader::MatchExact) {
461  if ((dir->type() == KIconLoader::Fixed) && (dir->size() != size)) {
462  continue;
463  }
464  if ((dir->type() == KIconLoader::Scalable) &&
465  ((size < dir->minSize()) || (size > dir->maxSize()))) {
466  continue;
467  }
468  if ((dir->type() == KIconLoader::Threshold) &&
469  (abs(dir->size() - size) > dir->threshold())) {
470  continue;
471  }
472  } else {
473  // dw < 0 means need to scale up to get an icon of the requested size.
474  // Upscaling should only be done if no larger icon is available.
475  if (dir->type() == KIconLoader::Fixed) {
476  dw = dir->size() - size;
477  } else if (dir->type() == KIconLoader::Scalable) {
478  if (size < dir->minSize()) {
479  dw = dir->minSize() - size;
480  } else if (size > dir->maxSize()) {
481  dw = dir->maxSize() - size;
482  } else {
483  dw = 0;
484  }
485  } else if (dir->type() == KIconLoader::Threshold) {
486  if (size < dir->size() - dir->threshold()) {
487  dw = dir->size() - dir->threshold() - size;
488  } else if (size > dir->size() + dir->threshold()) {
489  dw = dir->size() + dir->threshold() - size;
490  } else {
491  dw = 0;
492  }
493  }
494  // Usually if the delta (= 'dw') of the current directory is
495  // not smaller than the delta (= 'delta') of the currently best
496  // matching icon, this candidate can be skipped. But skipping
497  // the candidate may only be done, if this does not imply
498  // in an upscaling of the icon (it is OK to use a directory with
499  // smaller icons that what we've already found, however).
500  if ((abs(dw) >= abs(delta)) && ((dw < 0) || (delta > 0))) {
501  continue;
502  }
503  }
504 
505  // cache the result of iconPath() call which checks if file exists
506  tempPath = dir->iconPath(name);
507 
508  if (tempPath.isEmpty()) {
509  continue;
510  }
511  path = tempPath;
512 
513  // if we got in MatchExact that far, we find no better
514  if (match == KIconLoader::MatchExact) {
515  return path;
516  }
517  delta = dw;
518  if (delta == 0) {
519  return path; // We won't find a better match anyway
520  }
521  }
522  return path;
523 }

701  } else {

704  }
705  tmp = config.readEntry(QStringLiteral("Type"), QStringLiteral("Threshold"));

710  } else if (tmp == QLatin1String("Threshold")) {

715  }

717  mMinSize = config.readEntry(QStringLiteral("MinSize"), mSize);
718  mMaxSize = config.readEntry(QStringLiteral("MaxSize"), mSize);

721  }
