438 void KIconLoaderGlobalData::parseGenericIconsFiles(const QString &fileName)
439 {
440  QFile file(fileName);
441  if (file.open(QIODevice::ReadOnly)) {
442  QTextStream stream(&file);
443  stream.setCodec("ISO 8859-1");
444  while (!stream.atEnd()) {
445  const QString line = stream.readLine();
446  if (line.isEmpty() || line[0] == '#') {
447  continue;
448  }
449  const int pos = line.indexOf(':');
450  if (pos == -1) { // syntax error
451  continue;
452  }
453  QString mimeIcon = line.left(pos);
454  const int slashindex = mimeIcon.indexOf(QLatin1Char('/'));
455  if (slashindex != -1) {
456  mimeIcon[slashindex] = QLatin1Char('-');
457  }
458 
459  const QString genericIcon = line.mid(pos + 1);
460  m_genericIcons.insert(mimeIcon, genericIcon);
461  //qCDebug(KICONTHEMES) << mimeIcon << "->" << genericIcon;
462  }
463  }
464 }

468 void KIconLoaderPrivate::drawOverlays(const KIconLoader *iconLoader, KIconLoader::Group group, int state, QPixmap &pix, const QStringList &overlays)
469 {
470  if (overlays.isEmpty()) {
471  return;
472  }
473 
474  const int width = pix.size().width();
475  const int height = pix.size().height();
476  const int iconSize = qMin(width, height);
477  int overlaySize;
478 
479  if (iconSize < 32) {
480  overlaySize = 8;
481  } else if (iconSize <= 48) {
482  overlaySize = 16;
483  } else if (iconSize <= 96) {
484  overlaySize = 22;
485  } else if (iconSize < 256) {
486  overlaySize = 32;
487  } else {
488  overlaySize = 64;
489  }
490 
491  QPainter painter(&pix);
492 
493  int count = 0;
494  foreach (const QString &overlay, overlays) {
495  // Ensure empty strings fill up a emblem spot
496  // Needed when you have several emblems to ensure they're always painted
497  // at the same place, even if one is not here
498  if (overlay.isEmpty()) {
499  ++count;
500  continue;
501  }
502 
503  //TODO: should we pass in the kstate? it results in a slower
504  // path, and perhaps emblems should remain in the default state
505  // anyways?
506  const QPixmap pixmap = iconLoader->loadIcon(overlay, group, overlaySize, state, QStringList(), 0, true);
507 
508  if (pixmap.isNull()) {
509  continue;
510  }
511 
512  QPoint startPoint;
513  switch (count) {
514  case 0:
515  // bottom left corner
516  startPoint = QPoint(2, height - overlaySize - 2);
517  break;
518  case 1:
519  // bottom right corner
520  startPoint = QPoint(width - overlaySize - 2,
521  height - overlaySize - 2);
522  break;
523  case 2:
524  // top right corner
525  startPoint = QPoint(width - overlaySize - 2, 2);
526  break;
527  case 3:
528  // top left corner
529  startPoint = QPoint(2, 2);
530  break;
531  }
532 
533  painter.drawPixmap(startPoint, pixmap);
534 
535  ++count;
536  if (count > 3) {
537  break;
538  }
539  }
540 }

608 bool KIconLoaderPrivate::initIconThemes()
609 {
610  if (mIconThemeInited) {
611  // If mpThemeRoot isn't 0 then initing has succeeded
612  return (mpThemeRoot != 0);
613  }
614  //qCDebug(KICONTHEMES);
615  mIconThemeInited = true;
616 
617  // Add the default theme and its base themes to the theme tree
618  KIconTheme *def = new KIconTheme(KIconTheme::current(), appname);
619  if (!def->isValid()) {
620  delete def;
621  // warn, as this is actually a small penalty hit
622  qCDebug(KICONTHEMES) << "Couldn't find current icon theme, falling back to default.";
623  def = new KIconTheme(KIconTheme::defaultThemeName(), appname);
624  if (!def->isValid()) {
625  qWarning() << "Error: standard icon theme" << KIconTheme::defaultThemeName() << "not found!";
626  delete def;
627  return false;
628  }
629  }
630  mpThemeRoot = new KIconThemeNode(def);
631  mThemesInTree.append(def->internalName());
632  links.append(mpThemeRoot);
633  addBaseThemes(mpThemeRoot, appname);
634 
635  // Insert application specific themes at the top.
636  searchPaths.append(appname + "/pics");
637 
638  // Add legacy icon dirs.
639  searchPaths.append(QStringLiteral("icons")); // was xdgdata-icon in KStandardDirs
640  // These are not in the icon spec, but e.g. GNOME puts some icons there anyway.
641  searchPaths.append(QStringLiteral("pixmaps")); // was xdgdata-pixmaps in KStandardDirs
642 
643  return true;
644 }

689 void KIconLoaderPrivate::addBaseThemes(KIconThemeNode *node, const QString &appname)
690 {
691  // Quote from the icon theme specification:
692  // The lookup is done first in the current theme, and then recursively
693  // in each of the current theme's parents, and finally in the
694  // default theme called "hicolor" (implementations may add more
695  // default themes before "hicolor", but "hicolor" must be last).
696  //
697  // So we first make sure that all inherited themes are added, then we
698  // add the KDE default theme as fallback for all icons that might not be
699  // present in an inherited theme, and hicolor goes last.
700 
701  addInheritedThemes(node, appname);
702  addThemeByName(QStringLiteral("hicolor"), appname);
703 }

705 void KIconLoaderPrivate::addInheritedThemes(KIconThemeNode *node, const QString &appname)
706 {
707  const QStringList lst = node->theme->inherits();
708 
709  for (QStringList::ConstIterator it = lst.begin(); it != lst.end(); ++it) {
710  if ((*it) == QLatin1String("hicolor")) {
711  // The icon theme spec says that "hicolor" must be the very last
712  // of all inherited themes, so don't add it here but at the very end
713  // of addBaseThemes().
714  continue;
715  }
716  addThemeByName(*it, appname);
717  }

739  return;

745  const QStringList icnlibs = QStandardPaths::locateAll(QStandardPaths::GenericDataLocation, QStringLiteral("icons"), QStandardPaths::LocateDirectory);
746  QStringList::ConstIterator it;
747  char buf[1000];
748  for (it = icnlibs.begin(); it != icnlibs.end(); ++it) {
749  QDir dir(*it);
750  if (!dir.exists()) {
751  continue;

753  const QStringList lst = dir.entryList(QStringList(QStringLiteral("default.*")), QDir::Dirs);
754  QStringList::ConstIterator it2;
755  for (it2 = lst.begin(); it2 != lst.end(); ++it2) {
756  if (!QFileInfo::exists(*it + *it2 + "/index.desktop")
757  && !QFileInfo::exists(*it + *it2 + "/index.theme")) {
758  continue;
759  }
760  //TODO: Is any special handling required for NTFS symlinks?
761 #ifndef Q_OS_WIN

763  if (r > 0) {
764  buf[r] = 0;
765  const QDir dir2(buf);
766  QString themeName = dir2.dirName();
767 
768  if (!list.contains(themeName)) {

772 #endif
773  }

775 
776  for (it = list.constBegin(); it != list.constEnd(); ++it) {
777  // Don't add the KDE defaults once more, we have them anyways.
778  if (*it == QLatin1String("default.kde")

1034 QString KIconLoaderPrivate::findMatchingIcon(const QString &name, int size) const
1035 {
1036  const_cast<KIconLoaderPrivate *>(this)->initIconThemes();
1037 
1038  // Do two passes through themeNodes.
1039  //
1040  // The first pass looks for an exact match in each themeNode one after the other.
1041  // If one is found and it is an app icon then return that icon.
1042  //
1043  // In the next pass (assuming the first pass failed), it looks for
1044  // generic fallbacks in each themeNode one after the other.
1045 
1046  // In theory we should only do this for mimetype icons, not for app icons,
1047  // but that would require different APIs. The long term solution is under
1048  // development for Qt >= 5.8, QFileIconProvider calling QPlatformTheme::fileIcon,
1049  // using QMimeType::genericIconName() to get the proper -x-generic fallback.
1050  // Once everone uses that to look up mimetype icons, we can kill the fallback code
1051  // from this method.
1052 
1053  foreach (KIconThemeNode *themeNode, links) {
1054  const QString path = themeNode->theme->iconPathByName(name, size, KIconLoader::MatchBest);
1055  if (!path.isEmpty()) {
1056  return path;
1057  }
1058  }
1059 
1060  if (name.endsWith(QLatin1String("-x-generic"))) {
1061  return QString(); // no further fallback
1062  }
1063  bool genericFallback = false;
1064  QString path;
1065  foreach (KIconThemeNode *themeNode, links) {
1066  QString currentName = name;
1067 
1068  while (!currentName.isEmpty()) {
1069  if (genericFallback) {
1070  // we already tested the base name
1071  break;
1072  }
1073 
1074  int rindex = currentName.lastIndexOf('-');
1075  if (rindex > 1) { // > 1 so that we don't split x-content or x-epoc
1076  currentName.truncate(rindex);
1077 
1078  if (currentName.endsWith(QLatin1String("-x"))) {
1079  currentName.chop(2);
1080  }
1081  } else {
1082  // From update-mime-database.c
1083  static const QSet<QString> mediaTypes = QSet<QString>()
1084  << QStringLiteral("text") << QStringLiteral("application") << QStringLiteral("image") << QStringLiteral("audio")
1085  << QStringLiteral("inode") << QStringLiteral("video") << QStringLiteral("message") << QStringLiteral("model") << QStringLiteral("multipart")
1086  << QStringLiteral("x-content") << QStringLiteral("x-epoc");
1087  // Shared-mime-info spec says:
1088  // "If [generic-icon] is not specified then the mimetype is used to generate the
1089  // generic icon by using the top-level media type (e.g. "video" in "video/ogg")
1090  // and appending "-x-generic" (i.e. "video-x-generic" in the previous example)."
1091  if (mediaTypes.contains(currentName)) {
1092  currentName += QLatin1String("-x-generic");
1093  genericFallback = true;
1094  } else {
1095  break;
1096  }
1097  }
1098 
1099  if (currentName.isEmpty()) {
1100  break;
1101  }
1102 
1103  //qCDebug(KICONTHEMES) << "Looking up" << currentName;
1104  path = themeNode->theme->iconPathByName(currentName, size, KIconLoader::MatchBest);
1105  if (!path.isEmpty()) {
1106  return path;
1107  }
1108  }
1109  }
1110  return path;
1111 }

1230 QPixmap KIconLoader::loadIcon(const QString &_name, KIconLoader::Group group, int size,
1231  int state, const QStringList &overlays,
1232  QString *path_store, bool canReturnNull) const
1233 {
1234  QString name = _name;
1235  bool favIconOverlay = false;
1236 
1237  if (size < 0 || _name.isEmpty()) {
1238  return QPixmap();
1239  }
1240 
1241  /*
1242  * This method works in a kind of pipeline, with the following steps:
1243  * 1. Sanity checks.
1244  * 2. Convert _name, group, size, etc. to a key name.
1245  * 3. Check if the key is already cached.
1246  * 4. If not, initialize the theme and find/load the icon.
1247  * 4a Apply overlays
1248  * 4b Re-add to cache.
1249  */
1250 
1251  // Special case for absolute path icons.
1252  if (name.startsWith(QLatin1String("favicons/"))) {
1253  favIconOverlay = true;
1254  name = QStandardPaths::writableLocation(QStandardPaths::GenericCacheLocation) + '/' + name + ".png";
1255  }
1256 
1257  bool absolutePath = !pathIsRelative(name);
1258  if (!absolutePath) {
1259  name = d->removeIconExtension(name);
1260  }
1261 
1262  // Don't bother looking for an icon with no name.
1263  if (name.isEmpty()) {
1264  return QPixmap();
1265  }
1266 
1267  // May modify group, size, or state. This function puts them into sane
1268  // states.
1269  d->normalizeIconMetadata(group, size, state);
1270 
1271  // See if the image is already cached.
1272  QString key = d->makeCacheKey(name, group, overlays, size, state);
1273  QPixmap pix;
1274  bool iconWasUnknown = false;
1275  QString path;
1276 
1277  if (d->findCachedPixmapWithPath(key, pix, path)) {
1278  if (path_store) {
1279  *path_store = path;
1280  }
1281 
1282  if (!path.isEmpty()) {
1283  return pix;
1284  } else {
1285  // path is empty for "unknown" icons, which should be searched for
1286  // anew regularly
1287  if (!d->shouldCheckForUnknownIcons()) {
1288  return canReturnNull ? QPixmap() : pix;
1289  }
1290  }
1291  }
1292 
1293  // Image is not cached... go find it and apply effects.
1294  if (!d->initIconThemes()) {
1295  return QPixmap();
1296  }
1297 
1298  favIconOverlay = favIconOverlay && size > 22;
1299 
1300  // First we look for non-User icons. If we don't find one we'd search in
1301  // the User space anyways...
1302  if (group != KIconLoader::User) {
1303  if (absolutePath && !favIconOverlay) {
1304  path = name;
1305  } else {
1306  path = d->findMatchingIconWithGenericFallbacks(favIconOverlay ? QStringLiteral("text-html") : name, size);
1307  }
1308  }
1309 
1310  if (path.isEmpty()) {
1311  // We do have a "User" icon, or we couldn't find the non-User one.
1312  path = (absolutePath) ? name :
1313  iconPath(name, KIconLoader::User, canReturnNull);
1314  }
1315 
1316  // Still can't find it? Use "unknown" if we can't return null.
1317  // We keep going in the function so we can ensure this result gets cached.
1318  if (path.isEmpty() && !canReturnNull) {
1319  path = d->unknownIconPath(size);
1320  iconWasUnknown = true;
1321  }
1322 
1323  QImage img;
1324  if (!path.isEmpty()) {
1325  img = d->createIconImage(path, size, (KIconLoader::States)state);
1326  }
1327 
1328  if (group >= 0) {
1329  img = d->mpEffect.apply(img, group, state);
1330  }
1331 
1332  if (favIconOverlay) {
1333  QImage favIcon(name, "PNG");
1334  if (!favIcon.isNull()) { // if favIcon not there yet, don't try to blend it
1335  QPainter p(&img);
1336 
1337  // Align the favicon overlay
1338  QRect r(favIcon.rect());
1339  r.moveBottomRight(img.rect().bottomRight());
1340  r.adjust(-1, -1, -1, -1); // Move off edge
1341 
1342  // Blend favIcon over img.
1343  p.drawImage(r, favIcon);
1344  }
1345  }
1346 
1347  pix = QPixmap::fromImage(img);
1348 
1349  // TODO: If we make a loadIcon that returns the image we can convert
1350  // drawOverlays to use the image instead of pixmaps as well so we don't
1351  // have to transfer so much to the graphics card.
1352  d->drawOverlays(this, group, state, pix, overlays);
1353 
1354  // Don't add the path to our unknown icon to the cache, only cache the
1355  // actual image.
1356  if (iconWasUnknown) {
1357  path.clear();
1358  }
1359 
1360  d->insertCachedPixmapWithPath(key, pix, path);
1361 
1362  if (path_store) {
1363  *path_store = path;
1364  }
1365 
1366  return pix;
1367 }
