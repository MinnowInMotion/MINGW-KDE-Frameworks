253 {
254  QFileInfo pathInfo = path;

274  return KSambaShareData::UserSharePathNotAllowed;

302 KSambaShareData::UserShareError KSambaSharePrivate::add(const KSambaShareData &shareData)
303 {
304  // TODO:
305  // * check for usershare max shares
306 
307  if (!isSambaInstalled()) {
308  return KSambaShareData::UserShareSystemError;
309  }
310 
311  QStringList args;
312  QByteArray stdOut;
313  QByteArray stdErr;
314 
315  if (data.contains(shareData.name())) {
316  if (data.value(shareData.name()).path() != shareData.path()) {
317  return KSambaShareData::UserShareNameInUse;
318  }
319  } else {
320  // It needs to be added here, otherwise another instance of KSambaShareDataPrivate
321  // will be created and added to data.
322  data.insert(shareData.name(), shareData);
323  }
324 
325  QString guestok = QStringLiteral("guest_ok=%1").arg(
326  (shareData.guestPermission() == KSambaShareData::GuestsNotAllowed)
327  ? QStringLiteral("n") : QStringLiteral("y"));
328 
329  args << QStringLiteral("usershare") << QStringLiteral("add") << shareData.name()
330  << shareData.path() << shareData.comment() << shareData.acl() << guestok;
331 
332  int ret = runProcess(QStringLiteral("net"), args, stdOut, stdErr);
333 
334  //TODO: parse and process error messages.
335  if (!stdErr.isEmpty()) {
336  // create a parser for the error output and
337  // send error message somewhere
338  qWarning() << "We got some errors while running 'net usershare add'" << args;
339  qWarning() << stdErr;
340  }
341 
342  return (ret == 0) ? KSambaShareData::UserShareOk : KSambaShareData::UserShareSystemError;
343 }

346 {
347  if (!isSambaInstalled()) {

363 bool KSambaSharePrivate::sync()
364 {
365  const QRegExp headerRx(QString::fromLatin1("^\\s*\\["
366  "([^%<>*\?|/\\+=;:\",]+)"
367  "\\]"));
368 
369  const QRegExp OptValRx(QString::fromLatin1("^\\s*([\\w\\d\\s]+)"
370  "="
371  "(.*)$"));
372 
373  QTextStream stream(getNetUserShareInfo());
374  QString currentShare;
375  QStringList shareList;
376 
377  while (!stream.atEnd()) {
378  const QString line = stream.readLine().trimmed();
379 
380  if (headerRx.exactMatch(line)) {
381  currentShare = headerRx.cap(1).trimmed();
382  shareList << currentShare;
383 
384  if (!data.contains(currentShare)) {
385  KSambaShareData shareData;
386  shareData.dd->name = currentShare;
387  data.insert(currentShare, shareData);
388  }
389  } else if (OptValRx.exactMatch(line)) {
390  const QString key = OptValRx.cap(1).trimmed();
391  const QString value = OptValRx.cap(2).trimmed();
392  KSambaShareData shareData = getShareByName(currentShare);
393 
394  if (key == QLatin1String("path")) {
395  shareData.dd->path = value;
396  } else if (key == QLatin1String("comment")) {
397  shareData.dd->comment = value;
398  } else if (key == QLatin1String("usershare_acl")) {
399  shareData.dd->acl = value;
400  } else if (key == QLatin1String("guest_ok")) {
401  shareData.dd->guestPermission = value;
402  } else {
403  qWarning() << "Something nasty happen while parsing 'net usershare info'"
404  << "share:" << currentShare << "key:" << key;
405  }
406  } else if (line.trimmed().isEmpty()) {
407  continue;
408  } else {
409  return false;
410  }
411  }
412 
413  QMutableMapIterator<QString, KSambaShareData> i(data);
414  while (i.hasNext()) {
415  i.next();
416  if (!shareList.contains(i.key())) {
417  i.remove();
418  }
419  }
420 
421  return true;
422 }
