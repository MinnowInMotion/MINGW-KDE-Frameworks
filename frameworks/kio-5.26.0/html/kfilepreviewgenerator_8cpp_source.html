113 
114  TileSet()
115  {
116  QImage image(8 * 3, 8 * 3, QImage::Format_ARGB32_Premultiplied);
117 
118  QPainter p(&image);
119  p.setCompositionMode(QPainter::CompositionMode_Source);
120  p.fillRect(image.rect(), Qt::transparent);
121  p.fillRect(image.rect().adjusted(3, 3, -3, -3), Qt::black);

125 
126  QPixmap pixmap = QPixmap::fromImage(image);
127  m_tiles[TopLeftCorner] = pixmap.copy(0, 0, 8, 8);
128  m_tiles[TopSide] = pixmap.copy(8, 0, 8, 8);
129  m_tiles[TopRightCorner] = pixmap.copy(16, 0, 8, 8);
130  m_tiles[LeftSide] = pixmap.copy(0, 8, 8, 8);
131  m_tiles[RightSide] = pixmap.copy(16, 8, 8, 8);
132  m_tiles[BottomLeftCorner] = pixmap.copy(0, 16, 8, 8);
133  m_tiles[BottomSide] = pixmap.copy(8, 16, 8, 8);
134  m_tiles[BottomRightCorner] = pixmap.copy(16, 16, 8, 8);
135  }
136 
137  void paint(QPainter *p, const QRect &r)
138  {
139  p->drawPixmap(r.topLeft(), m_tiles[TopLeftCorner]);
140  if (r.width() - 16 > 0) {
141  p->drawTiledPixmap(r.x() + 8, r.y(), r.width() - 16, 8, m_tiles[TopSide]);
142  }
143  p->drawPixmap(r.right() - 8 + 1, r.y(), m_tiles[TopRightCorner]);
144  if (r.height() - 16 > 0) {
145  p->drawTiledPixmap(r.x(), r.y() + 8, 8, r.height() - 16, m_tiles[LeftSide]);
146  p->drawTiledPixmap(r.right() - 8 + 1, r.y() + 8, 8, r.height() - 16, m_tiles[RightSide]);

558 void KFilePreviewGenerator::Private::updateIcons(const QModelIndex &topLeft,
559  const QModelIndex &bottomRight)
560 {
561  if (m_internalDataChange > 0) {
562  // QAbstractItemModel::setData() has been invoked internally by the KFilePreviewGenerator.
563  // The signal dataChanged() is connected with this method, but previews only need
564  // to be generated when an external data change has occurred.
565  return;
566  }
567 
568  // dataChanged emitted for the root dir (e.g. permission changes)
569  if (!topLeft.isValid() || !bottomRight.isValid()) {
570  return;
571  }
572 
573  KDirModel *dirModel = m_dirModel.data();
574  if (!dirModel) {
575  return;
576  }
577 
578  KFileItemList itemList;
579  for (int row = topLeft.row(); row <= bottomRight.row(); ++row) {
580  const QModelIndex index = dirModel->index(row, 0);
581  if (!index.isValid()) {
582  continue;
583  }
584  const KFileItem item = dirModel->itemForIndex(index);
585  Q_ASSERT(!item.isNull());
586 
587  if (m_previewShown) {
588  const QUrl url = item.url();
589  const bool hasChanged = m_changedItems.contains(url); // O(1)
590  m_changedItems.insert(url, hasChanged);
591  if (!hasChanged) {
592  // only update the icon if it has not been already updated within
593  // the last 5 seconds (the other icons will be updated later with
594  // the help of m_changedItemsTimer)
595  itemList.append(item);
596  }
597  } else {
598  itemList.append(item);
599  }
600  }
601 
602  updateIcons(itemList);
603  m_changedItemsTimer->start();
604 }

606 void KFilePreviewGenerator::Private::addToPreviewQueue(const KFileItem &item, const QPixmap &pixmap)
607 {
608  KIO::PreviewJob *senderJob = qobject_cast<KIO::PreviewJob *>(q->sender());
609  Q_ASSERT(senderJob != 0);
610  if (senderJob != 0) {
611  QMap<QUrl, int>::iterator it = m_sequenceIndices.find(item.url());
612  if (senderJob->sequenceIndex() && (it == m_sequenceIndices.end() || *it != senderJob->sequenceIndex())) {
613  return; // the sequence index does not match the one we want
614  }
615  if (!senderJob->sequenceIndex() && it != m_sequenceIndices.end()) {
616  return; // the sequence index does not match the one we want
617  }
618 
619  m_sequenceIndices.erase(it);
620  }
621 
622  if (!m_previewShown) {
623  // the preview has been canceled in the meantime
624  return;
625  }
626 
627  KDirModel *dirModel = m_dirModel.data();
628  if (!dirModel) {
629  return;
630  }
631 
632  // check whether the item is part of the directory lister (it is possible
633  // that a preview from an old directory lister is received)
634  bool isOldPreview = true;
635 
636  const QUrl itemParentDir = item.url().adjusted(QUrl::RemoveFilename | QUrl::StripTrailingSlash);
637 
638  foreach (const QUrl &dir, dirModel->dirLister()->directories()) {
639  if (dir == itemParentDir || dir.path().isEmpty()) {
640  isOldPreview = false;
641  break;
642  }
643  }
644  if (isOldPreview) {
645  return;
646  }
647 
648  QPixmap icon = pixmap;
649 
650  const QString mimeType = item.mimetype();
651  const int slashIndex = mimeType.indexOf(QLatin1Char('/'));
652  const QString mimeTypeGroup = mimeType.left(slashIndex);
653  if ((mimeTypeGroup != QLatin1String("image")) || !applyImageFrame(icon)) {
654  limitToSize(icon, m_viewAdapter->iconSize());
655  }
656 
657  if (m_hasCutSelection && isCutItem(item)) {
658  // apply the disabled effect to the icon for marking it as "cut item"
659  // and apply the icon to the item
660  KIconEffect *iconEffect = KIconLoader::global()->iconEffect();
661  icon = iconEffect->apply(icon, KIconLoader::Desktop, KIconLoader::DisabledState);
662  }
663 
664  KIconLoader::global()->drawOverlays(item.overlays(), icon, KIconLoader::Desktop);
665 
666  // remember the preview and URL, so that it can be applied to the model
667  // in KFilePreviewGenerator::dispatchIconUpdateQueue()
668  ItemInfo preview;
669  preview.url = item.url();
670  preview.pixmap = icon;
671  m_previews.append(preview);
672 
673  m_pendingItems.removeOne(item);
674 
675  m_dispatchedItems.append(item);
676 }

678 void KFilePreviewGenerator::Private::slotPreviewJobFinished(KJob *job)
679 {
680  const int index = m_previewJobs.indexOf(job);
681  m_previewJobs.removeAt(index);
682 
683  if (m_previewJobs.isEmpty()) {
684  foreach (const KFileItem &item, m_pendingItems) {
685  if (item.isMimeTypeKnown()) {
686  m_resolvedMimeTypes.append(item);
687  }
688  }
689 
690  if (m_clearItemQueues) {
691  m_pendingItems.clear();
692  m_dispatchedItems.clear();
693  m_pendingVisibleIconUpdates = 0;
694  QMetaObject::invokeMethod(q, "dispatchIconUpdateQueue", Qt::QueuedConnection);
695  }
696  m_sequenceIndices.clear(); // just to be sure that we don't leak anything
697  }
698 }

719 void KFilePreviewGenerator::Private::clearCutItemsCache()
720 {
721  KDirModel *dirModel = m_dirModel.data();
722  if (!dirModel) {
723  return;
724  }
725 
726  DataChangeObtainer obt(this);
727  KFileItemList previews;
728  // Reset the icons of all items that are stored in the cache
729  // to use their default MIME type icon.
730  foreach (const QUrl &url, m_cutItemsCache.keys()) {
731  const QModelIndex index = dirModel->indexForUrl(url);
732  if (index.isValid()) {
733  dirModel->setData(index, QIcon(), Qt::DecorationRole);
734  if (m_previewShown) {
735  previews.append(dirModel->itemForIndex(index));
736  }
737  }
738  }
739  m_cutItemsCache.clear();
740 
741  if (previews.size() > 0) {
742  // assure that the previews gets restored
743  Q_ASSERT(m_previewShown);
744  orderItems(previews);
745  updateIcons(previews);
746  }

749 void KFilePreviewGenerator::Private::dispatchIconUpdateQueue()
750 {
751  KDirModel *dirModel = m_dirModel.data();
752  if (!dirModel) {
753  return;
754  }
755 
756  const int count = m_previews.count() + m_resolvedMimeTypes.count();
757  if (count > 0) {
758  LayoutBlocker blocker(m_itemView);
759  DataChangeObtainer obt(this);
760 
761  if (m_previewShown) {
762  // dispatch preview queue
763  foreach (const ItemInfo &preview, m_previews) {
764  const QModelIndex idx = dirModel->indexForUrl(preview.url);
765  if (idx.isValid() && (idx.column() == 0)) {
766  dirModel->setData(idx, QIcon(preview.pixmap), Qt::DecorationRole);
767  }
768  }
769  m_previews.clear();
770  }
771 
772  // dispatch mime type queue
773  foreach (const KFileItem &item, m_resolvedMimeTypes) {
774  const QModelIndex idx = dirModel->indexForItem(item);
775  dirModel->itemChanged(idx);
776  }
777  m_resolvedMimeTypes.clear();
778 
779  m_pendingVisibleIconUpdates -= count;
780  if (m_pendingVisibleIconUpdates < 0) {
781  m_pendingVisibleIconUpdates = 0;
782  }
783  }
784 
785  if (m_pendingVisibleIconUpdates > 0) {
786  // As long as there are pending previews for visible items, poll
787  // the preview queue periodically. If there are no pending previews,
788  // the queue is dispatched in slotPreviewJobFinished().
789  m_iconUpdateTimer->start();
790  }
791 }

803 void KFilePreviewGenerator::Private::resumeIconUpdates()
804 {
805  m_iconUpdatesPaused = false;
806 
807  // Before creating new preview jobs the m_pendingItems queue must be
808  // cleaned up by removing the already dispatched items. Implementation
809  // note: The order of the m_dispatchedItems queue and the m_pendingItems
810  // queue is usually equal. So even when having a lot of elements the
811  // nested loop is no performance bottle neck, as the inner loop is only
812  // entered once in most cases.
813  foreach (const KFileItem &item, m_dispatchedItems) {
814  KFileItemList::iterator begin = m_pendingItems.begin();
815  KFileItemList::iterator end = m_pendingItems.end();
816  for (KFileItemList::iterator it = begin; it != end; ++it) {
817  if ((*it).url() == item.url()) {
818  m_pendingItems.erase(it);
819  break;
820  }
821  }
822  }
823  m_dispatchedItems.clear();
824 
825  m_pendingVisibleIconUpdates = 0;
826  dispatchIconUpdateQueue();
827 
828  if (m_previewShown) {
829  KFileItemList orderedItems = m_pendingItems;
830  orderItems(orderedItems);
831 
832  // Kill all suspended preview jobs. Usually when a preview job
833  // has been finished, slotPreviewJobFinished() clears all item queues.
834  // This is not wanted in this case, as a new job is created afterwards
835  // for m_pendingItems.
836  m_clearItemQueues = false;
837  killPreviewJobs();
838  m_clearItemQueues = true;
839 
840  createPreviews(orderedItems);
841  } else {
842  orderItems(m_pendingItems);
843  startMimeTypeResolving();
844  }
845 }

853 void KFilePreviewGenerator::Private::resolveMimeType()
854 {
855  if (m_pendingItems.isEmpty()) {
856  return;
857  }
858 
859  // resolve at least one MIME type
860  bool resolved = false;
861  do {
862  KFileItem item = m_pendingItems.takeFirst();
863  if (item.isMimeTypeKnown()) {
864  if (m_pendingVisibleIconUpdates > 0) {
865  // The item is visible and the MIME type already known.
866  // Decrease the update counter for dispatchIconUpdateQueue():
867  --m_pendingVisibleIconUpdates;
868  }
869  } else {
870  // The MIME type is unknown and must get resolved. The
871  // directory model is not informed yet, as a single update
872  // would be very expensive. Instead the item is remembered in
873  // m_resolvedMimeTypes and will be dispatched later
874  // by dispatchIconUpdateQueue().
875  item.determineMimeType();
876  m_resolvedMimeTypes.append(item);
877  resolved = true;
878  }
879  } while (!resolved && !m_pendingItems.isEmpty());
880 
881  if (m_pendingItems.isEmpty()) {
882  // All MIME types have been resolved now. Assure
883  // that the directory model gets informed about
884  // this, so that an update of the icons is done.
885  dispatchIconUpdateQueue();
886  } else if (!m_iconUpdatesPaused) {
887  // assure that the MIME type of the next
888  // item will be resolved asynchronously
889  QMetaObject::invokeMethod(q, "resolveMimeType", Qt::QueuedConnection);
890  }
891 }

1023 void KFilePreviewGenerator::Private::createPreviews(const KFileItemList &items)
1024 {
1025  if (items.count() == 0) {
1026  return;
1027  }
1028 
1029  const QMimeData *mimeData = QApplication::clipboard()->mimeData();
1030  m_hasCutSelection = mimeData && KIO::isClipboardDataCut(mimeData);
1031 
1032  // PreviewJob internally caches items always with the size of
1033  // 128 x 128 pixels or 256 x 256 pixels. A downscaling is done
1034  // by PreviewJob if a smaller size is requested. For images KFilePreviewGenerator must
1035  // do a downscaling anyhow because of the frame, so in this case only the provided
1036  // cache sizes are requested.

1040  QString mimeTypeGroup;

1044  mimeTypeGroup = mimeType.left(slashIndex);

1046  imageItems.append(item);

1048  otherItems.append(item);
1049  }
1050  }
1051  const QSize size = m_viewAdapter->iconSize();
1052  startPreviewJob(otherItems, size.width(), size.height());

1055  startPreviewJob(imageItems, cacheSize, cacheSize);

1096 void KFilePreviewGenerator::Private::orderItems(KFileItemList &items)
1097 {
1098  KDirModel *dirModel = m_dirModel.data();
1099  if (!dirModel) {
1100  return;
1101  }
1102 
1103  // Order the items in a way that the preview for the visible items
1104  // is generated first, as this improves the feeled performance a lot.
1105  const bool hasProxy = (m_proxyModel != 0);
1106  const int itemCount = items.count();
1107  const QRect visibleArea = m_viewAdapter->visibleArea();
1108 
1109  QModelIndex dirIndex;
1110  QRect itemRect;
1111  int insertPos = 0;
1112  for (int i = 0; i < itemCount; ++i) {
1113  dirIndex = dirModel->indexForItem(items.at(i)); // O(n) (n = number of rows)
1114  if (hasProxy) {
1115  const QModelIndex proxyIndex = m_proxyModel->mapFromSource(dirIndex);
1116  itemRect = m_viewAdapter->visualRect(proxyIndex);
1117  } else {
1118  itemRect = m_viewAdapter->visualRect(dirIndex);
1119  }
1120 
1121  if (itemRect.intersects(visibleArea)) {
1122  // The current item is (at least partly) visible. Move it
1123  // to the front of the list, so that the preview is
1124  // generated earlier.
1125  items.insert(insertPos, items.at(i));
1126  items.removeAt(i + 1);
1127  ++insertPos;
1128  ++m_pendingVisibleIconUpdates;
1129  }
1130  }
1131 }
