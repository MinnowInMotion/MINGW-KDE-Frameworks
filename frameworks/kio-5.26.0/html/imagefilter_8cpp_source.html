85 inline static void blurHorizontal(QImage &image, unsigned int *stack, int div, int radius)
86 {
87  int stackindex;
88  int stackstart;
89 
90  quint32 *const pixels = reinterpret_cast<quint32 *>(image.bits());
91  quint32 pixel;
92 
93  int w = image.width();
94  int h = image.height();
95  int wm = w - 1;
96 
97  unsigned int mul_sum = stack_blur8_mul[radius];
98  unsigned int shr_sum = stack_blur8_shr[radius];
99 
100  unsigned int sum, sum_in, sum_out;
101 
102  for (int y = 0; y < h; y++) {
103  sum = 0;
104  sum_in = 0;
105  sum_out = 0;
106 
107  const int yw = y * w;
108  pixel = pixels[yw];
109  for (int i = 0; i <= radius; i++) {
110  stack[i] = qAlpha(pixel);
111 
112  sum += stack[i] * (i + 1);
113  sum_out += stack[i];
114  }
115 
116  for (int i = 1; i <= radius; i++) {
117  pixel = pixels[yw + qMin(i, wm)];
118 
119  unsigned int *stackpix = &stack[i + radius];
120  *stackpix = qAlpha(pixel);
121 
122  sum += *stackpix * (radius + 1 - i);
123  sum_in += *stackpix;
124  }
125 
126  stackindex = radius;
127  for (int x = 0, i = yw; x < w; x++) {
128  pixels[i++] = (((sum * mul_sum) >> shr_sum) << 24) & 0xff000000;
129 
130  sum -= sum_out;
131 
132  stackstart = stackindex + div - radius;
133  if (stackstart >= div) {
134  stackstart -= div;
135  }
136 
137  unsigned int *stackpix = &stack[stackstart];
138 
139  sum_out -= *stackpix;
140 
141  pixel = pixels[yw + qMin(x + radius + 1, wm)];
142 
143  *stackpix = qAlpha(pixel);
144 
145  sum_in += *stackpix;
146  sum += sum_in;
147 
148  if (++stackindex >= div) {
149  stackindex = 0;
150  }
151 
152  stackpix = &stack[stackindex];
153 
154  sum_out += *stackpix;
155  sum_in -= *stackpix;
156  } // for (x = 0, ...)
157  } // for (y = 0, ...)
158 }

234 static void stackBlur(QImage &image, float radius)
235 {
236  radius = qRound(radius);
237 
238  int div = int(radius * 2) + 1;
239  unsigned int *stack = new unsigned int[div];
240 
241  blurHorizontal(image, stack, div, radius);
242  blurVertical(image, stack, div, radius);
243 
244  delete [] stack;
245 }

247 void ImageFilter::shadowBlur(QImage &image, float radius, const QColor &color)
248 {
249  if (radius < 0) {
250  return;
251  }
252 
253  if (radius > 0) {
254  stackBlur(image, radius);
255  }
256 
257  // Correct the color and opacity of the shadow
258  QPainter p(&image);
259  p.setCompositionMode(QPainter::CompositionMode_SourceIn);
260  p.fillRect(image.rect(), color);
261 }
