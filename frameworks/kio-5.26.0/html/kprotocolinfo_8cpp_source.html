34 KProtocolInfoPrivate::KProtocolInfoPrivate(const QString &path)
35 {
36  KConfig sconfig(path, KConfig::SimpleConfig);
37  KConfigGroup config(&sconfig, "Protocol");
38 
39  m_name = config.readEntry("protocol");
40  m_exec = config.readPathEntry("exec", QString());
41  m_isSourceProtocol = config.readEntry("source", true);
42  m_isHelperProtocol = config.readEntry("helper", false);
43  m_supportsReading = config.readEntry("reading", false);
44  m_supportsWriting = config.readEntry("writing", false);
45  m_supportsMakeDir = config.readEntry("makedir", false);
46  m_supportsDeleting = config.readEntry("deleting", false);
47  m_supportsLinking = config.readEntry("linking", false);
48  m_supportsMoving = config.readEntry("moving", false);
49  m_supportsOpening = config.readEntry("opening", false);
50  m_canCopyFromFile = config.readEntry("copyFromFile", false);
51  m_canCopyToFile = config.readEntry("copyToFile", false);
52  m_canRenameFromFile = config.readEntry("renameFromFile", false);
53  m_canRenameToFile = config.readEntry("renameToFile", false);
54  m_canDeleteRecursive = config.readEntry("deleteRecursive", false);
55  const QString fnu = config.readEntry("fileNameUsedForCopying", "FromURL");
56  m_fileNameUsedForCopying = KProtocolInfo::FromUrl;
57  if (fnu == QLatin1String("Name")) {
58  m_fileNameUsedForCopying = KProtocolInfo::Name;
59  } else if (fnu == QLatin1String("DisplayName")) {
60  m_fileNameUsedForCopying = KProtocolInfo::DisplayName;
61  }
62 
63  m_listing = config.readEntry("listing", QStringList());
64  // Many .protocol files say "Listing=false" when they really mean "Listing=" (i.e. unsupported)
65  if (m_listing.count() == 1 && m_listing.first() == QLatin1String("false")) {
66  m_listing.clear();
67  }
68  m_supportsListing = (m_listing.count() > 0);
69  m_defaultMimetype = config.readEntry("defaultMimetype");
70  m_determineMimetypeFromExtension = config.readEntry("determineMimetypeFromExtension", true);
71  m_archiveMimeTypes = config.readEntry("archiveMimetype", QStringList());
72  m_icon = config.readEntry("Icon");
73  m_config = config.readEntry("config", m_name);
74  m_maxSlaves = config.readEntry("maxInstances", 1);
75  m_maxSlavesPerHost = config.readEntry("maxInstancesPerHost", 0);
76 
77  QString tmp = config.readEntry("input");
78  if (tmp == QLatin1String("filesystem")) {
79  m_inputType = KProtocolInfo::T_FILESYSTEM;
80  } else if (tmp == QLatin1String("stream")) {
81  m_inputType = KProtocolInfo::T_STREAM;
82  } else {
83  m_inputType = KProtocolInfo::T_NONE;
84  }
85 
86  tmp = config.readEntry("output");
87  if (tmp == QLatin1String("filesystem")) {
88  m_outputType = KProtocolInfo::T_FILESYSTEM;
89  } else if (tmp == QLatin1String("stream")) {
90  m_outputType = KProtocolInfo::T_STREAM;
91  } else {
92  m_outputType = KProtocolInfo::T_NONE;
93  }
94 
95  m_docPath = config.readPathEntry("X-DocPath", QString());
96  if (m_docPath.isEmpty()) {
97  m_docPath = config.readPathEntry("DocPath", QString());
98  }
99  m_protClass = config.readEntry("Class").toLower();
100  if (m_protClass[0] != QLatin1Char(':')) {
101  m_protClass.prepend(QLatin1Char(':'));
102  }
103 
104  const QStringList extraNames = config.readEntry("ExtraNames", QStringList());
105  const QStringList extraTypes = config.readEntry("ExtraTypes", QStringList());
106  QStringList::const_iterator it = extraNames.begin();
107  QStringList::const_iterator typeit = extraTypes.begin();
108  for (; it != extraNames.end() && typeit != extraTypes.end(); ++it, ++typeit) {
109  QVariant::Type type = QVariant::nameToType((*typeit).toLatin1());
110  // currently QVariant::Type and ExtraField::Type use the same subset of values, so we can just cast.
111  m_extraFields.append(KProtocolInfo::ExtraField(*it, static_cast<KProtocolInfo::ExtraField::Type>(type)));
112  }
113 
114  m_showPreviews = config.readEntry("ShowPreviews", m_protClass == QLatin1String(":local"));
115 
116  m_capabilities = config.readEntry("Capabilities", QStringList());
117 
118  m_slaveHandlesNotify = config.readEntry("slaveHandlesNotify", QStringList());
119 
120  m_proxyProtocol = config.readEntry("ProxiedBy");
121 }

237 bool KProtocolInfo::isFilterProtocol(const QString &_protocol)
238 {
239  // We call the findProtocol directly (not via KProtocolManager) to bypass any proxy settings.
240  KProtocolInfoPrivate *prot = KProtocolInfoFactory::self()->findProtocol(_protocol);
