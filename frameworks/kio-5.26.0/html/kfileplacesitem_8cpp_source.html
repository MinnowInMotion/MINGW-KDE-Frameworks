43 KFilePlacesItem::KFilePlacesItem(KBookmarkManager *manager,
44  const QString &address,
45  const QString &udi)
46  : m_manager(manager), m_folderIsEmpty(true), m_isCdrom(false),
47  m_isAccessible(false), m_device(udi)
48 {
49  setBookmark(m_manager->findByAddress(address));
50 
51  if (udi.isEmpty() && m_bookmark.metaDataItem(QStringLiteral("ID")).isEmpty()) {
52  m_bookmark.setMetaDataItem(QStringLiteral("ID"), generateNewId());
53  } else if (udi.isEmpty()) {
54  if (isTrash(m_bookmark)) {
55  KConfig cfg(QStringLiteral("trashrc"), KConfig::SimpleConfig);
56  const KConfigGroup group = cfg.group("Status");
57  m_folderIsEmpty = group.readEntry("Empty", true);
58  }
59  } else if (!udi.isEmpty() && m_device.isValid()) {
60  m_access = m_device.as<Solid::StorageAccess>();
61  m_volume = m_device.as<Solid::StorageVolume>();
62  m_disc = m_device.as<Solid::OpticalDisc>();
63  m_mtp = m_device.as<Solid::PortableMediaPlayer>();
64  if (m_access) {
65  connect(m_access, SIGNAL(accessibilityChanged(bool,QString)),
66  this, SLOT(onAccessibilityChanged(bool)));
67  onAccessibilityChanged(m_access->isAccessible());
68  }
69  m_iconPath = m_device.icon();
70  m_emblems = m_device.emblems();
71  }
72 }

145 QVariant KFilePlacesItem::bookmarkData(int role) const
146 {
147  KBookmark b = bookmark();
148 
149  if (b.isNull()) {
150  return QVariant();
151  }
152 
153  switch (role) {
154  case Qt::DisplayRole:
155  return m_text;
156  case Qt::DecorationRole:
157  return QIcon::fromTheme(iconNameForBookmark(b));
158  case Qt::BackgroundRole:
159  if (b.metaDataItem(QStringLiteral("IsHidden")) == QLatin1String("true")) {
160  return QColor(Qt::lightGray);
161  } else {
162  return QVariant();
163  }
164  case KFilePlacesModel::UrlRole:
165  return b.url();
166  case KFilePlacesModel::SetupNeededRole:
167  return false;
168  case KFilePlacesModel::HiddenRole:
169  return b.metaDataItem(QStringLiteral("IsHidden")) == QLatin1String("true");
170  default:
171  return QVariant();
172  }
173 }

175 QVariant KFilePlacesItem::deviceData(int role) const
176 {
177  Solid::Device d = device();
178 
179  if (d.isValid()) {
180  switch (role) {
181  case Qt::DisplayRole:
182  return d.description();
183  case Qt::DecorationRole:
184  return KDE::icon(m_iconPath, m_emblems);
185  case KFilePlacesModel::UrlRole:
186  if (m_access) {
187  const QString path = m_access->filePath();
188  return path.isEmpty() ? QUrl() : QUrl::fromLocalFile(path);
189  } else if (m_disc && (m_disc->availableContent() & Solid::OpticalDisc::Audio) != 0) {
190  Solid::Block *block = d.as<Solid::Block>();
191  if (block) {
192  QString device = block->device();
193  return QUrl(QStringLiteral("audiocd:/?device=%1").arg(device));
194  }
195  // We failed to get the block device. Assume audiocd:/ can
196  // figure it out, but cannot handle multiple disc drives.
197  // See https://bugs.kde.org/show_bug.cgi?id=314544#c40
198  return QUrl(QStringLiteral("audiocd:/"));
199  } else if (m_mtp) {
200  return QUrl(QStringLiteral("mtp:udi=%1").arg(d.udi()));
201  } else {
202  return QVariant();
203  }
204  case KFilePlacesModel::SetupNeededRole:
205  if (m_access) {
206  return !m_isAccessible;
207  } else {
208  return QVariant();
209  }
210 
211  case KFilePlacesModel::FixedDeviceRole: {
212  Solid::StorageDrive *drive = 0;
213  Solid::Device parentDevice = m_device;
214  while (parentDevice.isValid() && !drive) {
215  drive = parentDevice.as<Solid::StorageDrive>();
216  parentDevice = parentDevice.parent();
217  }
218  if (drive != 0) {
219  return !drive->isHotpluggable() && !drive->isRemovable();
220  }
221  return true;
222  }
223 
224  case KFilePlacesModel::CapacityBarRecommendedRole:
225  return m_isAccessible && !m_isCdrom;
226 
227  default:
228  return QVariant();
229  }
230  } else {
231  return QVariant();
232  }
233 }

235 KBookmark KFilePlacesItem::createBookmark(KBookmarkManager *manager,
236  const QString &label,
237  const QUrl &url,
238  const QString &iconName,
239  KFilePlacesItem *after)
240 {
241  KBookmarkGroup root = manager->root();
242  if (root.isNull()) {
243  return KBookmark();
244  }
245  QString empty_icon = iconName;
246  if (url.toString() == QLatin1String("trash:/")) {
247  if (empty_icon.endsWith(QLatin1String("-full"))) {
248  empty_icon.chop(5);
249  } else if (empty_icon.isEmpty()) {
250  empty_icon = QStringLiteral("user-trash");
251  }
252  }
253  KBookmark bookmark = root.addBookmark(label, url, empty_icon);
254  bookmark.setMetaDataItem(QStringLiteral("ID"), generateNewId());
255 
256  if (after) {
257  root.moveBookmark(bookmark, after->bookmark());
258  }
259 
260  return bookmark;
261 }
