207 {
208  Q_OBJECT
209 protected:
210  CompletionThread(KUrlCompletionPrivate *receiver) :

216 
217 public:
218  void requestTermination()
219  {
220  if (!isFinished()) {
221  qCDebug(KIO_WIDGETS) << "stopping thread" << this;
222  }
223  m_terminationRequested.store(true);
224  wait();
225  }

227  QStringList matches() const
228  {
229  QMutexLocker locker(&m_mutex);
230  return m_matches;
231  }
232 
233 Q_SIGNALS:
234  void completionThreadDone(QThread *thread, const QStringList &matches);
235 
236 protected:
237  void addMatch(const QString &match)
238  {
239  QMutexLocker locker(&m_mutex);
240  m_matches.append(match);
241  }
242  bool terminationRequested() const
243  {
244  return m_terminationRequested.load();
245  }
246  void done()
247  {
248  if (!terminationRequested()) {
249  qCDebug(KIO_WIDGETS) << "done, emitting signal with" << m_matches.count() << "matches";
250  emit completionThreadDone(this, m_matches);
251  }
252  }
253 
254  const QString m_prepend;
255  const bool m_complete_url; // if true completing a URL (i.e. 'm_prepend' is a URL), otherwise a path
256 
257 private:
258  mutable QMutex m_mutex; // protects m_matches
259  QStringList m_matches; // written by secondary thread, read by the matches() method
260  QAtomicInt m_terminationRequested; // used as a bool
261 };

335 void DirectoryListThread::run()
336 {
337  //qDebug() << "Entered DirectoryListThread::run(), m_filter=" << m_filter << ", m_onlyExe=" << m_onlyExe << ", m_onlyDir=" << m_onlyDir << ", m_appendSlashToDir=" << m_appendSlashToDir << ", m_dirList.size()=" << m_dirList.size();
338 
339  QDir::Filters iterator_filter = (m_noHidden ? QDir::Filter(0) : QDir::Hidden) | QDir::Readable | QDir::NoDotAndDotDot;
340  if (m_onlyExe) {
341  iterator_filter |= (QDir::Dirs | QDir::Files | QDir::Executable);
342  } else if (m_onlyDir) {
343  iterator_filter |= QDir::Dirs;
344  } else {
345  iterator_filter |= (QDir::Dirs | QDir::Files);
346  }
347 
348  const QStringList::const_iterator end = m_dirList.constEnd();
349  for (QStringList::const_iterator it = m_dirList.constBegin();
350  it != end && !terminationRequested();
351  ++it) {
352  //qDebug() << "Scanning directory" << *it;
353 
354  QDirIterator current_dir_iterator(*it, iterator_filter);
355 
356  while (current_dir_iterator.hasNext() && !terminationRequested()) {
357  current_dir_iterator.next();
358 
359  QFileInfo file_info = current_dir_iterator.fileInfo();
360  const QString file_name = file_info.fileName();
361 
362  //qDebug() << "Found" << file_name;
363 
364  if (m_filter.isEmpty() || file_name.startsWith(m_filter)) {
365 
366  QString toAppend = file_name;
367  // Add '/' to directories
368  if (m_appendSlashToDir && file_info.isDir()) {
369  toAppend.append(QLatin1Char('/'));
370  }
371 
372  if (m_complete_url) {
373  QUrl info(m_prepend);
374  info = addPathToUrl(info, toAppend);
375  addMatch(info.toDisplayString());
376  } else {
377  addMatch(m_prepend + toAppend);
378  }
379  }
380  }
381  }
382 
383  done();
384 }

459 void KUrlCompletionPrivate::MyURL::init(const QString &_url, const QUrl &cwd)
460 {
461  // Save the original text
462  m_url = _url;
463 
464  // Non-const copy
465  QString url_copy = _url;
466 
467  // Special shortcuts for "man:" and "info:"
468  if (url_copy.startsWith(QLatin1Char('#'))) {
469  if (url_copy.length() > 1 && url_copy.at(1) == QLatin1Char('#')) {
470  url_copy.replace(0, 2, QStringLiteral("info:"));
471  } else {
472  url_copy.replace(0, 1, QStringLiteral("man:"));
473  }
474  }
475 
476  // Look for a protocol in 'url'
477  QRegExp protocol_regex = QRegExp(QStringLiteral("^(?![A-Za-z]:)[^/\\s\\\\]*:"));
478 
479  // Assume "file:" or whatever is given by 'cwd' if there is
480  // no protocol. (QUrl does this only for absolute paths)
481  if (protocol_regex.indexIn(url_copy) == 0) {
482  m_kurl = QUrl(url_copy);
483  m_isURL = true;
484  } else { // relative path or ~ or $something
485  m_isURL = false;
486  if (!QDir::isRelativePath(url_copy) ||
487  url_copy.startsWith(QLatin1Char('~')) ||
488  url_copy.startsWith(QLatin1Char('$'))) {
489  m_kurl = QUrl::fromLocalFile(url_copy);
490  } else {
491  // Relative path
492  if (cwd.isEmpty()) {
493  m_kurl = QUrl(url_copy);
494  } else {
495  m_kurl = cwd;
496  m_kurl.setPath(m_kurl.path() + '/' + url_copy);
497  }
498  }
499  }
500 }

603 /*
604  * makeCompletion()
605  *
606  * Entry point for file name completion
607  */
608 QString KUrlCompletion::makeCompletion(const QString &text)
609 {
610  qCDebug(KIO_WIDGETS) << text << "d->cwd=" << d->cwd;
611 
612  KUrlCompletionPrivate::MyURL url(text, d->cwd);
613 
614  d->compl_text = text;
615 
616  // Set d->prepend to the original URL, with the filename [and ref/query] stripped.
617  // This is what gets prepended to the directory-listing matches.
618  if (url.isURL()) {
619  QUrl directoryUrl(url.kurl());
620  directoryUrl.setQuery(QString());
621  directoryUrl.setFragment(QString());
622  directoryUrl.setPath(url.dir());
623  d->prepend = directoryUrl.toString();
624  } else {
625  d->prepend = text.left(text.length() - url.file().length());
626  }
627 
628  d->complete_url = url.isURL();
629 
630  QString aMatch;
631 
632  // Environment variables
633  //
634  if (d->replace_env && d->envCompletion(url, &aMatch)) {
635  return aMatch;
636  }
637 
638  // User directories
639  //
640  if (d->replace_home && d->userCompletion(url, &aMatch)) {
641  return aMatch;
642  }
643 
644  // Replace user directories and variables
645  url.filter(d->replace_home, d->replace_env);
646 
647  //qDebug() << "Filtered: proto=" << url.scheme()
648  // << ", dir=" << url.dir()
649  // << ", file=" << url.file()
650  // << ", kurl url=" << *url.kurl();
651 
652  if (d->mode == ExeCompletion) {
653  // Executables
654  //
655  if (d->exeCompletion(url, &aMatch)) {
656  return aMatch;
657  }
658 
659  // KRun can run "man:" and "info:" etc. so why not treat them
660  // as executables...
661 
662  if (d->urlCompletion(url, &aMatch)) {
663  return aMatch;
664  }
665  } else {
666  // Local files, directories
667  //
668  if (d->fileCompletion(url, &aMatch)) {
669  return aMatch;
670  }
671 
672  // All other...
673  //
674  if (d->urlCompletion(url, &aMatch)) {
675  return aMatch;
676  }
677  }
678 
679  d->setListedUrl(CTNone);
680  stop();
681 
682  return QString();
683 }

750 bool KUrlCompletionPrivate::isListedUrl(ComplType complType,
751  const QString &directory,
752  const QString &filter,
753  bool no_hidden)
754 {
755  return last_compl_type == complType
756  && (last_path_listed == directory
757  || (directory.isEmpty() && last_path_listed.isEmpty()))
758  && (filter.startsWith(last_file_listed)

1403 /*
1404  * expandEnv
1405  *
1406  * Expand environment variables in text. Escaped '$' are ignored.
1407  * Return true if expansion was made.
1408  */
1409 static bool expandEnv(QString &text)
1410 {
1411  // Find all environment variables beginning with '$'
1412  //
1413  int pos = 0;
1414 
1415  bool expanded = false;
1416 
1417  while ((pos = text.indexOf(QLatin1Char('$'), pos)) != -1) {
1418 
1419  // Skip escaped '$'
1420  //
1421  if (pos > 0 && text.at(pos - 1) == QLatin1Char('\\')) {
1422  pos++;
1423  }
1424  // Variable found => expand
1425  //
1426  else {
1427  // Find the end of the variable = next '/' or ' '
1428  //
1429  int pos2 = text.indexOf(QLatin1Char(' '), pos + 1);
1430  int pos_tmp = text.indexOf(QLatin1Char('/'), pos + 1);
1431 
1432  if (pos2 == -1 || (pos_tmp != -1 && pos_tmp < pos2)) {
1433  pos2 = pos_tmp;
1434  }
1435 
1436  if (pos2 == -1) {
1437  pos2 = text.length();
1438  }
1439 
1440  // Replace if the variable is terminated by '/' or ' '
1441  // and defined
1442  //
1443  if (pos2 >= 0) {
1444  int len = pos2 - pos;
1445  QString key = text.mid(pos + 1, len - 1);
1446  QString value =
1447  QString::fromLocal8Bit(qgetenv(key.toLocal8Bit()));
1448 
1449  if (!value.isEmpty()) {
1450  expanded = true;
1451  text.replace(pos, len, value);
1452  pos = pos + value.length();
1453  } else {
1454  pos = pos2;
1455  }
1456  }
1457  }
1458  }
1459 
1460  return expanded;
1461 }

1463 /*
1464  * expandTilde
1465  *
1466  * Replace "~user" with the users home directory
1467  * Return true if expansion was made.
1468  */
1469 static bool expandTilde(QString &text)
1470 {
1471  if (text.isEmpty() || (text.at(0) != QLatin1Char('~'))) {
1472  return false;
1473  }
1474 
1475  bool expanded = false;
1476 
1477  // Find the end of the user name = next '/' or ' '
1478  //
1479  int pos2 = text.indexOf(QLatin1Char(' '), 1);
1480  int pos_tmp = text.indexOf(QLatin1Char('/'), 1);
1481 
1482  if (pos2 == -1 || (pos_tmp != -1 && pos_tmp < pos2)) {
1483  pos2 = pos_tmp;
1484  }
1485 
1486  if (pos2 == -1) {
1487  pos2 = text.length();
1488  }
1489 
1490  // Replace ~user if the user name is terminated by '/' or ' '
1491  //
1492  if (pos2 >= 0) {
1493 
1494  QString userName = text.mid(1, pos2 - 1);
1495  QString dir;
1496 
1497  // A single ~ is replaced with $HOME
1498  //
1499  if (userName.isEmpty()) {
1500  dir = QDir::homePath();
1501  }
1502  // ~user is replaced with the dir from passwd
1503  //
1504  else {
1505  KUser user(userName);
1506  dir = user.homeDir();
1507  }
