253 QStringList KIO::Job::detailedErrorStrings(const QUrl *reqUrl /*= 0*/,
254  int method /*= -1*/) const
255 {
256  QString errorName, techName, description, ret2;
257  QStringList causes, solutions, ret;
258 
259  QByteArray raw = rawErrorDetail(error(), errorText(), reqUrl, method);
260  QDataStream stream(raw);
261 
262  stream >> errorName >> techName >> description >> causes >> solutions;
263 
264  QString url, protocol, datetime;
265  if (reqUrl) {
266  QString prettyUrl;
267  prettyUrl = reqUrl->toDisplayString();
268  url = prettyUrl.toHtmlEscaped();
269  protocol = reqUrl->scheme();
270  } else {
271  url = i18nc("@info url", "(unknown)");
272  }
273 
274  datetime = QDateTime::currentDateTime().toString(Qt::DefaultLocaleLongDate);
275 
276  ret << errorName;
277  ret << i18nc("@info %1 error name, %2 description",
278  "<qt><p><b>%1</b></p><p>%2</p></qt>", errorName, description);
279 
280  ret2 = QStringLiteral("<qt>");
281  if (!techName.isEmpty())
282  ret2 += QLatin1String("<p>") + i18n("<b>Technical reason</b>: ") +
283  techName + QLatin1String("</p>");
284  ret2 += QLatin1String("<p>") + i18n("<b>Details of the request</b>:") +
285  QLatin1String("</p><ul>") + i18n("<li>URL: %1</li>", url);
286  if (!protocol.isEmpty()) {
287  ret2 += i18n("<li>Protocol: %1</li>", protocol);
288  }
289  ret2 += i18n("<li>Date and time: %1</li>", datetime) +
290  i18n("<li>Additional information: %1</li>", errorText()) +
291  QLatin1String("</ul>");
292  if (!causes.isEmpty()) {
293  ret2 += QLatin1String("<p>") + i18n("<b>Possible causes</b>:") +
294  QLatin1String("</p><ul><li>") + causes.join(QStringLiteral("</li><li>")) +
295  QLatin1String("</li></ul>");
296  }
297  if (!solutions.isEmpty()) {
298  ret2 += QLatin1String("<p>") + i18n("<b>Possible solutions</b>:") +
299  QLatin1String("</p><ul><li>") + solutions.join(QStringLiteral("</li><li>")) +
300  QLatin1String("</li></ul>");
301  }
302  ret2 += QLatin1String("</qt>");
303  ret << ret2;
304 
305  return ret;
306 }

308 KIOCORE_EXPORT QByteArray KIO::rawErrorDetail(int errorCode, const QString &errorText,
309  const QUrl *reqUrl /*= 0*/, int /*method = -1*/)
310 {
311  QString url, host, protocol, datetime, domain, path, filename;
312  bool isSlaveNetwork = false;
313  if (reqUrl) {
314  url = reqUrl->toDisplayString();
315  host = reqUrl->host();
316  protocol = reqUrl->scheme();
317 
318  if (host.startsWith(QLatin1String("www."))) {
319  domain = host.mid(4);
320  } else {
321  domain = host;
322  }
323 
324  filename = reqUrl->fileName();
325  path = reqUrl->path();
326 
327  // detect if protocol is a network protocol...
328  isSlaveNetwork = KProtocolInfo::protocolClass(protocol) == QLatin1String(":internet");
329  } else {
330  // assume that the errorText has the location we are interested in
331  url = host = domain = path = filename = errorText;
332  protocol = i18nc("@info protocol", "(unknown)");
333  }
334 
335  datetime = QDateTime::currentDateTime().toString(Qt::DefaultLocaleLongDate);
336 
337  QString errorName, techName, description;

1070 QFile::Permissions KIO::convertPermissions(int permissions)
1071 {
1072  QFile::Permissions qPermissions(0);
1073 
1074  if (permissions > 0) {
1075  if (permissions & S_IRUSR) {
1076  qPermissions |= QFile::ReadOwner;
1077  }
1078  if (permissions & S_IWUSR) {
1079  qPermissions |= QFile::WriteOwner;
1080  }
1081  if (permissions & S_IXUSR) {
1082  qPermissions |= QFile::ExeOwner;
1083  }
1084 
1085  if (permissions & S_IRGRP) {
1086  qPermissions |= QFile::ReadGroup;
1087  }
1088  if (permissions & S_IWGRP) {
1089  qPermissions |= QFile::WriteGroup;
1090  }
1091  if (permissions & S_IXGRP) {
1092  qPermissions |= QFile::ExeGroup;
1093  }
1094 
1095  if (permissions & S_IROTH) {
1096  qPermissions |= QFile::ReadOther;
1097  }
1098  if (permissions & S_IWOTH) {
1099  qPermissions |= QFile::WriteOther;
1100  }
1101  if (permissions & S_IXOTH) {
1102  qPermissions |= QFile::ExeOther;
1103  }
1104  }
1105 
1106  return qPermissions;
1107 }
