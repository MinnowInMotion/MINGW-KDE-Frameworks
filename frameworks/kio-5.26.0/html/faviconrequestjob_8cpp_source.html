44 static bool isIconOld(const QString &icon)
45 {
46  const QFileInfo info(icon);
47  if (!info.exists()) {
48  qCDebug(FAVICONS_LOG) << "isIconOld" << icon << "yes, no such file";
49  return true; // Trigger a new download on error
50  }
51  const QDate date = info.lastModified().date();
52 
53  qCDebug(FAVICONS_LOG) << "isIconOld" << icon << "?";
54  return date.daysTo(QDate::currentDate()) > 7; // arbitrary value (one week)
55 }

100 void FavIconRequestJob::doStart()
101 {
102  KIO::FavIconsCache *cache = KIO::FavIconsCache::instance();
103  QUrl iconUrl = d->m_iconUrl;
104  const bool isNewIconUrl = !iconUrl.isEmpty();
105  if (isNewIconUrl) {
106  cache->setIconForUrl(d->m_hostUrl, d->m_iconUrl);
107  } else {
108  iconUrl = cache->iconUrlForUrl(d->m_hostUrl);
109  }
110  if (d->m_reload == NoReload) {
111  const QString iconFile = cache->cachePathForIconUrl(iconUrl);
112  if (!isIconOld(iconFile)) {
113  qCDebug(FAVICONS_LOG) << "existing icon not old, reload not requested -> doing nothing";
114  d->m_iconFile = iconFile;
115  emitResult();
116  return;
117  }
118 
119  if (cache->isFailedDownload(iconUrl)) {
120  qCDebug(FAVICONS_LOG) << iconUrl << "already in failedDownloads, emitting error";
121  setError(KIO::ERR_DOES_NOT_EXIST);
122  setErrorText(i18n("No favicon found for %1", d->m_hostUrl.host()));
123  emitResult();
124  return;
125  }
126  }
127 
128  qCDebug(FAVICONS_LOG) << "downloading" << iconUrl;
129  KIO::TransferJob *job = KIO::get(iconUrl, d->m_reload, KIO::HideProgressInfo);
130  QMap<QString, QString> metaData;
131  metaData.insert(QStringLiteral("ssl_no_client_cert"), QStringLiteral("true"));
132  metaData.insert(QStringLiteral("ssl_no_ui"), QStringLiteral("true"));
133  metaData.insert(QStringLiteral("UseCache"), QStringLiteral("false"));
134  metaData.insert(QStringLiteral("cookies"), QStringLiteral("none"));
135  metaData.insert(QStringLiteral("no-www-auth"), QStringLiteral("true"));
136  metaData.insert(QStringLiteral("errorPage"), QStringLiteral("false"));
137  job->addMetaData(metaData);
138  QObject::connect(job, SIGNAL(data(KIO::Job *, QByteArray)), this, SLOT(slotData(KIO::Job *, QByteArray)));
139  addSubjob(job);
140 }

142 void FavIconRequestJob::slotResult(KJob *job)
143 {
144  KIO::TransferJob *tjob = static_cast<KIO::TransferJob *>(job);
145  const QUrl iconUrl = tjob->url();
146  KIO::FavIconsCache *cache = KIO::FavIconsCache::instance();
147  if (!job->error()) {
148  QBuffer buffer(&d->m_iconData);
149  buffer.open(QIODevice::ReadOnly);
150  QImageReader ir(&buffer);
151  QSize desired(16, 16);
152  if (ir.canRead()) {
153  while (ir.imageCount() > 1
154  && ir.currentImageRect() != QRect(0, 0, desired.width(), desired.height())) {
155  if (!ir.jumpToNextImage()) {
156  break;
157  }
158  }
159  ir.setScaledSize(desired);
160  const QImage img = ir.read();
161  if (!img.isNull()) {
162  cache->ensureCacheExists();
163  const QString localPath = cache->cachePathForIconUrl(iconUrl);
164  qCDebug(FAVICONS_LOG) << "Saving image to" << localPath;
165  QSaveFile saveFile(localPath);
166  if (saveFile.open(QIODevice::WriteOnly) &&
167  img.save(&saveFile, "PNG") && saveFile.commit()) {
168  d->m_iconFile = localPath;
169  } else {
170  setError(KIO::ERR_COULD_NOT_WRITE);
171  setErrorText(i18n("Error saving image to %1", localPath));
172  }
173  } else {
174  qCDebug(FAVICONS_LOG) << "QImageReader read() returned a null image";
175  }
176  } else {
177  qCDebug(FAVICONS_LOG) << "QImageReader canRead returned false";
178  }
179  } else if (job->error() == KJob::KilledJobError) { // we killed it in slotData
180  setError(KIO::ERR_SLAVE_DEFINED);
181  setErrorText(i18n("Icon file too big, download aborted"));
182  } else {
183  setError(job->error());
184  setErrorText(job->errorString()); // not errorText(), because "this" is a KJob, with no errorString building logic
185  }
186  d->m_iconData.clear(); // release memory
187  if (d->m_iconFile.isEmpty()) {
188  qCDebug(FAVICONS_LOG) << "adding" << iconUrl << "to failed downloads due to error:" << errorString();
189  cache->addFailedDownload(iconUrl);
190  } else {
191  cache->removeFailedDownload(iconUrl);
192  }
193  KCompositeJob::removeSubjob(job);
194  emitResult();
195 }

197 void FavIconRequestJobPrivate::slotData(Job *job, const QByteArray &data)
198 {
199  KIO::TransferJob *tjob = static_cast<KIO::TransferJob *>(job);
200  unsigned int oldSize = m_iconData.size();
201  // Size limit. Stop downloading if the file is huge.
202  // Testcase (as of june 2008, at least): http://planet-soc.com/favicon.ico, 136K and strange format.
203  // Another case: sites which redirect from "/favicon.ico" to "/" and return the main page.
204  if (oldSize > 0x10000) { // 65K
205  qCDebug(FAVICONS_LOG) << "Favicon too big, aborting download of" << tjob->url();
206  const QUrl iconUrl = tjob->url();
207  KIO::FavIconsCache::instance()->addFailedDownload(iconUrl);
208  tjob->kill(KJob::EmitResult);
209  } else {
210  m_iconData.resize(oldSize + data.size());
211  memcpy(m_iconData.data() + oldSize, data.data(), data.size());
212  }
213 }
