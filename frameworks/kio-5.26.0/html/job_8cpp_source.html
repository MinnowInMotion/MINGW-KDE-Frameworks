76 bool Job::addSubjob(KJob *jobBase)
77 {
78  //qDebug() << "addSubjob(" << jobBase << ") this=" << this;
79 
80  bool ok = KCompositeJob::addSubjob(jobBase);
81  KIO::Job *job = dynamic_cast<KIO::Job *>(jobBase);
82  if (ok && job) {
83  // Copy metadata into subjob (e.g. window-id, user-timestamp etc.)
84  Q_D(Job);
85  job->mergeMetaData(d->m_outgoingMetaData);
86 
87  // Forward information from that subjob.
88  connect(job, SIGNAL(speed(KJob*,ulong)),
89  SLOT(slotSpeed(KJob*,ulong)));
90 
91  job->setProperty("window", property("window")); // see KJobWidgets
92  job->setProperty("userTimestamp", property("userTimestamp")); // see KJobWidgets
93  job->setUiDelegateExtension(d->m_uiDelegateExtension);
94  }
95  return ok;
96 }

159 bool Job::doKill()
160 {
161  // kill all subjobs, without triggering their result slot
162  Q_FOREACH (KJob *it, subjobs()) {
163  it->kill(KJob::Quietly);
164  }
165  clearSubjobs();
166 
167  return true;
168 }

243 void Job::mergeMetaData(const QMap<QString, QString> &values)
244 {
245  Q_D(Job);
246  QMap<QString, QString>::const_iterator it = values.begin();
247  for (; it != values.end(); ++it)
248  // there's probably a faster way
249  if (!d->m_outgoingMetaData.contains(it.key())) {
250  d->m_outgoingMetaData.insert(it.key(), it.value());
251  }
252 }
