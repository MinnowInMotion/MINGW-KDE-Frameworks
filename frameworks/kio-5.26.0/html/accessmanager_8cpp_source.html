63  return -1;
64  }

213 QNetworkReply *AccessManager::createRequest(Operation op, const QNetworkRequest &req, QIODevice *outgoingData)
214 {
215  const QUrl reqUrl(req.url());
216 
217  if (!d->externalContentAllowed &&
218  !KDEPrivate::AccessManagerReply::isLocalRequest(reqUrl) &&
219  reqUrl.scheme() != QL1S("data")) {
220  //qDebug() << "Blocked: " << reqUrl;
221  return new KDEPrivate::AccessManagerReply(op, req, QNetworkReply::ContentAccessDenied, i18n("Blocked request."), this);
222  }
223 
224  // Check if the internal ignore content disposition header is set.
225  const bool ignoreContentDisposition = req.hasRawHeader("x-kdewebkit-ignore-disposition");
226 
227  // Retrieve the KIO meta data...
228  KIO::MetaData metaData;
229  d->setMetaDataForRequest(req, metaData);
230 
231  KIO::SimpleJob *kioJob = 0;
232 
233  switch (op) {
234  case HeadOperation: {
235  //qDebug() << "HeadOperation:" << reqUrl;
236  kioJob = KIO::mimetype(reqUrl, KIO::HideProgressInfo);
237  break;
238  }
239  case GetOperation: {
240  //qDebug() << "GetOperation:" << reqUrl;
241  if (!reqUrl.path().isEmpty() || reqUrl.host().isEmpty()) {
242  kioJob = KIO::storedGet(reqUrl, KIO::NoReload, KIO::HideProgressInfo);
243  } else {
244  kioJob = KIO::stat(reqUrl, KIO::HideProgressInfo);
245  }
246 
247  // WORKAROUND: Avoid the brain damaged stuff QtWebKit does when a POST
248  // operation is redirected! See BR# 268694.
249  metaData.remove(QStringLiteral("content-type")); // Remove the content-type from a GET/HEAD request!
250  break;
251  }
252  case PutOperation: {
253  //qDebug() << "PutOperation:" << reqUrl;
254  if (outgoingData) {
255  Q_ASSERT(outgoingData->isReadable());
256  StoredTransferJob* storedJob = KIO::storedPut(outgoingData, reqUrl, -1, KIO::HideProgressInfo);
257  storedJob->setAsyncDataEnabled(outgoingData->isSequential());
258 
259  QVariant len = req.header(QNetworkRequest::ContentLengthHeader);
260  if (len.isValid()) {
261  storedJob->setTotalSize(len.toInt());
262  }
263 
264  kioJob = storedJob;
265  } else {
266  kioJob = KIO::put(reqUrl, -1, KIO::HideProgressInfo);
267  }
268  break;
269  }
270  case PostOperation: {
271  kioJob = KIO::storedHttpPost(outgoingData, reqUrl, sizeFromRequest(req), KIO::HideProgressInfo);
272  if (!metaData.contains(QStringLiteral("content-type"))) {
273  const QVariant header = req.header(QNetworkRequest::ContentTypeHeader);
274  if (header.isValid()) {
275  metaData.insert(QStringLiteral("content-type"),
276  (QStringLiteral("Content-Type: ") + header.toString()));
277  } else {
278  metaData.insert(QStringLiteral("content-type"),
279  QStringLiteral("Content-Type: application/x-www-form-urlencoded"));
280  }
281  }
282  break;
283  }
284  case DeleteOperation: {
285  //qDebug() << "DeleteOperation:" << reqUrl;
286  kioJob = KIO::http_delete(reqUrl, KIO::HideProgressInfo);
287  break;
288  }
289  case CustomOperation: {
290  const QByteArray &method = req.attribute(QNetworkRequest::CustomVerbAttribute).toByteArray();
291  //qDebug() << "CustomOperation:" << reqUrl << "method:" << method << "outgoing data:" << outgoingData;
292 
293  if (method.isEmpty()) {
294  return new KDEPrivate::AccessManagerReply(op, req, QNetworkReply::ProtocolUnknownError, i18n("Unknown HTTP verb."), this);
295  }
296 
297  const qint64 size = sizeFromRequest(req);
298  if (size > 0) {
299  kioJob = KIO::http_post(reqUrl, outgoingData, size, KIO::HideProgressInfo);
300  } else {
301  kioJob = KIO::get(reqUrl, KIO::NoReload, KIO::HideProgressInfo);
302  }
303 
304  metaData.insert(QStringLiteral("CustomHTTPMethod"), method);
305  break;
306  }
307  default: {
308  qWarning() << "Unsupported KIO operation requested! Defering to QNetworkAccessManager...";
309  return QNetworkAccessManager::createRequest(op, req, outgoingData);
310  }
311  }
312 
313  // Set the job priority
314  switch (req.priority()) {
315  case QNetworkRequest::HighPriority:
316  KIO::Scheduler::setJobPriority(kioJob, -5);
317  break;
318  case QNetworkRequest::LowPriority:
319  KIO::Scheduler::setJobPriority(kioJob, 5);
320  break;
321  default:
322  break;
323  }
324 
325  KDEPrivate::AccessManagerReply *reply;
326 
327  /*
328  NOTE: Here we attempt to handle synchronous XHR requests. Unfortunately,
329  due to the fact that QNAM is both synchronous and multi-thread while KIO
330  is completely the opposite (asynchronous and not thread safe), the code
331  below might cause crashes like the one reported in bug# 287778 (nested
332  event loops are inherently dangerous).
333 
334  Unfortunately, all attempts to address the crash has so far failed due to
335  the many regressions they caused, e.g. bug# 231932 and 297954. Hence, until
336  a solution is found, we have to live with the side effects of creating
337  nested event loops.
338  */
339  if (req.attribute(gSynchronousNetworkRequestAttribute).toBool()) {
340  KJobWidgets::setWindow(kioJob, d->window);
341  kioJob->setRedirectionHandlingEnabled(true);
342  if (kioJob->exec()) {
343  QByteArray data;
344  if (StoredTransferJob *storedJob = qobject_cast< KIO::StoredTransferJob * >(kioJob)) {
345  data = storedJob->data();
346  }
347  reply = new KDEPrivate::AccessManagerReply(op, req, data, kioJob->url(), kioJob->metaData(), this);
348  //qDebug() << "Synchronous XHR:" << reply << reqUrl;
349  } else {
350  qWarning() << "Failed to create a synchronous XHR for" << reqUrl;
351  qWarning() << "REASON:" << kioJob->errorString();
352  reply = new KDEPrivate::AccessManagerReply(op, req, QNetworkReply::UnknownNetworkError, kioJob->errorText(), this);
353  }
354  } else {
355  // Set the window on the KIO ui delegate
356  if (d->window) {
357  KJobWidgets::setWindow(kioJob, d->window);
358  }
359 
360  // Disable internal automatic redirection handling
361  kioJob->setRedirectionHandlingEnabled(false);
362 
363  // Set the job priority
364  switch (req.priority()) {
365  case QNetworkRequest::HighPriority:
366  KIO::Scheduler::setJobPriority(kioJob, -5);
367  break;
368  case QNetworkRequest::LowPriority:
369  KIO::Scheduler::setJobPriority(kioJob, 5);
370  break;
371  default:
372  break;
373  }
374 
375  // Set the meta data for this job...
376  kioJob->setMetaData(metaData);
377 
378  // Create the reply...
379  reply = new KDEPrivate::AccessManagerReply(op, req, kioJob, d->emitReadyReadOnMetaDataChange, this);
380  //qDebug() << reply << reqUrl;
381  }
382 
383  if (ignoreContentDisposition && reply) {
384  //qDebug() << "Content-Disposition WILL BE IGNORED!";
385  reply->setIgnoreContentDisposition(ignoreContentDisposition);
386  }
387 
388  return reply;
389 }

391 void AccessManager::AccessManagerPrivate::setMetaDataForRequest(QNetworkRequest request, KIO::MetaData &metaData)
392 {
393  // Add any meta data specified within request...
394  QVariant userMetaData = request.attribute(static_cast<QNetworkRequest::Attribute>(MetaData));
395  if (userMetaData.isValid() && userMetaData.type() == QVariant::Map) {
396  metaData += userMetaData.toMap();
397  }
398 
399  metaData.insert(QStringLiteral("PropagateHttpHeader"), QStringLiteral("true"));
400 
401  if (request.hasRawHeader("User-Agent")) {
402  metaData.insert(QStringLiteral("UserAgent"), request.rawHeader("User-Agent"));
403  request.setRawHeader("User-Agent", QByteArray());
404  }
405 
406  if (request.hasRawHeader("Accept")) {
407  metaData.insert(QStringLiteral("accept"), request.rawHeader("Accept"));
408  request.setRawHeader("Accept", QByteArray());
409  }
410 
411  if (request.hasRawHeader("Accept-Charset")) {
412  metaData.insert(QStringLiteral("Charsets"), request.rawHeader("Accept-Charset"));
413  request.setRawHeader("Accept-Charset", QByteArray());
414  }
415 
416  if (request.hasRawHeader("Accept-Language")) {
417  metaData.insert(QStringLiteral("Languages"), request.rawHeader("Accept-Language"));
418  request.setRawHeader("Accept-Language", QByteArray());
419  }
420 
421  if (request.hasRawHeader("Referer")) {
422  metaData.insert(QStringLiteral("referrer"), request.rawHeader("Referer"));
423  request.setRawHeader("Referer", QByteArray());
424  }
425 
426  if (request.hasRawHeader("Content-Type")) {
427  metaData.insert(QStringLiteral("content-type"), request.rawHeader("Content-Type"));
428  request.setRawHeader("Content-Type", QByteArray());
429  }
430 
431  if (request.attribute(QNetworkRequest::AuthenticationReuseAttribute) == QNetworkRequest::Manual) {
432  metaData.insert(QStringLiteral("no-preemptive-auth-reuse"), QStringLiteral("true"));
433  }
434 
435  request.setRawHeader("Content-Length", QByteArray());
436  request.setRawHeader("Connection", QByteArray());
437  request.setRawHeader("If-None-Match", QByteArray());
438  request.setRawHeader("If-Modified-Since", QByteArray());
439  request.setRawHeader("x-kdewebkit-ignore-disposition", QByteArray());
440 
441  QStringList customHeaders;
442  Q_FOREACH (const QByteArray &key, request.rawHeaderList()) {
443  const QByteArray value = request.rawHeader(key);
444  if (value.length()) {
445  customHeaders << (key + QStringLiteral(": ") + value);
446  }
447  }
448 
449  if (!customHeaders.isEmpty()) {
450  metaData.insert(QStringLiteral("customHTTPHeader"), customHeaders.join(QStringLiteral("\r\n")));
451  }
452 
453  // Append per request meta data, if any...
454  if (!requestMetaData.isEmpty()) {
455  metaData += requestMetaData;
456  // Clear per request meta data...
457  requestMetaData.clear();
458  }
459 
460  // Append per session meta data, if any...
461  if (!sessionMetaData.isEmpty()) {
462  metaData += sessionMetaData;
463  }
464 }

468 static QSsl::SslProtocol qSslProtocolFromString(const QString &str)
469 {
470  if (str.compare(QStringLiteral("SSLv3"), Qt::CaseInsensitive) == 0) {

485 bool KIO::Integration::sslConfigFromMetaData(const KIO::MetaData &metadata, QSslConfiguration &sslconfig)
486 {
487  bool success = false;
488 
489  if (metadata.value(QStringLiteral("ssl_in_use")) == QStringLiteral("TRUE")) {
490  const QSsl::SslProtocol sslProto = qSslProtocolFromString(metadata.value(QStringLiteral("ssl_protocol_version")));
491  QList<QSslCipher> cipherList;
492  cipherList << QSslCipher(metadata.value(QStringLiteral("ssl_cipher_name")), sslProto);
493  sslconfig.setCaCertificates(QSslCertificate::fromData(metadata.value(QStringLiteral("ssl_peer_chain")).toUtf8()));
494  sslconfig.setCiphers(cipherList);
495  sslconfig.setProtocol(sslProto);
496  success = sslconfig.isNull();
497  }
498 
499  return success;

523 QList<QNetworkCookie> CookieJar::cookiesForUrl(const QUrl &url) const
524 {
525  QList<QNetworkCookie> cookieList;
526 
527  if (!d->isEnabled) {
528  return cookieList;
529  }
530  QDBusInterface kcookiejar(QStringLiteral("org.kde.kcookiejar5"), QStringLiteral("/modules/kcookiejar"), QStringLiteral("org.kde.KCookieServer"));
531  QDBusReply<QString> reply = kcookiejar.call(QStringLiteral("findDOMCookies"), url.toString(QUrl::RemoveUserInfo), (qlonglong)d->windowId);
532 
533  if (!reply.isValid()) {
534  qWarning() << "Unable to communicate with the cookiejar!";
535  return cookieList;
536  }
537 
538  const QString cookieStr = reply.value();
539  const QStringList cookies = cookieStr.split(QStringLiteral("; "), QString::SkipEmptyParts);
540  Q_FOREACH (const QString &cookie, cookies) {
541  const int index = cookie.indexOf(QL1C('='));
542  const QString name = cookie.left(index);
543  const QString value = cookie.right((cookie.length() - index - 1));
544  cookieList << QNetworkCookie(name.toUtf8(), value.toUtf8());
545  //qDebug() << "cookie: name=" << name << ", value=" << value;
546  }
547 
548  return cookieList;
549 }

551 bool CookieJar::setCookiesFromUrl(const QList<QNetworkCookie> &cookieList, const QUrl &url)
552 {
553  if (!d->isEnabled) {
554  return false;
555  }
556 
557  QDBusInterface kcookiejar(QStringLiteral("org.kde.kcookiejar5"), QStringLiteral("/modules/kcookiejar"), QStringLiteral("org.kde.KCookieServer"));
558  Q_FOREACH (const QNetworkCookie &cookie, cookieList) {
559  QByteArray cookieHeader("Set-Cookie: ");
560  if (d->isStorageDisabled && !cookie.isSessionCookie()) {
561  QNetworkCookie sessionCookie(cookie);
562  sessionCookie.setExpirationDate(QDateTime());
563  cookieHeader += sessionCookie.toRawForm();
564  } else {
565  cookieHeader += cookie.toRawForm();
566  }
567  kcookiejar.call(QStringLiteral("addCookies"), url.toString(QUrl::RemoveUserInfo), cookieHeader, (qlonglong)d->windowId);
568  //qDebug() << "[" << d->windowId << "]" << cookieHeader << " from " << url;
569  }
570 
571  return !kcookiejar.lastError().isValid();
572 }
