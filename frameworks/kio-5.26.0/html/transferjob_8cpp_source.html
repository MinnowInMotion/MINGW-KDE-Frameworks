65 // Slave got a redirection request
66 void TransferJob::slotRedirection(const QUrl &url)
67 {
68  Q_D(TransferJob);
69  //qDebug() << url;
70  if (!KUrlAuthorized::authorizeUrlAction(QStringLiteral("redirect"), d->m_url, url)) {
71  qWarning() << "Redirection from" << d->m_url << "to" << url << "REJECTED!";
72  return;
73  }
74 
75  // Some websites keep redirecting to themselves where each redirection
76  // acts as the stage in a state-machine. We define "endless redirections"
77  // as 5 redirections to the same URL.
78  if (d->m_redirectionList.count(url) > 5) {
79  //qDebug() << "CYCLIC REDIRECTION!";
80  setError(ERR_CYCLIC_LINK);
81  setErrorText(d->m_url.toDisplayString());
82  } else {
83  d->m_redirectionURL = url; // We'll remember that when the job finishes
84  d->m_redirectionList.append(url);
85  QString sslInUse = queryMetaData(QStringLiteral("ssl_in_use"));
86  if (!sslInUse.isNull()) { // the key is present
87  addMetaData(QStringLiteral("ssl_was_in_use"), sslInUse);
88  } else {
89  addMetaData(QStringLiteral("ssl_was_in_use"), QStringLiteral("FALSE"));
90  }
91  // Tell the user that we haven't finished yet
92  emit redirection(this, d->m_redirectionURL);
93  }
94 }

96 void TransferJob::slotFinished()
97 {
98  Q_D(TransferJob);
99 
100  //qDebug() << d->m_url;
101  if (!d->m_redirectionURL.isEmpty() && d->m_redirectionURL.isValid()) {
102 
103  //qDebug() << "Redirection to" << m_redirectionURL;
104  if (queryMetaData(QStringLiteral("permanent-redirect")) == QLatin1String("true")) {
105  emit permanentRedirection(this, d->m_url, d->m_redirectionURL);
106  }
107 
108  if (queryMetaData(QStringLiteral("redirect-to-get")) == QLatin1String("true")) {
109  d->m_command = CMD_GET;
110  d->m_outgoingMetaData.remove(QStringLiteral("CustomHTTPMethod"));
111  d->m_outgoingMetaData.remove(QStringLiteral("content-type"));
112  }
113 
114  if (d->m_redirectionHandlingEnabled) {
115  // Honour the redirection
116  // We take the approach of "redirecting this same job"
117  // Another solution would be to create a subjob, but the same problem
118  // happens (unpacking+repacking)
119  d->staticData.truncate(0);
120  d->m_incomingMetaData.clear();
121  if (queryMetaData(QStringLiteral("cache")) != QLatin1String("reload")) {
122  addMetaData(QStringLiteral("cache"), QStringLiteral("refresh"));
123  }
124  d->m_internalSuspended = false;
125  // The very tricky part is the packed arguments business
126  QString dummyStr;
127  QUrl dummyUrl;
128  QDataStream istream(d->m_packedArgs);
129  switch (d->m_command) {
130  case CMD_GET:
131  case CMD_STAT:
132  case CMD_DEL: {
133  d->m_packedArgs.truncate(0);
134  QDataStream stream(&d->m_packedArgs, QIODevice::WriteOnly);
135  stream << d->m_redirectionURL;
136  break;
137  }
138  case CMD_PUT: {
139  int permissions;
140  qint8 iOverwrite, iResume;
141  istream >> dummyUrl >> iOverwrite >> iResume >> permissions;
142  d->m_packedArgs.truncate(0);
143  QDataStream stream(&d->m_packedArgs, QIODevice::WriteOnly);
144  stream << d->m_redirectionURL << iOverwrite << iResume << permissions;
145  break;
146  }
147  case CMD_SPECIAL: {
148  int specialcmd;
149  istream >> specialcmd;
150  if (specialcmd == 1) { // HTTP POST
151  d->m_outgoingMetaData.remove(QStringLiteral("content-type"));
152  addMetaData(QStringLiteral("cache"), QStringLiteral("reload"));
153  d->m_packedArgs.truncate(0);
154  QDataStream stream(&d->m_packedArgs, QIODevice::WriteOnly);
155  stream << d->m_redirectionURL;
156  d->m_command = CMD_GET;
157  }
158  break;
159  }
160  }
161  d->restartAfterRedirection(&d->m_redirectionURL);
162  return;
163  }
164  }

166  SimpleJob::slotFinished();
167 }

222 // Slave requests data
223 void TransferJob::slotDataReq()
224 {
225  Q_D(TransferJob);
226  QByteArray dataForSlave;
227 
228  d->m_extraFlags |= JobPrivate::EF_TransferJobNeedData;
229 
230  if (!d->staticData.isEmpty()) {
231  dataForSlave = d->staticData;
232  d->staticData.clear();
233  } else {
234  emit dataReq(this, dataForSlave);
235 
236  if (d->m_extraFlags & JobPrivate::EF_TransferJobAsync) {
237  return;
238  }
239  }
240 
241  static const int max_size = 14 * 1024 * 1024;
242  if (dataForSlave.size() > max_size) {
243  //qDebug() << "send" << dataForSlave.size() / 1024 / 1024 << "MB of data in TransferJob::dataReq. This needs to be split, which requires a copy. Fix the application.";
244  d->staticData = QByteArray(dataForSlave.data() + max_size, dataForSlave.size() - max_size);
245  dataForSlave.truncate(max_size);
246  }
247 
248  sendAsyncData(dataForSlave);
249 
250  if (d->m_subJob) {
251  // Bitburger protocol in action
252  d->internalSuspend(); // Wait for more data from subJob.
253  d->m_subJob->d_func()->internalResume(); // Ask for more!
254  }
255 }

301 void TransferJobPrivate::start(Slave *slave)
302 {
303  Q_Q(TransferJob);
304  Q_ASSERT(slave);
305  JobPrivate::emitTransferring(q, m_url);
306  q->connect(slave, SIGNAL(data(QByteArray)),
307  SLOT(slotData(QByteArray)));
308 
309  if (m_outgoingDataSource) {
310  if (m_extraFlags & JobPrivate::EF_TransferJobAsync) {
311  q->connect(m_outgoingDataSource, SIGNAL(readyRead()),
312  SLOT(slotDataReqFromDevice()));
313  q->connect(m_outgoingDataSource, SIGNAL(readChannelFinished()),
314  SLOT(slotIODeviceClosed()));
315  if (m_outgoingDataSource->bytesAvailable() > 0) {
316  QMetaObject::invokeMethod(q, "slotDataReqFromDevice", Qt::QueuedConnection);
317  }
318  } else {
319  q->connect(slave, SIGNAL(dataReq()),
320  SLOT(slotDataReqFromDevice()));
321  }
322  } else
323  q->connect(slave, SIGNAL(dataReq()),
324  SLOT(slotDataReq()));
325 
326  q->connect(slave, SIGNAL(redirection(QUrl)),
327  SLOT(slotRedirection(QUrl)));
328 
329  q->connect(slave, SIGNAL(mimeType(QString)),
330  SLOT(slotMimetype(QString)));
331 
332  q->connect(slave, SIGNAL(errorPage()),
333  SLOT(slotErrorPage()));
334 
335  q->connect(slave, SIGNAL(needSubUrlData()),
336  SLOT(slotNeedSubUrlData()));
337 
338  q->connect(slave, SIGNAL(canResume(KIO::filesize_t)),
339  SLOT(slotCanResume(KIO::filesize_t)));
340 
341  if (slave->suspended()) {
342  m_mimetype = QStringLiteral("unknown");
343  // WABA: The slave was put on hold. Resume operation.
344  slave->resume();
345  }
346 
347  SimpleJobPrivate::start(slave);
348  if (m_internalSuspended) {
349  slave->suspend();
350  }
351 }

390 void TransferJobPrivate::slotDataReqFromDevice()
391 {
392  Q_Q(TransferJob);
393 
394  bool done = false;
395  QByteArray dataForSlave;
396 
397  m_extraFlags |= JobPrivate::EF_TransferJobNeedData;
398 
399  if (m_outgoingDataSource) {
400  dataForSlave.resize(MAX_READ_BUF_SIZE);
401 
402  //Code inspired in QNonContiguousByteDevice
403  qint64 bytesRead = m_outgoingDataSource->read(dataForSlave.data(), MAX_READ_BUF_SIZE);
404  if (bytesRead >= 0) {
405  dataForSlave.resize(bytesRead);
406  } else {
407  dataForSlave.clear();
408  }
409  done = ((bytesRead == -1) || (bytesRead == 0 && m_outgoingDataSource->atEnd() && !m_outgoingDataSource->isSequential()));
410 
411  if (!done) {
412  QMetaObject::invokeMethod(q, "slotDataReqFromDevice", Qt::QueuedConnection);
413  }
414  }
415 
416  if (dataForSlave.isEmpty()) {
417  emit q->dataReq(q, dataForSlave);
418  if (!done && (m_extraFlags & JobPrivate::EF_TransferJobAsync)) {
419  return;
420  }
421  }
422 
423  q->sendAsyncData(dataForSlave);
424 
425  if (m_subJob) {
426  // Bitburger protocol in action
427  internalSuspend(); // Wait for more data from subJob.
428  m_subJob->d_func()->internalResume(); // Ask for more!
429  }
430 }

436  if (!remainder.isEmpty()) {

439  }
440

442 
443  //We send an empty data array to indicate the stream is over
444  q->sendAsyncData(QByteArray());
445 
446  if (m_subJob) {
447  // Bitburger protocol in action
448  internalSuspend(); // Wait for more data from subJob.
449  m_subJob->d_func()->internalResume(); // Ask for more!
450  }
