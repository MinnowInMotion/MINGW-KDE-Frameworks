295 bool NetRC::lookup(const QUrl &url, AutoLogin &login, bool userealnetrc,
296  const QString &_type, LookUpMode mode)
297 {
298  //qDebug() << "AutoLogin lookup for: " << url.host();
299  if (!url.isValid()) {
300  return false;
301  }
302 
303  QString type = _type;
304  if (type.isEmpty()) {
305  type = url.scheme();
306  }
307 
308  if (d->loginMap.isEmpty() || d->isDirty) {
309  d->loginMap.clear();
310 
311  QString filename = QStandardPaths::writableLocation(QStandardPaths::GenericConfigLocation) + QLatin1Char('/') + QLatin1String("kionetrc");
312  bool kionetrcStatus = parse(filename);
313  bool netrcStatus = false;
314  if (userealnetrc) {
315  filename = QDir::homePath() + QLatin1String("/.netrc");
316  netrcStatus = parse(filename);
317  }
318 
319  if (!(kionetrcStatus || netrcStatus)) {
320  return false;
321  }
322  }
323 
324  if (!d->loginMap.contains(type)) {
325  return false;
326  }
327 
328  const LoginList &l = d->loginMap[type];
329  if (l.isEmpty()) {
330  return false;
331  }
332 
333  for (LoginList::ConstIterator it = l.begin(); it != l.end(); ++it) {
334  const AutoLogin &log = *it;
335 
336  if ((mode & defaultOnly) == defaultOnly &&
337  log.machine == QLatin1String("default") &&
338  (login.login.isEmpty() || login.login == log.login)) {
339  login.type = log.type;
340  login.machine = log.machine;
341  login.login = log.login;
342  login.password = log.password;
343  login.macdef = log.macdef;
344  }
345 
346  if ((mode & presetOnly) == presetOnly &&
347  log.machine == QLatin1String("preset") &&
348  (login.login.isEmpty() || login.login == log.login)) {
349  login.type = log.type;
350  login.machine = log.machine;
351  login.login = log.login;
352  login.password = log.password;
353  login.macdef = log.macdef;
354  }
355 
356  if ((mode & exactOnly) == exactOnly &&
357  log.machine == url.host() &&
358  (login.login.isEmpty() || login.login == log.login)) {
359  login.type = log.type;
360  login.machine = log.machine;
361  login.login = log.login;
362  login.password = log.password;
363  login.macdef = log.macdef;
364  break;
365  }
366  }
367 
368  return true;
369 }

376 bool NetRC::parse(const QString &fileName)
377 {
378  QFile file(fileName);
379  if (file.permissions() != (QFile::ReadOwner | QFile::WriteOwner
380  | QFile::ReadUser | QFile::WriteUser)) {
381  return false;
382 
383  }
384  if (!file.open(QIODevice::ReadOnly)) {
385  return false;
386  }
387 
388  d->fstream.setDevice(&file);
389 
390  QString line;
391 
392  while (!d->fstream.atEnd()) {
393  line = d->fstream.readLine().simplified();
394 
395  // If line is a comment or is empty, read next line
396  if ((line.startsWith(QLatin1String("#")) || line.isEmpty())) {
397  continue;
398  }
399 
400  // If line refers to a machine, maybe it is spread in more lines.
401  // getMachinePart() will take care of getting all the info and putting it into loginMap.
402  if ((line.startsWith(QLatin1String("machine"))
403  || line.startsWith(QLatin1String("default"))
404  || line.startsWith(QLatin1String("preset")))) {
405  d->getMachinePart(line);
406  continue;
407  }
408 
409  // If line refers to a macdef, it will be more than one line.
410  // getMacdefPart() will take care of getting all the lines of the macro
411  // and putting them into loginMap
412  if (line.startsWith(QLatin1String("macdef"))) {
413  d->getMacdefPart(line);
414  continue;
415  }
416  }
417  return true;
418 }

431 void NetRC::NetRCPrivate::getMachinePart(const QString &line)
432 {
433  QString buf = line;
434  while (!(buf.contains(QStringLiteral("login"))
435  && (buf.contains(QStringLiteral("password")) || buf.contains(QStringLiteral("account")) || buf.contains(QStringLiteral("type"))))) {
436  buf += QStringLiteral(" ");
437  buf += fstream.readLine().simplified();
438  }
439 
440  // Once we've got all the info, process it.
441  AutoLogin l;
442  l.machine = extract(buf, QStringLiteral("machine"));
443  if (l.machine.isEmpty()) {
444  if (buf.contains(QStringLiteral("default"))) {
445  l.machine = QStringLiteral("default");
446  } else if (buf.contains(QStringLiteral("preset"))) {
447  l.machine = QStringLiteral("preset");
448  }
449  }
450 
451  l.login = extract(buf, QStringLiteral("login"));
452  l.password = extract(buf, QStringLiteral("password"));
453  if (l.password.isEmpty()) {
454  l.password = extract(buf, QStringLiteral("account"));
455  }
456 
457  type = l.type = extract(buf, QStringLiteral("type"));
458  if (l.type.isEmpty() && !l.machine.isEmpty()) {
459  type = l.type = QStringLiteral("ftp");
460  }
461 
462  loginMap[l.type].append(l);
463  index = loginMap[l.type].count() - 1;
464 }

466 void NetRC::NetRCPrivate::getMacdefPart(const QString &line)
467 {
468  QString buf = line;
469  QString macro = extract(buf, QStringLiteral("macdef"));
470  QString newLine;
471  while (!fstream.atEnd()) {
472  newLine = fstream.readLine().simplified();
473  if (!newLine.isEmpty()) {
474  buf += QStringLiteral("\n");
475  buf += newLine;
476  } else {
477  break;
478  }
479  }
480  loginMap[type][index].macdef[macro].append(buf);
481 }
