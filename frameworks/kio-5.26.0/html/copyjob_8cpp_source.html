304 void CopyJobPrivate::slotStart()
305 {
306  Q_Q(CopyJob);
307  if (m_mode == CopyJob::CopyMode::Move) {
308  Q_FOREACH (const QUrl &url, m_srcList) {
309  if (m_dest.scheme() == url.scheme() && m_dest.host() == url.host()) {
310  QString srcPath = url.path();
311  if (!srcPath.endsWith(QLatin1Char('/')))
312  srcPath += QLatin1Char('/');
313  if (m_dest.path().startsWith(srcPath)) {
314  q->setError(KIO::ERR_CANNOT_MOVE_INTO_ITSELF);
315  q->emitResult();
316  return;
317  }
318  }
319  }
320  }
326  m_reportTimer = new QTimer(q);
327 
328  q->connect(m_reportTimer, SIGNAL(timeout()), q, SLOT(slotReport()));
329  m_reportTimer->start(REPORT_TIMEOUT);
330 
331  // Stat the dest

341 void CopyJobPrivate::slotResultStating(KJob *job)
342 {
343  Q_Q(CopyJob);
344  qCDebug(KIO_COPYJOB_DEBUG);
345  // Was there an error while stating the src ?
346  if (job->error() && destinationState != DEST_NOT_STATED) {
347  const QUrl srcurl = static_cast<SimpleJob *>(job)->url();
348  if (!srcurl.isLocalFile()) {
349  // Probably : src doesn't exist. Well, over some protocols (e.g. FTP)
350  // this info isn't really reliable (thanks to MS FTP servers).
351  // We'll assume a file, and try to download anyway.
352  qCDebug(KIO_COPYJOB_DEBUG) << "Error while stating source. Activating hack";
353  q->removeSubjob(job);
354  assert(!q->hasSubjobs()); // We should have only one job at a time ...
355  struct CopyInfo info;
356  info.permissions = (mode_t) - 1;
357  info.size = (KIO::filesize_t) - 1;
358  info.uSource = srcurl;
359  info.uDest = m_dest;
360  // Append filename or dirname to destination URL, if allowed
361  if (destinationState == DEST_IS_DIR && !m_asMethod) {
362  info.uDest = addPathToUrl(info.uDest, srcurl.fileName());
363  }
364 
365  files.append(info);
366  statNextSrc();
367  return;
368  }
369  // Local file. If stat fails, the file definitely doesn't exist.
370  // yes, q->Job::, because we don't want to call our override
371  q->Job::slotResult(job); // will set the error and emit result(this)
372  return;
373  }
374 
375  // Keep copy of the stat result
376  const UDSEntry entry = static_cast<StatJob *>(job)->statResult();
377 
378  if (destinationState == DEST_NOT_STATED) {
379  if (m_dest.isLocalFile()) { //works for dirs as well
380  QString path(m_dest.toLocalFile());
381  QFileInfo fileInfo(path);
382  if (m_asMethod || !fileInfo.exists()) {
383  // In copy-as mode, we want to check the directory to which we're
384  // copying. The target file or directory does not exist yet, which
385  // might confuse KDiskFreeSpaceInfo.
386  path = fileInfo.absolutePath();
387  }
388  KDiskFreeSpaceInfo freeSpaceInfo = KDiskFreeSpaceInfo::freeSpaceInfo(path);
389  if (freeSpaceInfo.isValid()) {
390  m_freeSpace = freeSpaceInfo.available();
391  } else {
392  qCDebug(KIO_COPYJOB_DEBUG) << "Couldn't determine free space information for" << path;
393  }
394  //TODO actually preliminary check is even more valuable for slow NFS/SMB mounts,
395  //but we need to find a way to report connection errors to user
396  }
397 
398  const bool isGlobalDest = m_dest == m_globalDest;
399  const bool isDir = entry.isDir();
400  // we were stating the dest
401  if (job->error()) {
402  destinationState = DEST_DOESNT_EXIST;
403  qCDebug(KIO_COPYJOB_DEBUG) << "dest does not exist";
404  } else {
405  // Treat symlinks to dirs as dirs here, so no test on isLink
406  destinationState = isDir ? DEST_IS_DIR : DEST_IS_FILE;
407  qCDebug(KIO_COPYJOB_DEBUG) << "dest is dir:" << isDir;
408 
409  const QString sLocalPath = entry.stringValue(KIO::UDSEntry::UDS_LOCAL_PATH);
410  if (!sLocalPath.isEmpty() && kio_resolve_local_urls && destinationState != DEST_DOESNT_EXIST) {
411  const QString fileName = m_dest.fileName();
412  m_dest = QUrl::fromLocalFile(sLocalPath);
413  if (m_asMethod) {
414  m_dest = addPathToUrl(m_dest, fileName);
415  }
416  qCDebug(KIO_COPYJOB_DEBUG) << "Setting m_dest to the local path:" << sLocalPath;
417  if (isGlobalDest) {
418  m_globalDest = m_dest;
419  }
420  }
421  }
422  if (isGlobalDest) {
423  m_globalDestinationState = destinationState;
424  }
425 
426  q->removeSubjob(job);
427  assert(!q->hasSubjobs());
428 
429  // After knowing what the dest is, we can start stat'ing the first src.
430  statCurrentSrc();
431  } else {
432  sourceStated(entry, static_cast<SimpleJob *>(job)->url());
433  q->removeSubjob(job);
434  }
435 }

437 void CopyJobPrivate::sourceStated(const UDSEntry &entry, const QUrl &sourceUrl)
438 {
439  const QString sLocalPath = entry.stringValue(KIO::UDSEntry::UDS_LOCAL_PATH);
440  const bool isDir = entry.isDir();
441 
442  // We were stating the current source URL
443  // Is it a file or a dir ?
444 
445  // There 6 cases, and all end up calling addCopyInfoFromUDSEntry first :
446  // 1 - src is a dir, destination is a directory,
447  // slotEntries will append the source-dir-name to the destination
448  // 2 - src is a dir, destination is a file -- will offer to overwrite, later on.
449  // 3 - src is a dir, destination doesn't exist, then it's the destination dirname,
450  // so slotEntries will use it as destination.
451 
452  // 4 - src is a file, destination is a directory,
453  // slotEntries will append the filename to the destination.
454  // 5 - src is a file, destination is a file, m_dest is the exact destination name
455  // 6 - src is a file, destination doesn't exist, m_dest is the exact destination name
456 
457  QUrl srcurl;
458  if (!sLocalPath.isEmpty() && destinationState != DEST_DOESNT_EXIST) {
459  qCDebug(KIO_COPYJOB_DEBUG) << "Using sLocalPath. destinationState=" << destinationState;
460  // Prefer the local path -- but only if we were able to stat() the dest.
461  // Otherwise, renaming a desktop:/ url would copy from src=file to dest=desktop (#218719)
462  srcurl = QUrl::fromLocalFile(sLocalPath);
463  } else {
464  srcurl = sourceUrl;
465  }
466  addCopyInfoFromUDSEntry(entry, srcurl, false, m_dest);
467 
468  m_currentDest = m_dest;
469  m_bCurrentSrcIsDir = false;
470 
471  if (isDir
472  // treat symlinks as files (no recursion)
473  && !entry.isLink()
474  && m_mode != CopyJob::Link) { // No recursion in Link mode either.
475  qCDebug(KIO_COPYJOB_DEBUG) << "Source is a directory";
476 
477  if (srcurl.isLocalFile()) {
478  const QString parentDir = srcurl.adjusted(QUrl::StripTrailingSlash).toLocalFile();
479  m_parentDirs.insert(parentDir);
480  }
481 
482  m_bCurrentSrcIsDir = true; // used by slotEntries
483  if (destinationState == DEST_IS_DIR) { // (case 1)
484  if (!m_asMethod) {
485  // Use <desturl>/<directory_copied> as destination, from now on
486  QString directory = srcurl.fileName();
487  const QString sName = entry.stringValue(KIO::UDSEntry::UDS_NAME);
488  KProtocolInfo::FileNameUsedForCopying fnu = KProtocolManager::fileNameUsedForCopying(srcurl);
489  if (fnu == KProtocolInfo::Name) {
490  if (!sName.isEmpty()) {
491  directory = sName;
492  }
493  } else if (fnu == KProtocolInfo::DisplayName) {
494  const QString dispName = entry.stringValue(KIO::UDSEntry::UDS_DISPLAY_NAME);
495  if (!dispName.isEmpty()) {
496  directory = dispName;
497  } else if (!sName.isEmpty()) {
498  directory = sName;
499  }
500  }
501  m_currentDest = addPathToUrl(m_currentDest, directory);
502  }
503  } else { // (case 3)
504  // otherwise dest is new name for toplevel dir
505  // so the destination exists, in fact, from now on.
506  // (This even works with other src urls in the list, since the
507  // dir has effectively been created)
508  destinationState = DEST_IS_DIR;
509  if (m_dest == m_globalDest) {
510  m_globalDestinationState = destinationState;
511  }
512  }
513 
514  startListing(srcurl);
515  } else {
516  qCDebug(KIO_COPYJOB_DEBUG) << "Source is a file (or a symlink), or we are linking -> no recursive listing";
517 
518  if (srcurl.isLocalFile()) {
519  const QString parentDir = srcurl.adjusted(QUrl::RemoveFilename | QUrl::StripTrailingSlash).path();
520  m_parentDirs.insert(parentDir);
521  }
522 
523  statNextSrc();
524  }

534 void CopyJobPrivate::slotReport()
535 {
536  Q_Q(CopyJob);
537  if (q->isSuspended()) {
538  return;
539  }
540  // If showProgressInfo was set, progressId() is > 0.
541  switch (state) {
542  case STATE_RENAMING:
543  q->setTotalAmount(KJob::Files, m_srcList.count());
544  // fall-through intended
545  case STATE_COPYING_FILES:
546  q->setProcessedAmount(KJob::Files, m_processedFiles);
547  if (m_bURLDirty) {
548  // Only emit urls when they changed. This saves time, and fixes #66281
549  m_bURLDirty = false;
550  if (m_mode == CopyJob::Move) {
551  emitMoving(q, m_currentSrcURL, m_currentDestURL);
552  emit q->moving(q, m_currentSrcURL, m_currentDestURL);
553  } else if (m_mode == CopyJob::Link) {
554  emitCopying(q, m_currentSrcURL, m_currentDestURL); // we don't have a delegate->linking
555  emit q->linking(q, m_currentSrcURL.path(), m_currentDestURL);
556  } else {
557  emitCopying(q, m_currentSrcURL, m_currentDestURL);
558  emit q->copying(q, m_currentSrcURL, m_currentDestURL);
559  }
560  }
561  break;
562 
563  case STATE_CREATING_DIRS:
564  q->setProcessedAmount(KJob::Directories, m_processedDirs);
565  if (m_bURLDirty) {
566  m_bURLDirty = false;
567  emit q->creatingDir(q, m_currentDestURL);
568  emitCreatingDir(q, m_currentDestURL);
569  }
570  break;
571 
572  case STATE_STATING:
573  case STATE_LISTING:
574  if (m_bURLDirty) {
575  m_bURLDirty = false;
576  if (m_mode == CopyJob::Move) {
577  emitMoving(q, m_currentSrcURL, m_currentDestURL);
578  } else {
579  emitCopying(q, m_currentSrcURL, m_currentDestURL);
580  }
581  }
582  q->setTotalAmount(KJob::Bytes, m_totalSize);
583  q->setTotalAmount(KJob::Files, files.count());
584  q->setTotalAmount(KJob::Directories, dirs.count());
585  break;
586 
587  default:
588  break;
589  }
590 }

841  q->addSubjob(job);
842  m_currentDestURL = m_dest;
843  m_bURLDirty = true;
844  } else {
845  // Finished the stat'ing phase
846  // First make sure that the totals were correctly emitted
847  state = STATE_STATING;
848  m_bURLDirty = true;
849  slotReport();

853 
854  if (!dirs.isEmpty()) {
855  emit q->aboutToCreate(q, dirs);
856  }

858  emit q->aboutToCreate(q, files);
859  }

864  createNextDir();
