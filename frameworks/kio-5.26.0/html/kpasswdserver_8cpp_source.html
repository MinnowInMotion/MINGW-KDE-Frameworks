129 static bool storeInWallet( KWallet::Wallet* wallet, const QString& key, const KIO::AuthInfo &info )
130 {
131  if ( !wallet->hasFolder( KWallet::Wallet::PasswordFolder() ) )
132  if ( !wallet->createFolder( KWallet::Wallet::PasswordFolder() ) )
133  return false;
134  wallet->setFolder( KWallet::Wallet::PasswordFolder() );
135  // Before saving, check if there's already an entry with this login.
136  // If so, replace it (with the new password). Otherwise, add a new entry.
137  typedef QMap<QString,QString> Map;
138  int entryNumber = 1;
139  Map map;
140  QString walletKey = makeWalletKey( key, info.realmValue );
141  qCDebug(category) << "walletKey =" << walletKey << " reading existing map";
142  if ( wallet->readMap( walletKey, map ) == 0 ) {
143  Map::ConstIterator end = map.constEnd();
144  Map::ConstIterator it = map.constFind( QStringLiteral("login") );
145  while ( it != end ) {
146  if ( it.value() == info.username ) {
147  break; // OK, overwrite this entry
148  }
149  it = map.constFind( QStringLiteral( "login-" ) + QString::number( ++entryNumber ) );
150  }
151  // If no entry was found, create a new entry - entryNumber is set already.
152  }
153  const QString loginKey = makeMapKey( "login", entryNumber );
154  const QString passwordKey = makeMapKey( "password", entryNumber );
155  qCDebug(category) << "writing to " << loginKey << "," << passwordKey;
156  // note the overwrite=true by default
157  map.insert( loginKey, info.username );
158  map.insert( passwordKey, info.password );
159  wallet->writeMap( walletKey, map );
160  return true;
161 }

163 static bool readFromWallet( KWallet::Wallet* wallet, const QString& key, const QString& realm, QString& username, QString& password, bool userReadOnly, QMap<QString,QString>& knownLogins )
164 {
165  //qCDebug(category) << "key =" << key << " username =" << username << " password =" /*<< password*/ << " userReadOnly =" << userReadOnly << " realm =" << realm;
166  if ( wallet->hasFolder( KWallet::Wallet::PasswordFolder() ) )
167  {
168  wallet->setFolder( KWallet::Wallet::PasswordFolder() );
169 
170  QMap<QString,QString> map;
171  if ( wallet->readMap( makeWalletKey( key, realm ), map ) == 0 )
172  {
173  typedef QMap<QString,QString> Map;
174  int entryNumber = 1;
175  Map::ConstIterator end = map.constEnd();
176  Map::ConstIterator it = map.constFind( QStringLiteral("login") );
177  while ( it != end ) {
178  //qCDebug(category) << "found " << it.key() << "=" << it.value();
179  Map::ConstIterator pwdIter = map.constFind( makeMapKey( "password", entryNumber ) );
180  if ( pwdIter != end ) {
181  if ( it.value() == username )
182  password = pwdIter.value();
183  knownLogins.insert( it.value(), pwdIter.value() );
184  }
185 
186  it = map.constFind( QStringLiteral( "login-" ) + QString::number( ++entryNumber ) );
187  }
188  //qCDebug(category) << knownLogins.count() << " known logins";
189 
190  if ( !userReadOnly && !knownLogins.isEmpty() && username.isEmpty() ) {
191  // Pick one, any one...
192  username = knownLogins.begin().key();
193  password = knownLogins.begin().value();
194  //qCDebug(category) << "picked the first one:" << username;
195  }
196 
197  return true;
198  }
199  }
200  return false;
201 }

515 void
516 KPasswdServer::processRequest()
517 {
518  if (m_authPending.isEmpty()) {
519  return;
520  }
521 
522  QScopedPointer<Request> request (m_authPending.takeFirst());
523 
524  // Prevent multiple prompts originating from the same window or the same
525  // key (server address).
526  const QString windowIdStr = QString::number(request->windowId);
527  if (m_authPrompted.contains(windowIdStr) || m_authPrompted.contains(request->key)) {
528  m_authPending.prepend(request.take()); // put it back.
529  return;
530  }
531 
532  m_authPrompted.append(windowIdStr);
533  m_authPrompted.append(request->key);
534 
535  KIO::AuthInfo &info = request->info;
536 
537  // NOTE: If info.username is empty and info.url.userName() is not, set
538  // info.username to info.url.userName() to ensure proper caching. See
539  // note passwordDialogDone.
540  if (info.username.isEmpty() && !info.url.userName().isEmpty()) {
541  info.username = info.url.userName();
542  }
543  const bool bypassCacheAndKWallet = info.getExtraField(AUTHINFO_EXTRAFIELD_BYPASS_CACHE_AND_KWALLET).toBool();
544 
545  const AuthInfoContainer *result = findAuthInfoItem(request->key, request->info);
546  qCDebug(category) << "key=" << request->key << ", user=" << info.username << "seqNr: request=" << request->seqNr << ", result=" << (result ? result->seqNr : -1);
547 
548  if (!bypassCacheAndKWallet && result && (request->seqNr < result->seqNr))
549  {
550  qCDebug(category) << "auto retry!";
551  if (result->isCanceled)
552  {
553  info.setModified(false);
554  }
555  else
556  {
557  updateAuthExpire(request->key, result, request->windowId, false);
558  copyAuthInfo(result, info);
559  }
560  }
561  else
562  {
563  m_seqNr++;
564  if (result && !request->errorMsg.isEmpty())
565  {
566  QString prompt (request->errorMsg.trimmed());
567  prompt += QLatin1Char('\n');
568  prompt += i18n("Do you want to retry?");
569 
570  QDialog* dlg = new QDialog;
571  connect(dlg, SIGNAL(finished(int)), this, SLOT(retryDialogDone(int)));
572  connect(this, SIGNAL(destroyed(QObject*)), dlg, SLOT(deleteLater()));
573  dlg->setWindowTitle(i18n("Retry Authentication"));
574  dlg->setWindowIcon(QIcon::fromTheme(QStringLiteral("dialog-password")));
575  dlg->setObjectName(QStringLiteral("warningOKCancel"));
576  QDialogButtonBox* buttonBox = new QDialogButtonBox(QDialogButtonBox::Yes|QDialogButtonBox::Cancel);
577  buttonBox->button(QDialogButtonBox::Yes)->setText(i18nc("@action:button filter-continue", "Retry"));
578 
579  KMessageBox::createKMessageBox(dlg, buttonBox, QMessageBox::Warning, prompt,
580  QStringList(), QString(), 0,
581  (KMessageBox::Notify | KMessageBox::NoExec));
582 
583  #ifndef Q_WS_WIN
584  KWindowSystem::setMainWindow(dlg, request->windowId);
585  #else
586  KWindowSystem::setMainWindow(dlg, (HWND)(long)request->windowId);
587  #endif
588 
589  qCDebug(category) << "Calling open on retry dialog" << dlg;
590  m_authRetryInProgress.insert(dlg, request.take());
591  dlg->open();
592  return;
593  }
594 
595  if (request->prompt)
596  {
597  showPasswordDialog(request.take());
598  return;
599  }
600  else
601  {
602  if (!bypassCacheAndKWallet && request->prompt)
603  {
604  addAuthInfoItem(request->key, info, 0, m_seqNr, true);
605  }
606  info.setModified( false );
607  }
608  }
609 
610  sendResponse(request.data());
611 }

613 QString KPasswdServer::createCacheKey( const KIO::AuthInfo &info )
614 {
615  if( !info.url.isValid() ) {
616  // Note that a null key will break findAuthInfoItem later on...
617  qCWarning(category) << "createCacheKey: invalid URL " << info.url ;
618  return QString();
619  }
620 
621  // Generate the basic key sequence.
622  QString key = info.url.scheme();
623  key += '-';
624  if (!info.url.userName().isEmpty())
625  {
626  key += info.url.userName();
627  key += '@';

909 void KPasswdServer::sendResponse (KPasswdServer::Request* request)
910 {
911  Q_ASSERT(request);
912  if (!request) {
913  return;
914  }
915 
916  qCDebug(category) << "key=" << request->key;
917  if (request->isAsync) {
918  emit queryAuthInfoAsyncResult(request->requestId, m_seqNr, request->info);
919  } else {
920  QByteArray replyData;
921  QDataStream stream2(&replyData, QIODevice::WriteOnly);
922  stream2 << request->info;
923  QDBusConnection::sessionBus().send(request->transaction.createReply(QVariantList() << replyData << m_seqNr));
924  }
925 
926  // Check all requests in the wait queue.
927  Request *waitRequest;
928  QMutableListIterator<Request*> it(m_authWait);
929  while (it.hasNext()) {
930  waitRequest = it.next();
931 
932  if (!hasPendingQuery(waitRequest->key, waitRequest->info))
933  {
934  const AuthInfoContainer *result = findAuthInfoItem(waitRequest->key,
935  waitRequest->info);
936  QByteArray replyData;
937 
938  QDataStream stream2(&replyData, QIODevice::WriteOnly);
939 
940  KIO::AuthInfo rcinfo;
941  if (!result || result->isCanceled)
942  {
943  waitRequest->info.setModified(false);
944  stream2 << waitRequest->info;
945  }
946  else
947  {
948  updateAuthExpire(waitRequest->key, result, waitRequest->windowId, false);
949  copyAuthInfo(result, rcinfo);
950  stream2 << rcinfo;
951  }
952 
953  if (waitRequest->isAsync) {
954  emit checkAuthInfoAsyncResult(waitRequest->requestId, m_seqNr, rcinfo);
955  } else {
956  QDBusConnection::sessionBus().send(waitRequest->transaction.createReply(QVariantList() << replyData << m_seqNr));
957  }
958 
959  delete waitRequest;
960  it.remove();
961  }
962  }
963 
964  // Re-enable password request processing for the current window id again.
965  m_authPrompted.removeAll(QString::number(request->windowId));
966  m_authPrompted.removeAll(request->key);
967 
968  if (m_authPending.count())
969  QTimer::singleShot(0, this, SLOT(processRequest()));
970 }

972 void KPasswdServer::passwordDialogDone(int result)
973 {
974  KPasswordDialog* dlg = qobject_cast<KPasswordDialog*>(sender());
975  Q_ASSERT(dlg);
976 
977  QScopedPointer<Request> request (m_authInProgress.take(dlg));
978  Q_ASSERT(request); // request should never be NULL.
979 
980  if (request) {
981  KIO::AuthInfo& info = request->info;
982  const bool bypassCacheAndKWallet = info.getExtraField(AUTHINFO_EXTRAFIELD_BYPASS_CACHE_AND_KWALLET).toBool();
983 
984  qCDebug(category) << "dialog result=" << result << ", bypassCacheAndKWallet?" << bypassCacheAndKWallet;
985  if (dlg && result == QDialog::Accepted) {
986  Q_ASSERT(dlg);
987  info.username = dlg->username();
988  info.password = dlg->password();
989  info.keepPassword = dlg->keepPassword();
990 
991  if (info.getExtraField(AUTHINFO_EXTRAFIELD_DOMAIN).isValid ())
992  info.setExtraField(AUTHINFO_EXTRAFIELD_DOMAIN, dlg->domain());
993  if (info.getExtraField(AUTHINFO_EXTRAFIELD_ANONYMOUS).isValid ())
994  info.setExtraField(AUTHINFO_EXTRAFIELD_ANONYMOUS, dlg->anonymousMode());
995 
996  // When the user checks "keep password", that means:
997  // * if the wallet is enabled, store it there for long-term, and in kpasswdserver
998  // only for the duration of the window (#92928)
999  // * otherwise store in kpasswdserver for the duration of the KDE session.
1000  if (!bypassCacheAndKWallet) {
1001  /*
1002  NOTE: The following code changes the key under which the auth
1003  info is stored in memory if the request url contains a username.
1004  e.g. "ftp://user@localhost", but the user changes that username
1005  in the password dialog.
1006 
1007  Since the key generated to store the credential contains the
1008  username from the request URL, the key must be updated on such
1009  changes. Otherwise, the key will not be found on subsequent
1010  requests and the user will be end up being prompted over and
1011  over to re-enter the password unnecessarily.
1012  */
1013  if (!info.url.userName().isEmpty() && info.username != info.url.userName()) {
1014  const QString oldKey(request->key);
1015  removeAuthInfoItem(oldKey, info);
1016  info.url.setUserName(info.username);
1017  request->key = createCacheKey(info);
1018  updateCachedRequestKey(m_authPending, oldKey, request->key);
1019  updateCachedRequestKey(m_authWait, oldKey, request->key);
1020  }
1021 
1022 #ifdef HAVE_KF5WALLET
1023  const bool skipAutoCaching = info.getExtraField(AUTHINFO_EXTRAFIELD_SKIP_CACHING_ON_QUERY).toBool();
1024  if (!skipAutoCaching && info.keepPassword && openWallet(request->windowId)) {
1025  if ( storeInWallet( m_wallet, request->key, info ) )
1026  // password is in wallet, don't keep it in memory after window is closed
1027  info.keepPassword = false;
1028  }
1029 #endif
1030  addAuthInfoItem(request->key, info, request->windowId, m_seqNr, false);
1031  }
1032  info.setModified( true );
1033  } else {
1034  if (!bypassCacheAndKWallet && request->prompt) {
1035  addAuthInfoItem(request->key, info, 0, m_seqNr, true);
1036  }
1037  info.setModified( false );
1038  }
1039 
1040  sendResponse(request.data());
1041  }
1042 
1043  dlg->deleteLater();
1044 }
