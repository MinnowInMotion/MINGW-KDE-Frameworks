129 bool TrashImpl::init()
130 {
131  if (m_initStatus == InitOK) {
132  return true;
133  }
134  if (m_initStatus == InitError) {
135  return false;
136  }
137 
138  // Check the trash directory and its info and files subdirs
139  // see also kdesktop/init.cc for first time initialization
140  m_initStatus = InitError;
141  // $XDG_DATA_HOME/Trash, i.e. ~/.local/share/Trash by default.
142  const QString xdgDataDir = QStandardPaths::writableLocation(QStandardPaths::GenericDataLocation) + QLatin1Char('/');
143  if (!QDir().mkpath(xdgDataDir)) {
144  qWarning() << "failed to create" << xdgDataDir;
145  return false;
146  }
147 
148  const QString trashDir = xdgDataDir + QLatin1String("Trash");
149  int err;
150  if ((err = testDir(trashDir))) {
151  error(err, trashDir);
152  return false;
153  }
154  if ((err = testDir(trashDir + QLatin1String("/info")))) {
155  error(err, trashDir + QLatin1String("/info"));
156  return false;
157  }
158  if ((err = testDir(trashDir + QLatin1String("/files")))) {
159  error(err, trashDir + QLatin1String("/files"));
160  return false;
161  }
162  m_trashDirectories.insert(0, trashDir);
163  m_initStatus = InitOK;
164  qDebug() << "initialization OK, home trash dir:" << trashDir;
165  return true;
166 }

168 void TrashImpl::migrateOldTrash()
169 {
170  qDebug();
171 
172  KConfigGroup g(KSharedConfig::openConfig(), "Paths");
173  const QString oldTrashDir = g.readPathEntry("Trash", QString());
174 
175  if (oldTrashDir.isEmpty()) {
176  return;
177  }
178 
179  const QStringList entries = listDir(oldTrashDir);
180  bool allOK = true;
181  for (QStringList::const_iterator entryIt = entries.begin(), entryEnd = entries.end();
182  entryIt != entryEnd; ++entryIt) {
183  QString srcPath = *entryIt;
184  if (srcPath == QLatin1String(".") || srcPath == QLatin1String("..") || srcPath == QLatin1String(".directory")) {
185  continue;
186  }
187  srcPath.prepend(oldTrashDir); // make absolute
188  int trashId;
189  QString fileId;
190  if (!createInfo(srcPath, trashId, fileId)) {
191  qWarning() << "Trash migration: failed to create info for" << srcPath;
192  allOK = false;
193  } else {
194  bool ok = moveToTrash(srcPath, trashId, fileId);
195  if (!ok) {
196  (void)deleteInfo(trashId, fileId);
197  qWarning() << "Trash migration: failed to create info for" << srcPath;
198  allOK = false;
199  } else {
200  qDebug() << "Trash migration: moved" << srcPath;
201  }
202  }
203  }
204  if (allOK) {
205  // We need to remove the old one, otherwise the desktop will have two trashcans...
206  qDebug() << "Trash migration: all OK, removing old trash directory";
207  synchronousDel(oldTrashDir, false, true);
208  }
209 }

211 bool TrashImpl::createInfo(const QString &origPath, int &trashId, QString &fileId)
212 {
213  qDebug() << origPath;
214  // Check source
215  const QByteArray origPath_c(QFile::encodeName(origPath));
216 
217  // off_t should be 64bit on Unix systems to have large file support
218  // FIXME: on windows this gets disabled until trash gets integrated
219  // BUG: 165449
220 #ifndef Q_OS_WIN
221  Q_STATIC_ASSERT(sizeof(off_t) >= 8);
222 #endif
223 
224  QT_STATBUF buff_src;
225  if (QT_LSTAT(origPath_c.data(), &buff_src) == -1) {
226  if (errno == EACCES) {
227  error(KIO::ERR_ACCESS_DENIED, origPath);
228  } else {
229  error(KIO::ERR_DOES_NOT_EXIST, origPath);
230  }
231  return false;
232  }
233 
234  // Choose destination trash
235  trashId = findTrashDirectory(origPath);
236  if (trashId < 0) {
237  qWarning() << "OUCH - internal error, TrashImpl::findTrashDirectory returned" << trashId;
238  return false; // ### error() needed?
239  }
240  qDebug() << "trashing to" << trashId;
241 
242  // Grab original filename
243  QUrl url = QUrl::fromLocalFile(origPath);
244  url = url.adjusted(QUrl::StripTrailingSlash);
245  const QString origFileName = url.fileName();
246 
247  // Make destination file in info/
248  url.setPath(infoPath(trashId, origFileName)); // we first try with origFileName
249  QUrl baseDirectory = QUrl::fromLocalFile(url.path());
250  // Here we need to use O_EXCL to avoid race conditions with other kioslave processes
251  int fd = 0;
252  QString fileName;
253  do {
254  qDebug() << "trying to create" << url.path();
255  fd = ::open(QFile::encodeName(url.path()), O_WRONLY | O_CREAT | O_EXCL, 0600);
256  if (fd < 0) {
257  if (errno == EEXIST) {
258  fileName = url.fileName();
259  url = url.adjusted(QUrl::RemoveFilename);
260  url.setPath(url.path() + KIO::suggestName(baseDirectory, fileName));
261  // and try again on the next iteration
262  } else {
263  error(KIO::ERR_COULD_NOT_WRITE, url.path());
264  return false;
265  }
266  }
267  } while (fd < 0);
268  const QString infoPath = url.path();
269  fileId = url.fileName();
270  Q_ASSERT(fileId.endsWith(QLatin1String(".trashinfo")));
271  fileId.truncate(fileId.length() - 10); // remove .trashinfo from fileId
272 
273  FILE *file = ::fdopen(fd, "w");
274  if (!file) { // can't see how this would happen
275  error(KIO::ERR_COULD_NOT_WRITE, infoPath);
276  return false;
277  }
278 
279  // Contents of the info file. We could use KSimpleConfig, but that would
280  // mean closing and reopening fd, i.e. opening a race condition...
281  QByteArray info = "[Trash Info]\n";
282  info += "Path=";
283  // Escape filenames according to the way they are encoded on the filesystem
284  // All this to basically get back to the raw 8-bit representation of the filename...
285  if (trashId == 0) { // home trash: absolute path
286  info += QUrl::toPercentEncoding(origPath, "/");
287  } else {
288  info += QUrl::toPercentEncoding(makeRelativePath(topDirectoryPath(trashId), origPath), "/");
289  }
290  info += '\n';
291  info += "DeletionDate=";
292  info += QDateTime::currentDateTime().toString(Qt::ISODate).toLatin1();
293  info += '\n';
294  size_t sz = info.size();
295 
296  size_t written = ::fwrite(info.data(), 1, sz, file);
297  if (written != sz) {
298  ::fclose(file);
299  QFile::remove(infoPath);
300  error(KIO::ERR_DISK_FULL, infoPath);
301  return false;
302  }
303 
304  ::fclose(file);
305 
306  qDebug() << "info file created in trashId=" << trashId << ":" << fileId;
307  return true;
308 }

310 QString TrashImpl::makeRelativePath(const QString &topdir, const QString &path)
311 {
312  QString realPath = QFileInfo(path).canonicalFilePath();
313  if (realPath.isEmpty()) { // shouldn't happen
314  realPath = path;
315  }
316  // topdir ends with '/'
317 #ifndef Q_OS_WIN
318  if (realPath.startsWith(topdir)) {
319 #else
320  if (realPath.startsWith(topdir, Qt::CaseInsensitive)) {
321 #endif
322  const QString rel = realPath.mid(topdir.length());
323  Q_ASSERT(rel[0] != QLatin1Char('/'));
324  return rel;
325  } else { // shouldn't happen...
326  qWarning() << "Couldn't make relative path for" << realPath << "(" << path << "), with topdir=" << topdir;
327  return realPath;
328  }

366 {
367  qDebug() << "Trashing" << origPath << trashId << fileId;
368  if (!adaptTrashSize(origPath, trashId)) {
369  return false;
370  }
371 
372  const qulonglong pathSize = DiscSpaceUtil::sizeOfPath(origPath);
373 
374  const QString dest = filesPath(trashId, fileId);
375  if (!move(origPath, dest)) {
376  // Maybe the move failed due to no permissions to delete source.
377  // In that case, delete dest to keep things consistent, since KIO doesn't do it.
378  if (QFileInfo(dest).isFile()) {
379  QFile::remove(dest);

381  synchronousDel(dest, false, true);
382  }
383  return false;
384  }
385 
386  if (QFileInfo(dest).isDir()) {
387  TrashSizeCache trashSize(trashDirectoryPath(trashId));
388  trashSize.add(fileId, pathSize);
389  }
390 
391  fileAdded();
392  return true;

494 bool TrashImpl::directRename(const QString &src, const QString &dest)
495 {
496  qDebug() << src << "->" << dest;
497  // Do not use QFile::rename here, we need to be able to move broken symlinks too
498  // (and we need to make sure errno is set)
499  if (::rename(QFile::encodeName(src), QFile::encodeName(dest)) != 0) {
500  if (errno == EXDEV) {
501  error(KIO::ERR_UNSUPPORTED_ACTION, QStringLiteral("rename"));
502  } else {
503  if ((errno == EACCES) || (errno == EPERM)) {
504  error(KIO::ERR_ACCESS_DENIED, dest);
505  } else if (errno == EROFS) { // The file is on a read-only filesystem
506  error(KIO::ERR_CANNOT_DELETE, src);
507  } else {
508  error(KIO::ERR_CANNOT_RENAME, src);
509  }
510  }
511  return false;
512  }
513  return true;
514 }

540 bool TrashImpl::del(int trashId, const QString &fileId)
541 {
542  QString info = infoPath(trashId, fileId);
543  QString file = filesPath(trashId, fileId);
544 
545  QByteArray info_c = QFile::encodeName(info);
546 
547  QT_STATBUF buff;
548  if (QT_LSTAT(info_c.data(), &buff) == -1) {
549  if (errno == EACCES) {
550  error(KIO::ERR_ACCESS_DENIED, file);
551  } else {
552  error(KIO::ERR_DOES_NOT_EXIST, file);
553  }
554  return false;
555  }
556 
557  const bool isDir = QFileInfo(file).isDir();
558  if (!synchronousDel(file, true, isDir)) {
559  return false;
560  }
561 
562  if (isDir) {
563  TrashSizeCache trashSize(trashDirectoryPath(trashId));
564  trashSize.remove(fileId);
565  }
566 
567  QFile::remove(info);
568  fileRemoved();
569  return true;

572 bool TrashImpl::synchronousDel(const QString &path, bool setLastErrorCode, bool isDir)
573 {
574  const int oldErrorCode = m_lastErrorCode;
575  const QString oldErrorMsg = m_lastErrorMessage;
576  QUrl url = QUrl::fromLocalFile(path);
577  // First ensure that all dirs have u+w permissions,
578  // otherwise we won't be able to delete files in them (#130780).
579  if (isDir) {
580  qDebug() << "chmod'ing" << url;
581  KFileItem fileItem(url, QStringLiteral("inode/directory"), KFileItem::Unknown);
582  KFileItemList fileItemList;
583  fileItemList.append(fileItem);
584  KIO::ChmodJob *chmodJob = KIO::chmod(fileItemList, 0200, 0200, QString(), QString(), true /*recursive*/, KIO::HideProgressInfo);
585  connect(chmodJob, SIGNAL(result(KJob*)),
586  this, SLOT(jobFinished(KJob*)));
587  enterLoop();
588  }
589 
590  KIO::DeleteJob *job = KIO::del(url, KIO::HideProgressInfo);
591  connect(job, SIGNAL(result(KJob*)),
592  this, SLOT(jobFinished(KJob*)));
593  enterLoop();
594  bool ok = m_lastErrorCode == 0;
595  if (!setLastErrorCode) {
596  m_lastErrorCode = oldErrorCode;
597  m_lastErrorMessage = oldErrorMsg;
598  }
599  return ok;
600 }

602 bool TrashImpl::emptyTrash()
603 {
604  qDebug();
605  // The naive implementation "delete info and files in every trash directory"
606  // breaks when deleted directories contain files owned by other users.
607  // We need to ensure that the .trashinfo file is only removed when the
608  // corresponding files could indeed be removed (#116371)
609 
610  // On the other hand, we certainly want to remove any file that has no associated
611  // .trashinfo file for some reason (#167051)
612 
613  QSet<QString> unremoveableFiles;
614 
615  int myErrorCode = 0;
616  QString myErrorMsg;
617  const TrashedFileInfoList fileInfoList = list();
618 
619  TrashedFileInfoList::const_iterator it = fileInfoList.begin();
620  const TrashedFileInfoList::const_iterator end = fileInfoList.end();
621  for (; it != end; ++it) {
622  const TrashedFileInfo &info = *it;
623  const QString filesPath = info.physicalPath;
624  if (synchronousDel(filesPath, true, true) || m_lastErrorCode == KIO::ERR_DOES_NOT_EXIST) {
625  QFile::remove(infoPath(info.trashId, info.fileId));
626  } else {
627  // error code is set by synchronousDel, let's remember it
628  // (so that successfully removing another file doesn't erase the error)
629  myErrorCode = m_lastErrorCode;
630  myErrorMsg = m_lastErrorMessage;
631  // and remember not to remove this file
632  unremoveableFiles.insert(filesPath);
633  qDebug() << "Unremoveable:" << filesPath;
634  }
635 
636  TrashSizeCache trashSize(trashDirectoryPath(info.trashId));
637  trashSize.clear();
638  }
639

1007 QString TrashImpl::trashForMountPoint(const QString &topdir, bool createIfNeeded) const
1008 {
1009  // (1) Administrator-created $topdir/.Trash directory
1010 
1011  const QString rootTrashDir = topdir + QLatin1String("/.Trash");
1012  const QByteArray rootTrashDir_c = QFile::encodeName(rootTrashDir);
1013  // Can't use QFileInfo here since we need to test for the sticky bit
1014  uid_t uid = getuid();
1015  QT_STATBUF buff;
1016  const unsigned int requiredBits = S_ISVTX; // Sticky bit required
1017  if (QT_LSTAT(rootTrashDir_c, &buff) == 0) {
1018  if ((S_ISDIR(buff.st_mode)) // must be a dir
1019  && (!S_ISLNK(buff.st_mode)) // not a symlink
1020  && ((buff.st_mode & requiredBits) == requiredBits)
1021  && (::access(rootTrashDir_c, W_OK) == 0) // must be user-writable
1022  ) {
1023  const QString trashDir = rootTrashDir + QLatin1Char('/') + QString::number(uid);
1024  const QByteArray trashDir_c = QFile::encodeName(trashDir);
1025  if (QT_LSTAT(trashDir_c, &buff) == 0) {
1026  if ((buff.st_uid == uid) // must be owned by user
1027  && (S_ISDIR(buff.st_mode)) // must be a dir
1028  && (!S_ISLNK(buff.st_mode)) // not a symlink
1029  && (buff.st_mode & 0777) == 0700) { // rwx for user
1030  return trashDir;
1031  }
1032  qDebug() << "Directory" << trashDir << "exists but didn't pass the security checks, can't use it";
1033  } else if (createIfNeeded && initTrashDirectory(trashDir_c)) {
1034  return trashDir;
1035  }
1036  } else {
1037  qDebug() << "Root trash dir" << rootTrashDir << "exists but didn't pass the security checks, can't use it";
1038  }
1039  }
1040 
1041  // (2) $topdir/.Trash-$uid
1042  const QString trashDir = topdir + QLatin1String("/.Trash-") + QString::number(uid);
1043  const QByteArray trashDir_c = QFile::encodeName(trashDir);
1044  if (QT_LSTAT(trashDir_c, &buff) == 0) {
1045  if ((buff.st_uid == uid) // must be owned by user
1046  && (S_ISDIR(buff.st_mode)) // must be a dir
1047  && (!S_ISLNK(buff.st_mode)) // not a symlink
1048  && ((buff.st_mode & 0777) == 0700)) { // rwx for user, ------ for group and others
1049 
1050  if (checkTrashSubdirs(trashDir_c)) {
1051  return trashDir;
1052  }
1053  }
1054  qDebug() << "Directory" << trashDir << "exists but didn't pass the security checks, can't use it";
1055  // Exists, but not useable
1056  return QString();
1057  }
1058  if (createIfNeeded && initTrashDirectory(trashDir_c)) {
1059  return trashDir;
1060  }
1061  return QString();
1062 }

1157 // Helper method. Parses a trash URL with the URL scheme defined in makeURL.
1158 // The trash:/ URL itself isn't parsed here, must be caught by the caller before hand.
1159 bool TrashImpl::parseURL(const QUrl &url, int &trashId, QString &fileId, QString &relativePath)
1160 {
1161  if (url.scheme() != QLatin1String("trash")) {
1162  return false;
1163  }
1164  const QString path = url.path();
1165  if (path.isEmpty()) {
1166  return false;
1167  }
1168  int start = 0;
1169  if (path[0] == QLatin1Char('/')) { // always true I hope
1170  start = 1;
1171  }
1172  int slashPos = path.indexOf(QLatin1Char('-'), 0); // don't match leading slash
1173  if (slashPos <= 0) {
1174  return false;
1175  }
1176  bool ok = false;
1177  trashId = path.midRef(start, slashPos - start).toInt(&ok);
1178  Q_ASSERT(ok);
1179  if (!ok) {
1180  return false;
1181  }
1182  start = slashPos + 1;
1183  slashPos = path.indexOf(QLatin1Char('/'), start);
1184  if (slashPos <= 0) {
1185  fileId = path.mid(start);
1186  relativePath.clear();
1187  return true;
1188  }
1189  fileId = path.mid(start, slashPos - start);
1190  relativePath = path.mid(slashPos + 1);
1191  return true;
1192 }

1194 bool TrashImpl::adaptTrashSize(const QString &origPath, int trashId)
1195 {
1196  KConfig config(QStringLiteral("ktrashrc"));
1197 
1198  const QString trashPath = trashDirectoryPath(trashId);
1199  KConfigGroup group = config.group(trashPath);
1200 
1201  bool useTimeLimit = group.readEntry("UseTimeLimit", false);
1202  bool useSizeLimit = group.readEntry("UseSizeLimit", true);
1203  double percent = group.readEntry("Percent", 10.0);
1204  int actionType = group.readEntry("LimitReachedAction", 0);
1205 
1206  if (useTimeLimit) { // delete all files in trash older than X days
1207  const int maxDays = group.readEntry("Days", 7);
1208  const QDateTime currentDate = QDateTime::currentDateTime();
1209 
1210  const TrashedFileInfoList trashedFiles = list();
1211  for (int i = 0; i < trashedFiles.count(); ++i) {
1212  struct TrashedFileInfo info = trashedFiles.at(i);
1213  if (info.trashId != trashId) {
1214  continue;
1215  }
1216 
1217  if (info.deletionDate.daysTo(currentDate) > maxDays) {
1218  del(info.trashId, info.fileId);
1219  }
1220  }
1221  }
1222 
1223  if (useSizeLimit) { // check if size limit exceeded
1224 
1225  // calculate size of the files to be put into the trash
1226  qulonglong additionalSize = DiscSpaceUtil::sizeOfPath(origPath);
1227 
1228  TrashSizeCache trashSize(trashPath);
1229  DiscSpaceUtil util(trashPath + QLatin1String("/files/"));
1230  if (util.usage(trashSize.calculateSize() + additionalSize) >= percent) {
1231  // before we start to remove any files from the trash,
1232  // check whether the new file will fit into the trash
1233  // at all...
1234  qulonglong partitionSize = util.size();
1235 
1236  if ((((double)additionalSize / (double)partitionSize) * 100) >= percent) {
1237  m_lastErrorCode = KIO::ERR_SLAVE_DEFINED;
1238  m_lastErrorMessage = i18n("The file is too large to be trashed.");
1239  return false;
1240  }
1241 
1242  if (actionType == 0) { // warn the user only
1243  m_lastErrorCode = KIO::ERR_SLAVE_DEFINED;
1244  m_lastErrorMessage = i18n("The trash has reached its maximum size!\nCleanup the trash manually.");
1245  return false;
1246  } else {
1247  // lets start removing some other files from the trash
1248 
1249  QDir dir(trashPath + QLatin1String("/files"));
1250  QFileInfoList infoList;
1251  if (actionType == 1) { // delete oldest files first
1252  infoList = dir.entryInfoList(QDir::Files | QDir::AllDirs | QDir::NoDotAndDotDot, QDir::Time | QDir::Reversed);
1253  } else if (actionType == 2) { // delete biggest files first
