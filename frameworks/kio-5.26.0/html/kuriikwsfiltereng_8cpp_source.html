57 SearchProvider* KURISearchFilterEngine::webShortcutQuery(const QString& typedString, QString &searchTerm) const
58 {
59  SearchProvider *provider = 0;
60 
61  if (m_bWebShortcutsEnabled)
62  {
63  const int pos = typedString.indexOf(m_cKeywordDelimiter);
64 
65  QString key;
66  if ( pos > -1 )
67  key = typedString.left(pos).toLower(); // #169801
68  else if ( !typedString.isEmpty() && m_cKeywordDelimiter == ' ')
69  key = typedString;
70 
71  if (!key.isEmpty() && !KProtocolInfo::isKnownProtocol(key))
72  {
73  provider = SearchProvider::findByKey(key);
74  if (provider)
75  {
76  if (!m_bUseOnlyPreferredWebShortcuts || m_preferredWebShortcuts.contains(provider->desktopEntryName()))
77  searchTerm = typedString.mid(pos+1);
78  else
79  {
80  delete provider;
81  provider = 0;
82  }
83  }
84  }
85  }
86 
87  return provider;
88 }

135 QStringList KURISearchFilterEngine::modifySubstitutionMap(SubstMap& map,
136  const QString& query) const
137 {
138  // Returns the number of query words
139  QString userquery = query;
140 
141  // Do some pre-encoding, before we can start the work:
142  {
143  int start = 0;
144  int pos = 0;
145  QRegExp qsexpr(QStringLiteral("\\\"[^\\\"]*\\\""));
146 
147  // Temporary substitute spaces in quoted strings (" " -> "%20")
148  // Needed to split user query into StringList correctly.
149  while ((pos = qsexpr.indexIn(userquery, start)) >= 0)
150  {
151  QString s = userquery.mid (pos, qsexpr.matchedLength());
152  s.replace (' ', QLatin1String("%20"));
153  start = pos + s.length(); // Move after last quote
154  userquery = userquery.replace (pos, qsexpr.matchedLength(), s);
155  }
156  }
157 
158  // Split user query between spaces:
159  QStringList l = userquery.simplified().split(' ', QString::SkipEmptyParts);
160 
161  // Back-substitute quoted strings (%20 -> " "):
162  userquery.replace (QLatin1String("%20"), QLatin1String(" "));
163  l.replaceInStrings(QStringLiteral("%20"), QStringLiteral(" "));
164 
165  qCDebug(category) << "Generating substitution map:\n";
166  // Generate substitution map from user query:
167  for (int i=0; i<=l.count(); i++)
168  {
169  int pos = 0;
170  QString v;
171  QString nr = QString::number(i);
172 
173  // Add whole user query (\{0}) to substitution map:
174  if (i==0)
175  v = userquery;
176  // Add partial user query items to substitution map:
177  else
178  v = l[i-1];
179 
180  // Insert partial queries (referenced by \1 ... \n) to map:
181  map.insert(QString::number(i), v);
182  PDVAR (" map['" + nr + "']", map[nr]);
183 
184  // Insert named references (referenced by \name) to map:
185  if ((i>0) && (pos = v.indexOf('=')) > 0)
186  {
187  QString s = v.mid(pos + 1);
188  QString k = v.left(pos);
189 
190  // Back-substitute references contained in references (e.g. '\refname' substitutes to 'thisquery=\0')
191  s.replace(QLatin1String("%5C"), QLatin1String("\\"));
192  map.insert(k, s);
193  PDVAR (" map['" + k + "']", map[k]);
194  }
195  }
196 
197  return l;
198 }

200 static QString encodeString(const QString& s, QTextCodec *codec)
201 {
202  // don't encode the space character, we replace it with + after the encoding
203  QByteArray encoded = codec->fromUnicode(s).toPercentEncoding(QByteArrayLiteral(" "));
204  encoded.replace(' ', '+');
205  return QString::fromUtf8(encoded);
206 }

208 QString KURISearchFilterEngine::substituteQuery(const QString& url, SubstMap &map, const QString& userquery, QTextCodec *codec) const
209 {
210  QString newurl = url;
211  QStringList ql = modifySubstitutionMap (map, userquery);
212  int count = ql.count();
213 
214  // Check, if old style '\1' is found and replace it with \{@} (compatibility mode):
215  {
216  int pos = -1;
217  if ((pos = newurl.indexOf(QStringLiteral("\\1"))) >= 0)
218  {
219  qCWarning(category) << "WARNING: Using compatibility mode for newurl='" << newurl
220  << "'. Please replace old style '\\1' with new style '\\{0}' "
221  "in the query definition.\n";
222  newurl = newurl.replace(pos, 2, QStringLiteral("\\{@}"));
223  }
224  }
225 
226  qCDebug(category) << "Substitute references:\n";
227  // Substitute references (\{ref1,ref2,...}) with values from user query:
228  {
229  int pos = 0;
230  QRegExp reflist(QStringLiteral("\\\\\\{[^\\}]+\\}"));
231 
232  // Substitute reflists (\{ref1,ref2,...}):
233  while ((pos = reflist.indexIn(newurl)) >= 0)
234  {
235  bool found = false;
236 
237  //bool rest = false;
238  QString v = QLatin1String("");
239  QString rlstring = newurl.mid(pos + 2, reflist.matchedLength() - 3);
240  PDVAR (" reference list", rlstring);
241 
242  // \{@} gets a special treatment later
243  if (rlstring == QLatin1String("@"))
244  {
245  v = QStringLiteral("\\@");
246  found = true;
247  }
248 
249  // TODO: strip whitespaces around commas
250  QStringList rl = rlstring.split(',', QString::SkipEmptyParts);
251  int i = 0;
252 
253  while ((i<rl.count()) && !found)
254  {
255  QString rlitem = rl[i];
256  QRegExp range(QStringLiteral("[0-9]*\\-[0-9]*"));
257 
258  // Substitute a range of keywords
259  if (range.indexIn(rlitem) >= 0)
260  {
261  int pos = rlitem.indexOf(QStringLiteral("-"));
262  int first = rlitem.leftRef(pos).toInt();
263  int last = rlitem.rightRef(rlitem.length()-pos-1).toInt();
264 
265  if (first == 0)
266  first = 1;
267 
268  if (last == 0)
269  last = count;
270 
271  for (int i=first; i<=last; i++)
272  {
273  v += map[QString::number(i)] + ' ';
274  // Remove used value from ql (needed for \{@}):
275  ql[i-1] = QLatin1String("");
276  }
277 
278  v = v.trimmed();
279  if (!v.isEmpty())
280  found = true;
281 
282  PDVAR (" range", QString::number(first) + '-' + QString::number(last) + " => '" + v + '\'');
283  v = encodeString(v, codec);
284  }
285  else if ( rlitem.startsWith('\"') && rlitem.endsWith('\"') )
286  {
287  // Use default string from query definition:
288  found = true;
289  QString s = rlitem.mid(1, rlitem.length() - 2);
290  v = encodeString(s, codec);
291  PDVAR (" default", s);
292  }
293  else if (map.contains(rlitem))
294  {
295  // Use value from substitution map:
296  found = true;
297  PDVAR (" map['" + rlitem + "']", map[rlitem]);
298  v = encodeString(map[rlitem], codec);
299 
300  // Remove used value from ql (needed for \{@}):
301  QString c = rlitem.left(1);
302  if (c==QLatin1String("0"))
303  {
304  // It's a numeric reference to '0'
305  for (QStringList::Iterator it = ql.begin(); it!=ql.end(); ++it)
306  (*it) = QLatin1String("");
307  }
308  else if ((c>=QLatin1String("0")) && (c<=QLatin1String("9"))) // krazy:excludeall=doublequote_chars
309  {
310  // It's a numeric reference > '0'
311  int n = rlitem.toInt();
312  ql[n-1] = QLatin1String("");
313  }
314  else
315  {
316  // It's a alphanumeric reference
317  QStringList::Iterator it = ql.begin();
318  while ((it != ql.end()) && !it->startsWith(rlitem + '='))
319  ++it;
320  if (it != ql.end())
321  it->clear();
322  }
323 
324  // Encode '+', otherwise it would be interpreted as space in the resulting url:
325  v.replace('+', QLatin1String("%2B"));
326  }
327  else if (rlitem == QLatin1String("@"))
328  {
329  v = QStringLiteral("\\@");
330  PDVAR (" v", v);
331  }
332 
333  i++;
334  }
335 
336  newurl.replace(pos, reflist.matchedLength(), v);
337  }
338 
339  // Special handling for \{@};
340  {
341  PDVAR (" newurl", newurl);
342  // Generate list of unmatched strings:
343  QString v = ql.join(QStringLiteral(" ")).simplified();
344 
345  PDVAR (" rest", v);
346  v = encodeString(v, codec);
347 
348  // Substitute \{@} with list of unmatched query strings
349  newurl.replace(QLatin1String("\\@"), v);
350  }
351  }
352 
353  return newurl;
354 }

366 QUrl KURISearchFilterEngine::formatResult( const QString& url,
367  const QString& cset1,
368  const QString& cset2,
369  const QString& userquery,
370  bool /* isMalformed */,
371  SubstMap& map ) const
372 {
373  // Return nothing if userquery is empty and it contains
374  // substitution strings...
375  if (userquery.isEmpty() && url.indexOf(QStringLiteral("\\{")) > 0)
376  return QUrl();
377 
378  // Debug info of map:
379  if (!map.isEmpty())
380  {
381  qCDebug(category) << "Got non-empty substitution map:\n";
382  for(SubstMap::Iterator it = map.begin(); it != map.end(); ++it)
383  PDVAR (" map['" + it.key() + "']", it.value());
384  }
385 
386  // Create a codec for the desired encoding so that we can transcode the user's "url".
387  QString cseta = cset1;
388  if (cseta.isEmpty())
389  cseta = QStringLiteral("UTF-8");
390 
391  QTextCodec *csetacodec = QTextCodec::codecForName(cseta.toLatin1());
392  if (!csetacodec)
393  {
394  cseta = QStringLiteral("UTF-8");
395  csetacodec = QTextCodec::codecForName(cseta.toLatin1());
396  }
397 
398  PDVAR ("user query", userquery);
399  PDVAR ("query definition", url);
400 
401  // Add charset indicator for the query to substitution map:
402  map.insert(QStringLiteral("ikw_charset"), cseta);
403 
404  // Add charset indicator for the fallback query to substitution map:
405  QString csetb = cset2;
406  if (csetb.isEmpty())
407  csetb = QStringLiteral("UTF-8");
408  map.insert(QStringLiteral("wsc_charset"), csetb);
409 
410  QString newurl = substituteQuery (url, map, userquery, csetacodec);
411 
412  PDVAR ("substituted query", newurl);
413 
414  return QUrl(newurl, QUrl::StrictMode);
415 }

417 void KURISearchFilterEngine::loadConfig()
418 {
419  qCDebug(category) << "Keywords Engine: Loading config...";
420 
421  // Load the config.
422  KConfig config( name() + QStringLiteral("rc"), KConfig::NoGlobals );
423  KConfigGroup group = config.group( "General" );
424 
425  m_cKeywordDelimiter = QString(group.readEntry("KeywordDelimiter", ":")).at(0).toLatin1();

427  m_defaultWebShortcut = group.readEntry("DefaultWebShortcut");
428  m_bUseOnlyPreferredWebShortcuts = group.readEntry("UsePreferredWebShortcutsOnly", false);

430  QStringList defaultPreferredShortcuts;

433  m_preferredWebShortcuts = group.readEntry("PreferredWebShortcuts", defaultPreferredShortcuts);
434 
435  // Use either a white space or a : as the keyword delimiter...
436  if (strchr (" :", m_cKeywordDelimiter) == 0)
437  m_cKeywordDelimiter = ':';

439  qCDebug(category) << "Web Shortcuts Enabled: " << m_bWebShortcutsEnabled;
440  qCDebug(category) << "Default Shortcut: " << m_defaultWebShortcut;
