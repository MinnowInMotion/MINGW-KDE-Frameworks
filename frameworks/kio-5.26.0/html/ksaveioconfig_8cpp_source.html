213 void KSaveIOConfig::updateRunningIOSlaves (QWidget *parent)
214 {
215  // Inform all running io-slaves about the changes...
216  // if we cannot update, ioslaves inform the end user...
217  QDBusMessage message = QDBusMessage::createSignal(QStringLiteral("/KIO/Scheduler"), QStringLiteral("org.kde.KIO.Scheduler"), QStringLiteral("reparseSlaveConfiguration"));
218  message << QString();
219  if (!QDBusConnection::sessionBus().send(message))
220  {
221  KMessageBox::information (parent,
222  i18n("You have to restart the running applications "
223  "for these changes to take effect."),
224  i18nc("@title:window", "Update Failed"));
225  }
226 }

228 void KSaveIOConfig::updateProxyScout(QWidget * parent)
229 {
230  // Inform the proxyscout kded module about changes if we cannot update,
231  // ioslaves inform the end user...
232  QDBusInterface kded(QStringLiteral("org.kde.kcookiejar5"), QStringLiteral("/modules/proxyscout"), QStringLiteral("org.kde.KPAC.ProxyScout"));
233  QDBusReply<void> reply = kded.call(QStringLiteral("reset"));
234  if (!reply.isValid())

236  KMessageBox::information (parent,
237  i18n("You have to restart KDE for these changes to take effect."),
238  i18nc("@title:window", "Update Failed"));
