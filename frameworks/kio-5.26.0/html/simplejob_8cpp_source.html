53 {
54  Q_D(SimpleJob);

61  return Job::doKill();

125 void SimpleJobPrivate::start(Slave *slave)
126 {
127  Q_Q(SimpleJob);
128  m_slave = slave;
129 
130  // Slave::setJob can send us SSL metadata if there is a persistent connection
131  q->connect(slave, SIGNAL(metaData(KIO::MetaData)),
132  SLOT(slotMetaData(KIO::MetaData)));
133 
134  slave->setJob(q);
135 
136  q->connect(slave, SIGNAL(error(int,QString)),
137  SLOT(slotError(int,QString)));
138 
139  q->connect(slave, SIGNAL(warning(QString)),
140  SLOT(slotWarning(QString)));
141 
142  q->connect(slave, SIGNAL(infoMessage(QString)),
143  SLOT(_k_slotSlaveInfoMessage(QString)));
144 
145  q->connect(slave, SIGNAL(connected()),
146  SLOT(slotConnected()));
147 
148  q->connect(slave, SIGNAL(finished()),
149  SLOT(slotFinished()));
150 
151  if ((m_extraFlags & EF_TransferJobDataSent) == 0) { // this is a "get" job
152  q->connect(slave, SIGNAL(totalSize(KIO::filesize_t)),
153  SLOT(slotTotalSize(KIO::filesize_t)));
154 
155  q->connect(slave, SIGNAL(processedSize(KIO::filesize_t)),
156  SLOT(slotProcessedSize(KIO::filesize_t)));
157 
158  q->connect(slave, SIGNAL(speed(ulong)),
159  SLOT(slotSpeed(ulong)));
160  }
161 
162  const QVariant windowIdProp = q->property("window-id"); // see KJobWidgets::setWindow
163  if (windowIdProp.isValid()) {
164  m_outgoingMetaData.insert(QStringLiteral("window-id"), QString::number(windowIdProp.toULongLong()));
165  }
166 
167  const QVariant userTimestampProp = q->property("userTimestamp"); // see KJobWidgets::updateUserTimestamp
168  if (userTimestampProp.isValid()) {
169  m_outgoingMetaData.insert(QStringLiteral("user-timestamp"), QString::number(userTimestampProp.toULongLong()));
170  }
171 
172  if (q->uiDelegate() == 0) { // not interactive
173  m_outgoingMetaData.insert(QStringLiteral("no-auth-prompt"), QStringLiteral("true"));
174  }
175 
176  if (!m_outgoingMetaData.isEmpty()) {
177  KIO_ARGS << m_outgoingMetaData;
178  slave->send(CMD_META_DATA, packedArgs);
179  }
180 
181  if (!m_subUrl.isEmpty()) {
182  KIO_ARGS << m_subUrl;
183  slave->send(CMD_SUBURL, packedArgs);
184  }
185 
186  slave->send(m_command, m_packedArgs);
187 }

189 void SimpleJobPrivate::slaveDone()
190 {
191  Q_Q(SimpleJob);
192  if (m_slave) {
193  if (m_command == CMD_OPEN) {
194  m_slave->send(CMD_CLOSE);
195  }
196  q->disconnect(m_slave); // Remove all signals between slave and job
197  }
198  // only finish a job once; Scheduler::jobFinished() resets schedSerial to zero.
199  if (m_schedSerial) {
200  Scheduler::jobFinished(q, m_slave);
201  }
202 }

204 void SimpleJob::slotFinished()
205 {
206  Q_D(SimpleJob);
207  // Return slave to the scheduler
208  d->slaveDone();
209 
210  if (!hasSubjobs()) {
211  if (!error() && (d->m_command == CMD_MKDIR || d->m_command == CMD_RENAME)) {
212  if (d->m_command == CMD_MKDIR) {
213  const QUrl urlDir = url().adjusted(QUrl::RemoveFilename | QUrl::StripTrailingSlash);
214  org::kde::KDirNotify::emitFilesAdded(urlDir);
215  } else { /*if ( m_command == CMD_RENAME )*/
216  QUrl src, dst;
217  QDataStream str(d->m_packedArgs);
218  str >> src >> dst;
219  if (src.adjusted(QUrl::RemoveFilename) == dst.adjusted(QUrl::RemoveFilename) // For the user, moving isn't renaming. Only renaming is.
220  && !KProtocolInfo::slaveHandlesNotify(dst.scheme()).contains(QLatin1String("Rename"))) {
221  org::kde::KDirNotify::emitFileRenamed(src, dst);
222  }
223 
224  org::kde::KDirNotify::emitFileMoved(src, dst);
225  if (d->m_uiDelegateExtension) {
226  d->m_uiDelegateExtension->updateUrlInClipboard(src, dst);
227  }
228  }
229  }
230  emitResult();
231  }
232 }

282 void SimpleJobPrivate::restartAfterRedirection(QUrl *redirectionUrl)
283 {
284  Q_Q(SimpleJob);
285  // Return slave to the scheduler while we still have the old URL in place; the scheduler
286  // requires a job URL to stay invariant while the job is running.
287  slaveDone();
288 
289  m_url = *redirectionUrl;
290  redirectionUrl->clear();
291  if ((m_extraFlags & EF_KillCalled) == 0) {
292  Scheduler::doJob(q);
293  }
294 }

311 void SimpleJob::slotMetaData(const KIO::MetaData &_metaData)
312 {
313  Q_D(SimpleJob);
314  QMapIterator<QString, QString> it(_metaData);
315  while (it.hasNext()) {
316  it.next();
317  if (it.key().startsWith(QLatin1String("{internal~"), Qt::CaseInsensitive)) {
318  d->m_internalMetaData.insert(it.key(), it.value());
319  } else {
320  d->m_incomingMetaData.insert(it.key(), it.value());
321  }
322  }
323 
324  // Update the internal meta-data values as soon as possible. Waiting until
325  // the ioslave is finished has unintended consequences if the client starts
326  // a new connection without waiting for the ioslave to finish.
327  if (!d->m_internalMetaData.isEmpty()) {
328  Scheduler::updateInternalMetaData(this);
329  }
330 }
