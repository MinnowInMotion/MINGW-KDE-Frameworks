95 // finished is a special case. If we emit it right away, then
96 // TransferJob::start can delete the job even before the end of the method
97 void DataSlave::dispatch_finished()
98 {
99  QueueStruct q(Queue_finished);
100  q.size = -1;

107 void DataSlave::dispatchNext()
108 {
109  if (dispatchQueue.empty()) {
110  timer->stop();
111  return;
112  }
113 
114  const QueueStruct &q = dispatchQueue.front();
115  //qDebug() << this << "dispatching" << q.type << dispatchQueue.size() << "left";
116  switch (q.type) {
117  case Queue_mimeType: mimeType(q.s); break;
118  case Queue_totalSize: totalSize(q.size); break;
119  case Queue_sendMetaData: sendMetaData(); break;
120  case Queue_data: data(q.ba); break;
121  case Queue_finished: finished(); break;
122  }/*end switch*/
123 
124  dispatchQueue.pop_front();
125 }

127 void DataSlave::send(int cmd, const QByteArray &arr)
128 {
129  QDataStream stream(arr);
130 
131  QUrl url;
132 
133  switch (cmd) {
134  case CMD_GET: {
135  stream >> url;
136  get(url);
137  break;
138  }
139  case CMD_MIMETYPE: {
140  stream >> url;
141  mimetype(url);
142  break;
143  }
144  // ignore these (must not emit error, otherwise SIGSEGV occurs)
145  case CMD_REPARSECONFIGURATION:
146  case CMD_META_DATA:
147  case CMD_SUBURL:
148  break;
149  default:
150  error(ERR_UNSUPPORTED_ACTION,
151  unsupportedActionErrorString(QStringLiteral("data"), cmd));
152  }/*end switch*/
153 }
