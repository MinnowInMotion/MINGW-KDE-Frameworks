46 static int findString(const QString &s, const char *const *values)
47 {
48  int index = 0;
49  const QString lower = s.toLower();
50  for (const char *const *p = values; *p; ++p, ++index) {
51  if (s.compare(QLatin1String(*p), Qt::CaseInsensitive) == 0) {
52  return index;
53  }
54  }
55  return -1;
56 }

119 static bool addressLessThanComparison(const QHostAddress &addr1, const QHostAddress &addr2)
120 {
121  if (addr1.protocol() == QAbstractSocket::IPv4Protocol &&
122  addr2.protocol() == QAbstractSocket::IPv4Protocol) {
123  return addr1.toIPv4Address() < addr2.toIPv4Address();
124  }
125 
126  if (addr1.protocol() == QAbstractSocket::IPv6Protocol &&
127  addr2.protocol() == QAbstractSocket::IPv6Protocol) {
128  const Q_IPV6ADDR ipv6addr1 = addr1.toIPv6Address();
129  const Q_IPV6ADDR ipv6addr2 = addr2.toIPv6Address();
130  for (int i = 0; i < 16; ++i) {
131  if (ipv6addr1[i] != ipv6addr2[i]) {
132  return ((ipv6addr1[i] & 0xff) - (ipv6addr2[i] & 0xff));
133  }
134  }
135  }
136 
137  return false;
138 }

176 private:
177  Address(const QString &host)
178  {
179  // Always try to see if it's already an IP first, to avoid Qt doing a
180  // needless reverse lookup
181  QHostAddress address(host);
182  if (address.isNull()) {
183  QHostInfo hostInfo = KIO::HostInfo::lookupCachedHostInfoFor(host);
184  if (hostInfo.hostName().isEmpty() || hostInfo.error() != QHostInfo::NoError) {
185  hostInfo = QHostInfo::fromName(host);
186  KIO::HostInfo::cacheLookup(hostInfo);
187  }
188  m_addressList = hostInfo.addresses();
189  } else {
190  m_addressList.clear();
191  m_addressList.append(address);
192  }
193  }
194 
195  QList<QHostAddress> m_addressList;
196 };

363 // weekdayRange(day [, "GMT" ])
364 // weekdayRange(day1, day2 [, "GMT" ])
365 // @returns true if the current day equals day or between day1 and day2 resp.
366 // If the last argument is "GMT", GMT timezone is used, otherwise local time
367 QScriptValue WeekdayRange(QScriptContext *context, QScriptEngine *engine)
368 {
369  if (context->argumentCount() < 1 || context->argumentCount() > 3) {
370  return engine->undefinedValue();
371  }
372 
373  static const char *const days[] = { "sun", "mon", "tue", "wed", "thu", "fri", "sat", 0 };
374 
375  const int d1 = findString(context->argument(0).toString(), days);
376  if (d1 == -1) {
377  return engine->undefinedValue();
378  }
379 
380  int d2 = findString(context->argument(1).toString(), days);
381  if (d2 == -1) {
382  d2 = d1;
383  }
384 
385  // Adjust the days of week coming from QDateTime since it starts
386  // counting with Monday as 1 and ends with Sunday as day 7.
387  int dayOfWeek = getTime(context).date().dayOfWeek();
388  if (dayOfWeek == 7) {
389  dayOfWeek = 0;
390  }
391  return engine->toScriptValue(checkRange(dayOfWeek, d1, d2));
392 }

394 // dateRange(day [, "GMT" ])
395 // dateRange(day1, day2 [, "GMT" ])
396 // dateRange(month [, "GMT" ])
397 // dateRange(month1, month2 [, "GMT" ])
398 // dateRange(year [, "GMT" ])
399 // dateRange(year1, year2 [, "GMT" ])
400 // dateRange(day1, month1, day2, month2 [, "GMT" ])
401 // dateRange(month1, year1, month2, year2 [, "GMT" ])
402 // dateRange(day1, month1, year1, day2, month2, year2 [, "GMT" ])
403 // @returns true if the current date (GMT or local time according to
404 // presence of "GMT" as last argument) is within the given range
405 QScriptValue DateRange(QScriptContext *context, QScriptEngine *engine)
406 {
407  if (context->argumentCount() < 1 || context->argumentCount() > 7) {
408  return engine->undefinedValue();
409  }
410 
411  static const char *const months[] = { "jan", "feb", "mar", "apr", "may", "jun",
412  "jul", "aug", "sep", "oct", "nov", "dec", 0
413  };
414 
415  QVector<int> values;
416  for (int i = 0; i < context->argumentCount(); ++i) {
417  int value = -1;
418  if (context->argument(i).isNumber()) {
419  value = context->argument(i).toInt32();
420  } else {
421  // QDate starts counting months from 1, so we add 1 here.
422  value = findString(context->argument(i).toString(), months) + 1;
423  }
424 
425  if (value > 0) {
426  values.append(value);
427  } else {
428  break;
429  }
430  }
431 
432  const QDate now = getTime(context).date();
433 
434  // day1, month1, year1, day2, month2, year2
435  if (values.size() == 6) {
436  const QDate d1(values[2], values[1], values[0]);
437  const QDate d2(values[5], values[4], values[3]);
438  return engine->toScriptValue(checkRange(now, d1, d2));
439  }
440  // day1, month1, day2, month2
441  else if (values.size() == 4 && values[ 1 ] < 13 && values[ 3 ] < 13) {
442  const QDate d1(now.year(), values[1], values[0]);
443  const QDate d2(now.year(), values[3], values[2]);
444  return engine->toScriptValue(checkRange(now, d1, d2));
445  }
446  // month1, year1, month2, year2
447  else if (values.size() == 4) {
448  const QDate d1(values[1], values[0], now.day());
449  const QDate d2(values[3], values[2], now.day());
450  return engine->toScriptValue(checkRange(now, d1, d2));
451  }
452  // year1, year2
453  else if (values.size() == 2 && values[0] >= 1000 && values[1] >= 1000) {
454  return engine->toScriptValue(checkRange(now.year(), values[0], values[1]));
455  }
456  // day1, day2
457  else if (values.size() == 2 && context->argument(0).isNumber() && context->argument(1).isNumber()) {
458  return engine->toScriptValue(checkRange(now.day(), values[0], values[1]));
459  }
460  // month1, month2
461  else if (values.size() == 2) {
462  return engine->toScriptValue(checkRange(now.month(), values[0], values[1]));
463  }
464  // year
465  else if (values.size() == 1 && values[ 0 ] >= 1000) {
466  return engine->toScriptValue(checkRange(now.year(), values[0], values[0]));
467  }
468  // day
469  else if (values.size() == 1 && context->argument(0).isNumber()) {
470  return engine->toScriptValue(checkRange(now.day(), values[0], values[0]));
471  }
472  // month
473  else if (values.size() == 1) {
474  return engine->toScriptValue(checkRange(now.month(), values[0], values[0]));
475  }
476 
477  return engine->undefinedValue();
478 }

480 // timeRange(hour [, "GMT" ])
481 // timeRange(hour1, hour2 [, "GMT" ])
482 // timeRange(hour1, min1, hour2, min2 [, "GMT" ])
483 // timeRange(hour1, min1, sec1, hour2, min2, sec2 [, "GMT" ])
484 // @returns true if the current time (GMT or local based on presence
485 // of "GMT" argument) is within the given range
486 QScriptValue TimeRange(QScriptContext *context, QScriptEngine *engine)
487 {
488  if (context->argumentCount() < 1 || context->argumentCount() > 7) {
489  return engine->undefinedValue();
490  }
491 
492  QVector<int> values;
493  for (int i = 0; i < context->argumentCount(); ++i) {
494  if (!context->argument(i).isNumber()) {
495  break;
496  }
497  values.append(context->argument(i).toNumber());
498  }
499 
500  const QTime now = getTime(context).time();
501 
502  // hour1, min1, sec1, hour2, min2, sec2
503  if (values.size() == 6) {
504  const QTime t1(values[0], values[1], values[2]);
505  const QTime t2(values[3], values[4], values[5]);
506  return engine->toScriptValue(checkRange(now, t1, t2));
507  }
508  // hour1, min1, hour2, min2
509  else if (values.size() == 4) {
510  const QTime t1(values[0], values[1]);
511  const QTime t2(values[2], values[3]);
512  return engine->toScriptValue(checkRange(now, t1, t2));
513  }
514  // hour1, hour2
515  else if (values.size() == 2) {
516  return engine->toScriptValue(checkRange(now.hour(), values[0], values[1]));
517  }
518  // hour
519  else if (values.size() == 1) {
520  return engine->toScriptValue(checkRange(now.hour(), values[0], values[0]));
521  }
522 
523  return engine->undefinedValue();
524 }

640 // @returns a sorted ipAddressList. If both IPv4 and IPv6 addresses are present in
641 // the list. The sorted IPv6 addresses will precede the sorted IPv4 addresses.
642 QScriptValue SortIpAddressList(QScriptContext *context, QScriptEngine *engine)
643 {
644  if (context->argumentCount() != 1) {
645  return engine->undefinedValue();
646  }
647 
648  QHash<QString, QString> actualEntryMap;
649  QList<QHostAddress> ipV4List, ipV6List;
650  const QStringList ipAddressList = context->argument(0).toString().split(QLatin1Char(';'));
651 
652  Q_FOREACH (const QString &ipAddress, ipAddressList) {
653  QHostAddress address(ipAddress);
654  switch (address.protocol()) {
655  case QAbstractSocket::IPv4Protocol:
656  ipV4List << address;
657  actualEntryMap.insert(address.toString(), ipAddress);
658  break;
659  case QAbstractSocket::IPv6Protocol:
660  ipV6List << address;
661  actualEntryMap.insert(address.toString(), ipAddress);
662  break;
663  default:
664  break;
665  }
666  }
667 
668  QString sortedAddress(QLatin1String(""));
669 
670  if (!ipV6List.isEmpty()) {
671  qSort(ipV6List.begin(), ipV6List.end(), addressLessThanComparison);
672  sortedAddress += addressListToString(ipV6List, actualEntryMap);
673  }
674 
675  if (!ipV4List.isEmpty()) {
676  qSort(ipV4List.begin(), ipV4List.end(), addressLessThanComparison);
677  if (!sortedAddress.isEmpty()) {
678  sortedAddress += QLatin1Char(';');
679  }
680  sortedAddress += addressListToString(ipV4List, actualEntryMap);
681  }
682 
683  return engine->toScriptValue(sortedAddress);
684 
685 }

746 {

748

753  return QString();
