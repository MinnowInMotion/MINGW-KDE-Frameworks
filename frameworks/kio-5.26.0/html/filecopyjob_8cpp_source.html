119  * The FileCopyJob works according to the famous Bavarian
120  * 'Alternating Bitburger Protocol': we either drink a beer or we
121  * we order a beer, but never both at the same time.
122  * Translated to io-slaves: We alternate between receiving a block of data
123  * and sending it away.

132 void FileCopyJobPrivate::slotStart()
133 {
134  Q_Q(FileCopyJob);
135  if (!m_move) {
136  JobPrivate::emitCopying(q, m_src, m_dest);
137  } else {
138  JobPrivate::emitMoving(q, m_src, m_dest);
139  }
140 
141  if (m_move) {
142  // The if() below must be the same as the one in startBestCopyMethod
143  if ((m_src.scheme() == m_dest.scheme()) &&
144  (m_src.host() == m_dest.host()) &&
145  (m_src.port() == m_dest.port()) &&
146  (m_src.userName() == m_dest.userName()) &&
147  (m_src.password() == m_dest.password())) {
148  startRenameJob(m_src);
149  return;
150  } else if (m_src.isLocalFile() && KProtocolManager::canRenameFromFile(m_dest)) {
151  startRenameJob(m_dest);
152  return;
153  } else if (m_dest.isLocalFile() && KProtocolManager::canRenameToFile(m_src)) {
154  startRenameJob(m_src);
155  return;
156  }
157  // No fast-move available, use copy + del.
158  }
159  startBestCopyMethod();
160 }

162 void FileCopyJobPrivate::startBestCopyMethod()
163 {
164  if ((m_src.scheme() == m_dest.scheme()) &&
165  (m_src.host() == m_dest.host()) &&
166  (m_src.port() == m_dest.port()) &&
167  (m_src.userName() == m_dest.userName()) &&
168  (m_src.password() == m_dest.password())) {
169  startCopyJob();
170  } else if (m_src.isLocalFile() && KProtocolManager::canCopyFromFile(m_dest)) {
171  startCopyJob(m_dest);
172  } else if (m_dest.isLocalFile() && KProtocolManager::canCopyToFile(m_src) &&
173  !KIO::Scheduler::isSlaveOnHoldFor(m_src)) {
174  startCopyJob(m_src);
175  } else {
176  startDataPump();

347 void FileCopyJobPrivate::slotCanResume(KIO::Job *job, KIO::filesize_t offset)
348 {
349  Q_Q(FileCopyJob);
350  if (job == m_putJob || job == m_copyJob) {
351  //qDebug() << "'can resume' from PUT job. offset=" << KIO::number(offset);
352  if (offset) {
353  RenameDialog_Result res = R_RESUME;
354 
355  if (!KProtocolManager::autoResume() && !(m_flags & Overwrite) && m_uiDelegateExtension) {
356  QString newPath;
357  KIO::Job *job = (q->parentJob()) ? q->parentJob() : q;
358  // Ask confirmation about resuming previous transfer
359  res = m_uiDelegateExtension->askFileRename(
360  job, i18n("File Already Exists"),
361  m_src,
362  m_dest,
363  RenameDialog_Options(RenameDialog_Overwrite | RenameDialog_Resume | RenameDialog_NoRename), newPath,
364  m_sourceSize, offset);
365  }
366 
367  if (res == R_OVERWRITE || (m_flags & Overwrite)) {
368  offset = 0;
369  } else if (res == R_CANCEL) {
370  if (job == m_putJob) {
371  m_putJob->kill(FileCopyJob::Quietly);
372  q->removeSubjob(m_putJob);
373  m_putJob = 0;
374  } else {
375  m_copyJob->kill(FileCopyJob::Quietly);
376  q->removeSubjob(m_copyJob);
377  m_copyJob = 0;
378  }
379  q->setError(ERR_USER_CANCELED);
380  q->emitResult();
381  return;
382  }
383  } else {
384  m_resumeAnswerSent = true; // No need for an answer
385  }
386 
387  if (job == m_putJob) {
388  m_getJob = KIO::get(m_src, NoReload, HideProgressInfo /* no GUI */);
389  //qDebug() << "m_getJob=" << m_getJob << m_src;
390  m_getJob->addMetaData(QStringLiteral("errorPage"), QStringLiteral("false"));
391  m_getJob->addMetaData(QStringLiteral("AllowCompressedPage"), QStringLiteral("false"));
392  // Set size in subjob. This helps if the slave doesn't emit totalSize.
393  if (m_sourceSize != (KIO::filesize_t) - 1) {
394  m_getJob->setTotalAmount(KJob::Bytes, m_sourceSize);
395  }
396  if (offset) {
397  //qDebug() << "Setting metadata for resume to" << (unsigned long) offset;
398  m_getJob->addMetaData(QStringLiteral("range-start"), KIO::number(offset));
399 
400  // Might or might not get emitted
401  q->connect(m_getJob, SIGNAL(canResume(KIO::Job*,KIO::filesize_t)),
402  SLOT(slotCanResume(KIO::Job*,KIO::filesize_t)));
403  }
404  jobSlave(m_putJob)->setOffset(offset);
405 
406  m_putJob->d_func()->internalSuspend();
407  q->addSubjob(m_getJob);
408  connectSubjob(m_getJob); // Progress info depends on get
409  m_getJob->d_func()->internalResume(); // Order a beer
410 
411  q->connect(m_getJob, SIGNAL(data(KIO::Job*,QByteArray)),
412  SLOT(slotData(KIO::Job*,QByteArray)));
413  q->connect(m_getJob, SIGNAL(mimetype(KIO::Job*,QString)),
414  SLOT(slotMimetype(KIO::Job*,QString)));
415  } else { // copyjob
416  jobSlave(m_copyJob)->sendResumeAnswer(offset != 0);
417  }
418  } else if (job == m_getJob) {
419  // Cool, the get job said ok, we can resume
420  m_canResume = true;
421  //qDebug() << "'can resume' from the GET job -> we can resume";
422 
423  jobSlave(m_getJob)->setOffset(jobSlave(m_putJob)->offset());
424  } else {
425  qWarning() << "unknown job=" << job << "m_getJob=" << m_getJob << "m_putJob=" << m_putJob;
426  }
427 }

429 void FileCopyJobPrivate::slotData(KIO::Job *, const QByteArray &data)
430 {
431  //qDebug() << "data size:" << data.size();
432  Q_ASSERT(m_putJob);
433  if (!m_putJob) {
434  return; // Don't crash
435  }
436  m_getJob->d_func()->internalSuspend();
437  m_putJob->d_func()->internalResume(); // Drink the beer
438  m_buffer += data;
439 
440  // On the first set of data incoming, we tell the "put" slave about our
441  // decision about resuming
442  if (!m_resumeAnswerSent) {
443  m_resumeAnswerSent = true;
444  //qDebug() << "(first time) -> send resume answer " << m_canResume;
445  jobSlave(m_putJob)->sendResumeAnswer(m_canResume);
446  }
447 }

449 void FileCopyJobPrivate::slotDataReq(KIO::Job *, QByteArray &data)
450 {
451  Q_Q(FileCopyJob);
452  //qDebug();
453  if (!m_resumeAnswerSent && !m_getJob) {
454  // This can't happen
455  q->setError(ERR_INTERNAL);
456  q->setErrorText(QStringLiteral("'Put' job did not send canResume or 'Get' job did not send data!"));
457  m_putJob->kill(FileCopyJob::Quietly);
458  q->removeSubjob(m_putJob);
459  m_putJob = 0;
460  q->emitResult();
461  return;
462  }
463  if (m_getJob) {
464  m_getJob->d_func()->internalResume(); // Order more beer
465  m_putJob->d_func()->internalSuspend();
466  }
467  data = m_buffer;
468  m_buffer = QByteArray();
469 }

477 void FileCopyJob::slotResult(KJob *job)
478 {
479  Q_D(FileCopyJob);
480  //qDebug() << "this=" << this << "job=" << job;
481  removeSubjob(job);
482  // Did job have an error ?
483  if (job->error()) {
484  if ((job == d->m_moveJob) && (job->error() == ERR_UNSUPPORTED_ACTION)) {
485  d->m_moveJob = 0;
486  d->startBestCopyMethod();
487  return;
488  } else if ((job == d->m_copyJob) && (job->error() == ERR_UNSUPPORTED_ACTION)) {
489  d->m_copyJob = 0;
490  d->startDataPump();
491  return;
492  } else if (job == d->m_getJob) {
493  d->m_getJob = 0L;
494  if (d->m_putJob) {
495  d->m_putJob->kill(Quietly);
496  removeSubjob(d->m_putJob);
497  }
498  } else if (job == d->m_putJob) {
499  d->m_putJob = 0L;
500  if (d->m_getJob) {
501  d->m_getJob->kill(Quietly);
502  removeSubjob(d->m_getJob);
503  }
504  }
505  setError(job->error());
506  setErrorText(job->errorText());
507  emitResult();
508  return;
509  }
510 
511  if (d->m_mustChmod) {
512  // If d->m_permissions == -1, keep the default permissions
513  if (d->m_permissions != -1) {
514  d->m_chmodJob = chmod(d->m_dest, d->m_permissions);
515  addSubjob(d->m_chmodJob);
516  }
517  d->m_mustChmod = false;
518  }
519 
520  if (job == d->m_moveJob) {
521  d->m_moveJob = 0; // Finished
522  }
523 
524  if (job == d->m_copyJob) {
525  d->m_copyJob = 0;
526  if (d->m_move) {
527  d->m_delJob = file_delete(d->m_src, HideProgressInfo/*no GUI*/); // Delete source
528  addSubjob(d->m_delJob);
529  }
530  }
531 
532  if (job == d->m_getJob) {
533  //qDebug() << "m_getJob finished";
534  d->m_getJob = 0; // No action required
535  if (d->m_putJob) {
536  d->m_putJob->d_func()->internalResume();
537  }
538  }
539 
540  if (job == d->m_putJob) {
541  //qDebug() << "m_putJob finished";
542  d->m_putJob = 0;
543  if (d->m_getJob) {
544  // The get job is still running, probably after emitting data(QByteArray())
545  // and before we receive its finished().
546  d->m_getJob->d_func()->internalResume();
547  }
548  if (d->m_move) {
549  d->m_delJob = file_delete(d->m_src, HideProgressInfo/*no GUI*/); // Delete source
550  addSubjob(d->m_delJob);
551  }
552  }
553 
554  if (job == d->m_delJob) {
555  d->m_delJob = 0; // Finished
556  }
557 
558  if (job == d->m_chmodJob) {
559  d->m_chmodJob = 0; // Finished
560  }
561 
562  if (!hasSubjobs()) {
563  emitResult();
564  }
565 }
