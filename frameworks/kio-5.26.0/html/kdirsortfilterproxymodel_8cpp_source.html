52 int KDirSortFilterProxyModel::KDirSortFilterProxyModelPrivate::compare(const QString &a,
53  const QString &b,
54  Qt::CaseSensitivity caseSensitivity)
55 {
56  int result;
57 
58  if (m_naturalSorting) {
59  m_collator.setCaseSensitivity(caseSensitivity);
60  result = m_collator.compare(a, b);
61  } else {
62  result = QString::compare(a, b, caseSensitivity);
63  }
64 
65  if (caseSensitivity == Qt::CaseSensitive || result != 0) {
66  // Only return the result, if the strings are not equal. If they are equal by a case insensitive
67  // comparison, still a deterministic sort order is required. A case sensitive
68  // comparison is done as fallback.
69  return result;
70  }
71 
72  return QString::compare(a, b, Qt::CaseSensitive);
73 }

146 bool KDirSortFilterProxyModel::subSortLessThan(const QModelIndex &left,
147  const QModelIndex &right) const
148 {
149  KDirModel *dirModel = static_cast<KDirModel *>(sourceModel());
150 
151  const KFileItem leftFileItem = dirModel->itemForIndex(left);
152  const KFileItem rightFileItem = dirModel->itemForIndex(right);
153 
154  const bool isLessThan = (sortOrder() == Qt::AscendingOrder);
155 
156  // Folders go before files if the corresponding setting is set.
157  if (d->m_sortFoldersFirst) {
158  const bool leftItemIsDir = leftFileItem.isDir();
159  const bool rightItemIsDir = rightFileItem.isDir();
160  if (leftItemIsDir && !rightItemIsDir) {
161  return isLessThan;
162  } else if (!leftItemIsDir && rightItemIsDir) {
163  return !isLessThan;
164  }
165  }
166 
167  // Hidden elements go before visible ones.
168  const bool leftItemIsHidden = leftFileItem.isHidden();
169  const bool rightItemIsHidden = rightFileItem.isHidden();
170  if (leftItemIsHidden && !rightItemIsHidden) {
171  return isLessThan;
172  } else if (!leftItemIsHidden && rightItemIsHidden) {
173  return !isLessThan;
174  }
175 
176  switch (left.column()) {
177  case KDirModel::Name: {
178  int result = d->compare(leftFileItem.text(), rightFileItem.text(), sortCaseSensitivity());
179  if (result == 0) {
180  // KFileItem::text() may not be unique in case UDS_DISPLAY_NAME is used
181  result = d->compare(leftFileItem.name(sortCaseSensitivity() == Qt::CaseInsensitive),
182  rightFileItem.name(sortCaseSensitivity() == Qt::CaseInsensitive),
183  sortCaseSensitivity());
184  if (result == 0) {
185  // If KFileItem::text() is also not unique most probably a search protocol is used
186  // that allows showing the same file names from different directories
187  result = d->compare(leftFileItem.url().toString(), rightFileItem.url().toString(), sortCaseSensitivity());
188  }
189  }
190 
191  return result < 0;
192  }
193 
194  case KDirModel::Size: {
195  // If we have two folders, what we have to measure is the number of
196  // items that contains each other
197  if (leftFileItem.isDir() && rightFileItem.isDir()) {
198  QVariant leftValue = dirModel->data(left, KDirModel::ChildCountRole);
199  int leftCount = (leftValue.type() == QVariant::Int) ? leftValue.toInt() : KDirModel::ChildCountUnknown;
200 
201  QVariant rightValue = dirModel->data(right, KDirModel::ChildCountRole);
202  int rightCount = (rightValue.type() == QVariant::Int) ? rightValue.toInt() : KDirModel::ChildCountUnknown;
203 
204  // In the case they two have the same child items, we sort them by
205  // their names. So we have always everything ordered. We also check
206  // if we are taking in count their cases.
207  if (leftCount == rightCount) {
208  return d->compare(leftFileItem.text(), rightFileItem.text(), sortCaseSensitivity()) < 0;
209  }
210 
211  // If one of them has unknown child items, place them on the end. If we
212  // were comparing two unknown childed items, the previous comparation
213  // sorted them by QCollator between them. This case is when we
214  // have an unknown childed item, and another known.
215  if (leftCount == KDirModel::ChildCountUnknown) {
216  return false;
217  }
218 
219  if (rightCount == KDirModel::ChildCountUnknown) {
220  return true;
221  }
222 
223  // If they had different number of items, we sort them depending
224  // on how many items had each other.
225  return leftCount < rightCount;
226  }
227 
228  // If what we are measuring is two files and they have the same size,
229  // sort them by their file names.
230  if (leftFileItem.size() == rightFileItem.size()) {
231  return d->compare(leftFileItem.text(), rightFileItem.text(), sortCaseSensitivity()) < 0;
232  }
233 
234  // If their sizes are different, sort them by their sizes, as expected.
235  return leftFileItem.size() < rightFileItem.size();
236  }
237 
238  case KDirModel::ModifiedTime: {
239  QDateTime leftModifiedTime = leftFileItem.time(KFileItem::ModificationTime).toLocalTime();
240  QDateTime rightModifiedTime = rightFileItem.time(KFileItem::ModificationTime).toLocalTime();
241 
242  if (leftModifiedTime == rightModifiedTime) {
243  return d->compare(leftFileItem.text(), rightFileItem.text(), sortCaseSensitivity()) < 0;
244  }
245 
246  return leftModifiedTime < rightModifiedTime;
247  }
248 
249  case KDirModel::Permissions: {
250  const int leftPermissions = leftFileItem.permissions();
251  const int rightPermissions = rightFileItem.permissions();
252 
253  if (leftPermissions == rightPermissions) {
254  return d->compare(leftFileItem.text(), rightFileItem.text(), sortCaseSensitivity()) < 0;
255  }
256 
257  return leftPermissions > rightPermissions;
258  }
259 
260  case KDirModel::Owner: {
261  if (leftFileItem.user() == rightFileItem.user()) {
262  return d->compare(leftFileItem.text(), rightFileItem.text(), sortCaseSensitivity()) < 0;
263  }
264 
265  return d->compare(leftFileItem.user(), rightFileItem.user()) < 0;
266  }
267 
268  case KDirModel::Group: {
269  if (leftFileItem.group() == rightFileItem.group()) {
270  return d->compare(leftFileItem.text(), rightFileItem.text(), sortCaseSensitivity()) < 0;
271  }
272 
273  return d->compare(leftFileItem.group(), rightFileItem.group()) < 0;
274  }
275 
276  case KDirModel::Type: {
277  if (leftFileItem.mimetype() == rightFileItem.mimetype()) {
278  return d->compare(leftFileItem.text(), rightFileItem.text(), sortCaseSensitivity()) < 0;
279  }
280 
281  return d->compare(leftFileItem.mimeComment(), rightFileItem.mimeComment()) < 0;
282  }
283 
284  }
285 
286  // We have set a SortRole and trust the ProxyModel to do
287  // the right thing for now.
288  return KCategorizedSortFilterProxyModel::subSortLessThan(left, right);
289 }
