56 static bool containsScheme(const char input[], int start, int end)
57 {
58  // skip any comma or white space
59  while (start < end && isWhiteSpaceOrComma(input[start])) {
60  start++;
61  }
62 
63  while (start < end) {
64  if (isWhiteSpace(input[start])) {
65  return true;
66  }
67  start++;
68  }
69 
70  return false;
71 }

76 static QList<QByteArray> parseChallenge(QByteArray &ba, QByteArray *scheme, QByteArray *nextAuth = 0)
77 {
78  QList<QByteArray> values;
79  const char *b = ba.constData();
80  int len = ba.count();
81  int start = 0, end = 0, pos = 0, pos2 = 0;
82 
83  // parse scheme
84  while (start < len && isWhiteSpaceOrComma(b[start])) {
85  start++;
86  }
87  end = start;
88  while (end < len && !isWhiteSpace(b[end])) {
89  end++;
90  }
91 
92  // drop empty stuff from the given string, it would have to be skipped over and over again
93  if (start != 0) {
94  ba = ba.mid(start);
95  end -= start;
96  len -= start;
97  start = 0;
98  b = ba.constData();
99  }
100  Q_ASSERT(scheme);
101  *scheme = ba.left(end);
102 
103  while (end < len) {
104  start = end;
105  while (end < len && b[end] != '=') {
106  end++;
107  }
108  pos = end; // save the end position
109  while (end - 1 > start && isWhiteSpace(b[end - 1])) { // trim whitespace
110  end--;
111  }
112  pos2 = start;
113  while (pos2 < end && isWhiteSpace(b[pos2])) { // skip whitespace
114  pos2++;
115  }
116  if (containsScheme(b, start, end) || (b[pos2] == ',' && b[pos] != '=' && pos == len)) {
117  if (nextAuth) {
118  *nextAuth = QByteArray(b + start);
119  }
120  break; // break on start of next scheme.
121  }
122  while (start < len && isWhiteSpaceOrComma(b[start])) {
123  start++;
124  }
125  values.append(QByteArray(b + start, end - start));
126  end = pos; // restore the end position
127  if (end == len) {
128  break;
129  }
130 
131  // parse value
132  start = end + 1; //skip '='
133  while (start < len && isWhiteSpace(b[start])) {
134  start++;
135  }
136 
137  if (b[start] == '"') {
138  //quoted string
139  bool hasBs = false;
140  bool hasErr = false;
141  end = ++start;
142  while (end < len) {
143  if (b[end] == '\\') {
144  end++;
145  if (end + 1 >= len) {
146  hasErr = true;
147  break;
148  } else {
149  hasBs = true;
150  end++;
151  }
152  } else if (b[end] == '"') {
153  break;
154  } else {
155  end++;
156  }
157  }
158  if (hasErr || (end == len)) {
159  // remove the key we already inserted
160  // qDebug() << "error in quoted text for key" << values.last();
161  values.removeLast();
162  break;
163  }
164  QByteArray value = QByteArray(b + start, end - start);
165  if (hasBs) {
166  // skip over the next character, it might be an escaped backslash
167  int i = -1;
168  while ((i = value.indexOf('\\', i + 1)) >= 0) {
169  value.remove(i, 1);
170  }
171  }
172  values.append(value);
173  end++;
174  } else {
175  //unquoted string
176  end = start;
177  while (end < len && b[end] != ',' && !isWhiteSpace(b[end])) {
178  end++;
179  }
180  values.append(QByteArray(b + start, end - start));
181  }
182 
183  //the quoted string has ended, but only a comma ends a key-value pair
184  while (end < len && isWhiteSpace(b[end])) {
185  end++;
186  }
187 
188  // garbage, here should be end or field delimiter (comma)
189  if (end < len && b[end] != ',') {
190  // qDebug() << "unexpected character" << b[end] << "found in WWW-authentication header where token boundary (,) was expected";
191  break;
192  }
193  }
194  // ensure every key has a value
195  // WARNING: Do not remove the > 1 check or parsing a Type 1 NTLM
196  // authentication challenge will surely fail.
197  if (values.count() > 1 && values.count() % 2) {
198  values.removeLast();
199  }
200  return values;
201 }

223 QByteArray KAbstractHttpAuthentication::bestOffer(const QList<QByteArray> &offers)
224 {
225  // choose the most secure auth scheme offered
226  QByteArray negotiateOffer;
227  QByteArray digestOffer;
228  QByteArray ntlmOffer;
229  QByteArray basicOffer;
230  Q_FOREACH (const QByteArray &offer, offers) {
231  const QByteArray scheme = offer.mid(0, offer.indexOf(' ')).toLower();
232 #if HAVE_LIBGSSAPI
233  if (scheme == "negotiate") { // krazy:exclude=strings
234  negotiateOffer = offer;
235  } else
236 #endif
237  if (scheme == "digest") { // krazy:exclude=strings
238  digestOffer = offer;
239  } else if (scheme == "ntlm") { // krazy:exclude=strings
240  ntlmOffer = offer;
241  } else if (scheme == "basic") { // krazy:exclude=strings
242  basicOffer = offer;
243  }
244  }
245 
246  if (!negotiateOffer.isEmpty()) {
247  return negotiateOffer;
248  }
249 
250  if (!digestOffer.isEmpty()) {
251  return digestOffer;
252  }
253 
254  if (!ntlmOffer.isEmpty()) {
255  return ntlmOffer;
256  }
257 
258  return basicOffer; //empty or not...

279 QList< QByteArray > KAbstractHttpAuthentication::splitOffers(const QList< QByteArray > &offers)
280 {
281  // first detect if one entry may contain multiple offers
282  QList<QByteArray> alloffers;
283  foreach (QByteArray offer, offers) {
284  QByteArray scheme, cont;
285 
286  parseChallenge(offer, &scheme, &cont);
287 
288  while (!cont.isEmpty()) {
289  offer.chop(cont.length());
290  alloffers << offer;
291  offer = cont;
292  cont.clear();
293  parseChallenge(offer, &scheme, &cont);
294  }
295  alloffers << offer;
296  }
297  return alloffers;
298 }

434 //calculateResponse() from the original HTTPProtocol
435 static QByteArray calculateResponse(const DigestAuthInfo &info, const QUrl &resource)
436 {
437  QCryptographicHash md(QCryptographicHash::Md5);
438  QByteArray HA1;
439  QByteArray HA2;
440 
441  // Calculate H(A1)
442  QByteArray authStr = info.username;
443  authStr += ':';
444  authStr += info.realm;
445  authStr += ':';
446  authStr += info.password;
447  md.addData(authStr);
448 
449  if (info.algorithm.toLower() == "md5-sess") {
450  authStr = md.result().toHex();
451  authStr += ':';
452  authStr += info.nonce;
453  authStr += ':';
454  authStr += info.cnonce;
455  md.reset();
456  md.addData(authStr);
457  }
458  HA1 = md.result().toHex();
459 
460  // qDebug() << "A1 => " << HA1;
461 
462  // Calcualte H(A2)
463  authStr = info.method;
464  authStr += ':';
465  authStr += resource.path(QUrl::FullyEncoded).toLatin1();
466  if (resource.hasQuery()) {
467  authStr += '?' + resource.query(QUrl::FullyEncoded).toLatin1();
468  }
469  if (info.qop == "auth-int") {
470  authStr += ':';
471  md.reset();
472  md.addData(info.entityBody);
473  authStr += md.result().toHex();
474  }
475  md.reset();
476  md.addData(authStr);
477  HA2 = md.result().toHex();
478 
479  // qDebug() << "A2 => " << HA2;
480 
481  // Calcualte the response.
482  authStr = HA1;
483  authStr += ':';
484  authStr += info.nonce;
485  authStr += ':';
486  if (!info.qop.isEmpty()) {
487  authStr += info.nc;
488  authStr += ':';
489  authStr += info.cnonce;
490  authStr += ':';
491  authStr += info.qop;
492  authStr += ':';
493  }
494  authStr += HA2;
495  md.reset();
496  md.addData(authStr);
497 
498  const QByteArray response = md.result().toHex();
499  // qDebug() << "Response =>" << response;
500  return response;
501 }

503 void KHttpDigestAuthentication::generateResponse(const QString &user, const QString &password)
504 {
505  generateResponseCommon(user, password);
506  if (m_isError) {
507  return;
508  }
509 
510  // magic starts here (this part is slightly modified from the original in HTTPProtocol)
511 
512  DigestAuthInfo info;
513 
514  info.username = m_username.toLatin1(); //### charset breakage
515  info.password = m_password.toLatin1(); //###
516 
517  // info.entityBody = p; // FIXME: send digest of data for POST action ??
518  info.realm = "";
519  info.nonce = "";
520  info.qop = "";
521 
522  // cnonce is recommended to contain about 64 bits of entropy
523 #ifdef ENABLE_HTTP_AUTH_NONCE_SETTER
524  info.cnonce = m_nonce;
525 #else
526  info.cnonce = KRandom::randomString(16).toLatin1();
527 #endif
528 
529  // HACK: Should be fixed according to RFC 2617 section 3.2.2
530  info.nc = "00000001";
531 
532  // Set the method used...
533  info.method = m_httpMethod;
534 
535  // Parse the Digest response....
536  info.realm = valueForKey(m_challenge, "realm");
537 
538  info.algorithm = valueForKey(m_challenge, "algorithm");
539  if (info.algorithm.isEmpty()) {
540  info.algorithm = valueForKey(m_challenge, "algorith");
541  }
542  if (info.algorithm.isEmpty()) {
543  info.algorithm = "MD5";
544  }
545 
546  Q_FOREACH (const QByteArray &path, valueForKey(m_challenge, "domain").split(' ')) {
547  QUrl u = m_resource.resolved(QUrl(path));
548  if (u.isValid()) {
549  info.digestURIs.append(u);
550  }
551  }
552 
553  info.nonce = valueForKey(m_challenge, "nonce");
554  QByteArray opaque = valueForKey(m_challenge, "opaque");
555  info.qop = valueForKey(m_challenge, "qop");
556 
557  // NOTE: Since we do not have access to the entity body, we cannot support
558  // the "auth-int" qop value ; so if the server returns a comma separated
559  // list of qop values, prefer "auth".See RFC 2617 sec 3.2.2 for the details.
560  // If "auth" is not present or it is set to "auth-int" only, then we simply
561  // print a warning message and disregard the qop option altogether.
562  if (info.qop.contains(',')) {
563  const QList<QByteArray> values = info.qop.split(',');
564  if (info.qop.contains("auth")) {
565  info.qop = "auth";
566  } else {
567  qWarning() << "Unsupported digest authentication qop parameters:" << values;
568  info.qop.clear();
569  }
570  } else if (info.qop == "auth-int") {
571  qWarning() << "Unsupported digest authentication qop parameter:" << info.qop;
572  info.qop.clear();
573  }
574 
575  if (info.realm.isEmpty() || info.nonce.isEmpty()) {
576  // ### proper error return
577  m_isError = true;
578  return;
579  }
580 
581  // If the "domain" attribute was not specified and the current response code
582  // is authentication needed, add the current request url to the list over which
583  // this credential can be automatically applied.
584  if (info.digestURIs.isEmpty() /*###&& (m_request.responseCode == 401 || m_request.responseCode == 407)*/) {
585  info.digestURIs.append(m_resource);
586  } else {
587  // Verify whether or not we should send a cached credential to the
588  // server based on the stored "domain" attribute...
589  bool send = true;
590 
591  // Determine the path of the request url...
592  QString requestPath = m_resource.adjusted(QUrl::RemoveFilename).path();
593  if (requestPath.isEmpty()) {
594  requestPath = QLatin1Char('/');
595  }
596 
597  Q_FOREACH (const QUrl &u, info.digestURIs) {
598  send &= (m_resource.scheme().toLower() == u.scheme().toLower());
599  send &= (m_resource.host().toLower() == u.host().toLower());
600 
601  if (m_resource.port() > 0 && u.port() > 0) {
602  send &= (m_resource.port() == u.port());
603  }
604 
605  QString digestPath = u.adjusted(QUrl::RemoveFilename).path();
606  if (digestPath.isEmpty()) {
607  digestPath = QLatin1Char('/');
608  }
609 
610  send &= (requestPath.startsWith(digestPath));
611 
612  if (send) {
613  break;
614  }
615  }
616 
617  if (!send) {
618  m_isError = true;
619  return;
620  }
621  }
622 
623  // qDebug() << "RESULT OF PARSING:";
624  // qDebug() << " algorithm: " << info.algorithm;
625  // qDebug() << " realm: " << info.realm;
626  // qDebug() << " nonce: " << info.nonce;
627  // qDebug() << " opaque: " << opaque;
628  // qDebug() << " qop: " << info.qop;
629 
630  // Calculate the response...
631  const QByteArray response = calculateResponse(info, m_resource);
632 
633  QByteArray auth = "Digest username=\"";
634  auth += info.username;
635 
636  auth += "\", realm=\"";
637  auth += info.realm;
638  auth += "\"";
639 
640  auth += ", nonce=\"";
641  auth += info.nonce;
642 
643  auth += "\", uri=\"";
644  auth += m_resource.path(QUrl::FullyEncoded).toLatin1();
645  if (m_resource.hasQuery()) {
646  auth += '?' + m_resource.query(QUrl::FullyEncoded).toLatin1();
647  }
648 
649  if (!info.algorithm.isEmpty()) {
650  auth += "\", algorithm=";
651  auth += info.algorithm;
652  }
653 
654  if (!info.qop.isEmpty()) {
655  auth += ", qop=";
656  auth += info.qop;
657  auth += ", cnonce=\"";
658  auth += info.cnonce;
659  auth += "\", nc=";
660  auth += info.nc;
661  }
662 
663  auth += ", response=\"";
664  auth += response;
665  if (!opaque.isEmpty()) {
666  auth += "\", opaque=\"";
667  auth += opaque;
668  }
669  auth += "\"\r\n";
670 
671  // magic ends here
672  // note that auth already contains \r\n
673  m_headerFragment = auth;
674 }

688 void KHttpNtlmAuthentication::setChallenge(const QByteArray &c, const QUrl &resource,

700  }
701  // The type 1 message we're going to send needs no credentials;
702  // they come later in the type 3 message.
703  m_needCredentials = !m_challenge.isEmpty();

715 void KHttpNtlmAuthentication::generateResponse(const QString &_user, const QString &password)
716 {
717  generateResponseCommon(_user, password);
718  if (m_isError) {
719  return;
720  }
721 
722  QByteArray buf;
723 
724  if (m_challenge.isEmpty()) {
725  m_finalAuthStage = false;
726  // first, send type 1 message (with empty domain, workstation..., but it still works)
727  switch (m_stage1State) {
728  case Init:
729  if (!KNTLM::getNegotiate(buf)) {
730  qWarning() << "Error while constructing Type 1 NTLMv1 authentication request";
731  m_isError = true;
732  return;
733  }
734  m_stage1State = SentNTLMv1;
735  break;
736  case SentNTLMv1:
737  if (!KNTLM::getNegotiate(buf, QString(), QString(), KNTLM::Negotiate_NTLM2_Key
738  | KNTLM::Negotiate_Always_Sign | KNTLM::Negotiate_Unicode
739  | KNTLM::Request_Target | KNTLM::Negotiate_NTLM)) {
740  qWarning() << "Error while constructing Type 1 NTLMv2 authentication request";
741  m_isError = true;
742  return;
743  }
744  m_stage1State = SentNTLMv2;
745  break;
746  default:
747  qWarning() << "Error - Type 1 NTLM already sent - no Type 2 response received.";
748  m_isError = true;
749  return;
750  }
751  } else {
752  m_finalAuthStage = true;
753  // we've (hopefully) received a valid type 2 message: send type 3 message as last step
754  QString user, domain;
755  if (m_username.contains(QLatin1Char('\\'))) {
756  domain = m_username.section(QLatin1Char('\\'), 0, 0);
757  user = m_username.section(QLatin1Char('\\'), 1);
758  } else {
759  user = m_username;
760  }
761 
762  m_forceKeepAlive = true;
763  const QByteArray challenge = QByteArray::fromBase64(m_challenge[0]);
764 
765  KNTLM::AuthFlags flags = KNTLM::Add_LM;
766  if ((!m_config || !m_config->readEntry("EnableNTLMv2Auth", false)) && (m_stage1State != SentNTLMv2)) {
767  flags |= KNTLM::Force_V1;
768  }
769 
770  if (!KNTLM::getAuth(buf, challenge, user, m_password, domain, QStringLiteral("WORKSTATION"), flags)) {
771  qWarning() << "Error while constructing Type 3 NTLM authentication request";
772  m_isError = true;
773  return;
774  }
775  }
776 
777  m_headerFragment = "NTLM ";
778  m_headerFragment += buf.toBase64();
779  m_headerFragment += "\r\n";
780 
781  return;
782 }

829 void KHttpNegotiateAuthentication::generateResponse(const QString &user, const QString &password)
830 {
831  generateResponseCommon(user, password);
832  if (m_isError) {
833  return;
834  }
835 
836  OM_uint32 major_status, minor_status;
837  gss_buffer_desc input_token = GSS_C_EMPTY_BUFFER;
838  gss_buffer_desc output_token = GSS_C_EMPTY_BUFFER;
839  gss_name_t server;
840  gss_ctx_id_t ctx;
841  gss_OID mech_oid;
842  static gss_OID_desc krb5_oid_desc = {9, (void *) "\x2a\x86\x48\x86\xf7\x12\x01\x02\x02"};
843  static gss_OID_desc spnego_oid_desc = {6, (void *) "\x2b\x06\x01\x05\x05\x02"};
844  gss_OID_set mech_set;
845  gss_OID tmp_oid;
846 
847  ctx = GSS_C_NO_CONTEXT;
848  mech_oid = &krb5_oid_desc;
849 
850  // see whether we can use the SPNEGO mechanism
851  major_status = gss_indicate_mechs(&minor_status, &mech_set);
852  if (GSS_ERROR(major_status)) {
853  qDebug() << "gss_indicate_mechs failed:" << gssError(major_status, minor_status);
854  } else {
855  for (uint i = 0; i < mech_set->count; i++) {
856  tmp_oid = &mech_set->elements[i];
857  if (tmp_oid->length == spnego_oid_desc.length &&
858  !memcmp(tmp_oid->elements, spnego_oid_desc.elements, tmp_oid->length)) {
859  // qDebug() << "found SPNEGO mech";
860  mech_oid = &spnego_oid_desc;
861  break;
862  }
863  }
864  gss_release_oid_set(&minor_status, &mech_set);
865  }
866 
867  // the service name is "HTTP/f.q.d.n"
868  QByteArray servicename = "HTTP@";
869  servicename += m_resource.host().toLatin1();
870 
871  input_token.value = (void *)servicename.data();
872  input_token.length = servicename.length() + 1;
873 
874  major_status = gss_import_name(&minor_status, &input_token,
875  GSS_C_NT_HOSTBASED_SERVICE, &server);
876 
877  input_token.value = NULL;
878  input_token.length = 0;
879 
880  if (GSS_ERROR(major_status)) {
881  qDebug() << "gss_import_name failed:" << gssError(major_status, minor_status);
882  m_isError = true;
883  return;
884  }
885 
886  OM_uint32 req_flags;
887  if (m_config && m_config->readEntry("DelegateCredentialsOn", false)) {
888  req_flags = GSS_C_DELEG_FLAG;
889  } else {
890  req_flags = 0;
891  }
892 
893  // GSSAPI knows how to get the credentials its own way, so don't ask for any
894  major_status = gss_init_sec_context(&minor_status, GSS_C_NO_CREDENTIAL,
895  &ctx, server, mech_oid,
896  req_flags, GSS_C_INDEFINITE,
897  GSS_C_NO_CHANNEL_BINDINGS,
898  GSS_C_NO_BUFFER, NULL, &output_token,
899  NULL, NULL);
900 
901  if (GSS_ERROR(major_status) || (output_token.length == 0)) {
902  qDebug() << "gss_init_sec_context failed:" << gssError(major_status, minor_status);
903  gss_release_name(&minor_status, &server);
904  if (ctx != GSS_C_NO_CONTEXT) {
905  gss_delete_sec_context(&minor_status, &ctx, GSS_C_NO_BUFFER);
906  ctx = GSS_C_NO_CONTEXT;
907  }
908  m_isError = true;
909  return;
910  }
911 
912  m_headerFragment = "Negotiate ";
913  m_headerFragment += QByteArray::fromRawData(static_cast<const char *>(output_token.value),
914  output_token.length).toBase64();
915  m_headerFragment += "\r\n";
916 
917  // free everything
918  gss_release_name(&minor_status, &server);
919  if (ctx != GSS_C_NO_CONTEXT) {
920  gss_delete_sec_context(&minor_status, &ctx, GSS_C_NO_BUFFER);
921  ctx = GSS_C_NO_CONTEXT;
922  }
923  gss_release_buffer(&minor_status, &output_token);
924 }
