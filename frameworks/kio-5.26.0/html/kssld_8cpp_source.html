91 void KSSLD::setRule(const KSslCertificateRule &rule)
92 {
93  if (rule.hostName().isEmpty()) {
94  return;
95  }
96  KConfigGroup group = d->config.group(rule.certificate().digest().toHex());
97 
98  QStringList sl;
99 
100  QString dtString = QStringLiteral("ExpireUTC ");
101  dtString.append(rule.expiryDateTime().toString(Qt::ISODate));
102  sl.append(dtString);
103 
104  if (rule.isRejected()) {
105  sl.append(QStringLiteral("Reject"));
106  } else {
107  foreach (KSslError::Error e, rule.ignoredErrors()) {
108  sl.append(d->sslErrorToString.value(e));
109  }
110  }
111 
112  if (!group.hasKey("CertificatePEM")) {
113  group.writeEntry("CertificatePEM", rule.certificate().toPem());
114  }
115 #ifdef PARANOIA
116  else if (group.readEntry("CertificatePEM") != rule.certificate().toPem()) {
117  return;
118  }
119 #endif
120  group.writeEntry(rule.hostName(), sl);
121  group.sync();
122 }

139 void KSSLD::pruneExpiredRules()
140 {
141  // expired rules are deleted when trying to load them, so we just try to load all rules.
142  // be careful about iterating over KConfig(Group) while changing it
143  foreach (const QString &groupName, d->config.groupList()) {
144  QByteArray certDigest = groupName.toLatin1();
145  foreach (const QString &key, d->config.group(groupName).keyList()) {
146  if (key == QLatin1String("CertificatePEM")) {
147  continue;
148  }
149  KSslCertificateRule r = rule(QSslCertificate(certDigest), key);
150  }
151  }
152 }

154 // check a domain name with subdomains for well-formedness and count the dot-separated parts
155 static QString normalizeSubdomains(const QString &hostName, int *namePartsCount)
156 {
157  QString ret;
158  int partsCount = 0;
159  bool wasPrevDot = true; // -> allow no dot at the beginning and count first name part
160  const int length = hostName.length();
161  for (int i = 0; i < length; i++) {
162  const QChar c = hostName.at(i);
163  if (c == QLatin1Char('.')) {
164  if (wasPrevDot || (i + 1 == hostName.length())) {
165  // consecutive dots or a dot at the end are forbidden
166  partsCount = 0;
167  ret.clear();
168  break;
169  }
170  wasPrevDot = true;
171  } else {
172  if (wasPrevDot) {
173  partsCount++;
174  }
175  wasPrevDot = false;
176  }
177  ret.append(c);
178  }
179 
180  *namePartsCount = partsCount;
181  return ret;
182 }

184 KSslCertificateRule KSSLD::rule(const QSslCertificate &cert, const QString &hostName) const
185 {
186  const QByteArray certDigest = cert.digest().toHex();
187  KConfigGroup group = d->config.group(certDigest);
188 
189  KSslCertificateRule ret(cert, hostName);
190  bool foundHostName = false;
191 
192  int needlePartsCount;
193  QString needle = normalizeSubdomains(hostName, &needlePartsCount);
194 
195  // Find a rule for the hostname, either...
196  if (group.hasKey(needle)) {
197  // directly (host, site.tld, a.site.tld etc)
198  if (needlePartsCount >= 1) {
199  foundHostName = true;
200  }
201  } else {
202  // or with wildcards
203  // "tld" <- "*." and "site.tld" <- "*.tld" are not valid matches,
204  // "a.site.tld" <- "*.site.tld" is
205  while (--needlePartsCount >= 2) {
206  const int dotIndex = needle.indexOf(QLatin1Char('.'));
207  Q_ASSERT(dotIndex > 0); // if this fails normalizeSubdomains() failed
208  needle.remove(0, dotIndex - 1);
209  needle[0] = QChar::fromLatin1('*');
210  if (group.hasKey(needle)) {
211  foundHostName = true;
212  break;
213  }
214  needle.remove(0, 2); // remove "*."
215  }
216  }
217 
218  if (!foundHostName) {
219  //Don't make a rule with the failed wildcard pattern - use the original hostname.
220  return KSslCertificateRule(cert, hostName);
221  }
222 
223  //parse entry of the format "ExpireUTC <date>, Reject" or
224  //"ExpireUTC <date>, HostNameMismatch, ExpiredCertificate, ..."
225  QStringList sl = group.readEntry(needle, QStringList());
226 
227  QDateTime expiryDt;
228  // the rule is well-formed if it contains at least the expire date and one directive
229  if (sl.size() >= 2) {
230  QString dtString = sl.takeFirst();
231  if (dtString.startsWith(QLatin1String("ExpireUTC "))) {
232  dtString.remove(0, 10/* length of "ExpireUTC " */);
233  expiryDt = QDateTime::fromString(dtString, Qt::ISODate);
234  }
235  }
236 
237  if (!expiryDt.isValid() || expiryDt < QDateTime::currentDateTime()) {
238  //the entry is malformed or expired so we remove it
239  group.deleteEntry(needle);
240  //the group is useless once only the CertificatePEM entry left
241  if (group.keyList().size() < 2) {
242  group.deleteGroup();
243  }
244  return ret;
245  }
246 
247  QList<KSslError::Error> ignoredErrors;
248  bool isRejected = false;
249  foreach (const QString &s, sl) {
250  if (s == QLatin1String("Reject")) {
251  isRejected = true;
252  ignoredErrors.clear();
253  break;
254  }
255  if (!d->stringToSslError.contains(s)) {
256  continue;
257  }
258  ignoredErrors.append(d->stringToSslError.value(s));
259  }
260 
261  //Everything is checked and we can make ret valid
262  ret.setExpiryDateTime(expiryDt);
263  ret.setRejected(isRejected);
264  ret.setIgnoredErrors(ignoredErrors);
265  return ret;
266 }
