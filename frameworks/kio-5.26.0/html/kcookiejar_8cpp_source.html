69 static QString removeWeekday(const QString &value)
70 {
71  const int index = value.indexOf(QL1C(' '));
72  if (index > -1) {
73  int pos = 0;
74  const QString weekday = value.left(index);
75  for (int i = 1; i < 8; ++i) {
76  if (weekday.startsWith(QDate::shortDayName(i), Qt::CaseInsensitive) ||
77  weekday.startsWith(QDate::longDayName(i), Qt::CaseInsensitive)) {
78  pos = index + 1;
79  break;
80  }
81  }
82  if (pos > 0) {
83  return value.mid(pos);
84  }
85  }
86  return value;
87 }

89 static QDateTime parseDate(const QString &_value)
90 {
91  // Handle sites sending invalid weekday as part of the date. #298660
92  const QString value(removeWeekday(_value));
93 
94  // Check if expiration date matches RFC dates as specified under
95  // RFC 2616 sec 3.3.1 & RFC 6265 sec 4.1.1
96  QDateTime dt = QDateTime::fromString(value, Qt::RFC2822Date);
97 
98  if (!dt.isValid()) {
99  static const char *const date_formats[] = {
100  // Other formats documented in RFC 2616 sec 3.3.1
101  // Note: the RFC says timezone information MUST be "GMT", hence the hardcoded timezone string
102  "MMM dd HH:mm:ss yyyy", /* ANSI C's asctime() format (#145244): Jan 01 00:00:00 1970 GMT */
103  "dd-MMM-yy HH:mm:ss 'GMT'", /* RFC 850 date: 06-Dec-39 00:30:42 GMT */
104  // Non-standard formats
105  "MMM dd yyyy HH:mm:ss", /* A variation on ANSI C format seen @ amazon.com: Jan 01 1970 00:00:00 GMT */
106  "dd-MMM-yyyy HH:mm:ss 'GMT'", /* cookies.test: Y2K38 problem: 06-Dec-2039 00:30:42 GMT */
107  "MMM dd HH:mm:ss yyyy 'GMT'", /* cookies.test: Non-standard expiration dates: Sep 12 07:00:00 2020 GMT */
108  "MMM dd yyyy HH:mm:ss 'GMT'", /* cookies.test: Non-standard expiration dates: Sep 12 2020 07:00:00 GMT */
109  0
110  };
111 
112  for (int i = 0; date_formats[i]; ++i) {
113  dt = QDateTime::fromString(value, QL1S(date_formats[i]));
114  if (dt.isValid()) {
115  break;
116  }
117  }
118  }
119 
120  return dt.toUTC(); // Per RFC 2616 sec 3.3.1 always convert to UTC.
121 }

244 //
245 // Returns whether this cookie should be send to this location.
246 bool KHttpCookie::match(const QString &fqdn, const QStringList &domains,
247  const QString &path, int port) const
248 {
249  // Cookie domain match check
250  if (mDomain.isEmpty()) {
251  if (fqdn != mHost) {
252  return false;
253  }
254  } else if (!domains.contains(mDomain)) {
255  if (mDomain[0] == '.') {
256  return false;
257  }
258 
259  // Maybe the domain needs an extra dot.
260  const QString domain = QL1C('.') + mDomain;
261  if (!domains.contains(domain))
262  if (fqdn != mDomain) {
263  return false;
264  }
265  } else if (mProtocolVersion != 0 && port != -1 &&
266  !mPorts.isEmpty() && !mPorts.contains(port)) {
267  return false;
268  }
269 
270  // Cookie path match check
271  if (mPath.isEmpty()) {
272  return true;
273  }
274 
275  // According to the netscape spec http://www.acme.com/foobar,
276  // http://www.acme.com/foo.bar and http://www.acme.com/foo/bar
277  // should all match http://www.acme.com/foo...
278  // We only match http://www.acme.com/foo/bar
279  if (path.startsWith(mPath) &&
280  (
281  (path.length() == mPath.length()) || // Paths are exact match
282  mPath.endsWith(QL1C('/')) || // mPath ended with a slash
283  (path[mPath.length()] == QL1C('/')) // A slash follows
284  )) {
285  return true; // Path of URL starts with cookie-path
286  }
287 
288  return false;
289 }

322 // cookiePtr is modified: the window ids of the existing cookie in the list are added to it
323 static void removeDuplicateFromList(KHttpCookieList *list, KHttpCookie &cookiePtr, bool nameMatchOnly = false, bool updateWindowId = false)
324 {
325  QString domain1 = cookiePtr.domain();
326  if (domain1.isEmpty()) {
327  domain1 = cookiePtr.host();
328  }
329 
330  QMutableListIterator<KHttpCookie> cookieIterator(*list);
331  while (cookieIterator.hasNext()) {
332  const KHttpCookie &cookie = cookieIterator.next();
333  QString domain2 = cookie.domain();
334  if (domain2.isEmpty()) {
335  domain2 = cookie.host();
336  }
337 
338  if (cookiePtr.name() == cookie.name() &&
339  (nameMatchOnly || (domain1 == domain2 && cookiePtr.path() == cookie.path()))) {
340  if (updateWindowId) {
341  Q_FOREACH (WId windowId, cookie.windowIds()) {
342  if (windowId && (!cookiePtr.windowIds().contains(windowId))) {
343  cookiePtr.windowIds().append(windowId);
344  }
345  }
346  }
347  cookieIterator.remove();
348  break;
349  }
350  }
351 }

358 QString KCookieJar::findCookies(const QString &_url, bool useDOMFormat, WId windowId, KHttpCookieList *pendingCookies)
359 {
360  QString cookieStr, fqdn, path;
361  QStringList domains;
362  int port = -1;
363 
364  if (!parseUrl(_url, fqdn, path, &port)) {
365  return cookieStr;
366  }
367 
368  const bool secureRequest = (_url.startsWith(QL1S("https://"), Qt::CaseInsensitive) ||
369  _url.startsWith(QL1S("webdavs://"), Qt::CaseInsensitive));
370  if (port == -1) {
371  port = (secureRequest ? 443 : 80);
372  }
373 
374  extractDomains(fqdn, domains);
375 
376  KHttpCookieList allCookies;
377  for (QStringList::ConstIterator it = domains.constBegin(), itEnd = domains.constEnd();; ++it) {
378  KHttpCookieList *cookieList = 0;
379  if (it == itEnd) {
380  cookieList = pendingCookies; // Add pending cookies
381  pendingCookies = 0;
382  if (!cookieList) {
383  break;
384  }
385  } else {
386  if ((*it).isNull()) {
387  cookieList = m_cookieDomains.value(QL1S(""));
388  } else {
389  cookieList = m_cookieDomains.value(*it);
390  }
391 
392  if (!cookieList) {
393  continue; // No cookies for this domain
394  }
395  }
396 
397  QMutableListIterator<KHttpCookie> cookieIt(*cookieList);
398  while (cookieIt.hasNext()) {
399  KHttpCookie &cookie = cookieIt.next();
400  if (cookieAdvice(cookie) == KCookieReject) {
401  continue;
402  }
403 
404  if (!cookie.match(fqdn, domains, path, port)) {
405  continue;
406  }
407 
408  if (cookie.isSecure() && !secureRequest) {
409  continue;
410  }
411 
412  if (cookie.isHttpOnly() && useDOMFormat) {
413  continue;
414  }
415 
416  // Do not send expired cookies.
417  if (cookie.isExpired()) {
418  // NOTE: there is no need to delete the cookie here because the
419  // cookieserver will invoke its saveCookieJar function as a result
420  // of the state change below. This will then result in the cookie
421  // being deleting at that point.
422  m_cookiesChanged = true;
423  continue;
424  }
425 
426  if (windowId && (cookie.windowIds().indexOf(windowId) == -1)) {
427  cookie.windowIds().append(windowId);
428  }
429 
430  if (it == itEnd) { // Only needed when processing pending cookies
431  removeDuplicateFromList(&allCookies, cookie);
432  }
433 
434  allCookies.append(cookie);
435  }
436 
437  if (it == itEnd) {
438  break; // Finished.
439  }
440  }
441 
442  int protVersion = 0;
443  Q_FOREACH (const KHttpCookie &cookie, allCookies) {
444  if (cookie.protocolVersion() > protVersion) {
445  protVersion = cookie.protocolVersion();
446  }
447  }
448 
449  if (!allCookies.isEmpty()) {
450  if (!useDOMFormat) {
451  cookieStr = QStringLiteral("Cookie: ");
452  }
453 
454  if (protVersion > 0) {
455  cookieStr = cookieStr + QStringLiteral("$Version=") + QString::number(protVersion) + QStringLiteral("; ");
456  }
457 
458  Q_FOREACH (const KHttpCookie &cookie, allCookies) {
459  cookieStr = cookieStr + cookie.cookieStr(useDOMFormat) + QStringLiteral("; ");
460  }
461 
462  cookieStr.truncate(cookieStr.length() - 2); // Remove the trailing ';'
463  }
464 
465  return cookieStr;
466 }

468 //
469 // This function parses a string like 'my_name="my_value";' and returns
470 // 'my_name' in Name and 'my_value' in Value.
471 //
472 // A pointer to the end of the parsed part is returned.
473 // This pointer points either to:
474 // '\0' - The end of the string has reached.
475 // ';' - Another my_name="my_value" pair follows
476 // ',' - Another cookie follows
477 // '\n' - Another header follows
478 static const char *parseNameValue(const char *header,
479  QString &Name,
480  QString &Value,
481  bool keepQuotes = false,
482  bool rfcQuotes = false)
483 {
484  const char *s = header;
485  // Parse 'my_name' part
486  for (; (*s != '='); s++) {
487  if ((*s == '\0') || (*s == ';') || (*s == '\n')) {
488  // No '=' sign -> use string as the value, name is empty
489  // (behavior found in Mozilla and IE)
490  Name = QL1S("");
491  Value = QL1S(header);
492  Value.truncate(s - header);
493  Value = Value.trimmed();
494  return s;
495  }
496  }
497 
498  Name = QL1S(header);
499  Name.truncate(s - header);
500  Name = Name.trimmed();
501 
502  // *s == '='
503  s++;
504 
505  // Skip any whitespace
506  for (; (*s == ' ') || (*s == '\t'); s++) {
507  if ((*s == '\0') || (*s == ';') || (*s == '\n')) {
508  // End of Name
509  Value = QLatin1String("");
510  return s;
511  }
512  }
513 
514  if ((rfcQuotes || !keepQuotes) && (*s == '\"')) {
515  // Parse '"my_value"' part (quoted value)
516  if (keepQuotes) {
517  header = s++;
518  } else {
519  header = ++s; // skip "
520  }
521  for (; (*s != '\"'); s++) {
522  if ((*s == '\0') || (*s == '\n')) {
523  // End of Name
524  Value = QL1S(header);
525  Value.truncate(s - header);
526  return s;
527  }
528  }
529  Value = QL1S(header);
530  // *s == '\"';
531  if (keepQuotes) {
532  Value.truncate(++s - header);
533  } else {
534  Value.truncate(s++ - header);
535  }
536 
537  // Skip any remaining garbage
538  for (;; s++) {
539  if ((*s == '\0') || (*s == ';') || (*s == '\n')) {
540  break;
541  }
542  }
543  } else {
544  // Parse 'my_value' part (unquoted value)
545  header = s;
546  while ((*s != '\0') && (*s != ';') && (*s != '\n')) {
547  s++;
548  }
549  // End of Name
550  Value = QL1S(header);
551  Value.truncate(s - header);
552  Value = Value.trimmed();
553  }
554  return s;
555

582 bool KCookieJar::parseUrl(const QString &_url, QString &_fqdn, QString &_path, int *port)
583 {
584  QUrl kurl(_url);
585  if (!kurl.isValid() || kurl.scheme().isEmpty()) {
586  return false;
587  }
588 
589  _fqdn = kurl.host().toLower();
590  // Cookie spoofing protection. Since there is no way a path separator,
591  // a space or the escape encoding character is allowed in the hostname
592  // according to RFC 2396, reject attempts to include such things there!
593  if (_fqdn.contains(QL1C('/')) || _fqdn.contains(QL1C('%'))) {
594  return false; // deny everything!!
595  }
596 
597  // Set the port number from the protocol when one is found...
598  if (port) {
599  *port = kurl.port();
600  }
601 
602  _path = kurl.path();
603  if (_path.isEmpty()) {
604  _path = QStringLiteral("/");
605  }
606 
607  return true;
608 }

610 // not static because it uses m_twoLevelTLD
611 void KCookieJar::extractDomains(const QString &_fqdn,
612  QStringList &_domains) const
613 {
614  if (_fqdn.isEmpty()) {
615  _domains.append(QStringLiteral("localhost"));
616  return;
617  }
618 
619  // Return numeric IPv6 addresses as is...
620  if (_fqdn[0] == '[') {
621  _domains.append(_fqdn);
622  return;
623  }
624  // Return numeric IPv4 addresses as is...
625  if (_fqdn[0] >= '0' && _fqdn[0] <= '9' && _fqdn.indexOf(QRegExp(IP_ADDRESS_EXPRESSION)) > -1) {
626  _domains.append(_fqdn);
627  return;
628  }
629 
630  // Always add the FQDN at the start of the list for
631  // hostname == cookie-domainname checks!
632  _domains.append(_fqdn);
633  _domains.append(QL1C('.') + _fqdn);
634 
635  QStringList partList = _fqdn.split(QL1C('.'), QString::SkipEmptyParts);
636 
637  if (partList.count()) {
638  partList.erase(partList.begin()); // Remove hostname
639  }
640 
641  while (partList.count()) {
642 
643  if (partList.count() == 1) {
644  break; // We only have a TLD left.
645  }
646 
647  if ((partList.count() == 2) && m_twoLevelTLD.contains(partList[1].toLower())) {
648  // This domain uses two-level TLDs in the form xxxx.yy
649  break;
650  }
651 
652  if ((partList.count() == 2) && (partList[1].length() == 2)) {
653  // If this is a TLD, we should stop. (e.g. co.uk)
654  // We assume this is a TLD if it ends with .xx.yy or .x.yy
655  if (partList[0].length() <= 2) {
656  break; // This is a TLD.
657  }
658 
659  // Catch some TLDs that we miss with the previous check
660  // e.g. com.au, org.uk, mil.co
661  if (m_gTLDs.contains(partList[0].toLower())) {
662  break;
663  }
664  }
665 
666  QString domain = partList.join(QLatin1Char('.'));
667  _domains.append(domain);
668  _domains.append(QL1C('.') + domain);
669  partList.erase(partList.begin()); // Remove part
670  }

673 //
674 // This function parses cookie_headers and returns a linked list of
675 // KHttpCookie objects for all cookies found in cookie_headers.
676 // If no cookies could be found 0 is returned.
677 //
678 // cookie_headers should be a concatenation of all lines of a HTTP-header
679 // which start with "Set-Cookie". The lines should be separated by '\n's.
680 //
681 KHttpCookieList KCookieJar::makeCookies(const QString &_url,
682  const QByteArray &cookie_headers,
683  WId windowId)
684 {
685  QString fqdn, path;
686 
687  if (!parseUrl(_url, fqdn, path)) {
688  return KHttpCookieList(); // Error parsing _url
689  }
690 
691  QString Name, Value;
692  KHttpCookieList cookieList, cookieList2;
693 
694  bool isRFC2965 = false;
695  bool crossDomain = false;
696  const char *cookieStr = cookie_headers.constData();
697 
698  QString defaultPath;
699  const int index = path.lastIndexOf(QL1C('/'));
700  if (index > 0) {
701  defaultPath = path.left(index);
702  }
703 
704  // Check for cross-domain flag from kio_http
705  if (qstrncmp(cookieStr, "Cross-Domain\n", 13) == 0) {
706  cookieStr += 13;
707  crossDomain = true;
708  }
709 
710  // The hard stuff :)
711  for (;;) {
712  // check for "Set-Cookie"
713  if (qstrnicmp(cookieStr, "Set-Cookie:", 11) == 0) {
714  cookieStr = parseNameValue(cookieStr + 11, Name, Value, true);
715 
716  // Host = FQDN
717  // Default domain = ""
718  // Default path according to rfc2109
719 
720  KHttpCookie cookie(fqdn, QL1S(""), defaultPath, Name, Value);
721  if (windowId) {
722  cookie.mWindowIds.append(windowId);
723  }
724  cookie.mCrossDomain = crossDomain;
725 
726  // Insert cookie in chain
727  cookieList.append(cookie);
728  } else if (qstrnicmp(cookieStr, "Set-Cookie2:", 12) == 0) {
729  // Attempt to follow rfc2965
730  isRFC2965 = true;
731  cookieStr = parseNameValue(cookieStr + 12, Name, Value, true, true);
732 
733  // Host = FQDN
734  // Default domain = ""
735  // Default path according to rfc2965
736 
737  KHttpCookie cookie(fqdn, QL1S(""), defaultPath, Name, Value);
738  if (windowId) {
739  cookie.mWindowIds.append(windowId);
740  }
741  cookie.mCrossDomain = crossDomain;
742 
743  // Insert cookie in chain
744  cookieList2.append(cookie);
745  } else {
746  // This is not the start of a cookie header, skip till next line.
747  while (*cookieStr && *cookieStr != '\n') {
748  cookieStr++;
749  }
750 
751  if (*cookieStr == '\n') {
752  cookieStr++;
753  }
754 
755  if (!*cookieStr) {
756  break; // End of cookie_headers
757  } else {
758  continue; // end of this header, continue with next.
759  }
760  }
761 
762  while ((*cookieStr == ';') || (*cookieStr == ' ')) {
763  cookieStr++;
764 
765  // Name-Value pair follows
766  cookieStr = parseNameValue(cookieStr, Name, Value);
767  KHttpCookie &lastCookie = (isRFC2965 ? cookieList2.last() : cookieList.last());
768 
769  if (Name.compare(QL1S("domain"), Qt::CaseInsensitive) == 0) {
770  QString dom = Value.toLower();
771  // RFC2965 3.2.2: If an explicitly specified value does not
772  // start with a dot, the user agent supplies a leading dot
773  if (dom.length() && dom[0] != '.') {
774  dom.prepend(".");
775  }
776  // remove a trailing dot
777  if (dom.length() > 2 && dom[dom.length() - 1] == '.') {
778  dom = dom.left(dom.length() - 1);
779  }
780 
781  if (dom.count(QL1C('.')) > 1 || dom == QLatin1String(".local")) {
782  lastCookie.mDomain = dom;
783  }
784  } else if (Name.compare(QL1S("max-age"), Qt::CaseInsensitive) == 0) {
785  int max_age = Value.toInt();
786  if (max_age == 0) {
787  lastCookie.mExpireDate = 1;
788  } else {
789  lastCookie.mExpireDate = toEpochSecs(QDateTime::currentDateTimeUtc().addSecs(max_age));
790  }
791  } else if (Name.compare(QL1S("expires"), Qt::CaseInsensitive) == 0) {
792  const QDateTime dt = parseDate(Value);
793 
794  if (dt.isValid()) {
795  lastCookie.mExpireDate = toEpochSecs(dt);
796  if (lastCookie.mExpireDate == 0) {
797  lastCookie.mExpireDate = 1;
798  }
799  }
800  } else if (Name.compare(QL1S("path"), Qt::CaseInsensitive) == 0) {
801  if (Value.isEmpty()) {
802  lastCookie.mPath.clear(); // Catch "" <> QString()
803  } else {
804  lastCookie.mPath = QUrl::fromPercentEncoding(Value.toLatin1());
805  }
806  lastCookie.mExplicitPath = true;
807  } else if (Name.compare(QL1S("version"), Qt::CaseInsensitive) == 0) {
808  lastCookie.mProtocolVersion = Value.toInt();
809  } else if (Name.compare(QL1S("secure"), Qt::CaseInsensitive) == 0 ||
810  (Name.isEmpty() && Value.compare(QL1S("secure"), Qt::CaseInsensitive) == 0)) {
811  lastCookie.mSecure = true;
812  } else if (Name.compare(QL1S("httponly"), Qt::CaseInsensitive) == 0 ||
813  (Name.isEmpty() && Value.compare(QL1S("httponly"), Qt::CaseInsensitive) == 0)) {
814  lastCookie.mHttpOnly = true;
815  } else if (isRFC2965 && (Name.compare(QL1S("port"), Qt::CaseInsensitive) == 0 ||
816  (Name.isEmpty() && Value.compare(QL1S("port"), Qt::CaseInsensitive) == 0))) {
817  // Based on the port selection rule of RFC 2965 section 3.3.4...
818  if (Name.isEmpty()) {
819  // We intentionally append a -1 first in order to distinguish
820  // between only a 'Port' vs a 'Port="80 443"' in the sent cookie.
821  lastCookie.mPorts.append(-1);
822  const bool secureRequest = (_url.startsWith(QL1S("https://"), Qt::CaseInsensitive) ||
823  _url.startsWith(QL1S("webdavs://"), Qt::CaseInsensitive));
824  if (secureRequest) {
825  lastCookie.mPorts.append(443);
826  } else {
827  lastCookie.mPorts.append(80);
828  }
829  } else {
830  bool ok;
831  const QStringList portNums = Value.split(QL1C(' '), QString::SkipEmptyParts);
832  Q_FOREACH (const QString &portNum, portNums) {
833  const int port = portNum.toInt(&ok);
834  if (ok) {
835  lastCookie.mPorts.append(port);
836  }
837  }
838  }
839  }
840  }
841 
842  if (*cookieStr == '\0') {
843  break; // End of header
844  }
845 
846  // Skip ';' or '\n'
847  cookieStr++;
848  }
849 
850  // RFC2965 cookies come last so that they override netscape cookies.
851  while (!cookieList2.isEmpty()) {
852  KHttpCookie &lastCookie = cookieList2.first();
853  removeDuplicateFromList(&cookieList, lastCookie, true);
854  cookieList.append(lastCookie);
855  cookieList2.removeFirst();
856  }
857 
858  return cookieList;
859 }

963  } else {

1360 {

1417 // Reloads all cookies from the file '_filename'.
1418 // On succes 'true' is returned.
1419 // On failure 'false' is returned.
1420 bool KCookieJar::loadCookies(const QString &_filename)
1421 {
1422  QFile cookieFile(_filename);
1423 
1424  if (!cookieFile.open(QIODevice::ReadOnly)) {
1425  return false;
1426  }
1427 
1428  int version = 1;
1429  bool success = false;
1430  char *buffer = new char[READ_BUFFER_SIZE];
1431  qint64 len = cookieFile.readLine(buffer, READ_BUFFER_SIZE - 1);
1432 
1433  if (len != -1) {
1434  if (qstrcmp(buffer, "# KDE Cookie File\n") == 0) {
1435  success = true;
1436  } else if (qstrcmp(buffer, "# KDE Cookie File v") > 0) {
1437  bool ok = false;
1438  const int verNum = QByteArray(buffer + 19, len - 19).trimmed().toInt(&ok);
1439  if (ok) {
1440  version = verNum;
1441  success = true;
1442  }
1443  }
1444  }
1445 
1446  if (success) {
1447  const qint64 currentTime = epoch();
1448  QList<int> ports;
1449 
1450  while (cookieFile.readLine(buffer, READ_BUFFER_SIZE - 1) != -1) {
1451  char *line = buffer;
1452  // Skip lines which begin with '#' or '['
1453  if ((line[0] == '#') || (line[0] == '[')) {
1454  continue;
1455  }
1456 
1457  const QString host = extractHostAndPorts(QL1S(parseField(line)), &ports);
1458  const QString domain = QL1S(parseField(line));
1459  if (host.isEmpty() && domain.isEmpty()) {
1460  continue;
1461  }
1462  const QString path = QL1S(parseField(line));
1463  const QString expStr = QL1S(parseField(line));
1464  if (expStr.isEmpty()) {
1465  continue;
1466  }
1467  const qint64 expDate = expStr.toLongLong();
1468  const QString verStr = QL1S(parseField(line));
1469  if (verStr.isEmpty()) {
1470  continue;
1471  }
1472  int protVer = verStr.toInt();
1473  QString name = QL1S(parseField(line));
1474  bool keepQuotes = false;
1475  bool secure = false;
1476  bool httpOnly = false;
1477  bool explicitPath = false;
1478  const char *value = 0;
1479  if ((version == 2) || (protVer >= 200)) {
1480  if (protVer >= 200) {
1481  protVer -= 200;
1482  }
1483  int i = atoi(parseField(line));
1484  secure = i & 1;
1485  httpOnly = i & 2;
1486  explicitPath = i & 4;
1487  if (i & 8) {
1488  name = QLatin1String("");
1489  }
1490  line[strlen(line) - 1] = '\0'; // Strip LF.
1491  value = line;
1492  } else {
1493  if (protVer >= 100) {
1494  protVer -= 100;
1495  keepQuotes = true;
1496  }
1497  value = parseField(line, keepQuotes);
1498  secure = QByteArray(parseField(line)).toShort();
1499  }
1500 
1501  // Expired or parse error
1502  if (!value || expDate == 0 || expDate < currentTime) {
1503  continue;
1504  }
1505 
1506  KHttpCookie cookie(host, domain, path, name, value, expDate,
1507  protVer, secure, httpOnly, explicitPath);
1508  if (ports.count()) {
1509  cookie.mPorts = ports;
1510  }
1511  addCookie(cookie);
1512  }
1513  }
1514 
1515  delete [] buffer;
1516  m_cookiesChanged = false;
1517  return success;
1518 }
