273 void SlaveBase::dispatchLoop()
274 {
275  while (!d->exit_loop) {
276  if (d->nextTimeout.isValid() && (d->nextTimeout < QDateTime::currentDateTime())) {
277  QByteArray data = d->timeoutData;
278  d->nextTimeout = QDateTime();
279  d->timeoutData = QByteArray();
280  special(data);
281  }
282 
283  Q_ASSERT(d->appConnection.inited());
284 
285  int ms = -1;
286  if (d->nextTimeout.isValid()) {
287  ms = qMax<int>(QDateTime::currentDateTime().msecsTo(d->nextTimeout), 1);
288  }
289 
290  int ret = -1;
291  if (d->appConnection.hasTaskAvailable() || d->appConnection.waitForIncomingTask(ms)) {
292  // dispatch application messages
293  int cmd;
294  QByteArray data;
295  ret = d->appConnection.read(&cmd, data);
296 
297  if (ret != -1) {
298  if (d->inOpenLoop) {
299  dispatchOpenCommand(cmd, data);
300  } else {
301  dispatch(cmd, data);
302  }
303  }
304  } else {
305  ret = d->appConnection.isConnected() ? 0 : -1;
306  }
307 
308  if (ret == -1) { // some error occurred, perhaps no more application
309  // When the app exits, should the slave be put back in the pool ?
310  if (!d->exit_loop && d->isConnectedToApp && !d->poolSocket.isEmpty()) {
311  disconnectSlave();
312  d->isConnectedToApp = false;
313  closeConnection();
314  connectSlave(d->poolSocket);
315  } else {
316  break;
317  }
318  }
319 
320  //I think we get here when we were killed in dispatch() and not in select()
321  if (wasKilled()) {
322  //qDebug() << "slave was killed, returning";
323  break;
324  }
325 
326  // execute deferred deletes
327  QCoreApplication::sendPostedEvents(NULL, QEvent::DeferredDelete);
328  }
329 
330  // execute deferred deletes
331  QCoreApplication::sendPostedEvents(NULL, QEvent::DeferredDelete);
332 }

462 {

465  qWarning() << "UDSEntry for '.' not found, creating a default one. Please fix the" << QCoreApplication::applicationName() << "KIO slave";

467  entry.insert(KIO::UDSEntry::UDS_NAME, QStringLiteral("."));
468  entry.insert(KIO::UDSEntry::UDS_FILE_TYPE, S_IFDIR);
469  entry.insert(KIO::UDSEntry::UDS_SIZE, 0);
470  entry.insert(KIO::UDSEntry::UDS_ACCESS, S_IRUSR | S_IWUSR | S_IXUSR | S_IRGRP | S_IWGRP | S_IXGRP | S_IROTH | S_IXOTH);
471  d->pendingListEntries.append(entry);

474  listEntries(d->pendingListEntries);
475  d->pendingListEntries.clear();
476  }
477 
478  if (d->m_state == d->FinishedCalled) {
479  qWarning() << "finished() called twice! Please fix the KIO slave.";
480  return;
481  } else if (d->m_state == d->ErrorCalled) {
482  qWarning() << "finished() called after error()! Please fix the KIO slave.";
483  return;
484  }
485 
486  d->m_state = d->FinishedCalled;
487  mIncomingMetaData.clear(); // Clear meta data
488  d->rebuildConfig();
489  sendMetaData();

492  // reset

494  d->inOpenLoop = false;
495  d->m_rootEntryListed = false;

599 void SlaveBase::mimeType(const QString &_type)
600 {
601  //qDebug() << _type;
602  int cmd;
603  do {
604  // Send the meta-data each time we send the mime-type.
605  if (!mOutgoingMetaData.isEmpty()) {
606  //qDebug() << "emitting meta data";
607  KIO_DATA << mOutgoingMetaData;
608  send(INF_META_DATA, data);
609  }
610  KIO_DATA << _type;
611  send(INF_MIME_TYPE, data);
612  while (true) {
613  cmd = 0;
614  int ret = -1;
615  if (d->appConnection.hasTaskAvailable() || d->appConnection.waitForIncomingTask(-1)) {
616  ret = d->appConnection.read(&cmd, data);
617  }
618  if (ret == -1) {
619  //qDebug() << "read error";
620  exit();
621  }
622  //qDebug() << "got" << cmd;
623  if (cmd == CMD_HOST) { // Ignore.
624  continue;
625  }
626  if (!isSubCommand(cmd)) {
627  break;
628  }
629 
630  dispatch(cmd, data);
631  }
632  } while (cmd != CMD_NONE);
633  mOutgoingMetaData.clear();
634 }

723  d->pendingListEntries.clear();
724

1076  SlaveBasePrivate::s_seqNr = 0;
1077  QString passwd;
1078  QString host, user;
1079  quint16 port;
1080  stream >> host >> port >> user >> passwd;

1083  d->verifyErrorFinishedNotCalled("setHost()");

1085  } break;
1086  case CMD_CONNECT: {
1087  openConnection();
1088  } break;
1089  case CMD_DISCONNECT: {
1090  closeConnection();
1091  } break;

1094  slave_status();
