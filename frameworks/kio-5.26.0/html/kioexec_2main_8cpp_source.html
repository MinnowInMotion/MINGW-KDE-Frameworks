60 KIOExec::KIOExec(const QStringList &args, bool tempFiles, const QString &suggestedFileName)
61  : mExited(false)
62  , mTempFiles(tempFiles)
63  , mSuggestedFileName(suggestedFileName)
64  , expectedCounter(0)
65  , command(args.first())
66  , jobCounter(0)
67 {
68  qDebug() << "command=" << command << "args=" << args;
69 
70  for ( int i = 1; i < args.count(); i++ )
71  {
72  const QUrl urlArg = QUrl::fromUserInput(args.value(i));
73  if (!urlArg.isValid()) {
74  KMessageBox::error( 0L, i18n("Invalid URL: %1", args.value(i)) );
75  exit(1);
76  }
77  KIO::StatJob* mostlocal = KIO::mostLocalUrl( urlArg );
78  bool b = mostlocal->exec();
79  if (!b) {
80  KMessageBox::error( 0L, i18n("File not found: %1", urlArg.toDisplayString()));
81  exit(1);
82  }
83  Q_ASSERT(b);
84  const QUrl url = mostlocal->mostLocalUrl();
85 
86  //kDebug() << "url=" << url.url() << " filename=" << url.fileName();
87  // A local file, not an URL ?
88  // => It is not encoded and not shell escaped, too.
89  if ( url.isLocalFile() )
90  {
91  FileInfo file;
92  file.path = url.toLocalFile();
93  file.url = url;
94  fileList.append(file);
95  }
96  // It is an URL
97  else
98  {
99  if ( !url.isValid() )
100  KMessageBox::error( 0L, i18n( "The URL %1\nis malformed" , url.url() ) );
101  else if ( mTempFiles )
102  KMessageBox::error( 0L, i18n( "Remote URL %1\nnot allowed with --tempfiles switch" , url.toDisplayString() ) );
103  else
104  // We must fetch the file
105  {
106  QString fileName = KIO::encodeFileName( url.fileName() );
107  if ( !suggestedFileName.isEmpty() )
108  fileName = suggestedFileName;
109  // Build the destination filename, in ~/.kde/cache-*/krun/
110  // Unlike KDE-1.1, we put the filename at the end so that the extension is kept
111  // (Some programs rely on it)
112  QString krun_writable = QStandardPaths::writableLocation(QStandardPaths::CacheLocation) + "/krun/";
113  QDir().mkpath(krun_writable); // error handling will be done by the job
114  QString tmp = krun_writable + QStringLiteral("%1_%2_%3").arg(QCoreApplication::applicationPid()).arg(jobCounter++).arg(fileName);
115  FileInfo file;
116  file.path = tmp;
117  file.url = url;
118  fileList.append(file);
119 
120  expectedCounter++;
121  const QUrl dest = QUrl::fromLocalFile(tmp);
122  qDebug() << "Copying" << url << " to" << dest;
123  KIO::Job *job = KIO::file_copy( url, dest );
124  jobList.append( job );
125 
126  connect( job, SIGNAL( result( KJob * ) ), SLOT( slotResult( KJob * ) ) );
127  }
128  }
129  }
130 
131  if ( mTempFiles )
132  {
133  slotRunApp();
134  return;
135  }
136 
137  counter = 0;
138  if ( counter == expectedCounter )
139  slotResult( 0L );
140 }

142 void KIOExec::slotResult( KJob * job )
143 {
144  if (job && job->error())
145  {
146  // That error dialog would be queued, i.e. not immediate...
147  //job->showErrorDialog();
148  if ( (job->error() != KIO::ERR_USER_CANCELED) )
149  KMessageBox::error( 0L, job->errorString() );
150 
151  QString path = static_cast<KIO::FileCopyJob*>(job)->destUrl().path();
152 
153  QList<FileInfo>::Iterator it = fileList.begin();
154  for(;it != fileList.end(); ++it)
155  {
156  if (it->path == path)
157  break;
158  }
159 
160  if ( it != fileList.end() )
161  fileList.erase( it );
162  else
163  qDebug() << path << " not found in list";
164  }
165 
166  counter++;
167 
168  if ( counter < expectedCounter )
169  return;
170 
171  qDebug() << "All files downloaded, will call slotRunApp shortly";
172  // We know we can run the app now - but let's finish the job properly first.
173  QTimer::singleShot( 0, this, SLOT( slotRunApp() ) );
174 
175  jobList.clear();
176 }

178 void KIOExec::slotRunApp()
179 {
180  if ( fileList.isEmpty() ) {
181  qDebug() << "No files downloaded -> exiting";
182  mExited = true;
183  QApplication::exit(1);
184  return;
185  }
186 
187  KService service(QStringLiteral("dummy"), command, QString());
188 
189  QList<QUrl> list;
190  // Store modification times
191  QList<FileInfo>::Iterator it = fileList.begin();
192  for ( ; it != fileList.end() ; ++it )
193  {
194  QFileInfo info(QFile::encodeName(it->path));
195  it->time = info.lastModified();
196  QUrl url = QUrl::fromLocalFile(it->path);
197  list << url;
198  }
199 
200  KIO::DesktopExecParser execParser(service, list);
201  QStringList params = execParser.resultingArguments();
202 
203  qDebug() << "EXEC " << params.join(QStringLiteral(" "));
204 
205 #if (QT_VERSION >= QT_VERSION_CHECK(5, 4, 0))
206  // propagate the startup identification to the started process
207  KStartupInfoId id;
208  QByteArray startupId;
209 #if HAVE_X11
210  if (QX11Info::isPlatformX11()) {
211  startupId = QX11Info::nextStartupId();
212  }
213 #endif
214  id.initId(startupId);
215  id.setupStartupEnv();
216 #endif
217 
218  QString exe( params.takeFirst() );
219  const int exit_code = QProcess::execute( exe, params );
220 
221 #if (QT_VERSION >= QT_VERSION_CHECK(5, 4, 0))
222  KStartupInfo::resetStartupEnv();
223 #endif
224 
225  qDebug() << "EXEC done";
226 
227  // Test whether one of the files changed
228  for(it = fileList.begin(); it != fileList.end(); ++it )
229  {
230  QString src = it->path;
231  const QUrl dest = it->url;
232  QFileInfo info(src);
233  if ( info.exists() && (it->time != info.lastModified()) )
234  {
235  if ( mTempFiles )
236  {
237  if ( KMessageBox::questionYesNo( 0L,
238  i18n( "The supposedly temporary file\n%1\nhas been modified.\nDo you still want to delete it?", dest.toDisplayString(QUrl::PreferLocalFile)),
239  i18n( "File Changed" ), KStandardGuiItem::del(), KGuiItem(i18n("Do Not Delete")) ) != KMessageBox::Yes )
240  continue; // don't delete the temp file
241  }
242  else if ( ! dest.isLocalFile() ) // no upload when it's already a local file
243  {
244  if ( KMessageBox::questionYesNo( 0L,
245  i18n( "The file\n%1\nhas been modified.\nDo you want to upload the changes?" , dest.toDisplayString()),
246  i18n( "File Changed" ), KGuiItem(i18n("Upload")), KGuiItem(i18n("Do Not Upload")) ) == KMessageBox::Yes )
247  {
248  qDebug() << "src='" << src << "' dest='" << dest << "'";
249  // Do it the synchronous way.
250  KIO::CopyJob* job = KIO::copy(QUrl::fromLocalFile(src), dest);
251  if ( !job->exec() )
252  {
253  KMessageBox::error( 0L, job->errorText() );
254  continue; // don't delete the temp file
255  }
256  }
257  }
258  }
259 
260  if ((!dest.isLocalFile() || mTempFiles) && exit_code == 0) {
261  // Wait for a reasonable time so that even if the application forks on startup (like OOo or amarok)
262  // it will have time to start up and read the file before it gets deleted. #130709.
263  qDebug() << "sleeping...";
264  QThread::currentThread()->sleep(180); // 3 mn
265  qDebug() << "about to delete " << src;
266  QFile( QFile::encodeName(src) ).remove();
267  }
268  }
269 
270  mExited = true;
271  QApplication::exit(exit_code);
272 }

306

308  const QString suggestedfilename = parser.value(QStringLiteral("suggestedfilename"));
309  KIOExec exec(parser.positionalArguments(), tempfiles, suggestedfilename);
310 
311  // Don't go into the event loop if we already want to exit (#172197)
312  if (exec.exited())
