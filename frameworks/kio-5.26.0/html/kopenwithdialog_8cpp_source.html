92 static bool AppNodeLessThan(KDEPrivate::AppNode *n1, KDEPrivate::AppNode *n2)
93 {
94  if (n1->isDir) {
95  if (n2->isDir) {
96  return n1->text.compare(n2->text, Qt::CaseInsensitive) < 0;
97  } else {
98  return true;
99  }
100  } else {
101  if (n2->isDir) {
102  return false;
103  } else {
104  return n1->text.compare(n2->text, Qt::CaseInsensitive) < 0;
105  }
106  }
107 }

130 void KApplicationModelPrivate::fillNode(const QString &_entryPath, KDEPrivate::AppNode *node)
131 {
132  KServiceGroup::Ptr root = KServiceGroup::group(_entryPath);
133  if (!root || !root->isValid()) {
134  return;
135  }
136 
137  const KServiceGroup::List list = root->entries();
138 
139  for (KServiceGroup::List::ConstIterator it = list.begin();
140  it != list.end(); ++it) {
141  QString icon;
142  QString text;
143  QString entryPath;
144  QString exec;
145  bool isDir = false;
146  const KSycocaEntry::Ptr p = (*it);
147  if (p->isType(KST_KService)) {
148  const KService::Ptr service(static_cast<KService*>(p.data()));
149 
150  if (service->noDisplay()) {
151  continue;
152  }
153 
154  icon = service->icon();
155  text = service->name();
156  exec = service->exec();
157  entryPath = service->entryPath();
158  } else if (p->isType(KST_KServiceGroup)) {
159  const KServiceGroup::Ptr serviceGroup(static_cast<KServiceGroup*>(p.data()));
160 
161  if (serviceGroup->noDisplay() || serviceGroup->childCount() == 0) {
162  continue;
163  }
164 
165  icon = serviceGroup->icon();
166  text = serviceGroup->caption();
167  entryPath = serviceGroup->entryPath();
168  isDir = true;
169  } else {
170  qWarning() << "KServiceGroup: Unexpected object in list!";
171  continue;
172  }
173 
174  KDEPrivate::AppNode *newnode = new KDEPrivate::AppNode();
175  newnode->icon = icon;
176  newnode->text = text;
177  newnode->entryPath = entryPath;
178  newnode->exec = exec;
179  newnode->isDir = isDir;
180  newnode->parent = node;
181  node->children.append(newnode);
182  }
183  qStableSort(node->children.begin(), node->children.end(), KDEPrivate::AppNodeLessThan);
184 }

781  exec.remove(QStringLiteral("%f"), Qt::CaseInsensitive);

819 bool KOpenWithDialogPrivate::checkAccept()
820 {
821  const QString typedExec(edit->text());
822  if (typedExec.isEmpty()) {
823  return false;
824  }
825  QString fullExec(typedExec);
826 
827  QString serviceName;
828  QString initialServiceName;
829  QString preferredTerminal;
830  QString configPath;
831  QString serviceExec;
832  m_pService = curService;
833  if (!m_pService) {
834  // No service selected - check the command line
835 
836  // Find out the name of the service from the command line, removing args and paths
837  serviceName = KIO::DesktopExecParser::executableName(typedExec);
838  if (serviceName.isEmpty()) {
839  KMessageBox::error(q, i18n("Could not extract executable name from '%1', please type a valid program name.", serviceName));
840  return false;
841  }
842  initialServiceName = serviceName;
843  // Also remember the executableName with a path, if any, for the
844  // check that the executable exists.
845  // qDebug() << "initialServiceName=" << initialServiceName;
846  int i = 1; // We have app, app-2, app-3... Looks better for the user.
847  bool ok = false;
848  // Check if there's already a service by that name, with the same Exec line
849  do {
850  // qDebug() << "looking for service" << serviceName;
851  KService::Ptr serv = KService::serviceByDesktopName(serviceName);
852  ok = !serv; // ok if no such service yet
853  // also ok if we find the exact same service (well, "kwrite" == "kwrite %U")
854  if (serv && !serv->noDisplay() /* #297720 */) {
855  if (serv->isApplication()) {
856  /*// qDebug() << "typedExec=" << typedExec
857  << "serv->exec=" << serv->exec()
858  << "simplifiedExecLineFromService=" << simplifiedExecLineFromService(fullExec);*/
859  serviceExec = simplifiedExecLineFromService(serv->exec());
860  if (typedExec == serviceExec) {
861  ok = true;
862  m_pService = serv;
863  // qDebug() << "OK, found identical service: " << serv->entryPath();
864  } else {
865  // qDebug() << "Exec line differs, service says:" << serviceExec;
866  configPath = serv->entryPath();
867  serviceExec = serv->exec();
868  }
869  } else {
870  // qDebug() << "Found, but not an application:" << serv->entryPath();
871  }
872  }
873  if (!ok) { // service was found, but it was different -> keep looking
874  ++i;
875  serviceName = initialServiceName + '-' + QString::number(i);
876  }
877  } while (!ok);
878  }
879  if (m_pService) {
880  // Existing service selected
881  serviceName = m_pService->name();
882  initialServiceName = serviceName;
883  fullExec = m_pService->exec();
884  } else {
885  const QString binaryName = KIO::DesktopExecParser::executablePath(typedExec);
886  // qDebug() << "binaryName=" << binaryName;
887  // Ensure that the typed binary name actually exists (#81190)
888  if (QStandardPaths::findExecutable(binaryName).isEmpty()) {
889  KMessageBox::error(q, i18n("'%1' not found, please type a valid program name.", binaryName));
890  return false;
891  }
892  }
893 
894  if (terminal->isChecked()) {
895  KConfigGroup confGroup(KSharedConfig::openConfig(), QStringLiteral("General"));
896  preferredTerminal = confGroup.readPathEntry("TerminalApplication", QStringLiteral("konsole"));
897  m_command = preferredTerminal;
898  // only add --noclose when we are sure it is konsole we're using
899  if (preferredTerminal == QStringLiteral("konsole") && nocloseonexit->isChecked()) {
900  m_command += QStringLiteral(" --noclose");
901  }
902  m_command += QStringLiteral(" -e ");
903  m_command += edit->text();
904  // qDebug() << "Setting m_command to" << m_command;
905  }
906  if (m_pService && terminal->isChecked() != m_pService->terminal()) {
907  m_pService = 0; // It's not exactly this service we're running
908  }
909 
910  const bool bRemember = remember && remember->isChecked();
911  // qDebug() << "bRemember=" << bRemember << "service found=" << m_pService;
912  if (m_pService) {
913  if (bRemember) {
914  // Associate this app with qMimeType in mimeapps.list
915  Q_ASSERT(!qMimeType.isEmpty()); // we don't show the remember checkbox otherwise
916  addToMimeAppsList(m_pService->storageId());
917  }
918  } else {
919  const bool createDesktopFile = bRemember || saveNewApps;
920  if (!createDesktopFile) {
921  // Create temp service
922  if (configPath.isEmpty()) {
923  m_pService = new KService(initialServiceName, fullExec, QString());
924  } else {
925  if (!typedExec.contains(QLatin1String("%u"), Qt::CaseInsensitive) &&
926  !typedExec.contains(QLatin1String("%f"), Qt::CaseInsensitive)) {
927  int index = serviceExec.indexOf(QLatin1String("%u"), 0, Qt::CaseInsensitive);
928  if (index == -1) {
929  index = serviceExec.indexOf(QLatin1String("%f"), 0, Qt::CaseInsensitive);
930  }
931  if (index > -1) {
932  fullExec += QLatin1Char(' ');
933  fullExec += serviceExec.midRef(index, 2);
934  }
935  }
936  // qDebug() << "Creating service with Exec=" << fullExec;
937  m_pService = new KService(configPath);
938  m_pService->setExec(fullExec);
939  }
940  if (terminal->isChecked()) {
941  m_pService->setTerminal(true);
942  // only add --noclose when we are sure it is konsole we're using
943  if (preferredTerminal == QLatin1String("konsole") && nocloseonexit->isChecked()) {
944  m_pService->setTerminalOptions(QStringLiteral("--noclose"));
945  }
946  }
947  } else {
948  // If we got here, we can't seem to find a service for what they wanted. Create one.
949 
950  QString menuId;
951 #ifdef Q_OS_WIN32
952  // on windows, do not use the complete path, but only the default name.
953  serviceName = QFileInfo(serviceName).fileName();
954 #endif
955  QString newPath = KService::newServicePath(false /* ignored argument */, serviceName, &menuId);
956  // qDebug() << "Creating new service" << serviceName << "(" << newPath << ")" << "menuId=" << menuId;
957 
958  KDesktopFile desktopFile(newPath);
959  KConfigGroup cg = desktopFile.desktopGroup();
960  cg.writeEntry("Type", "Application");
961  cg.writeEntry("Name", initialServiceName);
962  cg.writeEntry("Exec", fullExec);
963  cg.writeEntry("NoDisplay", true); // don't make it appear in the K menu
964  if (terminal->isChecked()) {
965  cg.writeEntry("Terminal", true);
966  // only add --noclose when we are sure it is konsole we're using
967  if (preferredTerminal == QLatin1String("konsole") && nocloseonexit->isChecked()) {
968  cg.writeEntry("TerminalOptions", "--noclose");
969  }
970  }
971  if (!qMimeType.isEmpty()) {
972  cg.writeXdgListEntry("MimeType", QStringList() << qMimeType);
973  }
974  cg.sync();
975 
976  if (!qMimeType.isEmpty()) {
977  addToMimeAppsList(menuId);
978  } else {
979  m_pService = new KService(newPath);
980  }
981  }
982  }
983 
984  saveComboboxHistory();
985  return true;
986 }
