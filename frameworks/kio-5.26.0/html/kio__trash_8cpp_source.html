94 void TrashProtocol::restore(const QUrl &trashURL)
95 {
96  int trashId;
97  QString fileId, relativePath;
98  bool ok = TrashImpl::parseURL(trashURL, trashId, fileId, relativePath);
99  if (!ok) {
100  error(KIO::ERR_SLAVE_DEFINED, i18n("Malformed URL %1", trashURL.toString()));
101  return;
102  }
103  TrashedFileInfo info;
104  ok = impl.infoForFile(trashId, fileId, info);
105  if (!ok) {
106  error(impl.lastErrorCode(), impl.lastErrorMessage());
107  return;
108  }
109  QUrl dest = QUrl::fromLocalFile(info.origPath);
110  if (!relativePath.isEmpty()) {
111  dest = dest.adjusted(QUrl::StripTrailingSlash);
112  // (u.path() + '/' + txt) '/' is unable to concate ?
113  dest.setPath(dest.path() + QLatin1String("/") + relativePath);
114  }
115 
116  // Check that the destination directory exists, to improve the error code in case it doesn't.
117  const QString destDir = dest.adjusted(QUrl::RemoveFilename).path();
118  QT_STATBUF buff;
119 
120  if (QT_LSTAT(QFile::encodeName(destDir), &buff) == -1) {
121  error(KIO::ERR_SLAVE_DEFINED,
122  i18n("The directory %1 does not exist anymore, so it is not possible to restore this item to its original location. "
123  "You can either recreate that directory and use the restore operation again, or drag the item anywhere else to restore it.", destDir));
124  return;
125  }
126 
127  copyOrMove(trashURL, dest, false /*overwrite*/, Move);
128 }

158 void TrashProtocol::copyOrMove(const QUrl &src, const QUrl &dest, bool overwrite, CopyOrMove action)
159 {
160  if (src.scheme() == QLatin1String("trash") && dest.isLocalFile()) {
161  // Extracting (e.g. via dnd). Ignore original location stored in info file.
162  int trashId;
163  QString fileId, relativePath;
164  bool ok = TrashImpl::parseURL(src, trashId, fileId, relativePath);
165  if (!ok) {
166  error(KIO::ERR_SLAVE_DEFINED, i18n("Malformed URL %1", src.toString()));
167  return;
168  }
169  const QString destPath = dest.path();
170  if (QFile::exists(destPath)) {
171  if (overwrite) {
172  ok = QFile::remove(destPath);
173  Q_ASSERT(ok); // ### TODO
174  } else {
175  error(KIO::ERR_FILE_ALREADY_EXIST, destPath);
176  return;
177  }
178  }
179 
180  if (action == Move) {
181  qDebug() << "calling moveFromTrash(" << destPath << " " << trashId << " " << fileId << ")";
182  ok = impl.moveFromTrash(destPath, trashId, fileId, relativePath);
183  } else { // Copy
184  qDebug() << "calling copyFromTrash(" << destPath << " " << trashId << " " << fileId << ")";
185  ok = impl.copyFromTrash(destPath, trashId, fileId, relativePath);
186  }
187  if (!ok) {
188  error(impl.lastErrorCode(), impl.lastErrorMessage());
189  } else {
190  if (action == Move && relativePath.isEmpty()) {
191  (void)impl.deleteInfo(trashId, fileId);
192  }
193  finished();
194  }
195  return;
196  } else if (src.isLocalFile() && dest.scheme() == QLatin1String("trash")) {
197  QString dir = dest.adjusted(QUrl::RemoveFilename).path();
198  qDebug() << "trashing a file to " << dir << src << dest;
199 
200  // Trashing a file
201  // We detect the case where this isn't normal trashing, but
202  // e.g. if kwrite tries to save (moving tempfile over destination)
203  if (dir.length() <= 1 && src.fileName() == dest.fileName()) { // new toplevel entry
204  const QString srcPath = src.path();
205  // In theory we should use TrashImpl::parseURL to give the right filename to createInfo,
206  // in case the trash URL didn't contain the same filename as srcPath.
207  // But this can only happen with copyAs/moveAs, not available in the GUI
208  // for the trash (New/... or Rename from iconview/listview).
209  int trashId;
210  QString fileId;
211  if (!impl.createInfo(srcPath, trashId, fileId)) {
212  error(impl.lastErrorCode(), impl.lastErrorMessage());
213  } else {
214  bool ok;
215  if (action == Move) {
216  qDebug() << "calling moveToTrash(" << srcPath << " " << trashId << " " << fileId << ")";
217  ok = impl.moveToTrash(srcPath, trashId, fileId);
218  } else { // Copy
219  qDebug() << "calling copyToTrash(" << srcPath << " " << trashId << " " << fileId << ")";
220  ok = impl.copyToTrash(srcPath, trashId, fileId);
221  }
222  if (!ok) {
223  (void)impl.deleteInfo(trashId, fileId);
224  error(impl.lastErrorCode(), impl.lastErrorMessage());
225  } else {
226  // Inform caller of the final URL. Used by konq_undo.
227  const QUrl url = impl.makeURL(trashId, fileId, QString());
228  setMetaData(QLatin1String("trashURL-") + srcPath, url.url());
229  finished();
230  }
231  }
232  return;
233  } else {
234  qDebug() << "returning KIO::ERR_ACCESS_DENIED, it's not allowed to add a file to an existing trash directory";
235  // It's not allowed to add a file to an existing trash directory.
236  error(KIO::ERR_ACCESS_DENIED, dest.toString());
237  return;
238  }
239  } else {
240  error(KIO::ERR_UNSUPPORTED_ACTION, i18n("Internal error in copyOrMove, should never happen"));
241  }

255 void TrashProtocol::stat(const QUrl &url)
256 {
257  INIT_IMPL;
258  const QString path = url.path();
259  if (path.isEmpty() || path == QLatin1String("/")) {
260  // The root is "virtual" - it's not a single physical directory
261  KIO::UDSEntry entry;
262  createTopLevelDirEntry(entry);
263  statEntry(entry);
264  finished();
265  } else {
266  int trashId;
267  QString fileId, relativePath;
268 
269  bool ok = TrashImpl::parseURL(url, trashId, fileId, relativePath);
270 
271  if (!ok) {
272  // ######## do we still need this?
273  qDebug() << url << " looks fishy, returning does-not-exist";
274  // A URL like trash:/file simply means that CopyJob is trying to see if
275  // the destination exists already (it made up the URL by itself).
276  error(KIO::ERR_DOES_NOT_EXIST, url.toString());
277  //error( KIO::ERR_SLAVE_DEFINED, i18n( "Malformed URL %1" ).arg( url.toString() ) );
278  return;
279  }
280 
281  qDebug() << "parsed" << url << "got" << trashId << fileId << relativePath;
282 
283  const QString filePath = impl.physicalPath(trashId, fileId, relativePath);
284  if (filePath.isEmpty()) {
285  error(impl.lastErrorCode(), impl.lastErrorMessage());
286  return;
287  }
288 
289  // For a toplevel file, use the fileId as display name (to hide the trashId)
290  // For a file in a subdir, use the fileName as is.
291  QString fileDisplayName = relativePath.isEmpty() ? fileId : url.fileName();
292 
293  QUrl fileURL;
294  if (url.path().length() > 1) {
295  fileURL = url;
296  }
297 
298  KIO::UDSEntry entry;
299  TrashedFileInfo info;
300  ok = impl.infoForFile(trashId, fileId, info);
301  if (ok) {
302  ok = createUDSEntry(filePath, fileDisplayName, fileURL.fileName(), entry, info);
303  }
304 
305  if (!ok) {
306  error(KIO::ERR_COULD_NOT_STAT, url.toString());
307  return;
308  }
309 
310  statEntry(entry);
311  finished();
312  }

343 {
344  INIT_IMPL;

393  listEntry(entry);
394  }
395  }
396  entry.clear();
397  finished();

400 bool TrashProtocol::createUDSEntry(const QString &physicalPath, const QString &displayFileName, const QString &internalFileName, KIO::UDSEntry &entry, const TrashedFileInfo &info)
401 {
402  QByteArray physicalPath_c = QFile::encodeName(physicalPath);
403  QT_STATBUF buff;
404  if (QT_LSTAT(physicalPath_c, &buff) == -1) {
405  qWarning() << "couldn't stat " << physicalPath;
406  return false;
407  }
408  if (S_ISLNK(buff.st_mode)) {
409  char buffer2[ 1000 ];
410  int n = readlink(physicalPath_c, buffer2, 999);
411  if (n != -1) {
412  buffer2[ n ] = 0;
413  }
414 
415  entry.insert(KIO::UDSEntry::UDS_LINK_DEST, QFile::decodeName(buffer2));
416  // Follow symlink
417  // That makes sense in kio_file, but not in the trash, especially for the size
418  // #136876
419 #if 0
420  if (KDE_stat(physicalPath_c, &buff) == -1) {
421  // It is a link pointing to nowhere
422  buff.st_mode = S_IFLNK | S_IRWXU | S_IRWXG | S_IRWXO;
423  buff.st_mtime = 0;
424  buff.st_atime = 0;
425  buff.st_size = 0;
426  }
427 #endif
428  }
429 
430  mode_t type = buff.st_mode & S_IFMT; // extract file type
431  mode_t access = buff.st_mode & 07777; // extract permissions
432  access &= 07555; // make it readonly, since it's in the trashcan
433  Q_ASSERT(!internalFileName.isEmpty());
434  entry.insert(KIO::UDSEntry::UDS_NAME, internalFileName); // internal filename, like "0-foo"
435  entry.insert(KIO::UDSEntry::UDS_DISPLAY_NAME, displayFileName); // user-visible filename, like "foo"
436  entry.insert(KIO::UDSEntry::UDS_FILE_TYPE, type);
437  //if ( !url.isEmpty() )
438  // entry.insert( KIO::UDSEntry::UDS_URL, url );
439 
440  QMimeDatabase db;
441  QMimeType mt = db.mimeTypeForFile(physicalPath);
442  if (mt.isValid()) {
443  entry.insert(KIO::UDSEntry::UDS_MIME_TYPE, mt.name());
444  }
445  entry.insert(KIO::UDSEntry::UDS_ACCESS, access);
446  entry.insert(KIO::UDSEntry::UDS_SIZE, buff.st_size);
447  entry.insert(KIO::UDSEntry::UDS_USER, m_userName); // assumption
448  entry.insert(KIO::UDSEntry::UDS_GROUP, m_groupName); // assumption
449  entry.insert(KIO::UDSEntry::UDS_MODIFICATION_TIME, buff.st_mtime);
450  entry.insert(KIO::UDSEntry::UDS_ACCESS_TIME, buff.st_atime); // ## or use it for deletion time?
451  entry.insert(KIO::UDSEntry::UDS_EXTRA, info.origPath);
452  entry.insert(KIO::UDSEntry::UDS_EXTRA + 1, info.deletionDate.toString(Qt::ISODate));
453  return true;
454 }

478 void TrashProtocol::special(const QByteArray &data)
479 {
480  INIT_IMPL;
481  QDataStream stream(data);
482  int cmd;
483  stream >> cmd;
484 
485  switch (cmd) {
486  case 1:
487  if (impl.emptyTrash()) {
488  finished();
489  } else {
490  error(impl.lastErrorCode(), impl.lastErrorMessage());
491  }
492  break;
493  case 2:
494  impl.migrateOldTrash();
495  finished();
496  break;
497  case 3: {
498  QUrl url;
499  stream >> url;
500  restore(url);
501  break;
502  }
503  default:

505  error(KIO::ERR_UNSUPPORTED_ACTION, QString::number(cmd));
506  break;
507  }
