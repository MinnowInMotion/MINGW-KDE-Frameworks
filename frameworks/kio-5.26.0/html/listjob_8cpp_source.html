95 void ListJobPrivate::slotListEntries(const KIO::UDSEntryList &list)
96 {
97  Q_Q(ListJob);
98  // Emit progress info (takes care of emit processedSize and percent)
99  m_processedEntries += list.count();
100  slotProcessedSize(m_processedEntries);
101 
102  if (recursive) {
103  UDSEntryList::ConstIterator it = list.begin();
104  const UDSEntryList::ConstIterator end = list.end();
105 
106  for (; it != end; ++it) {
107 
108  const UDSEntry &entry = *it;
109 
110  QUrl itemURL;
111  const QString udsUrl = entry.stringValue(KIO::UDSEntry::UDS_URL);
112  QString filename;
113  if (!udsUrl.isEmpty()) {
114  itemURL = QUrl(udsUrl);
115  filename = itemURL.fileName();
116  } else { // no URL, use the name
117  itemURL = q->url();
118  filename = entry.stringValue(KIO::UDSEntry::UDS_NAME);
119  Q_ASSERT(!filename.isEmpty()); // we'll recurse forever otherwise :)
120  itemURL.setPath(itemURL.path() + '/' + filename);
121  }
122 
123  if (entry.isDir() && !entry.isLink()) {
124  Q_ASSERT(!filename.isEmpty());
125  QString displayName = entry.stringValue(KIO::UDSEntry::UDS_DISPLAY_NAME);
126  if (displayName.isEmpty()) {
127  displayName = filename;
128  }
129  // skip hidden dirs when listing if requested
130  if (filename != QLatin1String("..") && filename != QLatin1String(".") && (includeHidden || filename[0] != '.')) {
131  ListJob *job = ListJobPrivate::newJobNoUi(itemURL,
132  true /*recursive*/,
133  m_prefix + filename + '/',
134  m_displayPrefix + displayName + '/',
135  includeHidden);
136  Scheduler::setJobPriority(job, 1);
137  q->connect(job, SIGNAL(entries(KIO::Job*,KIO::UDSEntryList)),
138  SLOT(gotEntries(KIO::Job*,KIO::UDSEntryList)));
139  q->connect(job, SIGNAL(subError(KIO::ListJob*,KIO::ListJob*)),
140  SLOT(slotSubError(KIO::ListJob*,KIO::ListJob*)));
141  q->addSubjob(job);
142  }
143  }
144  }
145  }
146 
147  // Not recursive, or top-level of recursive listing : return now (send . and .. as well)
148  // exclusion of hidden files also requires the full sweep, but the case for full-listing
149  // a single dir is probably common enough to justify the shortcut
150  if (m_prefix.isNull() && includeHidden) {
151  emit q->entries(q, list);
152  } else {
153  // cull the unwanted hidden dirs and/or parent dir references from the listing, then emit that
154  UDSEntryList newlist;
155 
156  UDSEntryList::const_iterator it = list.begin();
157  const UDSEntryList::const_iterator end = list.end();
158  for (; it != end; ++it) {
159 
160  // Modify the name in the UDSEntry
161  UDSEntry newone = *it;
162  const QString filename = newone.stringValue(KIO::UDSEntry::UDS_NAME);
163  QString displayName = newone.stringValue(KIO::UDSEntry::UDS_DISPLAY_NAME);
164  if (displayName.isEmpty()) {
165  displayName = filename;
166  }
167  // Avoid returning entries like subdir/. and subdir/.., but include . and .. for
168  // the toplevel dir, and skip hidden files/dirs if that was requested
169  if ((m_prefix.isNull() || (filename != QLatin1String("..") && filename != QLatin1String(".")))
170  && (includeHidden || (filename[0] != '.'))) {
171  // ## Didn't find a way to use the iterator instead of re-doing a key lookup
172  newone.insert(KIO::UDSEntry::UDS_NAME, m_prefix + filename);
173  newone.insert(KIO::UDSEntry::UDS_DISPLAY_NAME, m_displayPrefix + displayName);
174  newlist.append(newone);
175  }
176  }
177 
178  emit q->entries(q, newlist);
179  }
180 }

195 void ListJob::slotResult(KJob *job)
196 {
197  Q_D(ListJob);
198  if (job->error()) {
199  // If we can't list a subdir, the result is still ok
200  // This is why we override KCompositeJob::slotResult - to not set
201  // an error on parent job.
202  // Let's emit a signal about this though
203  emit subError(this, static_cast<KIO::ListJob *>(job));
204  }
205  removeSubjob(job);
206  if (!hasSubjobs() && !d->m_slave) { // if the main directory listing is still running, it will emit result in SimpleJob::slotFinished()
207  emitResult();
208  }
209 }

211 void ListJobPrivate::slotRedirection(const QUrl &url)
212 {
213  Q_Q(ListJob);
214  if (!KUrlAuthorized::authorizeUrlAction(QStringLiteral("redirect"), m_url, url)) {
215  qWarning() << "Redirection from" << m_url << "to" << url << "REJECTED!";
216  return;
217  }
218  m_redirectionURL = url; // We'll remember that when the job finishes
219  emit q->redirection(q, m_redirectionURL);
220 }

222 void ListJob::slotFinished()
223 {
224  Q_D(ListJob);
225 
226  if (!d->m_redirectionURL.isEmpty() && d->m_redirectionURL.isValid() && !error()) {
227 
228  //qDebug() << "Redirection to " << d->m_redirectionURL;
229  if (queryMetaData(QStringLiteral("permanent-redirect")) == QLatin1String("true")) {
230  emit permanentRedirection(this, d->m_url, d->m_redirectionURL);
231  }
232 
233  if (d->m_redirectionHandlingEnabled) {
234  d->m_packedArgs.truncate(0);
235  QDataStream stream(&d->m_packedArgs, QIODevice::WriteOnly);
236  stream << d->m_redirectionURL;
237 
238  d->restartAfterRedirection(&d->m_redirectionURL);
239  return;
240  }
241  }
242 
243  // Return slave to the scheduler
244  SimpleJob::slotFinished();
245 }

274 void ListJobPrivate::start(Slave *slave)
275 {
276  Q_Q(ListJob);
277  if (!KUrlAuthorized::authorizeUrlAction(QStringLiteral("list"), m_url, m_url) &&
278  !(m_extraFlags & EF_ListJobUnrestricted)) {
279  q->setError(ERR_ACCESS_DENIED);
280  q->setErrorText(m_url.toDisplayString());
281  QTimer::singleShot(0, q, SLOT(slotFinished()));
282  return;
283  }

285  SLOT(slotListEntries(KIO::UDSEntryList)));

287  SLOT(slotTotalSize(KIO::filesize_t)));
288  q->connect(slave, SIGNAL(redirection(QUrl)),
289  SLOT(slotRedirection(QUrl)));
290 
291  SimpleJobPrivate::start(slave);
