104 static QString schemeHandler(const QString &protocol)
105 {
106  // We have up to two sources of data, for protocols not handled by kioslaves (so called "helper") :
107  // 1) the exec line of the .protocol file, if there's one
108  // 2) the application associated with x-scheme-handler/<protocol> if there's one
109 
110  // If both exist, then:
111  // A) if the .protocol file says "launch an application", then the new-style handler-app has priority
112  // B) but if the .protocol file is for a kioslave (e.g. kio_http) then this has priority over
113  // firefox or chromium saying x-scheme-handler/http. Gnome people want to send all HTTP urls
114  // to a webbrowser, but we want mimetype-determination-in-calling-application by default
115  // (the user can configure a BrowserApplication though)
116 
117  const KService::Ptr service = KMimeTypeTrader::self()->preferredService(QLatin1String("x-scheme-handler/") + protocol);
118  if (service) {
119  return service->exec(); // for helper protocols, the handler app has priority over the hardcoded one (see A above)
120  }
121  Q_ASSERT(KProtocolInfo::isHelperProtocol(protocol));
122  return KProtocolInfo::exec(protocol);
123 }

167 bool KRun::runUrl(const QUrl &u, const QString &_mimetype, QWidget *window, bool tempFile, bool runExecutables, const QString &suggestedFileName, const QByteArray &asn)
168 {
169  bool noRun = false;
170  bool noAuth = false;
171  if (_mimetype == QLatin1String("inode/directory-locked")) {
172  KMessageBox::error(window,
173  i18n("<qt>Unable to enter <b>%1</b>.\nYou do not have access rights to this location.</qt>", u.toDisplayString().toHtmlEscaped()));
174  return false;
175  } else if (_mimetype == QLatin1String("application/x-desktop")) {
176  if (u.isLocalFile() && runExecutables) {
177  return KDesktopFileActions::runWithStartup(u, true, asn);
178  }
179  } else if (isExecutableFile(u, _mimetype)) {
180  if (u.isLocalFile() && runExecutables) {
181  if (KAuthorized::authorize(QStringLiteral("shell_access"))) {
182  return (KRun::runCommand(KShell::quoteArg(u.toLocalFile()), QString(), QString(),
183  window, asn, u.adjusted(QUrl::RemoveFilename).toLocalFile())); // just execute the url as a command
184  // ## TODO implement deleting the file if tempFile==true
185  } else {
186  noAuth = true;
187  }
188  } else if (_mimetype == QLatin1String("application/x-executable")) {
189  noRun = true;
190  }
191  } else if (isExecutable(_mimetype)) {
192  if (!runExecutables) {
193  noRun = true;
194  }
195 
196  if (!KAuthorized::authorize(QStringLiteral("shell_access"))) {
197  noAuth = true;
198  }
199  }
200 
201  if (noRun) {
202  KMessageBox::sorry(window,
203  i18n("<qt>The file <b>%1</b> is an executable program. "
204  "For safety it will not be started.</qt>", u.toDisplayString().toHtmlEscaped()));
205  return false;
206  }
207  if (noAuth) {
208  KMessageBox::error(window,
209  i18n("<qt>You do not have permission to run <b>%1</b>.</qt>", u.toDisplayString().toHtmlEscaped()));
210  return false;
211  }
212 
213  QList<QUrl> lst;
214  lst.append(u);
215 
216  KService::Ptr offer = KMimeTypeTrader::self()->preferredService(_mimetype);
217 
218  if (!offer) {
219 #ifdef Q_OS_WIN
220  // As KDE on windows doesnt know about the windows default applications offers will be empty in nearly all cases.
221  // So we use QDesktopServices::openUrl to let windows decide how to open the file
222  return QDesktopServices::openUrl(u);
223 #else
224  // Open-with dialog
225  // TODO : pass the mimetype as a parameter, to show it (comment field) in the dialog !
226  // Hmm, in fact KOpenWithDialog::setServiceType already guesses the mimetype from the first URL of the list...
227  return displayOpenWithDialog(lst, window, tempFile, suggestedFileName, asn);
228 #endif
229  }
230 
231  return KRun::runService(*offer, lst, window, tempFile, suggestedFileName, asn);
232 }

290 static qint64 runCommandInternal(KProcess *proc, const KService *service, const QString &executable,
291  const QString &userVisibleName, const QString &iconName, QWidget *window,
292  const QByteArray &asn)
293 {
294  if (window) {
295  window = window->topLevelWidget();
296  }
297  if (service && !service->entryPath().isEmpty()
298  && !KDesktopFile::isAuthorizedDesktopFile(service->entryPath())) {
299  qWarning() << "No authorization to execute " << service->entryPath();
300  KMessageBox::sorry(window, i18n("You are not authorized to execute this file."));
301  delete proc;
302  return 0;
303  }
304 
305  QString bin = KIO::DesktopExecParser::executableName(executable);
306 #if HAVE_X11 // Startup notification doesn't work with QT/E, service isn't needed without Startup notification
307  static bool isX11 = QGuiApplication::platformName() == QStringLiteral("xcb");
308  if (isX11) {
309  bool silent;
310  QByteArray wmclass;
311  KStartupInfoId id;
312  bool startup_notify = (asn != "0" && KRun::checkStartupNotify(QString() /*unused*/, service, &silent, &wmclass));
313  if (startup_notify) {
314  id.initId(asn);
315  id.setupStartupEnv();
316  KStartupInfoData data;
317  data.setHostname();
318  data.setBin(bin);
319  if (!userVisibleName.isEmpty()) {
320  data.setName(userVisibleName);
321  } else if (service && !service->name().isEmpty()) {
322  data.setName(service->name());
323  }
324  data.setDescription(i18n("Launching %1", data.name()));
325  if (!iconName.isEmpty()) {
326  data.setIcon(iconName);
327  } else if (service && !service->icon().isEmpty()) {
328  data.setIcon(service->icon());
329  }
330  if (!wmclass.isEmpty()) {
331  data.setWMClass(wmclass);
332  }
333  if (silent) {
334  data.setSilent(KStartupInfoData::Yes);
335  }
336  data.setDesktop(KWindowSystem::currentDesktop());
337  if (window) {
338  data.setLaunchedBy(window->winId());
339  }
340  if (service && !service->entryPath().isEmpty()) {
341  data.setApplicationId(service->entryPath());
342  }
343  KStartupInfo::sendStartup(id, data);
344  }
345  qint64 pid = KProcessRunner::run(proc, executable, id);
346  if (startup_notify && pid) {
347  KStartupInfoData data;
348  data.addPid(pid);
349  KStartupInfo::sendChange(id, data);
350  KStartupInfo::resetStartupEnv();
351  }
352  return pid;
353  }
354 #else
355  Q_UNUSED(userVisibleName);
356  Q_UNUSED(iconName);
357 #endif
358  return KProcessRunner::run(proc, bin, KStartupInfoId());
359 }

394  }

401 static qint64 runApplicationImpl(const KService &_service, const QList<QUrl> &_urls, QWidget *window,
402  KRun::RunFlags flags, const QString &suggestedFileName, const QByteArray &asn)
403 {
404  QList<QUrl> urlsToRun = _urls;
405  if ((_urls.count() > 1) && !_service.allowMultipleFiles()) {
406  // We need to launch the application N times. That sucks.
407  // We ignore the result for application 2 to N.
408  // For the first file we launch the application in the
409  // usual way. The reported result is based on this
410  // application.
411  QList<QUrl>::ConstIterator it = _urls.begin();
412  while (++it != _urls.end()) {
413  QList<QUrl> singleUrl;
414  singleUrl.append(*it);
415  runApplicationImpl(_service, singleUrl, window, flags, suggestedFileName, QByteArray());
416  }
417  urlsToRun.clear();
418  urlsToRun.append(_urls.first());
419  }
420  KIO::DesktopExecParser execParser(_service, urlsToRun);
421  execParser.setUrlsAreTempFiles(flags & KRun::DeleteTemporaryFiles);
422  execParser.setSuggestedFileName(suggestedFileName);
423  const QStringList args = execParser.resultingArguments();
424  if (args.isEmpty()) {
425  KMessageBox::sorry(window, i18n("Error processing Exec field in %1", _service.entryPath()));
426  return 0;
427  }
428  //qDebug() << "runTempService: KProcess args=" << args;
429 
430  KProcess * proc = new KProcess;
431  *proc << args;
432 
433  QString path(_service.path());
434  if (path.isEmpty() && !_urls.isEmpty() && _urls.first().isLocalFile()) {
435  path = _urls.first().adjusted(QUrl::RemoveFilename).toLocalFile();
436  }
437 
438  proc->setWorkingDirectory(path);
439 
440  return runCommandInternal(proc, &_service, KIO::DesktopExecParser::executablePath(_service.exec()),
441  _service.name(), _service.icon(), window, asn);
442 }

463  *it = localURL;

468  }
469  }

528 // Helper function to make the given .desktop file executable by ensuring
529 // that a #!/usr/bin/env xdg-open line is added if necessary and the file has
530 // the +x bit set for the user. Returns false if either fails.
531 static bool makeFileExecutable(const QString &fileName)
532 {
533  // Open the file and read the first two characters, check if it's
534  // #!. If not, create a new file, prepend appropriate lines, and copy
535  // over.
536  QFile desktopFile(fileName);
537  if (!desktopFile.open(QFile::ReadOnly)) {
538  qWarning() << "Error opening service" << fileName << desktopFile.errorString();
539  return false;
540  }
541 
542  QByteArray header = desktopFile.peek(2); // First two chars of file
543  if (header.size() == 0) {
544  qWarning() << "Error inspecting service" << fileName << desktopFile.errorString();
545  return false; // Some kind of error
546  }
547 
548  if (header != "#!") {
549  // Add header
550  QSaveFile saveFile;
551  saveFile.setFileName(fileName);
552  if (!saveFile.open(QIODevice::WriteOnly)) {
553  qWarning() << "Unable to open replacement file for" << fileName << saveFile.errorString();
554  return false;
555  }
556 
557  QByteArray shebang("#!/usr/bin/env xdg-open\n");
558  if (saveFile.write(shebang) != shebang.size()) {
559  qWarning() << "Error occurred adding header for" << fileName << saveFile.errorString();
560  saveFile.cancelWriting();
561  return false;
562  }
563 
564  // Now copy the one into the other and then close and reopen desktopFile
565  QByteArray desktopData(desktopFile.readAll());
566  if (desktopData.isEmpty()) {
567  qWarning() << "Unable to read service" << fileName << desktopFile.errorString();
568  saveFile.cancelWriting();
569  return false;
570  }
571 
572  if (saveFile.write(desktopData) != desktopData.size()) {
573  qWarning() << "Error copying service" << fileName << saveFile.errorString();
574  saveFile.cancelWriting();
575  return false;
576  }
577 
578  desktopFile.close();
579  if (!saveFile.commit()) { // Figures....
580  qWarning() << "Error committing changes to service" << fileName << saveFile.errorString();
581  return false;
582  }
583 
584  if (!desktopFile.open(QFile::ReadOnly)) {
