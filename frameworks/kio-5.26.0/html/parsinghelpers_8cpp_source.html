31 static void skipSpace(const char input[], int *pos, int end)
32 {
33  int idx = *pos;
34  while (idx < end && (input[idx] == ' ' || input[idx] == '\t')) {
35  idx++;
36  }
37  *pos = idx;
38  return;
39 }

41 // Advance *pos to start of next line while being forgiving about line endings.
42 // Return false if the end of the header has been reached, true otherwise.
43 static bool nextLine(const char input[], int *pos, int end)
44 {
45  int idx = *pos;
46  while (idx < end && input[idx] != '\r' && input[idx] != '\n') {
47  idx++;
48  }
49  int rCount = 0;
50  int nCount = 0;
51  while (idx < end && qMax(rCount, nCount) < 2 && (input[idx] == '\r' || input[idx] == '\n')) {
52  input[idx] == '\r' ? rCount++ : nCount++;
53  idx++;
54  }
55  if (idx < end && qMax(rCount, nCount) == 2 && qMin(rCount, nCount) == 1) {
56  // if just one of the others is missing eat it too.
57  // this ensures that conforming headers using the proper
58  // \r\n sequence (and also \n\r) will be parsed correctly.
59  if ((rCount == 1 && input[idx] == '\r') || (nCount == 1 && input[idx] == '\n')) {
60  idx++;
61  }
62  }
63 
64  *pos = idx;
65  return idx < end && rCount < 2 && nCount < 2;
66 }

68 // QByteArray::fromPercentEncoding() does not notify us about encoding errors so we need
69 // to check here if this is valid at all.
70 static bool isValidPercentEncoding(const QByteArray &data)
71 {
72  int i = 0;
73  const int last = data.length() - 1;
74  const char *d = data.constData();
75 
76  while ((i = data.indexOf('%', i)) != -1) {
77  if (i >= last - 2) {
78  return false;
79  }
80  if (! isxdigit(d[i + 1])) {
81  return false;
82  }
83  if (! isxdigit(d[i + 2])) {
84  return false;
85  }
86  i++;
87  }
88 
89  return true;
90 }

170 int HeaderTokenizer::tokenize(int begin, int end)
171 {
172  char *buf = m_buffer; //keep line length in check :/
173  int idx = begin;
174  int startIdx = begin; //multi-purpose start of current token
175  bool multiValuedEndedWithComma = false; //did the last multi-valued line end with a comma?
176  QByteArray headerKey;
177  do {
178 
179  if (buf[idx] == ' ' || buf [idx] == '\t') {
180  // line continuation; preserve startIdx except (see below)
181  if (headerKey.isEmpty()) {
182  continue;
183  }
184  // turn CR/LF into spaces for later parsing convenience
185  int backIdx = idx - 1;
186  while (backIdx >= begin && (buf[backIdx] == '\r' || buf[backIdx] == '\n')) {
187  buf[backIdx--] = ' ';
188  }
189 
190  // multiple values, comma-separated: add new value or continue previous?
191  if (operator[](headerKey).isMultiValued) {
192  if (multiValuedEndedWithComma) {
193  // start new value; this is almost like no line continuation
194  skipSpace(buf, &idx, end);
195  startIdx = idx;
196  } else {
197  // continue previous value; this is tricky. unit tests to the rescue!
198  if (operator[](headerKey).beginEnd.last().first == startIdx) {
199  // remove entry, it will be re-added because already idx != startIdx
200  operator[](headerKey).beginEnd.removeLast();
201  } else {
202  // no comma, no entry: the prev line was whitespace only - start new value
203  skipSpace(buf, &idx, end);
204  startIdx = idx;
205  }
206  }
207  }
208 
209  } else {
210  // new field
211  startIdx = idx;
212  // also make sure that there is at least one char after the colon
213  while (idx < (end - 1) && buf[idx] != ':' && buf[idx] != '\r' && buf[idx] != '\n') {
214  buf[idx] = tolower(buf[idx]);
215  idx++;
216  }
217  if (buf[idx] != ':') {
218  //malformed line: no colon
219  headerKey.clear();
220  continue;
221  }
222  headerKey = QByteArray(&buf[startIdx], idx - startIdx);
223  if (!contains(headerKey)) {
224  //we don't recognize this header line
225  headerKey.clear();
226  continue;
227  }
228  // skip colon & leading whitespace
229  idx++;
230  skipSpace(buf, &idx, end);
231  startIdx = idx;
232  }
233 
234  // we have the name/key of the field, now parse the value
235  if (!operator[](headerKey).isMultiValued) {
236 
237  // scan to end of line
238  while (idx < end && buf[idx] != '\r' && buf[idx] != '\n') {
239  idx++;
240  }
241  if (!operator[](headerKey).beginEnd.isEmpty()) {
242  // there already is an entry; are we just in a line continuation?
243  if (operator[](headerKey).beginEnd.last().first == startIdx) {
244  // line continuation: delete previous entry and later insert a new, longer one.
245  operator[](headerKey).beginEnd.removeLast();
246  }
247  }
248  operator[](headerKey).beginEnd.append(QPair<int, int>(startIdx, idx));
249 
250  } else {
251 
252  // comma-separated list
253  while (true) {
254  //skip one value
255  while (idx < end && buf[idx] != '\r' && buf[idx] != '\n' && buf[idx] != ',') {
256  idx++;
257  }
258  if (idx != startIdx) {
259  operator[](headerKey).beginEnd.append(QPair<int, int>(startIdx, idx));
260  }
261  multiValuedEndedWithComma = buf[idx] == ',';
262  //skip comma(s) and leading whitespace, if any respectively
263  while (idx < end && buf[idx] == ',') {
264  idx++;
265  }
266  skipSpace(buf, &idx, end);
267  //next value or end-of-line / end of header?
268  if (buf[idx] >= end || buf[idx] == '\r' || buf[idx] == '\n') {
269  break;
270  }
271  //next value
272  startIdx = idx;
273  }
274  }
275  } while (nextLine(buf, &idx, end));
276  return idx;
277 }

301 static bool specialChar(const QChar &ch, const char *specials)
302 {
303  // WORKAROUND: According to RFC 2616, any character other than ascii
304  // characters should NOT be allowed in unquoted content-disposition file
305  // names. However, since none of the major browsers follow this rule, we do
306  // the same thing here and allow all printable unicode characters. See
307  // https://bugs.kde.org/show_bug.cgi?id=261223 for the detials.
308  if (!ch.isPrint()) {
309  return true;
310  }
311 
312  for (int i = qstrlen(specials) - 1; i >= 0; i--) {
313  if (ch == QLatin1Char(specials[i])) {
314  return true;
315  }
316  }
317 
318  return false;
319 }

336 static QString extractUntil(const QString &str, QChar term, int &pos, const char *specials)
337 {
338  QString out;
339  skipLWS(str, pos);
340  bool valid = true;
341 
342  while (pos < str.length() && (str[pos] != term)) {
343  out += str[pos];
344  valid = (valid && !specialChar(str[pos], specials));
345  ++pos;
346  }
347 
348  if (pos < str.length()) { // Stopped due to finding term
349  ++pos;
350  }
351 
352  if (!valid) {
353  return QString();
354  }
355 
356  // Remove trailing linear whitespace...
357  while (out.endsWith(QLatin1Char(' ')) || out.endsWith(QLatin1Char('\t'))) {
358  out.chop(1);
359  }
360 
361  if (out.contains(QLatin1Char(' '))) {
362  out.clear();
363  }
364 
365  return out;
366 }

368 // As above, but also handles quotes..
369 // pos is set to -1 on parse error
370 static QString extractMaybeQuotedUntil(const QString &str, int &pos)
371 {
372  const QChar term = QLatin1Char(';');
373 
374  skipLWS(str, pos);
375 
376  // Are we quoted?
377  if (pos < str.length() && str[pos] == QLatin1Char('"')) {
378  QString out;
379 
380  // Skip the quote...
381  ++pos;
382 
383  // when quoted we also need an end-quote
384  bool endquote = false;
385 
386  // Parse until trailing quote...
387  while (pos < str.length()) {
388  if (str[pos] == QLatin1Char('\\') && pos + 1 < str.length()) {
389  // quoted-pair = "\" CHAR
390  out += str[pos + 1];
391  pos += 2; // Skip both...
392  } else if (str[pos] == QLatin1Char('"')) {
393  ++pos;
394  endquote = true;
395  break;
396  } else if (!str[pos].isPrint()) { // Don't allow CTL's RFC 2616 sec 2.2
397  break;
398  } else {
399  out += str[pos];
400  ++pos;
401  }
402  }
403 
404  if (!endquote) {
405  pos = -1;
406  return QString();
407  }
408 
409  // Skip until term..
410  while (pos < str.length() && (str[pos] != term)) {
411  if ((str[pos] != QLatin1Char(' ')) && (str[pos] != QLatin1Char('\t'))) {
412  pos = -1;
413  return QString();
414  }
415  ++pos;
416  }
417 
418  if (pos < str.length()) { // Stopped due to finding term
419  ++pos;
420  }
421 
422  return out;
423  } else {
424  return extractUntil(str, term, pos, valueSpecials);
425  }
426 }

428 static QMap<QString, QString> contentDispositionParserInternal(const QString &disposition)
429 {
430  // qDebug() << "disposition: " << disposition;
431  int pos = 0;
432  const QString strDisposition = extractUntil(disposition, QLatin1Char(';'), pos, typeSpecials).toLower();
433 
434  QMap<QString, QString> parameters;
435  QMap<QString, QString> contparams; // all parameters that contain continuations
436  QMap<QString, QString> encparams; // all parameters that have character encoding
437 
438  // the type is invalid, the complete header is junk
439  if (strDisposition.isEmpty()) {
440  return parameters;
441  }
442 
443  parameters.insert(QStringLiteral("type"), strDisposition);
444 
445  while (pos < disposition.length()) {
446  QString key = extractUntil(disposition, QLatin1Char('='), pos, attrSpecials).toLower();
447 
448  if (key.isEmpty()) {
449  // parse error in this key: do not parse more, but add up
450  // everything we already got
451  // qDebug() << "parse error in key, abort parsing";
452  break;
453  }
454 
455  QString val;
456  if (key.endsWith(QLatin1Char('*'))) {
457  val = extractUntil(disposition, QLatin1Char(';'), pos, valueSpecials);
458  } else {
459  val = extractMaybeQuotedUntil(disposition, pos);
460  }
461 
462  if (val.isEmpty()) {
463  if (pos == -1) {
464  // qDebug() << "parse error in value, abort parsing";
465  break;
466  }
467  continue;
468  }
469 
470  const int spos = key.indexOf(QLatin1Char('*'));
471  if (spos == key.length() - 1) {
472  key.chop(1);
473  encparams.insert(key, val);
474  } else if (spos >= 0) {
475  contparams.insert(key, val);
476  } else if (parameters.contains(key)) {
477  // qDebug() << "duplicate key" << key << "found, ignoring everything more";
478  parameters.remove(key);
479  return parameters;
480  } else {
481  parameters.insert(key, val);
482  }
483  }
484 
485  QMap<QString, QString>::iterator i = contparams.begin();
486  while (i != contparams.end()) {
487  QString key = i.key();
488  int spos = key.indexOf(QLatin1Char('*'));
489  bool hasencoding = false;
490 
491  if (key.at(spos + 1) != QLatin1Char('0')) {
492  ++i;
493  continue;
494  }
495 
496  // no leading zeros allowed, so delete the junk
497  int klen = key.length();
498  if (klen > spos + 2) {
499  // nothing but continuations and encodings may insert * into parameter name
500  if ((klen > spos + 3) || ((klen == spos + 3) && (key.at(spos + 2) != QLatin1Char('*')))) {
501  // qDebug() << "removing invalid key " << key << "with val" << i.value() << key.at(spos + 2);
502  i = contparams.erase(i);
503  continue;
504  }
505  hasencoding = true;
506  }
507 
508  int seqnum = 1;
509  QMap<QString, QString>::iterator partsi;
510  // we do not need to care about encoding specifications: only the first
511  // part is allowed to have one
512  QString val = i.value();
513 
514  key.chop(hasencoding ? 2 : 1);
515 
516  while ((partsi = contparams.find(key + QString::number(seqnum))) != contparams.end()) {
517  val += partsi.value();
518  contparams.erase(partsi);
519  }
520 
521  i = contparams.erase(i);
522 
523  key.chop(1);
524  if (hasencoding) {
525  encparams.insert(key, val);
526  } else {
527  if (parameters.contains(key)) {
528  // qDebug() << "duplicate key" << key << "found, ignoring everything more";
529  parameters.remove(key);
530  return parameters;
531  }
532 
533  parameters.insert(key, val);
534  }
535  }
536 
537  for (QMap<QString, QString>::iterator i = encparams.begin(); i != encparams.end(); ++i) {
538  QString val = i.value();
539 
540  // RfC 2231 encoded character set in filename
541  int spos = val.indexOf(QLatin1Char('\''));
542  if (spos == -1) {
543  continue;
544  }
545  int npos = val.indexOf(QLatin1Char('\''), spos + 1);
546  if (npos == -1) {
547  continue;
548  }
549 
550  const QString charset = val.left(spos);
551  const QString lang = val.mid(spos + 1, npos - spos - 1);
552  const QByteArray encodedVal = val.mid(npos + 1).toLatin1();
553 
554  if (! isValidPercentEncoding(encodedVal)) {
555  continue;
556  }
557 
558  const QByteArray rawval = QByteArray::fromPercentEncoding(encodedVal);
559 
560  if (charset.isEmpty() || (charset == QLatin1String("us-ascii"))) {
561  bool valid = true;
562  for (int j = rawval.length() - 1; (j >= 0) && valid; j--) {
563  valid = (rawval.at(j) >= 32);
564  }
565 
566  if (!valid) {
567  continue;
568  }
569  val = QString::fromLatin1(rawval.constData());
570  } else {
571  QTextCodec *codec = QTextCodec::codecForName(charset.toLatin1());
572  if (!codec) {
573  continue;
574  }
575  val = codec->toUnicode(rawval);
576  }
577 
578  parameters.insert(i.key(), val);
579  }
580 
581  return parameters;
582 }

584 static QMap<QString, QString> contentDispositionParser(const QString &disposition)
585 {
586  QMap<QString, QString> parameters = contentDispositionParserInternal(disposition);
587 
588  const QLatin1String fn("filename");
589  if (parameters.contains(fn)) {
590  // Content-Disposition is not allowed to dictate directory
591  // path, thus we extract the filename only.
592  const QString val = QDir::toNativeSeparators(parameters[fn]);
593  int slpos = val.lastIndexOf(QDir::separator());
594 
595  if (slpos > -1) {
596  parameters.insert(fn, val.mid(slpos + 1));
597  }
598  }
599
