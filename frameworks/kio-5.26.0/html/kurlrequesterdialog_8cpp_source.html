71 void KUrlRequesterDialogPrivate::initDialog(const QString &text, const QUrl &urlName)
72 {
73  QVBoxLayout *topLayout = new QVBoxLayout;
74  q->setLayout(topLayout);
75 
76  QLabel *label = new QLabel(text, q);
77  topLayout->addWidget(label);
78 
79  urlRequester = new KUrlRequester(urlName, q);
80  urlRequester->setMinimumWidth(urlRequester->sizeHint().width() * 3);
81  topLayout->addWidget(urlRequester);
82  urlRequester->setFocus();
83  QObject::connect(urlRequester->lineEdit(), SIGNAL(textChanged(QString)),
84  q, SLOT(_k_slotTextChanged(QString)));
85  /*
86  KFile::Mode mode = static_cast<KFile::Mode>( KFile::File |
87  KFile::ExistingOnly );
88  urlRequester_->setMode( mode );
89  */
90 
91  clearButton = new QPushButton;
92  KGuiItem::assign(clearButton, KStandardGuiItem::clear());
93  QObject::connect(clearButton, SIGNAL(clicked()), q, SLOT(_k_slotClear()));
94 
95  buttonBox = new QDialogButtonBox(q);
96  buttonBox->addButton(clearButton, QDialogButtonBox::ActionRole);
97  buttonBox->setStandardButtons(QDialogButtonBox::Ok | QDialogButtonBox::Cancel);
98  QObject::connect(buttonBox, SIGNAL(accepted()), q, SLOT(accept()));
99  QObject::connect(buttonBox, SIGNAL(rejected()), q, SLOT(reject()));
100  topLayout->addWidget(buttonBox);
101 
102  _k_slotTextChanged(urlName.toString());
103 }
