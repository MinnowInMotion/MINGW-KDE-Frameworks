140 void DropJobPrivate::slotStart()
141 {
142  Q_Q(DropJob);
143  if (!m_urls.isEmpty()) {
144  if (destIsDirectory()) {
145  handleCopyToDirectory();
146  } else { // local file
147  const QString destFile = m_destUrl.toLocalFile();
148  if (KDesktopFile::isDesktopFile(destFile)) {
149  handleDropToDesktopFile();
150  } else if (QFileInfo(destFile).isExecutable()) {
151  handleDropToExecutable();
152  } else {
153  // should not happen, if KDirModel::flags is correct
154  q->setError(KIO::ERR_ACCESS_DENIED);
155  q->emitResult();
156  }
157  }
158  } else {
159  // Dropping raw data
160  KIO::PasteJob *job = KIO::PasteJobPrivate::newJob(m_mimeData, m_destUrl, KIO::HideProgressInfo, false /*not clipboard*/);
161  QObject::connect(job, &KIO::PasteJob::itemCreated, q, &KIO::DropJob::itemCreated);
162  q->addSubjob(job);
163  }
164 }

166 // Input: m_dropAction as set by Qt at the time of the drop event
167 // Output: m_dropAction possibly modified
168 // Returns a KIO error code, in case of error.
169 int DropJobPrivate::determineDropAction()
170 {
171  Q_Q(DropJob);
172 
173  if (!KProtocolManager::supportsWriting(m_destUrl)) {
174  return KIO::ERR_CANNOT_WRITE;
175  }
176  if (!m_destItem.isNull() && !m_destItem.isWritable()) {
177  return KIO::ERR_WRITE_ACCESS_DENIED;
178  }
179 
180  bool allItemsAreFromTrash = true;
181  bool containsTrashRoot = false;
182  foreach (const QUrl &url, m_urls) {
183  const bool local = url.isLocalFile();
184  if (!local /*optimization*/ && url.scheme() == QLatin1String("trash")) {
185  if (url.path().isEmpty() || url.path() == QLatin1String("/")) {
186  containsTrashRoot = true;
187  }
188  } else {
189  allItemsAreFromTrash = false;
190  }
191  if (url.matches(m_destUrl, QUrl::StripTrailingSlash)) {
192  return KIO::ERR_DROP_ON_ITSELF;
193  }
194  }
195 
196  const bool trashing = m_destUrl.scheme() == QLatin1String("trash");
197  if (trashing) {
198  m_dropAction = Qt::MoveAction;
199  if (!q->uiDelegateExtension()->askDeleteConfirmation(m_urls, KIO::JobUiDelegate::Trash, KIO::JobUiDelegate::DefaultConfirmation)) {
200  return KIO::ERR_USER_CANCELED;
201  }
202  return KJob::NoError; // ok
203  }
204  if (containsTrashRoot) {
205  // Dropping a link to the trash: don't move the full contents, just make a link (#319660)
206  m_dropAction = Qt::LinkAction;
207  return KJob::NoError; // ok
208  }
209  if (allItemsAreFromTrash) {
210  // No point in asking copy/move/link when using dragging from the trash, just move the file out.
211  m_dropAction = Qt::MoveAction;
212  return KJob::NoError; // ok
213  }
214  if (m_keyboardModifiers & (Qt::ControlModifier | Qt::ShiftModifier | Qt::AltModifier)) {
215  // Qt determined m_dropAction from the modifiers already
216  return KJob::NoError; // ok
217  }
218 
219  // We need to ask the user with a popup menu. Let the caller know.
220  return KIO::ERR_UNKNOWN;
221 }

223 void DropJobPrivate::fillPopupMenu(QMenu *popup)
224 {
225  Q_Q(DropJob);
226 
227  // Check what the source can do
228  // TODO: Determining the mimetype of the source URLs is difficult for remote URLs,
229  // we would need to KIO::stat each URL in turn, asynchronously....
230  KFileItemList fileItems;
231  foreach (const QUrl &url, m_urls) {
232  fileItems.append(KFileItem(url));
233  }
234  const KFileItemListProperties itemProps(fileItems);
235 
236  emit q->popupMenuAboutToShow(fileItems);
237 
238  const bool sReading = itemProps.supportsReading();
239  const bool sDeleting = itemProps.supportsDeleting();
240  const bool sMoving = itemProps.supportsMoving();
241 
242  QString seq = QKeySequence(Qt::ShiftModifier).toString();
243  Q_ASSERT(seq.endsWith('+'));
244  seq.chop(1); // chop superfluous '+'
245  QAction* popupMoveAction = new QAction(i18n("&Move Here") + '\t' + seq, popup);
246  popupMoveAction->setIcon(QIcon::fromTheme(QStringLiteral("go-jump")));
247  popupMoveAction->setData(QVariant::fromValue(Qt::MoveAction));
248  seq = QKeySequence(Qt::ControlModifier).toString();
249  Q_ASSERT(seq.endsWith('+'));
250  seq.chop(1);
251  QAction* popupCopyAction = new QAction(i18n("&Copy Here") + '\t' + seq, popup);
252  popupCopyAction->setIcon(QIcon::fromTheme(QStringLiteral("edit-copy")));
253  popupCopyAction->setData(QVariant::fromValue(Qt::CopyAction));
254  seq = QKeySequence(Qt::ControlModifier + Qt::ShiftModifier).toString();
255  Q_ASSERT(seq.endsWith('+'));
256  seq.chop(1);
257  QAction* popupLinkAction = new QAction(i18n("&Link Here") + '\t' + seq, popup);
258  popupLinkAction->setIcon(QIcon::fromTheme(QStringLiteral("edit-link")));
259  popupLinkAction->setData(QVariant::fromValue(Qt::LinkAction));
260  QAction* popupCancelAction = new QAction(i18n("C&ancel") + '\t' + QKeySequence(Qt::Key_Escape).toString(), popup);
261  popupCancelAction->setIcon(QIcon::fromTheme(QStringLiteral("process-stop")));
262 
263  if (sMoving || (sReading && sDeleting)) {
264  bool equalDestination = true;
265  foreach (const QUrl &src, m_urls) {
266  if (!m_destUrl.matches(src.adjusted(QUrl::RemoveFilename), QUrl::StripTrailingSlash)) {
267  equalDestination = false;
268  break;
269  }
270  }
271 
272  if (!equalDestination) {
273  popup->addAction(popupMoveAction);
274  }
275  }
276 
277  if (sReading) {
278  popup->addAction(popupCopyAction);
279  }
280 
281  popup->addAction(popupLinkAction);
282 
283  addPluginActions(popup, itemProps);
284 
285  popup->addSeparator();
286  popup->addAction(popupCancelAction);
287 }

300  }

303  popup->addSeparator();
304  popup->addActions(m_appActions);
305  popup->addActions(m_pluginActions);
306  }

316 void DropJobPrivate::slotTriggered(QAction *action)
317 {
318  Q_Q(DropJob);
319  if (m_appActions.contains(action) || m_pluginActions.contains(action)) {
320  q->emitResult();
321  return;
322  }
323  const QVariant data = action->data();
324  if (!data.canConvert<Qt::DropAction>()) {
325  q->setError(KIO::ERR_USER_CANCELED);
326  q->emitResult();
327  return;
328  }
329  m_dropAction = data.value<Qt::DropAction>();
330  doCopyToDirectory();
331 }

333 void DropJobPrivate::handleCopyToDirectory()
334 {
335  Q_Q(DropJob);
336 
337  if (int error = determineDropAction()) {

341  fillPopupMenu(menu);

343  menu->popup(m_globalPos);

346  q->emitResult();
347  }
348  } else {

353 void DropJobPrivate::doCopyToDirectory()
354 {
355  Q_Q(DropJob);
356  KIO::CopyJob * job = 0;
357  switch (m_dropAction) {
358  case Qt::MoveAction:
359  job = KIO::move(m_urls, m_destUrl, m_flags);
360  KIO::FileUndoManager::self()->recordJob(
361  m_destUrl.scheme() == QLatin1String("trash") ? KIO::FileUndoManager::Trash : KIO::FileUndoManager::Move,
362  m_urls, m_destUrl, job);
363  break;
364  case Qt::CopyAction:
365  job = KIO::copy(m_urls, m_destUrl, m_flags);
366  KIO::FileUndoManager::self()->recordCopyJob(job);
367  break;
368  case Qt::LinkAction:
369  job = KIO::link(m_urls, m_destUrl, m_flags);
370  KIO::FileUndoManager::self()->recordCopyJob(job);
371  break;
372  default:
373  qWarning() << "Unknown drop action" << int(m_dropAction);
374  q->setError(KIO::ERR_UNSUPPORTED_ACTION);
375  q->emitResult();
376  return;
377  }
378  Q_ASSERT(job);
379  job->setMetaData(m_metaData);
380  QObject::connect(job, &KIO::CopyJob::copyingDone, q, [q](KIO::Job*, const QUrl &, const QUrl &to) {
381  emit q->itemCreated(to);
382  });
383  QObject::connect(job, &KIO::CopyJob::copyingLinkDone, q, [q](KIO::Job*, const QUrl&, const QString&, const QUrl &to) {
384  emit q->itemCreated(to);
385  });
386  q->addSubjob(job);
387 }

389 void DropJobPrivate::handleDropToDesktopFile()
390 {
391  Q_Q(DropJob);
392  const QString urlKey = QStringLiteral("URL");
393  const QString destFile = m_destUrl.toLocalFile();
394  const KDesktopFile desktopFile(destFile);
395  const KConfigGroup desktopGroup = desktopFile.desktopGroup();
396  if (desktopFile.hasApplicationType()) {
397  // Drop to application -> start app with urls as argument
398  KService service(destFile);
399  if (!KRun::runService(service, m_urls, KJobWidgets::window(q))) {
400  q->setError(KIO::ERR_CANNOT_LAUNCH_PROCESS);
401  q->setErrorText(destFile);
402  }
403  q->emitResult();
404  } else if (desktopFile.hasLinkType() && desktopGroup.hasKey(urlKey)) {
405  // Drop to link -> adjust destination directory
406  m_destUrl = QUrl::fromUserInput(desktopGroup.readPathEntry(urlKey, QString()));
407  handleCopyToDirectory();
408  } else {
409  if (desktopFile.hasDeviceType()) {
410  qWarning() << "Not re-implemented; please email kde-frameworks-devel@kde.org if you need this.";
411  // take code from libkonq's old konq_operations.cpp
412  // for now, fallback
413  }
414  // Some other kind of .desktop file (service, servicetype...)
415  q->setError(KIO::ERR_UNSUPPORTED_ACTION);
416  q->emitResult();
417  }
418 }
