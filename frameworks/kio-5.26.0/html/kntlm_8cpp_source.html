61 static QByteArray getBuf(const QByteArray &buf, const KNTLM::SecBuf &secbuf)
62 {
63  quint32 offset = qFromLittleEndian(secbuf.offset);
64  quint16 len = qFromLittleEndian(secbuf.len);
65 
66  //watch for buffer overflows
67  if (offset > (quint32) buf.size() || offset + len > (quint32) buf.size()) {
68  return QByteArray();
69  }
70 
71  return QByteArray(buf.data() + offset, len);
72 }

74 static void addBuf(QByteArray &buf, KNTLM::SecBuf &secbuf, const QByteArray &data)
75 {
76  quint32 offset = (buf.size() + 1) & 0xfffffffe;
77  quint16 len = data.size();
78  quint16 maxlen = data.size();
79 
80  secbuf.offset = qToLittleEndian((quint32) offset);
81  secbuf.len = qToLittleEndian(len);
82  secbuf.maxlen = qToLittleEndian(maxlen);
83  buf.resize(offset + len);
84  memcpy(buf.data() + offset, data.data(), data.size());
85 }

87 static QString getString(const QByteArray &buf, const KNTLM::SecBuf &secbuf, bool unicode)
88 {
89  //watch for buffer overflows
90  quint32 offset = qFromLittleEndian((quint32) secbuf.offset);
91  quint16 len = qFromLittleEndian(secbuf.len);
92 
93  if (offset > (quint32) buf.size() || offset + len > (quint32) buf.size()) {
94  return QString();
95  }
96 
97  const char *c = buf.data() + offset;
98 
99  if (unicode) {

101  }
102 
103  return QString::fromLatin1(c, len);

116 /*
117 * turns a 56 bit key into the 64 bit, odd parity key and sets the key.
118 * The key schedule ks is also set.
119 */
120 static void convertKey(unsigned char *key_56, void *ks)
121 {
122  unsigned char key[8];
123 
124  key[0] = key_56[0];
125  key[1] = ((key_56[0] << 7) & 0xFF) | (key_56[1] >> 1);
126  key[2] = ((key_56[1] << 6) & 0xFF) | (key_56[2] >> 2);
127  key[3] = ((key_56[2] << 5) & 0xFF) | (key_56[3] >> 3);
128  key[4] = ((key_56[3] << 4) & 0xFF) | (key_56[4] >> 4);
129  key[5] = ((key_56[4] << 3) & 0xFF) | (key_56[5] >> 5);
130  key[6] = ((key_56[5] << 2) & 0xFF) | (key_56[6] >> 6);
131  key[7] = (key_56[6] << 1) & 0xFF;
132 
133  for (uint i = 0; i < 8; i++) {
134  unsigned char b = key[i];
135  bool needsParity = ((((b >> 7) ^ (b >> 6) ^ (b >> 5) ^ (b >> 4) ^ (b >> 3) ^ (b >> 2) ^ (b >> 1)) & 0x01) == 0);
136 
137  if (needsParity) {
138  key[i] |= 0x01;
139  } else {
140  key[i] &= 0xfe;
141  }
142  }
143 
144  ntlm_des_set_key((DES_KEY *) ks, (char *) &key, sizeof(key));
145  memset(&key, 0, sizeof(key));
146 }

148 static QByteArray createBlob(const QByteArray &targetinfo)
149 {
150  QByteArray blob(NTLM_BLOB_SIZE + 4 + targetinfo.size(), 0);
151 
152  KNTLM::Blob *bl = (KNTLM::Blob *) blob.data();
153  bl->signature = qToBigEndian((quint32) 0x01010000);
154  quint64 now = QDateTime::currentDateTimeUtc().toTime_t();
155  now += (quint64) 3600 * (quint64) 24 * (quint64) 134774;
156  now *= (quint64) 10000000;
157  bl->timestamp = qToLittleEndian(now);
158 
159  for (uint i = 0; i < 8; i++) {
160  bl->challenge[i] = KRandom::random() % 0xff;
161  }
162 
163  memcpy(blob.data() + NTLM_BLOB_SIZE, targetinfo.data(), targetinfo.size());
164  return blob;
165 }

167 static QByteArray hmacMD5(const QByteArray &data, const QByteArray &key)
168 {
169  QByteArray ret;
170  QByteArray ipad(64, 0x36);
171  QByteArray opad(64, 0x5c);
172 
173  Q_ASSERT(key.size() <= 64);
174 
175  for (int i = qMin(key.size(), 64) - 1; i >= 0; i--) {
176  ipad.data()[i] ^= key[i];
177  opad.data()[i] ^= key[i];
178  }
179 
180  QByteArray content(ipad + data);
181 
182  QCryptographicHash md5(QCryptographicHash::Md5);
183  md5.addData(content);
184  content = opad + md5.result();
185 
186  md5.reset();
187  md5.addData(content);
188 
189  return md5.result();
190 }

194 bool KNTLM::getNegotiate(QByteArray &negotiate, const QString &domain, const QString &workstation, quint32 flags)
195 {
196  QByteArray rbuf(sizeof(Negotiate), 0);
197 
198  memcpy(rbuf.data(), NTLM_SIGNATURE, sizeof(NTLM_SIGNATURE));
199  ((Negotiate *) rbuf.data())->msgType = qToLittleEndian((quint32) 1);
200 
201  if (!domain.isEmpty()) {
202  flags |= Negotiate_Domain_Supplied;
203  addString(rbuf, ((Negotiate *) rbuf.data())->domain, domain);
204  }
205 
206  if (!workstation.isEmpty()) {
207  flags |= Negotiate_WS_Supplied;
208  addString(rbuf, ((Negotiate *) rbuf.data())->workstation, workstation);
209  }
210 
211  ((Negotiate *) rbuf.data())->flags = qToLittleEndian(flags);
212  negotiate = rbuf;
213  return true;
214 }

216 bool KNTLM::getAuth(QByteArray &auth, const QByteArray &challenge,
217  const QString &user, const QString &password, const QString &domain,
218  const QString &workstation, AuthFlags authflags)
219 {
220  QByteArray rbuf(sizeof(Auth), 0);
221  Challenge *ch = (Challenge *) challenge.data();
222  QByteArray response;
223  const uint chsize = challenge.size();
224  bool unicode = false;
225  QString dom;
226 
227  //challenge structure too small
228  if (chsize < 32) {
229  return false;
230  }
231 
232  unicode = qFromLittleEndian(ch->flags) & Negotiate_Unicode;
233 
234  if (domain.isEmpty()) {
235  dom = getString(challenge, ch->targetName, unicode);
236  } else {
237  dom = domain;
238  }
239 
240  memcpy(rbuf.data(), NTLM_SIGNATURE, sizeof(NTLM_SIGNATURE));
241  ((Auth *) rbuf.data())->msgType = qToLittleEndian((quint32) 3);
242  ((Auth *) rbuf.data())->flags = ch->flags;
243  QByteArray targetInfo;
244  if (chsize >= sizeof(Challenge)) {
245  targetInfo = getBuf(challenge, ch->targetInfo);
246  }
247 
248  if (!(authflags & Force_V1) &&
249  ((authflags & Force_V2) ||
250  (!targetInfo.isEmpty() && (qFromLittleEndian(ch->flags) & Negotiate_Target_Info))) /* may support NTLMv2 */) {
251  bool ret = false;
252 
253  if (qFromLittleEndian(ch->flags) & Negotiate_NTLM) {
254  if (targetInfo.isEmpty()) {
255  return false;
256  }
257 
258  response = getNTLMv2Response(dom, user, password, targetInfo, ch->challengeData);
259  addBuf(rbuf, ((Auth *) rbuf.data())->ntResponse, response);
260  ret = true;
261  }
262 
263  if (authflags & Add_LM) {
264  response = getLMv2Response(dom, user, password, ch->challengeData);
265  addBuf(rbuf, ((Auth *) rbuf.data())->lmResponse, response);
266  ret = true;
267  }
268 
269  if (!ret) {
270  return false;
271  }
272  } else { //if no targetinfo structure and NTLMv2 or LMv2 not forced, or v1 forced, try the older methods
273  bool ret = false;
274 
275  if (qFromLittleEndian(ch->flags) & Negotiate_NTLM) {
276  response = getNTLMResponse(password, ch->challengeData);
277  addBuf(rbuf, ((Auth *) rbuf.data())->ntResponse, response);
278  ret = true;
279  }
280 
281  if (authflags & Add_LM) {
282  response = getLMResponse(password, ch->challengeData);
283  addBuf(rbuf, ((Auth *) rbuf.data())->lmResponse, response);
284  ret = true;
285  }
286 
287  if (!ret) {
288  return false;
289  }
290  }
291 
292  if (!dom.isEmpty()) {
293  addString(rbuf, ((Auth *) rbuf.data())->domain, dom, unicode);
294  }
295 
296  addString(rbuf, ((Auth *) rbuf.data())->user, user, unicode);
297 
298  if (!workstation.isEmpty()) {
299  addString(rbuf, ((Auth *) rbuf.data())->workstation, workstation, unicode);
300  }
301 
302  auth = rbuf;
303  return true;
304 }

318 QByteArray KNTLM::lmHash(const QString &password)
319 {
320  QByteArray keyBytes(14, 0);
321  QByteArray hash(16, 0);
322  DES_KEY ks;
323  const char *magic = "KGS!@#$%";
324 
325  strncpy(keyBytes.data(), password.toUpper().toLocal8Bit().constData(), 14);
326 
327  convertKey((unsigned char *) keyBytes.data(), &ks);
328  ntlm_des_ecb_encrypt(magic, 8, &ks, (unsigned char *) hash.data());
329 
330  convertKey((unsigned char *) keyBytes.data() + 7, &ks);
331  ntlm_des_ecb_encrypt(magic, 8, &ks, (unsigned char *) hash.data() + 8);
332 
333  keyBytes.fill(0);
334  memset(&ks, 0, sizeof(ks));
335 
336  return hash;
337 }

339 QByteArray KNTLM::lmResponse(const QByteArray &hash, const unsigned char *challenge)
340 {
341  DES_KEY ks;
342  QByteArray answer(24, 0);
343 
344  convertKey((unsigned char *) hash.data(), &ks);
345  ntlm_des_ecb_encrypt(challenge, 8, &ks, (unsigned char *) answer.data());
346 
347  convertKey((unsigned char *) hash.data() + 7, &ks);
348  ntlm_des_ecb_encrypt(challenge, 8, &ks, (unsigned char *) answer.data() + 8);
349 
350  convertKey((unsigned char *) hash.data() + 14, &ks);
351  ntlm_des_ecb_encrypt(challenge, 8, &ks, (unsigned char *) answer.data() + 16);
352 
353  memset(&ks, 0, sizeof(ks));
