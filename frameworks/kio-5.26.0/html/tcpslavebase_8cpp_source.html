87  void setSslMetaData()
88  {
89  sslMetaData.insert(QStringLiteral("ssl_in_use"), QStringLiteral("TRUE"));
90  KSslCipher cipher = socket.sessionCipher();
91  sslMetaData.insert(QStringLiteral("ssl_protocol_version"), socket.negotiatedSslVersionName());
92  QString sslCipher = cipher.encryptionMethod() + '\n';
93  sslCipher += cipher.authenticationMethod() + '\n';
94  sslCipher += cipher.keyExchangeMethod() + '\n';
95  sslCipher += cipher.digestMethod();
96  sslMetaData.insert(QStringLiteral("ssl_cipher"), sslCipher);
97  sslMetaData.insert(QStringLiteral("ssl_cipher_name"), cipher.name());
98  sslMetaData.insert(QStringLiteral("ssl_cipher_used_bits"), QString::number(cipher.usedBits()));
99  sslMetaData.insert(QStringLiteral("ssl_cipher_bits"), QString::number(cipher.supportedBits()));
100  sslMetaData.insert(QStringLiteral("ssl_peer_ip"), ip);
101 
102  // try to fill in the blanks, i.e. missing certificates, and just assume that
103  // those belong to the peer (==website or similar) certificate.
104  for (int i = 0; i < sslErrors.count(); i++) {
105  if (sslErrors[i].certificate().isNull()) {
106  const QList<QSslCertificate> peerCertificateChain = socket.peerCertificateChain();
107  sslErrors[i] = KSslError(sslErrors[i].error(),
108  peerCertificateChain[0]);
109  }
110  }
111 
112  QString errorStr;
113  // encode the two-dimensional numeric error list using '\n' and '\t' as outer and inner separators
114  Q_FOREACH (const QSslCertificate &cert, socket.peerCertificateChain()) {
115  Q_FOREACH (const KSslError &error, sslErrors) {
116  if (error.certificate() == cert) {
117  errorStr += QString::number(static_cast<int>(error.error())) + '\t';
118  }
119  }
120  if (errorStr.endsWith('\t')) {
121  errorStr.chop(1);
122  }
123  errorStr += '\n';
124  }
125  errorStr.chop(1);
126  sslMetaData.insert(QStringLiteral("ssl_cert_errors"), errorStr);
127 
128  QString peerCertChain;
129  Q_FOREACH (const QSslCertificate &cert, socket.peerCertificateChain()) {
130  peerCertChain.append(cert.toPem());
131  peerCertChain.append('\x01');
132  }
133  peerCertChain.chop(1);
134  sslMetaData.insert(QStringLiteral("ssl_peer_chain"), peerCertChain);
135  sendSslMetaData();
136  }

263 ssize_t TCPSlaveBase::readLine(char *data, ssize_t len)
264 {
265  if (d->usingSSL && (d->socket.encryptionMode() != KTcpSocket::SslClientMode)) {
266  d->clearSslMetaData();
267  //qDebug() << "lost SSL connection.";
268  return -1;
269  }
270 
271  const int timeout = (d->isBlocking ? -1 : (readTimeout() * 1000));
272  ssize_t readTotal = 0;
273  do {
274  if (!d->socket.bytesAvailable()) {
275  d->socket.waitForReadyRead(timeout);
276  }
277  ssize_t readStep = d->socket.readLine(&data[readTotal], len - readTotal);
278  if (readStep == -1 || (readStep == 0 && d->socket.state() != KTcpSocket::ConnectedState)) {
279  return -1;
280  }
281  readTotal += readStep;
282  } while (readTotal == 0 || data[readTotal - 1] != '\n');
283 
284  return readTotal;

301 int TCPSlaveBase::connectToHost(const QString &host, quint16 port, QString *errorString)
302 {
303  d->clearSslMetaData(); //We have separate connection and SSL setup phases
304 
305  if (errorString) {
306  errorString->clear(); // clear prior error messages.
307  }
308 
309  d->socket.setVerificationPeerName(host); // Used for ssl certificate verification (SNI)
310 
311  // - leaving SSL - warn before we even connect
312  //### see if it makes sense to move this into the HTTP ioslave which is the only
313  // user.
314  if (metaData(QStringLiteral("main_frame_request")) == QLatin1String("TRUE") //### this looks *really* unreliable
315  && metaData(QStringLiteral("ssl_activate_warnings")) == QLatin1String("TRUE")
316  && metaData(QStringLiteral("ssl_was_in_use")) == QLatin1String("TRUE")
317  && !d->autoSSL) {
318  if (d->sslSettings.warnOnLeave()) {
319  int result = messageBox(i18n("You are about to leave secure "
320  "mode. Transmissions will no "
321  "longer be encrypted.\nThis "
322  "means that a third party could "
323  "observe your data in transit."),
324  WarningContinueCancel,
325  i18n("Security Information"),
326  i18n("C&ontinue Loading"), QString(),
327  QStringLiteral("WarnOnLeaveSSLMode"));
328 
329  if (result == SlaveBase::Cancel) {
330  if (errorString) {
331  *errorString = host;
332  }
333  return ERR_USER_CANCELED;
334  }
335  }
336  }
337 
338  /*
339  SSL handshake is attempted in the following order:
340 
341  1.) KTcpSocket::SecureProtocols
342  2.) KTcpSocket::TlsV1_2
343  3.) KTcpSocket::TlsV1_1
344  4.) KTcpSocket::TlsV1_0
345 
346  Note that we indivially attempt connection with each TLS version
347  because some sites don't support SSL negotiation. #275524
348 
349  The version used to successfully make encrypted connection with the
350  remote server is cached within the process to make subsequent
351  connection requests to the same server faster.
352  */
353 
354  const int lastSslVerson = config()->readEntry("LastUsedSslVersion", static_cast<int>(KTcpSocket::SecureProtocols));
355  KTcpSocket::SslVersion trySslVersion = static_cast<KTcpSocket::SslVersion>(lastSslVerson);
356  KTcpSocket::SslVersions alreadyTriedSslVersions = trySslVersion;
357 
358  const int timeout = (connectTimeout() * 1000); // 20 sec timeout value
359  while (true) {
360  disconnectFromHost(); //Reset some state, even if we are already disconnected
361  d->host = host;
362 
363  d->socket.connectToHost(host, port);
364  /*const bool connectOk = */d->socket.waitForConnected(timeout > -1 ? timeout : -1);
365 
366  /*qDebug() << "Socket: state=" << d->socket.state()
367  << ", error=" << d->socket.error()
368  << ", connected?" << connectOk;*/
369 
370  if (d->socket.state() != KTcpSocket::ConnectedState) {
371  if (errorString) {
372  *errorString = host + QLatin1String(": ") + d->socket.errorString();
373  }
374  switch (d->socket.error()) {
375  case KTcpSocket::UnsupportedSocketOperationError:
376  return ERR_UNSUPPORTED_ACTION;
377  case KTcpSocket::RemoteHostClosedError:
378  return ERR_CONNECTION_BROKEN;
379  case KTcpSocket::SocketTimeoutError:
380  return ERR_SERVER_TIMEOUT;
381  case KTcpSocket::HostNotFoundError:
382  return ERR_UNKNOWN_HOST;
383  default:
384  return ERR_CANNOT_CONNECT;
385  }
386  }
387 
388  //### check for proxyAuthenticationRequiredError
389 
390  d->ip = d->socket.peerAddress().toString();
391  d->port = d->socket.peerPort();
392 
393  if (d->autoSSL) {
394  SslResult res = d->startTLSInternal(trySslVersion, timeout);
395  if ((res & ResultFailed) && (res & ResultFailedEarly)) {
396  if (!(alreadyTriedSslVersions & KTcpSocket::SecureProtocols)) {
397  trySslVersion = KTcpSocket::SecureProtocols;
398  alreadyTriedSslVersions |= trySslVersion;
399  continue;
400  }
401 
402  if (!(alreadyTriedSslVersions & KTcpSocket::TlsV1_2)) {
403  trySslVersion = KTcpSocket::TlsV1;
404  alreadyTriedSslVersions |= trySslVersion;
405  continue;
406  }
407 
408  if (!(alreadyTriedSslVersions & KTcpSocket::TlsV1_1)) {
409  trySslVersion = KTcpSocket::TlsV1;
410  alreadyTriedSslVersions |= trySslVersion;
411  continue;
412  }
413 
414  if (!(alreadyTriedSslVersions & KTcpSocket::TlsV1_0)) {
415  trySslVersion = KTcpSocket::TlsV1_0;
416  alreadyTriedSslVersions |= trySslVersion;
417  continue;
418  }
419  }
420 
421  //### SSL 2.0 is (close to) dead and it's a good thing, too.
422  if (res & ResultFailed) {
423  if (errorString) {
424  *errorString = i18nc("%1 is a host name", "%1: SSL negotiation failed", host);
425  }
426  return ERR_CANNOT_CONNECT;
427  }
428  }
429  // If the SSL handshake was done with anything protocol other than the default,
430  // save that information so that any subsequent requests do not have to do thesame thing.
431  if (trySslVersion != KTcpSocket::SecureProtocols && lastSslVerson == KTcpSocket::SecureProtocols) {
432  setMetaData(QStringLiteral("{internal~currenthost}LastUsedSslVersion"),
433  QString::number(trySslVersion));
434  }
435  return 0;
436  }
437  Q_ASSERT(false);
438  // Code flow never gets here but let's make the compiler happy.
439  // More: the stack allocation of QSslSettings seems to be confusing the compiler;
440  // in fact, any non-POD allocation does.
441  // even a 'return 0;' directly after the allocation (so before the while(true))
442  // is ignored. definitely seems to be a compiler bug? - aseigo
443  return 0;
444 }

498 TCPSlaveBase::SslResult TCPSlaveBase::TcpSlaveBasePrivate::startTLSInternal(KTcpSocket::SslVersion version,
499  int waitForEncryptedTimeout)
500 {
501  q->selectClientCertificate();
502 
503  //setMetaData("ssl_session_id", d->kssl->session()->toString());
504  //### we don't support session reuse for now...
505  usingSSL = true;
506 
507  // Set the SSL version to use...
508  socket.setAdvertisedSslVersion(version);
509 
510  /* Usually ignoreSslErrors() would be called in the slot invoked by the sslErrors()
511  signal but that would mess up the flow of control. We will check for errors
512  anyway to decide if we want to continue connecting. Otherwise ignoreSslErrors()
513  before connecting would be very insecure. */
514  socket.ignoreSslErrors();
515  socket.startClientEncryption();
516  const bool encryptionStarted = socket.waitForEncrypted(waitForEncryptedTimeout);
517 
518  //Set metadata, among other things for the "SSL Details" dialog
519  KSslCipher cipher = socket.sessionCipher();
520 
521  if (!encryptionStarted || socket.encryptionMode() != KTcpSocket::SslClientMode
522  || cipher.isNull() || cipher.usedBits() == 0 || socket.peerCertificateChain().isEmpty()) {
523  usingSSL = false;
524  clearSslMetaData();
525  /*qDebug() << "Initial SSL handshake failed. encryptionStarted is"
526  << encryptionStarted << ", cipher.isNull() is" << cipher.isNull()
527  << ", cipher.usedBits() is" << cipher.usedBits()
528  << ", length of certificate chain is" << socket.peerCertificateChain().count()
529  << ", the socket says:" << socket.errorString()
530  << "and the list of SSL errors contains"
531  << socket.sslErrors().count() << "items.";*/
532  /*Q_FOREACH(const KSslError& sslError, socket.sslErrors()) {
533  qDebug() << "SSL ERROR: (" << sslError.error() << ")" << sslError.errorString();
534  }*/
535  return ResultFailed | ResultFailedEarly;
536  }
537 
538  /*qDebug() << "Cipher info - "
539  << " advertised SSL protocol version" << socket.advertisedSslVersion()
540  << " negotiated SSL protocol version" << socket.negotiatedSslVersion()
541  << " authenticationMethod:" << cipher.authenticationMethod()
542  << " encryptionMethod:" << cipher.encryptionMethod()
543  << " keyExchangeMethod:" << cipher.keyExchangeMethod()
544  << " name:" << cipher.name()
545  << " supportedBits:" << cipher.supportedBits()
546  << " usedBits:" << cipher.usedBits();*/
547 
548  sslErrors = socket.sslErrors();
549 
550  // TODO: review / rewrite / remove the comment
551  // The app side needs the metadata now for the SSL error dialog (if any) but
552  // the same metadata will be needed later, too. When "later" arrives the slave
553  // may actually be connected to a different application that doesn't know
554  // the metadata the slave sent to the previous application.
555  // The quite important SSL indicator icon in Konqi's URL bar relies on metadata
556  // from here, for example. And Konqi will be the second application to connect
557  // to the slave.
558  // Therefore we choose to have our metadata and send it, too :)
559  setSslMetaData();
560  q->sendAndKeepMetaData();
561 
562  SslResult rc = q->verifyServerCertificate();
563  if (rc & ResultFailed) {
564  usingSSL = false;
565  clearSslMetaData();
566  //qDebug() << "server certificate verification failed.";
567  socket.disconnectFromHost(); //Make the connection fail (cf. ignoreSslErrors())
568  return ResultFailed;
569  } else if (rc & ResultOverridden) {
570  //qDebug() << "server certificate verification failed but continuing at user's request.";
571  }
572 
573  //"warn" when starting SSL/TLS
574  if (q->metaData(QStringLiteral("ssl_activate_warnings")) == QLatin1String("TRUE")
575  && q->metaData(QStringLiteral("ssl_was_in_use")) == QLatin1String("FALSE")
576  && sslSettings.warnOnEnter()) {
577 
578  int msgResult = q->messageBox(i18n("You are about to enter secure mode. "
579  "All transmissions will be encrypted "
580  "unless otherwise noted.\nThis means "
581  "that no third party will be able to "
582  "easily observe your data in transit."),
583  WarningYesNo,
584  i18n("Security Information"),
585  i18n("Display SSL &Information"),
586  i18n("C&onnect"),
587  QStringLiteral("WarnOnEnterSSLMode"));
588  if (msgResult == SlaveBase::Yes) {
589  q->messageBox(SSLMessageBox /*==the SSL info dialog*/, host);
590  }
591  }
592 
593  return rc;
594 }

599  QString certname; // the cert to use this session
600  bool send = false, prompt = false, save = false, forcePrompt = false;
601  KSSLCertificateHome::KSSLAuthAction aa;
602 
603  setMetaData("ssl_using_client_cert", "FALSE"); // we change this if needed
604 
605  if (metaData("ssl_no_client_cert") == "TRUE") {
606  return;
607  }
608  forcePrompt = (metaData("ssl_force_cert_prompt") == "TRUE");
609 
610  // Delete the old cert since we're certainly done with it now
611  if (d->pkcs) {
612  delete d->pkcs;
613  d->pkcs = NULL;
614  }
615 
616  if (!d->kssl) {
617  return;
618  }
619 
620  // Look for a general certificate
621  if (!forcePrompt) {
622  certname = KSSLCertificateHome::getDefaultCertificateName(&aa);
623  switch (aa) {
624  case KSSLCertificateHome::AuthSend:
625  send = true; prompt = false;
626  break;
627  case KSSLCertificateHome::AuthDont:
628  send = false; prompt = false;
629  certname.clear();
630  break;
631  case KSSLCertificateHome::AuthPrompt:
632  send = false; prompt = true;
633  break;
634  default:
635  break;
636  }
637  }
638 
639  // Look for a certificate on a per-host basis as an override
640  QString tmpcn = KSSLCertificateHome::getDefaultCertificateName(d->host, &aa);
641  if (aa != KSSLCertificateHome::AuthNone) { // we must override
642  switch (aa) {
643  case KSSLCertificateHome::AuthSend:
644  send = true;
645  prompt = false;
646  certname = tmpcn;
647  break;
648  case KSSLCertificateHome::AuthDont:
649  send = false;
650  prompt = false;
651  certname.clear();
652  break;
653  case KSSLCertificateHome::AuthPrompt:
654  send = false;
655  prompt = true;
656  certname = tmpcn;
657  break;
658  default:
659  break;
660  }
661  }
662 
663  // Finally, we allow the application to override anything.
664  if (hasMetaData("ssl_demand_certificate")) {
665  certname = metaData("ssl_demand_certificate");
666  if (!certname.isEmpty()) {
667  forcePrompt = false;
668  prompt = false;
669  send = true;
670  }
671  }
672 
673  if (certname.isEmpty() && !prompt && !forcePrompt) {
674  return;
675  }
676 
677  // Ok, we're supposed to prompt the user....
678  if (prompt || forcePrompt) {
679  QStringList certs = KSSLCertificateHome::getCertificateList();
680 
681  QStringList::const_iterator it = certs.begin();
682  while (it != certs.end()) {
683  KSSLPKCS12 *pkcs = KSSLCertificateHome::getCertificateByName(*it);
684  if (pkcs && (!pkcs->getCertificate() ||
685  !pkcs->getCertificate()->x509V3Extensions().certTypeSSLClient())) {
686  it = certs.erase(it);
687  } else {
688  ++it;
689  }
690  delete pkcs;
691  }
692 
693  if (certs.isEmpty()) {
694  return; // we had nothing else, and prompt failed
695  }
696 
697  QDBusConnectionInterface *bus = QDBusConnection::sessionBus().interface();
698  if (!bus->isServiceRegistered("org.kde.kio.uiserver")) {
699  bus->startService("org.kde.kuiserver");
700  }
701 
702  QDBusInterface uis("org.kde.kio.uiserver", "/UIServer", "org.kde.KIO.UIServer");
703 
704  QDBusMessage retVal = uis.call("showSSLCertDialog", d->host, certs, metaData("window-id").toLongLong());
705  if (retVal.type() == QDBusMessage::ReplyMessage) {
706  if (retVal.arguments().at(0).toBool()) {
707  send = retVal.arguments().at(1).toBool();
708  save = retVal.arguments().at(2).toBool();
709  certname = retVal.arguments().at(3).toString();
710  }
711  }
712  }
713 
714  // The user may have said to not send the certificate,
715  // but to save the choice
716  if (!send) {
717  if (save) {
718  KSSLCertificateHome::setDefaultCertificate(certname, d->host,
719  false, false);
720  }
721  return;
722  }
723 
724  // We're almost committed. If we can read the cert, we'll send it now.
725  KSSLPKCS12 *pkcs = KSSLCertificateHome::getCertificateByName(certname);
726  if (!pkcs && KSSLCertificateHome::hasCertificateByName(certname)) { // We need the password
727  KIO::AuthInfo ai;
728  bool first = true;
729  do {
730  ai.prompt = i18n("Enter the certificate password:");
731  ai.caption = i18n("SSL Certificate Password");
732  ai.url.setScheme("kssl");
733  ai.url.setHost(certname);
734  ai.username = certname;
735  ai.keepPassword = true;
736 
737  bool showprompt;
738  if (first) {
739  showprompt = !checkCachedAuthentication(ai);
740  } else {
741  showprompt = true;
742  }
743  if (showprompt) {
744  if (!openPasswordDialog(ai, first ? QString() :
745  i18n("Unable to open the certificate. Try a new password?"))) {
746  break;
747  }
748  }
749 
750  first = false;
751  pkcs = KSSLCertificateHome::getCertificateByName(certname, ai.password);
752  } while (!pkcs);
753 
754  }
755 
756  // If we could open the certificate, let's send it
757  if (pkcs) {
758  if (!d->kssl->setClientCertificate(pkcs)) {
759  messageBox(Information, i18n("The procedure to set the "
760  "client certificate for the session "
761  "failed."), i18n("SSL"));
762  delete pkcs; // we don't need this anymore
763  pkcs = 0L;
764  } else {
765  //qDebug() << "Client SSL certificate is being used.";
766  setMetaData("ssl_using_client_cert", "TRUE");
767  if (save) {
768  KSSLCertificateHome::setDefaultCertificate(certname, d->host,
769  true, false);
770  }
771  }
772  d->pkcs = pkcs;
773  }
774 #endif
