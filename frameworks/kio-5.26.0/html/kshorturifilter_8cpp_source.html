113 bool KShortUriFilter::filterUri( KUriFilterData& data ) const
114 {
115  /*
116  * Here is a description of how the shortURI deals with the supplied
117  * data. First it expands any environment variable settings and then
118  * deals with special shortURI cases. These special cases are the "smb:"
119  * URL scheme which is very specific to KDE, "#" and "##" which are
120  * shortcuts for man:/ and info:/ protocols respectively. It then handles
121  * local files. Then it checks to see if the URL is valid and one that is
122  * supported by KDE's IO system. If all the above checks fails, it simply
123  * lookups the URL in the user-defined list and returns without filtering
124  * if it is not found. TODO: the user-defined table is currently only manually
125  * hackable and is missing a config dialog.
126  */
127 
128  //QUrl url = data.uri();
129  QString cmd = data.typedString();
130 
131  int firstNonSlash = 0;
132  while (firstNonSlash < cmd.length() && (cmd.at(firstNonSlash) == '/')) {
133  firstNonSlash++;
134  }
135  if (firstNonSlash > 1) {
136  cmd = cmd.mid(firstNonSlash - 1);
137  }
138 
139  // Replicate what KUrl(cmd) did in KDE4. This could later be folded into the checks further down...
140  QUrl url;
141  if (QDir::isAbsolutePath(cmd)) {
142  url = QUrl::fromLocalFile(cmd);
143  } else {
144  url.setUrl(cmd);
145  }
146 
147  // WORKAROUND: Allow the use of '@' in the username component of a URL since
148  // other browsers such as firefox in their infinite wisdom allow such blatant
149  // violations of RFC 3986. BR# 69326/118413.
150  if (cmd.count(QLatin1Char('@')) > 1) {
151  const int lastIndex = cmd.lastIndexOf(QLatin1Char('@'));
152  // Percent encode all but the last '@'.
153  QString encodedCmd = QUrl::toPercentEncoding(cmd.left(lastIndex), ":/");
154  encodedCmd += cmd.midRef(lastIndex);
155  cmd = encodedCmd;
156  url.setUrl(encodedCmd);
157  }
158 
159  const bool isMalformed = !url.isValid();
160  QString protocol = url.scheme();
161 
162  qCDebug(category) << cmd;
163 
164  // Fix misparsing of "foo:80", QUrl thinks "foo" is the protocol and "80" is the path.
165  // However, be careful not to do that for valid hostless URLs, e.g. file:///foo!
166  if (!protocol.isEmpty() && url.host().isEmpty() && !url.path().isEmpty()
167  && cmd.contains(':') && !isKnownProtocol(protocol)) {
168  protocol.clear();
169  }
170 
171  qCDebug(category) << "url=" << url << "cmd=" << cmd << "isMalformed=" << isMalformed;
172 
173  // TODO: Make this a bit more intelligent for Minicli! There
174  // is no need to make comparisons if the supplied data is a local
175  // executable and only the argument part, if any, changed! (Dawit)
176  // You mean caching the last filtering, to try and reuse it, to save stat()s? (David)
177 
178  const QString starthere_proto = QStringLiteral("start-here:");
179  if (cmd.indexOf(starthere_proto) == 0 )
180  {
181  setFilteredUri( data, QUrl(QStringLiteral("system:/")) );
182  setUriType( data, KUriFilterData::LocalDir );
183  return true;
184  }
185 
186  // Handle MAN & INFO pages shortcuts...
187  const QString man_proto = QStringLiteral("man:");
188  const QString info_proto = QStringLiteral("info:");
189  if( cmd[0] == '#' ||
190  cmd.indexOf( man_proto ) == 0 ||
191  cmd.indexOf( info_proto ) == 0 )
192  {
193  if( cmd.leftRef(2) == QLatin1String("##") )
194  cmd = QStringLiteral("info:/") + cmd.mid(2);
195  else if ( cmd[0] == '#' )
196  cmd = QStringLiteral("man:/") + cmd.mid(1);
197 
198  else if ((cmd==info_proto) || (cmd==man_proto))
199  cmd+='/';
200 
201  setFilteredUri( data, QUrl( cmd ));
202  setUriType( data, KUriFilterData::Help );
203  return true;
204  }
205 
206  // Detect UNC style (aka windows SMB) URLs
207  if ( cmd.startsWith( QLatin1String( "\\\\") ) )
208  {
209  // make sure path is unix style
210  cmd.replace('\\', '/');
211  cmd.prepend( QLatin1String( "smb:" ) );
212  setFilteredUri( data, QUrl( cmd ));
213  setUriType( data, KUriFilterData::NetProtocol );
214  return true;
215  }
216 
217  bool expanded = false;
218 
219  // Expanding shortcut to HOME URL...
220  QString path;
221  QString ref;
222  QString query;
223  QString nameFilter;
224 
225  if (QDir::isRelativePath(cmd) && QUrl(cmd).isRelative()) {
226  path = cmd;
227  qCDebug(category) << "path=cmd=" << path;
228  } else {
229  if (url.isLocalFile())
230  {
231  qCDebug(category) << "hasRef=" << url.hasFragment();
232  // Split path from ref/query
233  // but not for "/tmp/a#b", if "a#b" is an existing file,
234  // or for "/tmp/a?b" (#58990)
235  if( ( url.hasFragment() || !url.query().isEmpty() )
236  && !url.path().endsWith(QLatin1Char('/')) ) // /tmp/?foo is a namefilter, not a query
237  {
238  path = url.path();
239  ref = url.fragment();
240  qCDebug(category) << "isLocalFile set path to" << path << "and ref to" << ref;
241  query = url.query();
242  if (path.isEmpty() && !url.host().isEmpty())
243  path = '/';
244  }
245  else
246  {
247  if (cmd.startsWith(QLatin1String("file://"))) {
248  path = cmd.mid(strlen("file://"));
249  } else {
250  path = cmd;
251  }
252  qCDebug(category) << "(2) path=cmd=" << path;
253  }
254  }
255  }
256 
257  if( path[0] == '~' )
258  {
259  int slashPos = path.indexOf('/');
260  if( slashPos == -1 )
261  slashPos = path.length();
262  if( slashPos == 1 ) // ~/
263  {
264  path.replace ( 0, 1, QDir::homePath() );
265  }
266  else // ~username/
267  {
268  const QString userName (path.mid( 1, slashPos-1 ));
269  KUser user (userName);
270  if( user.isValid() && !user.homeDir().isEmpty())
271  {
272  path.replace (0, slashPos, user.homeDir());
273  }
274  else
275  {
276  if (user.isValid()) {
277  setErrorMsg(data, i18n("<qt><b>%1</b> does not have a home folder.</qt>", userName));
278  } else {
279  setErrorMsg(data, i18n("<qt>There is no user called <b>%1</b>.</qt>", userName));
280  }
281  setUriType( data, KUriFilterData::Error );
282  // Always return true for error conditions so
283  // that other filters will not be invoked !!
284  return true;
285  }
286  }
287  expanded = true;
288  }
289  else if ( path[0] == '$' ) {
290  // Environment variable expansion.
291  auto match = sEnvVarExp.match(path);
292  if ( match.hasMatch() )
293  {
294  QByteArray exp = qgetenv( path.mid( 1, match.capturedLength() - 1 ).toLocal8Bit().data() );
295  if (!exp.isEmpty()) {
296  path.replace( 0, match.capturedLength(), QFile::decodeName(exp) );
297  expanded = true;
298  }
299  }
300  }
301 
302  if ( expanded || cmd.startsWith( '/' ) )
303  {
304  // Look for #ref again, after $ and ~ expansion (testcase: $QTDIR/doc/html/functions.html#s)
305  // Can't use QUrl here, setPath would escape it...
306  const int pos = path.indexOf('#');
307  if ( pos > -1 )
308  {
309  const QString newPath = path.left( pos );
310  if ( QFile::exists( newPath ) )
311  {
312  ref = path.mid( pos + 1 );
313  path = newPath;
314  qCDebug(category) << "Extracted ref: path=" << path << " ref=" << ref;
315  }
316  }
317  }
318 
319 
320 
321  bool isLocalFullPath = QDir::isAbsolutePath(path);
322 
323  // Checking for local resource match...
324  // Determine if "uri" is an absolute path to a local resource OR
325  // A local resource with a supplied absolute path in KUriFilterData
326  const QString abs_path = data.absolutePath();
327 
328  const bool canBeAbsolute = (protocol.isEmpty() && !abs_path.isEmpty());
329  const bool canBeLocalAbsolute = (canBeAbsolute && abs_path[0] =='/' && !isMalformed);
330  bool exists = false;
331 
332  /*qCDebug(category) << "abs_path=" << abs_path
333  << "protocol=" << protocol
334  << "canBeAbsolute=" << canBeAbsolute
335  << "canBeLocalAbsolute=" << canBeLocalAbsolute
336  << "isLocalFullPath=" << isLocalFullPath;*/
337 
338  QT_STATBUF buff;
339  if ( canBeLocalAbsolute )
340  {
341  QString abs = QDir::cleanPath( abs_path );
342  // combine absolute path (abs_path) and relative path (cmd) into abs_path
343  int len = path.length();
344  if( (len==1 && path[0]=='.') || (len==2 && path[0]=='.' && path[1]=='.') )
345  path += '/';
346  qCDebug(category) << "adding " << abs << " and " << path;
347  abs = QDir::cleanPath(abs + '/' + path);
348  qCDebug(category) << "checking whether " << abs << " exists.";
349  // Check if it exists
350  if(QT_STAT(QFile::encodeName(abs), &buff) == 0) {
351  path = abs; // yes -> store as the new cmd
352  exists = true;
353  isLocalFullPath = true;
354  }
355  }
356 
357  if (isLocalFullPath && !exists && !isMalformed) {
358  exists = QT_STAT(QFile::encodeName(path), &buff) == 0;
359 
360  if ( !exists ) {
361  // Support for name filter (/foo/*.txt), see also KonqMainWindow::detectNameFilter
362  // If the app using this filter doesn't support it, well, it'll simply error out itself
363  int lastSlash = path.lastIndexOf( '/' );
364  if ( lastSlash > -1 && path.indexOf( ' ', lastSlash ) == -1 ) // no space after last slash, otherwise it's more likely command-line arguments
365  {
366  QString fileName = path.mid( lastSlash + 1 );
367  QString testPath = path.left(lastSlash);
368  if ((fileName.indexOf('*') != -1 || fileName.indexOf('[') != -1 || fileName.indexOf( '?' ) != -1)
369  && QT_STAT(QFile::encodeName(testPath), &buff) == 0) {
370  nameFilter = fileName;
371  qCDebug(category) << "Setting nameFilter to" << nameFilter << "and path to" << testPath;
372  path = testPath;
373  exists = true;
374  }
375  }
376  }
377  }
378 
379  qCDebug(category) << "path =" << path << " isLocalFullPath=" << isLocalFullPath << " exists=" << exists << " url=" << url;
380  if( exists )
381  {
382  QUrl u = QUrl::fromLocalFile(path);
383  qCDebug(category) << "ref=" << ref << "query=" << query;
384  u.setFragment(ref);
385  u.setQuery(query);
386 
387  if (!KUrlAuthorized::authorizeUrlAction( QStringLiteral("open"), QUrl(), u))
388  {
389  // No authorization, we pretend it's a file will get
390  // an access denied error later on.
391  setFilteredUri( data, u );
392  setUriType( data, KUriFilterData::LocalFile );
393  return true;
394  }
395 
396  // Can be abs path to file or directory, or to executable with args
397  bool isDir = ((buff.st_mode & QT_STAT_MASK) == QT_STAT_DIR);
398  if( !isDir && access ( QFile::encodeName(path).data(), X_OK) == 0 )
399  {
400  qCDebug(category) << "Abs path to EXECUTABLE";
401  setFilteredUri( data, u );
402  setUriType( data, KUriFilterData::Executable );
403  return true;
404  }
405 
406  // Open "uri" as file:/xxx if it is a non-executable local resource.
407  if( isDir || (( buff.st_mode & QT_STAT_MASK ) == QT_STAT_REG) )
408  {
409  qCDebug(category) << "Abs path as local file or directory";
410  if ( !nameFilter.isEmpty() )
411  u.setPath( u.path() + '/' + nameFilter );
412  setFilteredUri( data, u );
413  setUriType( data, ( isDir ) ? KUriFilterData::LocalDir : KUriFilterData::LocalFile );
414  return true;
415  }
416 
417  // Should we return LOCAL_FILE for non-regular files too?
418  qCDebug(category) << "File found, but not a regular file nor dir... socket?";
419  }
420 
421  if( data.checkForExecutables())
422  {
423  // Let us deal with possible relative URLs to see
424  // if it is executable under the user's $PATH variable.
425  // We try hard to avoid parsing any possible command
426  // line arguments or options that might have been supplied.
427  QString exe = removeArgs( cmd );
428  qCDebug(category) << "findExe with" << exe;
429 
430  if (!QStandardPaths::findExecutable( exe ).isNull() )
431  {
432  qCDebug(category) << "EXECUTABLE exe=" << exe;
433  setFilteredUri( data, QUrl::fromLocalFile( exe ));
434  // check if we have command line arguments
435  if( exe != cmd )
436  setArguments(data, cmd.right(cmd.length() - exe.length()));
437  setUriType( data, KUriFilterData::Executable );
438  return true;
439  }
440  }
441 
442  // Process URLs of known and supported protocols so we don't have
443  // to resort to the pattern matching scheme below which can possibly
444  // slow things down...
445  if ( !isMalformed && !isLocalFullPath && !protocol.isEmpty() )
446  {
447  qCDebug(category) << "looking for protocol" << protocol;
448  if (isKnownProtocol(protocol))
449  {
450  setFilteredUri( data, url );
451  if ( protocol == QLatin1String("man") || protocol == QLatin1String("help") )
452  setUriType( data, KUriFilterData::Help );
453  else
454  setUriType( data, KUriFilterData::NetProtocol );
455  return true;
456  }
457  }
458 
459  // Short url matches
460  if ( !cmd.contains( ' ' ) )
461  {
462  // Okay this is the code that allows users to supply custom matches for
463  // specific URLs using Qt's regexp class. This is hard-coded for now.
464  // TODO: Make configurable at some point...
465  Q_FOREACH(const URLHint& hint, m_urlHints)
466  {
467  qCDebug(category) << "testing regexp for" << hint.prepend;
468  if (hint.regexp.indexIn(cmd) == 0)
469  {
470  const QString cmdStr = hint.prepend + cmd;
471  QUrl url(cmdStr);
472  qCDebug(category) << "match - prepending" << hint.prepend << "->" << cmdStr << "->" << url;
473  setFilteredUri( data, url );
474  setUriType( data, hint.type );
475  return true;
476  }
477  }
478 
479  // No protocol and not malformed means a valid short URL such as kde.org or
480  // user@192.168.0.1. However, it might also be valid only because it lacks
481  // the scheme component, e.g. www.kde,org (illegal ',' before 'org'). The
482  // check below properly deciphers the difference between the two and sends
483  // back the proper result.
484  if (protocol.isEmpty() && isPotentialShortURL(cmd))
485  {
486  QString urlStr = data.defaultUrlScheme();
487  if (urlStr.isEmpty())
488  urlStr = m_strDefaultUrlScheme;
489 
490  const int index = urlStr.indexOf(QLatin1Char(':'));
491  if (index == -1 || !isKnownProtocol(urlStr.left(index)))
492  urlStr += QStringLiteral("://");
493  urlStr += cmd;
494 
495  QUrl url (urlStr);
496  if (url.isValid())
497  {
498  setFilteredUri(data, url);
499  setUriType(data, KUriFilterData::NetProtocol);
500  }
501  else if (isKnownProtocol(url.scheme()))
502  {
503  setFilteredUri(data, data.uri());
504  setUriType(data, KUriFilterData::Error);
505  }
506  return true;
507  }
508  }
509 
510  // If we previously determined that the URL might be a file,
511  // and if it doesn't exist, then error
512  if( isLocalFullPath && !exists )
513  {
514  QUrl u = QUrl::fromLocalFile(path);
515  u.setFragment(ref);
516 
517  if (!KUrlAuthorized::authorizeUrlAction( QStringLiteral("open"), QUrl(), u))
518  {
519  // No authorization, we pretend it exists and will get
520  // an access denied error later on.
521  setFilteredUri( data, u );
522  setUriType( data, KUriFilterData::LocalFile );
523  return true;
524  }
525  qCDebug(category) << "fileNotFound -> ERROR";
526  setErrorMsg( data, i18n( "<qt>The file or folder <b>%1</b> does not exist.</qt>", data.uri().toDisplayString() ) );
527  setUriType( data, KUriFilterData::Error );
528  return true;
529  }
530 
531  // If we reach this point, we cannot filter this thing so simply return false
532  // so that other filters, if present, can take a crack at it.
533  return false;
534 }

547 void KShortUriFilter::configure()
548 {

550  KConfigGroup cg( config.group("") );
551 
552  m_strDefaultUrlScheme = cg.readEntry( "DefaultProtocol", QStringLiteral("http://") );
553  const EntryMap patterns = config.entryMap( QStringLiteral("Pattern") );
554  const EntryMap protocols = config.entryMap( QStringLiteral("Protocol") );
555  if (cg.readEntry("Verbose", false)) {
556  category.setEnabled(QtDebugMsg, true);
557  }
558  KConfigGroup typeGroup(&config, "Type");
559 
560  for( EntryMap::ConstIterator it = patterns.begin(); it != patterns.end(); ++it )
561  {

564  {
565  int type = typeGroup.readEntry(it.key(), -1);
566  if (type > -1 && type <= KUriFilterData::Unknown)
567  m_urlHints.append( URLHint(it.value(), protocol, static_cast<KUriFilterData::UriTypes>(type) ) );
568  else
569  m_urlHints.append( URLHint(it.value(), protocol) );
570  }
571  }
