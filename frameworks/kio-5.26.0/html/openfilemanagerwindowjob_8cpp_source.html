114 void OpenFileManagerWindowDBusStrategy::start(const QList<QUrl> &urls, const QByteArray &asn)
115 {
116  // see the spec at: https://www.freedesktop.org/wiki/Specifications/file-manager-interface/
117 
118  QDBusMessage msg = QDBusMessage::createMethodCall(QStringLiteral("org.freedesktop.FileManager1"),
119  QStringLiteral("/org/freedesktop/FileManager1"),
120  QStringLiteral("org.freedesktop.FileManager1"),
121  QStringLiteral("ShowItems"));
122 
123  msg << QUrl::toStringList(urls) << QString::fromUtf8(asn);
124 
125  QDBusPendingReply<void> reply = QDBusConnection::sessionBus().asyncCall(msg);
126  QDBusPendingCallWatcher *watcher = new QDBusPendingCallWatcher(reply, job);
127  QObject::connect(watcher, &QDBusPendingCallWatcher::finished, job, [=](QDBusPendingCallWatcher *watcher) {
128  QDBusPendingReply<void> reply = *watcher;
129  watcher->deleteLater();
130 
131  if (reply.isError()) {
132  // Try the KRun strategy as fallback, also calls emitResult inside
133  OpenFileManagerWindowKRunStrategy kRunStrategy(job);
134  kRunStrategy.start(urls, asn);
135  return;
136  }
137 
138  emitResultProxy();
139  });
140 }

142 void OpenFileManagerWindowKRunStrategy::start(const QList<QUrl> &urls, const QByteArray &asn)
143 {
144  if (!KRun::runUrl(urls.value(0).adjusted(QUrl::RemoveFilename),
145  QStringLiteral("inode/directory"),
146  KJobWidgets::window(job), // window
147  false, // delete temp files
148  false, // run executables
149  QString(), // sugested file name
150  asn)) {
151  emitResultProxy(OpenFileManagerWindowJob::LaunchFailedError);
152  return;
153  }
154 
155  emitResultProxy();
156 }
