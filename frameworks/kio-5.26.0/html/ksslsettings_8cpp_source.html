29 class CipherNode
30 {
31 public:
32  CipherNode(const char *_name, int _keylen) :
33  name(_name), keylen(_keylen) {}
34  QString name;
35  int keylen;
36  inline int operator==(CipherNode &x)
37  {
38  return ((x.keylen == keylen) && (x.name == name));
39  }
40  inline int operator< (CipherNode &x)
41  {
42  return keylen < x.keylen;
43  }
44  inline int operator<=(CipherNode &x)
45  {
46  return keylen <= x.keylen;
47  }
48  inline int operator> (CipherNode &x)
49  {
50  return keylen > x.keylen;
51  }
52  inline int operator>=(CipherNode &x)
53  {
54  return keylen >= x.keylen;
55  }
56 };

115 void KSSLSettings::load()
116 {
117  d->m_cfg->reparseConfiguration();
118 
119  KConfigGroup cfg(d->m_cfg, "Warnings");
120  d->m_bWarnOnEnter = cfg.readEntry("OnEnter", false);
121  d->m_bWarnOnLeave = cfg.readEntry("OnLeave", true);
122  d->m_bWarnOnUnencrypted = cfg.readEntry("OnUnencrypted", false);
123  d->m_bWarnOnMixed = cfg.readEntry("OnMixed", true);
124 
125  cfg = KConfigGroup(d->m_cfg, "Validation");
126  d->m_bWarnSelfSigned = cfg.readEntry("WarnSelfSigned", true);
127  d->m_bWarnExpired = cfg.readEntry("WarnExpired", true);
128  d->m_bWarnRevoked = cfg.readEntry("WarnRevoked", true);
129 
130  cfg = KConfigGroup(d->m_cfg, "EGD");
131  d->m_bUseEGD = cfg.readEntry("UseEGD", false);
132  d->m_bUseEFile = cfg.readEntry("UseEFile", false);
133  d->m_EGDPath = cfg.readPathEntry("EGDPath", QString());
134 
135  cfg = KConfigGroup(d->m_cfg, "Auth");
136  d->m_bSendX509 = ("send" == cfg.readEntry("AuthMethod", ""));
137  d->m_bPromptX509 = ("prompt" == cfg.readEntry("AuthMethod", ""));
138 }

154 void KSSLSettings::save()
155 {
156  KConfigGroup cfg(d->m_cfg, "Warnings");
157  cfg.writeEntry("OnEnter", d->m_bWarnOnEnter);
158  cfg.writeEntry("OnLeave", d->m_bWarnOnLeave);
159  cfg.writeEntry("OnUnencrypted", d->m_bWarnOnUnencrypted);
160  cfg.writeEntry("OnMixed", d->m_bWarnOnMixed);
161 
162  cfg = KConfigGroup(d->m_cfg, "Validation");
163  cfg.writeEntry("WarnSelfSigned", d->m_bWarnSelfSigned);
164  cfg.writeEntry("WarnExpired", d->m_bWarnExpired);
165  cfg.writeEntry("WarnRevoked", d->m_bWarnRevoked);
166 
167  cfg = KConfigGroup(d->m_cfg, "EGD");
168  cfg.writeEntry("UseEGD", d->m_bUseEGD);
169  cfg.writeEntry("UseEFile", d->m_bUseEFile);
170  cfg.writePathEntry("EGDPath", d->m_EGDPath);
171 
172  d->m_cfg->sync();
173  // FIXME - ciphers
174 #if 0
175 #if KSSL_HAVE_SSL
176  cfg.setGroup("SSLv3");
177  for (unsigned int i = 0; i < v3ciphers.count(); i++) {
178  QString ciphername;
179  ciphername.sprintf("cipher_%s", v3ciphers[i].ascii());
180  if (v3selectedciphers.contains(v3ciphers[i])) {
181  cfg.writeEntry(ciphername, true);
182  } else {
183  cfg.writeEntry(ciphername, false);
184  }
185  }
186  d->m_cfg->sync();
187 #endif
188 
189  // insure proper permissions -- contains sensitive data
190  QString cfgName(QStandardPaths::locate(QStandardPaths::GenericConfigLocation, "cryptodefaults"));
191  if (!cfgName.isEmpty()) {
192  KDE::chmod(cfgName, 0600);
193  }
194 #endif
195 }
