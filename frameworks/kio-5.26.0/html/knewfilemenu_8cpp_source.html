143 void KNewFileMenuSingleton::parseFiles()
144 {
145  //qDebug();
146  filesParsed = true;
147  QMutableListIterator<KNewFileMenuSingleton::Entry> templIter(*templatesList);
148  while (templIter.hasNext()) {
149  KNewFileMenuSingleton::Entry &templ = templIter.next();
150  const QString filePath = templ.filePath;
151  if (!filePath.isEmpty()) {
152  QString text;
153  QString templatePath;
154  // If a desktop file, then read the name from it.
155  // Otherwise (or if no name in it?) use file name
156  if (KDesktopFile::isDesktopFile(filePath)) {
157  KDesktopFile desktopFile(filePath);
158  if (desktopFile.noDisplay()) {
159  templIter.remove();
160  continue;
161  }
162  text = desktopFile.readName();
163  templ.icon = desktopFile.readIcon();
164  templ.comment = desktopFile.readComment();
165  QString type = desktopFile.readType();
166  if (type == QLatin1String("Link")) {
167  templatePath = desktopFile.desktopGroup().readPathEntry("URL", QString());
168  if (templatePath[0] != '/' && !templatePath.startsWith(QLatin1String("__"))) {
169  if (templatePath.startsWith(QLatin1String("file:/"))) {
170  templatePath = QUrl(templatePath).toLocalFile();
171  } else {
172  // A relative path, then (that's the default in the files we ship)
173  QString linkDir = filePath.left(filePath.lastIndexOf('/') + 1 /*keep / */);
174  //qDebug() << "linkDir=" << linkDir;
175  templatePath = linkDir + templatePath;
176  }
177  }
178  }
179  if (templatePath.isEmpty()) {
180  // No URL key, this is an old-style template
181  templ.entryType = KNewFileMenuSingleton::Template;
182  templ.templatePath = templ.filePath; // we'll copy the file
183  } else {
184  templ.entryType = KNewFileMenuSingleton::LinkToTemplate;
185  templ.templatePath = templatePath;
186  }
187 
188  }
189  if (text.isEmpty()) {
190  text = QUrl(filePath).fileName();
191  if (text.endsWith(QLatin1String(".desktop"))) {
192  text.truncate(text.length() - 8);
193  }
194  }
195  templ.text = text;
196  /*// qDebug() << "Updating entry with text=" << text
197  << "entryType=" << templ.entryType
198  << "templatePath=" << templ.templatePath;*/
199  } else {
200  templ.entryType = KNewFileMenuSingleton::Separator;
201  }
202  }
203 }

442 void KNewFileMenuPrivate::executeOtherDesktopFile(const KNewFileMenuSingleton::Entry &entry)
443 {
444  if (!checkSourceExists(entry.templatePath)) {
445  return;
446  }
447 
448  QList<QUrl>::const_iterator it = m_popupFiles.constBegin();
449  for (; it != m_popupFiles.constEnd(); ++it) {
450  QString text = entry.text;
451  text.remove(QStringLiteral("...")); // the ... is fine for the menu item but not for the default filename
452  text = text.trimmed(); // In some languages, there is a space in front of "...", see bug 268895
453  // KDE5 TODO: remove the "..." from link*.desktop files and use i18n("%1...") when making
454  // the action.
455  QString name = text;
456  text.append(QStringLiteral(".desktop"));
457 
458  const QUrl directory = mostLocalUrl(*it);
459  const QUrl defaultFile = QUrl::fromLocalFile(directory.toLocalFile() + '/' + KIO::encodeFileName(text));
460  if (defaultFile.isLocalFile() && QFile::exists(defaultFile.toLocalFile())) {
461  text = KIO::suggestName(directory, text);
462  }
463 
464  QUrl templateUrl;
465  bool usingTemplate = false;
466  if (entry.templatePath.startsWith(QLatin1String(":/"))) {
467  QTemporaryFile *tmpFile = QTemporaryFile::createNativeFile(entry.templatePath);
468  tmpFile->setAutoRemove(false);
469  QString tempFileName = tmpFile->fileName();
470  tmpFile->close();
471 
472  KDesktopFile df(tempFileName);
473  KConfigGroup group = df.desktopGroup();
474  group.writeEntry("Name", name);
475  templateUrl = QUrl::fromLocalFile(tempFileName);
476  m_tempFileToDelete = tempFileName;
477  usingTemplate = true;
478  } else {
479  templateUrl = QUrl::fromLocalFile(entry.templatePath);
480  }
481  QDialog *dlg = new KPropertiesDialog(templateUrl, directory, text, m_parentWidget);
482  dlg->setModal(q->isModal());
483  dlg->setAttribute(Qt::WA_DeleteOnClose);
484  QObject::connect(dlg, SIGNAL(applied()), q, SLOT(_k_slotOtherDesktopFile()));
485  if (usingTemplate) {
486  QObject::connect(dlg, SIGNAL(propertiesClosed()), q, SLOT(_k_slotOtherDesktopFileClosed()));
487  }
488  dlg->show();
489  }
490  // We don't set m_src here -> there will be no copy, we are done.
491 }

551 void KNewFileMenuPrivate::executeStrategy()
552 {
553  m_tempFileToDelete = m_copyData.tempFileToDelete();
554  const QString src = m_copyData.sourceFileToCopy();
555  QString chosenFileName = expandTilde(m_copyData.chosenFileName(), true);
556 
557  if (src.isEmpty()) {
558  return;
559  }
560  QUrl uSrc(QUrl::fromLocalFile(src));
561 
562  // In case the templates/.source directory contains symlinks, resolve
563  // them to the target files. Fixes bug #149628.
564  KFileItem item(uSrc, QString(), KFileItem::Unknown);
565  if (item.isLink()) {
566  uSrc.setPath(item.linkDest());
567  }
568 
569  if (!m_copyData.m_isSymlink) {
570  // If the file is not going to be detected as a desktop file, due to a
571  // known extension (e.g. ".pl"), append ".desktop". #224142.
572  QFile srcFile(uSrc.toLocalFile());
573  if (srcFile.open(QIODevice::ReadOnly)) {
574  QMimeDatabase db;
575  QMimeType wantedMime = db.mimeTypeForUrl(uSrc);
576  QMimeType mime = db.mimeTypeForFileNameAndData(m_copyData.m_chosenFileName, srcFile.read(1024));
577  //qDebug() << "mime=" << mime->name() << "wantedMime=" << wantedMime->name();
578  if (!mime.inherits(wantedMime.name()))
579  if (!wantedMime.preferredSuffix().isEmpty()) {
580  chosenFileName += QLatin1Char('.') + wantedMime.preferredSuffix();
581  }
582  }
583  }
584 
585  // The template is not a desktop file [or it's a URL one]
586  // Copy it.
587  QList<QUrl>::const_iterator it = m_popupFiles.constBegin();
588  for (; it != m_popupFiles.constEnd(); ++it) {
589  QUrl dest = *it;
590  dest.setPath(dest.path() + '/' + KIO::encodeFileName(chosenFileName));
591 
592  QList<QUrl> lstSrc;
593  lstSrc.append(uSrc);
594  KIO::Job *kjob;
595  if (m_copyData.m_isSymlink) {
596  KIO::CopyJob *linkJob = KIO::linkAs(uSrc, dest);
597  kjob = linkJob;
598  KIO::FileUndoManager::self()->recordCopyJob(linkJob);
599  } else if (src.startsWith(QLatin1String(":/"))) {
600  QFile srcFile(src);
601  if (!srcFile.open(QIODevice::ReadOnly)) {
602  return;
603  }
604  // The QFile won't live long enough for the job, so let's buffer the contents
605  const QByteArray srcBuf(srcFile.readAll());
606  KIO::StoredTransferJob* putJob = KIO::storedPut(srcBuf, dest, -1);
607  kjob = putJob;
608  KIO::FileUndoManager::self()->recordJob(KIO::FileUndoManager::Put, QList<QUrl>(), dest, putJob);
609  } else {
610  //qDebug() << "KIO::copyAs(" << uSrc.url() << "," << dest.url() << ")";
611  KIO::CopyJob *job = KIO::copyAs(uSrc, dest);
612  job->setDefaultPermissions(true);
613  kjob = job;
614  KIO::FileUndoManager::self()->recordCopyJob(job);
615  }
616  KJobWidgets::setWindow(kjob, m_parentWidget);
617  QObject::connect(kjob, SIGNAL(result(KJob*)), q, SLOT(slotResult(KJob*)));
618  }
619 }

633 void KNewFileMenuPrivate::fillMenu()
634 {
635  QMenu *menu = q->menu();
636  menu->clear();
637  m_menuDev->menu()->clear();
638  m_newDirAction = 0;
639 
640  QSet<QString> seenTexts;
641  // these shall be put at special positions
642  QAction *linkURL = 0;
643  QAction *linkApp = 0;
644  QAction *linkPath = 0;
645 
646  KNewFileMenuSingleton *s = kNewMenuGlobals();
647  int i = 1;
648  KNewFileMenuSingleton::EntryList::iterator templ = s->templatesList->begin();
649  const KNewFileMenuSingleton::EntryList::iterator templ_end = s->templatesList->end();
650  for (; templ != templ_end; ++templ, ++i) {
651  KNewFileMenuSingleton::Entry &entry = *templ;
652  if (entry.entryType != KNewFileMenuSingleton::Separator) {
653  // There might be a .desktop for that one already, if it's a kdelnk
654  // This assumes we read .desktop files before .kdelnk files ...
655 
656  // In fact, we skip any second item that has the same text as another one.
657  // Duplicates in a menu look bad in any case.
658 
659  const bool bSkip = seenTexts.contains(entry.text);
660  if (bSkip) {
661  // qDebug() << "skipping" << entry.filePath;
662  } else {
663  seenTexts.insert(entry.text);
664  //const KNewFileMenuSingleton::Entry entry = templatesList->at(i-1);
665 
666  const QString templatePath = entry.templatePath;
667  // The best way to identify the "Create Directory", "Link to Location", "Link to Application" was the template
668  if (templatePath.endsWith(QLatin1String("emptydir"))) {
669  QAction *act = new QAction(q);
670  m_newDirAction = act;
671  act->setIcon(QIcon::fromTheme(entry.icon));
672  act->setText(i18nc("@item:inmenu Create New", "%1", entry.text));
673  act->setActionGroup(m_newMenuGroup);
674  menu->addAction(act);
675 
676  QAction *sep = new QAction(q);
677  sep->setSeparator(true);
678  menu->addAction(sep);
679  } else {
680 
681  if (!m_supportedMimeTypes.isEmpty()) {
682  bool keep = false;
683 
684  // We need to do mimetype filtering, for real files.
685  const bool createSymlink = entry.templatePath == QLatin1String("__CREATE_SYMLINK__");
686  if (createSymlink) {
687  keep = true;
688  } else if (!KDesktopFile::isDesktopFile(entry.templatePath)) {
689 
690  // Determine mimetype on demand
691  QMimeDatabase db;
692  QMimeType mime;
693  if (entry.mimeType.isEmpty()) {
694  mime = db.mimeTypeForFile(entry.templatePath);
695  //qDebug() << entry.templatePath << "is" << mime.name();
696  entry.mimeType = mime.name();
697  } else {
698  mime = db.mimeTypeForName(entry.mimeType);
699  }
700  Q_FOREACH (const QString &supportedMime, m_supportedMimeTypes) {
701  if (mime.inherits(supportedMime)) {
702  keep = true;
703  break;
704  }
705  }
706  }
707 
708  if (!keep) {
709  //qDebug() << "Not keeping" << entry.templatePath;
710  continue;
711  }

876 void KNewFileMenuPrivate::_k_slotFillTemplates()
877 {
878  KNewFileMenuSingleton *s = kNewMenuGlobals();
879  //qDebug();
880 
881  QStringList dirs = { QStringLiteral(":/kio5/newfile-templates") };
882  dirs += QStandardPaths::locateAll(QStandardPaths::GenericDataLocation, QStringLiteral("templates"), QStandardPaths::LocateDirectory);
883 
884  // Ensure any changes in the templates dir will call this
885  if (! s->dirWatch) {
886  s->dirWatch = new KDirWatch;
887  for (QStringList::const_iterator it = dirs.constBegin(); it != dirs.constEnd(); ++it) {
888  //qDebug() << "Templates resource dir:" << *it;
889  s->dirWatch->addDir(*it);
890  }
891  QObject::connect(s->dirWatch, SIGNAL(dirty(QString)),
892  q, SLOT(_k_slotFillTemplates()));
893  QObject::connect(s->dirWatch, SIGNAL(created(QString)),
894  q, SLOT(_k_slotFillTemplates()));
895  QObject::connect(s->dirWatch, SIGNAL(deleted(QString)),
896  q, SLOT(_k_slotFillTemplates()));
897  // Ok, this doesn't cope with new dirs in XDG_DATA_DIRS, but that's another story
898  }
899  ++s->templatesVersion;
900  s->filesParsed = false;
901 
902  s->templatesList->clear();
903 
904  // Look into "templates" dirs.
905  QStringList files;
906  QDir dir;
907 
908  Q_FOREACH (const QString &path, dirs) {
909  dir.setPath(path);
910  const QStringList &entryList(dir.entryList(QStringList() << QStringLiteral("*.desktop"), QDir::Files));
911  Q_FOREACH (const QString &entry, entryList) {
912  const QString file = dir.path() + QLatin1Char('/') + entry;
913  files.append(file);
914  }
915  }
916 
917  QMap<QString, KNewFileMenuSingleton::Entry> slist; // used for sorting
918  QMap<QString, EntryWithName> ulist; // entries with unique URLs
919  Q_FOREACH (const QString &file, files) {
920  //qDebug() << file;
921  if (file[0] != '.') {
922  KNewFileMenuSingleton::Entry e;
923  e.filePath = file;
924  e.entryType = KNewFileMenuSingleton::Unknown; // not parsed yet
925 
926  // Put Directory first in the list (a bit hacky),
927  // and TextFile before others because it's the most used one.
928  // This also sorts by user-visible name.
929  // The rest of the re-ordering is done in fillMenu.
930  const KDesktopFile config(file);
931  QString url = config.desktopGroup().readEntry("URL");
932  QString key = config.desktopGroup().readEntry("Name");
933  if (file.endsWith(QLatin1String("Directory.desktop"))) {
934  key.prepend('0');
935  } else if (file.endsWith(QLatin1String("TextFile.desktop"))) {
936  key.prepend('1');
937  } else {
938  key.prepend('2');
939  }
940  EntryWithName en = { key, e };
941  if (ulist.contains(url)) {
942  ulist.remove(url);
943  }
944  ulist.insert(url, en);
945  }
946  }
947  QMap<QString, EntryWithName>::iterator it = ulist.begin();
948  for (; it != ulist.end(); ++it) {
949  EntryWithName ewn = *it;
950  slist.insert(ewn.key, ewn.entry);
951  }
952  (*s->templatesList) += slist.values();
953 }

994 void KNewFileMenuPrivate::_k_slotUrlDesktopFile()
995 {
996  KNameAndUrlInputDialog *dlg = static_cast<KNameAndUrlInputDialog *>(m_fileDialog);
997 
998  m_copyData.m_chosenFileName = dlg->name(); // no path
999  QUrl linkUrl = dlg->url();
1000 
1001  // Filter user input so that short uri entries, e.g. www.kde.org, are
1002  // handled properly. This not only makes the icon detection below work
1003  // properly, but opening the URL link where the short uri will not be
1004  // sent to the application (opening such link Konqueror fails).
1005  KUriFilterData uriData;
1006  uriData.setData(linkUrl); // the url to put in the file
1007  uriData.setCheckForExecutables(false);
1008 
1009  if (KUriFilter::self()->filterUri(uriData, QStringList() << QStringLiteral("kshorturifilter"))) {
1010  linkUrl = uriData.uri();
1011  }
1012 
1013  if (m_copyData.m_chosenFileName.isEmpty() || linkUrl.isEmpty()) {
1014  return;
1015  }
1016 
1017  // It's a "URL" desktop file; we need to make a temp copy of it, to modify it
1018  // before copying it to the final destination [which could be a remote protocol]
1019  QTemporaryFile tmpFile;
1020  tmpFile.setAutoRemove(false); // done below
1021  if (!tmpFile.open()) {
1022  qCritical() << "Couldn't create temp file!";
1023  return;
1024  }
1025 
1026  if (!checkSourceExists(m_copyData.m_templatePath)) {
1027  return;
1028  }
1029 
1030  // First copy the template into the temp file
1031  QFile file(m_copyData.m_templatePath);
1032  if (!file.open(QIODevice::ReadOnly)) {
1033  qCritical() << "Couldn't open template" << m_copyData.m_templatePath;
1034  return;
1035  }
1036  const QByteArray data = file.readAll();
1037  tmpFile.write(data);
1038  const QString tempFileName = tmpFile.fileName();
1039  Q_ASSERT(!tempFileName.isEmpty());
1040  tmpFile.close();
1041  file.close();
1042 
1043  KDesktopFile df(tempFileName);
1044  KConfigGroup group = df.desktopGroup();
1045  group.writeEntry("Icon", KProtocolInfo::icon(linkUrl.scheme()));
1046  group.writePathEntry("URL", linkUrl.toDisplayString());
1047  df.sync();
1048 
1049  m_copyData.m_src = tempFileName;
1050  m_copyData.m_tempFileToDelete = tempFileName;
1051 
1052  executeStrategy();
1053 }
