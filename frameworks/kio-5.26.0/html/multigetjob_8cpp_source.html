95 void MultiGetJobPrivate::flushQueue(RequestQueue &queue)
96 {
97  // Use multi-get
98  // Scan all jobs in m_waitQueue
99  RequestQueue::iterator wqit = m_waitQueue.begin();
100  const RequestQueue::iterator wqend = m_waitQueue.end();
101  while (wqit != wqend) {
102  const GetRequest &entry = *wqit;
103  if ((m_url.scheme() == entry.url.scheme()) &&
104  (m_url.host() == entry.url.host()) &&
105  (m_url.port() == entry.url.port()) &&
106  (m_url.userName() == entry.url.userName())) {
107  queue.append(entry);
108  wqit = m_waitQueue.erase(wqit);
109  } else {
110  ++wqit;
111  }
112  }
113  // Send number of URLs, (URL, metadata)*
114  KIO_ARGS << (qint32) queue.count();
115  RequestQueue::const_iterator qit = queue.begin();
116  const RequestQueue::const_iterator qend = queue.end();
117  for (; qit != qend; ++qit) {
118  stream << (*qit).url << (*qit).metaData;
119  }
120  m_packedArgs = packedArgs;
121  m_command = CMD_MULTI_GET;
122  m_outgoingMetaData.clear();
123 }

125 void MultiGetJobPrivate::start(Slave *slave)
126 {
127  // Add first job from m_waitQueue and add it to m_activeQueue
128  GetRequest entry = m_waitQueue.takeFirst();
129  m_activeQueue.append(entry);
130 
131  m_url = entry.url;
132 
133  if (!entry.url.scheme().startsWith(QLatin1String("http"))) {
134  // Use normal get
135  KIO_ARGS << entry.url;
136  m_packedArgs = packedArgs;
137  m_outgoingMetaData = entry.metaData;
138  m_command = CMD_GET;
139  b_multiGetActive = false;
140  } else {
141  flushQueue(m_activeQueue);
142  b_multiGetActive = true;
143  }
144 
145  TransferJobPrivate::start(slave); // Anything else to do??
146 }

148 bool MultiGetJobPrivate::findCurrentEntry()
149 {
150  if (b_multiGetActive) {
151  long id = m_incomingMetaData[QStringLiteral("request-id")].toLong();
152  RequestQueue::const_iterator qit = m_activeQueue.begin();
153  const RequestQueue::const_iterator qend = m_activeQueue.end();
154  for (; qit != qend; ++qit) {
155  if ((*qit).id == id) {
156  m_currentEntry = *qit;
157  return true;
158  }
159  }
160  m_currentEntry.id = 0;
161  return false;
162  } else {
163  if (m_activeQueue.isEmpty()) {
164  return false;
165  }
166  m_currentEntry = m_activeQueue.first();
167  return true;
168  }
169 }

171 void MultiGetJob::slotRedirection(const QUrl &url)
172 {
173  Q_D(MultiGetJob);
174  if (!d->findCurrentEntry()) {
175  return; // Error
176  }
177  if (!KUrlAuthorized::authorizeUrlAction(QStringLiteral("redirect"), d->m_url, url)) {
178  qWarning() << "Redirection from" << d->m_currentEntry.url << "to" << url << "REJECTED!";
179  return;
180  }
181  d->m_redirectionURL = url;
182  get(d->m_currentEntry.id, d->m_redirectionURL, d->m_currentEntry.metaData); // Try again
183 }

185 void MultiGetJob::slotFinished()
186 {
187  Q_D(MultiGetJob);
188  if (!d->findCurrentEntry()) {
189  return;
190  }
191  if (d->m_redirectionURL.isEmpty()) {
192  // No redirection, tell the world that we are finished.
193  emit result(d->m_currentEntry.id);
194  }
195  d->m_redirectionURL = QUrl();
196  setError(0);
197  d->m_incomingMetaData.clear();
198  d->m_activeQueue.removeAll(d->m_currentEntry);
199  if (d->m_activeQueue.count() == 0) {
200  if (d->m_waitQueue.count() == 0) {
201  // All done
202  TransferJob::slotFinished();
203  } else {
204  // return slave to pool
205  // fetch new slave for first entry in d->m_waitQueue and call start
206  // again.
207  d->slaveDone();
208 
209  d->m_url = d->m_waitQueue.first().url;
210  if ((d->m_extraFlags & JobPrivate::EF_KillCalled) == 0) {
211  Scheduler::doJob(this);
212  }
213  }
214  }
215 }

225 void MultiGetJob::slotMimetype(const QString &_mimetype)
226 {
227  Q_D(MultiGetJob);
228  if (d->b_multiGetActive) {
229  MultiGetJobPrivate::RequestQueue newQueue;
230  d->flushQueue(newQueue);
231  if (!newQueue.isEmpty()) {
232  d->m_activeQueue += newQueue;
233  d->m_slave->send(d->m_command, d->m_packedArgs);
234  }
235  }
236  if (!d->findCurrentEntry()) {
237  return; // Error, unknown request!
238  }
239  emit mimetype(d->m_currentEntry.id, _mimetype);
240 }
