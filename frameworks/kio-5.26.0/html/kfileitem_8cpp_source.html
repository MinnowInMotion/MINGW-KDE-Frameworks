177 void KFileItemPrivate::init()
178 {
179  m_access.clear();
180  // metaInfo = KFileMetaInfo();
181 
182  // stat() local files if needed
183  // TODO: delay this until requested
184  if (m_fileMode == KFileItem::Unknown || m_permissions == KFileItem::Unknown || m_entry.count() == 0) {
185  if (m_url.isLocalFile()) {
186  /* directories may not have a slash at the end if
187  * we want to stat() them; it requires that we
188  * change into it .. which may not be allowed
189  * stat("/is/unaccessible") -> rwx------
190  * stat("/is/unaccessible/") -> EPERM H.Z.
191  * This is the reason for the StripTrailingSlash
192  */
193  QT_STATBUF buf;
194  const QString path = m_url.adjusted(QUrl::StripTrailingSlash).toLocalFile();
195  const QByteArray pathBA = QFile::encodeName(path);
196  if (QT_LSTAT(pathBA.constData(), &buf) == 0) {
197  m_entry.insert(KIO::UDSEntry::UDS_DEVICE_ID, buf.st_dev);
198  m_entry.insert(KIO::UDSEntry::UDS_INODE, buf.st_ino);
199 
200  mode_t mode = buf.st_mode;
201  if ((buf.st_mode & QT_STAT_MASK) == QT_STAT_LNK) {
202  m_bLink = true;
203  if (QT_STAT(pathBA, &buf) == 0) {
204  mode = buf.st_mode;
205  } else {// link pointing to nowhere (see FileProtocol::createUDSEntry() in ioslaves/file/file.cpp)
206  mode = (QT_STAT_MASK - 1) | S_IRWXU | S_IRWXG | S_IRWXO;
207  }
208  }
209  m_entry.insert(KIO::UDSEntry::UDS_SIZE, buf.st_size);
210  m_entry.insert(KIO::UDSEntry::UDS_FILE_TYPE, buf.st_mode & QT_STAT_MASK); // extract file type
211  m_entry.insert(KIO::UDSEntry::UDS_ACCESS, buf.st_mode & 07777); // extract permissions
212  m_entry.insert(KIO::UDSEntry::UDS_MODIFICATION_TIME, buf.st_mtime); // TODO: we could use msecs too...
213  m_entry.insert(KIO::UDSEntry::UDS_ACCESS_TIME, buf.st_atime);
214 #ifndef Q_OS_WIN
215  m_entry.insert(KIO::UDSEntry::UDS_USER, KUser(buf.st_uid).loginName());
216  m_entry.insert(KIO::UDSEntry::UDS_GROUP, KUserGroup(buf.st_gid).name());
217 #endif
218 
219  // TODO: these can be removed, we can use UDS_FILE_TYPE and UDS_ACCESS everywhere
220  if (m_fileMode == KFileItem::Unknown) {
221  m_fileMode = mode & QT_STAT_MASK; // extract file type
222  }
223  if (m_permissions == KFileItem::Unknown) {
224  m_permissions = mode & 07777; // extract permissions
225  }
226  }
227  }
228  }

230 void KFileItemPrivate::readUDSEntry(bool _urlIsDirectory)
231 {
232  // extract fields from the KIO::UDS Entry
233 
234  m_fileMode = m_entry.numberValue(KIO::UDSEntry::UDS_FILE_TYPE, KFileItem::Unknown);
235  m_permissions = m_entry.numberValue(KIO::UDSEntry::UDS_ACCESS, KFileItem::Unknown);
236  m_strName = m_entry.stringValue(KIO::UDSEntry::UDS_NAME);
237 
238  const QString displayName = m_entry.stringValue(KIO::UDSEntry::UDS_DISPLAY_NAME);
239  if (!displayName.isEmpty()) {
240  m_strText = displayName;
241  } else {
242  m_strText = KIO::decodeFileName(m_strName);
243  }
244 
245  const QString urlStr = m_entry.stringValue(KIO::UDSEntry::UDS_URL);
246  const bool UDS_URL_seen = !urlStr.isEmpty();
247  if (UDS_URL_seen) {
248  m_url = QUrl(urlStr);
249  if (m_url.isLocalFile()) {
250  m_bIsLocalUrl = true;
251  }
252  }
253  QMimeDatabase db;
254  const QString mimeTypeStr = m_entry.stringValue(KIO::UDSEntry::UDS_MIME_TYPE);
255  m_bMimeTypeKnown = !mimeTypeStr.isEmpty();
256  if (m_bMimeTypeKnown) {
257  m_mimeType = db.mimeTypeForName(mimeTypeStr);
258  }
259 
260  m_guessedMimeType = m_entry.stringValue(KIO::UDSEntry::UDS_GUESSED_MIME_TYPE);
261  m_bLink = !m_entry.stringValue(KIO::UDSEntry::UDS_LINK_DEST).isEmpty(); // we don't store the link dest
262 
263  const int hiddenVal = m_entry.numberValue(KIO::UDSEntry::UDS_HIDDEN, -1);
264  m_hidden = hiddenVal == 1 ? Hidden : (hiddenVal == 0 ? Shown : Auto);
265 
266  if (_urlIsDirectory && !UDS_URL_seen && !m_strName.isEmpty() && m_strName != QLatin1String(".")) {
267  if (!m_url.path().endsWith('/')) {
268  m_url.setPath(m_url.path() + '/');
269  }
270  m_url.setPath(m_url.path() + m_strName);
271  }
272 
273  m_iconName.clear();
274 }

367 QString KFileItemPrivate::parsePermissions(mode_t perm) const
368 {
369  static char buffer[ 12 ];
370 
371  char uxbit, gxbit, oxbit;
372 
373  if ((perm & (S_IXUSR | S_ISUID)) == (S_IXUSR | S_ISUID)) {
374  uxbit = 's';
375  } else if ((perm & (S_IXUSR | S_ISUID)) == S_ISUID) {
376  uxbit = 'S';
377  } else if ((perm & (S_IXUSR | S_ISUID)) == S_IXUSR) {
378  uxbit = 'x';
379  } else {
380  uxbit = '-';
381  }
382 
383  if ((perm & (S_IXGRP | S_ISGID)) == (S_IXGRP | S_ISGID)) {
384  gxbit = 's';
385  } else if ((perm & (S_IXGRP | S_ISGID)) == S_ISGID) {
386  gxbit = 'S';
387  } else if ((perm & (S_IXGRP | S_ISGID)) == S_IXGRP) {
388  gxbit = 'x';
389  } else {
390  gxbit = '-';
391  }
392 
393  if ((perm & (S_IXOTH | S_ISVTX)) == (S_IXOTH | S_ISVTX)) {
394  oxbit = 't';
395  } else if ((perm & (S_IXOTH | S_ISVTX)) == S_ISVTX) {
396  oxbit = 'T';
397  } else if ((perm & (S_IXOTH | S_ISVTX)) == S_IXOTH) {
398  oxbit = 'x';
399  } else {
400  oxbit = '-';
401  }
402 
403  // Include the type in the first char like ls does; people are more used to seeing it,
404  // even though it's not really part of the permissions per se.
405  if (m_bLink) {
406  buffer[0] = 'l';
407  } else if (m_fileMode != KFileItem::Unknown) {
408  if ((m_fileMode & QT_STAT_MASK) == QT_STAT_DIR) {
409  buffer[0] = 'd';
410  }
411 #ifdef Q_OS_UNIX
412  else if (S_ISSOCK(m_fileMode)) {
413  buffer[0] = 's';
414  } else if (S_ISCHR(m_fileMode)) {
415  buffer[0] = 'c';
416  } else if (S_ISBLK(m_fileMode)) {
417  buffer[0] = 'b';
418  } else if (S_ISFIFO(m_fileMode)) {
419  buffer[0] = 'p';
420  }
421 #endif // Q_OS_UNIX
422  else {
423  buffer[0] = '-';
424  }
425  } else {
426  buffer[0] = '-';
427  }
428 
429  buffer[1] = (((perm & S_IRUSR) == S_IRUSR) ? 'r' : '-');
430  buffer[2] = (((perm & S_IWUSR) == S_IWUSR) ? 'w' : '-');
431  buffer[3] = uxbit;
432  buffer[4] = (((perm & S_IRGRP) == S_IRGRP) ? 'r' : '-');
433  buffer[5] = (((perm & S_IWGRP) == S_IWGRP) ? 'w' : '-');
434  buffer[6] = gxbit;
435  buffer[7] = (((perm & S_IROTH) == S_IROTH) ? 'r' : '-');
436  buffer[8] = (((perm & S_IWOTH) == S_IWOTH) ? 'w' : '-');
437  buffer[9] = oxbit;
438  // if (hasExtendedACL())
439  if (m_entry.contains(KIO::UDSEntry::UDS_EXTENDED_ACL)) {
440  buffer[10] = '+';
441  buffer[11] = 0;
442  } else {
443  buffer[10] = 0;
444  }
445 
446  return QString::fromLatin1(buffer);
447 }

768  return false;

1482 {
1483  if (!d) {
1484  return QMimeType();
1485  }

1487  if (!d->m_mimeType.isValid()) {
1488  // On-demand fast (but not always accurate) mimetype determination
1489  Q_ASSERT(!d->m_url.isEmpty());
1490  QMimeDatabase db;
1491  if (isDir()) {
1492  d->m_mimeType = db.mimeTypeForName(QStringLiteral("inode/directory"));
