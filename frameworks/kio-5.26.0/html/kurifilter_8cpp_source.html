37 static QString lookupIconNameFor(const QUrl &url, KUriFilterData::UriTypes type)
38 {
39  QString iconName;
40 
41  switch (type) {
42  case KUriFilterData::NetProtocol:
43  iconName = KIO::iconNameForUrl(url);
44  break;
45  case KUriFilterData::Executable: {
46  QString exeName = url.path();
47  exeName = exeName.mid(exeName.lastIndexOf('/') + 1); // strip path if given
48  KService::Ptr service = KService::serviceByDesktopName(exeName);
49  if (service && service->icon() != QLatin1String("unknown")) {
50  iconName = service->icon();
51  }
52  // Try to find an icon with the same name as the binary (useful for non-kde apps)
53  // Use iconPath rather than loadIcon() as the latter uses QPixmap (not threadsafe)
54  else if (!KIconLoader::global()->iconPath(exeName, KIconLoader::NoGroup, true).isNull()) {
55  iconName = exeName;
56  } else
57  // not found, use default
58  {
59  iconName = QStringLiteral("system-run");
60  }
61  break;
62  }
63  case KUriFilterData::Help: {
64  iconName = QStringLiteral("khelpcenter");
65  break;
66  }
67  case KUriFilterData::Shell: {
68  iconName = QStringLiteral("konsole");
69  break;
70  }
71  case KUriFilterData::Error:
72  case KUriFilterData::Blocked: {
73  iconName = QStringLiteral("error");
74  break;
75  }
76  default:
77  break;
78  }
79 
80  return iconName;
81 }

585 bool KUriFilter::filterUri(KUriFilterData &data, const QStringList &filters)
586 {
587  bool filtered = false;
588 
589  // If no specific filters were requested, iterate through all the plugins.
590  // Otherwise, only use available filters.
591  if (filters.isEmpty()) {
592  QStringListIterator it(d->pluginNames);
593  while (it.hasNext()) {
594  KUriFilterPlugin *plugin = d->plugins.value(it.next());
595  if (plugin && plugin->filterUri(data)) {
596  filtered = true;
597  }
598  }
599  } else {
600  QStringListIterator it(filters);
601  while (it.hasNext()) {
602  KUriFilterPlugin *plugin = d->plugins.value(it.next());
603  if (plugin && plugin->filterUri(data)) {
604  filtered = true;
605  }
606  }
607  }
608 
609  return filtered;
610 }

673 void KUriFilter::loadPlugins()
674 {
675  const KService::List offers = KServiceTypeTrader::self()->query(QStringLiteral("KUriFilter/Plugin"));
676 
677  // NOTE: Plugin priority is determined by the InitialPreference entry in
678  // the .desktop files, so the trader result is already sorted and should
679  // not be manually sorted.
680  Q_FOREACH (const KService::Ptr &ptr, offers) {
681  KUriFilterPlugin *plugin = ptr->createInstance<KUriFilterPlugin>();
682  if (plugin) {
683  const QString &pluginName = plugin->objectName();
684  Q_ASSERT(!pluginName.isEmpty());
685  d->plugins.insert(pluginName, plugin);
686  // Needed to ensure the order of filtering is honored since
687  // items are ordered arbitrarily in a QHash and QMap always
688  // sorts by keys. Both undesired behavior.
689  d->pluginNames << pluginName;
690  }
691  }
692 }
