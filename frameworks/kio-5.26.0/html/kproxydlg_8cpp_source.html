105 static QString proxyUrlFromInput(KProxyDialog::DisplayUrlFlags* flags,
106  const QLineEdit* edit, const QSpinBox* spinBox,
107  const QString& defaultScheme = QString(),
108  KProxyDialog::DisplayUrlFlag flag = KProxyDialog::HideNone)
109 {
110  Q_ASSERT(edit);
111  Q_ASSERT(spinBox);
112 
113  QString proxyStr;
114 
115  if (edit->text().isEmpty())
116  return proxyStr;
117 
118  if (flags && !edit->text().contains(QL1S("://"))) {
119  *flags |= flag;
120  }
121 
122  KUriFilterData data;
123  data.setData(edit->text());
124  data.setCheckForExecutables(false);
125  if (!defaultScheme.isEmpty()) {
126  data.setDefaultUrlScheme(defaultScheme);
127  }
128 
129  if (KUriFilter::self()->filterUri(data, QStringList() << QStringLiteral("kshorturifilter"))) {
130  QUrl url = data.uri();
131  const int portNum = (spinBox->value() > 0 ? spinBox->value() : url.port());
132  url.setPort(-1);
133 
134  proxyStr = url.url();
135  proxyStr += QL1C(' ');
136  if (portNum > -1) {
137  proxyStr += QString::number(portNum);
138  }
139  } else {
140  proxyStr = edit->text();
141  if (spinBox->value() > 0) {
142  proxyStr += QL1C(' ');
143  proxyStr += QString::number(spinBox->value());
144  }
145  }
146 
147  return proxyStr;
148 }

150 static void setProxyInformation(const QString& value,
151  int proxyType,
152  QLineEdit* manEdit,
153  QLineEdit* sysEdit,
154  QSpinBox* spinBox ,
155  const QString& defaultScheme,
156  KProxyDialog::DisplayUrlFlag flag)
157 {
158  const bool isSysProxy = (!value.contains(QL1C(' ')) &&
159  !value.contains(QL1C('.')) &&
160  !value.contains(QL1C(',')) &&
161  !value.contains(QL1C(':')));
162 
163  if (proxyType == KProtocolManager::EnvVarProxy || isSysProxy) {
164 #if defined(Q_OS_LINUX) || defined (Q_OS_UNIX)
165  sysEdit->setText(value);
166 #endif
167  return;
168  }
169 
170  if (spinBox) {
171  QString urlStr;
172  int portNum = -1;
173  int index = value.lastIndexOf(QL1C(' '));
174  if (index == -1)
175  index = value.lastIndexOf(QL1C(':'));
176 
177  if (index > 0) {
178  bool ok = false;
179  portNum = value.midRef(index+1).toInt(&ok);
180  if (!ok) {
181  portNum = -1;
182  }
183  urlStr = value.left(index).trimmed();
184  } else {
185  urlStr = value.trimmed();
186  }
187 
188  KUriFilterData data;
189  data.setData(urlStr);
190  data.setCheckForExecutables(false);
191  if (!defaultScheme.isEmpty()) {
192  data.setDefaultUrlScheme(defaultScheme);
193  }
194 
195  if (KUriFilter::self()->filterUri(data, QStringList() << QStringLiteral("kshorturifilter"))) {
196  QUrl url (data.uri());
197  if (portNum == -1 && url.port() > -1) {
198  portNum = url.port();
199  }
200 
201  url.setPort(-1);
202  url.setUserName(QString());
203  url.setPassword(QString());
204  url.setPath(QString());
205 
206  manEdit->setText(((KSaveIOConfig::proxyDisplayUrlFlags() & flag) ? url.host(): url.url()));
207  } else {
208  QUrl url(urlStr);
209  manEdit->setText((KSaveIOConfig::proxyDisplayUrlFlags() & flag) ? url.host() : urlStr);
210  }
211 
212  if (spinBox && portNum > -1) {
213  spinBox->setValue(portNum);
214  }
215  return;
216  }
217 
218  manEdit->setText(value); // Manual proxy exception...
219 }

281 void KProxyDialog::load()
282 {
283  mProxyMap[QStringLiteral("HttpProxy")] = KProtocolManager::proxyFor(QStringLiteral("http"));
284  mProxyMap[QStringLiteral("HttpsProxy")] = KProtocolManager::proxyFor(QStringLiteral("https"));
285  mProxyMap[QStringLiteral("FtpProxy")] = KProtocolManager::proxyFor(QStringLiteral("ftp"));
286  mProxyMap[QStringLiteral("SocksProxy")] = KProtocolManager::proxyFor(QStringLiteral("socks"));
287  mProxyMap[QStringLiteral("ProxyScript")] = KProtocolManager::proxyConfigScript();
288  mProxyMap[QStringLiteral("NoProxy")] = KSaveIOConfig::noProxyFor();
289 
290  const int proxyType = KProtocolManager::proxyType();
291 
292  // Make sure showEnvValueCheckBox is unchecked before setting proxy env var names
293  mUi.showEnvValueCheckBox->setChecked(false);
294 
295  setProxyInformation(mProxyMap.value(QStringLiteral("HttpProxy")), proxyType, mUi.manualProxyHttpEdit, mUi.systemProxyHttpEdit, mUi.manualProxyHttpSpinBox, QStringLiteral("http"), HideHttpUrlScheme);
296  setProxyInformation(mProxyMap.value(QStringLiteral("HttpsProxy")), proxyType, mUi.manualProxyHttpsEdit, mUi.systemProxyHttpsEdit, mUi.manualProxyHttpsSpinBox, QStringLiteral("http"), HideHttpsUrlScheme);
297  setProxyInformation(mProxyMap.value(QStringLiteral("FtpProxy")), proxyType, mUi.manualProxyFtpEdit, mUi.systemProxyFtpEdit, mUi.manualProxyFtpSpinBox, QStringLiteral("ftp"), HideFtpUrlScheme);
298  setProxyInformation(mProxyMap.value(QStringLiteral("SocksProxy")), proxyType, mUi.manualProxySocksEdit, mUi.systemProxySocksEdit, mUi.manualProxySocksSpinBox, QStringLiteral("socks"), HideSocksUrlScheme);
299  setProxyInformation(mProxyMap.value(QStringLiteral("NoProxy")), proxyType, mUi.manualNoProxyEdit, mUi.systemNoProxyEdit, 0, QString(), HideNone);
300 
301  // Check the "Use this proxy server for all protocols" if all the proxy URLs are the same...
302  const QString httpProxy(mUi.manualProxyHttpEdit->text());
303  if (!httpProxy.isEmpty()) {
304  const int httpProxyPort = mUi.manualProxyHttpSpinBox->value();
305  mUi.useSameProxyCheckBox->setChecked(httpProxy == mUi.manualProxyHttpsEdit->text() &&
306  httpProxy == mUi.manualProxyFtpEdit->text() &&
307  httpProxy == mUi.manualProxySocksEdit->text() &&
308  httpProxyPort == mUi.manualProxyHttpsSpinBox->value() &&
309  httpProxyPort == mUi.manualProxyFtpSpinBox->value() &&
310  httpProxyPort == mUi.manualProxySocksSpinBox->value());
311  }
312 
313  // Validate and Set the automatic proxy configuration script url.
314  QUrl u (mProxyMap.value(QStringLiteral("ProxyScript")));
315  if (u.isValid() && !u.isEmpty()) {
316  u.setUserName (QString());
317  u.setPassword (QString());
318  mUi.proxyScriptUrlRequester->setUrl(u);
319  }
320 
321  // Set use reverse proxy checkbox...
322  mUi.useReverseProxyCheckBox->setChecked((!mProxyMap.value(QStringLiteral("NoProxy")).isEmpty()
323  && KProtocolManager::useReverseProxy()));
324 
325  switch (proxyType) {
326  case KProtocolManager::WPADProxy:
327  mUi.autoDiscoverProxyRadioButton->setChecked(true);
328  break;
329  case KProtocolManager::PACProxy:

332  case KProtocolManager::ManualProxy:

334  break;
335  case KProtocolManager::EnvVarProxy:
336  mUi.systemProxyRadioButton->setChecked(true);
337  break;
338  case KProtocolManager::NoProxy:
339  default:
340  mUi.noProxyRadioButton->setChecked(true);
341  break;
342  }

350 void KProxyDialog::save()
351 {
352  const KProtocolManager::ProxyType lastProxyType = KProtocolManager::proxyType();
353  KProtocolManager::ProxyType proxyType = KProtocolManager::NoProxy;
354  DisplayUrlFlags displayUrlFlags = static_cast<DisplayUrlFlags>(KSaveIOConfig::proxyDisplayUrlFlags());
355 
356  if (mUi.manualProxyRadioButton->isChecked()) {
357  DisplayUrlFlags flags = HideNone;
358  proxyType = KProtocolManager::ManualProxy;
359  mProxyMap[QStringLiteral("HttpProxy")] = proxyUrlFromInput(&flags, mUi.manualProxyHttpEdit, mUi.manualProxyHttpSpinBox, QStringLiteral("http"), HideHttpUrlScheme);
360  mProxyMap[QStringLiteral("HttpsProxy")] = proxyUrlFromInput(&flags, mUi.manualProxyHttpsEdit, mUi.manualProxyHttpsSpinBox, QStringLiteral("http"), HideHttpsUrlScheme);
361  mProxyMap[QStringLiteral("FtpProxy")] = proxyUrlFromInput(&flags, mUi.manualProxyFtpEdit, mUi.manualProxyFtpSpinBox, QStringLiteral("ftp"), HideFtpUrlScheme);
362  mProxyMap[QStringLiteral("SocksProxy")] = proxyUrlFromInput(&flags, mUi.manualProxySocksEdit, mUi.manualProxySocksSpinBox, QStringLiteral("socks"), HideSocksUrlScheme);
363  mProxyMap[QStringLiteral("NoProxy")] = mUi.manualNoProxyEdit->text();
364  displayUrlFlags = flags;
365  } else if (mUi.systemProxyRadioButton->isChecked()) {
366  proxyType = KProtocolManager::EnvVarProxy;
367  if (!mUi.showEnvValueCheckBox->isChecked()) {
368  mProxyMap[QStringLiteral("HttpProxy")] = mUi.systemProxyHttpEdit->text();
369  mProxyMap[QStringLiteral("HttpsProxy")] = mUi.systemProxyHttpsEdit->text();
370  mProxyMap[QStringLiteral("FtpProxy")] = mUi.systemProxyFtpEdit->text();
371  mProxyMap[QStringLiteral("SocksProxy")] = mUi.systemProxySocksEdit->text();
372  mProxyMap[QStringLiteral("NoProxy")] = mUi.systemNoProxyEdit->text();
373  }
374  else {
375  mProxyMap[QStringLiteral("HttpProxy")] = mProxyMap.take(mUi.systemProxyHttpEdit->objectName());
376  mProxyMap[QStringLiteral("HttpsProxy")] = mProxyMap.take(mUi.systemProxyHttpsEdit->objectName());
377  mProxyMap[QStringLiteral("FtpProxy")] = mProxyMap.take(mUi.systemProxyFtpEdit->objectName());
378  mProxyMap[QStringLiteral("SocksProxy")] = mProxyMap.take(mUi.systemProxySocksEdit->objectName());
379  mProxyMap[QStringLiteral("NoProxy")] = mProxyMap.take(mUi.systemNoProxyEdit->objectName());
380  }
381  } else if (mUi.autoScriptProxyRadioButton->isChecked()) {
382  proxyType = KProtocolManager::PACProxy;
383  mProxyMap[QStringLiteral("ProxyScript")] = mUi.proxyScriptUrlRequester->text();
384  } else if (mUi.autoDiscoverProxyRadioButton->isChecked()) {
385  proxyType = KProtocolManager::WPADProxy;
386  }
387 
388  KSaveIOConfig::setProxyType(proxyType);
389  KSaveIOConfig::setProxyDisplayUrlFlags(displayUrlFlags);
390  KSaveIOConfig::setUseReverseProxy(mUi.useReverseProxyCheckBox->isChecked());
391 
392  // Save the common proxy setting...
393  KSaveIOConfig::setProxyFor(QStringLiteral("http"), mProxyMap.value(QStringLiteral("HttpProxy")));
394  KSaveIOConfig::setProxyFor(QStringLiteral("https"), mProxyMap.value(QStringLiteral("HttpsProxy")));
395  KSaveIOConfig::setProxyFor(QStringLiteral("ftp"), mProxyMap.value(QStringLiteral("FtpProxy")));
396  KSaveIOConfig::setProxyFor(QStringLiteral("socks"), mProxyMap.value(QStringLiteral("SocksProxy")));
397 
398  KSaveIOConfig::setProxyConfigScript (mProxyMap.value(QStringLiteral("ProxyScript")));
399  KSaveIOConfig::setNoProxyFor (mProxyMap.value(QStringLiteral("NoProxy")));
400 
401  KSaveIOConfig::updateRunningIOSlaves (this);
402  if (isPACProxyType(lastProxyType) || isPACProxyType(proxyType)) {
403  KSaveIOConfig::updateProxyScout (this);
404  }
405 
406  emit changed (false);
407 }

433 bool KProxyDialog::autoDetectSystemProxy(QLineEdit* edit, const QString& envVarStr, bool showValue)
434 {
435  const QStringList envVars = envVarStr.split(QL1S(","), QString::SkipEmptyParts);
436  Q_FOREACH (const QString & envVar, envVars) {
437  const QByteArray envVarUtf8(envVar.toUtf8());
438  const QByteArray envVarValue = qgetenv(envVarUtf8.constData());
439  if (!envVarValue.isEmpty()) {
440  if (showValue) {
441  mProxyMap[edit->objectName()] = envVar;
442  edit->setText(envVarValue);
443  } else {
444  edit->setText(envVar);
445  }
446  edit->setEnabled(!showValue);
447  return true;
448  }
449  }
450  return false;
451 }
