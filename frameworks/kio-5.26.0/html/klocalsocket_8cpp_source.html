66 void KLocalSocket::connectToHostImplementation(const QString &path, quint16 type, OpenMode mode)
67 {
68  if (state() == ConnectedState || state() == ConnectingState) {
69  return;
70  }
71 
72  d->localPath.clear();
73  d->peerPath.clear();
74 
75  setSocketState(ConnectingState);
76  emit stateChanged(ConnectingState);
77 
78  d->connectToPath(path, LocalSocketType(type), mode);
79 }

136 bool KLocalSocketServer::listen(const QString &path, KLocalSocket::LocalSocketType type)
137 {

139  return false; // already created

142  if (!d->listen(path, type)) {

147  d->localPath = path;

195 KLocalSocket *KLocalSocketServer::nextPendingConnection()
196 {
197  if (hasPendingConnections()) {
198  d->readNotifier->setEnabled((d->pendingConnections.size() - 1) < d->maxPendingConnections);
