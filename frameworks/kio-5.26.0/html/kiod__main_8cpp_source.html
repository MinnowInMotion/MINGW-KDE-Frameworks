46 void KIOD::loadModule(const QString &name)
47 {
48  // Make sure this method is only called with valid module names.
49  Q_ASSERT(name.indexOf('/') == -1);
50 
51  KDEDModule *module = m_modules.value(name, 0);
52  if (module) {
53  return;
54  }
55 
56  KPluginLoader loader("kf5/kiod/" + name);
57  KPluginFactory *factory = loader.factory();
58  if (factory) {
59  module = factory->create<KDEDModule>();
60  }
61  if (!module) {
62  qCWarning(KIOD_CATEGORY) << "Error loading plugin:" << loader.errorString();
63  return;
64  }
65  module->setModuleName(name); // makes it register to DBus
66  m_modules.insert(name, module);
67 }

71 // on-demand module loading
72 // this function is called by the D-Bus message processing function before
73 // calls are delivered to objects
74 static void messageFilter(const QDBusMessage &message)
75 {
76  const QString name = KDEDModule::moduleForMessage(message);
77  if (name.isEmpty()) {
78  return;
79  }
80 
81  qCDebug(KIOD_CATEGORY) << "loadModule" << name;
82  self()->loadModule(name);
83 }

87 int main(int argc, char *argv[])
88 {
89  qunsetenv("SESSION_MANAGER"); // disable session management
90 
91  QApplication app(argc, argv); // GUI needed for kpasswdserver's dialogs
92  app.setApplicationName(QStringLiteral("kiod5"));
93  app.setOrganizationDomain(QStringLiteral("kde.org"));
94  app.setQuitOnLastWindowClosed(false);
95  KDBusService service(KDBusService::Unique);
96 
97  QDBusConnectionInterface *bus = QDBusConnection::sessionBus().interface();
98  // Also register as all the names we should respond to (org.kde.kssld, org.kde.kcookiejar, etc.)
99  // so that the calling code is independent from the physical "location" of the service.
100  const QVector<KPluginMetaData> plugins = KPluginLoader::findPlugins(QStringLiteral("kf5/kiod"));
101  foreach (const KPluginMetaData &metaData, plugins) {
102  const QString serviceName = metaData.rawData().value(QStringLiteral("X-KDE-DBus-ServiceName")).toString();
103  if (serviceName.isEmpty()) {
104  qCWarning(KIOD_CATEGORY) << "No X-KDE-DBus-ServiceName found in" << metaData.fileName();
105  continue;
106  }
107  if (!bus->registerService(serviceName)) {
108  qCWarning(KIOD_CATEGORY) << "Couldn't register name" << serviceName << "with DBUS - another process owns it already!";
109  }
110  }
111 
112  self(); // create it in this thread
113  qDBusAddSpyHook(messageFilter);
114 
115  return app.exec();
116 }
