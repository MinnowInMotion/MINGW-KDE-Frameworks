72 /*
73  Domain suffix match. E.g. return true if host is "cuzco.inka.de" and
74  nplist is "inka.de,hadiko.de" or if host is "localhost" and nplist is
75  "localhost".
76 */
77 static bool revmatch(const char *host, const char *nplist)
78 {
79  if (host == 0) {
80  return false;
81  }
82 
83  const char *hptr = host + strlen(host) - 1;
84  const char *nptr = nplist + strlen(nplist) - 1;
85  const char *shptr = hptr;
86 
87  while (nptr >= nplist) {
88  if (*hptr != *nptr) {
89  hptr = shptr;
90 
91  // Try to find another domain or host in the list
92  while (--nptr >= nplist && *nptr != ',' && *nptr != ' ');
93 
94  // Strip out multiple spaces and commas
95  while (--nptr >= nplist && (*nptr == ',' || *nptr == ' '));
96  } else {
97  if (nptr == nplist || nptr[-1] == ',' || nptr[-1] == ' ') {
98  return true;
99  }
100  if (nptr[-1] == '/' && hptr == host) { // "bugs.kde.org" vs "http://bugs.kde.org", the config UI says URLs are ok
101  return true;
102  }
103  if (hptr == host) { // e.g. revmatch("bugs.kde.org","mybugs.kde.org")
104  return false;
105  }
106 
107  hptr--;
108  nptr--;
109  }
110  }
111 
112  return false;
113 }

178 /*
179  * Returns true if url is in the no proxy list.
180  */
181 bool KProtocolManagerPrivate::shouldIgnoreProxyFor(const QUrl &url)
182 {
183  bool isMatch = false;
184  const KProtocolManager::ProxyType type = proxyType();
185  const bool useRevProxy = ((type == KProtocolManager::ManualProxy) && useReverseProxy());
186  const bool useNoProxyList = (type == KProtocolManager::ManualProxy || type == KProtocolManager::EnvVarProxy);
187 
188  // No proxy only applies to ManualProxy and EnvVarProxy types...
189  if (useNoProxyList && noProxyFor.isEmpty()) {
190  QStringList noProxyForList(readNoProxyFor().split(QL1C(',')));
191  QMutableStringListIterator it(noProxyForList);
192  while (it.hasNext()) {
193  SubnetPair subnet = QHostAddress::parseSubnet(it.next());
194  if (!subnet.first.isNull()) {
195  noProxySubnets << subnet;
196  it.remove();
197  }
198  }
199  noProxyFor = noProxyForList.join(QLatin1Char(','));
200  }
201 
202  if (!noProxyFor.isEmpty()) {
203  QString qhost = url.host().toLower();
204  QByteArray host = qhost.toLatin1();
205  const QString qno_proxy = noProxyFor.trimmed().toLower();
206  const QByteArray no_proxy = qno_proxy.toLatin1();
207  isMatch = revmatch(host, no_proxy);
208 
209  // If no match is found and the request url has a port
210  // number, try the combination of "host:port". This allows
211  // users to enter host:port in the No-proxy-For list.
212  if (!isMatch && url.port() > 0) {
213  qhost += QL1C(':');
214  qhost += QString::number(url.port());
215  host = qhost.toLatin1();
216  isMatch = revmatch(host, no_proxy);
217  }
218 
219  // If the hostname does not contain a dot, check if
220  // <local> is part of noProxy.
221  if (!isMatch && !host.isEmpty() && (strchr(host, '.') == NULL)) {
222  isMatch = revmatch("<local>", no_proxy);
223  }
224  }
225 
226  const QString host(url.host());
227 
228  if (!noProxySubnets.isEmpty() && !host.isEmpty()) {
229  QHostAddress address(host);
230  // If request url is not IP address, do a DNS lookup of the hostname.
231  // TODO: Perhaps we should make configurable ?
232  if (address.isNull()) {
233  //qDebug() << "Performing DNS lookup for" << host;
234  QHostInfo info = KIO::HostInfo::lookupHost(host, 2000);
235  const QList<QHostAddress> addresses = info.addresses();
236  if (!addresses.isEmpty()) {
237  address = addresses.first();
238  }
239  }
240 
241  if (!address.isNull()) {
242  Q_FOREACH (const SubnetPair &subnet, noProxySubnets) {
243  if (address.isInSubnet(subnet)) {
244  isMatch = true;
245  break;
246  }
247  }
248  }
249  }
250 
251  return (useRevProxy != isMatch);
252 }

502 QStringList KProtocolManagerPrivate::getSystemProxyFor(const QUrl &url)
503 {
504  QStringList proxies;
505 
506 #if !defined(QT_NO_NETWORKPROXY) && (defined(Q_OS_WIN32) || defined(Q_OS_MAC))
507  QNetworkProxyQuery query(url);
508  const QList<QNetworkProxy> proxyList = QNetworkProxyFactory::systemProxyForQuery(query);
509  Q_FOREACH (const QNetworkProxy &proxy, proxyList) {
510  QUrl url;
511  const QNetworkProxy::ProxyType type = proxy.type();
512  if (type == QNetworkProxy::NoProxy || type == QNetworkProxy::DefaultProxy) {
513  proxies << QL1S("DIRECT");
514  continue;
515  }
516 
517  if (type == QNetworkProxy::HttpProxy || type == QNetworkProxy::HttpCachingProxy) {
518  url.setScheme(QL1S("http"));
519  } else if (type == QNetworkProxy::Socks5Proxy) {
520  url.setScheme(QL1S("socks"));
521  } else if (type == QNetworkProxy::FtpCachingProxy) {
522  url.setScheme(QL1S("ftp"));
523  }
524 
525  url.setHost(proxy.hostName());
526  url.setPort(proxy.port());
527  url.setUserName(proxy.user());
528  proxies << url.url();
529  }
530 #else
531  // On Unix/Linux use system environment variables if any are set.
532  QString proxyVar(proxyFor(url.scheme()));
533  // Check for SOCKS proxy, if not proxy is found for given url.
534  if (!proxyVar.isEmpty()) {
535  const QString proxy(QString::fromLocal8Bit(qgetenv(proxyVar.toLocal8Bit())).trimmed());
536  if (!proxy.isEmpty()) {
537  proxies << proxy;
538  }
539  }
540  // Add the socks proxy as an alternate proxy if it exists,
541  proxyVar = proxyFor(QStringLiteral("socks"));
542  if (!proxyVar.isEmpty()) {
543  QString proxy = QString::fromLocal8Bit(qgetenv(proxyVar.toLocal8Bit())).trimmed();
544  // Make sure the scheme of SOCKS proxy is always set to "socks://".
545  const int index = proxy.indexOf(QL1S("://"));
546  proxy = QL1S("socks://") + (index == -1 ? proxy : proxy.mid(index + 3));
547  if (!proxy.isEmpty()) {
548  proxies << proxy;
549  }
550  }
551 #endif
552  return proxies;
553 }

555 QStringList KProtocolManager::proxiesForUrl(const QUrl &url)
556 {
557  QStringList proxyList;
558 
559  PRIVATE_DATA;
560  QMutexLocker lock(&d->mutex);
561  if (!d->shouldIgnoreProxyFor(url)) {
562  switch (d->proxyType()) {
563  case PACProxy:
564  case WPADProxy: {
565  QUrl u(url);
566  const QString protocol = adjustProtocol(u.scheme());
567  u.setScheme(protocol);
568 
569  if (protocol.startsWith(QLatin1String("http")) || protocol.startsWith(QLatin1String("ftp"))) {
570  QDBusReply<QStringList> reply = QDBusInterface(QStringLiteral("org.kde.kded5"),
571  QStringLiteral("/modules/proxyscout"),
572  QStringLiteral("org.kde.KPAC.ProxyScout"))
573  .call(QStringLiteral("proxiesForUrl"), u.toString());
574  proxyList = reply;
575  }
576  break;
577  }
578  case EnvVarProxy:
579  proxyList = d->getSystemProxyFor(url);
580  break;
581  case ManualProxy: {
582  QString proxy(d->proxyFor(url.scheme()));
583  if (!proxy.isEmpty()) {
584  proxyList << proxy;
585  }
586  // Add the socks proxy as an alternate proxy if it exists,
587  proxy = d->proxyFor(QStringLiteral("socks"));
588  if (!proxy.isEmpty()) {
589  // Make sure the scheme of SOCKS proxy is always set to "socks://".
590  const int index = proxy.indexOf(QL1S("://"));
591  proxy = QStringLiteral("socks://") + (index == -1 ? proxy : proxy.mid(index + 3));
592  proxyList << proxy;
593  }
594  }
595  break;
596  case NoProxy:
597  break;
598  }
599  }
600 
601  if (proxyList.isEmpty()) {
602  proxyList << QStringLiteral("DIRECT");
603  }
604 
605  return proxyList;
606 }

646 QString KProtocolManager::slaveProtocol(const QUrl &url, QStringList &proxyList)
647 {
648 #if 0
649  if (url.hasSubUrl()) { // We don't want the suburl's protocol
650  const QUrl::List list = QUrl::split(url);
651  return slaveProtocol(list.last(), proxyList);
652  }
653 #endif
654 
655  proxyList.clear();
656 
657  // Do not perform a proxy lookup for any url classified as a ":local" url or
658  // one that does not have a host component or if proxy is disabled.
659  QString protocol(url.scheme());
660  if (url.host().isEmpty()
661  || KProtocolInfo::protocolClass(protocol) == QL1S(":local")
662  || KProtocolManager::proxyType() == KProtocolManager::NoProxy) {
663  return protocol;
664  }
665 
666  QString proxyCacheKey;
667  extractProxyCacheKeyFromUrl(url, &proxyCacheKey);
668 
669  PRIVATE_DATA;
670  QMutexLocker lock(&d->mutex);
671  // Look for cached proxy information to avoid more work.
672  if (d->cachedProxyData.contains(proxyCacheKey)) {
673  KProxyData *data = d->cachedProxyData.object(proxyCacheKey);
674  proxyList = data->proxyList;
675  return data->protocol;
676  }
677  lock.unlock();
678 
679  const QStringList proxies = proxiesForUrl(url);
680  const int count = proxies.count();
681 
682  if (count > 0 && !(count == 1 && proxies.first() == QL1S("DIRECT"))) {
683  Q_FOREACH (const QString &proxy, proxies) {
684  if (proxy == QL1S("DIRECT")) {
685  proxyList << proxy;
686  } else {
687  QUrl u(proxy);

706  }
707  }
708  }

775  modifiers = QStringLiteral(DEFAULT_USER_AGENT_KEYS);
776  }

779  return d->useragent;
780  }

782  d->modifiers = modifiers;
783 
784  /*
785  The following code attempts to determine the default user agent string
786  from the 'X-KDE-UA-DEFAULT-STRING' property of the desktop file
787  for the preferred service that was configured to handle the 'text/html'
788  mime type. If the preferred service's desktop file does not specify this
789  property, the long standing default user agent string will be used.
790  The following keyword placeholders are automatically converted when the
791  user agent string is read from the property:
792 
793  %SECURITY% Expands to"N" when SSL is not supported, otherwise it is ignored.
794  %OSNAME% Expands to operating system name, e.g. Linux.
795  %OSVERSION% Expands to operating system version, e.g. 2.6.32
796  %SYSTYPE% Expands to machine or system type, e.g. i386
797  %PLATFORM% Expands to windowing system, e.g. X11 on Unix/Linux.
798  %LANGUAGE% Expands to default language in use, e.g. en-US.
799  %APPVERSION% Expands to QCoreApplication applicationName()/applicationVerison(),
800  e.g. Konqueror/4.5.0. If application name and/or application version
801  number are not set, then "KDE" and the runtime KDE version numbers
802  are used respectively.
803 
804  All of the keywords are handled case-insensitively.
805  */
806 
807  QString systemName, systemVersion, machine, supp;
808  const bool sysInfoFound = getSystemNameVersionAndMachine(systemName, systemVersion, machine);
809  QString agentStr = defaultUserAgentFromPreferredService();
810 
811  if (agentStr.isEmpty()) {
812  supp += platform();
813 
814  if (sysInfoFound) {
815  if (modifiers.contains('o')) {
816  supp += QL1S("; ");
817  supp += systemName;
818  if (modifiers.contains('v')) {
819  supp += QL1C(' ');
820  supp += systemVersion;
821  }
822 
823  if (modifiers.contains('m')) {
824  supp += QL1C(' ');
825  supp += machine;
826  }
827  }
828 
829  if (modifiers.contains('l')) {
830  supp += QL1S("; ");
831  supp += QLocale::languageToString(QLocale().language());
832  }
833  }
834 
835  // Full format: Mozilla/5.0 (Linux
836  d->useragent = QStringLiteral("Mozilla/5.0 (");
837  d->useragent += supp;
838  d->useragent += QStringLiteral(") KHTML/");
839  d->useragent += QString::number(KIO_VERSION_MAJOR);
840  d->useragent += QL1C('.');
841  d->useragent += QString::number(KIO_VERSION_MINOR);
842  d->useragent += QL1C('.');
843  d->useragent += QString::number(KIO_VERSION_PATCH);
844  d->useragent += QStringLiteral(" (like Gecko) Konqueror/");
845  d->useragent += QString::number(KIO_VERSION_MAJOR);
846  d->useragent += QL1C('.');
847  d->useragent += QString::number(KIO_VERSION_MINOR);
848  } else {
849  QString appName = QCoreApplication::applicationName();
850  if (appName.isEmpty() || appName.startsWith(QLatin1String("kcmshell"), Qt::CaseInsensitive)) {
851  appName = QStringLiteral("KDE");
852  }
853 
854  QString appVersion = QCoreApplication::applicationVersion();
855  if (appVersion.isEmpty()) {
856  appVersion += QString::number(KIO_VERSION_MAJOR);
857  appVersion += QL1C('.');
858  appVersion += QString::number(KIO_VERSION_MINOR);
859  appVersion += QL1C('.');
860  appVersion += QString::number(KIO_VERSION_PATCH);
861  }
862 
863  appName += QL1C('/');
864  appName += appVersion;
865 
866  agentStr.replace(QL1S("%appversion%"), appName, Qt::CaseInsensitive);
867 
868  if (!QSslSocket::supportsSsl()) {
869  agentStr.replace(QLatin1String("%security%"), QL1S("N"), Qt::CaseInsensitive);
870  } else {
871  agentStr.remove(QStringLiteral("%security%"), Qt::CaseInsensitive);
872  }
873 
874  if (sysInfoFound) {
875  // Platform (e.g. X11). It is no longer configurable from UI.
876  agentStr.replace(QL1S("%platform%"), platform(), Qt::CaseInsensitive);
877 
878  // Operating system (e.g. Linux)
879  if (modifiers.contains('o')) {
880  agentStr.replace(QL1S("%osname%"), systemName, Qt::CaseInsensitive);
881 
882  // OS version (e.g. 2.6.36)
883  if (modifiers.contains('v')) {
884  agentStr.replace(QL1S("%osversion%"), systemVersion, Qt::CaseInsensitive);
885  } else {
886  agentStr.remove(QStringLiteral("%osversion%"), Qt::CaseInsensitive);
887  }
888 
889  // Machine type (i686, x86-64, etc.)
890  if (modifiers.contains('m')) {
891  agentStr.replace(QL1S("%systype%"), machine, Qt::CaseInsensitive);
892  } else {
893  agentStr.remove(QStringLiteral("%systype%"), Qt::CaseInsensitive);
894  }
895  } else {
896  agentStr.remove(QStringLiteral("%osname%"), Qt::CaseInsensitive);
897  agentStr.remove(QStringLiteral("%osversion%"), Qt::CaseInsensitive);
898  agentStr.remove(QStringLiteral("%systype%"), Qt::CaseInsensitive);
899  }
900 
901  // Language (e.g. en_US)
902  if (modifiers.contains('l')) {
903  agentStr.replace(QL1S("%language%"), QLocale::languageToString(QLocale().language()), Qt::CaseInsensitive);
904  } else {
905  agentStr.remove(QStringLiteral("%language%"), Qt::CaseInsensitive);
906  }
907 
908  // Clean up unnecessary separators that could be left over from the
909  // possible keyword removal above...
910  agentStr.replace(QRegExp(QL1S("[(]\\s*[;]\\s*")), QStringLiteral("("));
911  agentStr.replace(QRegExp(QL1S("[;]\\s*[;]\\s*")), QStringLiteral("; "));
912  agentStr.replace(QRegExp(QL1S("\\s*[;]\\s*[)]")), QStringLiteral(")"));
913  } else {
914  agentStr.remove(QStringLiteral("%osname%"));
915  agentStr.remove(QStringLiteral("%osversion%"));
916  agentStr.remove(QStringLiteral("%platform%"));
917  agentStr.remove(QStringLiteral("%systype%"));
918  agentStr.remove(QStringLiteral("%language%"));
919  }
920 
921  d->useragent = agentStr.simplified();
922  }
923

925  return d->useragent;

991 QString KProtocolManager::acceptLanguagesHeader()
992 {
993  const QLatin1String english("en");
994 
995  // User's desktop language preference.
996  QStringList languageList = QLocale().uiLanguages();
997 
998  // Replace possible "C" in the language list with "en", unless "en" is
999  // already pressent. This is to keep user's priorities in order.
1000  // If afterwards "en" is still not present, append it.
1001  int idx = languageList.indexOf(QStringLiteral("C"));
1002  if (idx != -1) {
1003  if (languageList.contains(english)) {
1004  languageList.removeAt(idx);
1005  } else {
1006  languageList[idx] = english;
1007  }
1008  }
1009  if (!languageList.contains(english)) {
1010  languageList += english;
1011  }
1012 
1013  // Some languages may have web codes different from locale codes,
1014  // read them from the config and insert in proper order.
1015  KConfig acclangConf(QStringLiteral("accept-languages.codes"), KConfig::NoGlobals);
1016  KConfigGroup replacementCodes(&acclangConf, "ReplacementCodes");
1017  QStringList languageListFinal;
1018  Q_FOREACH (const QString &lang, languageList) {
1019  const QStringList langs = replacementCodes.readEntry(lang, QStringList());
1020  if (langs.isEmpty()) {
1021  languageListFinal += lang;
1022  } else {
1023  languageListFinal += langs;
1024  }
1025  }
1026 
1027  // The header is composed of comma separated languages, with an optional
1028  // associated priority estimate (q=1..0) defaulting to 1.
1029  // As our language tags are already sorted by priority, we'll just decrease
1030  // the value evenly
1031  int prio = 10;
1032  QString header;
1033  Q_FOREACH (const QString &lang, languageListFinal) {
1034  header += lang;
1035  if (prio < 10) {
1036  header += QL1S(";q=0.");
1037  header += QString::number(prio);
1038  }
1039  // do not add cosmetic whitespace in here : it is less compatible (#220677)
1040  header += QL1S(",");
1041  if (prio > 1) {
1042  --prio;
1043  }
1044  }
1045  header.chop(1);
1046 
1047  // Some of the languages may have country specifier delimited by
1048  // underscore, or modifier delimited by at-sign.
1049  // The header should use dashes instead.
1050  header.replace('_', '-');
1051  header.replace('@', '-');
1052 
1053  return header;
1054 }
