76 QString KUrlComboBoxPrivate::textForItem(const KUrlComboItem *item) const
77 {
78  if (!item->text.isEmpty()) {
79  return item->text;
80  }
81  QString text;
82  QUrl url = item->url;
83 
84  if (myMode == KUrlComboBox::Directories) {
85  if (!url.path().isEmpty() && !url.path().endsWith(QLatin1Char('/'))) {
86  url.setPath(url.path() + QLatin1Char('/'));
87  }
88  } else {
89  url = url.adjusted(QUrl::StripTrailingSlash);
90  }
91  if (url.isLocalFile()) {
92  return url.toLocalFile();
93  } else {
94  return url.toDisplayString();
95  }
96 }

179 void KUrlComboBox::setUrls(const QStringList &_urls, OverLoadResolving remove)
180 {
181  setDefaults();
182  qDeleteAll(d->itemList);
183  d->itemList.clear();
184  d->urlAdded = false;
185 
186  if (_urls.isEmpty()) {
187  return;
188  }
189 
190  QStringList urls;
191  QStringList::ConstIterator it = _urls.constBegin();
192 
193  // kill duplicates
194  while (it != _urls.constEnd()) {
195  if (!urls.contains(*it)) {
196  urls += *it;
197  }
198  ++it;
199  }
200 
201  // limit to myMaximum items
202  /* Note: overload is an (old) C++ keyword, some compilers (KCC) choke
203  on that, so call it Overload (capital 'O'). (matz) */
204  int Overload = urls.count() - d->myMaximum + d->defaultList.count();
205  while (Overload > 0) {
206  if (remove == RemoveBottom) {
207  if (!urls.isEmpty()) {
208  urls.removeLast();
209  }
210  } else {
211  if (!urls.isEmpty()) {
212  urls.removeFirst();
213  }
214  }
215  Overload--;
216  }
217 
218  it = urls.constBegin();
219 
220  KUrlComboBoxPrivate::KUrlComboItem *item = 0L;
221 
222  while (it != urls.constEnd()) {
223  if ((*it).isEmpty()) {
224  ++it;
225  continue;
226  }
227  QUrl u;
228  if (QDir::isAbsolutePath(*it)) {
229  u = QUrl::fromLocalFile(*it);
230  } else {
231  u.setUrl(*it);
232  }
233 
234  // Don't restore if file doesn't exist anymore
235  if (u.isLocalFile() && !QFile::exists(u.toLocalFile())) {
236  ++it;
237  continue;
238  }
239 
240  item = new KUrlComboBoxPrivate::KUrlComboItem(u, d->getIcon(u));
241  d->insertUrlItem(item);
242  d->itemList.append(item);
243  ++it;
244  }
245 }

247 void KUrlComboBox::setUrl(const QUrl &url)
248 {
249  if (url.isEmpty()) {
250  return;
251  }
252 
253  bool blocked = blockSignals(true);
254 
255  // check for duplicates
256  QMap<int, const KUrlComboBoxPrivate::KUrlComboItem *>::ConstIterator mit = d->itemMapper.constBegin();
257  QString urlToInsert = url.toString(QUrl::StripTrailingSlash);
258  while (mit != d->itemMapper.constEnd()) {
259  Q_ASSERT(mit.value());
260 
261  if (urlToInsert == mit.value()->url.toString(QUrl::StripTrailingSlash)) {
262  setCurrentIndex(mit.key());
263 
264  if (d->myMode == Directories) {
265  d->updateItem(mit.value(), mit.key(), d->opendirIcon);
266  }
267 
268  blockSignals(blocked);
269  return;
270  }
271  ++mit;
272  }
273 
274  // not in the combo yet -> create a new item and insert it
275 
276  // first remove the old item
277  if (d->urlAdded) {
278  Q_ASSERT(!d->itemList.isEmpty());
279  d->itemList.removeLast();
280  d->urlAdded = false;
281  }
282 
283  setDefaults();
284 
285  int offset = qMax(0, d->itemList.count() - d->myMaximum + d->defaultList.count());
286  for (int i = offset; i < d->itemList.count(); i++) {
287  d->insertUrlItem(d->itemList[i]);
288  }
289 
290  KUrlComboBoxPrivate::KUrlComboItem *item = new KUrlComboBoxPrivate::KUrlComboItem(url, d->getIcon(url));
291 
292  const int id = count();
293  const QString text = d->textForItem(item);
294  if (d->myMode == Directories) {
295  KComboBox::insertItem(id, d->opendirIcon, text);
296  } else {
297  KComboBox::insertItem(id, item->icon, text);
298  }
299 
300  d->itemMapper.insert(id, item);
301  d->itemList.append(item);
302 
303  setCurrentIndex(id);
304  Q_ASSERT(!d->itemList.isEmpty());
305  d->urlAdded = true;
306  blockSignals(blocked);
307 }

329 void KUrlComboBox::setMaxItems(int max)
330 {
331  d->myMaximum = max;
332 
333  if (count() > d->myMaximum) {
334  int oldCurrent = currentIndex();

336  setDefaults();

339  for (int i = offset; i < d->itemList.count(); i++) {
340  d->insertUrlItem(d->itemList[i]);
341  }
342 
343  if (count() > 0) { // restore the previous currentItem

357 void KUrlComboBox::removeUrl(const QUrl &url, bool checkDefaultURLs)
358 {
359  QMap<int, const KUrlComboBoxPrivate::KUrlComboItem *>::ConstIterator mit = d->itemMapper.constBegin();
360  while (mit != d->itemMapper.constEnd()) {
361  if (url.toString(QUrl::StripTrailingSlash) == mit.value()->url.toString(QUrl::StripTrailingSlash)) {
362  if (!d->itemList.removeAll(mit.value()) && checkDefaultURLs) {
363  d->defaultList.removeAll(mit.value());
364  }
365  }
366  ++mit;
367  }
368 
369  bool blocked = blockSignals(true);
370  setDefaults();
371  QListIterator<const KUrlComboBoxPrivate::KUrlComboItem *> it(d->itemList);
372  while (it.hasNext()) {
373  d->insertUrlItem(it.next());
374  }
375  blockSignals(blocked);
376 }

378 void KUrlComboBox::setCompletionObject(KCompletion *compObj, bool hsig)
379 {
380  if (compObj) {
381  // on a url combo box we want completion matches to be sorted. This way, if we are given
382  // a suggestion, we match the "best" one. For instance, if we have "foo" and "foobar",
383  // and we write "foo", the match is "foo" and never "foobar". (ereslibre)
384  compObj->setOrder(KCompletion::Sorted);
385  }
386  KComboBox::setCompletionObject(compObj, hsig);
387 }

406 void KUrlComboBox::mouseMoveEvent(QMouseEvent *event)
407 {
408  const int index = currentIndex();
409  const KUrlComboBoxPrivate::KUrlComboItem *item = d->itemMapper.value(index);
410 
411  if (item && !d->m_dragPoint.isNull() && event->buttons() & Qt::LeftButton &&
412  (event->pos() - d->m_dragPoint).manhattanLength() > QApplication::startDragDistance()) {
413  QDrag *drag = new QDrag(this);
414  QMimeData *mime = new QMimeData();
415  mime->setUrls(QList<QUrl>() << item->url);
416  mime->setText(itemText(index));
417  if (!itemIcon(index).isNull()) {
418  drag->setPixmap(itemIcon(index).pixmap(KIconLoader::SizeMedium));
419  }
420  drag->setMimeData(mime);
421  drag->exec();
422  }
423 
424  KComboBox::mouseMoveEvent(event);
