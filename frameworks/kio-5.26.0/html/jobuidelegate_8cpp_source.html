59 /*
60  Returns the top most window associated with widget.
61 
62  Unlike QWidget::window(), this function does its best to find and return the
63  main application window associated with the given widget.
64 
65  If widget itself is a dialog or its parent is a dialog, and that dialog has a
66  parent widget then this function will iterate through all those widgets to
67  find the top most window, which most of the time is the main window of the
68  application. By contrast, QWidget::window() would simply return the first
69  file dialog it encountered since it is the "next ancestor widget that has (or
70  could have) a window-system frame".
71 */
72 static QWidget *topLevelWindow(QWidget *widget)
73 {
74  QWidget *w = widget;
75  while (w && w->parentWidget()) {
76  w = w->parentWidget();
77  }
78  return (w ? w->window() : 0);
79 }

81 class JobUiDelegateStatic : public QObject
82 {
83  Q_OBJECT
84 public:
85  void registerWindow(QWidget *wid)
86  {
87  if (!wid) {
88  return;
89  }
90 
91  QWidget *window = topLevelWindow(wid);
92  QObject *obj = static_cast<QObject *>(window);
93  if (!m_windowList.contains(obj)) {
94  // We must store the window Id because by the time
95  // the destroyed signal is emitted we can no longer
96  // access QWidget::winId() (already destructed)
97  WId windowId = window->winId();
98  m_windowList.insert(obj, windowId);
99  connect(window, SIGNAL(destroyed(QObject*)),
100  this, SLOT(slotUnregisterWindow(QObject*)));
101  QDBusInterface(QStringLiteral("org.kde.kded5"), QStringLiteral("/kded"), QStringLiteral("org.kde.kded5")).
102  call(QDBus::NoBlock, QStringLiteral("registerWindowId"), qlonglong(windowId));
103  }
104  }
105 public Q_SLOTS:
106  void slotUnregisterWindow(QObject *obj)
107  {
108  if (!obj) {
109  return;
110  }
111 
112  QMap<QObject *, WId>::Iterator it = m_windowList.find(obj);
113  if (it == m_windowList.end()) {
114  return;
115  }
116  WId windowId = it.value();
117  disconnect(it.key(), SIGNAL(destroyed(QObject*)),
118  this, SLOT(slotUnregisterWindow(QObject*)));
119  m_windowList.erase(it);
120  QDBusInterface(QStringLiteral("org.kde.kded5"), QStringLiteral("/kded"), QStringLiteral("org.kde.kded5")).
121  call(QDBus::NoBlock, QStringLiteral("unregisterWindowId"), qlonglong(windowId));
122  }
123 private:
124  QMap<QObject *, WId> m_windowList;
125 };

140 KIO::RenameDialog_Result KIO::JobUiDelegate::askFileRename(KJob *job,
141  const QString &caption,
142  const QUrl &src,
143  const QUrl &dest,
144  KIO::RenameDialog_Options options,
145  QString &newDest,
146  KIO::filesize_t sizeSrc,
147  KIO::filesize_t sizeDest,
148  const QDateTime &ctimeSrc,
149  const QDateTime &ctimeDest,
150  const QDateTime &mtimeSrc,
151  const QDateTime &mtimeDest)
152 {
153  //qDebug() << "job=" << job;
154  // We now do it in process, so that opening the rename dialog
155  // doesn't start uiserver for nothing if progressId=0 (e.g. F2 in konq)
156  KIO::RenameDialog dlg(KJobWidgets::window(job), caption, src, dest, options,
157  sizeSrc, sizeDest,
158  ctimeSrc, ctimeDest, mtimeSrc,
159  mtimeDest);
160  dlg.setWindowModality(Qt::WindowModal);
161  connect(job, SIGNAL(finished(KJob*)), &dlg, SLOT(reject())); // #192976
162  KIO::RenameDialog_Result res = static_cast<RenameDialog_Result>(dlg.exec());
163  if (res == R_AUTO_RENAME) {
164  newDest = dlg.autoDestUrl().path();
165  } else {
166  newDest = dlg.newDestUrl().path();
167  }
168  return res;
169 }

181 bool KIO::JobUiDelegate::askDeleteConfirmation(const QList<QUrl> &urls,
182  DeletionType deletionType,
183  ConfirmationType confirmationType)
184 {
185  QString keyName;
186  bool ask = (confirmationType == ForceConfirmation);
187  if (!ask) {
188  KSharedConfigPtr kioConfig = KSharedConfig::openConfig(QStringLiteral("kiorc"), KConfig::NoGlobals);
189 
190  switch (deletionType) {
191  case Delete:
192  keyName = QStringLiteral("ConfirmDelete");
193  break;
194  case Trash:
195  keyName = QStringLiteral("ConfirmTrash");
196  break;
197  case EmptyTrash:
198  keyName = QStringLiteral("ConfirmEmptyTrash");
199  break;
200  }
201 
202  // The default value for confirmations is true (for both delete and trash)
203  // If you change this, update kdebase/apps/konqueror/settings/konq/behaviour.cpp
204  const bool defaultValue = true;
205  ask = kioConfig->group("Confirmations").readEntry(keyName, defaultValue);
206  }
207  if (ask) {
208  QStringList prettyList;
209  Q_FOREACH (const QUrl &url, urls) {
210  if (url.scheme() == QLatin1String("trash")) {
211  QString path = url.path();
212  // HACK (#98983): remove "0-foo". Note that it works better than
213  // displaying KFileItem::name(), for files under a subdir.
214  path.remove(QRegExp(QStringLiteral("^/[0-9]*-")));
215  prettyList.append(path);
216  } else {
217  prettyList.append(url.toDisplayString(QUrl::PreferLocalFile));
218  }
219  }
220 
221  int result;
222  QWidget *widget = window();
223  const KMessageBox::Options options(KMessageBox::Notify | KMessageBox::WindowModal);
224  switch (deletionType) {
225  case Delete:
226  result = KMessageBox::warningContinueCancelList(
227  widget,
228  i18np("Do you really want to delete this item?", "Do you really want to delete these %1 items?", prettyList.count()),
229  prettyList,
230  i18n("Delete Files"),
231  KStandardGuiItem::del(),
232  KStandardGuiItem::cancel(),
233  keyName, options);
234  break;
235  case EmptyTrash:
236  result = KMessageBox::warningContinueCancel(
237  widget,
238  i18nc("@info", "Do you want to permanently delete all items from Trash? This action cannot be undone."),
239  QString(),
240  KGuiItem(i18nc("@action:button", "Empty Trash"),
241  QIcon::fromTheme(QStringLiteral("user-trash"))),
242  KStandardGuiItem::cancel(),
243  keyName, options);
244  break;
245  case Trash:
246  default:
247  result = KMessageBox::warningContinueCancelList(
248  widget,
249  i18np("Do you really want to move this item to the trash?", "Do you really want to move these %1 items to the trash?", prettyList.count()),
250  prettyList,
251  i18n("Move to Trash"),
252  KGuiItem(i18nc("Verb", "&Trash"), QStringLiteral("user-trash")),
253  KStandardGuiItem::cancel(),
254  keyName, options);
255  }
256  if (!keyName.isEmpty()) {
257  // Check kmessagebox setting... erase & copy to konquerorrc.
258  KSharedConfig::Ptr config = KSharedConfig::openConfig();
259  KConfigGroup notificationGroup(config, "Notification Messages");
260  if (!notificationGroup.readEntry(keyName, true)) {
261  notificationGroup.writeEntry(keyName, true);
262  notificationGroup.sync();
263 
264  KSharedConfigPtr kioConfig = KSharedConfig::openConfig(QStringLiteral("kiorc"), KConfig::NoGlobals);
265  kioConfig->group("Confirmations").writeEntry(keyName, false);
266  }
267  }
268  return (result == KMessageBox::Continue);
269  }
270  return true;
271 }

273 int KIO::JobUiDelegate::requestMessageBox(KIO::JobUiDelegate::MessageBoxType type,
274  const QString &text, const QString &caption,
275  const QString &buttonYes, const QString &buttonNo,
276  const QString &iconYes, const QString &iconNo,
277  const QString &dontAskAgainName,
278  const KIO::MetaData &sslMetaData)
279 {
280  int result = -1;
281 
282  //qDebug() << type << text << "caption=" << caption;
283 
284  KConfig config(QStringLiteral("kioslaverc"));
285  KMessageBox::setDontShowAgainConfig(&config);
286 
287  const KGuiItem buttonYesGui(buttonYes, iconYes);
288  const KGuiItem buttonNoGui(buttonNo, iconNo);
289  KMessageBox::Options options(KMessageBox::Notify | KMessageBox::WindowModal);
290 
291  switch (type) {
292  case QuestionYesNo:
293  result = KMessageBox::questionYesNo(
294  window(), text, caption, buttonYesGui,
295  buttonNoGui, dontAskAgainName, options);
296  break;
297  case WarningYesNo:
298  result = KMessageBox::warningYesNo(
299  window(), text, caption, buttonYesGui,
300  buttonNoGui, dontAskAgainName,
301  options | KMessageBox::Dangerous);
302  break;
303  case WarningYesNoCancel:
304  result = KMessageBox::warningYesNoCancel(
305  window(), text, caption, buttonYesGui, buttonNoGui,
306  KStandardGuiItem::cancel(), dontAskAgainName, options);
307  break;
308  case WarningContinueCancel:
309  result = KMessageBox::warningContinueCancel(
310  window(), text, caption, buttonYesGui,
311  KStandardGuiItem::cancel(), dontAskAgainName, options);
312  break;
313  case Information:
314  KMessageBox::information(window(), text, caption, dontAskAgainName, options);
315  result = 1; // whatever
316  break;
317  case SSLMessageBox: {
318  QPointer<KSslInfoDialog> kid(new KSslInfoDialog(window()));
319  //### this is boilerplate code and appears in khtml_part.cpp almost unchanged!
320  const QStringList sl = sslMetaData.value(QStringLiteral("ssl_peer_chain")).split('\x01', QString::SkipEmptyParts);
321  QList<QSslCertificate> certChain;
322  bool decodedOk = true;
323  foreach (const QString &s, sl) {
324  certChain.append(QSslCertificate(s.toLatin1())); //or is it toLocal8Bit or whatever?
325  if (certChain.last().isNull()) {
326  decodedOk = false;
327  break;
328  }
329  }
330 
331  if (decodedOk) {
332  result = 1; // whatever
333  kid->setSslInfo(certChain,
334  sslMetaData.value(QStringLiteral("ssl_peer_ip")),
335  text, // the URL
336  sslMetaData.value(QStringLiteral("ssl_protocol_version")),
337  sslMetaData.value(QStringLiteral("ssl_cipher")),
338  sslMetaData.value(QStringLiteral("ssl_cipher_used_bits")).toInt(),
339  sslMetaData.value(QStringLiteral("ssl_cipher_bits")).toInt(),
340  KSslInfoDialog::errorsFromString(sslMetaData.value(QStringLiteral("ssl_cert_errors"))));
341  kid->exec();
342  } else {
343  result = -1;
344  KMessageBox::information(window(),
345  i18n("The peer SSL certificate chain appears to be corrupt."),
346  i18n("SSL"), QString(), options);
347  }
348  // KSslInfoDialog deletes itself (Qt::WA_DeleteOnClose).
349  delete kid;
350  break;
351  }
352  default:
353  qWarning() << "Unknown type" << type;
354  result = 0;
355  break;
356  }
357  KMessageBox::setDontShowAgainConfig(0);
358  return result;
359 }
