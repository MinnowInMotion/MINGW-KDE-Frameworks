120 static QFile::Permissions modeToQFilePermissions(int mode)
121 {
122  QFile::Permissions perms;
123  if (mode & S_IRUSR) {
124  perms |= QFile::ReadOwner;
125  }
126  if (mode & S_IWUSR) {
127  perms |= QFile::WriteOwner;
128  }
129  if (mode & S_IXUSR) {
130  perms |= QFile::ExeOwner;
131  }
132  if (mode & S_IRGRP) {
133  perms |= QFile::ReadGroup;
134  }
135  if (mode & S_IWGRP) {

138  if (mode & S_IXGRP) {
139  perms |= QFile::ExeGroup;

174 int FileProtocol::setACL(const char *path, mode_t perm, bool directoryDefault)
175 {
176  int ret = 0;
177 #if HAVE_POSIX_ACL
178 
179  const QString ACLString = metaData(QStringLiteral("ACL_STRING"));
180  const QString defaultACLString = metaData(QStringLiteral("DEFAULT_ACL_STRING"));
181  // Empty strings mean leave as is
182  if (!ACLString.isEmpty()) {
183  acl_t acl = 0;
184  if (ACLString == QLatin1String("ACL_DELETE")) {
185  // user told us to delete the extended ACL, so let's write only
186  // the minimal (UNIX permission bits) part
187  acl = acl_from_mode(perm);
188  }
189  acl = acl_from_text(ACLString.toLatin1());
190  if (acl_valid(acl) == 0) { // let's be safe
191  ret = acl_set_file(path, ACL_TYPE_ACCESS, acl);
192  // qDebug() << "Set ACL on:" << path << "to:" << aclToText(acl);
193  }
194  acl_free(acl);
195  if (ret != 0) {
196  return ret; // better stop trying right away
197  }
198  }
199 
200  if (directoryDefault && !defaultACLString.isEmpty()) {
201  if (defaultACLString == QLatin1String("ACL_DELETE")) {
202  // user told us to delete the default ACL, do so
203  ret += acl_delete_def_file(path);
204  } else {
205  acl_t acl = acl_from_text(defaultACLString.toLatin1());
206  if (acl_valid(acl) == 0) { // let's be safe
207  ret += acl_set_file(path, ACL_TYPE_DEFAULT, acl);
208  // qDebug() << "Set Default ACL on:" << path << "to:" << aclToText(acl);
209  }
210  acl_free(acl);
211  }
212  }
213 #else
214  Q_UNUSED(path);
215  Q_UNUSED(perm);
216  Q_UNUSED(directoryDefault);
217 #endif
218  return ret;
219 }

221 void FileProtocol::chmod(const QUrl &url, int permissions)
222 {
223  const QString path(url.toLocalFile());
224  const QByteArray _path(QFile::encodeName(path));
225  /* FIXME: Should be atomic */
226  if (!QFile::setPermissions(path, modeToQFilePermissions(permissions)) ||
227  (setACL(_path.data(), permissions, false) == -1) ||
228  /* if not a directory, cannot set default ACLs */
229  (setACL(_path.data(), permissions, true) == -1 && errno != ENOTDIR)) {
230 
231  switch (errno) {
232  case EPERM:
233  case EACCES:
234  error(KIO::ERR_ACCESS_DENIED, path);
235  break;
236 #if defined(ENOTSUP)
237  case ENOTSUP: // from setACL since chmod can't return ENOTSUP
238  error(KIO::ERR_UNSUPPORTED_ACTION, i18n("Setting ACL for %1", path));
239  break;
240 #endif
241  case ENOSPC:
242  error(KIO::ERR_DISK_FULL, path);
243  break;
244  default:
245  error(KIO::ERR_CANNOT_CHMOD, path);
246  }
247  } else {
248  finished();
249  }

307 void FileProtocol::get(const QUrl &url)
308 {
309  if (!url.isLocalFile()) {
310  QUrl redir(url);
311  redir.setScheme(config()->readEntry("DefaultRemoteProtocol", "smb"));
312  redirection(redir);
313  finished();
314  return;
315  }
316 
317  const QString path(url.toLocalFile());
318  QT_STATBUF buff;
319  if (QT_STAT(QFile::encodeName(path).constData(), &buff) == -1) {
320  if (errno == EACCES) {
321  error(KIO::ERR_ACCESS_DENIED, path);
322  } else {
323  error(KIO::ERR_DOES_NOT_EXIST, path);
324  }
325  return;
326  }
327 
328  if ((buff.st_mode & QT_STAT_MASK) == QT_STAT_DIR) {
329  error(KIO::ERR_IS_DIRECTORY, path);
330  return;
331  }
332  if ((buff.st_mode & QT_STAT_MASK) != QT_STAT_REG) {
333  error(KIO::ERR_CANNOT_OPEN_FOR_READING, path);
334  return;
335  }
336 
337  QFile f(path);
338  if (!f.open(QIODevice::ReadOnly)) {
339  error(KIO::ERR_CANNOT_OPEN_FOR_READING, path);
340  return;
341  }
342 
343 #if HAVE_FADVISE
344  //TODO check return code
345  posix_fadvise(f.handle(), 0, 0, POSIX_FADV_SEQUENTIAL);
346 #endif
347 
348  // Determine the mimetype of the file to be retrieved, and emit it.
349  // This is mandatory in all slaves (for KRun/BrowserRun to work)
350  // In real "remote" slaves, this is usually done using mimeTypeForFileNameAndData
351  // after receiving some data. But we don't know how much data the mimemagic rules
352  // need, so for local files, better use mimeTypeForFile.
353  QMimeDatabase db;
354  QMimeType mt = db.mimeTypeForFile(url.toLocalFile());
355  emit mimeType(mt.name());
356  // Emit total size AFTER mimetype
357  totalSize(buff.st_size);
358 
359  KIO::filesize_t processed_size = 0;
360 
361  QString resumeOffset = metaData(QStringLiteral("range-start"));
362  if (resumeOffset.isEmpty()) {
363  resumeOffset = metaData(QStringLiteral("resume")); // old name
364  }
365  if (!resumeOffset.isEmpty()) {
366  bool ok;
367  KIO::fileoffset_t offset = resumeOffset.toLongLong(&ok);
368  if (ok && (offset > 0) && (offset < buff.st_size)) {
369  if (f.seek(offset)) {
370  canResume();
371  processed_size = offset;
372  // qDebug() << "Resume offset:" << KIO::number(offset);
373  }
374  }
375  }
376 
377  char buffer[ MAX_IPC_SIZE ];
378  QByteArray array;
379 
380  while (1) {
381  int n = f.read(buffer, MAX_IPC_SIZE);
382  if (n == -1) {
383  if (errno == EINTR) {
384  continue;
385  }
386  error(KIO::ERR_CANNOT_READ, path);
387  f.close();
388  return;
389  }
390  if (n == 0) {
391  break; // Finished
392  }
393 
394  array = QByteArray::fromRawData(buffer, n);
395  data(array);
396  array.clear();
397 
398  processed_size += n;
399  processedSize(processed_size);
400 
401  //qDebug() << "Processed: " << KIO::number (processed_size);
402  }
403 
404  data(QByteArray());
405 
406  f.close();
407 
408  processedSize(buff.st_size);
409  finished();
410 }

421  } else {
422  error(KIO::ERR_DOES_NOT_EXIST, openPath);
423  }
424  return;
425  }
426 
427  if ((buff.st_mode & QT_STAT_MASK) == QT_STAT_DIR) {
428  error(KIO::ERR_IS_DIRECTORY, openPath);
429  return;
430  }
431  if ((buff.st_mode & QT_STAT_MASK) != QT_STAT_REG) {
432  error(KIO::ERR_CANNOT_OPEN_FOR_READING, openPath);
433  return;
434  }
435 
436  mFile = new QFile(openPath);
437  if (!mFile->open(mode)) {
438  if (mode & QIODevice::ReadOnly) {
439  error(KIO::ERR_CANNOT_OPEN_FOR_READING, openPath);
440  } else {
441  error(KIO::ERR_CANNOT_OPEN_FOR_WRITING, openPath);
442  }
443

445  }
446  // Determine the mimetype of the file to be retrieved, and emit it.

451  QMimeDatabase db;
452  QMimeType mt = db.mimeTypeForFile(url.toLocalFile());
453  emit mimeType(mt.name());
454  }
455 
456  totalSize(buff.st_size);
457  position(0);
458 
459  emit opened();

532 void FileProtocol::put(const QUrl &url, int _mode, KIO::JobFlags _flags)
533 {
534  const QString dest_orig = url.toLocalFile();
535 
536  // qDebug() << dest_orig << "mode=" << _mode;
537 
538  QString dest_part(dest_orig + QLatin1String(".part"));
539 
540  QT_STATBUF buff_orig;
541  const bool bOrigExists = (QT_LSTAT(QFile::encodeName(dest_orig).constData(), &buff_orig) != -1);
542  bool bPartExists = false;
543  const bool bMarkPartial = config()->readEntry("MarkPartial", true);
544 
545  if (bMarkPartial) {
546  QT_STATBUF buff_part;
547  bPartExists = (QT_LSTAT(QFile::encodeName(dest_part).constData(), &buff_part) != -1);
548 
549  if (bPartExists && !(_flags & KIO::Resume) && !(_flags & KIO::Overwrite) && buff_part.st_size > 0 && ((buff_part.st_mode & QT_STAT_MASK) == QT_STAT_REG)) {
550  // qDebug() << "calling canResume with" << KIO::number(buff_part.st_size);
551 
552  // Maybe we can use this partial file for resuming
553  // Tell about the size we have, and the app will tell us
554  // if it's ok to resume or not.
555  _flags |= canResume(buff_part.st_size) ? KIO::Resume : KIO::DefaultFlags;
556 
557  // qDebug() << "got answer" << (_flags & KIO::Resume);
558  }
559  }
560 
561  if (bOrigExists && !(_flags & KIO::Overwrite) && !(_flags & KIO::Resume)) {
562  if ((buff_orig.st_mode & QT_STAT_MASK) == QT_STAT_DIR) {
563  error(KIO::ERR_DIR_ALREADY_EXIST, dest_orig);
564  } else {
565  error(KIO::ERR_FILE_ALREADY_EXIST, dest_orig);
566  }
567  return;
568  }
569 
570  int result;
571  QString dest;
572  QByteArray _dest;
573  QFile f;
574 
575  // Loop until we got 0 (end of data)
576  do {
577  QByteArray buffer;
578  dataReq(); // Request for data
579  result = readData(buffer);
580 
581  if (result >= 0) {
582  if (dest.isEmpty()) {
583  if (bMarkPartial) {
584  // qDebug() << "Appending .part extension to" << dest_orig;
585  dest = dest_part;
586  if (bPartExists && !(_flags & KIO::Resume)) {
587  // qDebug() << "Deleting partial file" << dest_part;
588  QFile::remove(dest_part);
589  // Catch errors when we try to open the file.
590  }
591  } else {
592  dest = dest_orig;
593  if (bOrigExists && !(_flags & KIO::Resume)) {
594  // qDebug() << "Deleting destination file" << dest_orig;
595  QFile::remove(dest_orig);
596  // Catch errors when we try to open the file.
597  }
598  }
599 
600  f.setFileName(dest);
601 
602  if ((_flags & KIO::Resume)) {
603  f.open(QIODevice::ReadWrite | QIODevice::Append);
604  } else {
605  f.open(QIODevice::Truncate | QIODevice::WriteOnly);
606  if (_mode != -1) {
607  // WABA: Make sure that we keep writing permissions ourselves,
608  // otherwise we can be in for a surprise on NFS.
609  mode_t initialMode = _mode | S_IWUSR | S_IRUSR;
610  f.setPermissions(modeToQFilePermissions(initialMode));
611  }
612  }
613 
614  if (!f.isOpen()) {
615  // qDebug() << "####################### COULD NOT WRITE" << dest << "_mode=" << _mode;
616  // qDebug() << "QFile error==" << f.error() << "(" << f.errorString() << ")";
617 
618  if (f.error() == QFileDevice::PermissionsError) {
619  error(KIO::ERR_WRITE_ACCESS_DENIED, dest);
620  } else {
621  error(KIO::ERR_CANNOT_OPEN_FOR_WRITING, dest);
622  }
623  return;
624  }
625  }
626 
627  if (f.write(buffer) == -1) {
628  if (f.error() == QFile::ResourceError) { // disk full
629  error(KIO::ERR_DISK_FULL, dest_orig);
630  result = -2; // means: remove dest file
631  } else {
632  qWarning() << "Couldn't write. Error:" << f.errorString();
633  error(KIO::ERR_CANNOT_WRITE, dest_orig);
634  result = -1;
635  }
636  }
637  }
638  } while (result > 0);
639 
640  // An error occurred deal with it.
641  if (result < 0) {
642  // qDebug() << "Error during 'put'. Aborting.";
643 
644  if (f.isOpen()) {
645  f.close();
646 
647  QT_STATBUF buff;
648  if (QT_STAT(QFile::encodeName(dest).constData(), &buff) == 0) {
649  int size = config()->readEntry("MinimumKeepSize", DEFAULT_MINIMUM_KEEP_SIZE);
650  if (buff.st_size < size) {
651  QFile::remove(dest);
652  }
653  }
654  }
655 
656  ::exit(255);
657  }
658 
659  if (!f.isOpen()) { // we got nothing to write out, so we never opened the file
660  finished();
661  return;
662  }
663 
664  f.close();
665 
666  if (f.error() != QFile::NoError) {
667  qWarning() << "Error when closing file descriptor:" << f.errorString();
668  error(KIO::ERR_CANNOT_WRITE, dest_orig);
669  return;
670  }
671 
672  // after full download rename the file back to original name
673  if (bMarkPartial) {
674  //QFile::rename() never overwrites the destination file unlike ::remove,
675  //so we must remove it manually first
676  if (_flags & KIO::Overwrite) {
677  QFile::remove(dest_orig);
678  }
679 
680  if (!QFile::rename(dest, dest_orig)) {
681  qWarning() << " Couldn't rename " << dest << " to " << dest_orig;
682  error(KIO::ERR_CANNOT_RENAME_PARTIAL, dest_orig);
683  return;
684  }
685  org::kde::KDirNotify::emitFileRenamed(QUrl::fromLocalFile(dest), QUrl::fromLocalFile(dest_orig));
686  }
687 
688  // set final permissions
689  if (_mode != -1 && !(_flags & KIO::Resume)) {
690  if (!QFile::setPermissions(dest_orig, modeToQFilePermissions(_mode))) {
691  // couldn't chmod. Eat the error if the filesystem apparently doesn't support it.
692  KMountPoint::Ptr mp = KMountPoint::currentMountPoints().findByPath(dest_orig);
693  if (mp && mp->testFileSystemFlag(KMountPoint::SupportsChmod)) {
694  warning(i18n("Could not change permissions for\n%1", dest_orig));
695  }
696  }
697  }
698 
699  // set modification time
700  const QString mtimeStr = metaData(QStringLiteral("modified"));
701  if (!mtimeStr.isEmpty()) {
702  QDateTime dt = QDateTime::fromString(mtimeStr, Qt::ISODate);
703  if (dt.isValid()) {
704  QT_STATBUF dest_statbuf;
705  if (QT_STAT(QFile::encodeName(dest_orig).constData(), &dest_statbuf) == 0) {
706 #ifndef Q_OS_WIN
707  struct timeval utbuf[2];
708  // access time
709  utbuf[0].tv_sec = dest_statbuf.st_atime; // access time, unchanged ## TODO preserve msec
710  utbuf[0].tv_usec = 0;
711  // modification time
712  utbuf[1].tv_sec = dt.toTime_t();
713  utbuf[1].tv_usec = dt.time().msec() * 1000;
714  utimes(QFile::encodeName(dest_orig).constData(), utbuf);
715 #else
716  struct utimbuf utbuf;
717  utbuf.actime = dest_statbuf.st_atime;
718  utbuf.modtime = dt.toTime_t();
719  utime(QFile::encodeName(dest_orig).constData(), &utbuf);
720 #endif
721  }
722  }
723 
724  }
725 
726  // We have done our job => finish
727  finished();
728 }

764 bool FileProtocol::createUDSEntry(const QString &filename, const QByteArray &path, UDSEntry &entry,
765  short int details)
766 {
767  assert(entry.count() == 0); // by contract :-)
768  entry.reserve(8);
769 
770  entry.insert(KIO::UDSEntry::UDS_NAME, filename);
771 
772  mode_t type;
773  mode_t access;
774  QT_STATBUF buff;
775 
776  if (QT_LSTAT(path.data(), &buff) == 0) {
777 
778  if (details > 2) {
779  entry.insert(KIO::UDSEntry::UDS_DEVICE_ID, buff.st_dev);
780  entry.insert(KIO::UDSEntry::UDS_INODE, buff.st_ino);
781  }
782 
783  if ((buff.st_mode & QT_STAT_MASK) == QT_STAT_LNK) {
784 
785 #ifdef Q_OS_WIN
786  const QString linkTarget = QFile::symLinkTarget(QFile::decodeName(path));
787 #else
788  // Use readlink on Unix because symLinkTarget turns relative targets into absolute (#352927)
789  QByteArray linkTargetBuffer;
790  linkTargetBuffer.resize(buff.st_size);
791  const int n = readlink(path.constData(), linkTargetBuffer.data(), buff.st_size);
792  if (n == -1) {
793  qWarning() << "readlink failed!" << path;
794  return false;
795  }
796  const QString linkTarget = QFile::decodeName(linkTargetBuffer);
797 #endif
798  entry.insert(KIO::UDSEntry::UDS_LINK_DEST, linkTarget);
799 
800  // A symlink -> follow it only if details>1
801  if (details > 1 && QT_STAT(path.constData(), &buff) == -1) {
802  // It is a link pointing to nowhere
803  type = S_IFMT - 1;
804  access = S_IRWXU | S_IRWXG | S_IRWXO;
805 
806  entry.insert(KIO::UDSEntry::UDS_FILE_TYPE, type);
807  entry.insert(KIO::UDSEntry::UDS_ACCESS, access);
808  entry.insert(KIO::UDSEntry::UDS_SIZE, 0LL);
809  goto notype;
810 
811  }
812  }
813  } else {
814  // qWarning() << "lstat didn't work on " << path.data();
815  return false;
816  }
817 
818  type = buff.st_mode & S_IFMT; // extract file type
819  access = buff.st_mode & 07777; // extract permissions
820 
821  entry.insert(KIO::UDSEntry::UDS_FILE_TYPE, type);
822  entry.insert(KIO::UDSEntry::UDS_ACCESS, access);
823 
824  entry.insert(KIO::UDSEntry::UDS_SIZE, buff.st_size);
825 
826 #if HAVE_POSIX_ACL
827  if (details > 1) {
828  /* Append an atom indicating whether the file has extended acl information
829  * and if withACL is specified also one with the acl itself. If it's a directory
830  * and it has a default ACL, also append that. */
831  appendACLAtoms(path, entry, type);
832  }
833 #endif
834 
835 notype:
836  if (details > 0) {
837  entry.insert(KIO::UDSEntry::UDS_MODIFICATION_TIME, buff.st_mtime);
838 #ifndef Q_OS_WIN
839  entry.insert(KIO::UDSEntry::UDS_USER, getUserName(KUserId(buff.st_uid)));
840  entry.insert(KIO::UDSEntry::UDS_GROUP, getGroupName(KGroupId(buff.st_gid)));
841 #else
842 #pragma message("TODO: st_uid and st_gid are always zero, use GetSecurityInfo to find the owner")
843 #endif
844  entry.insert(KIO::UDSEntry::UDS_ACCESS_TIME, buff.st_atime);
845  }
846 
847  // Note: buff.st_ctime isn't the creation time !
848  // We made that mistake for KDE 2.0, but it's in fact the
849  // "file status" change time, which we don't care about.
850 
851  return true;
852 }

901 void FileProtocol::mount(bool _ro, const char *_fstype, const QString &_dev, const QString &_point)
902 {
903  // qDebug() << "fstype=" << _fstype;
904 
905 #ifndef _WIN32_WCE
906 #if HAVE_VOLMGT
907  /*
908  * support for Solaris volume management
909  */
910  QString err;
911  QByteArray devname = QFile::encodeName(_dev);
912 
913  if (volmgt_running()) {
914 // qDebug() << "VOLMGT: vold ok.";
915  if (volmgt_check(devname.data()) == 0) {
916  // qDebug() << "VOLMGT: no media in " << devname.data();
917  err = i18n("No Media inserted or Media not recognized.");
918  error(KIO::ERR_CANNOT_MOUNT, err);

920  } else {
921  // qDebug() << "VOLMGT: " << devname.data() << ": media ok";
922  finished();
923  return;
924  }
925  } else {
926  err = i18n("\"vold\" is not running.");
927  // qDebug() << "VOLMGT: " << err;
928  error(KIO::ERR_CANNOT_MOUNT, err);
929  return;
930  }
931 #else
932 
933  QTemporaryFile tmpFile;
934  tmpFile.setAutoRemove(false);
935  tmpFile.open();
936  QByteArray tmpFileName = QFile::encodeName(tmpFile.fileName());
937  QByteArray dev;
938  if (_dev.startsWith(QLatin1String("LABEL="))) { // turn LABEL=foo into -L foo (#71430)
939  QString labelName = _dev.mid(6);
940  dev = "-L ";
941  dev += QFile::encodeName(KShell::quoteArg(labelName)); // is it correct to assume same encoding as filesystem?
942  } else if (_dev.startsWith(QLatin1String("UUID="))) { // and UUID=bar into -U bar
943  QString uuidName = _dev.mid(5);
944  dev = "-U ";

946  } else {
947  dev = QFile::encodeName(KShell::quoteArg(_dev)); // get those ready to be given to a shell
948  }

950  QByteArray point = QFile::encodeName(KShell::quoteArg(_point));
951  bool fstype_empty = !_fstype || !*_fstype;
952  QByteArray fstype = KShell::quoteArg(QString::fromLatin1(_fstype)).toLatin1(); // good guess
953  QByteArray readonly = _ro ? "-r" : "";
954  QByteArray mountProg = QStandardPaths::findExecutable(QStringLiteral("mount")).toLocal8Bit();
955  if (mountProg.isEmpty()) {
956  mountProg = QStandardPaths::findExecutable(QStringLiteral("mount"), fallbackSystemPath()).toLocal8Bit();
957  }
958  if (mountProg.isEmpty()) {
959  error(KIO::ERR_CANNOT_MOUNT, i18n("Could not find program \"mount\""));
960  return;
961  }
962 
963  // Two steps, in case mount doesn't like it when we pass all options
964  for (int step = 0; step <= 1; step++) {
965  QByteArray buffer = mountProg + ' ';
966  // Mount using device only if no fstype nor mountpoint (KDE-1.x like)
967  if (!dev.isEmpty() && _point.isEmpty() && fstype_empty) {
968  buffer += dev;
969  } else
970  // Mount using the mountpoint, if no fstype nor device (impossible in first step)
971  if (!_point.isEmpty() && dev.isEmpty() && fstype_empty) {
972  buffer += point;
973  } else
974  // mount giving device + mountpoint but no fstype
975  if (!_point.isEmpty() && !dev.isEmpty() && fstype_empty) {
976  buffer += readonly + ' ' + dev + ' ' + point;
977  } else
978  // mount giving device + mountpoint + fstype
979 #if defined(__svr4__) && defined(Q_OS_SOLARIS) // MARCO for Solaris 8 and I
980  // believe this is true for SVR4 in general
981  buffer += "-F " + fstype + ' ' + (_ro ? "-oro" : "") + ' ' + dev + ' ' + point;
982 #else
983  buffer += readonly + " -t " + fstype + ' ' + dev + ' ' + point;
984 #endif
985  buffer += " 2>" + tmpFileName;
986  // qDebug() << buffer;
987 
988  int mount_ret = system(buffer.constData());
989 
990  QString err = readLogFile(tmpFileName);
991  if (err.isEmpty() && mount_ret == 0) {
992  finished();
993  return;
994  } else {
995  // Didn't work - or maybe we just got a warning
996  KMountPoint::Ptr mp = KMountPoint::currentMountPoints().findByDevice(_dev);
997  // Is the device mounted ?
998  if (mp && mount_ret == 0) {
999  // qDebug() << "mount got a warning:" << err;
1000  warning(err);
1001  finished();
1002  return;
1003  } else {
1004  if ((step == 0) && !_point.isEmpty()) {
1005  // qDebug() << err;
1006  // qDebug() << "Mounting with those options didn't work, trying with only mountpoint";
1007  fstype = "";
1008  fstype_empty = true;
1009  dev = "";
1010  // The reason for trying with only mountpoint (instead of
1011  // only device) is that some people (hi Malte!) have the
1012  // same device associated with two mountpoints
1013  // for different fstypes, like /dev/fd0 /mnt/e2floppy and
1014  // /dev/fd0 /mnt/dosfloppy.
1015  // If the user has the same mountpoint associated with two
1016  // different devices, well they shouldn't specify the
1017  // mountpoint but just the device.
1018  } else {
1019  error(KIO::ERR_CANNOT_MOUNT, err);
1020  return;
1021  }
1022  }
1023  }
1024  }
1025 #endif /* ! HAVE_VOLMGT */
1026 #else
1027  QString err;
1028  err = i18n("mounting is not supported by wince.");
1029  error(KIO::ERR_CANNOT_MOUNT, err);
1030 #endif
1031

1034 void FileProtocol::unmount(const QString &_point)
1035 {

1037  QByteArray buffer;
1038 
1039  QTemporaryFile tmpFile;
1040  tmpFile.setAutoRemove(false);
1041  tmpFile.open();
1042  QByteArray tmpFileName = QFile::encodeName(tmpFile.fileName());
1043  QString err;
1044 
1045 #if HAVE_VOLMGT
1046  /*
1047  * support for Solaris volume management
1048  */
1049  char *devname;
1050  char *ptr;
1051  FILE *mnttab;
1052  struct mnttab mnt;
1053 
1054  if (volmgt_running()) {
1055  // qDebug() << "VOLMGT: looking for "
1056  << _point.toLocal8Bit();
1057 
1058  if ((mnttab = QT_FOPEN(MNTTAB, "r")) == NULL) {
1059  err = QLatin1String("could not open mnttab");
1060  // qDebug() << "VOLMGT: " << err;
1061  error(KIO::ERR_CANNOT_UNMOUNT, err);
1062  return;
1063  }
1064 
1065  /*
1066  * since there's no way to derive the device name from
1067  * the mount point through the volmgt library (and
1068  * media_findname() won't work in this case), we have to
1069  * look ourselves...
1070  */
1071  devname = NULL;
1072  rewind(mnttab);
1073  while (getmntent(mnttab, &mnt) == 0) {
1074  if (strcmp(_point.toLocal8Bit(), mnt.mnt_mountp) == 0) {
1075  devname = mnt.mnt_special;

1082  err = QLatin1String("not in mnttab");

1086  error(KIO::ERR_CANNOT_UNMOUNT, err);
1087  return;
1088  }
1089

1091  * strip off the directory name (volume name)
1092  * the eject(1) command will handle unmounting and
1093  * physically eject the media (if possible)
1094  */
1095  ptr = strrchr(devname, '/');
1096  *ptr = '\0';
1097  QByteArray qdevname(QFile::encodeName(KShell::quoteArg(QFile::decodeName(QByteArray(devname)))).data());
1098  buffer = "/usr/bin/eject " + qdevname + " 2>" + tmpFileName;
1099  // qDebug() << "VOLMGT: eject " << qdevname;
1100 
1101  /*
1102  * from eject(1): exit status == 0 => need to manually eject
1103  * exit status == 4 => media was ejected
1104  */
1105  if (WEXITSTATUS(system(buffer.constData())) == 4) {
1106  /*
1107  * this is not an error, so skip "readLogFile()"
1108  * to avoid wrong/confusing error popup. The
1109  * temporary file is removed by QTemporaryFile's

1112  finished();

1114  }
1115  } else {

1118  * so we probably could call eject anyway, but since the
1119  * media is mounted now, vold must've died for some reason
1120  * during the user's session, so it should be restarted...
1121  */
1122  err = i18n("\"vold\" is not running.");

1124  error(KIO::ERR_CANNOT_UNMOUNT, err);
1125  return;
1126  }
1127 #else
1128  QByteArray umountProg = QStandardPaths::findExecutable(QStringLiteral("umount")).toLocal8Bit();

1130  umountProg = QStandardPaths::findExecutable(QStringLiteral("umount"), fallbackSystemPath()).toLocal8Bit();
1131  }
1132  if (umountProg.isEmpty()) {
1133  error(KIO::ERR_CANNOT_UNMOUNT, i18n("Could not find program \"umount\""));
1134  return;
1135  }

1137  system(buffer.constData());

1142  finished();
1143  } else {

1149  error(KIO::ERR_CANNOT_MOUNT, err);
