108 void ChmodJobPrivate::_k_processList()
109 {
110  Q_Q(ChmodJob);
111  while (!m_lstItems.isEmpty()) {
112  const KFileItem item = m_lstItems.first();
113  if (!item.isLink()) { // don't do anything with symlinks
114  // File or directory -> remember to chmod
115  ChmodInfo info;
116  info.url = item.url();
117  // This is a toplevel file, we apply changes directly (no +X emulation here)
118  const mode_t permissions = item.permissions() & 0777; // get rid of "set gid" and other special flags
119  info.permissions = (m_permissions & m_mask) | (permissions & ~m_mask);
120  /*//qDebug() << "toplevel url:" << info.url << "\n current permissions=" << QString::number(permissions,8)
121  << "\n wanted permission=" << QString::number(m_permissions,8)
122  << "\n with mask=" << QString::number(m_mask,8)
123  << "\n with ~mask (mask bits we keep) =" << QString::number((uint)~m_mask,8)
124  << "\n bits we keep =" << QString::number(permissions & ~m_mask,8)
125  << "\n new permissions = " << QString::number(info.permissions,8);*/
126  m_infos.prepend(info);
127  //qDebug() << "processList : Adding info for " << info.url;
128  // Directory and recursive -> list
129  if (item.isDir() && m_recursive) {
130  //qDebug() << "ChmodJob::processList dir -> listing";
131  KIO::ListJob *listJob = KIO::listRecursive(item.url(), KIO::HideProgressInfo);
132  q->connect(listJob, SIGNAL(entries(KIO::Job *,
133  const KIO::UDSEntryList &)),
134  SLOT(_k_slotEntries(KIO::Job*,KIO::UDSEntryList)));
135  q->addSubjob(listJob);
136  return; // we'll come back later, when this one's finished
137  }
138  }
139  m_lstItems.removeFirst();
140  }
141  //qDebug() << "ChmodJob::processList -> going to STATE_CHMODING";
142  // We have finished, move on
143  state = CHMODJOB_STATE_CHMODING;
144  _k_chmodNextFile();
145 }

147 void ChmodJobPrivate::_k_slotEntries(KIO::Job *, const KIO::UDSEntryList &list)
148 {
149  KIO::UDSEntryList::ConstIterator it = list.begin();
150  KIO::UDSEntryList::ConstIterator end = list.end();
151  for (; it != end; ++it) {
152  const KIO::UDSEntry &entry = *it;
153  const bool isLink = !entry.stringValue(KIO::UDSEntry::UDS_LINK_DEST).isEmpty();
154  const QString relativePath = entry.stringValue(KIO::UDSEntry::UDS_NAME);
155  if (!isLink && relativePath != QLatin1String("..")) {
156  const mode_t permissions = entry.numberValue(KIO::UDSEntry::UDS_ACCESS)
157  & 0777; // get rid of "set gid" and other special flags
158 
159  ChmodInfo info;
160  info.url = m_lstItems.first().url(); // base directory
161  info.url.setPath(info.url.path() + '/' + relativePath);
162  int mask = m_mask;
163  // Emulate -X: only give +x to files that had a +x bit already
164  // So the check is the opposite : if the file had no x bit, don't touch x bits
165  // For dirs this doesn't apply
166  if (!entry.isDir()) {
167  int newPerms = m_permissions & mask;
168  if ((newPerms & 0111) && !(permissions & 0111)) {
169  // don't interfere with mandatory file locking
170  if (newPerms & 02000) {
171  mask = mask & ~0101;
172  } else {
173  mask = mask & ~0111;
174  }
175  }
176  }
177  info.permissions = (m_permissions & mask) | (permissions & ~mask);
178  /*//qDebug() << info.url << "\n current permissions=" << QString::number(permissions,8)
179  << "\n wanted permission=" << QString::number(m_permissions,8)
180  << "\n with mask=" << QString::number(mask,8)
181  << "\n with ~mask (mask bits we keep) =" << QString::number((uint)~mask,8)
182  << "\n bits we keep =" << QString::number(permissions & ~mask,8)
183  << "\n new permissions = " << QString::number(info.permissions,8);*/
184  // Prepend this info in our todo list.
185  // This way, the toplevel dirs are done last.
186  m_infos.prepend(info);
187  }
188  }
189 }

191 void ChmodJobPrivate::_k_chmodNextFile()
192 {
193  Q_Q(ChmodJob);
194  if (!m_infos.isEmpty()) {
195  ChmodInfo info = m_infos.takeFirst();
196  // First update group / owner (if local file)
197  // (permissions have to set after, in case of suid and sgid)
198  if (info.url.isLocalFile() && (m_newOwner.isValid() || m_newGroup.isValid())) {
199  QString path = info.url.toLocalFile();
200  if (!KIOPrivate::changeOwnership(path, m_newOwner, m_newGroup)) {
201  if (!m_uiDelegateExtension) {
202  emit q->warning(q, i18n("Could not modify the ownership of file %1", path));
203  } else if (!m_bAutoSkipFiles) {
204  const QString errMsg = i18n("<qt>Could not modify the ownership of file <b>%1</b>. You have insufficient access to the file to perform the change.</qt>", path);
205  SkipDialog_Options options;
206  if (m_infos.count() > 1) {
207  options |= SkipDialog_MultipleItems;
208  }
209  const SkipDialog_Result skipResult = m_uiDelegateExtension->askSkip(q, options, errMsg);
210  switch (skipResult) {
211  case Result_AutoSkip:
212  m_bAutoSkipFiles = true;
213  // fall through
214  case Result_Skip:
215  QMetaObject::invokeMethod(q, "_k_chmodNextFile", Qt::QueuedConnection);
216  return;
217  case Result_Retry:
218  m_infos.prepend(info);
219  QMetaObject::invokeMethod(q, "_k_chmodNextFile", Qt::QueuedConnection);
220  return;
221  case Result_Cancel:
222  default:
223  q->setError(ERR_USER_CANCELED);
224  q->emitResult();
225  return;
226  }
227  }
228  }
229  }
230 
231  /*qDebug() << "chmod'ing" << info.url
232  << "to" << QString::number(info.permissions,8);*/
233  KIO::SimpleJob *job = KIO::chmod(info.url, info.permissions);
234  // copy the metadata for acl and default acl
235  const QString aclString = q->queryMetaData(QStringLiteral("ACL_STRING"));
236  const QString defaultAclString = q->queryMetaData(QStringLiteral("DEFAULT_ACL_STRING"));
237  if (!aclString.isEmpty()) {
238  job->addMetaData(QStringLiteral("ACL_STRING"), aclString);
239  }
240  if (!defaultAclString.isEmpty()) {
241  job->addMetaData(QStringLiteral("DEFAULT_ACL_STRING"), defaultAclString);
242  }
243  q->addSubjob(job);
244  } else
245  // We have finished
246  {
247  q->emitResult();
248  }

251 void ChmodJob::slotResult(KJob *job)
252 {
253  Q_D(ChmodJob);
254  removeSubjob(job);
255  if (job->error()) {
256  setError(job->error());
257  setErrorText(job->errorText());
258  emitResult();
259  return;
260  }
261  //qDebug() << "d->m_lstItems:" << d->m_lstItems.count();
262  switch (d->state) {
263  case CHMODJOB_STATE_LISTING:
264  d->m_lstItems.removeFirst();
265  //qDebug() << "-> processList";
266  d->_k_processList();
267  return;
268  case CHMODJOB_STATE_CHMODING:
269  //qDebug() << "-> chmodNextFile";
270  d->_k_chmodNextFile();
271  return;
272  default:
273  Q_ASSERT(false);
274  return;
275  }
276 }
