52 void CacheConfigModule::load()
53 {
54  ui.cbUseCache->setChecked(KProtocolManager::useCache());
55  ui.sbMaxCacheSize->setValue( KProtocolManager::maxCacheSize() );
56 
57  KIO::CacheControl cc = KProtocolManager::cacheControl();
58 
59  if (cc==KIO::CC_Verify)
60  ui.rbVerifyCache->setChecked( true );
61  else if (cc==KIO::CC_Refresh)
62  ui.rbVerifyCache->setChecked( true );
63  else if (cc==KIO::CC_CacheOnly)
64  ui.rbOfflineMode->setChecked( true );
65  else if (cc==KIO::CC_Cache)
66  ui.rbCacheIfPossible->setChecked( true );
67 
68  // Config changed notifications...
69  connect ( ui.cbUseCache, SIGNAL(toggled(bool)), SLOT(configChanged()) );
70  connect ( ui.rbVerifyCache, SIGNAL(toggled(bool)), SLOT(configChanged()) );
71  connect ( ui.rbOfflineMode, SIGNAL(toggled(bool)), SLOT(configChanged()) );
72  connect ( ui.rbCacheIfPossible, SIGNAL(toggled(bool)), SLOT(configChanged()) );
73  connect ( ui.sbMaxCacheSize, SIGNAL(valueChanged(int)), SLOT(configChanged()) );
74  emit changed( false );
75 }

77 void CacheConfigModule::save()
78 {
79  KSaveIOConfig::setUseCache( ui.cbUseCache->isChecked() );
80  KSaveIOConfig::setMaxCacheSize( ui.sbMaxCacheSize->value() );
81 
82  if ( !ui.cbUseCache->isChecked() )
83  KSaveIOConfig::setCacheControl(KIO::CC_Refresh);
84  else if ( ui.rbVerifyCache->isChecked() )
85  KSaveIOConfig::setCacheControl(KIO::CC_Refresh);
86  else if ( ui.rbOfflineMode->isChecked() )
87  KSaveIOConfig::setCacheControl(KIO::CC_CacheOnly);
88  else if ( ui.rbCacheIfPossible->isChecked() )
89  KSaveIOConfig::setCacheControl(KIO::CC_Cache);
90 
91  KProtocolManager::reparseConfiguration();
92 
93  // Update running io-slaves...
94  KSaveIOConfig::updateRunningIOSlaves (this);
95 
96  emit changed( false );
97 }
