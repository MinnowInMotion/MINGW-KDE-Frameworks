58 static QSsl::SslProtocol qSslProtocolFromK(KTcpSocket::SslVersion sslVersion)
59 {
60  //### this lowlevel bit-banging is a little dangerous and a likely source of bugs
61  if (sslVersion == KTcpSocket::AnySslVersion) {
62  return QSsl::AnyProtocol;
63  }
64  //does it contain any valid protocol?
65  KTcpSocket::SslVersions validVersions(KTcpSocket::SslV2 | KTcpSocket::SslV3 | KTcpSocket::TlsV1);
66  validVersions |= KTcpSocket::TlsV1_1;
67  validVersions |= KTcpSocket::TlsV1_2;
68  validVersions |= KTcpSocket::TlsV1SslV3;
69  validVersions |= KTcpSocket::SecureProtocols;
70 
71  if (!(sslVersion & validVersions)) {
72  return QSsl::UnknownProtocol;
73  }
74 
75  switch (sslVersion) {
76  case KTcpSocket::SslV2:
77  return QSsl::SslV2;
78  case KTcpSocket::SslV3:
79  return QSsl::SslV3;
80  case KTcpSocket::TlsV1_0:
81  return QSsl::TlsV1_0;
82  case KTcpSocket::TlsV1_1:
83  return QSsl::TlsV1_1;
84  case KTcpSocket::TlsV1_2:
85  return QSsl::TlsV1_2;
86  case KTcpSocket::TlsV1SslV3:
87  return QSsl::TlsV1SslV3;
88  case KTcpSocket::SecureProtocols:
89  return QSsl::SecureProtocols;
90 
91  default:
92  //QSslSocket doesn't really take arbitrary combinations. It's one or all.
93  return QSsl::AnyProtocol;
94  }
95 }
