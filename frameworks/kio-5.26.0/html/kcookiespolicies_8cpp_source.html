47 // QUrl::fromAce/toAce don't accept a domain that starts with a '.', like we do here.
48 // So we use these wrappers.
49 QString tolerantFromAce (const QByteArray& _domain)
50 {
51  QByteArray domain (_domain);
52  const bool hasDot = domain.startsWith ('.');
53  if (hasDot)
54  domain.remove (0, 1);
55  QString ret = QUrl::fromAce(domain);
56  if (hasDot) {
57  ret.prepend ('.');
58  }
59  return ret;
60 }

169 void KCookiesPolicies::changePressed(QTreeWidgetItem* item, bool state)
170 {
171  Q_ASSERT(item);
172  const QString oldDomain(item->text (0));
173 
174  KCookiesPolicySelectionDlg pdlg (this);
175  pdlg.setWindowTitle (i18nc ("@title:window", "Change Cookie Policy"));
176  pdlg.setPolicy (KCookieAdvice::strToAdvice (mDomainPolicyMap.value(oldDomain)));
177  pdlg.setEnableHostEdit (state, oldDomain);
178 
179  if (pdlg.exec() && !pdlg.domain().isEmpty()) {
180  const QString newDomain = tolerantFromAce (pdlg.domain().toLatin1());
181  int advice = pdlg.advice();

183  mDomainPolicyMap[newDomain] = KCookieAdvice::adviceToStr(advice);
184  item->setText(0, newDomain);
185  item->setText(1, i18n (mDomainPolicyMap.value(newDomain)));
186  configChanged();
187  }

216 bool KCookiesPolicies::handleDuplicate (const QString& domain, int advice)
217 {
218  QTreeWidgetItem* item = mUi.policyTreeWidget->topLevelItem (0);
219  while (item != 0) {
220  if (item->text (0) == domain) {
221  const int res = KMessageBox::warningContinueCancel (this,
222  i18n ("<qt>A policy already exists for"
223  "<center><b>%1</b></center>"
224  "Do you want to replace it?</qt>", domain),
225  i18nc ("@title:window", "Duplicate Policy"),
226  KGuiItem (i18n ("Replace")));
227  if (res == KMessageBox::Continue) {
228  mDomainPolicyMap[domain] = KCookieAdvice::adviceToStr(advice);
229  item->setText (0, domain);
230  item->setText (1, i18n (mDomainPolicyMap.value(domain)));
231  configChanged();
232  return true;
233  } else
234  return true; // User Cancelled!!
235  }
236  item = mUi.policyTreeWidget->itemBelow (item);
237  }
238  return false;
239 }

278 void KCookiesPolicies::updateDomainList (const QStringList& domainConfig)
279 {

304 void KCookiesPolicies::load()
305 {
306  mSelectedItemsCount = 0;
307 
308  KConfig cfg (QStringLiteral("kcookiejarrc"));
309  KConfigGroup group = cfg.group ("Cookie Policy");
310 
311  bool enableCookies = group.readEntry ("Cookies", true);
312  mUi.cbEnableCookies->setChecked (enableCookies);
313  cookiesEnabled (enableCookies);
314 
315  // Warning: the default values are duplicated in kcookiejar.cpp
316  KCookieAdvice::Value advice = KCookieAdvice::strToAdvice (group.readEntry (
317  "CookieGlobalAdvice", "Accept"));
318  switch (advice) {
319  case KCookieAdvice::Accept:
320  mUi.rbPolicyAccept->setChecked (true);
321  break;
322  case KCookieAdvice::AcceptForSession:
323  mUi.rbPolicyAcceptForSession->setChecked (true);
324  break;
325  case KCookieAdvice::Reject:
326  mUi.rbPolicyReject->setChecked (true);
327  break;
328  case KCookieAdvice::Ask:
329  case KCookieAdvice::Dunno:
330  default:
331  mUi.rbPolicyAsk->setChecked (true);
332  }
333 
334  bool enable = group.readEntry ("RejectCrossDomainCookies", true);
335  mUi.cbRejectCrossDomainCookies->setChecked (enable);
336 
337  bool sessionCookies = group.readEntry ("AcceptSessionCookies", true);
338  mUi.cbAutoAcceptSessionCookies->setChecked (sessionCookies);
339  updateDomainList (group.readEntry ("CookieDomainAdvice", QStringList()));
340 
341  if (enableCookies) {
342  updateButtons();
343  }
344 }

346 void KCookiesPolicies::save()
347 {
348  KConfig cfg (QStringLiteral("kcookiejarrc"));
349  KConfigGroup group = cfg.group ("Cookie Policy");
350 
351  bool state = mUi.cbEnableCookies->isChecked();
352  group.writeEntry ("Cookies", state);
353  state = mUi.cbRejectCrossDomainCookies->isChecked();
354  group.writeEntry ("RejectCrossDomainCookies", state);
355  state = mUi.cbAutoAcceptSessionCookies->isChecked();
356  group.writeEntry ("AcceptSessionCookies", state);
357 
358  QString advice;
359  if (mUi.rbPolicyAccept->isChecked())
360  advice = KCookieAdvice::adviceToStr (KCookieAdvice::Accept);
361  else if (mUi.rbPolicyAcceptForSession->isChecked())
362  advice = KCookieAdvice::adviceToStr (KCookieAdvice::AcceptForSession);
363  else if (mUi.rbPolicyReject->isChecked())
364  advice = KCookieAdvice::adviceToStr (KCookieAdvice::Reject);
365  else
366  advice = KCookieAdvice::adviceToStr (KCookieAdvice::Ask);
367 
368  group.writeEntry ("CookieGlobalAdvice", advice);
369 
370  QStringList domainConfig;
371  QMapIterator<QString, const char*> it (mDomainPolicyMap);
372  while (it.hasNext()) {
373  it.next();
374  QString policy = tolerantToAce(it.key());
375  policy += QLatin1Char (':');
376  policy += QLatin1String (it.value());
377  domainConfig << policy;
378  }
379 
380  group.writeEntry ("CookieDomainAdvice", domainConfig);
381  group.sync();
382 
383  // Update the cookiejar...
384  if (!mUi.cbEnableCookies->isChecked()) {
385  QDBusInterface kded (QStringLiteral("org.kde.kcookiejar5"), QStringLiteral("/modules/kcookiejar"), QStringLiteral("org.kde.KCookieServer"), QDBusConnection::sessionBus());
386  kded.call (QStringLiteral("shutdown"));
387  } else {
388  QDBusInterface kded (QStringLiteral("org.kde.kcookiejar5"), QStringLiteral("/modules/kcookiejar"), QStringLiteral("org.kde.KCookieServer"), QDBusConnection::sessionBus());
389  QDBusReply<void> reply = kded.call (QStringLiteral("reloadPolicy"));
390  if (!reply.isValid())
391  KMessageBox::sorry (0, i18n ("Unable to communicate with the cookie handler service.\n"
392  "Any changes you made will not take effect until the service "
393  "is restarted."));
394  }
395 
396  // Force running io-slave to reload configurations...
397  KSaveIOConfig::updateRunningIOSlaves (this);
398  emit changed (false);
399 }

418 void KCookiesPolicies::splitDomainAdvice (const QString& cfg, QString& domain,
419  KCookieAdvice::Value& advice)
420 {
421  int sepPos = cfg.lastIndexOf (':');
422 
423  // Ignore any policy that does not contain a domain...
424  if (sepPos <= 0)
425  return;
426 
427  domain = cfg.left (sepPos);
428  advice = KCookieAdvice::strToAdvice (cfg.mid (sepPos + 1));
429 }
