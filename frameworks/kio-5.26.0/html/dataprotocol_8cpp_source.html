81 static int find(const QByteArray &buf, int begin, const char c1)
82 {
83  static const char comma = ',';
84  static const char semicolon = ';';
85  int pos = begin;
86  int size = buf.length();
87  while (pos < size) {
88  const char ch = buf[pos];
89  if (ch == comma || ch == semicolon || (c1 != '\0' && ch == c1)) {
90  break;
91  }
92  pos++;
93  }/*wend*/
94  return pos;
95 }

135 static QString parseQuotedString(const QString &buf, int &pos)
136 {
137  int size = buf.length();
138  QString res;
139  res.reserve(size); // can't be larger than buf
140  pos++; // jump over leading quote
141  bool escaped = false; // if true means next character is literal
142  bool parsing = true; // true as long as end quote not found
143  while (parsing && pos < size) {
144  const QChar ch = buf[pos++];
145  if (escaped) {
146  res += ch;
147  escaped = false;
148  } else {
149  switch (ch.unicode()) {
150  case '"': parsing = false; break;
151  case '\\': escaped = true; break;
152  default: res += ch; break;
153  }/*end switch*/
154  }/*end if*/
155  }/*wend*/
156  res.squeeze();
157  return res;
158 }

165 static DataHeader parseDataHeader(const QUrl &url, const bool mimeOnly)
166 {
167  DataHeader header_info;
168 
169  // initialize header info members
170  header_info.mime_type = QStringLiteral("text/plain");
171  header_info.attributes.insert(QStringLiteral("charset"), QStringLiteral("us-ascii"));
172  header_info.is_base64 = false;
173 
174  // decode url and save it
175  const QByteArray &raw_url = header_info.url = QByteArray::fromPercentEncoding(url.path(QUrl::FullyEncoded).toLatin1());
176  const int raw_url_len = raw_url.length();
177 
178  header_info.data_offset = 0;
179 
180  // read mime type
181  if (raw_url_len == 0) {
182  return header_info;
183  }
184  const QString mime_type = extract(raw_url, header_info.data_offset).trimmed();
185  if (!mime_type.isEmpty()) {
186  header_info.mime_type = mime_type;
187  }
188  if (mimeOnly) {
189  return header_info;
190  }
191 
192  if (header_info.data_offset >= raw_url_len) {
193  return header_info;
194  }
195  // jump over delimiter token and return if data reached
196  if (raw_url[header_info.data_offset++] == QLatin1Char(',')) {
197  return header_info;
198  }
199 
200  // read all attributes and store them
201  bool data_begin_reached = false;
202  while (!data_begin_reached && header_info.data_offset < raw_url_len) {
203  // read attribute
204  const QString attribute = extract(raw_url, header_info.data_offset, '=').trimmed();
205  if (header_info.data_offset >= raw_url_len
206  || raw_url[header_info.data_offset] != QLatin1Char('=')) {
207  // no assigment, must be base64 option
208  if (attribute == QLatin1String("base64")) {
209  header_info.is_base64 = true;
210  }
211  } else {
212  header_info.data_offset++; // jump over '=' token
213 
214  // read value
215  ignoreWS(raw_url, header_info.data_offset);
216  if (header_info.data_offset >= raw_url_len) {
217  return header_info;
218  }
219 
220  QString value;
221  if (raw_url[header_info.data_offset] == QLatin1Char('"')) {
222  value = parseQuotedString(raw_url, header_info.data_offset);
223  ignoreWS(raw_url, header_info.data_offset);
224  } else {
225  value = extract(raw_url, header_info.data_offset).trimmed();
226  }
227 
228  // add attribute to map
229  header_info.attributes[attribute.toLower()] = value;
230 
231  }/*end if*/
232  if (header_info.data_offset < raw_url_len
233  && raw_url[header_info.data_offset] == QLatin1Char(',')) {
234  data_begin_reached = true;
235  }
236  header_info.data_offset++; // jump over separator token
237  }/*wend*/
238 
239  return header_info;
240 }

262 void DataProtocol::get(const QUrl &url)
263 {
264  ref();
265  //qDebug() << this;
266 
267  const DataHeader hdr = parseDataHeader(url, false);
268 
269  const int size = hdr.url.length();
270  const int data_ofs = qMin(hdr.data_offset, size);
271  // FIXME: string is copied, would be nice if we could have a reference only
272  const QByteArray url_data = hdr.url.mid(data_ofs);
273  QByteArray outData;
274 
275  if (hdr.is_base64) {
276  // base64 stuff is expected to contain the correct charset, so we just
277  // decode it and pass it to the receiver
278  outData = QByteArray::fromBase64(url_data);
279  } else {
280  QTextCodec *codec = QTextCodec::codecForName(hdr.attributes[QStringLiteral("charset")].toLatin1());
281  if (codec != 0) {
282  outData = codec->toUnicode(url_data).toUtf8();
283  } else {
284  outData = url_data;
285  }/*end if*/
286  }/*end if*/
287 
288  //qDebug() << "emit mimeType@"<<this;
289  mimeType(hdr.mime_type);
290  //qDebug() << "emit totalSize@"<<this;
291  totalSize(outData.size());
292 
293  //qDebug() << "emit setMetaData@"<<this;
294 #if defined(DATAKIOSLAVE)
295  MetaData::ConstIterator it;
296  for (it = hdr.attributes.constBegin(); it != hdr.attributes.constEnd(); ++it) {
297  setMetaData(it.key(), it.value());
298  }/*next it*/
299 #else
300  setAllMetaData(hdr.attributes);
301 #endif
302 
303  //qDebug() << "emit sendMetaData@"<<this;
304  sendMetaData();
305 //qDebug() << "(1) queue size " << dispatchQueue.size();
306  // empiric studies have shown that this shouldn't be queued & dispatched
307  data(outData);
308 //qDebug() << "(2) queue size " << dispatchQueue.size();
309  DISPATCH(data(QByteArray()));
310 //qDebug() << "(3) queue size " << dispatchQueue.size();
311  DISPATCH(finished());
312 //qDebug() << "(4) queue size " << dispatchQueue.size();
313  deref();
314 }
