60 KPreviewWidgetBase *KFileMetaPreview::findExistingProvider(const QString &mimeType, const QMimeType &mimeInfo) const
61 {
62  KPreviewWidgetBase *provider = m_previewProviders.value(mimeType);
63  if (provider) {
64  return provider;
65  }
66 
67  if (mimeInfo.isValid()) {
68  // check mime type inheritance
69  const QStringList parentMimeTypes = mimeInfo.allAncestors();
70  Q_FOREACH (const QString &parentMimeType, parentMimeTypes) {
71  provider = m_previewProviders.value(parentMimeType);
72  if (provider) {
73  return provider;
74  }
75  }
76  }
77 
78  // ### mimetype may be image/* for example, try that
79  const int index = mimeType.indexOf('/');
80  if (index > 0) {
81  provider = m_previewProviders.value(mimeType.left(index + 1) + '*');
82  if (provider) {
83  return provider;
84  }
85  }
86 
87  return 0;
88 }

90 KPreviewWidgetBase *KFileMetaPreview::previewProviderFor(const QString &mimeType)
91 {
92  QMimeDatabase db;
93  QMimeType mimeInfo = db.mimeTypeForName(mimeType);
94 
95  // qDebug("### looking for: %s", mimeType.toLatin1().constData());
96  // often the first highlighted item, where we can be sure, there is no plugin
97  // (this "folders reflect icons" is a konq-specific thing, right?)
98  if (mimeInfo.inherits(QStringLiteral("inode/directory"))) {
99  return 0L;
100  }
101 
102  KPreviewWidgetBase *provider = findExistingProvider(mimeType, mimeInfo);
103  if (provider) {
104  return provider;
105  }
106 
107 //qDebug("#### didn't find anything for: %s", mimeType.toLatin1().constData());
108 
109  if (s_tryAudioPreview &&
110  !mimeType.startsWith(QLatin1String("text/")) &&
111  !mimeType.startsWith(QLatin1String("image/"))) {
112  if (!haveAudioPreview) {
113  KPreviewWidgetBase *audioPreview = createAudioPreview(m_stack);
114  if (audioPreview) {
115  haveAudioPreview = true;
116  (void) m_stack->addWidget(audioPreview);
117  const QStringList mimeTypes = audioPreview->supportedMimeTypes();
118  QStringList::ConstIterator it = mimeTypes.begin();
119  for (; it != mimeTypes.end(); ++it) {
120  // only add non already handled mimetypes
121  if (m_previewProviders.constFind(*it) == m_previewProviders.constEnd()) {
122  m_previewProviders.insert(*it, audioPreview);
123  }
124  }
125  }
126  }
127  }
128 
129  // with the new mimetypes from the audio-preview, try again
130  provider = findExistingProvider(mimeType, mimeInfo);
131  if (provider) {
132  return provider;
133  }
134 
135  // The logic in this code duplicates the logic in PreviewJob.
136  // But why do we need multiple KPreviewWidgetBase instances anyway?
137 
138  return 0L;
139 }

141 void KFileMetaPreview::showPreview(const QUrl &url)
142 {
143  QMimeDatabase db;
144  QMimeType mt = db.mimeTypeForUrl(url);
145  KPreviewWidgetBase *provider = previewProviderFor(mt.name());
146  if (provider) {
147  if (provider != m_stack->currentWidget()) { // stop the previous preview
148  clearPreview();
149  }
150 
151  m_stack->setEnabled(true);
152  m_stack->setCurrentWidget(provider);
153  provider->showPreview(url);
154  } else {
155  clearPreview();
156  m_stack->setEnabled(false);
157  }
158 }
