86 bool ForwardingSlaveBasePrivate::internalRewriteUrl(const QUrl &url, QUrl &newURL)
87 {
88  bool result = true;

101 void ForwardingSlaveBase::prepareUDSEntry(KIO::UDSEntry &entry,
102  bool listing) const
103 {
104  //qDebug() << "listing==" << listing;
105 
106  const QString name = entry.stringValue(KIO::UDSEntry::UDS_NAME);
107  QString mimetype = entry.stringValue(KIO::UDSEntry::UDS_MIME_TYPE);
108  QUrl url;
109  const QString urlStr = entry.stringValue(KIO::UDSEntry::UDS_URL);
110  const bool url_found = !urlStr.isEmpty();
111  if (url_found) {
112  url = QUrl(urlStr);
113  QUrl new_url(d->m_requestedURL);
114  if (listing) {
115  new_url.setPath(new_url.path() + '/' + url.fileName());
116  }
117  // ## Didn't find a way to use an iterator instead of re-doing a key lookup
118  entry.insert(KIO::UDSEntry::UDS_URL, new_url.toString());
119  //qDebug() << "URL =" << url;
120  //qDebug() << "New URL =" << new_url;
121  }
122 
123  if (mimetype.isEmpty()) {
124  QUrl new_url(d->m_processedURL);
125  if (url_found && listing) {
126  new_url.setPath(new_url.path() + '/' + url.fileName());
127  } else if (listing) {
128  new_url.setPath(new_url.path() + '/' + name);
129  }
130 
131  QMimeDatabase db;
132  mimetype = db.mimeTypeForUrl(new_url).name();
133 
134  entry.insert(KIO::UDSEntry::UDS_MIME_TYPE, mimetype);
135 
136  //qDebug() << "New Mimetype = " << mimetype;
137  }
138 
139  if (d->m_processedURL.isLocalFile()) {
140  QUrl new_url(d->m_processedURL);
141  if (listing) {
142  new_url.setPath(new_url.path() + '/' + name);
143  }
144 
145  entry.insert(KIO::UDSEntry::UDS_LOCAL_PATH, new_url.toLocalFile());
146  }
147 }

415 void ForwardingSlaveBasePrivate::_k_slotResult(KJob *job)
416 {
417  if (job->error() != 0) {
418  q->error(job->error(), job->errorText());
419  } else {
420  KIO::StatJob *stat_job = qobject_cast<KIO::StatJob *>(job);
421  if (stat_job != 0L) {
422  KIO::UDSEntry entry = stat_job->statResult();
423  q->prepareUDSEntry(entry);
424  q->statEntry(entry);
425  }
426  q->finished();
427  }
428 
429  eventLoop.exit();
430 }
