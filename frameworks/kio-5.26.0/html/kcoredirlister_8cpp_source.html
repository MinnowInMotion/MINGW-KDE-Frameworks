100 bool KCoreDirListerCache::listDir(KCoreDirLister *lister, const QUrl &_u,
101  bool _keep, bool _reload)
102 {
103  QUrl _url(_u);
104  _url.setPath(QDir::cleanPath(_url.path())); // kill consecutive slashes
105 
106  if (!_url.host().isEmpty() && KProtocolInfo::protocolClass(_url.scheme()) == QLatin1String(":local")
107  && _url.scheme() != QLatin1String("file")) {
108  // ":local" protocols ignore the hostname, so strip it out preventively - #160057
109  // kio_file is special cased since it does honor the hostname (by redirecting to e.g. smb)
110  _url.setHost(QString());
111  if (_keep == false) {
112  emit lister->redirection(_url);
113  }
114  }
115 
116  // like this we don't have to worry about trailing slashes any further
117  _url = _url.adjusted(QUrl::StripTrailingSlash);
118 
119  const QString urlStr = _url.toString();
120 
121  QString resolved;
122  if (_url.isLocalFile()) {
123  // Resolve symlinks (#213799)
124  const QString local = _url.toLocalFile();
125  resolved = QFileInfo(local).canonicalFilePath();
126  if (local != resolved) {
127  canonicalUrls[QUrl::fromLocalFile(resolved)].append(_url);
128  }
129  // TODO: remove entry from canonicalUrls again in forgetDirs
130  // Note: this is why we use a QStringList value in there rather than a QSet:
131  // we can just remove one entry and not have to worry about other dirlisters
132  // (the non-unicity of the stringlist gives us the refcounting, basically).
133  }
134 
135  if (!validUrl(lister, _url)) {
136  qCDebug(KIO_CORE_DIRLISTER) << lister << "url=" << _url << "not a valid url";
137  return false;
138  }
139 
140  qCDebug(KIO_CORE_DIRLISTER) << lister << "url=" << _url << "keep=" << _keep << "reload=" << _reload;
141 #ifdef DEBUG_CACHE
142  printDebug();
143 #endif
144 
145  if (!_keep) {
146  // stop any running jobs for lister
147  stop(lister, true /*silent*/);
148 
149  // clear our internal list for lister
150  forgetDirs(lister);
151 
152  lister->d->rootFileItem = KFileItem();
153  } else if (lister->d->lstDirs.contains(_url)) {
154  // stop the job listing _url for this lister
155  stopListingUrl(lister, _url, true /*silent*/);
156 
157  // remove the _url as well, it will be added in a couple of lines again!
158  // forgetDirs with three args does not do this
159  // TODO: think about moving this into forgetDirs
160  lister->d->lstDirs.removeAll(_url);
161 
162  // clear _url for lister
163  forgetDirs(lister, _url, true);
164 
165  if (lister->d->url == _url) {
166  lister->d->rootFileItem = KFileItem();
167  }
168  }
169 
170  lister->d->complete = false;
171 
172  lister->d->lstDirs.append(_url);
173 
174  if (lister->d->url.isEmpty() || !_keep) { // set toplevel URL only if not set yet
175  lister->d->url = _url;
176  }
177 
178  DirItem *itemU = itemsInUse.value(urlStr);
179 
180  KCoreDirListerCacheDirectoryData &dirData = directoryData[urlStr]; // find or insert
181 
182  if (dirData.listersCurrentlyListing.isEmpty()) {
183  // if there is an update running for _url already we get into
184  // the following case - it will just be restarted by updateDirectory().
185 
186  dirData.listersCurrentlyListing.append(lister);
187 
188  DirItem *itemFromCache = 0;
189  if (itemU || (!_reload && (itemFromCache = itemsCached.take(urlStr)))) {
190  if (itemU) {
191  qCDebug(KIO_CORE_DIRLISTER) << "Entry already in use:" << _url;
192  // if _reload is set, then we'll emit cached items and then updateDirectory.
193  } else {
194  qCDebug(KIO_CORE_DIRLISTER) << "Entry in cache:" << _url;
195  itemsInUse.insert(urlStr, itemFromCache);
196  itemU = itemFromCache;
197  }
198  if (lister->d->autoUpdate) {
199  itemU->incAutoUpdate();
200  }
201  if (itemFromCache && itemFromCache->watchedWhileInCache) {
202  itemFromCache->watchedWhileInCache = false;;
203  itemFromCache->decAutoUpdate();
204  }
205 
206  emit lister->started(_url);
207 
208  // List items from the cache in a delayed manner, just like things would happen
209  // if we were not using the cache.
210  new KCoreDirLister::Private::CachedItemsJob(lister, _url, _reload);
211 
212  } else {
213  // dir not in cache or _reload is true
214  if (_reload) {
215  qCDebug(KIO_CORE_DIRLISTER) << "Reloading directory:" << _url;
216  itemsCached.remove(urlStr);
217  } else {
218  qCDebug(KIO_CORE_DIRLISTER) << "Listing directory:" << _url;
219  }
220 
221  itemU = new DirItem(_url, resolved);
222  itemsInUse.insert(urlStr, itemU);
223  if (lister->d->autoUpdate) {
224  itemU->incAutoUpdate();
225  }
226 
227 // // we have a limit of MAX_JOBS_PER_LISTER concurrently running jobs
228 // if ( lister->d->numJobs() >= MAX_JOBS_PER_LISTER )
229 // {
230 // pendingUpdates.insert( _url );
231 // }
232 // else
233  {
234  KIO::ListJob *job = KIO::listDir(_url, KIO::HideProgressInfo);
235  runningListJobs.insert(job, KIO::UDSEntryList());
236 
237  lister->jobStarted(job);
238  lister->d->connectJob(job);
239 
240  connect(job, SIGNAL(entries(KIO::Job*,KIO::UDSEntryList)),
241  this, SLOT(slotEntries(KIO::Job*,KIO::UDSEntryList)));
242  connect(job, SIGNAL(result(KJob*)),
243  this, SLOT(slotResult(KJob*)));
244  connect(job, SIGNAL(redirection(KIO::Job*,QUrl)),
245  this, SLOT(slotRedirection(KIO::Job*,QUrl)));
246 
247  emit lister->started(_url);
248  }
249  qCDebug(KIO_CORE_DIRLISTER) << "Entry now being listed by" << dirData.listersCurrentlyListing;
250  }
251  } else {
252 
253  qCDebug(KIO_CORE_DIRLISTER) << "Entry currently being listed:" << _url << "by" << dirData.listersCurrentlyListing;
254 #ifdef DEBUG_CACHE
255  printDebug();
256 #endif
257 
258  emit lister->started(_url);
259 
260  // Maybe listersCurrentlyListing/listersCurrentlyHolding should be QSets?
261  Q_ASSERT(!dirData.listersCurrentlyListing.contains(lister));
262  dirData.listersCurrentlyListing.append(lister);
263 
264  KIO::ListJob *job = jobForUrl(urlStr);
265  // job will be 0 if we were listing from cache rather than listing from a kio job.
266  if (job) {
267  lister->jobStarted(job);
268  lister->d->connectJob(job);
269  }
270  Q_ASSERT(itemU);
271 
272  // List existing items in a delayed manner, just like things would happen
273  // if we were not using the cache.
274  qCDebug(KIO_CORE_DIRLISTER) << "Listing" << itemU->lstItems.count() << "cached items soon";
275  KCoreDirLister::Private::CachedItemsJob* cachedItemsJob = new KCoreDirLister::Private::CachedItemsJob(lister, _url, _reload);
276  if (job) {
277  // The ListJob will take care of emitting completed.
278  // ### If it finishes before the CachedItemsJob, then we'll emit cached items after completed(), not sure how bad this is.
279  cachedItemsJob->setEmitCompleted(false);
280  }
281 
282 #ifdef DEBUG_CACHE
283  printDebug();
284 #endif
285  }
286 
287  return true;

336 void KCoreDirListerCache::emitItemsFromCache(KCoreDirLister::Private::CachedItemsJob *cachedItemsJob, KCoreDirLister *lister, const QUrl &_url, bool _reload, bool _emitCompleted)
337 {
338  const QString urlStr = _url.toString();
339  KCoreDirLister::Private *kdl = lister->d;
340  kdl->complete = false;
341 
342  DirItem *itemU = kDirListerCache()->itemsInUse.value(urlStr);
343  if (!itemU) {
344  qWarning() << "Can't find item for directory" << urlStr << "anymore";
345  } else {
346  const NonMovableFileItemList items = itemU->lstItems;
347  const KFileItem rootItem = itemU->rootItem;
348  _reload = _reload || !itemU->complete;
349 
350  if (kdl->rootFileItem.isNull() && !rootItem.isNull() && kdl->url == _url) {
351  kdl->rootFileItem = rootItem;
352  }
353  if (!items.isEmpty()) {
354  qCDebug(KIO_CORE_DIRLISTER) << "emitting" << items.count() << "for lister" << lister;
355  kdl->addNewItems(_url, items);
356  kdl->emitItems();
357  }
358  }
359 
360  forgetCachedItemsJob(cachedItemsJob, lister, _url);
361 
362  // Emit completed, unless we were told not to,
363  // or if listDir() was called while another directory listing for this dir was happening,
364  // so we "joined" it. We detect that using jobForUrl to ensure it's a real ListJob,
365  // not just a lister-specific CachedItemsJob (which wouldn't emit completed for us).
366  if (_emitCompleted) {
367 
368  kdl->complete = true;
369  emit lister->completed(_url);
370  emit lister->completed();
371 
372  if (_reload) {
373  updateDirectory(_url);
374  }
375  }
376 }

379 {
380  // Modifications to data structures only below this point;
381  // so that addNewItems is called with a consistent state
382

394  dirData.listersCurrentlyListing.removeAll(lister);
395  } else {
396  qCDebug(KIO_CORE_DIRLISTER) << "Still having a listjob" << listJob << ", so not moving to currently-holding.";
397  }

481 {
482  // Old idea: if it's an update job, let's just leave the job running.
483  // After all, update jobs do run for "listersCurrentlyHolding",

486  // However it could be a long-running non-local job (e.g. filenamesearch), which

490

536 static bool manually_mounted(const QString &path, const KMountPoint::List &possibleMountPoints)
537 {
538  KMountPoint::Ptr mp = possibleMountPoints.findByPath(path);
539  if (!mp) { // not listed in fstab -> yes, manually mounted
540  if (possibleMountPoints.isEmpty()) { // no fstab at all -> don't assume anything
541  return false;
542  }
543  return true;
544  }
545  const bool supermount = mp->mountType() == QLatin1String("supermount");
546  if (supermount) {
547  return true;
548  }
549  // noauto -> manually mounted. Otherwise, mounted at boot time, won't be unmounted any time soon hopefully.
550  return mp->mountOptions().contains(QStringLiteral("noauto"));
551 }

648  }

651  if (item && insertIntoCache) {
