346 void FileUndoManager::undo()
347 {
348  // Make a copy of the command to undo before slotPop() pops it.
349  UndoCommand cmd = d->m_commands.last();
350  assert(cmd.m_valid);
351  d->m_current = cmd;
352  const CommandType commandType = cmd.m_type;
353 
354  BasicOperation::Stack &opStack = d->m_current.m_opStack;
355  // Note that opStack is empty for simple operations like Mkdir.
356 
357  // Let's first ask for confirmation if we need to delete any file (#99898)
358  QList<QUrl> itemsToDelete;
359  BasicOperation::Stack::Iterator it = opStack.begin();
360  for (; it != opStack.end(); ++it) {
361  BasicOperation::Type type = (*it).m_type;
362  if (type == BasicOperation::File && commandType == FileUndoManager::Copy) {
363  itemsToDelete.append((*it).m_dst);
364  } else if (commandType == FileUndoManager::Mkpath) {
365  itemsToDelete.append((*it).m_dst);
366  }
367  }
368  if (commandType == FileUndoManager::Mkdir || commandType == FileUndoManager::Put) {
369  itemsToDelete.append(d->m_current.m_dst);
370  }
371  if (!itemsToDelete.isEmpty()) {
372  if (!d->m_uiInterface->confirmDeletion(itemsToDelete)) {
373  return;
374  }
375  }
376 
377  d->slotPop();
378  d->slotLock();
379 
380  d->m_dirCleanupStack.clear();
381  d->m_dirStack.clear();
382  d->m_dirsToUpdate.clear();
383 
384  d->m_undoState = MOVINGFILES;
385 
386  // Let's have a look at the basic operations we need to undo.
387  // While we're at it, collect all links that should be deleted.
388 
389  it = opStack.begin();
390  while (it != opStack.end()) { // don't cache end() here, erase modifies it
391  bool removeBasicOperation = false;
392  BasicOperation::Type type = (*it).m_type;
393  if (type == BasicOperation::Directory && !(*it).m_renamed) {
394  // If any directory has to be created/deleted, we'll start with that
395  d->m_undoState = MAKINGDIRS;
396  // Collect all the dirs that have to be created in case of a move undo.
397  if (d->m_current.isMoveCommand()) {
398  d->m_dirStack.push((*it).m_src);
399  }
400  // Collect all dirs that have to be deleted
401  // from the destination in both cases (copy and move).
402  d->m_dirCleanupStack.prepend((*it).m_dst);
403  removeBasicOperation = true;
404  } else if (type == BasicOperation::Link) {
405  d->m_fileCleanupStack.prepend((*it).m_dst);
406 
407  removeBasicOperation = !d->m_current.isMoveCommand();
408  }
409 
410  if (removeBasicOperation) {
411  it = opStack.erase(it);
412  } else {
413  ++it;
414  }
415  }
416 
417  if (commandType == FileUndoManager::Put) {
418  d->m_fileCleanupStack.append(d->m_current.m_dst);
419  }
420 
421  //qDebug() << "starting with" << undoStateToString(d->m_undoState);
422  d->m_undoJob = new UndoJob(d->m_uiInterface->showProgressInfo());
423  QMetaObject::invokeMethod(d, "undoStep", Qt::QueuedConnection);
424 }

427 {

445 void FileUndoManagerPrivate::slotResult(KJob *job)
446 {
447  m_currentJob = 0;
448  if (job->error()) {
449  m_uiInterface->jobError(static_cast<KIO::Job *>(job));
450  delete m_undoJob;
451  stopUndo(false);
452  } else if (m_undoState == STATINGFILE) {
453  BasicOperation op = m_current.m_opStack.last();
454  //qDebug() << "stat result for " << op.m_dst;
455  KIO::StatJob *statJob = static_cast<KIO::StatJob *>(job);
456  const QDateTime mtime = QDateTime::fromMSecsSinceEpoch(1000 * statJob->statResult().numberValue(KIO::UDSEntry::UDS_MODIFICATION_TIME, -1));
457  if (mtime != op.m_mtime) {
458  //qDebug() << op.m_dst << " was modified after being copied!";
459  QDateTime srcTime = op.m_mtime.toLocalTime();
460  QDateTime destTime = mtime.toLocalTime();
461  if (!m_uiInterface->copiedFileWasModified(op.m_src, op.m_dst, srcTime, destTime)) {
462  stopUndo(false);
463  }
464  }
465  }
466 
467  undoStep();
468 }

477 void FileUndoManagerPrivate::undoStep()
478 {
479  m_currentJob = 0;
480 
481  if (m_undoState == MAKINGDIRS) {
482  stepMakingDirectories();
483  }
484 
485  if (m_undoState == MOVINGFILES || m_undoState == STATINGFILE) {
486  stepMovingFiles();
487  }
488 
489  if (m_undoState == REMOVINGLINKS) {
490  stepRemovingLinks();
491  }
492 
493  if (m_undoState == REMOVINGDIRS) {
494  stepRemovingDirectories();
495  }
496 
497  if (m_currentJob) {
498  if (m_uiInterface) {
499  KJobWidgets::setWindow(m_currentJob, m_uiInterface->parentWidget());
500  }
501  QObject::connect(m_currentJob, SIGNAL(result(KJob*)),
502  this, SLOT(slotResult(KJob*)));
503  }
504 }

506 void FileUndoManagerPrivate::stepMakingDirectories()

518 // Misnamed method: It moves files back, but it also
519 // renames directories back, recreates symlinks,
520 // deletes copied files, and restores trashed files.
521 void FileUndoManagerPrivate::stepMovingFiles()
522 {
523  if (!m_current.m_opStack.isEmpty()) {
524  BasicOperation op = m_current.m_opStack.last();
525  BasicOperation::Type type = op.m_type;
526 
527  assert(op.m_valid);
528  if (type == BasicOperation::Directory) {
529  if (op.m_renamed) {
530  //qDebug() << "rename" << op.m_dst << op.m_src;
531  m_currentJob = KIO::rename(op.m_dst, op.m_src, KIO::HideProgressInfo);
532  m_undoJob->emitMoving(op.m_dst, op.m_src);
533  } else {
534  assert(0); // this should not happen!
535  }
536  } else if (type == BasicOperation::Link) {
537  //qDebug() << "symlink" << op.m_target << op.m_src;
538  m_currentJob = KIO::symlink(op.m_target, op.m_src, KIO::Overwrite | KIO::HideProgressInfo);
539  } else if (m_current.m_type == FileUndoManager::Copy) {
540  if (m_undoState == MOVINGFILES) { // dest not stat'ed yet
541  // Before we delete op.m_dst, let's check if it was modified (#20532)
542  //qDebug() << "stat" << op.m_dst;
543  m_currentJob = KIO::stat(op.m_dst, KIO::HideProgressInfo);
544  m_undoState = STATINGFILE; // temporarily
545  return; // no pop() yet, we'll finish the work in slotResult
546  } else { // dest was stat'ed, and the deletion was approved in slotResult
547  m_currentJob = KIO::file_delete(op.m_dst, KIO::HideProgressInfo);
548  m_undoJob->emitDeleting(op.m_dst);
549  m_undoState = MOVINGFILES;
550  }
551  } else if (m_current.isMoveCommand()
552  || m_current.m_type == FileUndoManager::Trash) {
553  //qDebug() << "file_move" << op.m_dst << op.m_src;
554  m_currentJob = KIO::file_move(op.m_dst, op.m_src, -1, KIO::Overwrite | KIO::HideProgressInfo);
555  m_currentJob->uiDelegateExtension()->createClipboardUpdater(m_currentJob, JobUiDelegateExtension::UpdateContent);
556  m_undoJob->emitMoving(op.m_dst, op.m_src);
557  }
558 
559  m_current.m_opStack.removeLast();
560  // The above KIO jobs are lowlevel, they don't trigger KDirNotify notification
561  // So we need to do it ourselves (but schedule it to the end of the undo, to compress them)
562  QUrl url = op.m_dst.adjusted(QUrl::RemoveFilename | QUrl::StripTrailingSlash);
563  addDirToUpdate(url);
564 
565  url = op.m_src.adjusted(QUrl::RemoveFilename | QUrl::StripTrailingSlash);
566  addDirToUpdate(url);
567  } else {
568  m_undoState = REMOVINGLINKS;
569  }
570 }

572 void FileUndoManagerPrivate::stepRemovingLinks()
573 {
574  //qDebug() << "REMOVINGLINKS";
575  if (!m_fileCleanupStack.isEmpty()) {
576  QUrl file = m_fileCleanupStack.pop();
577  //qDebug() << "file_delete" << file;
578  m_currentJob = KIO::file_delete(file, KIO::HideProgressInfo);
579  m_undoJob->emitDeleting(file);
580 
581  QUrl url = file.adjusted(QUrl::RemoveFilename | QUrl::StripTrailingSlash);
582  addDirToUpdate(url);
583  } else {
584  m_undoState = REMOVINGDIRS;
585 
586  if (m_dirCleanupStack.isEmpty() && m_current.m_type == FileUndoManager::Mkdir) {
587  m_dirCleanupStack << m_current.m_dst;
588  }
589  }
590 }

592 void FileUndoManagerPrivate::stepRemovingDirectories()
593 {
594  if (!m_dirCleanupStack.isEmpty()) {
595  QUrl dir = m_dirCleanupStack.pop();
596  //qDebug() << "rmdir" << dir;
597  m_currentJob = KIO::rmdir(dir);
598  m_undoJob->emitDeleting(dir);
599  addDirToUpdate(dir);
600  } else {
601  m_current.m_valid = false;
602  m_currentJob = 0;
603  if (m_undoJob) {
604  //qDebug() << "deleting undojob";
605  m_undoJob->emitResult();
606  m_undoJob = 0;
607  }
608  QList<QUrl>::ConstIterator it = m_dirsToUpdate.constBegin();
609  for (; it != m_dirsToUpdate.constEnd(); ++it) {
610  //qDebug() << "Notifying FilesAdded for " << *it;
611  org::kde::KDirNotify::emitFilesAdded((*it));
612  }
613  emit q->undoJobFinished();
614  slotUnlock();
615  }
