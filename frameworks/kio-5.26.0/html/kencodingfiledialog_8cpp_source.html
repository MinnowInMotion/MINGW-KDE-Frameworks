53 KEncodingFileDialog::KEncodingFileDialog(const QUrl &startDir,
54  const QString &encoding,
55  const QString &filter,
56  const QString &caption,
57  QFileDialog::AcceptMode type,
58  QWidget *parent)
59  : QDialog(parent, Qt::Dialog)
60  , d(new KEncodingFileDialogPrivate)
61 {
62  d->w = new KFileWidget(startDir, this);
63  d->w->setFilter(filter);
64  if (type == QFileDialog::AcceptOpen) {
65  d->w->setOperationMode(KFileWidget::Opening);
66  } else {
67  d->w->setOperationMode(KFileWidget::Saving);
68  }
69 
70  setWindowTitle(caption);
71  //ops->clearHistory();
72 
73  KWindowConfig::restoreWindowSize(windowHandle(), d->cfgGroup);
74 
75  QBoxLayout *mainLayout = new QVBoxLayout;
76  setLayout(mainLayout);
77  mainLayout->addWidget(d->w);
78 
79  d->w->okButton()->show();
80  connect(d->w->okButton(), SIGNAL(clicked()), SLOT(slotOk()));
81  d->w->cancelButton()->show();
82  connect(d->w->cancelButton(), SIGNAL(clicked()), SLOT(slotCancel()));
83  connect(d->w, SIGNAL(accepted()), SLOT(accept()));
84 
85  d->encoding = new KComboBox(this);
86  d->w->setCustomWidget(i18n("Encoding:"), d->encoding);
87 
88  d->encoding->clear();
89  QString sEncoding = encoding;
90  QString systemEncoding = QTextCodec::codecForLocale()->name();
91  if (sEncoding.isEmpty() || sEncoding == QLatin1String("System")) {
92  sEncoding = systemEncoding;
93  }
94 
95  const QStringList encodings(KCharsets::charsets()->availableEncodingNames());
96  int insert = 0, system = 0;
97  bool foundRequested = false;
98  foreach (const QString &encoding, encodings) {
99  bool found = false;
100  QTextCodec *codecForEnc = KCharsets::charsets()->codecForName(encoding, found);
101 
102  if (found) {
103  d->encoding->addItem(encoding);
104  if ((codecForEnc->name() == sEncoding) || (encoding == sEncoding)) {
105  d->encoding->setCurrentIndex(insert);
106  foundRequested = true;
107  }
108 
109  if ((codecForEnc->name() == systemEncoding) || (encoding == systemEncoding)) {
110  system = insert;
111  }
112  insert++;
113  }
114  }
115 
116  if (!foundRequested) {
117  d->encoding->setCurrentIndex(system);
118  }
119 }
