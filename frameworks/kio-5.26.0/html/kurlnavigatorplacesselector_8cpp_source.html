68 void KUrlNavigatorPlacesSelector::updateMenu()
69 {
70  m_placesMenu->clear();
71 
72  updateSelection(m_selectedUrl);
73  const int rowCount = m_placesModel->rowCount();
74  for (int i = 0; i < rowCount; ++i) {
75  QModelIndex index = m_placesModel->index(i, 0);
76  QAction *action = new QAction(m_placesModel->icon(index),
77  m_placesModel->text(index),
78  m_placesMenu);
79  m_placesMenu->addAction(action);
80  action->setData(i);
81  if (i == m_selectedItem) {
82  setIcon(m_placesModel->icon(index));
83  }
84  }
85 
86  updateTeardownAction();
87 }

89 void KUrlNavigatorPlacesSelector::updateTeardownAction()
90 {
91  const int rowCount = m_placesModel->rowCount();
92  if (m_placesMenu->actions().size() == rowCount + 2) {
93  // remove teardown action
94  QAction *action = m_placesMenu->actions().at(rowCount + 1);
95  m_placesMenu->removeAction(action);
96  delete action;
97 
98  // remove separator
99  action = m_placesMenu->actions().at(rowCount);
100  m_placesMenu->removeAction(action);
101  delete action;
102  }
103 
104  const QModelIndex index = m_placesModel->index(m_selectedItem, 0);
105  QAction *teardown = m_placesModel->teardownActionForIndex(index);
106  if (teardown != 0) {
107  teardown->setParent(m_placesMenu);
108  teardown->setData("teardownAction");
109 
110  m_placesMenu->addSeparator();
111  m_placesMenu->addAction(teardown);
112  }
113 }

115 void KUrlNavigatorPlacesSelector::updateSelection(const QUrl &url)
116 {
117  const QModelIndex index = m_placesModel->closestItem(url);
118  if (index.isValid()) {
119  m_selectedItem = index.row();
120  m_selectedUrl = url;
121  setIcon(m_placesModel->icon(index));
122  } else {
123  m_selectedItem = -1;
124  // No bookmark has been found which matches to the given Url. Show
125  // a generic folder icon as pixmap for indication:
126  setIcon(QIcon::fromTheme(QStringLiteral("folder")));
127  }
128  updateTeardownAction();
129 }

178 void KUrlNavigatorPlacesSelector::dropEvent(QDropEvent *event)
179 {
180  setDisplayHintEnabled(DraggedHint, false);
181  update();
182 
183  QMimeDatabase db;
184  const QList<QUrl> urlList = KUrlMimeData::urlsFromMimeData(event->mimeData());
185  foreach (const QUrl &url, urlList) {
186  QMimeType mimetype = db.mimeTypeForUrl(url);
187  if (mimetype.inherits(QStringLiteral("inode/directory"))) {
188  m_placesModel->addPlace(url.fileName(), url);
189  }

193 void KUrlNavigatorPlacesSelector::activatePlace(QAction *action)
194 {
195  Q_ASSERT(action != 0);
196  if (action->data().toString() == QLatin1String("teardownAction")) {
197  QModelIndex index = m_placesModel->index(m_selectedItem, 0);
198  m_placesModel->requestTeardown(index);
199  return;
200  }
201 
202  QModelIndex index = m_placesModel->index(action->data().toInt(), 0);
203 
204  m_lastClickedIndex = QPersistentModelIndex();
205 
206  if (m_placesModel->setupNeeded(index)) {
207  connect(m_placesModel, SIGNAL(setupDone(QModelIndex,bool)),
208  this, SLOT(onStorageSetupDone(QModelIndex,bool)));
209 
210  m_lastClickedIndex = index;
211  m_placesModel->requestSetup(index);
212  return;
213  } else if (index.isValid()) {
214  m_selectedItem = index.row();
215  setIcon(m_placesModel->icon(index));
216  updateTeardownAction();
217  emit placeActivated(m_placesModel->url(index));
218  }
219 }
220 
221 void KUrlNavigatorPlacesSelector::onStorageSetupDone(const QModelIndex &index, bool success)
222 {
223  if (m_lastClickedIndex == index) {
224  if (success) {
225  m_selectedItem = index.row();
226  setIcon(m_placesModel->icon(index));
227  updateTeardownAction();
228  emit placeActivated(m_placesModel->url(index));
229  }
230  m_lastClickedIndex = QPersistentModelIndex();
231  }
232 }
