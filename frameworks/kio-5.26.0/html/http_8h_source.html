192  void updateCredentials(const HTTPRequest &request)
193  {
194  if (url.host() == request.url.host() && url.port() == request.url.port()) {
195  url.setUserName(request.url.userName());
196  url.setPassword(request.url.password());
197  }
198  if (proxyUrl.host() == request.proxyUrl.host() &&
199  proxyUrl.port() == request.proxyUrl.port()) {
200  proxyUrl.setUserName(request.proxyUrl.userName());
201  proxyUrl.setPassword(request.proxyUrl.password());
202  }
203  }
