115 void KSslInfoDialog::updateWhichPartsEncrypted()

138 void KSslInfoDialog::setSslInfo(const QList<QSslCertificate> &certificateChain,
139  const QString &ip, const QString &host,
140  const QString &sslProtocol, const QString &cipher,
141  int usedBits, int bits,
142  const QList<QList<KSslError::Error> > &validationErrors)
143 {
144 
145  d->certificateChain = certificateChain;
146  d->certificateErrors = validationErrors;
147 
148  d->ui.certSelector->clear();
149  for (int i = 0; i < certificateChain.size(); i++) {
150  const QSslCertificate &cert = certificateChain[i];
151  QString name;
152  static const QSslCertificate::SubjectInfo si[] = {
153  QSslCertificate::CommonName,
154  QSslCertificate::Organization,
155  QSslCertificate::OrganizationalUnitName
156  };
157  for (int j = 0; j < 3 && name.isEmpty(); j++)
158  {
159  name = cert.subjectInfo(si[j]).join(QStringLiteral(", "));
160  }
161  d->ui.certSelector->addItem(name);
162  }
163  if (certificateChain.size() < 2) {
164  d->ui.certSelector->setEnabled(false);
165  }
166  connect(d->ui.certSelector, SIGNAL(currentIndexChanged(int)),
167  this, SLOT(displayFromChain(int)));
168  if (d->certificateChain.isEmpty()) {
169  d->certificateChain.append(QSslCertificate());
170  }
171  displayFromChain(0);
172 
173  d->ui.ip->setText(ip);
174  d->ui.address->setText(host);
175  d->ui.sslVersion->setText(sslProtocol);
176 
177  const QStringList cipherInfo = cipher.split('\n', QString::SkipEmptyParts);
178  if (cipherInfo.size() >= 4) {
179  d->ui.encryption->setText(i18nc("%1, using %2 bits of a %3 bit key", "%1, %2 %3", cipherInfo[0],
180  i18ncp("Part of: %1, using %2 bits of a %3 bit key",
181  "using %1 bit", "using %1 bits", usedBits),
182  i18ncp("Part of: %1, using %2 bits of a %3 bit key",
183  "of a %1 bit key", "of a %1 bit key", bits)));
184  d->ui.details->setText(QStringLiteral("Auth = %1, Kx = %2, MAC = %3")
185  .arg(cipherInfo[1], cipherInfo[2],
186  cipherInfo[3]));
187  } else {
188  d->ui.encryption->setText(QLatin1String(""));
189  d->ui.details->setText(QLatin1String(""));
190  }
191 }

193 void KSslInfoDialog::displayFromChain(int i)
194 {
195  const QSslCertificate &cert = d->certificateChain[i];
196 
197  QString trusted;
198  if (!d->certificateErrors[i].isEmpty()) {
199  trusted = i18nc("The certificate is not trusted", "NO, there were errors:");
200  foreach (KSslError::Error e, d->certificateErrors[i]) {
201  KSslError classError(e);
202  trusted.append('\n');
203  trusted.append(classError.errorString());
204  }
205  } else {
206  trusted = i18nc("The certificate is trusted", "Yes");
207  }
208  d->ui.trusted->setText(trusted);
209 
210  QString vp = i18nc("%1 is the effective date of the certificate, %2 is the expiry date", "%1 to %2",
211  cert.effectiveDate().toString(),
212  cert.expiryDate().toString());
213  d->ui.validityPeriod->setText(vp);
214 
215  d->ui.serial->setText(cert.serialNumber());
216  d->ui.digest->setText(cert.digest().toHex());
217  d->ui.sha1Digest->setText(cert.digest(QCryptographicHash::Sha1).toHex());
218 
219  d->subject->setCertificate(cert, KSslCertificateBox::Subject);
220  d->issuer->setCertificate(cert, KSslCertificateBox::Issuer);
221 }

224 QList<QList<KSslError::Error> > KSslInfoDialog::errorsFromString(const QString &es)
225 {
226  QStringList sl = es.split('\n', QString::KeepEmptyParts);
227  QList<QList<KSslError::Error> > ret;
228  foreach (const QString &s, sl) {
229  QList<KSslError::Error> certErrors;
230  QStringList sl2 = s.split('\t', QString::SkipEmptyParts);
231  foreach (const QString &s2, sl2) {
232  bool didConvert;
233  KSslError::Error error = static_cast<KSslError::Error>(s2.toInt(&didConvert));
234  if (didConvert) {
235  certErrors.append(error);
236  }
237  }
238  ret.append(certErrors);
239  }
240  return ret;
241 }
