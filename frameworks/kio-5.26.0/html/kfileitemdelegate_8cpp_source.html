223 // Returns the additional information string, if one should be shown, or an empty string otherwise
224 QString KFileItemDelegate::Private::information(const QStyleOptionViewItem &option, const QModelIndex &index,
225  const KFileItem &item) const
226 {
227  QString string;
228 
229  if (informationList.isEmpty() || item.isNull() || !isListView(option)) {
230  return string;
231  }
232 
233  foreach (KFileItemDelegate::Information info, informationList) {
234  if (info == KFileItemDelegate::NoInformation) {
235  continue;
236  }
237 
238  if (!string.isEmpty()) {
239  string += QChar::LineSeparator;
240  }
241 
242  switch (info) {
243  case KFileItemDelegate::Size:
244  string += itemSize(index, item);
245  break;
246 
247  case KFileItemDelegate::Permissions:
248  string += item.permissionsString();
249  break;
250 
251  case KFileItemDelegate::OctalPermissions:
252  string += QString('0') + QString::number(item.permissions(), 8);
253  break;
254 
255  case KFileItemDelegate::Owner:
256  string += item.user();
257  break;
258 
259  case KFileItemDelegate::OwnerAndGroup:
260  string += item.user() + ':' + item.group();
261  break;
262 
263  case KFileItemDelegate::CreationTime:
264  string += item.timeString(KFileItem::CreationTime);
265  break;
266 
267  case KFileItemDelegate::ModificationTime:
268  string += item.timeString(KFileItem::ModificationTime);
269  break;
270 
271  case KFileItemDelegate::AccessTime:
272  string += item.timeString(KFileItem::AccessTime);
273  break;
274 
275  case KFileItemDelegate::MimeType:
276  string += item.isMimeTypeKnown() ? item.mimetype() : i18nc("@info mimetype", "Unknown");
277  break;
278 
279  case KFileItemDelegate::FriendlyMimeType:
280  string += item.isMimeTypeKnown() ? item.mimeComment() : i18nc("@info mimetype", "Unknown");
281  break;
282 
283  case KFileItemDelegate::LinkDest:
284  string += item.linkDest();
285  break;
286 
287  case KFileItemDelegate::LocalPathOrUrl:
288  if (!item.localPath().isEmpty()) {
289  string += item.localPath();
290  } else {
291  string += item.url().toDisplayString();
292  }
293  break;
294 
295  case KFileItemDelegate::Comment:
296  string += item.comment();
297  break;
298 
299  default:
300  break;
301  } // switch (info)
302  } // foreach (info, list)
303 
304  return string;
305 }

342 // Lays the text out in a rectangle no wider than maxWidth
343 QSize KFileItemDelegate::Private::layoutText(QTextLayout &layout, const QString &text, int maxWidth) const
344 {
345  QFontMetrics metrics(layout.font());
346  int leading = metrics.leading();
347  int height = 0;
348  qreal widthUsed = 0;
349  QTextLine line;
350 
351  layout.setText(text);
352 
353  layout.beginLayout();
354  while ((line = layout.createLine()).isValid()) {
355  line.setLineWidth(maxWidth);
356  height += leading;
357  line.setPosition(QPoint(0, height));
358  height += int(line.height());
359  widthUsed = qMax(widthUsed, line.naturalTextWidth());
360  }
361  layout.endLayout();
362 
363  return QSize(qCeil(widthUsed), height);
364 }

366 // Elides the text in the layout, by iterating over each line in the layout, eliding
367 // or word breaking the line if it's wider than the max width, and finally adding an
368 // ellipses at the end of the last line, if there are more lines than will fit within
369 // the vertical size constraints.
370 QString KFileItemDelegate::Private::elidedText(QTextLayout &layout, const QStyleOptionViewItem &option,
371  const QSize &size) const
372 {
373  const QString text = layout.text();
374  int maxWidth = size.width();
375  int maxHeight = size.height();
376  qreal height = 0;
377  bool wrapText = (option.features & QStyleOptionViewItem::WrapText);
378 
379  // If the string contains a single line of text that shouldn't be word wrapped
380  if (!wrapText && text.indexOf(QChar::LineSeparator) == -1) {
381  return option.fontMetrics.elidedText(text, option.textElideMode, maxWidth);
382  }
383 
384  // Elide each line that has already been laid out in the layout.
385  QString elided;
386  elided.reserve(text.length());
387 
388  for (int i = 0; i < layout.lineCount(); i++) {
389  QTextLine line = layout.lineAt(i);
390  int start = line.textStart();
391  int length = line.textLength();
392 
393  height += option.fontMetrics.leading();
394  if (height + line.height() + option.fontMetrics.lineSpacing() > maxHeight) {
395  // Unfortunately, if the line ends because of a line separator, elidedText() will be too
396  // clever and keep adding lines until it finds one that's too wide.
397  if (line.naturalTextWidth() < maxWidth && text[start + length - 1] == QChar::LineSeparator) {
398  elided += text.midRef(start, length - 1);
399  } else {
400  elided += option.fontMetrics.elidedText(text.mid(start), option.textElideMode, maxWidth);
401  }
402  break;
403  } else if (line.naturalTextWidth() > maxWidth) {
404  elided += option.fontMetrics.elidedText(text.mid(start, length), option.textElideMode, maxWidth);
405  if (!elided.endsWith(QChar::LineSeparator)) {
406  elided += QChar::LineSeparator;
407  }
408  } else {
409  elided += text.midRef(start, length);
410  }
411 
412  height += line.height();
413  }
414 
415  return elided;
416 }

429 QSize KFileItemDelegate::Private::displaySizeHint(const QStyleOptionViewItem &option,
430  const QModelIndex &index) const
431 {
432  QString label = option.text;
433  int maxWidth = 0;
434  if (maximumSize.isEmpty()) {
435  maxWidth = verticalLayout(option) && (option.features & QStyleOptionViewItem::WrapText)
436  ? option.decorationSize.width() + 10 : 32757;
437  } else {
438  const Margin &itemMargin = activeMargins[ItemMargin];
439  const Margin &textMargin = activeMargins[TextMargin];
440  maxWidth = maximumSize.width() -
441  (itemMargin.left + itemMargin.right) -
442  (textMargin.left + textMargin.right);
443  }
444 
445  KFileItem item = fileItem(index);
446 
447  // To compute the nominal size for the label + info, we'll just append
448  // the information string to the label
449  const QString info = information(option, index, item);
450  if (!info.isEmpty()) {
451  label += QString(QChar::LineSeparator) + info;
452  }
453 
454  QTextLayout layout;
455  setLayoutOptions(layout, option);
456 
457  QSize size = layoutText(layout, label, maxWidth);
458  if (!info.isEmpty()) {
459  // As soon as additional information is shown, it might be necessary that
460  // the label and/or the additional information must get elided. To prevent
461  // an expensive eliding in the scope of displaySizeHint, the maximum
462  // width is reserved instead.
463  size.setWidth(maxWidth);
464  }
465 
466  return addMargin(size, TextMargin);
467 }

582 QPixmap KFileItemDelegate::Private::transition(const QPixmap &from, const QPixmap &to, qreal amount) const
583 {
584  int value = int(0xff * amount);
585 
586  if (value == 0 || to.isNull()) {
587  return from;
588  }
589 
590  if (value == 0xff || from.isNull()) {
591  return to;
592  }
593 
594  QColor color;
595  color.setAlphaF(amount);
596 
597 // FIXME: Somehow this doesn't work on Mac OS..
598 #if defined(Q_OS_MAC)
599  const bool usePixmap = false;
600 #else
601  const bool usePixmap = from.paintEngine()->hasFeature(QPaintEngine::PorterDuff) &&
602  from.paintEngine()->hasFeature(QPaintEngine::BlendModes);
603 #endif
604 
605  // If the native paint engine supports Porter/Duff compositing and CompositionMode_Plus
606  if (usePixmap) {
607  QPixmap under = from;
608  QPixmap over = to;
609 
610  QPainter p;
611  p.begin(&over);
612  p.setCompositionMode(QPainter::CompositionMode_DestinationIn);
613  p.fillRect(over.rect(), color);
614  p.end();
615 
616  p.begin(&under);
617  p.setCompositionMode(QPainter::CompositionMode_DestinationOut);
618  p.fillRect(under.rect(), color);
619  p.setCompositionMode(QPainter::CompositionMode_Plus);
620  p.drawPixmap(0, 0, over);
621  p.end();
622 
623  return under;
624  }
625 // Cannot use XRender with QPixmap anymore.
626 #if 0 // HAVE_X11 && HAVE_XRENDER
627  else if (from.paintEngine()->hasFeature(QPaintEngine::PorterDuff)) { // We have Xrender support
628  // QX11PaintEngine doesn't implement CompositionMode_Plus in Qt 4.3,
629  // which we need to be able to do a transition from one pixmap to
630  // another.
631  //
632  // In order to avoid the overhead of converting the pixmaps to images
633  // and doing the operation entirely in software, this function has a
634  // specialized path for X11 that uses Xrender directly to do the
635  // transition. This operation can be fully accelerated in HW.
636  //
637  // This specialization can be removed when QX11PaintEngine supports
638  // CompositionMode_Plus.
639  QPixmap source(to), destination(from);
640 
641  source.detach();
642  destination.detach();
643 
644  Display *dpy = QX11Info::display();
645 
646  XRenderPictFormat *format = XRenderFindStandardFormat(dpy, PictStandardA8);
647  XRenderPictureAttributes pa;
648  pa.repeat = 1; // RepeatNormal
649 
650  // Create a 1x1 8 bit repeating alpha picture
651  Pixmap pixmap = XCreatePixmap(dpy, destination.handle(), 1, 1, 8);
652  Picture alpha = XRenderCreatePicture(dpy, pixmap, format, CPRepeat, &pa);
653  XFreePixmap(dpy, pixmap);
654 
655  // Fill the alpha picture with the opacity value
656  XRenderColor xcolor;
657  xcolor.alpha = quint16(0xffff * amount);
658  XRenderFillRectangle(dpy, PictOpSrc, alpha, &xcolor, 0, 0, 1, 1);
659 
660  // Reduce the alpha of the destination with 1 - opacity
661  XRenderComposite(dpy, PictOpOutReverse, alpha, None, destination.x11PictureHandle(),
662  0, 0, 0, 0, 0, 0, destination.width(), destination.height());
663 
664  // Add source * opacity to the destination
665  XRenderComposite(dpy, PictOpAdd, source.x11PictureHandle(), alpha,
666  destination.x11PictureHandle(),
667  0, 0, 0, 0, 0, 0, destination.width(), destination.height());
668 
669  XRenderFreePicture(dpy, alpha);
670  return destination;
671  }
672 #endif
673  else {
674  // Fall back to using QRasterPaintEngine to do the transition.
675  QImage under = from.toImage();
676  QImage over = to.toImage();
677 
678  QPainter p;
679  p.begin(&over);
680  p.setCompositionMode(QPainter::CompositionMode_DestinationIn);
681  p.fillRect(over.rect(), color);
682  p.end();
683 
684  p.begin(&under);
685  p.setCompositionMode(QPainter::CompositionMode_DestinationOut);
686  p.fillRect(under.rect(), color);
687  p.setCompositionMode(QPainter::CompositionMode_Plus);
688  p.drawImage(0, 0, over);
689  p.end();
690 
691  return QPixmap::fromImage(under);
692  }

1187 void KFileItemDelegate::paint(QPainter *painter, const QStyleOptionViewItem &option,
1188  const QModelIndex &index) const
1189 {

1191  return;
1192  }
1193 
1194  QStyleOptionViewItem opt(option);
1195  d->initStyleOption(&opt, index);
1196  d->setActiveMargins(d->verticalLayout(opt) ? Qt::Vertical : Qt::Horizontal);
1197 
1198  if (!(option.state & QStyle::State_Enabled)) {
1199  opt.palette.setCurrentColorGroup(QPalette::Disabled);
1200  }
1201 
1202  // Unset the mouse over bit if we're not drawing the first column
1203  if (index.column() > 0) {
1204  opt.state &= ~QStyle::State_MouseOver;
1205  } else {
1206  opt.viewItemPosition = QStyleOptionViewItem::OnlyOne;
1207  }
1208 
1209  const QAbstractItemView *view = qobject_cast<const QAbstractItemView *>(opt.widget);
1210 
1211  // Check if the item is being animated
1212  // ========================================================================
1213  KIO::AnimationState *state = d->animationState(opt, index, view);
1214  KIO::CachedRendering *cache = 0;
1215  qreal progress = ((opt.state & QStyle::State_MouseOver) &&
1216  index.column() == KDirModel::Name) ? 1.0 : 0.0;
1217  const QPoint iconPos = d->iconPosition(opt);
1218  QIcon::Mode iconMode;
1219 
1220  if (!(option.state & QStyle::State_Enabled)) {
1221  iconMode = QIcon::Disabled;
1222  } else if ((option.state & QStyle::State_Selected) && (option.state & QStyle::State_Active)) {
1223  iconMode = QIcon::Selected;
1224  } else {
1225  iconMode = QIcon::Normal;
1226  }
1227 
1228  QIcon::State iconState = option.state & QStyle::State_Open ? QIcon::On : QIcon::Off;
1229  QPixmap icon = opt.icon.pixmap(opt.decorationSize, iconMode, iconState);
1230 
1231  if (state && !state->hasJobAnimation()) {
1232  cache = state->cachedRendering();
1233  progress = state->hoverProgress();
1234  // Clear the mouse over bit temporarily
1235  opt.state &= ~QStyle::State_MouseOver;
1236 
1237  // If we have a cached rendering, draw the item from the cache
1238  if (cache) {
1239  if (cache->checkValidity(opt.state) && cache->regular.size() == opt.rect.size()) {
1240  QPixmap pixmap = d->transition(cache->regular, cache->hover, progress);
1241 
1242  if (state->cachedRenderingFadeFrom() && state->fadeProgress() != 1.0) {
1243  // Apply icon fading animation
1244  KIO::CachedRendering *fadeFromCache = state->cachedRenderingFadeFrom();
1245  const QPixmap fadeFromPixmap = d->transition(fadeFromCache->regular, fadeFromCache->hover, progress);
1246 
1247  pixmap = d->transition(fadeFromPixmap, pixmap, state->fadeProgress());
1248  }
1249  painter->drawPixmap(option.rect.topLeft(), pixmap);
1250  if (d->jobTransfersVisible && index.column() == 0) {
1251  if (index.data(KDirModel::HasJobRole).toBool()) {
1252  d->paintJobTransfers(painter, state->jobAnimationAngle(), iconPos, opt);
1253  }
1254  }
1255  return;
1256  }
1257 
1258  if (!cache->checkValidity(opt.state)) {
1259  if (opt.widget->style()->styleHint(QStyle::SH_Widget_Animate, 0, opt.widget)) {
1260  // Fade over from the old icon to the new one

1324  d->drawFocusRect(&p, opt, focusRect);
1325  p.end();

1335

1339  }

1342  painter->setRenderHint(QPainter::Antialiasing);

1345  d->paintJobTransfers(painter, state->jobAnimationAngle(), iconPos, opt);
1346  }

1352  // ========================================================================
1353  painter->save();
1354  painter->setRenderHint(QPainter::Antialiasing);
1355  painter->setPen(pen);
1356

1360  }
1361

1363  painter->drawPixmap(iconPos, icon);
1364 
1365  d->drawTextItems(painter, labelLayout, infoLayout, textBoundingRect);
1366  d->drawFocusRect(painter, opt, focusRect);
