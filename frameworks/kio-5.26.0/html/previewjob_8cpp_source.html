257 void PreviewJobPrivate::startPreview()
258 {
259  Q_Q(PreviewJob);
260  // Load the list of plugins to determine which mimetypes are supported
261  const KService::List plugins = KServiceTypeTrader::self()->query(QStringLiteral("ThumbCreator"));
262  QMap<QString, KService::Ptr> mimeMap;
263  QHash<QString, QHash<QString, KService::Ptr> > protocolMap;
264  for (KService::List::ConstIterator it = plugins.constBegin(); it != plugins.constEnd(); ++it) {
265  QStringList protocols = (*it)->property(QStringLiteral("X-KDE-Protocols")).toStringList();
266  const QString p = (*it)->property(QStringLiteral("X-KDE-Protocol")).toString();
267  if (!p.isEmpty()) {
268  protocols.append(p);
269  }
270  foreach (const QString &protocol, protocols) {
271  // We cannot use mimeTypes() here, it doesn't support groups such as: text/*
272  const QStringList mtypes = (*it)->serviceTypes();
273  // Add supported mimetype for this protocol
274  QStringList &_ms = m_remoteProtocolPlugins[protocol];
275  foreach (const QString &_m, mtypes) {
276  if (_m != QLatin1String("ThumbCreator")) {
277  protocolMap[protocol].insert(_m, *it);
278  if (!_ms.contains(_m)) {
279  _ms.append(_m);
280  }
281  }
282  }
283  }
284  if (enabledPlugins.contains((*it)->desktopEntryName())) {
285  const QStringList mimeTypes = (*it)->serviceTypes();
286  for (QStringList::ConstIterator mt = mimeTypes.constBegin(); mt != mimeTypes.constEnd(); ++mt) {
287  if (*mt != QLatin1String("ThumbCreator")) {
288  mimeMap.insert(*mt, *it);
289  }
290  }
291  }
292  }
293 
294  // Look for images and store the items in our todo list :)
295  bool bNeedCache = false;
296  KFileItemList::const_iterator kit = initialItems.constBegin();
297  const KFileItemList::const_iterator kend = initialItems.constEnd();
298  for (; kit != kend; ++kit) {
299  PreviewItem item;
300  item.item = *kit;
301  const QString mimeType = item.item.mimetype();
302  KService::Ptr plugin(0);
303 
304  // look for protocol-specific thumbnail plugins first
305  QHash<QString, QHash<QString, KService::Ptr> >::const_iterator it = protocolMap.constFind(item.item.url().scheme());
306  if (it != protocolMap.constEnd()) {
307  plugin = it.value().value(mimeType);
308  }
309 
310  if (!plugin) {
311  QMap<QString, KService::Ptr>::ConstIterator pluginIt = mimeMap.constFind(mimeType);
312  if (pluginIt == mimeMap.constEnd()) {
313  QString groupMimeType = mimeType;
314  groupMimeType.replace(QRegExp(QStringLiteral("/.*")), QStringLiteral("/*"));
315  pluginIt = mimeMap.constFind(groupMimeType);
316 
317  if (pluginIt == mimeMap.constEnd()) {
318  QMimeDatabase db;
319  // check mime type inheritance, resolve aliases
320  const QMimeType mimeInfo = db.mimeTypeForName(mimeType);
321  if (mimeInfo.isValid()) {
322  const QStringList parentMimeTypes = mimeInfo.allAncestors();
323  Q_FOREACH (const QString &parentMimeType, parentMimeTypes) {
324  pluginIt = mimeMap.constFind(parentMimeType);
325  if (pluginIt != mimeMap.constEnd()) {
326  break;
327  }
328  }
329  }
330  }
331  }
332 
333  if (pluginIt != mimeMap.constEnd()) {
334  plugin = *pluginIt;
335  }
336  }
337 
338  if (plugin) {
339  item.plugin = plugin;
340  items.append(item);
341  if (!bNeedCache && bSave && plugin->property(QStringLiteral("CacheThumbnail")).toBool()) {
342  const QUrl url = (*kit).url();
343  if (!url.isLocalFile() ||
344  !url.adjusted(QUrl::RemoveFilename).toLocalFile().startsWith(thumbRoot)) {
345  bNeedCache = true;
346  }
347  }
348  } else {
349  emit q->failed(*kit);
350  }
351  }
352 
353  KConfigGroup cg(KSharedConfig::openConfig(), "PreviewSettings");
354  maximumLocalSize = cg.readEntry("MaximumSize", 5 * 1024 * 1024LL /* 5MB */);
355  maximumRemoteSize = cg.readEntry("MaximumRemoteSize", 0);
356 
357  if (bNeedCache) {
358  if (width <= 128 && height <= 128) {
359  cacheWidth = cacheHeight = 128;
360  } else {
361  cacheWidth = cacheHeight = 256;
362  }
363  thumbPath = thumbRoot + (cacheWidth == 128 ? "normal/" : "large/");
364  if (!QDir(thumbPath).exists()) {
365  if (QDir().mkpath(thumbPath)) { // Qt5 TODO: mkpath(dirPath, permissions)
366  QFile f(thumbPath);
367  f.setPermissions(QFile::ReadUser | QFile::WriteUser | QFile::ExeUser); // 0700
368  }
369  }
370  } else {
371  bSave = false;
372  }
373 
374  initialItems.clear();
375  determineNextFile();
376 }

410 void PreviewJobPrivate::determineNextFile()
411 {
412  Q_Q(PreviewJob);
413  if (!currentItem.item.isNull()) {
414  if (!succeeded) {
415  emit q->failed(currentItem.item);
416  }
417  }
418  // No more items ?
419  if (items.isEmpty()) {
420  q->emitResult();
421  return;
422  } else {
423  // First, stat the orig file
424  state = PreviewJobPrivate::STATE_STATORIG;
425  currentItem = items.first();
426  succeeded = false;
427  items.removeFirst();
428  KIO::Job *job = KIO::stat(currentItem.item.url(), KIO::HideProgressInfo);
429  job->addMetaData(QStringLiteral("no-auth-prompt"), QStringLiteral("true"));
430  q->addSubjob(job);
431  }
432 }

434 void PreviewJob::slotResult(KJob *job)
435 {
436  Q_D(PreviewJob);
437 
438  removeSubjob(job);
439  Q_ASSERT(!hasSubjobs()); // We should have only one job at a time ...
440  switch (d->state) {
441  case PreviewJobPrivate::STATE_STATORIG: {
442  if (job->error()) { // that's no good news...
443  // Drop this one and move on to the next one
444  d->determineNextFile();
445  return;
446  }
447  const KIO::UDSEntry entry = static_cast<KIO::StatJob *>(job)->statResult();
448  d->tOrig = QDateTime::fromTime_t(entry.numberValue(KIO::UDSEntry::UDS_MODIFICATION_TIME, 0));
449 
450  bool skipCurrentItem = false;
451  const KIO::filesize_t size = (KIO::filesize_t)entry.numberValue(KIO::UDSEntry::UDS_SIZE, 0);
452  const QUrl itemUrl = d->currentItem.item.mostLocalUrl();
453 
454  if (itemUrl.isLocalFile() || KProtocolInfo::protocolClass(itemUrl.scheme()) == QLatin1String(":local")) {
455  skipCurrentItem = !d->ignoreMaximumSize && size > d->maximumLocalSize
456  && !d->currentItem.plugin->property(QStringLiteral("IgnoreMaximumSize")).toBool();
457  } else {
458  // For remote items the "IgnoreMaximumSize" plugin property is not respected
459  skipCurrentItem = !d->ignoreMaximumSize && size > d->maximumRemoteSize;
460 
461  // Remote directories are not supported, don't try to do a file_copy on them
462  if (!skipCurrentItem) {
463  // TODO update item.mimeType from the UDS entry, in case it wasn't set initially
464  // But we don't use the mimetype anymore, we just use isDir().
465  if (d->currentItem.item.isDir()) {
466  skipCurrentItem = true;
467  }
468  }
469  }
470  if (skipCurrentItem) {
471  d->determineNextFile();
472  return;
473  }
474 
475  bool pluginHandlesSequences = d->currentItem.plugin->property(QStringLiteral("HandleSequences"), QVariant::Bool).toBool();
476  if (!d->currentItem.plugin->property(QStringLiteral("CacheThumbnail")).toBool() || (d->sequenceIndex && pluginHandlesSequences)) {
477  // This preview will not be cached, no need to look for a saved thumbnail
478  // Just create it, and be done
479  d->getOrCreateThumbnail();
480  return;
481  }
482 
483  if (d->statResultThumbnail()) {
484  return;
485  }
486 
487  d->getOrCreateThumbnail();
488  return;
489  }
490  case PreviewJobPrivate::STATE_GETORIG: {
491  if (job->error()) {
492  d->determineNextFile();
493  return;
494  }
495 
496  d->createThumbnail(static_cast<KIO::FileCopyJob *>(job)->destUrl().toLocalFile());
497  return;
498  }
499  case PreviewJobPrivate::STATE_CREATETHUMB: {
500  if (!d->tempName.isEmpty()) {
501  QFile::remove(d->tempName);
502  d->tempName.clear();
503  }
504  d->determineNextFile();
505  return;
506  }
507  }
508 }

510 bool PreviewJobPrivate::statResultThumbnail()
511 {
512  if (thumbPath.isEmpty()) {
513  return false;
514  }
515 
516  QUrl url = currentItem.item.mostLocalUrl();
517  // Don't include the password if any
518  url.setPassword(QString());
519  origName = url.toString();
520 
521  QCryptographicHash md5(QCryptographicHash::Md5);
522  md5.addData(QFile::encodeName(origName));
523  thumbName = QFile::encodeName(md5.result().toHex()) + ".png";
524 
525  QImage thumb;
526  if (!thumb.load(thumbPath + thumbName)) {
527  return false;
528  }
529 
530  if (thumb.text(QStringLiteral("Thumb::URI")) != origName ||
531  thumb.text(QStringLiteral("Thumb::MTime")).toLongLong() != tOrig.toTime_t()) {
532  return false;
533  }
534 
535  QString thumbnailerVersion = currentItem.plugin->property(QStringLiteral("ThumbnailerVersion"), QVariant::String).toString();
536 
537  if (!thumbnailerVersion.isEmpty() && thumb.text(QStringLiteral("Software")).startsWith(QStringLiteral("KDE Thumbnail Generator"))) {
538  //Check if the version matches
539  //The software string should read "KDE Thumbnail Generator pluginName (vX)"
540  QString softwareString = thumb.text(QStringLiteral("Software")).remove(QStringLiteral("KDE Thumbnail Generator")).trimmed();
541  if (softwareString.isEmpty()) {
542  // The thumbnail has been created with an older version, recreating
543  return false;
544  }
545  int versionIndex = softwareString.lastIndexOf(QLatin1String("(v"));
546  if (versionIndex < 0) {
547  return false;
548  }
549 
550  QString cachedVersion = softwareString.remove(0, versionIndex + 2);
551  cachedVersion.chop(1);
552  uint thumbnailerMajor = thumbnailerVersion.toInt();
553  uint cachedMajor = cachedVersion.toInt();
554  if (thumbnailerMajor > cachedMajor) {
555  return false;
556  }
557  }
558 
559  // Found it, use it
560  emitPreview(thumb);
561  succeeded = true;
562  determineNextFile();
563  return true;
564 }

566 void PreviewJobPrivate::getOrCreateThumbnail()
567 {
568  Q_Q(PreviewJob);
569  // We still need to load the orig file ! (This is getting tedious) :)
570  const KFileItem &item = currentItem.item;
571  const QString localPath = item.localPath();
572  if (!localPath.isEmpty()) {
573  createThumbnail(localPath);
574  } else {
575  const QUrl fileUrl = item.url();
576  // heuristics for remote URL support
577  bool supportsProtocol = false;
578  if (m_remoteProtocolPlugins.value(fileUrl.scheme()).contains(item.mimetype())) {
579  // There's a plugin supporting this protocol and mimetype
580  supportsProtocol = true;
581  } else if (m_remoteProtocolPlugins.value(QStringLiteral("KIO")).contains(item.mimetype())) {
582  // Assume KIO understands any URL, ThumbCreator slaves who have
583  // X-KDE-Protocols=KIO will get fed the remote URL directly.
584  supportsProtocol = true;
585  }
586 
587  if (supportsProtocol) {
588  createThumbnail(fileUrl.toString());
589  return;
590  }
591  // No plugin support access to this remote content, copy the file
592  // to the local machine, then create the thumbnail
593  state = PreviewJobPrivate::STATE_GETORIG;
594  QTemporaryFile localFile;
595  localFile.setAutoRemove(false);
596  localFile.open();
597  tempName = localFile.fileName();
598  const QUrl currentURL = item.mostLocalUrl();
599  KIO::Job *job = KIO::file_copy(currentURL, QUrl::fromLocalFile(tempName), -1, KIO::Overwrite | KIO::HideProgressInfo /* No GUI */);
600  job->addMetaData(QStringLiteral("thumbnail"), QStringLiteral("1"));
601  q->addSubjob(job);
602  }
603 }

605 void PreviewJobPrivate::createThumbnail(const QString &pixPath)
606 {
607  Q_Q(PreviewJob);
608  state = PreviewJobPrivate::STATE_CREATETHUMB;
609  QUrl thumbURL;
610  thumbURL.setScheme(QStringLiteral("thumbnail"));
611  thumbURL.setPath(pixPath);
612  KIO::TransferJob *job = KIO::get(thumbURL, NoReload, HideProgressInfo);
613  q->addSubjob(job);
614  q->connect(job, SIGNAL(data(KIO::Job*,QByteArray)), SLOT(slotThumbData(KIO::Job*,QByteArray)));
615  bool save = bSave && currentItem.plugin->property(QStringLiteral("CacheThumbnail")).toBool() && !sequenceIndex;
616  job->addMetaData(QStringLiteral("mimeType"), currentItem.item.mimetype());
617  job->addMetaData(QStringLiteral("width"), QString().setNum(save ? cacheWidth : width));
618  job->addMetaData(QStringLiteral("height"), QString().setNum(save ? cacheHeight : height));
619  job->addMetaData(QStringLiteral("iconSize"), QString().setNum(save ? 64 : iconSize));
620  job->addMetaData(QStringLiteral("iconAlpha"), QString().setNum(iconAlpha));
621  job->addMetaData(QStringLiteral("plugin"), currentItem.plugin->library());
622  if (sequenceIndex) {
623  job->addMetaData(QStringLiteral("sequence-index"), QString().setNum(sequenceIndex));
624  }
625 
626 #ifdef Q_OS_UNIX
627  if (shmid == -1) {
628  if (shmaddr) {
629  shmdt((char *)shmaddr);
630  shmctl(shmid, IPC_RMID, 0);
631  }
632  shmid = shmget(IPC_PRIVATE, cacheWidth * cacheHeight * 4, IPC_CREAT | 0600);
633  if (shmid != -1) {
634  shmaddr = (uchar *)(shmat(shmid, 0, SHM_RDONLY));
635  if (shmaddr == (uchar *) - 1) {
636  shmctl(shmid, IPC_RMID, 0);
637  shmaddr = 0;
638  shmid = -1;
639  }
640  } else {
641  shmaddr = 0;

650 void PreviewJobPrivate::slotThumbData(KIO::Job *, const QByteArray &data)
651 {
652  bool save = bSave &&
653  currentItem.plugin->property(QStringLiteral("CacheThumbnail")).toBool() &&
654  (!currentItem.item.url().isLocalFile() ||
655  !currentItem.item.url().adjusted(QUrl::RemoveFilename).toLocalFile().startsWith(thumbRoot))
656  && !sequenceIndex;
657  QImage thumb;
658 #ifdef Q_OS_UNIX
659  if (shmaddr) {
660  // Keep this in sync with kdebase/kioslave/thumbnail.cpp
661  QDataStream str(data);
662  int width, height;
663  quint8 iFormat;
664  str >> width >> height >> iFormat;
665  QImage::Format format = static_cast<QImage::Format>(iFormat);
666  thumb = QImage(shmaddr, width, height, format).copy();
667  } else
668 #endif
669  thumb.loadFromData(data);
670 
671  if (thumb.isNull()) {
672  QDataStream s(data);
673  s >> thumb;
674  }
675 
676  if (save) {
677  thumb.setText(QStringLiteral("Thumb::URI"), origName);
678  thumb.setText(QStringLiteral("Thumb::MTime"), QString::number(tOrig.toTime_t()));
679  thumb.setText(QStringLiteral("Thumb::Size"), number(currentItem.item.size()));
680  thumb.setText(QStringLiteral("Thumb::Mimetype"), currentItem.item.mimetype());
681  QString thumbnailerVersion = currentItem.plugin->property(QStringLiteral("ThumbnailerVersion"), QVariant::String).toString();
682  QString signature = QString("KDE Thumbnail Generator " + currentItem.plugin->name());
683  if (!thumbnailerVersion.isEmpty()) {
684  signature.append(" (v" + thumbnailerVersion + ')');
685  }
686  thumb.setText(QStringLiteral("Software"), signature);
687  QSaveFile saveFile(thumbPath + thumbName);
688  if (saveFile.open(QIODevice::WriteOnly)) {
689  if (thumb.save(&saveFile, "PNG")) {
690  saveFile.commit();
691  }
692  }
693  }
694  emitPreview(thumb);
695  succeeded = true;
696 }
