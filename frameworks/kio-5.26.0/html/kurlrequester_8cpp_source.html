167  QUrl url() const
168  {
169  const QString txt = text();
170  KUrlCompletion *comp;
171  if (combo) {
172  comp = qobject_cast<KUrlCompletion *>(combo->completionObject());
173  } else {
174  comp = qobject_cast<KUrlCompletion *>(edit->completionObject());
175  }
176 
177  QString enteredPath;
178  if (comp)
179  enteredPath = comp->replacedPath(txt);
180  else
181  enteredPath = txt;
182 
183  if (QDir::isAbsolutePath(enteredPath)) {
184  return QUrl::fromLocalFile(enteredPath);
185  }
186 
187  const QUrl enteredUrl = QUrl(enteredPath); // absolute or relative
188  if (enteredUrl.isRelative() && !txt.isEmpty()) {
189  QUrl finalUrl(m_startDir);
190  finalUrl.setPath(finalUrl.path() + '/' + enteredPath);
191  return finalUrl;
192  } else {
193  return enteredUrl;
194  }
195  }

219  // Converts from "*.foo *.bar|Comment" to "Comment (*.foo *.bar)"
220  QStringList kToQFilters(const QString &filters) const
221  {
222  QStringList qFilters = filters.split('\n', QString::SkipEmptyParts);
223 
224  for (QStringList::iterator it = qFilters.begin(); it != qFilters.end(); ++it) {
225  int sep = it->indexOf('|');
226  QString globs = it->left(sep);
227  QString desc = it->mid(sep + 1);
228  *it = QStringLiteral("%1 (%2)").arg(desc, globs);
229  }
230 
231  return qFilters;
232  }

396 void KUrlRequester::KUrlRequesterPrivate::_k_slotOpenDialog()
397 {
398  if (myFileDialog)
399  if (myFileDialog->isVisible()) {
400  //The file dialog is already being shown, raise it and exit
401  myFileDialog->raise();
402  myFileDialog->activateWindow();
403  return;
404  }
405 
406  if (((fileDialogMode & KFile::Directory) && !(fileDialogMode & KFile::File)) ||
407  /* catch possible fileDialog()->setMode( KFile::Directory ) changes */
408  (myFileDialog && (myFileDialog->fileMode() == QFileDialog::Directory &&
409  myFileDialog->testOption(QFileDialog::ShowDirsOnly)))) {
410  const QUrl openUrl = (!m_parent->url().isEmpty() && !m_parent->url().isRelative())
411  ? m_parent->url() : m_startDir;
412 
413  /* FIXME We need a new abstract interface for using KDirSelectDialog in a non-modal way */
414 
415  QUrl newUrl;
416  if (fileDialogMode & KFile::LocalOnly) {
417  newUrl = QFileDialog::getExistingDirectoryUrl(m_parent, QString(), openUrl, QFileDialog::ShowDirsOnly, QStringList() << QStringLiteral("file"));
418  } else {
419  newUrl = getDirFromFileDialog(openUrl);
420  }
421 
422  if (newUrl.isValid()) {
423  m_parent->setUrl(newUrl);
424  emit m_parent->urlSelected(url());
425  }
426  } else {
427  emit m_parent->openFileDialog(m_parent);
428 
429  //Creates the fileDialog if it doesn't exist yet
430  QFileDialog *dlg = m_parent->fileDialog();
431 
432  if (!url().isEmpty() && !url().isRelative()) {
433  QUrl u(url());
434  // If we won't be able to list it (e.g. http), then don't try :)
435  if (KProtocolManager::supportsListing(u)) {
436  dlg->selectUrl(u);
437  }
438  } else {
439  dlg->setDirectoryUrl(m_startDir);
440  }
441 
442  //Update the file dialog window modality
443  if (dlg->windowModality() != fileDialogModality) {
444  dlg->setWindowModality(fileDialogModality);
445  }
446 
447  if (fileDialogModality == Qt::NonModal) {
448  dlg->show();
449  } else {
450  dlg->exec();
451  }
452  }
453 }

455 void KUrlRequester::KUrlRequesterPrivate::_k_slotFileDialogAccepted()
456 {
457  if (!myFileDialog) {
458  return;

460 
461  QUrl newUrl = myFileDialog->selectedUrls().first();
462  if (newUrl.isValid()) {
463  m_parent->setUrl(newUrl);
464  emit m_parent->urlSelected(url());
465  // remember url as defaultStartDir and update startdir for autocompletion
466  if (newUrl.isLocalFile() && !m_startDirCustomized) {

468  updateCompletionStartDir(m_startDir);
