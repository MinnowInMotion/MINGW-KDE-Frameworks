62 DavJob::DavJob(DavJobPrivate &dd, int method, const QString &request)
63  : TransferJob(dd)
64 {
65  // We couldn't set the args when calling the parent constructor,
66  // so do it now.
67  Q_D(DavJob);
68  QDataStream stream(&d->m_packedArgs, QIODevice::WriteOnly);
69  stream << (int) 7 << d->m_url << method;
70  // Same for static data
71  if (! request.isEmpty()) {
72  d->staticData = "<?xml version=\"1.0\" encoding=\"utf-8\"?>\r\n" + request.toUtf8();
73  d->staticData.truncate(d->staticData.size() - 1);
74  d->savedStaticData = d->staticData;
75  stream << static_cast<qint64>(d->staticData.size());
76  } else {
77  stream << static_cast<qint64>(-1);
78  }
79 }

86 void DavJob::slotData(const QByteArray &data)
87 {
88  Q_D(DavJob);
89  if (d->m_redirectionURL.isEmpty() || !d->m_redirectionURL.isValid() || error()) {
90  unsigned int oldSize = d->str_response.size();
91  d->str_response.resize(oldSize + data.size());
92  memcpy(d->str_response.data() + oldSize, data.data(), data.size());
93  }
94 }

96 void DavJob::slotFinished()
97 {
98  Q_D(DavJob);
99  //qDebug() << d->str_response;
100  if (!d->m_redirectionURL.isEmpty() && d->m_redirectionURL.isValid() &&
101  (d->m_command == CMD_SPECIAL)) {
102  QDataStream istream(d->m_packedArgs);
103  int s_cmd, s_method;
104  qint64 s_size;
105  QUrl s_url;
106  istream >> s_cmd;
107  istream >> s_url;
108  istream >> s_method;
109  istream >> s_size;
110  // PROPFIND
111  if ((s_cmd == 7) && (s_method == (int)KIO::DAV_PROPFIND)) {
112  d->m_packedArgs.truncate(0);
113  QDataStream stream(&d->m_packedArgs, QIODevice::WriteOnly);
114  stream << (int)7 << d->m_redirectionURL << (int)KIO::DAV_PROPFIND << s_size;
115  }
116  } else if (! d->m_response.setContent(d->str_response, true)) {
117  // An error occurred parsing the XML response
118  QDomElement root = d->m_response.createElementNS(QStringLiteral("DAV:"), QStringLiteral("error-report"));
119  d->m_response.appendChild(root);
120 
121  QDomElement el = d->m_response.createElementNS(QStringLiteral("DAV:"), QStringLiteral("offending-response"));
122  QDomText textnode = d->m_response.createTextNode(d->str_response);
123  el.appendChild(textnode);
124  root.appendChild(el);
125  }
126  //qDebug() << d->m_response.toString();
127  TransferJob::slotFinished();
128  d->staticData = d->savedStaticData; // Need to send DAV request to this host too
129 }
