58 static UDSEntry createUDSEntryWin(const QFileInfo &fileInfo)
59 {
60  UDSEntry entry;
61 
62  entry.insert(KIO::UDSEntry::UDS_NAME, fileInfo.fileName());
63  if (fileInfo.isSymLink()) {
64  entry.insert(KIO::UDSEntry::UDS_TARGET_URL, fileInfo.symLinkTarget());
65  /* TODO - or not useful on windows?
66  if ( details > 1 ) {
67  // It is a link pointing to nowhere
68  type = S_IFMT - 1;
69  access = S_IRWXU | S_IRWXG | S_IRWXO;
70 
71  entry.insert( KIO::UDSEntry::UDS_FILE_TYPE, type );
72  entry.insert( KIO::UDSEntry::UDS_ACCESS, access );
73  entry.insert( KIO::UDSEntry::UDS_SIZE, 0LL );
74  goto notype;
75 
76  }
77  */
78  }
79  int type = S_IFREG;
80  int access = 0;
81  if (fileInfo.isDir()) {
82  type = S_IFDIR;
83  } else if (fileInfo.isSymLink()) {
84  type = QT_STAT_LNK;
85  }
86  if (fileInfo.isReadable()) {
87  access |= S_IRUSR;
88  }
89  if (fileInfo.isWritable()) {
90  access |= S_IWUSR;
91  }
92  if (fileInfo.isExecutable()) {
93  access |= S_IXUSR;
94  }
95 
96  entry.insert(KIO::UDSEntry::UDS_FILE_TYPE, type);
97  entry.insert(KIO::UDSEntry::UDS_ACCESS, access);
98  entry.insert(KIO::UDSEntry::UDS_SIZE, fileInfo.size());
99  if (fileInfo.isHidden()) {
100  entry.insert(KIO::UDSEntry::UDS_HIDDEN, true);
101  }
102 
103  entry.insert(KIO::UDSEntry::UDS_MODIFICATION_TIME, fileInfo.lastModified().toTime_t());
104  entry.insert(KIO::UDSEntry::UDS_USER, fileInfo.owner());
105  entry.insert(KIO::UDSEntry::UDS_GROUP, fileInfo.group());
106  entry.insert(KIO::UDSEntry::UDS_ACCESS_TIME, fileInfo.lastRead().toTime_t());
107 
108  return entry;
109 }

111 void FileProtocol::copy(const QUrl &src, const QUrl &dest,
112  int _mode, JobFlags _flags)
113 {
114  // qDebug() << "copy(): " << src << " -> " << dest << ", mode=" << _mode;
115 
116  QFileInfo _src(src.toLocalFile());
117  QFileInfo _dest(dest.toLocalFile());
118  DWORD dwFlags = COPY_FILE_FAIL_IF_EXISTS;
119 
120  if (_src == _dest) {
121  error(KIO::ERR_IDENTICAL_FILES, _dest.filePath());
122  return;
123  }
124 
125  if (!_src.exists()) {
126  error(KIO::ERR_DOES_NOT_EXIST, _src.filePath());
127  return;
128  }
129 
130  if (_src.isDir()) {
131  error(KIO::ERR_IS_DIRECTORY, _src.filePath());
132  return;
133  }
134 
135  if (_dest.exists()) {
136  if (_dest.isDir()) {
137  error(KIO::ERR_DIR_ALREADY_EXIST, _dest.filePath());
138  return;
139  }
140 
141  if (!(_flags & KIO::Overwrite)) {
142  error(KIO::ERR_FILE_ALREADY_EXIST, _dest.filePath());
143  return;
144  }
145 
146  dwFlags = 0;
147  }
148 
149  if (!QFileInfo(_dest.dir().absolutePath()).exists()) {
150  _dest.dir().mkdir(_dest.dir().absolutePath());
151  }
152 
153  if (CopyFileExW((LPCWSTR) _src.filePath().utf16(),
154  (LPCWSTR) _dest.filePath().utf16(),
155  CopyProgressRoutine,
156  (LPVOID) this,
157  FALSE,
158  dwFlags) == 0) {
159  DWORD dwLastErr = GetLastError();
160  if (dwLastErr == ERROR_FILE_NOT_FOUND) {
161  error(KIO::ERR_DOES_NOT_EXIST, _src.filePath());
162  } else if (dwLastErr == ERROR_ACCESS_DENIED) {
163  error(KIO::ERR_ACCESS_DENIED, _dest.filePath());
164  } else {
165 #if 0
166  LPVOID lpMsgBuf;
167 
168  FormatMessage(
169  FORMAT_MESSAGE_ALLOCATE_BUFFER |
170  FORMAT_MESSAGE_FROM_SYSTEM |
171  FORMAT_MESSAGE_IGNORE_INSERTS,
172  NULL,
173  dwLastErr,
174  MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT),
175  (LPTSTR) &lpMsgBuf,
176  0, NULL);
177  OutputDebugString((WCHAR *)lpMsgBuf);
178 #endif
179  error(KIO::ERR_CANNOT_RENAME, _src.filePath());
180  // qDebug() << "Copying file " << _src.filePath() << " failed (" << dwLastErr << ")";
181  }
182  return;
183  }
184 
185  finished();
186 }

188 void FileProtocol::listDir(const QUrl &url)
189 {
190  // qDebug() << "========= LIST " << url << " =========";
191 
192  if (!url.isLocalFile()) {
193  QUrl redir(url);
194  redir.setScheme(config()->readEntry("DefaultRemoteProtocol", "smb"));
195  redirection(redir);
196  // qDebug() << "redirecting to " << redir;
197  finished();
198  return;
199  }
200 
201  QDir dir(url.toLocalFile());
202  dir.setFilter(QDir::AllEntries | QDir::Hidden);
203 
204  if (!dir.exists()) {
205  // qDebug() << "========= ERR_DOES_NOT_EXIST =========";
206  error(KIO::ERR_DOES_NOT_EXIST, url.toLocalFile());
207  return;
208  }
209 
210  if (!dir.isReadable()) {
211  // qDebug() << "========= ERR_CANNOT_ENTER_DIRECTORY =========";
212  error(KIO::ERR_CANNOT_ENTER_DIRECTORY, url.toLocalFile());
213  return;
214  }
215  QDirIterator it(dir);
216  UDSEntry entry;
217  while (it.hasNext()) {
218  it.next();
219  UDSEntry entry = createUDSEntryWin(it.fileInfo());
220 
221  listEntry(entry);
222  entry.clear();
223  }
224 
225  // qDebug() << "============= COMPLETED LIST ============";
226 
227  finished();
228 }

237  DWORD dwFlags = 0;

257  return;
258  }
259 
260 #ifndef _WIN32_WCE
261  dwFlags = MOVEFILE_REPLACE_EXISTING;
262 #endif
263  }
264  // To avoid error 17 - The system cannot move the file to a different disk drive.
265 #ifndef _WIN32_WCE
266  dwFlags |= MOVEFILE_COPY_ALLOWED;
267 
268  if (MoveFileExW((LPCWSTR) _src.filePath().utf16(),
269  (LPCWSTR) _dest.filePath().utf16(), dwFlags) == 0)
270 #else
271  if (MoveFileW((LPCWSTR) _src.filePath().utf16(),
272  (LPCWSTR) _dest.filePath().utf16()) == 0)
273 #endif
274  {
275  DWORD dwLastErr = GetLastError();
276  if (dwLastErr == ERROR_FILE_NOT_FOUND) {
277  error(KIO::ERR_DOES_NOT_EXIST, _src.filePath());
278  } else if (dwLastErr == ERROR_ACCESS_DENIED) {
279  error(KIO::ERR_ACCESS_DENIED, _dest.filePath());
280  } else {
281  error(KIO::ERR_CANNOT_RENAME, _src.filePath());
282  qDebug() << "Renaming file "
283  << _src.filePath()
284  << " failed ("
285  << dwLastErr << ")";
286  }
287  return;
288  }

290  finished();
