117 void KACLEditWidget::KACLEditWidgetPrivate::_k_slotUpdateButtons()
118 {
119  bool atLeastOneIsNotDeletable = false;
120  bool atLeastOneIsNotAllowedToChangeType = false;
121  int selectedCount = 0;
122  QList<QTreeWidgetItem *> selected = m_listView->selectedItems();
123  QListIterator<QTreeWidgetItem *> it(selected);
124  while (it.hasNext()) {
125  KACLListViewItem *item = static_cast<KACLListViewItem *>(it.next());
126  ++selectedCount;
127  if (!item->isDeletable()) {
128  atLeastOneIsNotDeletable = true;
129  }
130  if (!item->isAllowedToChangeType()) {
131  atLeastOneIsNotAllowedToChangeType = true;
132  }
133  }
134  m_EditBtn->setEnabled(selectedCount && !atLeastOneIsNotAllowedToChangeType);
135  m_DelBtn->setEnabled(selectedCount && !atLeastOneIsNotDeletable);
136 }

315 void KACLListViewItem::calcEffectiveRights()
316 {
317  QString strEffective = QStringLiteral("---");
318 
319  // Do we need to worry about the mask entry? It applies to named users,
320  // owning group, and named groups
321  if (m_pACLListView->hasMaskEntry()
322  && (type == KACLListView::NamedUser
323  || type == KACLListView::Group
324  || type == KACLListView::NamedGroup)
325  && !isDefault) {
326 
327  strEffective[0] = (m_pACLListView->maskPermissions() & value & ACL_READ) ? 'r' : '-';
328  strEffective[1] = (m_pACLListView->maskPermissions() & value & ACL_WRITE) ? 'w' : '-';
329  strEffective[2] = (m_pACLListView->maskPermissions() & value & ACL_EXECUTE) ? 'x' : '-';
330  /*
331  // What about any partial perms?
332  if ( maskPerms & partialPerms & ACL_READ || // Partial perms on entry
333  maskPartialPerms & perms & ACL_READ || // Partial perms on mask
334  maskPartialPerms & partialPerms & ACL_READ ) // Partial perms on mask and entry
335  strEffective[0] = 'R';
336  if ( maskPerms & partialPerms & ACL_WRITE || // Partial perms on entry
337  maskPartialPerms & perms & ACL_WRITE || // Partial perms on mask
338  maskPartialPerms & partialPerms & ACL_WRITE ) // Partial perms on mask and entry
339  strEffective[1] = 'W';
340  if ( maskPerms & partialPerms & ACL_EXECUTE || // Partial perms on entry
341  maskPartialPerms & perms & ACL_EXECUTE || // Partial perms on mask
342  maskPartialPerms & partialPerms & ACL_EXECUTE ) // Partial perms on mask and entry
343  strEffective[2] = 'X';
344  */
345  } else {
346  // No, the effective value are just the value in this entry
347  strEffective[0] = (value & ACL_READ) ? 'r' : '-';
348  strEffective[1] = (value & ACL_WRITE) ? 'w' : '-';
349  strEffective[2] = (value & ACL_EXECUTE) ? 'x' : '-';
350 
351  /*
352  // What about any partial perms?
353  if ( partialPerms & ACL_READ )
354  strEffective[0] = 'R';
355  if ( partialPerms & ACL_WRITE )
356  strEffective[1] = 'W';
357  if ( partialPerms & ACL_EXECUTE )
358  strEffective[2] = 'X';
359  */
360  }
361  setText(5, strEffective);
362 }

364 bool KACLListViewItem::isDeletable() const
365 {
366  bool isMaskAndDeletable = false;
367  if (type == KACLListView::Mask) {
368  if (!isDefault && m_pACLListView->maskCanBeDeleted()) {
369  isMaskAndDeletable = true;
370  } else if (isDefault && m_pACLListView->defaultMaskCanBeDeleted()) {
371  isMaskAndDeletable = true;
372  }
373  }
374  return type != KACLListView::User &&
375  type != KACLListView::Group &&
376  type != KACLListView::Others &&
377  (type != KACLListView::Mask || isMaskAndDeletable);
378 }

562 void EditACLEntryDialog::slotUpdateAllowedUsersAndGroups()
563 {
564  const QString oldUser = m_usersCombo->currentText();
565  const QString oldGroup = m_groupsCombo->currentText();
566  m_usersCombo->clear();
567  m_groupsCombo->clear();
568  if (m_defaultCB && m_defaultCB->isChecked()) {
569  m_usersCombo->addItems(m_defaultUsers);
570  if (m_defaultUsers.contains(oldUser)) {
571  m_usersCombo->setItemText(m_usersCombo->currentIndex(), oldUser);
572  }
573  m_groupsCombo->addItems(m_defaultGroups);
574  if (m_defaultGroups.contains(oldGroup)) {
575  m_groupsCombo->setItemText(m_groupsCombo->currentIndex(), oldGroup);
576  }
577  } else {
578  m_usersCombo->addItems(m_users);

794 KACL KACLListView::itemsToACL(bool defaults) const
795 {
796  KACL newACL(0);
797  bool atLeastOneEntry = false;
798  ACLUserPermissionsList users;
799  ACLGroupPermissionsList groups;
800  QTreeWidgetItemIterator it(const_cast<KACLListView *>(this));
801  while (QTreeWidgetItem *qlvi = *it) {
802  ++it;
803  const KACLListViewItem *item = static_cast<KACLListViewItem *>(qlvi);
804  if (item->isDefault != defaults) {
805  continue;
806  }
807  atLeastOneEntry = true;
808  switch (item->type) {
809  case User:
810  newACL.setOwnerPermissions(item->value);
811  break;
812  case Group:
813  newACL.setOwningGroupPermissions(item->value);
814  break;
815  case Others:
816  newACL.setOthersPermissions(item->value);
817  break;
818  case Mask:
819  newACL.setMaskPermissions(item->value);
820  break;
821  case NamedUser:
822  users.append(qMakePair(item->text(1), item->value));
823  break;
824  case NamedGroup:
825  groups.append(qMakePair(item->text(1), item->value));
826  break;
827  default:
828  break;
829  }
830  }
831  if (atLeastOneEntry) {
832  newACL.setAllUserPermissions(users);
833  newACL.setAllGroupPermissions(groups);
834  if (newACL.isValid()) {
835  return newACL;
836  }
837  }
838  return KACL();

1012 void KACLListView::slotAddEntry()
1013 {
1014  int allowedTypes = NamedUser | NamedGroup;
1015  if (!m_hasMask) {
1016  allowedTypes |= Mask;
1017  }
1018  int allowedDefaultTypes = NamedUser | NamedGroup;
1019  if (!findDefaultItemByType(Mask)) {
1020  allowedDefaultTypes |= Mask;
1021  }
1022  if (!hasDefaultEntries()) {
1023  allowedDefaultTypes |= User | Group;
1024  }
1025  EditACLEntryDialog dlg(this, 0,
1026  allowedUsers(false), allowedGroups(false),
1027  allowedUsers(true), allowedGroups(true),
1028  allowedTypes, allowedDefaultTypes, m_allowDefaults);
1029  dlg.exec();
1030  KACLListViewItem *item = dlg.item();
1031  if (!item) {
1032  return; // canceled
1033  }
1034  if (item->type == Mask && !item->isDefault) {
1035  m_hasMask = true;
1036  m_mask = item->value;
1037  }
1038  if (item->isDefault && !hasDefaultEntries()) {
1039  // first default entry, fill in what is needed
1040  if (item->type != User) {
1041  unsigned short v = findDefaultItemByType(User)->value;
1042  new KACLListViewItem(this, User, v, true);
1043  }
1044  if (item->type != Group) {
1045  unsigned short v = findDefaultItemByType(Group)->value;
1046  new KACLListViewItem(this, Group, v, true);
1047  }
1048  if (item->type != Others) {
1049  unsigned short v = findDefaultItemByType(Others)->value;
1050  new KACLListViewItem(this, Others, v, true);
1051  }
1052  }
1053  const KACLListViewItem *defaultMaskItem = findDefaultItemByType(Mask);
1054  if (item->isDefault && !defaultMaskItem) {
1055  unsigned short v = calculateMaskValue(true);
1056  new KACLListViewItem(this, Mask, v, true);
1057  }
1058  if (!item->isDefault && !m_hasMask &&
1059  (item->type == Group
1060  || item->type == NamedUser
1061  || item->type == NamedGroup)) {
1062  // auto-add a mask entry
1063  unsigned short v = calculateMaskValue(false);
1064  new KACLListViewItem(this, Mask, v, false);
1065  m_hasMask = true;
1066  m_mask = v;
1067  }
1068  calculateEffectiveRights();
1069  sortItems(sortColumn(), Qt::AscendingOrder);
1070  setCurrentItem(item);
1071  // QTreeWidget doesn't seem to emit, in this case, and we need to update
1072  // the buttons...
1073  if (topLevelItemCount() == 1) {
1074  emit currentItemChanged(item, item);
1075  }

1116 {

1118  while (*it) {
1119  KACLListViewItem *item = static_cast<KACLListViewItem *>(*it);
1120  ++it;
1121  /* First check if it's a mask entry and if so, make sure that there is

1142  (item->type == User

1144  || item->type == Others)) {
