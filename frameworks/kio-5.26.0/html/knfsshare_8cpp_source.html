60 bool KNFSShare::KNFSSharePrivate::findExportsFile()
61 {
62  KConfig knfsshare(QStringLiteral("knfsshare"));
63  KConfigGroup config(&knfsshare, "General");
64  exportsFile = config.readPathEntry("exportsFile", QString());
65 
66  if (QFile::exists(exportsFile)) {
67  return true;
68  }
69 
70  if (QFile::exists(QStringLiteral("/etc/exports"))) {
71  exportsFile = QStringLiteral("/etc/exports");
72  } else {
73  //qDebug() << "Could not find exports file! /etc/exports doesn't exist. Configure it in share/config/knfsshare, [General], exportsFile=....";
74  return false;
75  }
76 
77  config.writeEntry("exportsFile", exportsFile);
78  return true;
79 }

85 bool KNFSShare::KNFSSharePrivate::readExportsFile()
86 {
87  QFile f(exportsFile);
88 
89  //qDebug() << exportsFile;
90 
91  if (!f.open(QIODevice::ReadOnly)) {
92  qWarning() << "KNFSShare: Could not open" << exportsFile;
93  return false;
94  }
95 
96  sharedPaths.clear();
97 
98  QTextStream s(&f);
99 
100  bool continuedLine = false; // is true if the line before ended with a backslash
101  QString completeLine;
102 
103  while (!s.atEnd()) {
104  QString currentLine = s.readLine().trimmed();
105 
106  if (continuedLine) {
107  completeLine += currentLine;
108  continuedLine = false;
109  } else {
110  completeLine = currentLine;
111  }
112 
113  // is the line continued in the next line ?
114  if (completeLine.endsWith(QLatin1Char('\\'))) {
115  continuedLine = true;
116  // remove the ending backslash
117  completeLine.chop(1);
118  continue;
119  }
120 
121  // comments or empty lines
122  if (completeLine.startsWith(QLatin1Char('#')) || completeLine.isEmpty()) {
123  continue;
124  }
125 
126  QString path;
127 
128  // Handle quotation marks
129  if (completeLine[0] == QLatin1Char('\"')) {
130  int i = completeLine.indexOf(QLatin1Char('"'), 1);
131  if (i == -1) {
132  qWarning() << "KNFSShare: Parse error: Missing quotation mark:" << completeLine;
133  continue;
134  }
135  path = completeLine.mid(1, i - 1);
136 
137  } else { // no quotation marks
138  int i = completeLine.indexOf(QLatin1Char(' '));
139  if (i == -1) {
140  i = completeLine.indexOf(QLatin1Char('\t'));
141  }
142 
143  if (i == -1) {
144  path = completeLine;
145  } else {
146  path = completeLine.left(i);
147  }
148 
149  }
150 
151  //qDebug() << "KNFSShare: Found path: " << path;
152 
153  if (!path.isEmpty()) {
154  // normalize path
155  if (!path.endsWith(QLatin1Char('/'))) {
156  path += QLatin1Char('/');
157  }
158 
159  sharedPaths.insert(path);
160  }
161  }
162 
163  return true;
164 }
