249 const char *Ftp::ftpResponse(int iOffset)
250 {
251  Q_ASSERT(m_control); // must have control connection socket
252  const char *pTxt = m_lastControlLine.data();
253 
254  // read the next line ...
255  if (iOffset < 0) {
256  int iMore = 0;
257  m_iRespCode = 0;
258 
259  if (!pTxt) {
260  return 0; // avoid using a NULL when calling atoi.
261  }
262 
263  // If the server sends a multiline response starting with
264  // "nnn-text" we loop here until a final "nnn text" line is
265  // reached. Only data from the final line will be stored.
266  do {
267  while (!m_control->canReadLine() && m_control->waitForReadyRead((readTimeout() * 1000))) {}
268  m_lastControlLine = m_control->readLine();
269  pTxt = m_lastControlLine.data();
270  int iCode = atoi(pTxt);
271  if (iMore == 0) {
272  // first line
273  qCDebug(KIO_FTP) << " > " << pTxt;
274  if (iCode >= 100) {
275  m_iRespCode = iCode;
276  if (pTxt[3] == '-') {
277  // marker for a multiple line response
278  iMore = iCode;
279  }
280  } else {
281  qCWarning(KIO_FTP) << "Cannot parse valid code from line" << pTxt;
282  }
283  } else {
284  // multi-line
285  qCDebug(KIO_FTP) << " > " << pTxt;
286  if (iCode >= 100 && iCode == iMore && pTxt[3] == ' ') {
287  iMore = 0;
288  }
289  }
290  } while (iMore != 0);
291  qCDebug(KIO_FTP) << "resp> " << pTxt;
292 
293  m_iRespType = (m_iRespCode > 0) ? m_iRespCode / 100 : 0;
294  }
295 
296  // return text with offset ...
297  while (iOffset-- > 0 && pTxt[0]) {
298  pTxt++;
299  }
300  return pTxt;
301 }

426  QString errorMessage;

428  Q_FOREACH (const QString &proxyUrl, m_proxyUrls) {

463 bool Ftp::ftpOpenControlConnection(const QString &host, int port)
464 {
465  // implicitly close, then try to open a new connection ...
466  closeConnection();
467  QString sErrorMsg;
468 
469  // now connect to the server and read the login message ...
470  if (port == 0) {
471  port = 21; // default FTP port
472  }
473  m_control = synchronousConnectToHost(host, port);
474  connect(m_control, SIGNAL(proxyAuthenticationRequired(QNetworkProxy,QAuthenticator*)),
475  this, SLOT(proxyAuthentication(QNetworkProxy,QAuthenticator*)));
476  int iErrorCode = m_control->state() == QAbstractSocket::ConnectedState ? 0 : ERR_CANNOT_CONNECT;
477 
478  // on connect success try to read the server message...
479  if (iErrorCode == 0) {
480  const char *psz = ftpResponse(-1);
481  if (m_iRespType != 2) {
482  // login not successful, do we have an message text?
483  if (psz[0]) {
484  sErrorMsg = i18n("%1 (Error %2)", host, remoteEncoding()->decode(psz).trimmed());
485  }
486  iErrorCode = ERR_CANNOT_CONNECT;
487  }
488  } else {
489  if (m_control->error() == QAbstractSocket::HostNotFoundError) {
490  iErrorCode = ERR_UNKNOWN_HOST;
491  }
492 
493  sErrorMsg = QStringLiteral("%1: %2").arg(host, m_control->errorString());
494  }
495 
496  // if there was a problem - report it ...
497  if (iErrorCode == 0) { // OK, return success
498  return true;
499  }
500  closeConnection(); // clean-up on error
501  error(iErrorCode, sErrorMsg);
502  return false;
503 }

512 bool Ftp::ftpLogin(bool *userChanged)
513 {
514  infoMessage(i18n("Sending login information"));
515 
516  Q_ASSERT(!m_bLoggedOn);
517 
518  QString user(m_user);
519  QString pass(m_pass);
520 
521  if (config()->readEntry("EnableAutoLogin", false)) {
522  QString au = config()->readEntry("autoLoginUser");
523  if (!au.isEmpty()) {
524  user = au;
525  pass = config()->readEntry("autoLoginPass");
526  }
527  }
528 
529  AuthInfo info;
530  info.url.setScheme(QStringLiteral("ftp"));
531  info.url.setHost(m_host);
532  if (m_port > 0 && m_port != DEFAULT_FTP_PORT) {
533  info.url.setPort(m_port);
534  }
535  if (!user.isEmpty()) {
536  info.url.setUserName(user);
537  }
538 
539  // Check for cached authentication first and fallback to
540  // anonymous login when no stored credentials are found.
541  if (!config()->readEntry("TryAnonymousLoginFirst", false) &&
542  pass.isEmpty() && checkCachedAuthentication(info)) {
543  user = info.username;
544  pass = info.password;
545  }
546 
547  // Try anonymous login if both username/password
548  // information is blank.
549  if (user.isEmpty() && pass.isEmpty()) {
550  user = QStringLiteral(FTP_LOGIN);
551  pass = QStringLiteral(FTP_PASSWD);
552  }
553 
554  QByteArray tempbuf;
555  QString lastServerResponse;
556  int failedAuth = 0;
557  bool promptForRetry = false;
558 
559  // Give the user the option to login anonymously...
560  info.setExtraField(QStringLiteral("anonymous"), false);
561 
562  do {
563  // Check the cache and/or prompt user for password if 1st
564  // login attempt failed OR the user supplied a login name,
565  // but no password.
566  if (failedAuth > 0 || (!user.isEmpty() && pass.isEmpty())) {
567  QString errorMsg;
568  qCDebug(KIO_FTP) << "Prompting user for login info...";
569 
570  // Ask user if we should retry after when login fails!
571  if (failedAuth > 0 && promptForRetry) {
572  errorMsg = i18n("Message sent:\nLogin using username=%1 and "
573  "password=[hidden]\n\nServer replied:\n%2\n\n"
574  , user, lastServerResponse);
575  }
576 
577  if (user != QStringLiteral(FTP_LOGIN)) {
578  info.username = user;
579  }
580 
581  info.prompt = i18n("You need to supply a username and a password "
582  "to access this site.");
583  info.commentLabel = i18n("Site:");
584  info.comment = i18n("<b>%1</b>", m_host);
585  info.keepPassword = true; // Prompt the user for persistence as well.
586  info.setModified(false); // Default the modified flag since we reuse authinfo.
587 
588  const bool disablePassDlg = config()->readEntry("DisablePassDlg", false);
589  if (disablePassDlg) {
590  error(ERR_USER_CANCELED, m_host);
591  return false;
592  }
593  const int errorCode = openPasswordDialogV2(info, errorMsg);
594  if (errorCode) {
595  error(errorCode, QString());
596  return false;
597  } else {
598  // User can decide go anonymous using checkbox
599  if (info.getExtraField(QStringLiteral("anonymous")).toBool()) {
600  user = QStringLiteral(FTP_LOGIN);
601  pass = QStringLiteral(FTP_PASSWD);
602  } else {
603  user = info.username;
604  pass = info.password;
605  }
606  promptForRetry = true;
607  }
608  }
609 
610  tempbuf = "USER ";
611  tempbuf += user.toLatin1();
612  if (m_proxyURL.isValid()) {
613  tempbuf += '@';
614  tempbuf += m_host.toLatin1();
615  if (m_port > 0 && m_port != DEFAULT_FTP_PORT) {
616  tempbuf += ':';
617  tempbuf += QString::number(m_port).toLatin1();
618  }
619  }
620 
621  qCDebug(KIO_FTP) << "Sending Login name: " << tempbuf;
622 
623  bool loggedIn = (ftpSendCmd(tempbuf) && (m_iRespCode == 230));
624  bool needPass = (m_iRespCode == 331);
625  // Prompt user for login info if we do not
626  // get back a "230" or "331".
627  if (!loggedIn && !needPass) {
628  lastServerResponse = ftpResponse(0);
629  qCDebug(KIO_FTP) << "Login failed: " << lastServerResponse;
630  ++failedAuth;
631  continue; // Well we failed, prompt the user please!!
632  }
633 
634  if (needPass) {
635  tempbuf = "PASS ";
636  tempbuf += pass.toLatin1();
637  qCDebug(KIO_FTP) << "Sending Login password: " << "[protected]";
638  loggedIn = (ftpSendCmd(tempbuf) && (m_iRespCode == 230));
639  }
640 
641  if (loggedIn) {
642  // Make sure the user name changed flag is properly set.
643  if (userChanged) {
644  *userChanged = (!m_user.isEmpty() && (m_user != user));
645  }
646 
647  // Do not cache the default login!!
648  if (user != QLatin1String(FTP_LOGIN) && pass != QLatin1String(FTP_PASSWD)) {
649  // Update the username in case it was changed during login.
650  if (!m_user.isEmpty()) {
651  info.url.setUserName(user);
652  m_user = user;
653  }
654 
655  // Cache the password if the user requested it.
656  if (info.keepPassword) {
657  cacheAuthentication(info);
658  }
659  }
660  failedAuth = -1;
661  } else {
662  // some servers don't let you login anymore
663  // if you fail login once, so restart the connection here
664  lastServerResponse = ftpResponse(0);
665  if (!ftpOpenControlConnection()) {
666  return false;
667  }
668  }
669  } while (++failedAuth);
670 
671  qCDebug(KIO_FTP) << "Login OK";
672  infoMessage(i18n("Login OK"));
673 
674  // Okay, we're logged in. If this is IIS 4, switch dir listing style to Unix:
675  // Thanks to jk@soegaard.net (Jens Kristian Sgaard) for this hint
676  if (ftpSendCmd(QByteArrayLiteral("SYST")) && (m_iRespType == 2)) {
677  if (!qstrncmp(ftpResponse(0), "215 Windows_NT", 14)) { // should do for any version
678  ftpSendCmd(QByteArrayLiteral("site dirstyle"));
679  // Check if it was already in Unix style
680  // Patch from Keith Refson <Keith.Refson@earth.ox.ac.uk>
681  if (!qstrncmp(ftpResponse(0), "200 MSDOS-like directory output is on", 37))
682  //It was in Unix style already!
683  {
684  ftpSendCmd(QByteArrayLiteral("site dirstyle"));
685  }
686  // windows won't support chmod before KDE konquers their desktop...
687  m_extControl |= chmodUnknown;
688  }
689  } else {
690  qCWarning(KIO_FTP) << "SYST failed";
691  }
692 
693  if (config()->readEntry("EnableAutoLoginMacro", false)) {
694  ftpAutoLoginMacro();
695  }
696 
697  // Get the current working directory
698  qCDebug(KIO_FTP) << "Searching for pwd";
699  if (!ftpSendCmd(QByteArrayLiteral("PWD")) || (m_iRespType != 2)) {
700  qCDebug(KIO_FTP) << "Couldn't issue pwd command";
701  error(ERR_CANNOT_LOGIN, i18n("Could not login to %1.", m_host)); // or anything better ?
702  return false;

757 bool Ftp::ftpSendCmd(const QByteArray &cmd, int maxretries)
758 {
759  Q_ASSERT(m_control); // must have control connection socket
760 
761  if (cmd.indexOf('\r') != -1 || cmd.indexOf('\n') != -1) {
762  qCWarning(KIO_FTP) << "Invalid command received (contains CR or LF):"
763  << cmd.data();

765  return false;
766  }
767 
768  // Don't print out the password...
769  bool isPassCmd = (cmd.left(4).toLower() == "pass");
770 #if 0
771  if (!isPassCmd) {
772  qDebug() << "send> " << cmd.data();
773  } else {
774  qDebug() << "send> pass [protected]";
775  }
776 #endif
777 
778  // Send the message...
779  QByteArray buf = cmd;
780  buf += "\r\n"; // Yes, must use CR/LF - see http://cr.yp.to/ftp/request.html
781  int num = m_control->write(buf);
782  while (m_control->bytesToWrite() && m_control->waitForBytesWritten()) {}
783 
784  // If we were able to successfully send the command, then we will
785  // attempt to read the response. Otherwise, take action to re-attempt
786  // the login based on the maximum number of retries specified...
787  if (num > 0) {
788  ftpResponse(-1);
789  } else {
790  m_iRespType = m_iRespCode = 0;
791  }
792 
793  // If respCh is NULL or the response is 421 (Timed-out), we try to re-send
794  // the command based on the value of maxretries.
795  if ((m_iRespType <= 0) || (m_iRespCode == 421)) {
796  // We have not yet logged on...
797  if (!m_bLoggedOn) {
798  // The command was sent from the ftpLogin function, i.e. we are actually
799  // attempting to login in. NOTE: If we already sent the username, we
800  // return false and let the user decide whether (s)he wants to start from
801  // the beginning...
802  if (maxretries > 0 && !isPassCmd) {
803  closeConnection();
804  if (ftpOpenConnection(loginDefered)) {
805  ftpSendCmd(cmd, maxretries - 1);
806  }
807  }
808 
809  return false;
810  } else {
811  if (maxretries < 1) {
812  return false;
813  } else {
814  qCDebug(KIO_FTP) << "Was not able to communicate with " << m_host
815  << "Attempting to re-establish connection.";
816 
817  closeConnection(); // Close the old connection...
818  openConnection(); // Attempt to re-establish a new connection...
819 
820  if (!m_bLoggedOn) {
821  if (m_control) { // if openConnection succeeded ...
822  qCDebug(KIO_FTP) << "Login failure, aborting";
823  error(ERR_CANNOT_LOGIN, m_host);
824  closeConnection();
825  }
826  return false;
827  }
828 
829  qCDebug(KIO_FTP) << "Logged back in, re-issuing command";
830 
831  // If we were able to login, resend the command...
832  if (maxretries) {
833  maxretries--;
834  }
835 
836  return ftpSendCmd(cmd, maxretries);
837  }
838  }
839  }
