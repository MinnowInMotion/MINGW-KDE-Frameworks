349 void KUrlNavigator::Private::openPathSelectorMenu()
350 {
351  if (m_navButtons.count() <= 0) {
352  return;
353  }
354 
355  const QUrl firstVisibleUrl = m_navButtons.first()->url();
356 
357  QString spacer;
358  QPointer<QMenu> popup = new QMenu(q);
359  popup->setLayoutDirection(Qt::LeftToRight);
360 
361  const QUrl placeUrl = retrievePlaceUrl();
362  int idx = placeUrl.path().count(QLatin1Char('/')); // idx points to the first directory
363  // after the place path
364 
365  const QString path = m_history[m_historyIndex].url.path();
366  QString dirName = path.section(QLatin1Char('/'), idx, idx);
367  if (dirName.isEmpty()) {
368  if (placeUrl.isLocalFile()) {
369  dirName = QStringLiteral("/");
370  } else {
371  dirName = placeUrl.toDisplayString();
372  }
373  }
374  do {
375  const QString text = spacer + dirName;
376 
377  QAction *action = new QAction(text, popup);
378  const QUrl currentUrl = buttonUrl(idx);
379  if (currentUrl == firstVisibleUrl) {
380  popup->addSeparator();
381  }
382  action->setData(QVariant(currentUrl.toString()));
383  popup->addAction(action);
384 
385  ++idx;
386  spacer.append(" ");
387  dirName = path.section('/', idx, idx);
388  } while (!dirName.isEmpty());
389 
390  const QPoint pos = q->mapToGlobal(m_dropDownButton->geometry().bottomRight());
391  const QAction *activatedAction = popup->exec(pos);
392  if (activatedAction != 0) {
393  const QUrl url(activatedAction->data().toString());
394  q->setLocationUrl(url);
395  }
396 
397  // Delete the menu, unless it has been deleted in its own nested event loop already.
398  if (popup) {
399  popup->deleteLater();
400  }

558 void KUrlNavigator::Private::updateButtons(int startIndex)
559 {
560  QUrl currentUrl = q->locationUrl();
561  if (!currentUrl.isValid()) { // QFileDialog::setDirectory not called yet
562  return;
563  }
564 
565  const QString path = currentUrl.path();
566 
567  bool createButton = false;
568  const int oldButtonCount = m_navButtons.count();
569 
570  int idx = startIndex;
571  bool hasNext = true;
572  do {
573  createButton = (idx - startIndex >= oldButtonCount);
574  const bool isFirstButton = (idx == startIndex);
575  const QString dirName = path.section(QLatin1Char('/'), idx, idx);
576  hasNext = isFirstButton || !dirName.isEmpty();
577  if (hasNext) {
578  KUrlNavigatorButton *button = 0;
579  if (createButton) {
580  button = new KUrlNavigatorButton(buttonUrl(idx), q);
581  button->installEventFilter(q);
582  button->setForegroundRole(QPalette::WindowText);
583  connect(button, SIGNAL(urlsDropped(QUrl,QDropEvent*)),
584  q, SLOT(dropUrls(QUrl,QDropEvent*)));
585  connect(button, SIGNAL(clicked(QUrl,Qt::MouseButton)),
586  q, SLOT(slotNavigatorButtonClicked(QUrl,Qt::MouseButton)));
587  connect(button, SIGNAL(finishedTextResolving()),
588  q, SLOT(updateButtonVisibility()));
589  appendWidget(button);
590  } else {
591  button = m_navButtons[idx - startIndex];
592  button->setUrl(buttonUrl(idx));
593  }
594 
595  if (isFirstButton) {
596  button->setText(firstButtonText());
597  }
598  button->setActive(q->isActive());
599 
600  if (createButton) {
601  if (!isFirstButton) {
602  setTabOrder(m_navButtons.last(), button);
603  }
604  m_navButtons.append(button);
605  }
606 
607  ++idx;
608  button->setActiveSubDirectory(path.section(QLatin1Char('/'), idx, idx));
609  }
610  } while (hasNext);
611 
612  // delete buttons which are not used anymore
613  const int newButtonCount = idx - startIndex;
614  if (newButtonCount < oldButtonCount) {
615  const QList<KUrlNavigatorButton *>::iterator itBegin = m_navButtons.begin() + newButtonCount;
616  const QList<KUrlNavigatorButton *>::iterator itEnd = m_navButtons.end();
617  QList<KUrlNavigatorButton *>::iterator it = itBegin;
618  while (it != itEnd) {
619  (*it)->hide();
620  (*it)->deleteLater();
621  ++it;
622  }
623  m_navButtons.erase(itBegin, itEnd);
624  }
625 
626  setTabOrder(m_dropDownButton, m_navButtons.first());
627  setTabOrder(m_navButtons.last(), m_toggleEditableMode);
628 
629  updateButtonVisibility();
630 }

632 void KUrlNavigator::Private::updateButtonVisibility()
633 {
634  if (m_editable) {
635  return;
636  }
637 
638  const int buttonsCount = m_navButtons.count();
639  if (buttonsCount == 0) {
640  m_dropDownButton->hide();
641  return;
642  }
643 
644  // Subtract all widgets from the available width, that must be shown anyway
645  int availableWidth = q->width() - m_toggleEditableMode->minimumWidth();
646 
647  if ((m_placesSelector != 0) && m_placesSelector->isVisible()) {
648  availableWidth -= m_placesSelector->width();
649  }
650 
651  if ((m_protocols != 0) && m_protocols->isVisible()) {
652  availableWidth -= m_protocols->width();
653  }
654 
655  // Check whether buttons must be hidden at all...
656  int requiredButtonWidth = 0;
657  foreach (const KUrlNavigatorButton *button, m_navButtons) {
658  requiredButtonWidth += button->minimumWidth();
659  }
660 
661  if (requiredButtonWidth > availableWidth) {
662  // At least one button must be hidden. This implies that the
663  // drop-down button must get visible, which again decreases the
664  // available width.
665  availableWidth -= m_dropDownButton->width();
666  }
667 
668  // Hide buttons...
669  QList<KUrlNavigatorButton *>::const_iterator it = m_navButtons.constEnd();
670  const QList<KUrlNavigatorButton *>::const_iterator itBegin = m_navButtons.constBegin();
671  bool isLastButton = true;
672  bool hasHiddenButtons = false;
673 
674  QLinkedList<KUrlNavigatorButton *> buttonsToShow;
675  while (it != itBegin) {
676  --it;
677  KUrlNavigatorButton *button = (*it);
678  availableWidth -= button->minimumWidth();
679  if ((availableWidth <= 0) && !isLastButton) {
680  button->hide();
681  hasHiddenButtons = true;
682  } else {
683  // Don't show the button immediately, as setActive()
684  // might change the size and a relayout gets triggered
685  // after showing the button. So the showing of all buttons
686  // is postponed until all buttons have the correct
687  // activation state.
688  buttonsToShow.append(button);
689  }
690  isLastButton = false;
691  }
692 
693  // All buttons have the correct activation state and
694  // can be shown now
695  foreach (KUrlNavigatorButton *button, buttonsToShow) {
696  button->show();
697  }
698 
699  if (hasHiddenButtons) {
700  m_dropDownButton->show();
701  } else {
702  // Check whether going upwards is possible. If this is the case, show the drop-down button.
703  QUrl url(m_navButtons.front()->url());
704  const bool visible = !url.matches(KIO::upUrl(url), QUrl::StripTrailingSlash) && (url.scheme() != QLatin1String("nepomuksearch"));
705  m_dropDownButton->setVisible(visible);
706  }
707 }

709 QString KUrlNavigator::Private::firstButtonText() const
710 {
711  QString text;
712 
713  // The first URL navigator button should get the name of the
714  // place instead of the directory name
715  if ((m_placesSelector != 0) && !m_showFullPath) {
716  text = m_placesSelector->selectedPlaceText();
717  }

719  if (text.isEmpty()) {

1008 void KUrlNavigator::setLocationUrl(const QUrl &newUrl)
1009 {
1010  if (newUrl == locationUrl()) {
1011  return;
1012  }
1013 
1014  QUrl url = newUrl.adjusted(QUrl::NormalizePathSegments);
1015 
1016  if ((url.scheme() == QLatin1String("tar")) || (url.scheme() == QLatin1String("zip"))) {
1017  // The URL represents a tar- or zip-file. Check whether
1018  // the URL is really part of the tar- or zip-file, otherwise
1019  // replace it by the local path again.
1020  bool insideCompressedPath = d->isCompressedPath(url);
1021  if (!insideCompressedPath) {
1022  QUrl prevUrl = url;
1023  QUrl parentUrl = KIO::upUrl(url);
1024  while (parentUrl != prevUrl) {
1025  if (d->isCompressedPath(parentUrl)) {
1026  insideCompressedPath = true;
1027  break;
1028  }
1029  prevUrl = parentUrl;
1030  parentUrl = KIO::upUrl(parentUrl);
1031  }
1032  }
1033  if (!insideCompressedPath) {
1034  // drop the tar: or zip: protocol since we are not
1035  // inside the compressed path
1036  url.setScheme(QStringLiteral("file"));
1037  }
1038  }
1039 
1040  // Check whether current history element has the same URL.
1041  // If this is the case, just ignore setting the URL.
1042  const LocationData &data = d->m_history[d->m_historyIndex];
1043  const bool isUrlEqual = url.matches(locationUrl(), QUrl::StripTrailingSlash) ||
1044  (!url.isValid() && url.matches(data.url, QUrl::StripTrailingSlash));
1045  if (isUrlEqual) {
1046  return;
1047  }
1048 
1049  emit urlAboutToBeChanged(url);
1050 
1051  if (d->m_historyIndex > 0) {
1052  // If an URL is set when the history index is not at the end (= 0),
1053  // then clear all previous history elements so that a new history
1054  // tree is started from the current position.
1055  QList<LocationData>::iterator begin = d->m_history.begin();
1056  QList<LocationData>::iterator end = begin + d->m_historyIndex;
1057  d->m_history.erase(begin, end);
1058  d->m_historyIndex = 0;
1059  }
1060 
1061  Q_ASSERT(d->m_historyIndex == 0);
1062  LocationData newData;
1063  newData.url = url;
1064  d->m_history.insert(0, newData);
1065 
1066  // Prevent an endless growing of the history: remembering
1067  // the last 100 Urls should be enough...
1068  const int historyMax = 100;
1069  if (d->m_history.size() > historyMax) {
1070  QList<LocationData>::iterator begin = d->m_history.begin() + historyMax;
1071  QList<LocationData>::iterator end = d->m_history.end();
1072  d->m_history.erase(begin, end);
1073  }
1074 
1075  emit historyChanged();
1076  emit urlChanged(url);
1077 
1078  d->updateContent();
1079 
1080  requestActivation();
1081 }
