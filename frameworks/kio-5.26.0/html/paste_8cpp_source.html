58 static KIO::Job *pasteClipboardUrls(const QMimeData *mimeData, const QUrl &destDir, KIO::JobFlags flags = KIO::DefaultFlags)
59 {
60  const QList<QUrl> urls = KUrlMimeData::urlsFromMimeData(mimeData, KUrlMimeData::PreferLocalUrls);
61  if (!urls.isEmpty()) {
62  const bool move = KIO::isClipboardDataCut(mimeData);
63  KIO::Job *job = 0;
64  if (move) {
65  job = KIO::move(urls, destDir, flags);
66  } else {
67  job = KIO::copy(urls, destDir, flags);
68  }
69  return job;
70  }
71  return 0;
72 }

74 static QUrl getNewFileName(const QUrl &u, const QString &text, const QString &suggestedFileName, QWidget *widget)
75 {
76  bool ok;
77  QString dialogText(text);
78  if (dialogText.isEmpty()) {
79  dialogText = i18n("Filename for clipboard content:");
80  }
81  QString file = QInputDialog::getText(widget, QString(), dialogText, QLineEdit::Normal, suggestedFileName, &ok);
82  if (!ok) {
83  return QUrl();
84  }
85 
86  QUrl myurl(u);
87  myurl.setPath(myurl.path() + '/' + file);
88 
89  KIO::StatJob *job = KIO::stat(myurl, myurl.isLocalFile() ? KIO::HideProgressInfo : KIO::DefaultFlags);
90  job->setDetails(0);
91  job->setSide(KIO::StatJob::DestinationSide);
92  KJobWidgets::setWindow(job, widget);
93 
94  // Check for existing destination file.
95  // When we were using CopyJob, we couldn't let it do that (would expose
96  // an ugly tempfile name as the source URL)
97  // And now we're using a put job anyway, no destination checking included.
98  if (job->exec()) {
99  //qDebug() << "Paste will overwrite file. Prompting...";
100  KIO::RenameDialog dlg(widget,
101  i18n("File Already Exists"),
102  u,
103  myurl,
104  KIO::RenameDialog_Overwrite);
105  KIO::RenameDialog_Result res = static_cast<KIO::RenameDialog_Result>(dlg.exec());
106 
107  if (res == KIO::Result_Rename) {
108  myurl = dlg.newDestUrl();
109  } else if (res == KIO::Result_Cancel) {
110  return QUrl();
111  } else if (res == KIO::Result_Overwrite) {
112  // OK, proceed
113  }
114  }
115 
116  return myurl;
117 }

126 static QByteArray chooseFormatAndUrl(const QUrl &u, const QMimeData *mimeData,
127  const QStringList &formats,
128  const QString &text,
129  const QString &suggestedFileName,
130  QWidget *widget,
131  bool clipboard,
132  QUrl *newUrl)
133 {
134  QMimeDatabase db;
135  QStringList formatLabels;
136  for (int i = 0; i < formats.size(); ++i) {
137  const QString &fmt = formats[i];
138  QMimeType mime = db.mimeTypeForName(fmt);
139  if (mime.isValid()) {
140  formatLabels.append(i18n("%1 (%2)", mime.comment(), fmt));
141  } else {
142  formatLabels.append(fmt);
143  }
144  }
145 
146  QString dialogText(text);
147  if (dialogText.isEmpty()) {
148  dialogText = i18n("Filename for clipboard content:");
149  }
150  //using QString() instead of QString::null didn't compile (with gcc 3.2.3), because the ctor was mistaken as a function declaration, Alex //krazy:exclude=nullstrassign
151  KIO::PasteDialog dlg(QString::null, dialogText, suggestedFileName, formatLabels, widget, clipboard); //krazy:exclude=nullstrassign
152 
153  if (dlg.exec() != QDialog::Accepted) {
154  return QByteArray();
155  }
156 
157  if (clipboard && dlg.clipboardChanged()) {
158  KMessageBox::sorry(widget,
159  i18n("The clipboard has changed since you used 'paste': "
160  "the chosen data format is no longer applicable. "
161  "Please copy again what you wanted to paste."));
162  return QByteArray();
163  }
164 
165  const QString result = dlg.lineEditText();
166  const QString chosenFormat = formats[ dlg.comboItem() ];
167 
168  //qDebug() << " result=" << result << " chosenFormat=" << chosenFormat;
169  *newUrl = u;
170  newUrl->setPath(newUrl->path() + '/' + result);
171  // In Qt3, the result of clipboard()->mimeData() only existed until the next
172  // event loop run (see dlg.exec() above), so we re-fetched it.
173  // TODO: This should not be necessary with Qt5; remove this conditional
174  // and test that it still works.
175  if (clipboard) {
176  mimeData = QApplication::clipboard()->mimeData();
177  }
178  const QByteArray ba = mimeData->data(chosenFormat);
179  return ba;
180 }

182 static QStringList extractFormats(const QMimeData *mimeData)
183 {
184  QStringList formats;
185  const QStringList allFormats = mimeData->formats();
186  Q_FOREACH (const QString &format, allFormats) {
187  if (format == QLatin1String("application/x-qiconlist")) { // Q3IconView and kde4's libkonq
188  continue;
189  }
190  if (format == QLatin1String("application/x-kde-cutselection")) { // see isClipboardDataCut
191  continue;
192  }
193  if (format == QLatin1String("application/x-kde-suggestedfilename")) {
194  continue;
195  }
196  if (format.startsWith(QLatin1String("application/x-qt-"))) { // Qt-internal
197  continue;
198  }
199  if (format.startsWith(QLatin1String("x-kmail-drag/"))) { // app-internal
200  continue;
201  }
202  if (!format.contains(QLatin1Char('/'))) { // e.g. TARGETS, MULTIPLE, TIMESTAMP
203  continue;
204  }
205  formats.append(format);
206  }
207  return formats;
208 }

215 KIO::Job *pasteMimeDataImpl(const QMimeData *mimeData, const QUrl &destUrl,
216  const QString &dialogText, QWidget *widget,
217  bool clipboard)
218 {
219  QByteArray ba;
220  const QString suggestedFilename = QString::fromUtf8(mimeData->data(QStringLiteral("application/x-kde-suggestedfilename")));
221 
222  // Now check for plain text
223  // We don't want to display a mimetype choice for a QTextDrag, those mimetypes look ugly.
224  if (mimeData->hasText()) {
225  ba = mimeData->text().toLocal8Bit(); // encoding OK?
226  } else {
227  const QStringList formats = extractFormats(mimeData);
228  if (formats.isEmpty()) {
229  return 0;
230  } else if (formats.size() > 1) {
231  QUrl newUrl;
232  ba = chooseFormatAndUrl(destUrl, mimeData, formats, dialogText, suggestedFilename, widget, clipboard, &newUrl);
233  if (ba.isEmpty()) {
234  return 0;
235  }
236  return putDataAsyncTo(newUrl, ba, widget, KIO::Overwrite);
237  }
238  ba = mimeData->data(formats.first());
239  }
240  if (ba.isEmpty()) {
241  return 0;
242  }
243 
244  const QUrl newUrl = getNewFileName(destUrl, dialogText, suggestedFilename, widget);
245  if (newUrl.isEmpty()) {
246  return 0;
247  }
248 
249  return putDataAsyncTo(newUrl, ba, widget, KIO::Overwrite);
250 }

253 KIOWIDGETS_EXPORT KIO::Job *KIO::pasteClipboard(const QUrl &destUrl, QWidget *widget, bool move)
254 {
255  Q_UNUSED(move);
256 
257  if (!destUrl.isValid()) {
258  KMessageBox::error(widget, i18n("Malformed URL\n%1", destUrl.errorString()));
259  qWarning() << destUrl.errorString();
260  return 0;
261  }
262 
263  // TODO: if we passed mimeData as argument, we could write unittests that don't
264  // mess up the clipboard and that don't need QtGui.
265  const QMimeData *mimeData = QApplication::clipboard()->mimeData();
266 
267  if (mimeData->hasUrls()) {
268  // We can ignore the bool move, KIO::paste decodes it
269  KIO::Job *job = pasteClipboardUrls(mimeData, destUrl);
270  if (job) {
271  KJobWidgets::setWindow(job, widget);
272  return job;
273  }
274  }
275 
276  return pasteMimeDataImpl(mimeData, destUrl, QString(), widget, true /*clipboard*/);
277 }

297 KIOWIDGETS_EXPORT QString KIO::pasteActionText(const QMimeData *mimeData, bool *enable, const KFileItem &destItem)
298 {
299  bool canPasteData = false;
300  QList<QUrl> urls;
301 
302  // mimeData can be 0 according to https://bugs.kde.org/show_bug.cgi?id=335053
303  if (mimeData) {
304  canPasteData = KIO::canPasteMimeData(mimeData);
305  urls = KUrlMimeData::urlsFromMimeData(mimeData);
306  } else {
307  qWarning() << "QApplication::clipboard()->mimeData() is 0!";
308  }
309 
310  QString text;
311  if (!urls.isEmpty() || canPasteData) {
312  // disable the paste action if no writing is supported
313  if (!destItem.isNull()) {
314  if (destItem.url().isEmpty()) {
315  *enable = false;
316  } else {
317  *enable = KFileItemListProperties(KFileItemList() << destItem).supportsWriting();
318  }
319  } else {
320  *enable = false;
321  }
322 
323  if (urls.count() == 1 && urls.first().isLocalFile()) {
324  const bool isDir = QFileInfo(urls.first().toLocalFile()).isDir();
325  text = isDir ? i18nc("@action:inmenu", "Paste One Folder") :
326  i18nc("@action:inmenu", "Paste One File");
327  } else if (!urls.isEmpty()) {
328  text = i18ncp("@action:inmenu", "Paste One Item", "Paste %1 Items", urls.count());
329  } else {
330  text = i18nc("@action:inmenu", "Paste Clipboard Contents...");
331  }
332  } else {
333  *enable = false;
334  text = i18nc("@action:inmenu", "Paste");
335  }
336  return text;
337 }
