93 // Check if the user wants to assign shorthands that are already assigned to
94 // another search provider. Invoked on every change to the shortcuts field.
95 void SearchProviderDialog::shortcutsChanged(const QString& newShorthands) {
96  // Convert all spaces to commas. A shorthand should be a single word.
97  // Assume that the user wanted to enter an alternative shorthand and hit
98  // space instead of the comma key. Save cursor position beforehand because
99  // setText() will reset it to the end, which is not what we want when
100  // backspacing something in the middle.
101  int savedCursorPosition = m_dlg.leShortcut->cursorPosition();
102  QString normalizedShorthands = QString(newShorthands).replace(' ', ',');
103  m_dlg.leShortcut->setText(normalizedShorthands);
104  m_dlg.leShortcut->setCursorPosition(savedCursorPosition);
105 
106  QHash<QString, const SearchProvider*> contenders;
107  QSet<QString> shorthands = normalizedShorthands.split(',').toSet();
108 
109  // Look at each shorthand the user entered and wade through the search
110  // provider list in search of a conflicting shorthand. Do not continue
111  // search after finding one, because shorthands should be assigned only
112  // once. Act like data inconsistencies regarding this don't exist (should
113  // probably be handled on load).
114  Q_FOREACH (const QString &shorthand, shorthands) {
115  Q_FOREACH (const SearchProvider* provider, m_providers) {
116  if (provider != m_provider && provider->keys().contains(shorthand)) {
117  contenders.insert(shorthand, provider);
118  break;
119  }
120  }
121  }
122 
123  if (!contenders.isEmpty()) {
124  if (contenders.size() == 1) {
125  m_dlg.noteLabel->setText(i18n("The shortcut \"%1\" is already assigned to \"%2\". Please choose a different one.", contenders.keys().at(0), contenders.values().at(0)->name()));
126  } else {
127  QStringList contenderList;
128  QHash<QString, const SearchProvider*>::const_iterator i = contenders.constBegin();
129  while (i != contenders.constEnd()) {
130  contenderList.append(i18nc("- web short cut (e.g. gg): what it refers to (e.g. Google)", "- %1: \"%2\"", i.key(), i.value()->name()));
131  ++i;
132  }
133 
134  m_dlg.noteLabel->setText(i18n("The following shortcuts are already assigned. Please choose different ones.\n%1", contenderList.join(QLatin1Char('\n'))));
135  }
136  m_buttons->button(QDialogButtonBox::Ok)->setEnabled(false);
137  } else {
138  m_dlg.noteLabel->clear();
139  }
140 }

142 void SearchProviderDialog::accept()
143 {
144  if ((m_dlg.leQuery->text().indexOf(QStringLiteral("\\{")) == -1)
145  && KMessageBox::warningContinueCancel(0,
146  i18n("The Shortcut URL does not contain a \\{...} placeholder for the user query.\n"
147  "This means that the same page is always going to be visited, "
148  "regardless of the text typed in with the shortcut."),
149  QString(), KGuiItem(i18n("Keep It"))) == KMessageBox::Cancel) {
150  return;
151  }
152 
153  if (!m_provider)
154  m_provider = new SearchProvider;
155 
156  const QString name = m_dlg.leName->text().trimmed();
157  const QString query = m_dlg.leQuery->text().trimmed();
158  QStringList keys = m_dlg.leShortcut->text().trimmed().toLower().split(',', QString::SkipEmptyParts);
159  keys.removeDuplicates();// #169801. Remove duplicates...
160  const QString charset = (m_dlg.cbCharset->currentIndex() ? m_dlg.cbCharset->currentText().trimmed() : QString());
161 
162  m_provider->setDirty((name != m_provider->name() || query != m_provider->query() ||
163  keys != m_provider->keys() || charset != m_provider->charset()));
164  m_provider->setName(name);
165  m_provider->setQuery(query);
166  m_provider->setKeys(keys);
167  m_provider->setCharset(charset);
168  QDialog::accept();
169 }
