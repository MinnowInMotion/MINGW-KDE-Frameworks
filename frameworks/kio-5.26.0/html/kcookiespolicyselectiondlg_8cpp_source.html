44  State validate (QString& input, int&) const Q_DECL_OVERRIDE
45  {
46  if (input.isEmpty() || (input == QLatin1String("."))) {
47  return Intermediate;
48  }
49 
50  const int length = input.length();
51 
52  for (int i = 0 ; i < length; i++) {
53  if (!input[i].isLetterOrNumber() && input[i] != '.' && input[i] != '-') {
54  return Invalid;
55  }
56  }
57 
58  return Acceptable;
59  }
