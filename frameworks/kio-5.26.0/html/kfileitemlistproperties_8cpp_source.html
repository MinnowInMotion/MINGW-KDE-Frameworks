73 void KFileItemListPropertiesPrivate::setItems(const KFileItemList &items)
74 {
75  const bool initialValue = !items.isEmpty();
76  m_items = items;
77  m_urlList = items.targetUrlList();
78  m_supportsReading = initialValue;
79  m_supportsDeleting = initialValue;
80  m_supportsWriting = initialValue;
81  m_supportsMoving = initialValue;
82  m_isDirectory = initialValue;
83  m_isLocal = true;
84  m_mimeType.clear();
85  m_mimeGroup.clear();
86 
87  QFileInfo parentDirInfo;
88  foreach (const KFileItem &item, items) {
89  const QUrl url = item.url();
90  m_isLocal = m_isLocal && url.isLocalFile();
91  m_supportsReading = m_supportsReading && KProtocolManager::supportsReading(url);
92  m_supportsDeleting = m_supportsDeleting && KProtocolManager::supportsDeleting(url);
93  m_supportsWriting = m_supportsWriting && KProtocolManager::supportsWriting(url) && item.isWritable();
94  m_supportsMoving = m_supportsMoving && KProtocolManager::supportsMoving(url);
95 
96  // For local files we can do better: check if we have write permission in parent directory
97  // TODO: if we knew about the parent KFileItem, we could even do that for remote protocols too
98 #ifndef Q_OS_WIN
99  if (m_isLocal && (m_supportsDeleting || m_supportsMoving)) {
100  const QString directory = url.adjusted(QUrl::RemoveFilename | QUrl::StripTrailingSlash).toLocalFile();
101  if (parentDirInfo.filePath() != directory) {
102  parentDirInfo.setFile(directory);
103  }
104  if (!parentDirInfo.isWritable()) {
105  m_supportsDeleting = false;
106  m_supportsMoving = false;
107  }
108  }
109 #else
110  if (m_isLocal && m_supportsDeleting) {
111  if (!QFileInfo(url.toLocalFile()).isWritable())
112  m_supportsDeleting = false;
113  }
114 #endif
115  if (m_isDirectory && !item.isDir()) {
116  m_isDirectory = false;
117  }
118  }
119 }

191 void KFileItemListPropertiesPrivate::determineMimeTypeAndGroup() const
192 {
193  if (!m_items.isEmpty()) {
194  m_mimeType = m_items.first().mimetype();
195  m_mimeGroup = m_mimeType.left(m_mimeType.indexOf('/'));
196  }
197  foreach (const KFileItem &item, m_items) {
198  const QString itemMimeType = item.mimetype();
199  // Determine if common mimetype among all items
200  if (m_mimeType != itemMimeType) {
201  m_mimeType.clear();
202  if (m_mimeGroup != itemMimeType.left(itemMimeType.indexOf('/'))) {
203  m_mimeGroup.clear(); // mimetype groups are different as well!
204  }
205  }
206  }
207 }
