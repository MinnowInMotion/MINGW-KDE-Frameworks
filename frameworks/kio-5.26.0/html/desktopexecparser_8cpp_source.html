58 int KRunMX1::expandEscapedMacro(const QString &str, int pos, QStringList &ret)
59 {
60  uint option = str[pos + 1].unicode();
61  switch (option) {
62  case 'c':
63  ret << service.name().replace('%', QLatin1String("%%"));
64  break;
65  case 'k':
66  ret << service.entryPath().replace('%', QLatin1String("%%"));
67  break;
68  case 'i':
69  ret << QStringLiteral("--icon") << service.icon().replace('%', QLatin1String("%%"));
70  break;
71  case 'm':
72 // ret << "-miniicon" << service.icon().replace( '%', "%%" );
73  qWarning() << "-miniicon isn't supported anymore (service"
74  << service.name() << ')';
75  break;
76  case 'u':
77  case 'U':
78  hasUrls = true;
79  /* fallthrough */
80  case 'f':
81  case 'F':
82  case 'n':
83  case 'N':
84  case 'd':
85  case 'D':
86  case 'v':
87  hasSpec = true;
88  /* fallthrough */
89  default:
90  return -2; // subst with same and skip
91  }
92  return 2;
93 }

112 void KRunMX2::subst(int option, const QUrl &url, QStringList &ret)
113 {
114  switch (option) {
115  case 'u':
116  ret << ((url.isLocalFile() && url.fragment().isNull() && url.query().isNull()) ?
117  QDir::toNativeSeparators(url.toLocalFile()) : url.toString());
118  break;
119  case 'd':
120  ret << url.adjusted(QUrl::RemoveFilename).path();
121  break;
122  case 'f':
123  ret << QDir::toNativeSeparators(url.toLocalFile());
124  break;
125  case 'n':
126  ret << url.fileName();
127  break;
128  case 'v':
129  if (url.isLocalFile() && QFile::exists(url.toLocalFile())) {
130  ret << KDesktopFile(url.toLocalFile()).desktopGroup().readEntry("Dev");
131  }
132  break;
133  }
134  return;
135 }

137 int KRunMX2::expandEscapedMacro(const QString &str, int pos, QStringList &ret)
138 {
139  uint option = str[pos + 1].unicode();
140  switch (option) {
141  case 'f':
142  case 'u':
143  case 'n':
144  case 'd':
145  case 'v':
146  if (urls.isEmpty()) {
147  if (!ignFile) {
148  //qCDebug(KIO_CORE) << "No URLs supplied to single-URL service" << str;
149  }
150  } else if (urls.count() > 1) {
151  qWarning() << urls.count() << "URLs supplied to single-URL service" << str;
152  } else {
153  subst(option, urls.first(), ret);
154  }
155  break;
156  case 'F':
157  case 'U':
158  case 'N':
159  case 'D':
160  option += 'a' - 'A';
161  for (QList<QUrl>::ConstIterator it = urls.begin(); it != urls.end(); ++it) {
162  subst(option, *it, ret);
163  }
164  break;
165  case '%':
166  ret = QStringList(QStringLiteral("%"));
167  break;
168  default:
169  return -2; // subst with same and skip
170  }
171  return 2;
172 }

174 QStringList KIO::DesktopExecParser::supportedProtocols(const KService &service)
175 {
176  QStringList supportedProtocols = service.property(QStringLiteral("X-KDE-Protocols")).toStringList();
177  KRunMX1 mx1(service);
178  QString exec = service.exec();
179  if (mx1.expandMacrosShellQuote(exec) && !mx1.hasUrls) {
180  if (!supportedProtocols.isEmpty()) {
181  qWarning() << service.entryPath() << "contains a X-KDE-Protocols line but doesn't use %u or %U in its Exec line! This is inconsistent.";
182  }
183  return QStringList();
184  } else {
185  if (supportedProtocols.isEmpty()) {
186  // compat mode: assume KIO if not set and it's a KDE app (or a KDE service)
187  const QStringList categories = service.property(QStringLiteral("Categories")).toStringList();
188  if (categories.contains(QStringLiteral("KDE"))
189  || !service.isApplication()
190  || service.entryPath().isEmpty() /*temp service*/) {
191  supportedProtocols.append(QStringLiteral("KIO"));
192  } else { // if no KDE app, be a bit over-generic
193  supportedProtocols.append(QStringLiteral("http"));
194  supportedProtocols.append(QStringLiteral("https")); // #253294
195  supportedProtocols.append(QStringLiteral("ftp"));
196  }
197  }
198  }
199  //qCDebug(KIO_CORE) << "supportedProtocols:" << supportedProtocols;
200  return supportedProtocols;
201 }

211 bool KIO::DesktopExecParser::hasSchemeHandler(const QUrl &url)
212 {
213  if (KProtocolInfo::isHelperProtocol(url)) {
214  return true;
215  }
216  if (KProtocolInfo::isKnownProtocol(url)) {
217  return false; // see schemeHandler()... this is case B, we prefer kioslaves over the competition
218  }
219  const KService::Ptr service = KMimeTypeTrader::self()->preferredService(QLatin1String("x-scheme-handler/") + url.scheme());
220  if (service) {
221  qCDebug(KIO_CORE) << "preferred service for x-scheme-handler/" + url.scheme() << service->desktopEntryName();
222  }
223  return service;

257 QStringList KIO::DesktopExecParser::resultingArguments() const
258 {
259  QString exec = d->service.exec();
260  if (exec.isEmpty()) {
261  qWarning() << "KRun: no Exec field in `" << d->service.entryPath() << "' !";
262  return QStringList();
263  }
264 
265  QStringList result;
266  bool appHasTempFileOption;
267 
268  KRunMX1 mx1(d->service);
269  KRunMX2 mx2(d->urls);
270 
271  if (!mx1.expandMacrosShellQuote(exec)) { // Error in shell syntax
272  qWarning() << "KRun: syntax error in command" << d->service.exec() << ", service" << d->service.name();
273  return QStringList();
274  }
275 
276  // FIXME: the current way of invoking kioexec disables term and su use
277 
278  // Check if we need "tempexec" (kioexec in fact)
279  appHasTempFileOption = d->tempFiles && d->service.property(QStringLiteral("X-KDE-HasTempFileOption")).toBool();
280  if (d->tempFiles && !appHasTempFileOption && d->urls.size()) {
281  const QString kioexec = QFile::decodeName(CMAKE_INSTALL_FULL_LIBEXECDIR_KF5 "/kioexec");
282  Q_ASSERT(QFile::exists(kioexec));
283  result << kioexec << QStringLiteral("--tempfiles") << exec;
284  if (!d->suggestedFileName.isEmpty()) {
285  result << QStringLiteral("--suggestedfilename");
286  result << d->suggestedFileName;
287  }
288  result += QUrl::toStringList(d->urls);
289  return result;
290  }
291 
292  // Check if we need kioexec
293  bool useKioexec = false;
294  if (!mx1.hasUrls) {
295  Q_FOREACH (const QUrl &url, d->urls)
296  if (!url.isLocalFile() && !hasSchemeHandler(url)) {
297  useKioexec = true;
298  //qCDebug(KIO_CORE) << "non-local files, application does not support urls, using kioexec";
299  break;
300  }
301  } else { // app claims to support %u/%U, check which protocols
302  QStringList appSupportedProtocols = supportedProtocols(d->service);
303  Q_FOREACH (const QUrl &url, d->urls) {
304  if (!isProtocolInSupportedList(url, appSupportedProtocols) && !hasSchemeHandler(url)) {
305  useKioexec = true;
306  //qCDebug(KIO_CORE) << "application does not support url, using kioexec:" << url;
307  break;
308  }
309  }
310  }
311  if (useKioexec) {
312  // We need to run the app through kioexec
313  const QString kioexec = CMAKE_INSTALL_FULL_LIBEXECDIR_KF5 "/kioexec";
314  Q_ASSERT(QFile::exists(kioexec));
315  result << kioexec;
316  if (d->tempFiles) {
317  result << QStringLiteral("--tempfiles");
318  }
319  if (!d->suggestedFileName.isEmpty()) {
320  result << QStringLiteral("--suggestedfilename");
321  result << d->suggestedFileName;
322  }
323  result << exec;
324  result += QUrl::toStringList(d->urls);
325  return result;
326  }
327 
328  if (appHasTempFileOption) {
329  exec += QLatin1String(" --tempfile");
330  }
331 
332  // Did the user forget to append something like '%f'?
333  // If so, then assume that '%f' is the right choice => the application
334  // accepts only local files.
335  if (!mx1.hasSpec) {
336  exec += QLatin1String(" %f");
337  mx2.ignFile = true;
338  }
339 
340  mx2.expandMacrosShellQuote(exec); // syntax was already checked, so don't check return value
341 
342  /*
343  1 = need_shell, 2 = terminal, 4 = su
344 
345  0 << split(cmd)
346  1 << "sh" << "-c" << cmd
347  2 << split(term) << "-e" << split(cmd)
348  3 << split(term) << "-e" << "sh" << "-c" << cmd
349 
350  4 << "kdesu" << "-u" << user << "-c" << cmd
351  5 << "kdesu" << "-u" << user << "-c" << ("sh -c " + quote(cmd))
352  6 << split(term) << "-e" << "su" << user << "-c" << cmd
353  7 << split(term) << "-e" << "su" << user << "-c" << ("sh -c " + quote(cmd))
354 
355  "sh -c" is needed in the "su" case, too, as su uses the user's login shell, not sh.
356  this could be optimized with the -s switch of some su versions (e.g., debian linux).
357  */
358 
359  if (d->service.terminal()) {
360  KConfigGroup cg(KSharedConfig::openConfig(), "General");
361  QString terminal = cg.readPathEntry("TerminalApplication", QStringLiteral("konsole"));
362  if (terminal == QLatin1String("konsole")) {
363  if (!d->service.path().isEmpty()) {
364  terminal += " --workdir " + KShell::quoteArg(d->service.path());
365  }
366  terminal += QLatin1String(" -caption=%c %i %m");
367  }
368  terminal += ' ';
369  terminal += d->service.terminalOptions();
370  if (!mx1.expandMacrosShellQuote(terminal)) {
371  qWarning() << "KRun: syntax error in command" << terminal << ", service" << d->service.name();
372  return QStringList();
373  }
374  mx2.expandMacrosShellQuote(terminal);
375  result = KShell::splitArgs(terminal); // assuming that the term spec never needs a shell!
376  result << QStringLiteral("-e");
377  }
378 
379  KShell::Errors err;
380  QStringList execlist = KShell::splitArgs(exec, KShell::AbortOnMeta | KShell::TildeExpand, &err);
381  if (err == KShell::NoError && !execlist.isEmpty()) { // mx1 checked for syntax errors already
382  // Resolve the executable to ensure that helpers in libexec are found.
383  // Too bad for commands that need a shell - they must reside in $PATH.
384  QString exePath = QStandardPaths::findExecutable(execlist.first());
385  if (exePath.isEmpty()) {
386  exePath = QFile::decodeName(CMAKE_INSTALL_FULL_LIBEXECDIR_KF5 "/") + execlist.first();
387  }
388  if (QFile::exists(exePath)) {
389  execlist[0] = exePath;
390  }
391  }
392  if (d->service.substituteUid()) {
393  if (d->service.terminal()) {
394  result << QStringLiteral("su");
395  } else {
396  QString kdesu = QFile::decodeName(CMAKE_INSTALL_FULL_LIBEXECDIR_KF5 "/kdesu");
397  if (!QFile::exists(kdesu)) {
398  kdesu = QStandardPaths::findExecutable(QStringLiteral("kdesu"));
399  }
400  if (!QFile::exists(kdesu)) {
401  // Insert kdesu as string so we show a nice warning: 'Could not launch kdesu'
402  result << QStringLiteral("kdesu");
403  return result;
404  } else {
405  result << kdesu << QStringLiteral("-u");
406  }
407  }
408 
409  result << d->service.username() << QStringLiteral("-c");
410  if (err == KShell::FoundMeta) {
411  exec = "/bin/sh -c " + KShell::quoteArg(exec);
412  } else {
413  exec = KShell::joinArgs(execlist);
414  }
415  result << exec;
416  } else {
417  if (err == KShell::FoundMeta) {
418  result << QStringLiteral("/bin/sh") << QStringLiteral("-c") << exec;
419  } else {
420  result += execlist;
421  }
422  }
423 
424  return result;
425 }
