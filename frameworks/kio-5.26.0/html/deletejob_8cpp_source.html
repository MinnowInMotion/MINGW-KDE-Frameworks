146 void DeleteJobPrivate::slotReport()
147 {
148  Q_Q(DeleteJob);
149  emit q->deleting(q, m_currentURL);
150 
151  // TODO: maybe we could skip everything else when (flags & HideProgressInfo) ?
152  JobPrivate::emitDeleting(q, m_currentURL);
153 
154  switch (state) {
155  case DELETEJOB_STATE_STATING:
156  q->setTotalAmount(KJob::Files, files.count());
157  q->setTotalAmount(KJob::Directories, dirs.count());
158  break;
159  case DELETEJOB_STATE_DELETING_DIRS:
160  q->setProcessedAmount(KJob::Directories, m_processedDirs);
161  q->emitPercent(m_processedFiles + m_processedDirs, m_totalFilesDirs);
162  break;
163  case DELETEJOB_STATE_DELETING_FILES:
164  q->setProcessedAmount(KJob::Files, m_processedFiles);
165  q->emitPercent(m_processedFiles, m_totalFilesDirs);
166  break;
167  }
168 }

170 void DeleteJobPrivate::slotEntries(KIO::Job *job, const UDSEntryList &list)
171 {
172  UDSEntryList::ConstIterator it = list.begin();
173  const UDSEntryList::ConstIterator end = list.end();
174  for (; it != end; ++it) {
175  const UDSEntry &entry = *it;
176  const QString displayName = entry.stringValue(KIO::UDSEntry::UDS_NAME);
177 
178  Q_ASSERT(!displayName.isEmpty());
179  if (displayName != QLatin1String("..") && displayName != QLatin1String(".")) {
180  QUrl url;
181  const QString urlStr = entry.stringValue(KIO::UDSEntry::UDS_URL);
182  if (!urlStr.isEmpty()) {
183  url = QUrl(urlStr);
184  } else {
185  url = static_cast<SimpleJob *>(job)->url(); // assumed to be a dir
186  url.setPath(url.path() + '/' + displayName);
187  }
188 
189  //qDebug() << displayName << "(" << url << ")";
190  if (entry.isLink()) {
191  symlinks.append(url);
192  } else if (entry.isDir()) {
193  dirs.append(url);
194  } else {
195  files.append(url);
196  }
197  }

201 void DeleteJobPrivate::statNextSrc()
202 {
203  Q_Q(DeleteJob);
204  //qDebug();
205  if (m_currentStat != m_srcList.end()) {
206  m_currentURL = (*m_currentStat);
207 
208  // if the file system doesn't support deleting, we do not even stat
209  if (!KProtocolManager::supportsDeleting(m_currentURL)) {
210  QPointer<DeleteJob> that = q;
211  ++m_currentStat;
212  emit q->warning(q, buildErrorString(ERR_CANNOT_DELETE, m_currentURL.toDisplayString()));
213  if (that) {
214  statNextSrc();
215  }
216  return;
217  }
218  // Stat it
219  state = DELETEJOB_STATE_STATING;
220 
221  // Fast path for KFileItems in directory views
222  while (m_currentStat != m_srcList.end()) {
223  m_currentURL = (*m_currentStat);
224  const KFileItem cachedItem = KCoreDirLister::cachedItemForUrl(m_currentURL);
225  if (cachedItem.isNull()) {
226  break;
227  }
228  //qDebug() << "Found cached info about" << m_currentURL << "isDir=" << cachedItem.isDir() << "isLink=" << cachedItem.isLink();
229  currentSourceStated(cachedItem.isDir(), cachedItem.isLink());
230  ++m_currentStat;
231  }
232 
233  // Hook for unit test to disable the fast path.
234  if (!kio_resolve_local_urls) {
235 
236  // Fast path for local files
237  // (using a loop, instead of a huge recursion)
238  while (m_currentStat != m_srcList.end() && (*m_currentStat).isLocalFile()) {
239  m_currentURL = (*m_currentStat);
240  QFileInfo fileInfo(m_currentURL.toLocalFile());
241  currentSourceStated(fileInfo.isDir(), fileInfo.isSymLink());
242  ++m_currentStat;
243  }
244  }
245  if (m_currentStat == m_srcList.end()) {
246  // Done, jump to the last else of this method
247  statNextSrc();
248  } else {
249  KIO::SimpleJob *job = KIO::stat(m_currentURL, StatJob::SourceSide, 0, KIO::HideProgressInfo);
250  Scheduler::setJobPriority(job, 1);
251  //qDebug() << "stat'ing" << m_currentURL;
252  q->addSubjob(job);
253  }
254  } else {
255  if (!q->hasSubjobs()) { // don't go there yet if we're still listing some subdirs
256  finishedStatPhase();
257  }
258  }
259 }

261 void DeleteJobPrivate::finishedStatPhase()
262 {
263  m_totalFilesDirs = files.count() + symlinks.count() + dirs.count();
264  slotReport();
265  // Now we know which dirs hold the files we're going to delete.
266  // To speed things up and prevent double-notification, we disable KDirWatch
267  // on those dirs temporarily (using KDirWatch::self, that's the instance
268  // used by e.g. kdirlister).
269  const QSet<QString>::const_iterator itEnd = m_parentDirs.constEnd();
270  for (QSet<QString>::const_iterator it = m_parentDirs.constBegin(); it != itEnd; ++it) {
271  KDirWatch::self()->stopDirScan(*it);
272  }
273  state = DELETEJOB_STATE_DELETING_FILES;
274  deleteNextFile();
275 }

277 void DeleteJobPrivate::deleteNextFile()
278 {
279  Q_Q(DeleteJob);
280  //qDebug();
281  if (!files.isEmpty() || !symlinks.isEmpty()) {
282  SimpleJob *job;
283  do {
284  // Take first file to delete out of list
285  QList<QUrl>::iterator it = files.begin();
286  bool isLink = false;
287  if (it == files.end()) { // No more files
288  it = symlinks.begin(); // Pick up a symlink to delete
289  isLink = true;
290  }
291  // Normal deletion
292  // If local file, try do it directly
293  if ((*it).isLocalFile() && QFile::remove((*it).toLocalFile())) {
294  //kdDebug(7007) << "DeleteJob deleted" << (*it).toLocalFile();
295  job = 0;
296  m_processedFiles++;
297  if (m_processedFiles % 300 == 1 || m_totalFilesDirs < 300) { // update progress info every 300 files
298  m_currentURL = *it;
299  slotReport();
300  }
301  } else {
302  // if remote - or if unlink() failed (we'll use the job's error handling in that case)
303  //qDebug() << "calling file_delete on" << *it;
304  if (isHttpProtocol(it->scheme())) {
305  job = KIO::http_delete(*it, KIO::HideProgressInfo);
306  } else {
307  job = KIO::file_delete(*it, KIO::HideProgressInfo);
308  }
309  Scheduler::setJobPriority(job, 1);
310  m_currentURL = (*it);
311  }
312  if (isLink) {
313  symlinks.erase(it);
314  } else {
315  files.erase(it);
316  }
317  if (job) {
318  q->addSubjob(job);
319  return;
320  }
321  // loop only if direct deletion worked (job=0) and there is something else to delete
322  } while (!job && (!files.isEmpty() || !symlinks.isEmpty()));
323  }
324  state = DELETEJOB_STATE_DELETING_DIRS;
325  deleteNextDir();
326 }

328 void DeleteJobPrivate::deleteNextDir()
329 {
330  Q_Q(DeleteJob);
331  if (!dirs.isEmpty()) { // some dirs to delete ?
332  do {
333  // Take first dir to delete out of list - last ones first !
334  QList<QUrl>::iterator it = --dirs.end();
335  // If local dir, try to rmdir it directly
336  if ((*it).isLocalFile() && QDir().rmdir((*it).toLocalFile())) {
337  m_processedDirs++;
338  if (m_processedDirs % 100 == 1) { // update progress info every 100 dirs
339  m_currentURL = *it;
340  slotReport();
341  }
342  } else {
343  // Call rmdir - works for kioslaves with canDeleteRecursive too,
344  // CMD_DEL will trigger the recursive deletion in the slave.
345  SimpleJob *job = KIO::rmdir(*it);
346  job->addMetaData(QStringLiteral("recurse"), QStringLiteral("true"));
347  Scheduler::setJobPriority(job, 1);
348  dirs.erase(it);
349  q->addSubjob(job);
350  return;
351  }
352  dirs.erase(it);
353  } while (!dirs.isEmpty());
354  }
355 
356  // Re-enable watching on the dirs that held the deleted files
357  restoreDirWatch();
358 
359  // Finished - tell the world
360  if (!m_srcList.isEmpty()) {
361  //qDebug() << "KDirNotify'ing FilesRemoved" << m_srcList;
362  org::kde::KDirNotify::emitFilesRemoved(m_srcList);
363  }
364  if (m_reportTimer != 0) {
365  m_reportTimer->stop();
366  }
367  q->emitResult();
368 }

378 void DeleteJobPrivate::currentSourceStated(bool isDir, bool isLink)
379 {
380  Q_Q(DeleteJob);
381  const QUrl url = (*m_currentStat);
382  if (isDir && !isLink) {
383  // Add toplevel dir in list of dirs
384  dirs.append(url);
385  if (url.isLocalFile()) {
386  // We are about to delete this dir, no need to watch it
387  // Maybe we should ask kdirwatch to remove all watches recursively?
388  // But then there would be no feedback (things disappearing progressively) during huge deletions
389  KDirWatch::self()->stopDirScan(url.adjusted(QUrl::StripTrailingSlash).toLocalFile());
390  }
391  if (!KProtocolManager::canDeleteRecursive(url)) {
392  //qDebug() << url << "is a directory, let's list it";
393  ListJob *newjob = KIO::listRecursive(url, KIO::HideProgressInfo);
394  newjob->addMetaData(QStringLiteral("details"), QStringLiteral("0"));
395  newjob->setUnrestricted(true); // No KIOSK restrictions
396  Scheduler::setJobPriority(newjob, 1);
397  QObject::connect(newjob, SIGNAL(entries(KIO::Job*,KIO::UDSEntryList)),
398  q, SLOT(slotEntries(KIO::Job*,KIO::UDSEntryList)));
399  q->addSubjob(newjob);
400  // Note that this listing job will happen in parallel with other stat jobs.
401  }
402  } else {
403  if (isLink) {
404  //qDebug() << "Target is a symlink";
405  symlinks.append(url);
406  } else {
407  //qDebug() << "Target is a file";
408  files.append(url);
409  }
410  }
411  if (url.isLocalFile()) {
412  const QString parentDir = url.adjusted(QUrl::RemoveFilename | QUrl::StripTrailingSlash).path();
413  m_parentDirs.insert(parentDir);
414  }
415 }

417 void DeleteJob::slotResult(KJob *job)
418 {
419  Q_D(DeleteJob);
420  switch (d->state) {
421  case DELETEJOB_STATE_STATING:
422  removeSubjob(job);
423 
424  // Was this a stat job or a list job? We do both in parallel.
425  if (StatJob *statJob = qobject_cast<StatJob *>(job)) {
426  // Was there an error while stating ?
427  if (job->error()) {
428  // Probably : doesn't exist
429  Job::slotResult(job); // will set the error and emit result(this)
430  d->restoreDirWatch();
431  return;
432  }
433 
434  const UDSEntry entry = statJob->statResult();
435  // Is it a file or a dir ?
436  const bool isLink = entry.isLink();
437  const bool isDir = entry.isDir();
438  d->currentSourceStated(isDir, isLink);
439 
440  ++d->m_currentStat;
441  d->statNextSrc();
442  } else {
443  if (job->error()) {
444  // Try deleting nonetheless, it may be empty (and non-listable)
445  }
446  if (!hasSubjobs()) {
447  d->finishedStatPhase();
448  }
449  }
450  break;
451  case DELETEJOB_STATE_DELETING_FILES:
452  // Propagate the subjob's metadata (a SimpleJob) to the real DeleteJob
453  // FIXME: setMetaData() in the KIO API only allows access to outgoing metadata,
454  // but we need to alter the incoming one
455  d->m_incomingMetaData = dynamic_cast<KIO::Job *>(job)->metaData();
456 
457  if (job->error()) {
458  Job::slotResult(job); // will set the error and emit result(this)
459  d->restoreDirWatch();
460  return;
461  }
462  removeSubjob(job);
463  Q_ASSERT(!hasSubjobs());
464  d->m_processedFiles++;
465 
466  d->deleteNextFile();
467  break;
468  case DELETEJOB_STATE_DELETING_DIRS:
469  if (job->error()) {
470  Job::slotResult(job); // will set the error and emit result(this)
471  d->restoreDirWatch();
472  return;
473  }
474  removeSubjob(job);
475  Q_ASSERT(!hasSubjobs());
476  d->m_processedDirs++;
477  //emit processedAmount( this, KJob::Directories, d->m_processedDirs );
478  //emitPercent( d->m_processedFiles + d->m_processedDirs, d->m_totalFilesDirs );
479 
480  d->deleteNextDir();
481  break;
482  default:
483  Q_ASSERT(0);
484  }
485 }
