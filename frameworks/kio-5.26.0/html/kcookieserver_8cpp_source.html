53 static QDir getOrCreateCookieJarDir()
54 {
55  const QDir dataDir(QStandardPaths::writableLocation(QStandardPaths::GenericDataLocation));
56  const QString kcookiejarDirName = dataDir.absoluteFilePath(QStringLiteral("kcookiejar"));
57 
58  if (dataDir.exists(QStringLiteral("kcookiejar"))) {
59  const QFileInfo cookiejarDirInfo(kcookiejarDirName);
60 
61  if (!cookiejarDirInfo.isDir()) {
62  QFile kcookieBogusFile(kcookiejarDirName);
63 
64  if (!kcookieBogusFile.remove()) {
65  QMessageBox::warning(Q_NULLPTR, i18n("Cannot Save Cookies"), i18n("Could not remove %1, check permissions").arg(cookiejarDirInfo.absoluteFilePath()));
66  }
67  }
68  else {
69  return QDir(kcookiejarDirName);
70  }
71  }
72 
73  if (!dataDir.mkpath(QStringLiteral("kcookiejar"))) {
74  QMessageBox::warning(Q_NULLPTR, i18n("Cannot Save Cookies"), i18n("Could not create directory %1").arg(kcookiejarDirName));
75  }
76 
77  return QDir(kcookiejarDirName);
78 }

130 bool KCookieServer::cookiesPending(const QString &url, KHttpCookieList *cookieList)
131 {
132  QString fqdn;
133  QString path;
134  // Check whether 'url' has cookies on the pending list
135  if (mPendingCookies->isEmpty()) {
136  return false;
137  }
138  if (!KCookieJar::parseUrl(url, fqdn, path)) {
139  return false;
140  }
141 
142  QStringList domains;
143  mCookieJar->extractDomains(fqdn, domains);
144  Q_FOREACH (const KHttpCookie &cookie, *mPendingCookies) {
145  if (cookie.match(fqdn, domains, path)) {
146  if (!cookieList) {
147  return true;
148  }
149  cookieList->append(cookie);
150  }
151  }
152  if (!cookieList) {
153  return false;
154  }
155  return cookieList->isEmpty();
156 }

158 void KCookieServer::addCookies(const QString &url, const QByteArray &cookieHeader,
159  qlonglong windowId, bool useDOMFormat)
160 {
161  KHttpCookieList cookieList;
162  if (useDOMFormat) {
163  cookieList = mCookieJar->makeDOMCookies(url, cookieHeader, windowId);
164  } else {
165  cookieList = mCookieJar->makeCookies(url, cookieHeader, windowId);
166  }
167 
168  checkCookies(&cookieList, windowId);
169 
170  *mPendingCookies += cookieList;
171 
172  if (!mAdvicePending) {
173  mAdvicePending = true;
174  while (!mPendingCookies->isEmpty()) {
175  checkCookies(0, windowId);
176  }
177  mAdvicePending = false;
178  }
179 }

186 void KCookieServer::checkCookies(KHttpCookieList *cookieList, qlonglong windowId)
187 {
188  KHttpCookieList *list;
189 
190  if (cookieList) {
191  list = cookieList;
192  } else {
193  list = mPendingCookies;
194  }
195 
196  QMutableListIterator<KHttpCookie> cookieIterator(*list);
197  while (cookieIterator.hasNext()) {
198  KHttpCookie &cookie = cookieIterator.next();
199  const KCookieAdvice advice = mCookieJar->cookieAdvice(cookie);
200  switch (advice) {
201  case KCookieAccept:
202  case KCookieAcceptForSession:
203  mCookieJar->addCookie(cookie);
204  cookieIterator.remove();
205  break;
206  case KCookieReject:
207  cookieIterator.remove();
208  break;
209  case KCookieDunno:
210  case KCookieAsk:
211  break;
212  }
213  }
214 
215  if (cookieList || list->isEmpty()) {
216  return;
217  }
218 
219  // Collect all pending cookies with the same host as the first pending cookie
220  const KHttpCookie &currentCookie = mPendingCookies->first();
221  KHttpCookieList currentList;
222  currentList.append(currentCookie);
223  const QString currentHost = currentCookie.host();
224  QList<int> shownCookies; shownCookies << 0;
225  for (int i = 1 /*first already done*/; i < mPendingCookies->count(); ++i) {
226  const KHttpCookie &cookie = (*mPendingCookies)[i];
227  if (cookie.host() == currentHost) {
228  currentList.append(cookie);
229  shownCookies << i;
230  }
231  }
232  //qDebug() << shownCookies;
233 
234  KCookieWin *kw = new KCookieWin(0L, currentList,
235  mCookieJar->preferredDefaultPolicy(),
236  mCookieJar->showCookieDetails());
237  if (windowId > 0) {
238  KWindowSystem::setMainWindow(kw, windowId);
239  }
240 
241  KCookieAdvice userAdvice = kw->advice(mCookieJar, currentCookie);
242  delete kw;
243  // Save the cookie config if it has changed
244  mCookieJar->saveConfig(mConfig);
245 
246  // Apply the user's choice to all cookies that are currently
247  // queued for this host (or just the first one, if the user asks for that).
248  QMutableListIterator<KHttpCookie> cookieIterator2(*mPendingCookies);
249  int pendingCookieIndex = -1;
250  while (cookieIterator2.hasNext()) {
251  ++pendingCookieIndex;
252  KHttpCookie &cookie = cookieIterator2.next();
253  if (cookie.host() != currentHost) {
254  continue;
255  }
256  if (mCookieJar->preferredDefaultPolicy() == KCookieJar::ApplyToShownCookiesOnly
257  && !shownCookies.contains(pendingCookieIndex)) {
258  // User chose "only those cookies", and this one was added while the dialog was up -> skip
259  break;
260  }
261  switch (userAdvice) {
262  case KCookieAccept:
263  case KCookieAcceptForSession:
264  // Store the user's choice for the cookie.
265  // This is only used to check later if the cookie should expire
266  // at the end of the session. The choice is not saved on disk.
267  cookie.setUserSelectedAdvice(userAdvice);
268  mCookieJar->addCookie(cookie);
269  cookieIterator2.remove();
270  break;
271 
272  case KCookieReject:
273  cookieIterator2.remove();
274  break;
275 
276  case KCookieDunno:
277  case KCookieAsk:
278  qWarning() << "userAdvice not accept or reject, this should never happen!";
279  break;
280  }
281  }
282 
283  // Check if we can handle any request
284  QMutableListIterator<CookieRequest *> requestIterator(*mRequestList);
285  while (requestIterator.hasNext()) {
286  CookieRequest *request = requestIterator.next();
287  if (!cookiesPending(request->url)) {
288  const QString res = mCookieJar->findCookies(request->url, request->DOM, request->windowId);
289 
290  QDBusConnection::sessionBus().send(request->reply.createReply(res));
291  delete request;
292  requestIterator.remove();
293  }
294  }
295 
296  saveCookieJar();
297 }

315 void KCookieServer::putCookie(QStringList &out, const KHttpCookie &cookie,
316  const QList<int> &fields)
317 {
318  foreach (int i, fields) {
319  switch (i) {
320  case CF_DOMAIN :
321  out << cookie.domain();
322  break;
323  case CF_NAME :
324  out << cookie.name();
325  break;
326  case CF_PATH :
327  out << cookie.path();
328  break;
329  case CF_HOST :
330  out << cookie.host();
331  break;
332  case CF_VALUE :
333  out << cookie.value();
334  break;
335  case CF_EXPIRE :
336  out << QString::number(cookie.expireDate());
337  break;
338  case CF_PROVER :
339  out << QString::number(cookie.protocolVersion());
340  break;
341  case CF_SECURE :
342  out << QString::number(cookie.isSecure() ? 1 : 0);
343  break;
344  default :
345  out << QString();
346  }
347  }
348 }

350 bool KCookieServer::cookieMatches(const KHttpCookie &c,
351  const QString &domain, const QString &fqdn,
352  const QString &path, const QString &name)
353 {
354  const bool hasDomain = !domain.isEmpty();
355  return (((hasDomain && c.domain() == domain) || fqdn == c.host()) &&
356  (c.path() == path) && (c.name() == name) &&
357  (!c.isExpired()));
358 }

367 QString KCookieServer::findCookies(const QString &url, qlonglong windowId)
368 {
369  if (cookiesPending(url)) {
370  CookieRequest *request = new CookieRequest;
371  message().setDelayedReply(true);
372  request->reply = message();
373  request->url = url;
374  request->DOM = false;
375  request->windowId = windowId;
376  mRequestList->append(request);
377  return QString(); // Talk to you later :-)
378  }
379 
380  QString cookies = mCookieJar->findCookies(url, false, windowId);
381  saveCookieJar();
382  return cookies;

401 // DBUS function
402 QStringList
403 KCookieServer::findCookies(const QList<int> &fields,
404  const QString &_domain,
405  const QString &fqdn,
406  const QString &path,
407  const QString &name)
408 {
409  QStringList result;
410  const bool allCookies = name.isEmpty();
411  const QStringList domainList = _domain.split(QLatin1Char(' '));
412 
413  if (allCookies) {
414  Q_FOREACH (const QString &domain, domainList) {
415  const KHttpCookieList *list = mCookieJar->getCookieList(domain, fqdn);
416  if (!list) {
417  continue;
418  }
419  Q_FOREACH (const KHttpCookie &cookie, *list) {
420  if (cookie.isExpired()) {
421  continue;
422  }
423  putCookie(result, cookie, fields);
424  }
425  }
426  } else {
427  Q_FOREACH (const QString &domain, domainList) {
428  const KHttpCookieList *list = mCookieJar->getCookieList(domain, fqdn);
429  if (!list) {
430  continue;
431  }
432  Q_FOREACH (const KHttpCookie &cookie, *list) {
433  if (cookie.isExpired()) {
434  continue;
435  }
436  if (cookieMatches(cookie, domain, fqdn, path, name)) {
437  putCookie(result, cookie, fields);
438  break;
439  }
440  }
441  }
442  }
443 
444  return result;
445 }

476 KCookieServer::deleteCookie(const QString &domain, const QString &fqdn,
477  const QString &path, const QString &name)
478 {
479  KHttpCookieList *cookieList = mCookieJar->getCookieList(domain, fqdn);
480  if (cookieList && !cookieList->isEmpty()) {
481  KHttpCookieList::Iterator itEnd = cookieList->end();
482  for (KHttpCookieList::Iterator it = cookieList->begin(); it != itEnd; ++it) {
483  if (cookieMatches(*it, domain, fqdn, path, name)) {
484  mCookieJar->eatCookie(it);
485  saveCookieJar();
486  break;
487  }
488  }
489  }

556 // DBUS function
557 QString
558 KCookieServer::getDomainAdvice(const QString &url)
559 {
560  KCookieAdvice advice = KCookieDunno;
561  QString fqdn;
562  QString dummy;
563  if (KCookieJar::parseUrl(url, fqdn, dummy)) {
564  QStringList domains;
565  mCookieJar->extractDomains(fqdn, domains);
566 
567  QStringListIterator it(domains);
568  while ((advice == KCookieDunno) && it.hasNext()) {
569  // Always check advice in both ".domain" and "domain". Note
570  // that we only want to check "domain" if it matches the
571  // fqdn of the requested URL.
572  const QString &domain = it.next();
573  if (domain.at(0) == '.' || domain == fqdn) {
574  advice = mCookieJar->getDomainAdvice(domain);
575  }
576  }
577  if (advice == KCookieDunno) {
578  advice = mCookieJar->getGlobalAdvice();
579  }
580  }
581  return KCookieJar::adviceToStr(advice);
582 }
