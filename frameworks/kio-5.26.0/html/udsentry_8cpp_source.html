37 class KIO::UDSEntryPrivate : public QSharedData
38 {
39 public:
40  struct Field {
41  inline Field(const QString &value) : m_str(value), m_long(0) {}
42  inline Field(long long value = 0) : m_long(value) { }
43  QString m_str;
44  long long m_long;
45  };
46 
47  QVector<Field> fields;
48 
49  // If udsIndexes[i] == uds, then fields[i] contains the value for 'uds'. Example:
50  // udsIndexes = {UDS_NAME, UDS_FILE_SIZE, ...}
51  // fields = {Field("filename"), Field(1234), ...}
52  QVector<uint> udsIndexes;
53 
54  void insert(uint uds, const Field& field)
55  {
56  const int index = udsIndexes.indexOf(uds);
57  if (index >= 0) {
58  fields[index] = field;
59  } else {
60  udsIndexes.append(uds);
61  fields.append(field);
62  }
63  }
64 
65  static void save(QDataStream &, const UDSEntry &);
66  static void load(QDataStream &, UDSEntry &);
67 };

195 void UDSEntryPrivate::save(QDataStream &s, const UDSEntry &a)
196 {
197  const QVector<uint> &udsIndexes = a.d->udsIndexes;
198  const QVector<Field> &fields = a.d->fields;
199  const int size = udsIndexes.size();
200 
201  s << size;
202 
203  for (int index = 0; index < size; ++index) {
204  uint uds = udsIndexes.at(index);
205  s << uds;
206 
207  if (uds & KIO::UDSEntry::UDS_STRING) {
208  s << fields.at(index).m_str;
209  } else if (uds & KIO::UDSEntry::UDS_NUMBER) {
210  s << fields.at(index).m_long;
211  } else {
212  Q_ASSERT_X(false, "KIO::UDSEntry", "Found a field with an invalid type");
213  }
214  }
215 }

223 void UDSEntryPrivate::load(QDataStream &s, UDSEntry &a)
224 {
225  a.clear();
226 
227  QVector<Field> &fields = a.d->fields;
228  QVector<uint> &udsIndexes = a.d->udsIndexes;
229 
230  quint32 size;
231  s >> size;
232  fields.reserve(size);
233  udsIndexes.reserve(size);
234 
235  // We cache the loaded strings. Some of them, like, e.g., the user,
236  // will often be the same for many entries in a row. Caching them
237  // permits to use implicit sharing to save memory.
238  static QVector<QString> cachedStrings;
239  if (quint32(cachedStrings.size()) < size) {
240  cachedStrings.resize(size);
241  }
242 
243  for (quint32 i = 0; i < size; ++i) {
244  quint32 uds;
245  s >> uds;
246  udsIndexes.append(uds);
247 
248  if (uds & KIO::UDSEntry::UDS_STRING) {
249  // If the QString is the same like the one we read for the
250  // previous UDSEntry at the i-th position, use an implicitly
251  // shared copy of the same QString to save memory.
252  QString buffer;
253  s >> buffer;
254 
255  if (buffer != cachedStrings.at(i)) {
256  cachedStrings[i] = buffer;
257  }
258 
259  fields.append(Field(cachedStrings.at(i)));
260  } else if (uds & KIO::UDSEntry::UDS_NUMBER) {
261  long long value;
262  s >> value;
263  fields.append(Field(value));
264  } else {
265  Q_ASSERT_X(false, "KIO::UDSEntry", "Found a field with an invalid type");
266  }
267  }
268 }

270 void debugUDSEntry(QDebug stream, const KIO::UDSEntry &entry)
271 {
272  const QVector<uint> &udsIndexes = entry.d->udsIndexes;
273  const QVector<KIO::UDSEntryPrivate::Field> &fields = entry.d->fields;
274  const int size = udsIndexes.size();
275  QDebugStateSaver saver(stream);
276  stream.nospace() << "[";
277  for (int index = 0; index < size; ++index) {
278  const uint uds = udsIndexes.at(index);
279  stream << " " << (uds & 0xffff) << "="; // we could use a switch statement for readability :-)
280  if (uds & KIO::UDSEntry::UDS_STRING) {
281  stream << fields.at(index).m_str;
282  } else if (uds & KIO::UDSEntry::UDS_NUMBER) {
283  stream << fields.at(index).m_long;
284  } else {
285  Q_ASSERT_X(false, "KIO::UDSEntry", "Found a field with an invalid type");
286  }
287  }
