53 void SearchProvider::setKeys(const QStringList &keys)
54 {
55  if (KUriFilterSearchProvider::keys() == keys)
56  return;
57 
58  KUriFilterSearchProvider::setKeys(keys);
59 
60  QString name = desktopEntryName();
61  if (!name.isEmpty())
62  return;
63 
64  // New provider. Set the desktopEntryName.
65  // Take the longest search shortcut as filename,
66  // if such a file already exists, append a number and increase it
67  // until the name is unique
68  Q_FOREACH(const QString& key, keys)
69  {
70  if (key.length() > name.length())
71  name = key.toLower();
72  }
73 
74  const QString path = QStandardPaths::writableLocation(QStandardPaths::GenericDataLocation) + "/kservices5/searchproviders/";
75  bool firstRun = true;
76 
77  while (true)
78  {
79  QString check(name);
80 
81  if (!firstRun)
82  check += KRandom::randomString(4);
83 
84  const QString located = QStandardPaths::locate(QStandardPaths::GenericDataLocation, QStringLiteral("kservices5/searchproviders/") + check + ".desktop");
85  if (located.isEmpty())
86  {
87  name = check;
88  break;
89  }
90  else if (located.startsWith(path))
91  {
92  // If it's a deleted (hidden) entry, overwrite it
93  if (KService(located).isDeleted())
94  break;
95  }
96  }
97 
98  setDesktopEntryName(name);
99 }
