130 // Slave got a redirection request
131 void StatJobPrivate::slotRedirection(const QUrl &url)
132 {
133  Q_Q(StatJob);
134  //qDebug() << m_url << "->" << url;
135  if (!KUrlAuthorized::authorizeUrlAction(QStringLiteral("redirect"), m_url, url)) {
136  qWarning() << "Redirection from" << m_url << "to" << url << "REJECTED!";
137  q->setError(ERR_ACCESS_DENIED);
138  q->setErrorText(url.toDisplayString());
139  return;
140  }
141  m_redirectionURL = url; // We'll remember that when the job finishes
142  // Tell the user that we haven't finished yet
143  emit q->redirection(q, m_redirectionURL);
144 }

146 void StatJob::slotFinished()
147 {
148  Q_D(StatJob);
149 
150  if (!d->m_redirectionURL.isEmpty() && d->m_redirectionURL.isValid()) {
151  //qDebug() << "StatJob: Redirection to " << m_redirectionURL;
152  if (queryMetaData(QStringLiteral("permanent-redirect")) == QLatin1String("true")) {
153  emit permanentRedirection(this, d->m_url, d->m_redirectionURL);
154  }
155 
156  if (d->m_redirectionHandlingEnabled) {
157  d->m_packedArgs.truncate(0);
158  QDataStream stream(&d->m_packedArgs, QIODevice::WriteOnly);
159  stream << d->m_redirectionURL;
160 
161  d->restartAfterRedirection(&d->m_redirectionURL);
162  return;
163  }
164  }
165 
166  // Return slave to the scheduler
167  SimpleJob::slotFinished();
168 }

183 StatJob *KIO::mostLocalUrl(const QUrl &url, JobFlags flags)
184 {
185  StatJob *job = stat(url, StatJob::SourceSide, 2, flags);
186  if (url.isLocalFile()) {
187  QTimer::singleShot(0, job, SLOT(slotFinished()));
188  Scheduler::cancelJob(job); // deletes the slave if not 0
189  }
