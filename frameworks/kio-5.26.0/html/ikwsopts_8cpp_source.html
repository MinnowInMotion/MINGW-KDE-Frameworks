75 bool ProvidersModel::setData (const QModelIndex& index, const QVariant& value, int role)
76 {
77  if (role==Qt::CheckStateRole)

89 QVariant ProvidersModel::data(const QModelIndex& index, int role) const
90 {
91  if (index.isValid())
92  {
93  if (role == Qt::CheckStateRole && index.column()==Preferred)
94  return (m_favoriteEngines.contains(m_providers.at(index.row())->desktopEntryName()) ? Qt::Checked : Qt::Unchecked);
95 
96  if (role == Qt::DisplayRole)
97  {
98  if (index.column()==Name)
99  return m_providers.at(index.row())->name();
100  if (index.column()==Shortcuts)
101  return m_providers.at(index.row())->keys().join(QStringLiteral(","));
102  }
103 
104  if (role == Qt::ToolTipRole || role == Qt::WhatsThisRole)
105  {
106  if (index.column() == Preferred)
107  return xi18nc("@info:tooltip", "Check this box to select the highlighted web shortcut "
108  "as preferred.<nl/>Preferred web shortcuts are used in "
109  "places where only a few select shortcuts can be shown "
110  "at one time.");
111  }
112 
113  if (role == Qt::UserRole)
114  return index.row();//a nice way to bypass proxymodel
115  }
116 
117  return QVariant();
118 }

286 void FilterOptions::load()
287 {
288  KConfig config(KURISearchFilterEngine::self()->name() + QStringLiteral("rc"), KConfig::NoGlobals);
289  KConfigGroup group = config.group("General");
290 
291  const QString defaultSearchEngine = group.readEntry("DefaultWebShortcut");
292  const QStringList favoriteEngines = group.readEntry("PreferredWebShortcuts", DEFAULT_PREFERRED_SEARCH_PROVIDERS);
293 
294  QList<SearchProvider*> providers;
295  const KService::List services = KServiceTypeTrader::self()->query(QStringLiteral("SearchProvider"));
296  int defaultProviderIndex = services.size(); //default is "None", it is last in the list
297 
298  Q_FOREACH(const KService::Ptr &service, services)
299  {
300  SearchProvider* provider = new SearchProvider(service);
301  if (defaultSearchEngine == provider->desktopEntryName())
302  defaultProviderIndex = providers.size();
303  providers.append(provider);
304  }
305 
306  m_providersModel->setProviders(providers, favoriteEngines);
307  m_dlg.lvSearchProviders->setColumnWidth(0,200);
308  m_dlg.lvSearchProviders->resizeColumnToContents(1);
309  m_dlg.lvSearchProviders->sortByColumn(0,Qt::AscendingOrder);
310  m_dlg.cmbDefaultEngine->model()->sort(0,Qt::AscendingOrder);
311  setDefaultEngine(defaultProviderIndex);
312 
313  m_dlg.cbEnableShortcuts->setChecked(group.readEntry("EnableWebShortcuts", true));
314  m_dlg.cbUseSelectedShortcutsOnly->setChecked(group.readEntry("UsePreferredWebShortcutsOnly", false));
315 
316  const QString delimiter = group.readEntry ("KeywordDelimiter", ":");
317  setDelimiter(delimiter.at(0).toLatin1());
318 }

331 void FilterOptions::save()
332 {
333  KConfig config(KURISearchFilterEngine::self()->name() + QStringLiteral("rc"), KConfig::NoGlobals );
334 
335  KConfigGroup group = config.group("General");
336  group.writeEntry("EnableWebShortcuts", m_dlg.cbEnableShortcuts->isChecked());
337  group.writeEntry("KeywordDelimiter", QString(QLatin1Char(delimiter())));
338  group.writeEntry("DefaultWebShortcut", m_dlg.cmbDefaultEngine->view()->currentIndex().data(ProvidersListModel::ShortNameRole));
339  group.writeEntry("PreferredWebShortcuts", m_providersModel->favoriteEngines());
340  group.writeEntry("UsePreferredWebShortcutsOnly", m_dlg.cbUseSelectedShortcutsOnly->isChecked());
341 
342  int changedProviderCount = 0;
343  QList<SearchProvider*> providers = m_providersModel->providers();
344  const QString path = QStandardPaths::writableLocation(QStandardPaths::GenericDataLocation) + "/kservices5/searchproviders/";
345 
346  Q_FOREACH(SearchProvider* provider, providers)
347  {
348  if (!provider->isDirty())
349  continue;
350 
351  changedProviderCount++;
352 
353  KConfig _service(path + provider->desktopEntryName() + ".desktop", KConfig::SimpleConfig );
354  KConfigGroup service(&_service, "Desktop Entry");
355  service.writeEntry("Type", "Service");
356  service.writeEntry("ServiceTypes", "SearchProvider");
357  service.writeEntry("Name", provider->name());
358  service.writeEntry("Query", provider->query());
359  service.writeEntry("Keys", provider->keys());
360  service.writeEntry("Charset", provider->charset());
361  service.writeEntry("Hidden", false); // we might be overwriting a hidden entry
362  }
363 
364  const QStringList servicesDirs = QStandardPaths::locateAll(QStandardPaths::GenericDataLocation, QStringLiteral("kservices5/searchproviders/"), QStandardPaths::LocateDirectory);
365  Q_FOREACH(const QString& providerName, m_deletedProviders)
366  {
367  QStringList matches;
368  foreach(const QString& dir, servicesDirs) {
369  QString current = dir + '/' + providerName + ".desktop";
370  if(QFile::exists(current))
371  matches += current;
372  }
373 
374  // Shouldn't happen
375  if (!matches.size())
376  continue;
377 
378  changedProviderCount++;
379 
380  if (matches.size() == 1 && matches.first().startsWith(path))
381  {
382  // If only the local copy existed, unlink it
383  // TODO: error handling
384  QFile::remove(matches.first());
385  continue;
386  }
387 
388  KConfig _service(path + providerName + ".desktop", KConfig::SimpleConfig );
389  KConfigGroup service(&_service, "Desktop Entry");
390  service.writeEntry("Type", "Service");
391  service.writeEntry("ServiceTypes", "SearchProvider");
392  service.writeEntry("Hidden", true);
393  }
394 
395  config.sync();
396 
397  emit changed(false);
398 
399  // Update filters in running applications...
400  QDBusMessage msg = QDBusMessage::createSignal(QStringLiteral("/"), QStringLiteral("org.kde.KUriFilterPlugin"), QStringLiteral("configure"));
401  QDBusConnection::sessionBus().send(msg);
402 
403  // If the providers changed, tell sycoca to rebuild its database...
404  if (changedProviderCount)
405  KBuildSycocaProgressDialog::rebuildKSycoca(this);
406 }
