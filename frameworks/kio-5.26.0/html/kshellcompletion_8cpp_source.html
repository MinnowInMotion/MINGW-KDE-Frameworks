69 QString KShellCompletion::makeCompletion(const QString &text)
70 {
71  // Split text at the last unquoted space
72  //
73  d->splitText(text, d->m_text_start, d->m_text_compl);
74 
75  // Remove quotes from the text to be completed
76  //
77  QString tmp = d->unquote(d->m_text_compl);
78  d->m_text_compl = tmp;
79 
80  // Do exe-completion if there was no unquoted space
81  //
82  bool is_exe_completion = true;
83 
84  for (int i = 0; i < d->m_text_start.length(); i++) {
85  if (d->m_text_start[i] != d->m_word_break_char) {
86  is_exe_completion = false;
87  break;
88  }
89  }
90 
91  Mode mode = (is_exe_completion ? ExeCompletion : FileCompletion);
92 
93  setMode(mode);
94 
95  // Make completion on the last part of text
96  //
97  return KUrlCompletion::makeCompletion(d->m_text_compl);
98 }

100 /*
101  * postProcessMatch, postProcessMatches
102  *
103  * Called by KCompletion before emitting match() and matches()
104  *
105  * Add add the part of the text that was not completed
106  * Add quotes when needed
107  */
108 void KShellCompletion::postProcessMatch(QString *match) const
109 {
110  KUrlCompletion::postProcessMatch(match);
111 
112  if (match->isNull()) {
113  return;
114  }
115 
116  if (match->endsWith(QLatin1Char('/'))) {
117  d->quoteText(match, false, true); // don't quote the trailing '/'
118  } else {
119  d->quoteText(match, false, false); // quote the whole text
120  }
121 
122  match->prepend(d->m_text_start);
123 }

161 /*
162  * splitText
163  *
164  * Split text at the last unquoted space
165  *
166  * text_start = [out] text at the left, including the space
167  * text_compl = [out] text at the right
168  */
169 void KShellCompletionPrivate::splitText(const QString &text, QString &text_start,
170  QString &text_compl) const
171 {
172  bool in_quote = false;
173  bool escaped = false;
174  QChar p_last_quote_char;
175  int last_unquoted_space = -1;
176  int end_space_len = 0;
177 
178  for (int pos = 0; pos < text.length(); pos++) {
179 
180  end_space_len = 0;
181 
182  if (escaped) {
183  escaped = false;
184  } else if (in_quote && text[pos] == p_last_quote_char) {
185  in_quote = false;
186  } else if (!in_quote && text[pos] == m_quote_char1) {
187  p_last_quote_char = m_quote_char1;
188  in_quote = true;
189  } else if (!in_quote && text[pos] == m_quote_char2) {
190  p_last_quote_char = m_quote_char2;
191  in_quote = true;
192  } else if (text[pos] == m_escape_char) {
193  escaped = true;
194  } else if (!in_quote && text[pos] == m_word_break_char) {
195 
196  end_space_len = 1;
197 
198  while (pos + 1 < text.length() && text[pos + 1] == m_word_break_char) {
199  end_space_len++;
200  pos++;
201  }
202 
203  if (pos + 1 == text.length()) {
204  break;
205  }
206 
207  last_unquoted_space = pos;
208  }
209  }
210 
211  text_start = text.left(last_unquoted_space + 1);
212 
213  // the last part without trailing blanks
214  text_compl = text.mid(last_unquoted_space + 1);
215 }

217 /*
218  * quoteText()
219  *
220  * Add quotations to 'text' if needed or if 'force' = true
221  * Returns true if quotes were added
222  *
223  * skip_last => ignore the last charachter (we add a space or '/' to all filenames)
224  */
225 bool KShellCompletionPrivate::quoteText(QString *text, bool force, bool skip_last) const
226 {
227  int pos = 0;
228 
229  if (!force) {
230  pos = text->indexOf(m_word_break_char);
231  if (skip_last && (pos == (int)(text->length()) - 1)) {
232  pos = -1;
233  }
234  }
235 
236  if (!force && pos == -1) {
237  pos = text->indexOf(m_quote_char1);
238  if (skip_last && (pos == (int)(text->length()) - 1)) {
239  pos = -1;
240  }
241  }
242 
243  if (!force && pos == -1) {
244  pos = text->indexOf(m_quote_char2);
245  if (skip_last && (pos == (int)(text->length()) - 1)) {
246  pos = -1;
247  }
248  }
249 
250  if (!force && pos == -1) {
251  pos = text->indexOf(m_escape_char);
252  if (skip_last && (pos == (int)(text->length()) - 1)) {
253  pos = -1;
254  }
255  }
256 
257  if (force || (pos >= 0)) {
258 
259  // Escape \ in the string
260  text->replace(m_escape_char,
261  QString(m_escape_char) + m_escape_char);
262 
263  // Escape " in the string
264  text->replace(m_quote_char1,
265  QString(m_escape_char) + m_quote_char1);
266 
267  // " at the beginning
268  text->insert(0, m_quote_char1);
269 
270  // " at the end
271  if (skip_last) {
272  text->insert(text->length() - 1, m_quote_char1);
273  } else {
274  text->insert(text->length(), m_quote_char1);
275  }
276 
277  return true;
278  }
279 
280  return false;
281 }

288  */
289 QString KShellCompletionPrivate::unquote(const QString &text) const
290 {
291  bool in_quote = false;
292  bool escaped = false;
293  QChar p_last_quote_char;
294  QString result;
295 
296  for (int pos = 0; pos < text.length(); pos++) {
297 
298  if (escaped) {
299  escaped = false;
300  result.insert(result.length(), text[pos]);
301  } else if (in_quote && text[pos] == p_last_quote_char) {
302  in_quote = false;
303  } else if (!in_quote && text[pos] == m_quote_char1) {
304  p_last_quote_char = m_quote_char1;
305  in_quote = true;
306  } else if (!in_quote && text[pos] == m_quote_char2) {
307  p_last_quote_char = m_quote_char2;
308  in_quote = true;
309  } else if (text[pos] == m_escape_char) {
310  escaped = true;
311  result.insert(result.length(), text[pos]);
312  } else {
313  result.insert(result.length(), text[pos]);
314  }
315 
316  }
317 
318  return result;
319 }
