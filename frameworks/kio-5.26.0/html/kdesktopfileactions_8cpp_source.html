64 bool KDesktopFileActions::runWithStartup(const QUrl &u, bool _is_local, const QByteArray &asn)
65 {
66  // It might be a security problem to run external untrusted desktop
67  // entry files
68  if (!_is_local) {
69  return false;
70  }
71 
72  if (u.fileName() == QLatin1String(".directory")) {
73  // We cannot execute a .directory file. Open with a text editor instead.
74  return KRun::runUrl(u, QStringLiteral("text/plain"), 0, false /*tempFile*/, false /*runExecutables*/, QString(), asn);
75  }
76 
77  KDesktopFile cfg(u.toLocalFile());
78  if (!cfg.desktopGroup().hasKey("Type")) {
79  QString tmp = i18n("The desktop entry file %1 "
80  "has no Type=... entry.", u.toLocalFile());
81  KMessageBox::error(0, tmp);
82  return false;
83  }
84 
85  //qDebug() << "TYPE = " << type.data();
86 
87  if (cfg.hasDeviceType()) {
88  return runFSDevice(u, cfg, asn);
89  } else if (cfg.hasApplicationType()
90  || (cfg.readType() == QLatin1String("Service") && !cfg.desktopGroup().readEntry("Exec").isEmpty())) { // for kio_settings
91  return runApplication(u, u.toLocalFile(), asn);
92  } else if (cfg.hasLinkType()) {
93  return runLink(u, cfg, asn);
94  }
95 
96  QString tmp = i18n("The desktop entry of type\n%1\nis unknown.", cfg.readType());
97  KMessageBox::error(0, tmp);
98 
99  return false;
100 }

102 static bool runFSDevice(const QUrl &_url, const KDesktopFile &cfg, const QByteArray &asn)
103 {
104  bool retval = false;
105 
106  QString dev = cfg.readDevice();
107 
108  if (dev.isEmpty()) {
109  QString tmp = i18n("The desktop entry file\n%1\nis of type FSDevice but has no Dev=... entry.", _url.toLocalFile());
110  KMessageBox::error(0, tmp);
111  return retval;
112  }
113 
114  KMountPoint::Ptr mp = KMountPoint::currentMountPoints().findByDevice(dev);
115  // Is the device already mounted ?
116  if (mp) {
117  const QUrl mpURL = QUrl::fromLocalFile(mp->mountPoint());
118  // Open a new window
119  retval = KRun::runUrl(mpURL, QStringLiteral("inode/directory"), 0 /*TODO - window*/, false, true, QString(), asn);
120  } else {
121  KConfigGroup cg = cfg.desktopGroup();
122  bool ro = cg.readEntry("ReadOnly", false);
123  QString fstype = cg.readEntry("FSType");
124  if (fstype == QLatin1String("Default")) { // KDE-1 thing
125  fstype.clear();
126  }
127  QString point = cg.readEntry("MountPoint");
128 #ifndef Q_OS_WIN
129  (void) new KAutoMount(ro, fstype.toLatin1(), dev, point, _url.toLocalFile());
130 #endif
131  retval = false;
132  }
133 
134  return retval;
135 }

151 static bool runLink(const QUrl &_url, const KDesktopFile &cfg, const QByteArray &asn)
152 {
153  QString u = cfg.readUrl();
154  if (u.isEmpty()) {
155  QString tmp = i18n("The desktop entry file\n%1\nis of type Link but has no URL=... entry.", _url.toString());
156  KMessageBox::error(0, tmp);
157  return false;
158  }
159 
160  QUrl url = QUrl::fromUserInput(u);
161  KRun *run = new KRun(url, (QWidget *)0, true, asn);
162 
163  // X-KDE-LastOpenedWith holds the service desktop entry name that
164  // was should be preferred for opening this URL if possible.
165  // This is used by the Recent Documents menu for instance.
166  QString lastOpenedWidth = cfg.desktopGroup().readEntry("X-KDE-LastOpenedWith");
167  if (!lastOpenedWidth.isEmpty()) {
168  run->setPreferredService(lastOpenedWidth);
169  }
170 
171  return false;
172 }

174 QList<KServiceAction> KDesktopFileActions::builtinServices(const QUrl &_url)
175 {
176  QList<KServiceAction> result;
177 
178  if (!_url.isLocalFile()) {
179  return result;
180  }
181 
182  bool offerMount = false;
183  bool offerUnmount = false;
184 
185  KDesktopFile cfg(_url.toLocalFile());
186  if (cfg.hasDeviceType()) { // url to desktop file
187  const QString dev = cfg.readDevice();
188  if (dev.isEmpty()) {
189  QString tmp = i18n("The desktop entry file\n%1\nis of type FSDevice but has no Dev=... entry.", _url.toLocalFile());
190  KMessageBox::error(0, tmp);
191  return result;
192  }
193 
194  KMountPoint::Ptr mp = KMountPoint::currentMountPoints().findByDevice(dev);
195  if (mp) {
196  offerUnmount = true;
197  } else {
198  offerMount = true;
199  }
200  }
201 #if ! KIO_NO_SOLID
202  else { // url to device
203  Solid::Predicate predicate(Solid::DeviceInterface::Block, "device", _url.toLocalFile());
204  const QList<Solid::Device> devList = Solid::Device::listFromQuery(predicate, QString());
205  if (devList.empty()) {
206  //qDebug() << "Device" << _url.toLocalFile() << "not found";
207  return result;
208  }
209  Solid::Device device = devList[0];
210  Solid::StorageAccess *access = device.as<Solid::StorageAccess>();
211  Solid::StorageDrive *drive = device.parent().as<Solid::StorageDrive>();
212  bool mounted = access && access->isAccessible();
213 
214  if ((mounted || device.is<Solid::OpticalDisc>()) && drive && drive->isRemovable()) {
215  offerUnmount = true;
216  }
217 
218  if (!mounted && ((drive && drive->isHotpluggable()) || device.is<Solid::OpticalDisc>())) {
219  offerMount = true;
220  }
221  }
222 #endif
223 
224  if (offerMount) {
225  KServiceAction mount(QStringLiteral("mount"), i18n("Mount"), QString(), QString(), false);
226  mount.setData(QVariant(ST_MOUNT));
227  result.append(mount);
228  }
229 
230  if (offerUnmount) {
231  QString text;
232 #ifdef HAVE_VOLMGT
233  /*
234  * Solaris' volume management can only umount+eject
235  */
236  text = i18n("Eject");
237 #else
238  text = i18n("Unmount");
239 #endif
240  KServiceAction unmount(QStringLiteral("unmount"), text, QString(), QString(), false);
241  unmount.setData(QVariant(ST_UNMOUNT));
242  result.append(unmount);
243  }
244 
245  return result;
246 }

261 QList<KServiceAction> KDesktopFileActions::userDefinedServices(const KService &service, bool bLocalFiles, const QList<QUrl> &file_list)
262 {
263  QList<KServiceAction> result;
264 
265  if (!service.isValid()) { // e.g. TryExec failed
266  return result;
267  }
268 
269  QStringList keys;
270  const QString actionMenu = service.property(QStringLiteral("X-KDE-GetActionMenu"), QVariant::String).toString();
271  if (!actionMenu.isEmpty()) {
272  const QStringList dbuscall = actionMenu.split(QChar(' '));
273  if (dbuscall.count() >= 4) {
274  const QString &app = dbuscall.at(0);
275  const QString &object = dbuscall.at(1);
276  const QString &interface = dbuscall.at(2);
277  const QString &function = dbuscall.at(3);
278 
279  QDBusInterface remote(app, object, interface);
280  // Do NOT use QDBus::BlockWithGui here. It runs a nested event loop,
281  // in which timers can fire, leading to crashes like #149736.
282  QDBusReply<QStringList> reply = remote.call(function, QUrl::toStringList(file_list));
283  keys = reply; // ensures that the reply was a QStringList
284  if (keys.isEmpty()) {
285  return result;
286  }
287  } else {
288  qWarning() << "The desktop file" << service.entryPath()
289  << "has an invalid X-KDE-GetActionMenu entry."
290  << "Syntax is: app object interface function";
291  }
292  }
293 
294  // Now, either keys is empty (all actions) or it's set to the actions we want
295 
296  foreach (const KServiceAction &action, service.actions()) {
297  if (keys.isEmpty() || keys.contains(action.name())) {
298  const QString exec = action.exec();
299  if (bLocalFiles || exec.contains(QStringLiteral("%U")) || exec.contains(QStringLiteral("%u"))) {
300  result.append(action);
301  }
302  }
303  }
304 
305  return result;
306 }

308 void KDesktopFileActions::executeService(const QList<QUrl> &urls, const KServiceAction &action)
309 {
310  //qDebug() << "EXECUTING Service " << action.name();
311 
312  int actionData = action.data().toInt();
313  if (actionData == ST_MOUNT || actionData == ST_UNMOUNT) {
314  Q_ASSERT(urls.count() == 1);
315  const QString path = urls.first().toLocalFile();
316  //qDebug() << "MOUNT&UNMOUNT";
317 
318  KDesktopFile cfg(path);
319  if (cfg.hasDeviceType()) { // path to desktop file
320  const QString dev = cfg.readDevice();
321  if (dev.isEmpty()) {
322  QString tmp = i18n("The desktop entry file\n%1\nis of type FSDevice but has no Dev=... entry.", path);
323  KMessageBox::error(0, tmp);
324  return;
325  }
326  KMountPoint::Ptr mp = KMountPoint::currentMountPoints().findByDevice(dev);
327 
328  if (actionData == ST_MOUNT) {
329  // Already mounted? Strange, but who knows ...
330  if (mp) {
331  //qDebug() << "ALREADY Mounted";
332  return;
333  }
334 
335  const KConfigGroup group = cfg.desktopGroup();
336  bool ro = group.readEntry("ReadOnly", false);
337  QString fstype = group.readEntry("FSType");
338  if (fstype == QLatin1String("Default")) { // KDE-1 thing
339  fstype.clear();
340  }
341  QString point = group.readEntry("MountPoint");
342 #ifndef Q_OS_WIN
343  (void)new KAutoMount(ro, fstype.toLatin1(), dev, point, path, false);
344 #endif
345  } else if (actionData == ST_UNMOUNT) {
346  // Not mounted? Strange, but who knows ...
347  if (!mp) {
348  return;
349  }
350 
351 #ifndef Q_OS_WIN
352  (void)new KAutoUnmount(mp->mountPoint(), path);
353 #endif
354  }
355  }
356 #if ! KIO_NO_SOLID
357  else { // path to device
358  Solid::Predicate predicate(Solid::DeviceInterface::Block, "device", path);
359  const QList<Solid::Device> devList = Solid::Device::listFromQuery(predicate, QString());
360  if (!devList.empty()) {
361  Solid::Device device = devList[0];
362  if (actionData == ST_MOUNT) {
363  if (device.is<Solid::StorageVolume>()) {
364  Solid::StorageAccess *access = device.as<Solid::StorageAccess>();
365  if (access) {
366  access->setup();
367  }
368  }
369  } else if (actionData == ST_UNMOUNT) {
370  if (device.is<Solid::OpticalDisc>()) {
371  Solid::OpticalDrive *drive = device.parent().as<Solid::OpticalDrive>();
372  if (drive != 0) {
373  drive->eject();
374  }
375  } else if (device.is<Solid::StorageVolume>()) {
376  Solid::StorageAccess *access = device.as<Solid::StorageAccess>();
377  if (access && access->isAccessible()) {
378  access->teardown();
379  }
380  }
381  }
382  } else {
383  //qDebug() << "Device" << path << "not found";
384  }
385  }
386 #endif
387  } else {
388  //qDebug() << action.name() << "first url's path=" << urls.first().toLocalFile() << "exec=" << action.exec();
389  KRun::run(action.exec(), urls, 0, action.text(), action.icon());
390  // The action may update the desktop file. Example: eject unmounts (#5129).
391  org::kde::KDirNotify::emitFilesChanged(urls);
392  }
393 }
