71 void KUrlNavigatorButton::setUrl(const QUrl &url)
72 {
73  m_url = url;
74 
75  // Doing a text-resolving with KIO::stat() for all non-local
76  // URLs leads to problems for protocols where a limit is given for
77  // the number of parallel connections. A black-list
78  // is given where KIO::stat() should not be used:
79  static const QSet<QString> protocolBlacklist = QSet<QString>()
80  << QStringLiteral("nfs") << QStringLiteral("fish")
81  << QStringLiteral("ftp") << QStringLiteral("sftp")
82  << QStringLiteral("smb") << QStringLiteral("webdav");
83 
84  const bool startTextResolving = m_url.isValid() && !m_url.isLocalFile()
85  && !protocolBlacklist.contains(m_url.scheme());
86 
87  if (startTextResolving) {
88  m_pendingTextChange = true;
89  KIO::StatJob *job = KIO::stat(m_url, KIO::HideProgressInfo);
90  connect(job, SIGNAL(result(KJob*)),
91  this, SLOT(statFinished(KJob*)));
92  emit startedTextResolving();
93  } else {
94  setText(m_url.fileName().replace('&', QLatin1String("&&")));
95  }
96 }

157 void KUrlNavigatorButton::paintEvent(QPaintEvent *event)
158 {
159  Q_UNUSED(event);
160 
161  QPainter painter(this);
162 
163  QFont adjustedFont(font());
164  adjustedFont.setBold(m_subDir.isEmpty());
165  painter.setFont(adjustedFont);
166 
167  int buttonWidth = width();
168  int preferredWidth = sizeHint().width();
169  if (preferredWidth < minimumWidth()) {
170  preferredWidth = minimumWidth();
171  }
172  if (buttonWidth > preferredWidth) {
173  buttonWidth = preferredWidth;
174  }
175  const int buttonHeight = height();
176 
177  const QColor fgColor = foregroundColor();
178  drawHoverBackground(&painter);
179 
180  int textLeft = 0;
181  int textWidth = buttonWidth;
182 
183  const bool leftToRight = (layoutDirection() == Qt::LeftToRight);
184 
185  if (!m_subDir.isEmpty()) {
186  // draw arrow
187  const int arrowSize = arrowWidth();
188  const int arrowX = leftToRight ? (buttonWidth - arrowSize) - BorderWidth : BorderWidth;
189  const int arrowY = (buttonHeight - arrowSize) / 2;
190 
191  QStyleOption option;
192  option.initFrom(this);
193  option.rect = QRect(arrowX, arrowY, arrowSize, arrowSize);
194  option.palette = palette();
195  option.palette.setColor(QPalette::Text, fgColor);
196  option.palette.setColor(QPalette::WindowText, fgColor);
197  option.palette.setColor(QPalette::ButtonText, fgColor);
198 
199  if (m_hoverArrow) {
200  // highlight the background of the arrow to indicate that the directories
201  // popup can be opened by a mouse click
202  QColor hoverColor = palette().color(QPalette::HighlightedText);
203  hoverColor.setAlpha(96);
204  painter.setPen(Qt::NoPen);
205  painter.setBrush(hoverColor);
206 
207  int hoverX = arrowX;
208  if (!leftToRight) {
209  hoverX -= BorderWidth;
210  }
211  painter.drawRect(QRect(hoverX, 0, arrowSize + BorderWidth, buttonHeight));
212  }
213 
214  if (leftToRight) {
215  style()->drawPrimitive(QStyle::PE_IndicatorArrowRight, &option, &painter, this);
216  } else {
217  style()->drawPrimitive(QStyle::PE_IndicatorArrowLeft, &option, &painter, this);
218  textLeft += arrowSize + 2 * BorderWidth;
219  }
220 
221  textWidth -= arrowSize + 2 * BorderWidth;
222  }
223 
224  painter.setPen(fgColor);
225  const bool clipped = isTextClipped();
226  const QRect textRect(textLeft, 0, textWidth, buttonHeight);
227  if (clipped) {
228  QColor bgColor = fgColor;
229  bgColor.setAlpha(0);
230  QLinearGradient gradient(textRect.topLeft(), textRect.topRight());
231  if (leftToRight) {
232  gradient.setColorAt(0.8, fgColor);
233  gradient.setColorAt(1.0, bgColor);
234  } else {
235  gradient.setColorAt(0.0, bgColor);
236  gradient.setColorAt(0.2, fgColor);
237  }
238 
239  QPen pen;
240  pen.setBrush(QBrush(gradient));
241  painter.setPen(pen);
242  }
243 
244  int textFlags = clipped ? Qt::AlignVCenter : Qt::AlignCenter;
245  if (m_showMnemonic) {
246  textFlags |= Qt::TextShowMnemonic;
247  painter.drawText(textRect, textFlags, text());
248  } else {
249  painter.drawText(textRect, textFlags, plainText());
250  }
251 }

398 void KUrlNavigatorButton::startSubDirsJob()
399 {
400  if (m_subDirsJob != 0) {
401  return;
402  }
403 
404  const QUrl url = m_replaceButton ? KIO::upUrl(m_url) : m_url;
405  m_subDirsJob = KIO::listDir(url, KIO::HideProgressInfo, false /*no hidden files*/);
406  m_subDirs.clear(); // just to be ++safe
407 
408  connect(m_subDirsJob, SIGNAL(entries(KIO::Job*,KIO::UDSEntryList)),
409  this, SLOT(addEntriesToSubDirs(KIO::Job*,KIO::UDSEntryList)));
410 
411  if (m_replaceButton) {
412  connect(m_subDirsJob, SIGNAL(result(KJob*)), this, SLOT(replaceButton(KJob*)));
413  } else {
414  connect(m_subDirsJob, SIGNAL(result(KJob*)), this, SLOT(openSubDirsMenu(KJob*)));

418 void KUrlNavigatorButton::addEntriesToSubDirs(KIO::Job *job, const KIO::UDSEntryList &entries)
419 {
420  Q_ASSERT(job == m_subDirsJob);
421  Q_UNUSED(job);
422 
423  foreach (const KIO::UDSEntry &entry, entries) {
424  if (entry.isDir()) {
425  const QString name = entry.stringValue(KIO::UDSEntry::UDS_NAME);
426  QString displayName = entry.stringValue(KIO::UDSEntry::UDS_DISPLAY_NAME);
427  if (displayName.isEmpty()) {
428  displayName = name;
429  }
430  if ((name != QLatin1String(".")) && (name != QLatin1String(".."))) {
431  m_subDirs.append(qMakePair(name, displayName));
432  }
433  }
434  }

490 void KUrlNavigatorButton::openSubDirsMenu(KJob *job)
491 {
492  Q_ASSERT(job == m_subDirsJob);
493  m_subDirsJob = 0;
494 
495  if (job->error() || m_subDirs.isEmpty()) {
496  // clear listing
497  return;
498  }
499 
500  NaturalLessThan nlt;
501  qSort(m_subDirs.begin(), m_subDirs.end(), nlt);
502  setDisplayHintEnabled(PopupActiveHint, true);
503  update(); // ensure the button is drawn highlighted
504 
505  if (m_subDirsMenu != 0) {
506  m_subDirsMenu->close();
507  m_subDirsMenu->deleteLater();
508  m_subDirsMenu = 0;
509  }
510 
511  m_subDirsMenu = new KUrlNavigatorMenu(this);
512  initMenu(m_subDirsMenu, 0);
513 
514  const bool leftToRight = (layoutDirection() == Qt::LeftToRight);
515  const int popupX = leftToRight ? width() - arrowWidth() - BorderWidth : 0;
516  const QPoint popupPos = parentWidget()->mapToGlobal(geometry().bottomLeft() + QPoint(popupX, 0));
517 
518  QPointer<QObject> guard(this);
519 
520  m_subDirsMenu->exec(popupPos);
521 
522  // If 'this' has been deleted in the menu's nested event loop, we have to return
523  // immediatedely because any access to a member variable might cause a crash.
524  if (!guard) {
525  return;
526  }
527 
528  m_subDirs.clear();
529  delete m_subDirsMenu;
530  m_subDirsMenu = 0;
531 
532  setDisplayHintEnabled(PopupActiveHint, false);
533 }

535 void KUrlNavigatorButton::replaceButton(KJob *job)
536 {
537  Q_ASSERT(job == m_subDirsJob);
538  m_subDirsJob = 0;
539  m_replaceButton = false;
540 
541  if (job->error() || m_subDirs.isEmpty()) {
542  return;
543  }
544 
545  NaturalLessThan nlt;
546  qSort(m_subDirs.begin(), m_subDirs.end(), nlt);
547 
548  // Get index of the directory that is shown currently in the button
549  const QString currentDir = m_url.fileName();
550  int currentIndex = 0;
551  const int subDirsCount = m_subDirs.count();
552  while (currentIndex < subDirsCount) {
553  if (m_subDirs[currentIndex].first == currentDir) {
554  break;
555  }
556  ++currentIndex;
557  }
558 
559  // Adjust the index by respecting the wheel steps and
560  // trigger a replacing of the button content
561  int targetIndex = currentIndex - m_wheelSteps;
562  if (targetIndex < 0) {
563  targetIndex = 0;
564  } else if (targetIndex >= subDirsCount) {
565  targetIndex = subDirsCount - 1;
566  }
567 
568  QUrl url(KIO::upUrl(m_url));
569  url.setPath(url.path() + '/' + m_subDirs[targetIndex].first);
570  emit clicked(url, Qt::LeftButton);
571 
572  m_subDirs.clear();
573 }

584 QString KUrlNavigatorButton::plainText() const
585 {
586  // Replace all "&&" by '&' and remove all single
587  // '&' characters
588  const QString source = text();
589  const int sourceLength = source.length();
590 
591  QString dest;
592  dest.reserve(sourceLength);
593 
594  int sourceIndex = 0;
595  int destIndex = 0;
596  while (sourceIndex < sourceLength) {
597  if (source.at(sourceIndex) == QLatin1Char('&')) {
598  ++sourceIndex;
599  if (sourceIndex >= sourceLength) {
600  break;
601  }
602  }
603  dest[destIndex] = source.at(sourceIndex);
604  ++sourceIndex;
605  ++destIndex;
606  }
607 
608  return dest;
609 }

659 void KUrlNavigatorButton::initMenu(KUrlNavigatorMenu *menu, int startIndex)
660 {
661  connect(menu, SIGNAL(mouseButtonClicked(QAction*, Qt::MouseButton)),
662  this, SLOT(slotMenuActionClicked(QAction*, Qt::MouseButton)));
663  connect(menu, SIGNAL(urlsDropped(QAction*,QDropEvent*)),
664  this, SLOT(urlsDropped(QAction*,QDropEvent*)));
665 
666  menu->setLayoutDirection(Qt::LeftToRight);
667 
668  const int maxIndex = startIndex + 30; // Don't show more than 30 items in a menu
669  const int lastIndex = qMin(m_subDirs.count() - 1, maxIndex);
670  for (int i = startIndex; i <= lastIndex; ++i) {
671  const QString subDirName = m_subDirs[i].first;
672  const QString subDirDisplayName = m_subDirs[i].second;
673  QString text = KStringHandler::csqueeze(subDirDisplayName, 60);
674  text.replace(QLatin1Char('&'), QLatin1String("&&"));
675  QAction *action = new QAction(text, this);
676  if (m_subDir == subDirName) {
677  QFont font(action->font());
678  font.setBold(true);
679  action->setFont(font);
680  }
681  action->setData(i);
