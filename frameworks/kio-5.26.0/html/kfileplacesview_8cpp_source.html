137 void KFilePlacesViewDelegate::paint(QPainter *painter, const QStyleOptionViewItem &option, const QModelIndex &index) const
138 {
139  painter->save();
140 
141  if (m_appearingItems.contains(index)) {
142  painter->setOpacity(m_appearingOpacity);
143  } else if (m_disappearingItems.contains(index)) {
144  painter->setOpacity(m_disappearingOpacity);
145  }
146 
147  QStyleOptionViewItem opt = option;
148  if (!m_showHoverIndication) {
149  opt.state &= ~QStyle::State_MouseOver;
150  }
151  QApplication::style()->drawPrimitive(QStyle::PE_PanelItemViewItem, &opt, painter);
152  const KFilePlacesModel *placesModel = static_cast<const KFilePlacesModel *>(index.model());
153 
154  bool isLTR = option.direction == Qt::LeftToRight;
155 
156  QIcon icon = index.model()->data(index, Qt::DecorationRole).value<QIcon>();
157  QPixmap pm = icon.pixmap(m_iconSize, m_iconSize, (option.state & QStyle::State_Selected) && (option.state & QStyle::State_Active) ? QIcon::Selected : QIcon::Normal);
158  QPoint point(isLTR ? option.rect.left() + LATERAL_MARGIN
159  : option.rect.right() - LATERAL_MARGIN - m_iconSize, option.rect.top() + (option.rect.height() - m_iconSize) / 2);
160  painter->drawPixmap(point, pm);
161 
162  if (option.state & QStyle::State_Selected) {
163  QPalette::ColorGroup cg = QPalette::Active;
164  if (!(option.state & QStyle::State_Enabled)) {
165  cg = QPalette::Disabled;
166  } else if (!(option.state & QStyle::State_Active)) {
167  cg = QPalette::Inactive;
168  }
169  painter->setPen(option.palette.color(cg, QPalette::HighlightedText));
170  }
171 
172  QRect rectText;
173 
174  const QUrl url = placesModel->url(index);
175  bool drawCapacityBar = false;
176  if (url.isLocalFile()) {
177  const QString mountPointPath = placesModel->url(index).toLocalFile();
178  const KDiskFreeSpaceInfo info = KDiskFreeSpaceInfo::freeSpaceInfo(mountPointPath);
179  drawCapacityBar = info.size() != 0 &&
180  placesModel->data(index, KFilePlacesModel::CapacityBarRecommendedRole).toBool();
181 
182  if (drawCapacityBar && contentsOpacity(index) > 0) {
183  painter->save();
184  painter->setOpacity(painter->opacity() * contentsOpacity(index));
185 
186  int height = option.fontMetrics.height() + CAPACITYBAR_HEIGHT;
187  rectText = QRect(isLTR ? m_iconSize + LATERAL_MARGIN * 2 + option.rect.left()
188  : 0, option.rect.top() + (option.rect.height() / 2 - height / 2), option.rect.width() - m_iconSize - LATERAL_MARGIN * 2, option.fontMetrics.height());
189  painter->drawText(rectText, Qt::AlignLeft | Qt::AlignTop, option.fontMetrics.elidedText(index.model()->data(index).toString(), Qt::ElideRight, rectText.width()));
190  QRect capacityRect(isLTR ? rectText.x() : LATERAL_MARGIN, rectText.bottom() - 1, rectText.width() - LATERAL_MARGIN, CAPACITYBAR_HEIGHT);
191  KCapacityBar capacityBar(KCapacityBar::DrawTextInline);
192  capacityBar.setValue((info.used() * 100) / info.size());
193  capacityBar.drawCapacityBar(painter, capacityRect);
194 
195  painter->restore();
196 
197  painter->save();
198  painter->setOpacity(painter->opacity() * (1 - contentsOpacity(index)));
199  }
200  }
201 
202  rectText = QRect(isLTR ? m_iconSize + LATERAL_MARGIN * 2 + option.rect.left()
203  : 0, option.rect.top(), option.rect.width() - m_iconSize - LATERAL_MARGIN * 2, option.rect.height());
204  painter->drawText(rectText, Qt::AlignLeft | Qt::AlignVCenter, option.fontMetrics.elidedText(index.model()->data(index).toString(), Qt::ElideRight, rectText.width()));
205 
206  if (drawCapacityBar && contentsOpacity(index) > 0) {
207  painter->restore();
208  }
209 
210  painter->restore();
211 }

228 void KFilePlacesViewDelegate::setAppearingItemProgress(qreal value)
229 {
230  if (value <= 0.25) {
231  m_appearingOpacity = 0.0;
232  m_appearingIconSize = iconSize() * value * 4;
233 
234  if (m_appearingIconSize >= m_iconSize) {
235  m_appearingIconSize = m_iconSize;
236  }
237  } else {
238  m_appearingIconSize = m_iconSize;
239  m_appearingOpacity = (value - 0.25) * 4 / 3;
240 
241  if (value >= 1.0) {
242  m_appearingItems.clear();
243  }
244  }
245 }

459 void KFilePlacesView::setUrl(const QUrl &url)
460 {
461  QUrl oldUrl = d->currentUrl;
462  KFilePlacesModel *placesModel = qobject_cast<KFilePlacesModel *>(model());
463 
464  if (placesModel == 0) {
465  return;
466  }
467 
468  QModelIndex index = placesModel->closestItem(url);
469  QModelIndex current = selectionModel()->currentIndex();
470 
471  if (index.isValid()) {
472  if (current != index && placesModel->isHidden(current) && !d->showAll) {
473  KFilePlacesViewDelegate *delegate = static_cast<KFilePlacesViewDelegate *>(itemDelegate());
474  delegate->addDisappearingItem(current);
475 
476  if (d->itemDisappearTimeline.state() != QTimeLine::Running) {
477  delegate->setDisappearingItemProgress(0.0);
478  d->itemDisappearTimeline.start();
479  }
480  }
481 
482  if (current != index && placesModel->isHidden(index) && !d->showAll) {
483  KFilePlacesViewDelegate *delegate = static_cast<KFilePlacesViewDelegate *>(itemDelegate());
484  delegate->addAppearingItem(index);
485 
486  if (d->itemAppearTimeline.state() != QTimeLine::Running) {
487  delegate->setAppearingItemProgress(0.0);
488  d->itemAppearTimeline.start();
489  }
490 
491  setRowHidden(index.row(), false);
492  }
493 
494  d->currentUrl = url;
495  selectionModel()->setCurrentIndex(index, QItemSelectionModel::ClearAndSelect);
496  } else {
497  d->currentUrl = QUrl();
498  selectionModel()->clear();
499  }
500 
501  if (!current.isValid()) {
502  d->updateHiddenRows();
503  }
504 }

506 void KFilePlacesView::setShowAll(bool showAll)
507 {
508  KFilePlacesModel *placesModel = qobject_cast<KFilePlacesModel *>(model());
509 
510  if (placesModel == 0) {
511  return;
512  }
513 
514  d->showAll = showAll;
515 
516  KFilePlacesViewDelegate *delegate = static_cast<KFilePlacesViewDelegate *>(itemDelegate());
517 
518  int rowCount = placesModel->rowCount();
519  QModelIndex current = placesModel->closestItem(d->currentUrl);
520 
521  if (showAll) {
522  d->updateHiddenRows();
523 
524  for (int i = 0; i < rowCount; ++i) {
525  QModelIndex index = placesModel->index(i, 0);
526  if (index != current && placesModel->isHidden(index)) {
527  delegate->addAppearingItem(index);
528  }
529  }
530 
531  if (d->itemAppearTimeline.state() != QTimeLine::Running) {
532  delegate->setAppearingItemProgress(0.0);
533  d->itemAppearTimeline.start();
534  }
535  } else {
536  for (int i = 0; i < rowCount; ++i) {
537  QModelIndex index = placesModel->index(i, 0);
538  if (index != current && placesModel->isHidden(index)) {
539  delegate->addDisappearingItem(index);
540  }
541  }
542 
543  if (d->itemDisappearTimeline.state() != QTimeLine::Running) {
544  delegate->setDisappearingItemProgress(0.0);
545  d->itemDisappearTimeline.start();
546  }
547  }

558 void KFilePlacesView::contextMenuEvent(QContextMenuEvent *event)
559 {
560  KFilePlacesModel *placesModel = qobject_cast<KFilePlacesModel *>(model());
561  KFilePlacesViewDelegate *delegate = dynamic_cast<KFilePlacesViewDelegate *>(itemDelegate());
562 
563  if (placesModel == 0) {
564  return;
565  }
566 
567  QModelIndex index = indexAt(event->pos());
568  QString label = placesModel->text(index).replace('&', QLatin1String("&&"));
569 
570  QMenu menu;
571 
572  QAction *edit = 0;
573  QAction *hide = 0;
574  QAction *emptyTrash = 0;
575  QAction *eject = 0;
576  QAction *teardown = 0;
577  QAction *add = 0;
578  QAction *mainSeparator = 0;
579 
580  if (index.isValid()) {
581  if (!placesModel->isDevice(index)) {
582  if (placesModel->url(index).toString() == QLatin1String("trash:/")) {
583  emptyTrash = menu.addAction(QIcon::fromTheme(QStringLiteral("trash-empty")), i18nc("@action:inmenu", "Empty Trash"));
584  KConfig trashConfig(QStringLiteral("trashrc"), KConfig::SimpleConfig);
585  emptyTrash->setEnabled(!trashConfig.group("Status").readEntry("Empty", true));
586  menu.addSeparator();
587  }
588  add = menu.addAction(QIcon::fromTheme(QStringLiteral("document-new")), i18n("Add Entry..."));
589  mainSeparator = menu.addSeparator();
590  edit = menu.addAction(QIcon::fromTheme(QStringLiteral("document-properties")), i18n("&Edit Entry '%1'...", label));
591  } else {
592  eject = placesModel->ejectActionForIndex(index);
593  if (eject != 0) {
594  eject->setParent(&menu);
595  menu.addAction(eject);
596  }
597 
598  teardown = placesModel->teardownActionForIndex(index);
599  if (teardown != 0) {
600  teardown->setParent(&menu);
601  menu.addAction(teardown);
602  }
603 
604  if (teardown != 0 || eject != 0) {
605  mainSeparator = menu.addSeparator();
606  }
607  }
608  if (add == 0) {
609  add = menu.addAction(QIcon::fromTheme(QStringLiteral("document-new")), i18n("Add Entry..."));

920 void KFilePlacesView::Private::adaptItemSize()
921 {
922  KFilePlacesViewDelegate *delegate = dynamic_cast<KFilePlacesViewDelegate *>(q->itemDelegate());
923  if (!delegate) {
924  return;
925  }
926 
927  if (!autoResizeItems) {
928  int size = q->iconSize().width(); // Assume width == height
929  delegate->setIconSize(size);
930  q->scheduleDelayedItemsLayout();
931  return;
932  }
933 
934  KFilePlacesModel *placesModel = qobject_cast<KFilePlacesModel *>(q->model());
935 
936  if (placesModel == 0) {
937  return;
938  }
939 
940  int rowCount = placesModel->rowCount();
941 
942  if (!showAll) {
943  rowCount -= placesModel->hiddenCount();
944 
945  QModelIndex current = placesModel->closestItem(currentUrl);
946 
947  if (placesModel->isHidden(current)) {
948  rowCount++;
949  }
950  }
951 
952  if (rowCount == 0) {
953  return; // We've nothing to display anyway
954  }
955 
956  const int minSize = IconSize(KIconLoader::Small);
957  const int maxSize = 64;
958 
959  int textWidth = 0;
960  QFontMetrics fm = q->fontMetrics();
961  for (int i = 0; i < placesModel->rowCount(); ++i) {
962  QModelIndex index = placesModel->index(i, 0);
963 
964  if (!placesModel->isHidden(index)) {
965  textWidth = qMax(textWidth, fm.width(index.data(Qt::DisplayRole).toString()));
966  }
967  }
968 
969  const int margin = q->style()->pixelMetric(QStyle::PM_FocusFrameHMargin, 0, q) + 1;
970  const int maxWidth = q->viewport()->width() - textWidth - 4 * margin - 1;
971  const int maxHeight = ((q->height() - (fm.height() / 2) * rowCount) / rowCount) - 1;
972 
973  int size = qMin(maxHeight, maxWidth);
974 
975  if (size < minSize) {
976  size = minSize;
977  } else if (size > maxSize) {
978  size = maxSize;
979  } else {
980  // Make it a multiple of 16
981  size &= ~0xf;
982  }
983 
984  if (size == delegate->iconSize()) {
985  return;
986  }
987 
988  if (smoothItemResizing) {
989  oldSize = delegate->iconSize();
990  endSize = size;
991  if (adaptItemsTimeline.state() != QTimeLine::Running) {
992  adaptItemsTimeline.start();
993  }
994  } else {
995  delegate->setIconSize(size);
996  q->scheduleDelayedItemsLayout();
997  }
998 }
