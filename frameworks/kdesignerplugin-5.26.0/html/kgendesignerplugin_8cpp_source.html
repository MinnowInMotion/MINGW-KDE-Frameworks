143 void buildFile(QTextStream &ts, const QString &group, const QString &fileName, const QString &pluginName)
144 {
145  KConfig input(fileName, KConfig::NoGlobals);
146  KConfigGroup cg(&input, "Global");
147  ts << classHeader << endl;
148 
149  QString defaultGroup = cg.readEntry("DefaultGroup", group);
150  QStringList includes = cg.readEntry("Includes", QStringList());
151  QStringList classes = input.groupList();
152  classes.removeAll("Global");
153 
154  foreach (const QString &myInclude, classes) {
155  includes += buildWidgetInclude(myInclude, input);
156  }
157 
158  foreach (const QString &myInclude, includes) {
159  ts << "#include <" << myInclude << ">" << endl;
160  }
161 
162  ts << QLatin1String("\n\n");
163 
164  // Autogenerate widget defs here
165  foreach (const QString &myClass, classes) {
166  ts << buildWidgetClass(myClass, input, defaultGroup) << endl;
167  }
168 
169  ts << buildCollClass(input, classes, pluginName);
170 
171 }

180 QString buildCollClass(KConfig &_input, const QStringList &classes, const QString &pluginName)
181 {
182  KConfigGroup input(&_input, "Global");
183  QHash<QString, QString> defMap;
184  const QString collName = input.readEntry("PluginName", pluginName);
185  Q_ASSERT(!collName.isEmpty());
186  defMap.insert("CollName", collName);
187  QString genCode;
188 
189  foreach (const QString &myClass, classes) {
190  genCode += QString("\t\tm_plugins.append( new %1(this) );\n").arg(denamespace(myClass) + "Plugin");
191  }
192 
193  defMap.insert("CollectionAdd", genCode);
194 
195  QString str = KMacroExpander::expandMacros(collClassDef, defMap);
196  str += KMacroExpander::expandMacros(collClassImpl, defMap);
197  return str;
198 }

200 QString buildWidgetClass(const QString &name, KConfig &_input, const QString &group)
201 {
202  KConfigGroup input(&_input, name);
203  QHash<QString, QString> defMap;
204 
205  defMap.insert("Group", input.readEntry("Group", group).replace('\"', "\\\""));
206  defMap.insert("IncludeFile", input.readEntry("IncludeFile", QString(name.toLower() + ".h")).remove(':'));
207  defMap.insert("ToolTip", input.readEntry("ToolTip", QString(name + " Widget")).replace('\"', "\\\""));
208  defMap.insert("WhatsThis", input.readEntry("WhatsThis", QString(name + " Widget")).replace('\"', "\\\""));
209  defMap.insert("IsContainer", input.readEntry("IsContainer", "false"));
210  defMap.insert("IconName", input.readEntry("IconName", QString::fromLatin1(":/pics/%1.png").arg(denamespace(name).toLower())));
211  defMap.insert("Class", name);
212  defMap.insert("PluginName", denamespace(name) + QLatin1String("Plugin"));
213 
214  // FIXME: ### make this more useful, i.e. outsource to separate file
215  QString domXml = input.readEntry("DomXML", QString());
216  // If domXml is empty then we should call base class function
217  if (domXml.isEmpty()) {
218  domXml = QStringLiteral("QDesignerCustomWidgetInterface::domXml()");
219  } else {
220  domXml = QStringLiteral("QStringLiteral(\"%1\")").arg(domXml.replace('\"', "\\\""));
221  }
222  defMap.insert("DomXml", domXml);
223  defMap.insert("CodeTemplate", input.readEntry("CodeTemplate"));
224  defMap.insert("CreateWidget", input.readEntry("CreateWidget",
225  QString("\n\t\treturn new %1%2;")
226  .arg(input.readEntry("ImplClass", name))
227  .arg(input.readEntry("ConstructorArgs", "( parent )"))));
228  defMap.insert("Initialize", input.readEntry("Initialize", "\n\t\tQ_UNUSED(core);\n\t\tif (mInitialized) return;\n\t\tmInitialized=true;"));
229 
230  return KMacroExpander::expandMacros(classDef, defMap);
231 }
