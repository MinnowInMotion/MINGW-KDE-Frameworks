38 QString Kuit::escape(const QString &text)
39 {
40  int tlen = text.length();
41  QString ntext;
42  ntext.reserve(tlen);
43  for (int i = 0; i < tlen; ++i) {
44  QChar c = text[i];
45  if (c == QL1C('&')) {
46  ntext += QStringLiteral("&amp;");
47  } else if (c == QL1C('<')) {
48  ntext += QStringLiteral("&lt;");
49  } else if (c == QL1C('>')) {
50  ntext += QStringLiteral("&gt;");
51  } else if (c == QL1C('\'')) {
52  ntext += QStringLiteral("&apos;");
53  } else if (c == QL1C('"')) {
54  ntext += QStringLiteral("&quot;");
55  } else {
56  ntext += c;
57  }
58  }
59 
60  return ntext;
61 }

76 static void parseUiMarker(const QString &context_,
77  QString &roleName,
78  QString &cueName,
79  QString &formatName)
80 {
81  // UI marker is in the form @role:cue/format,
82  // and must start just after any leading whitespace in the context string.
83  QString context = context_.trimmed();
84  if (context.startsWith(QL1C('@'))) { // found UI marker
85  static QRegExp staticWsRx(QStringLiteral("\\s"));
86  QRegExp wsRx = staticWsRx; // QRegExp not thread-safe
87  context = context.mid(1, wsRx.indexIn(context) - 1);
88 
89  // Possible format.
90  int pfmt = context.indexOf(QL1C('/'));
91  if (pfmt >= 0) {
92  formatName = context.mid(pfmt + 1);
93  context.truncate(pfmt);
94  }
95 
96  // Possible subcue.
97  int pcue = context.indexOf(QL1C(':'));
98  if (pcue >= 0) {
99  cueName = context.mid(pcue + 1);
100  context.truncate(pcue);
101  }
102 
103  // Role.
104  roleName = context;
105  }
106  // Names remain untouched if marker was not found, which is fine.
107 
108  // Normalize names.
109  roleName = roleName.trimmed().toLower();
110  cueName = cueName.trimmed().toLower();
111  formatName = formatName.trimmed().toLower();
112 }

378 QString KuitStaticData::toKeyCombo(const QStringList &languages,
379  const QString &shstr,
380  Kuit::VisualFormat format)
381 {
382  // Take '+' or '-' as input shortcut delimiter,
383  // whichever is first encountered.
384  static QRegExp staticDelimRx(QStringLiteral("[+-]"));
385  QRegExp delimRx = staticDelimRx; // QRegExp not thread-safe
386 
387  int p = delimRx.indexIn(shstr); // find delimiter
388  QStringList keys;
389  if (p < 0) { // single-key shortcut, no delimiter found
390  keys.append(shstr);
391  } else { // multi-key shortcut
392  QChar oldDelim = shstr[p];
393  keys = shstr.split(oldDelim, QString::SkipEmptyParts);
394  }
395 
396  for (int i = 0; i < keys.size(); ++i) {
397  // Normalize key, trim and all lower-case.
398  QString nkey = keys[i].trimmed().toLower();
399  keys[i] = keyNames.contains(nkey) ? keyNames[nkey].toString(languages)
400  : keys[i].trimmed();
401  }
402  QString delim = comboKeyDelim.value(format).toString(languages);
403  return keys.join(delim);
404 }

406 QString KuitStaticData::toInterfacePath(const QStringList &languages,
407  const QString &inpstr,
408  Kuit::VisualFormat format)
409 {
410  // Take '/', '|' or "->" as input path delimiter,
411  // whichever is first encountered.
412  static QRegExp staticDelimRx(QStringLiteral("\\||->"));
413  QRegExp delimRx = staticDelimRx; // QRegExp not thread-safe

421  QString delim = guiPathDelim.value(format).toString(languages);
422  return guiels.join(delim);

454 QString KuitTag::format(const QStringList &languages,
455  const QHash<QString, QString> &attributes,
456  const QString &text,
457  const QStringList &tagPath,
458  Kuit::VisualFormat format) const
459 {
460  KuitStaticData *s = staticData();
461  QString formattedText = text;
462  QString attribKey = attributeSetKey(attributes.keys());
463  if (patterns.contains(attribKey) && patterns.value(attribKey).contains(format)) {
464  QString modText;
465  Kuit::TagFormatter formatter = formatters.value(attribKey).value(format);
466  if (formatter != NULL) {
467  modText = formatter(languages, name, attributes, text, tagPath, format);
468  } else {
469  modText = text;
470  }
471  KLocalizedString aggText = patterns.value(attribKey).value(format);
472  // line below is first-aid fix.for e.g. <emphasis strong='true'>.
473  // TODO: proper handling of boolean attributes still needed
474  aggText = aggText.relaxSubs();
475  if (!aggText.isEmpty()) {
476  aggText = aggText.subs(modText);
477  QStringList attributeOrder = attributeOrders.value(attribKey).value(format);
478  foreach (const QString &attribName, attributeOrder) {
479  aggText = aggText.subs(attributes.value(attribName));
480  }
481  formattedText = aggText.ignoreMarkup().toString(languages);
482  } else {
483  formattedText = modText;
484  }
485  } else if (patterns.contains(attribKey)) {
486  qWarning() << QStringLiteral(
487  "Undefined visual format for tag <%1> and attribute combination %2: %3.")
488  .arg(name, attribKey, s->namesByFormat.value(format));
489  } else {
490  qWarning() << QStringLiteral(
491  "Undefined attribute combination for tag <%1>: %2.")
492  .arg(name, attribKey);
493  }
494  return formattedText;
495 }
