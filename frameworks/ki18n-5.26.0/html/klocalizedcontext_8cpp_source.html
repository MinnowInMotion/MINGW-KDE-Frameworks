61 static void subsVariant(KLocalizedString &trMessage, const QVariant &value)
62 {
63  switch(value.type()) {
64  case QVariant::String:
65  trMessage = trMessage.subs(value.toString());
66  break;
67  case QVariant::Int:
68  trMessage = trMessage.subs(value.toInt());
69  break;
70  case QVariant::Double:
71  trMessage = trMessage.subs(value.toDouble());
72  break;
73  case QVariant::Char:
74  trMessage = trMessage.subs(value.toChar());
75  break;
76  default:
77  if (value.canConvert(QVariant::String)) {
78  trMessage = trMessage.subs(value.toString());
79  } else {
80  trMessage = trMessage.subs(QStringLiteral("???"));
81  qWarning() << "couldn't convert" << value << "to translate";
82  }
83 
84  }
85 }
