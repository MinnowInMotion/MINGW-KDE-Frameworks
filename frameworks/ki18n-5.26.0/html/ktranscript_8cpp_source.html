257 // Normalize string key for hash lookups,
258 QByteArray normKeystr(const QString &raw, bool mayHaveAcc = true)
259 {
260  // NOTE: Regexes should not be used here for performance reasons.
261  // This function may potentially be called thousands of times
262  // on application startup.
263 
264  QString key = raw;
265 
266  // Strip all whitespace.
267  int len = key.length();
268  QString nkey;
269  for (int i = 0; i < len; ++i) {
270  QChar c = key[i];
271  if (!c.isSpace()) {
272  nkey.append(c);
273  }
274  }
275  key = nkey;
276 
277  // Strip accelerator marker.
278  if (mayHaveAcc) {
279  key = removeAcceleratorMarker(key);
280  }
281 
282  // Convert to lower case.
283  key = key.toLower();
284 
285  return key.toUtf8();
286 }

288 // ----------------------------------------------------------------------
289 // Trim multiline string in a "smart" way:
290 // Remove leading and trailing whitespace up to and including first
291 // newline from that side, if there is one; otherwise, don't touch.
292 QString trimSmart(const QString &raw)
293 {
294  // NOTE: This could be done by a single regex, but is not due to
295  // performance reasons.
296  // This function may potentially be called thousands of times
297  // on application startup.
298 
299  int len = raw.length();
300 
301  int is = 0;
302  while (is < len && raw[is].isSpace() && raw[is] != QLatin1Char('\n')) {
303  ++is;
304  }
305  if (is >= len || raw[is] != QLatin1Char('\n')) {
306  is = -1;
307  }
308 
309  int ie = len - 1;
310  while (ie >= 0 && raw[ie].isSpace() && raw[ie] != QLatin1Char('\n')) {
311  --ie;
312  }
313  if (ie < 0 || raw[ie] != QLatin1Char('\n')) {
314  ie = len;
315  }
316 
317  return raw.mid(is + 1, ie - is - 1);
318 }

323 QScriptValue variantToJsValue(const QVariant &val)
324 {
325  QVariant::Type vtype = val.type();
326  if (vtype == QVariant::String) {
327  return QScriptValue(val.toString());

329  return QScriptValue(val.toBool());
330  } else if (vtype == QVariant::Double
331  || vtype == QVariant::Int || vtype == QVariant::UInt
332  || vtype == QVariant::LongLong || vtype == QVariant::ULongLong) {

339 // ----------------------------------------------------------------------
340 // Parse ini-style config file,
341 // returning content as hash of hashes by group and key.
342 // Parsing is not fussy, it will read what it can.
343 TsConfig readConfig(const QString &fname)
344 {
345  TsConfig config;
346  // Add empty group.
347  TsConfig::iterator configGroup;
348  configGroup = config.insert(QString(), TsConfigGroup());
349 
350  QFile file(fname);
351  if (!file.open(QIODevice::ReadOnly)) {
352  return config;
353  }
354  QTextStream stream(&file);
355  stream.setCodec("UTF-8");
356  while (!stream.atEnd()) {
357  QString line = stream.readLine();
358  int p1, p2;
359 
360  // Remove comment from the line.
361  p1 = line.indexOf(QLatin1Char('#'));
362  if (p1 >= 0) {
363  line.truncate(p1);
364  }
365  line = line.trimmed();
366  if (line.isEmpty()) {
367  continue;
368  }
369 
370  if (line[0] == QLatin1Char('[')) {
371  // Group switch.
372  p1 = 0;
373  p2 = line.indexOf(QLatin1Char(']'), p1 + 1);
374  if (p2 < 0) {
375  continue;
376  }
377  QString group = line.mid(p1 + 1, p2 - p1 - 1).trimmed();
378  configGroup = config.find(group);
379  if (configGroup == config.end()) {
380  // Add new group.
381  configGroup = config.insert(group, TsConfigGroup());
382  }
383  } else {
384  // Field.
385  p1 = line.indexOf(QLatin1Char('='));
386  if (p1 < 0) {
387  continue;
388  }
389  QStringRef field = line.leftRef(p1).trimmed();
390  QStringRef value = line.midRef(p1 + 1).trimmed();
391  if (!field.isEmpty()) {
392  (*configGroup)[field.toString()] = value.toString();
393  }
394  }
395  }
396  file.close();
397 
398  return config;
399 }

488 QString KTranscriptImp::eval(const QList<QVariant> &argv,
489  const QString &lang,
490  const QString &ctry,
491  const QString &msgctxt,
492  const QHash<QString, QString> &dynctxt,
493  const QString &msgid,
494  const QStringList &subs,
495  const QList<QVariant> &vals,
496  const QString &ftrans,
497  QList<QStringList> &mods,
498  QString &error,
499  bool &fallback)
500 {
501  //error = "debug"; return QString();
502 
503  error.clear(); // empty error message means successful evaluation
504  fallback = false; // fallback not requested
505 
506 #if 0
507  // FIXME: Maybe not needed, as QScriptEngine has no native outside access?
508  // Unportable (needs unistd.h)?
509 
510  // If effective user id is root and real user id is not root.
511  if (geteuid() == 0 && getuid() != 0) {
512  // Since scripts are user input, and the program is running with
513  // root permissions while real user is not root, do not invoke
514  // scripting at all, to prevent exploits.
515  error = "Security block: trying to execute a script in suid environment.";
516  return QString();
517  }
518 #endif
519 
520  // Load any new modules and clear the list.
521  if (!mods.isEmpty()) {
522  loadModules(mods, error);
523  mods.clear();
524  if (!error.isEmpty()) {
525  return QString();
526  }
527  }
528 
529  // Add interpreters for new languages.
530  // (though it should never happen here, but earlier when loading modules;
531  // this also means there are no calls set, so the unregistered call error
532  // below will be reported).
533  if (!m_sface.contains(lang)) {
534  setupInterpreter(lang);
535  }
536 
537  // Shortcuts.
538  Scriptface *sface = m_sface[lang];
539 
540  QScriptEngine *engine = sface->scriptEngine;
541  QScriptValue gobj = engine->globalObject();
542 
543  // Link current message data for script-side interface.
544  sface->msgcontext = &msgctxt;
545  sface->dyncontext = &dynctxt;
546  sface->msgId = &msgid;
547  sface->subList = &subs;
548  sface->valList = &vals;
549  sface->ftrans = &ftrans;
550  sface->fallbackRequest = &fallback;
551  sface->ctry = &ctry;
552 
553  // Find corresponding JS function.
554  int argc = argv.size();
555  if (argc < 1) {
556  //error = "At least the call name must be supplied.";
557  // Empty interpolation is OK, possibly used just to initialize
558  // at a given point (e.g. for Ts.setForall() to start having effect).
559  return QString();
560  }
561  QString funcName = argv[0].toString();
562  if (!sface->funcs.contains(funcName)) {
563  error = QStringLiteral("Unregistered call to '%1'.").arg(funcName);
564  return QString();
565  }
566 
567  QScriptValue func = sface->funcs[funcName];
568  QScriptValue fval = sface->fvals[funcName];
569 
570  // Recover module path from the time of definition of this call,
571  // for possible load calls.
572  currentModulePath = sface->fpaths[funcName];
573 
574  // Execute function.
575  QScriptValueList arglist;
576  arglist.reserve(argc-1);
577  for (int i = 1; i < argc; ++i) {
578  arglist.append(variantToJsValue(argv[i]));
579  }
580 
581  QScriptValue val;
582  if (fval.isObject()) {
583  val = func.call(fval.toObject(), arglist);
584  } else { // no object associated to this function, use global
585  val = func.call(gobj, arglist);
586  }
587 
588  if (fallback) {
589  // Fallback to ordinary translation requested.
590 
591  // Possibly clear exception state.
592  if (engine->hasUncaughtException()) {
593  engine->clearExceptions();
594  }
595 
596  return QString();
597  } else if (!engine->hasUncaughtException()) {
598  // Evaluation successful.
599 
600  if (val.isString()) {
601  // Good to go.
602 
603  return val.toString();
604  } else {
605  // Accept only strings.
606 
607  QString strval = val.toString();
608  error = QStringLiteral("Non-string return value: %1").arg(strval);
609  return QString();
610  }
611  } else {
612  // Exception raised.
613 
614  error = expt2str(engine);
615 
616  engine->clearExceptions();
617 
618  return QString();
619  }
620 }

636 void KTranscriptImp::loadModules(const QList<QStringList> &mods,
637  QString &error)
638 {
639  QList<QString> modErrors;
640 
641  foreach (const QStringList &mod, mods) {
642  QString mpath = mod[0];
643  QString mlang = mod[1];
644 
645  // Add interpreters for new languages.
646  if (!m_sface.contains(mlang)) {
647  setupInterpreter(mlang);
648  }
649 
650  // Setup current module path for loading submodules.
651  // (sort of closure over invocations of loadf)
652  int posls = mpath.lastIndexOf(QLatin1Char('/'));
653  if (posls < 1) {
654  modErrors.append(QStringLiteral(
655  "Funny module path '%1', skipping.").arg(mpath));
656  continue;
657  }
658  currentModulePath = mpath.left(posls);
659  QString fname = mpath.mid(posls + 1);
660  // Scriptface::loadf() wants no extension on the filename
661  fname = fname.left(fname.lastIndexOf(QLatin1Char('.')));
662 
663  // Load the module.
664  QScriptEngine *engine = m_sface[mlang]->scriptEngine;
665  QScriptValueList alist;
666  alist.append(QScriptValue(fname));
667 
668  m_sface[mlang]->load(alist);
669 
670  // Handle any exception.
671  if (engine->hasUncaughtException()) {
672  modErrors.append(expt2str(engine));
673  engine->clearExceptions();
674  }
675  }
676 
677  // Unset module path.
678  currentModulePath.clear();
679 
680  foreach (const QString &merr, modErrors) {
681  error.append(merr + QLatin1Char('\n'));
682  }

1090 static QString toCaseFirst(const QString &qstr, int qnalt, bool toupper)
1091 {
1092  static const QLatin1String head("~@");
1093  static const int hlen = 2; //head.length()
1094 
1095  // If the first letter is found within an alternatives directive,
1096  // change case of the first letter in each of the alternatives.
1097  QString qstrcc = qstr;
1098  int len = qstr.length();
1099  QChar altSep;
1100  int remainingAlts = 0;
1101  bool checkCase = true;
1102  int numChcased = 0;
1103  int i = 0;
1104  while (i < len) {
1105  QChar c = qstr[i];
1106 
1107  if (qnalt && !remainingAlts && qstr.midRef(i, hlen) == head) {
1108  // An alternatives directive is just starting.
1109  i += 2;
1110  if (i >= len) {
1111  break; // malformed directive, bail out
1112  }
1113  // Record alternatives separator, set number of remaining
1114  // alternatives, reactivate case checking.
1115  altSep = qstrcc[i];
1116  remainingAlts = qnalt;
1117  checkCase = true;
1118  } else if (remainingAlts && c == altSep) {
1119  // Alternative separator found, reduce number of remaining
1120  // alternatives and reactivate case checking.
1121  --remainingAlts;
1122  checkCase = true;
1123  } else if (checkCase && c.isLetter()) {
1124  // Case check is active and the character is a letter; change case.
1125  if (toupper) {
1126  qstrcc[i] = c.toUpper();
1127  } else {
1128  qstrcc[i] = c.toLower();
1129  }
1130  ++numChcased;
1131  // No more case checks until next alternatives separator.
1132  checkCase = false;
1133  }
1134 
1135  // If any letter has been changed, and there are no more alternatives
1136  // to be processed, we're done.
1137  if (numChcased > 0 && remainingAlts == 0) {
1138  break;
1139  }
1140 
1141  // Go to next character.
1142  ++i;
1143  }
1144 
1145  return qstrcc;
1146 }

1354  ++i;

1451  state = s_nextKey;
1452  } else {

1454  .arg(fpath).arg(countLines(s, i));
1455  }

1461  }
1462  }
1463

1465 
1466  if (state != s_nextEntry) {
1467  return SPREF("loadProps: unexpected end of file in %1")

1470

1474 // Read big-endian integer of nbytes length at position pos
1475 // in character array fc of length len.
1476 // Update position to point after the number.
1477 // In case of error, pos is set to -1.
1478 template <typename T>
1479 static int bin_read_int_nbytes(const char *fc, qlonglong len, qlonglong &pos, int nbytes)
1480 {
1481  if (pos + nbytes > len) {
1482  pos = -1;
1483  return 0;
1484  }
1485  T num = qFromBigEndian<T>((uchar *) fc + pos);
1486  pos += nbytes;
1487  return num;
1488 }

1491 static quint64 bin_read_int64(const char *fc, qlonglong len, qlonglong &pos)
1492 {

1494 }

1496 // Read 32-bit big-endian integer.
1497 static quint32 bin_read_int(const char *fc, qlonglong len, qlonglong &pos)
1498 {
1499  return bin_read_int_nbytes<quint32>(fc, len, pos, 4);
1500 }
1501 
1502 // Read string at position pos of character array fc of length n.
1503 // String is represented as 32-bit big-endian byte length followed by bytes.
1504 // Update position to point after the string.
1505 // In case of error, pos is set to -1.
1506 static QByteArray bin_read_string(const char *fc, qlonglong len, qlonglong &pos)
1507 {
1508  // Binary format stores strings as length followed by byte sequence.
1509  // No null-termination.
1510  int nbytes = bin_read_int(fc, len, pos);
1511  if (pos < 0) {
1512  return QByteArray();
1513  }
1514  if (nbytes < 0 || pos + nbytes > len) {
1515  pos = -1;
1516  return QByteArray();
1517  }
1518  QByteArray s(fc + pos, nbytes);
1519  pos += nbytes;
1520  return s;
1521 }
