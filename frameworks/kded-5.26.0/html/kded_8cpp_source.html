158 {
159  QVector<KPluginMetaData> plugins = KPluginLoader::findPlugins(QStringLiteral("kf5/kded"));
160  QSet<QString> moduleIds;
161  foreach (const KPluginMetaData &md, plugins) {
162  moduleIds.insert(md.pluginId());
163  }
164  // also search for old .desktop based kded modules
165  KPluginInfo::List oldStylePlugins = KPluginInfo::fromServices(KServiceTypeTrader::self()->query(QStringLiteral("KDEDModule")));

167  if (moduleIds.contains(info.pluginName())) {
168  qCWarning(KDED).nospace() << "kded module " << info.pluginName() << " has already been found using "
169  "JSON metadata, please don't install the now unneeded .desktop file (" << info.entryPath() << ").";
170  } else {
171  qCDebug(KDED).nospace() << "kded module " << info.pluginName() << " still uses .desktop files ("
172  << info.entryPath() << "). Please port it to JSON metadata.";
173  plugins.append(info.toMetaData());
174  }
175  }
176  return plugins;

180 {
181  KPluginMetaData module(QStringLiteral("kf5/kded/") + id);
182  if (module.isValid()) {
183  return module;

187  if (oldStyleModule) {

192  qCWarning(KDED) << "could not find kded module with id" << id;
193  return KPluginMetaData();

196 void Kded::initModules()
197 {
198  m_dontLoad.clear();
199 #ifdef Q_OS_OSX
200  // it seems there is no reason to honour KDE_FULL_SESSION on OS X even if it's set for some reason.
201  // That way kdeinit5 is always auto-started if required (and the kded is as good a candidate
202  // for starting this service as any other application).
203  bool kde_running = false;
204 #else
205  bool kde_running = !qEnvironmentVariableIsEmpty("KDE_FULL_SESSION");
206 #endif
207  if (kde_running) {
208 #ifndef Q_OS_WIN
209  // not the same user like the one running the session (most likely we're run via sudo or something)
210  const QByteArray sessionUID = qgetenv("KDE_SESSION_UID");
211  if (!sessionUID.isEmpty() && uid_t(sessionUID.toInt()) != getuid()) {
212  kde_running = false;
213  }
214 #endif
215  //TODO: Change 5 to KDED_VERSION_MAJOR the moment KF5 are stable
216  // not the same kde version as the current desktop
217  const QByteArray kdeSession = qgetenv("KDE_SESSION_VERSION");
218  if (kdeSession.toInt() != 5) {
219  kde_running = false;
220  }
221  }
222 
223  // There will be a "phase 2" only if we're in the KDE startup.
224  // If kded is restarted by its crashhandled or by hand,
225  // then there will be no second phase autoload, so load
226  // these modules now, if in a KDE session.
227  const bool loadPhase2Now = (kde_running && qgetenv("KDED_STARTED_BY_KDEINIT").toInt() == 0);
228 
229  // Preload kded modules.
230  const QVector<KPluginMetaData> kdedModules = availableModules();
231  foreach (const KPluginMetaData &module, kdedModules) {
232  // Should the service load on startup?
233  const bool autoload = isModuleAutoloaded(module);
234 
235  // see ksmserver's README for description of the phases
236  bool prevent_autoload = false;
237  switch (phaseForModule(module)) {
238  case 0: // always autoload
239  break;
240  case 1: // autoload only in KDE
241  if (!kde_running) {
242  prevent_autoload = true;
243  }
244  break;
245  case 2: // autoload delayed, only in KDE
246  default:
247  if (!loadPhase2Now) {
248  prevent_autoload = true;
249  }
250  break;
251  }
252 
253  // Load the module if necessary and allowed
254  if (autoload && !prevent_autoload) {
255  if (!loadModule(module, false)) {
256  continue;
257  }
258  }
259 
260  // Remember if the module is allowed to load on demand
261  bool loadOnDemand = isModuleLoadedOnDemand(module);
262  if (!loadOnDemand) {
263  noDemandLoad(module.pluginId());
264  }
265 
266  // In case of reloading the configuration it is possible for a module
267  // to run even if it is now allowed to. Stop it then.
268  if (!loadOnDemand && !autoload) {
269  unloadModule(module.pluginId());
270  }
271  }
272 }

275 {
276  qCDebug(KDED) << "Loading second phase autoload";
277  KSharedConfig::Ptr config = KSharedConfig::openConfig();
278  QVector<KPluginMetaData> kdedModules = availableModules();
279  foreach (const KPluginMetaData &module, kdedModules) {

357 KDEDModule *Kded::loadModule(const KPluginMetaData &module, bool onDemand)
358 {
359  if (!module.isValid() || module.fileName().isEmpty()) {
360  qCWarning(KDED) << "attempted to load an invalid module.";
361  return nullptr;
362  }
363  const QString moduleId = module.pluginId();
364  KDEDModule *oldModule = m_modules.value(moduleId, 0);
365  if (oldModule) {
366  qCDebug(KDED) << "kded module" << moduleId << "is already loaded.";
367  return oldModule;
368  }
369 
370  if (onDemand) {
371  // use toVariant() since it could be string or bool in the json and QJsonObject does not convert
372  QVariant p = module.rawData().value(QStringLiteral("X-KDE-Kded-load-on-demand")).toVariant();
373  if (p.isValid() && (p.toBool() == false)) {
374  noDemandLoad(moduleId);
375  return nullptr;
376  }
377  }
378 
379  KDEDModule *kdedModule = nullptr;
380 
381  KPluginLoader loader(module.fileName());
382  KPluginFactory *factory = loader.factory();
383  if (factory) {
384  kdedModule = factory->create<KDEDModule>(this);
385  } else {
386  // TODO: remove this fallback code, the kded modules should all be fixed instead
387  KPluginLoader loader2(QStringLiteral("kded_") + module.fileName());
388  factory = loader2.factory();
389  if (factory) {
390  qCWarning(KDED).nospace() << "found kded module " << moduleId
391  << " by prepending 'kded_' to the library path, please fix your metadata.";
392  kdedModule = factory->create<KDEDModule>(this);
393  } else {
394  qCWarning(KDED).nospace() << "Could not load kded module " << moduleId << ":"
395  << loader.errorString() << " (library path was:" << module.fileName() << ")";
396  }
397  }
398 
399  if (kdedModule) {
400  kdedModule->setModuleName(moduleId);
401  m_modules.insert(moduleId, kdedModule);
402  //m_libs.insert(moduleId, lib);
403  connect(kdedModule, &KDEDModule::moduleDeleted, this, &Kded::slotKDEDModuleRemoved);
404  qCDebug(KDED) << "Successfully loaded module" << moduleId;
405  return kdedModule;
406  }
407  return nullptr;
408 }

513 void Kded::recreate(bool initial)
514 {
515  // Using KLauncher here is difficult since we might not have a
516  // database
517 
518  if (!initial) {
519  updateDirWatch(); // Update tree first, to be sure to miss nothing.
520  KSycoca::self()->ensureCacheValid();
521  recreateDone();
522  } else {
523  if (!delayedCheck) {
524  updateDirWatch(); // this would search all the directories
525  }
526  if (bCheckSycoca) {
527  KSycoca::self()->ensureCacheValid();
528  }
529  recreateDone();
530  if (delayedCheck) {
531  // do a proper ksycoca check after a delay
532  QTimer::singleShot(60000, this, SLOT(runDelayedCheck()));
533  m_needDelayedCheck = true;
534  delayedCheck = false;
535  } else {
536  m_needDelayedCheck = false;
537  }
538  }
539 }

688 extern "C" Q_DECL_EXPORT int kdemain(int argc, char *argv[])
689 {
690 #ifdef Q_OS_OSX
691  CFBundleRef mainBundle = CFBundleGetMainBundle();
692  if (mainBundle) {
693  // get the application's Info Dictionary. For app bundles this would live in the bundle's Info.plist,
694  // for regular executables it is obtained in another way.
695  CFMutableDictionaryRef infoDict = (CFMutableDictionaryRef) CFBundleGetInfoDictionary(mainBundle);
696  if (infoDict) {
697  // Add or set the "LSUIElement" key with/to value "1". This can simply be a CFString.
698  CFDictionarySetValue(infoDict, CFSTR("LSUIElement"), CFSTR("1"));
699  // That's it. We're now considered as an "agent" by the window server, and thus will have
700  // neither menubar nor presence in the Dock or App Switcher.
701  }
702  }
703 #endif
704  //options.add("check", qi18n("Check Sycoca database only once"));
705 
706  // WABA: Make sure not to enable session management.
707  qunsetenv("SESSION_MANAGER");
708 
709  // In older versions, QApplication creation was postponed until after
710  // testing for --check, in which case, only a QCoreApplication was created.
711  // Since that option is no longer used at startup, we removed that speed
712  // optimization for code clarity and easier support of standard parameters.
713  QApplication app(argc, argv);
714  setupAppInfo(&app);
715  app.setQuitOnLastWindowClosed(false);
716 
717  QCommandLineParser parser;
718  parser.addHelpOption();
719  parser.addVersionOption();
720  parser.addOption(QCommandLineOption(QStringLiteral("check"), QStringLiteral("Check cache validity")));
721  parser.process(app);
722 
723  // Parse command line before checking D-Bus
724  if (parser.isSet(QStringLiteral("check"))) {
725  // KDBusService not wanted here.
726  KSycoca::self()->ensureCacheValid();
727  runKonfUpdate();
728  return 0;
729  }
730 
731  KDBusService service(KDBusService::Unique);
732 
733  QDBusConnectionInterface *bus = QDBusConnection::sessionBus().interface();
734  // Also register as all the names we should respond to (org.kde.kcookiejar, org.kde.khotkeys etc.)
735  // so that the calling code is independent from the physical "location" of the service.
736  const QVector<KPluginMetaData> plugins = KPluginLoader::findPlugins(QStringLiteral("kf5/kded"));
737  foreach (const KPluginMetaData &metaData, plugins) {
738  const QString serviceName = metaData.rawData().value(QStringLiteral("X-KDE-DBus-ServiceName")).toString();
739  if (serviceName.isEmpty()) {
740  qCWarning(KDED) << "No X-KDE-DBus-ServiceName found in" << metaData.fileName();
741  continue;
742  }
743  if (!bus->registerService(serviceName)) {
744  qCWarning(KDED) << "Couldn't register name" << serviceName << "with DBUS - another process owns it already!";
745  }
746  }
747 
748  KSharedConfig::Ptr config = KSharedConfig::openConfig();
749  KConfigGroup cg(config, "General");
750 
751  bCheckSycoca = cg.readEntry("CheckSycoca", true);
752  bCheckUpdates = cg.readEntry("CheckUpdates", true);
753  delayedCheck = cg.readEntry("DelayedCheck", false);
754 
755 #ifndef Q_OS_WIN
756  signal(SIGTERM, sighandler);
757  signal(SIGHUP, sighandler);
758 #endif
759 
760  KCrash::setFlags(KCrash::AutoRestart);
761 
762  Kded *kded = new Kded();
763 
764  kded->recreate(true); // initial
765 
766  if (bCheckUpdates) {
767  (void) new KUpdateD; // Watch for updates
768  }
769 
770 //NOTE: We are going to change how KDE starts and this certanly doesn't fit on the new design.
771 #ifdef Q_OS_LINUX
772  // Tell KSplash that KDED has started
773  QDBusMessage ksplashProgressMessage = QDBusMessage::createMethodCall(QStringLiteral("org.kde.KSplash"),
774  QStringLiteral("/KSplash"),
775  QStringLiteral("org.kde.KSplash"),
776  QStringLiteral("setStage"));
777  ksplashProgressMessage.setArguments(QList<QVariant>() << QStringLiteral("kded"));
778  QDBusConnection::sessionBus().asyncCall(ksplashProgressMessage);
779 #endif
780 
781  runKonfUpdate(); // Run it once.
782 
783 #ifdef Q_OS_LINUX
784  ksplashProgressMessage = QDBusMessage::createMethodCall(QStringLiteral("org.kde.KSplash"),
785  QStringLiteral("/KSplash"),
786  QStringLiteral("org.kde.KSplash"),
787  QStringLiteral("setStage"));
788  ksplashProgressMessage.setArguments(QList<QVariant>() << QStringLiteral("confupdate"));
789  QDBusConnection::sessionBus().asyncCall(ksplashProgressMessage);
790 #endif
791 
792  int result = app.exec(); // keep running
793 
794  delete kded;
795 
796  return result;
797 }
