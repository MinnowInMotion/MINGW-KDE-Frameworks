57 static qreal contrastRatioForLuma(qreal y1, qreal y2)
58 {
59  if (y1 > y2) {
60  return (y1 + 0.05) / (y2 + 0.05);
61  } else {
62  return (y2 + 0.05) / (y1 + 0.05);

64 }

103 QColor KColorUtils::tint(const QColor &base, const QColor &color, qreal amount)
104 {
105  if (amount <= 0.0) {
106  return base;
107  }
108  if (amount >= 1.0) {
109  return color;
110  }
111  if (qIsNaN(amount)) {
112  return base;
113  }
114 
115  qreal baseLuma = luma(base); //cache value because luma call is expensive
116  double ri = contrastRatioForLuma(baseLuma, luma(color));
117  double rg = 1.0 + ((ri + 1.0) * amount * amount * amount);
118  double u = 1.0, l = 0.0;
119  QColor result;
120  for (int i = 12; i; --i) {
121  double a = 0.5 * (l + u);
122  result = tintHelper(base, baseLuma, color, a);
123  double ra = contrastRatioForLuma(baseLuma, luma(result));
124  if (ra > rg) {
125  u = a;
126  } else {
127  l = a;
128  }
129  }
130  return result;
131 }

133 QColor KColorUtils::mix(const QColor &c1, const QColor &c2, qreal bias)
134 {
135  if (bias <= 0.0) {
136  return c1;
137  }
138  if (bias >= 1.0) {
139  return c2;
140  }
141  if (qIsNaN(bias)) {
142  return c1;
143  }
144 
145  qreal r = mixQreal(c1.redF(), c2.redF(), bias);
146  qreal g = mixQreal(c1.greenF(), c2.greenF(), bias);
147  qreal b = mixQreal(c1.blueF(), c2.blueF(), bias);
148  qreal a = mixQreal(c1.alphaF(), c2.alphaF(), bias);
149 
150  return QColor::fromRgbF(r, g, b, a);
151 }

153 QColor KColorUtils::overlayColors(const QColor &base, const QColor &paint,
154  QPainter::CompositionMode comp)
155 {
156  // This isn't the fastest way, but should be "fast enough".
157  // It's also the only safe way to use QPainter::CompositionMode
158  QImage img(1, 1, QImage::Format_ARGB32_Premultiplied);
159  QPainter p(&img);
160  QColor start = base;
161  start.setAlpha(255); // opaque
162  p.fillRect(0, 0, 1, 1, start);
163  p.setCompositionMode(comp);
164  p.fillRect(0, 0, 1, 1, paint);
165  p.end();
166  return img.pixel(0, 0);
167 }
