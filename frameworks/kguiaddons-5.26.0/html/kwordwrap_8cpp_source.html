40 KWordWrap KWordWrap::formatText(QFontMetrics &fm, const QRect &r, int /*flags*/, const QString &str, int len)
41 {
42  KWordWrap kw(r);
43  // The wordwrap algorithm
44  // The variable names and the global shape of the algorithm are inspired
45  // from QTextFormatterBreakWords::format().
46  //qDebug() << "KWordWrap::formatText " << str << " r=" << r.x() << "," << r.y() << " " << r.width() << "x" << r.height();
47  int height = fm.height();
48  if (len == -1) {
49  kw.d->m_text = str;
50  } else {
51  kw.d->m_text = str.left(len);
52  }
53  if (len == -1) {
54  len = str.length();
55  }
56  int lastBreak = -1;
57  int lineWidth = 0;
58  int x = 0;
59  int y = 0;
60  int w = r.width();
61  int textwidth = 0;
62  bool isBreakable = false;
63  bool wasBreakable = false; // value of isBreakable for last char (i-1)
64  bool isParens = false; // true if one of ({[
65  bool wasParens = false; // value of isParens for last char (i-1)
66  QString inputString = str;
67 
68  for (int i = 0; i < len; ++i) {
69  const QChar c = inputString.at(i);
70  const int ww = fm.charWidth(inputString, i);
71 
72  isParens = (c == QLatin1Char('(') || c == QLatin1Char('[')
73  || c == QLatin1Char('{'));
74  // isBreakable is true when we can break _after_ this character.
75  isBreakable = (c.isSpace() || c.isPunct() || c.isSymbol()) & !isParens;
76 
77  // Special case for '(', '[' and '{': we want to break before them
78  if (!isBreakable && i < len - 1) {
79  const QChar nextc = inputString.at(i + 1); // look at next char
80  isBreakable = (nextc == QLatin1Char('(')
81  || nextc == QLatin1Char('[')
82  || nextc == QLatin1Char('{'));
83  }
84  // Special case for '/': after normal chars it's breakable (e.g. inside a path),
85  // but after another breakable char it's not (e.g. "mounted at /foo")
86  // Same thing after a parenthesis (e.g. "dfaure [/fool]")
87  if (c == QLatin1Char('/') && (wasBreakable || wasParens)) {
88  isBreakable = false;
89  }
90 
91  /*qDebug() << "c='" << QString(c) << "' i=" << i << "/" << len
92  << " x=" << x << " ww=" << ww << " w=" << w
93  << " lastBreak=" << lastBreak << " isBreakable=" << isBreakable << endl;*/
94  int breakAt = -1;
95  if (x + ww > w && lastBreak != -1) { // time to break and we know where
96  breakAt = lastBreak;
97  }
98  if (x + ww > w - 4 && lastBreak == -1) { // time to break but found nowhere [-> break here]
99  breakAt = i;
100  }
101  if (i == len - 2 && x + ww + fm.charWidth(inputString, i + 1) > w) { // don't leave the last char alone
102  breakAt = lastBreak == -1 ? i - 1 : lastBreak;
103  }
104  if (c == QLatin1Char('\n')) { // Forced break here
105  if (breakAt == -1 && lastBreak != -1) { // only break if not already breaking
106  breakAt = i - 1;
107  lastBreak = -1;
108  }
109  // remove the line feed from the string
110  kw.d->m_text.remove(i, 1);
111  inputString.remove(i, 1);
112  len--;
113  }
114  if (breakAt != -1) {
115  //qDebug() << "KWordWrap::formatText breaking after " << breakAt;
116  kw.d->m_breakPositions.append(breakAt);
117  int thisLineWidth = lastBreak == -1 ? x + ww : lineWidth;
118  kw.d->m_lineWidths.append(thisLineWidth);
119  textwidth = qMax(textwidth, thisLineWidth);
120  x = 0;
121  y += height;
122  wasBreakable = true;
123  wasParens = false;
124  if (lastBreak != -1) {
125  // Breakable char was found, restart from there
126  i = lastBreak;
127  lastBreak = -1;
128  continue;
129  }
130  } else if (isBreakable) {
131  lastBreak = i;
132  lineWidth = x + ww;
133  }
134  x += ww;
135  wasBreakable = isBreakable;
136  wasParens = isParens;
137  }
138  textwidth = qMax(textwidth, x);
139  kw.d->m_lineWidths.append(x);
140  y += height;
141  //qDebug() << "KWordWrap::formatText boundingRect:" << r.x() << "," << r.y() << " " << textwidth << "x" << y;
142  if (r.height() >= 0 && y > r.height()) {
143  textwidth = r.width();
144  }
145  int realY = y;
146  if (r.height() >= 0) {
147  while (realY > r.height()) {
148  realY -= height;
149  }
150  realY = qMax(realY, 0);
151  }
152  kw.d->m_boundingRect.setRect(0, 0, textwidth, realY);
153  return kw;
154 }

171 QString KWordWrap::wrappedString() const
172 {
173  // We use the calculated break positions to insert '\n' into the string
174  QString ws;
175  int start = 0;
176  for (int i = 0; i < d->m_breakPositions.count(); ++i) {
177  int end = d->m_breakPositions.at(i);
178  ws += d->m_text.midRef(start, end - start + 1);
179  ws += QLatin1Char('\n');
180  start = end + 1;
181  }
182  ws += d->m_text.midRef(start);
183  return ws;
184 }

206 void KWordWrap::drawFadeoutText(QPainter *p, int x, int y, int maxW,
207  const QString &t)
208 {
209  QFontMetrics fm = p->fontMetrics();
210  QColor bgColor = p->background().color();
211  QColor textColor = p->pen().color();
212 
213  if ((fm.boundingRect(t).width() > maxW) && (t.length() > 1)) {
214  int tl = 0;
215  int w = 0;
216  while (tl < t.length()) {
217  w += fm.charWidth(t, tl);
218  if (w >= maxW) {
219  break;
220  }
221  tl++;
222  }
223 
224  int n = qMin(tl, 3);
225  if (t.isRightToLeft()) {
226  x += maxW; // start from the right side for RTL string
227  if (tl > 3) {
228  x -= fm.width(t.left(tl - 3));
229  p->drawText(x, y, t.left(tl - 3));
230  }
231  for (int i = 0; i < n; i++) {
232  p->setPen(mixColors(0.70 - i * 0.25, textColor, bgColor));
233  QString s(t.at(tl - n + i));
234  x -= fm.width(s);
235  p->drawText(x, y, s);
236  }
237  } else {
238  if (tl > 3) {
239  p->drawText(x, y, t.left(tl - 3));
240  x += fm.width(t.left(tl - 3));
241  }
242  for (int i = 0; i < n; i++) {
243  p->setPen(mixColors(0.70 - i * 0.25, textColor, bgColor));
244  QString s(t.at(tl - n + i));
245  p->drawText(x, y, s);
246  x += fm.width(s);
247  }
248  }
249  } else {
250  p->drawText(x, y, t);
251  }
252 }

261 void KWordWrap::drawText(QPainter *painter, int textX, int textY, int flags) const
262 {
263  //qDebug() << "KWordWrap::drawText text=" << wrappedString() << " x=" << textX << " y=" << textY;
264  // We use the calculated break positions to draw the text line by line using QPainter
265  int start = 0;
266  int y = 0;
267  QFontMetrics fm = painter->fontMetrics();
268  int height = fm.height(); // line height
269  int ascent = fm.ascent();
270  int maxwidth = d->m_boundingRect.width();
271  int i;
272  int lwidth = 0;
273  int end = 0;
274  for (i = 0; i < d->m_breakPositions.count(); ++i) {
275  // if this is the last line, leave the loop
276  if ((d->m_constrainingRect.height() >= 0) &&
277  ((y + 2 * height) > d->m_constrainingRect.height())) {
278  break;
279  }
280  end = d->m_breakPositions.at(i);
281  lwidth = d->m_lineWidths.at(i);
282  int x = textX;
283  if (flags & Qt::AlignHCenter) {
284  x += (maxwidth - lwidth) / 2;
285  } else if (flags & Qt::AlignRight) {
286  x += maxwidth - lwidth;
287  }
288  painter->drawText(x, textY + y + ascent, d->m_text.mid(start, end - start + 1));
289  y += height;
290  start = end + 1;
291  }
292 
293  // Draw the last line
294  lwidth = d->m_lineWidths.last();
295  int x = textX;
296  if (flags & Qt::AlignHCenter) {
297  x += (maxwidth - lwidth) / 2;
298  } else if (flags & Qt::AlignRight) {
299  x += maxwidth - lwidth;
300  }
301  if ((d->m_constrainingRect.height() < 0) ||
302  ((y + height) <= d->m_constrainingRect.height())) {
303  if (i == d->m_breakPositions.count()) {
304  painter->drawText(x, textY + y + ascent, d->m_text.mid(start));
305  } else if (flags & FadeOut)
306  drawFadeoutText(painter, textX, textY + y + ascent,
307  d->m_constrainingRect.width(),
308  d->m_text.mid(start));
309  else if (flags & Truncate)
310  drawTruncateText(painter, textX, textY + y + ascent,
311  d->m_constrainingRect.width(),
312  d->m_text.mid(start));
313  else
314  painter->drawText(x, textY + y + ascent,
315  d->m_text.mid(start));
316  }
317 }
