90 void KOverlayIconEngine::paint(QPainter *painter, const QRect &rect, QIcon::Mode mode, QIcon::State state)
91 {
92  // Paint the base icon as the first layer
93  m_base.paint(painter, rect, Qt::AlignCenter, mode, state);
94 
95  if (m_overlays.isEmpty()) {
96  return;
97  }
98 
99  const int width = rect.width();
100  const int height = rect.height();
101  const int iconSize = qMin(width, height);
102  // Determine the overlay icon size
103  int overlaySize;
104  if (iconSize < 32) {
105  overlaySize = 8;
106  } else if (iconSize <= 48) {
107  overlaySize = 16;
108  } else if (iconSize <= 64) {
109  overlaySize = 22;
110  } else if (iconSize <= 96) {
111  overlaySize = 32;
112  } else if (iconSize <= 128) {
113  overlaySize = 48;
114  } else {
115  overlaySize = (int)(iconSize / 4);
116  }
117 
118  // Iterate over stored overlays
119  QHash<Qt::Corner, QIcon>::const_iterator i = m_overlays.constBegin();
120  while (i != m_overlays.constEnd()) {
121 
122  const QPixmap overlayPixmap = i.value().pixmap(overlaySize, overlaySize, mode, state);
123  if (overlayPixmap.isNull()) {
124  ++i;
125  continue;
126  }
127 
128  QPoint startPoint;
129  switch (i.key()) {
130  case Qt::BottomLeftCorner:
131  startPoint = QPoint(2, height - overlaySize - 2);
132  break;
133  case Qt::BottomRightCorner:
134  startPoint = QPoint(width - overlaySize - 2, height - overlaySize - 2);
135  break;
136  case Qt::TopRightCorner:
137  startPoint = QPoint(width - overlaySize - 2, 2);
138  break;
139  case Qt::TopLeftCorner:
140  startPoint = QPoint(2, 2);
141  break;
142  }
143 
144  // Draw the overlay pixmap
145  painter->drawPixmap(startPoint, overlayPixmap);
146 
147  ++i;
148  }
149 }
