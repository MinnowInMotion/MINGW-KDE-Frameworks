48 /*
49  * Get the real modifiers related to a virtual modifier.
50  */
51 unsigned int xkbVirtualModifier(XkbDescPtr xkb, const char *name)
52 {
53  Q_ASSERT(xkb != 0);
54 
55  unsigned int mask = 0;
56  bool nameEqual;
57  for (int i = 0; i < XkbNumVirtualMods; ++i) {
58  char *modStr = XGetAtomName(xkb->dpy, xkb->names->vmods[i]);
59  if (modStr != 0) {
60  nameEqual = (strcmp(name, modStr) == 0);
61  XFree(modStr);
62  if (nameEqual) {
63  XkbVirtualModsToReal(xkb, 1 << i, &mask);
64  break;
65  }
66  }
67  }
68  return mask;
69 }

71 KModifierKeyInfoProvider::KModifierKeyInfoProvider()
72  : QObject(0)
73  , QAbstractNativeEventFilter()
74  , m_xkbEv(0)
75  , m_xkbAvailable(false)
76 {
77  if (qApp) {
78  if (qApp->platformName() == QStringLiteral("xcb")) {
79  int code, xkberr, maj, min;
80  m_xkbAvailable = XkbQueryExtension(QX11Info::display(), &code, &m_xkbEv, &xkberr, &maj, &min);
81  }
82  }
83  if (m_xkbAvailable) {
84  XkbSelectEvents(QX11Info::display(), XkbUseCoreKbd,
85  XkbStateNotifyMask | XkbMapNotifyMask,
86  XkbStateNotifyMask | XkbMapNotifyMask);
87  unsigned long int stateMask = XkbModifierStateMask | XkbModifierBaseMask |
88  XkbModifierLatchMask | XkbModifierLockMask |
89  XkbPointerButtonMask;
90  XkbSelectEventDetails(QX11Info::display(), XkbUseCoreKbd, XkbStateNotifyMask,
91  stateMask, stateMask);
92  }
93 
94  xkbUpdateModifierMapping();
95 
96  // add known pointer buttons
97  m_xkbButtons.insert(Qt::LeftButton, Button1Mask);
98  m_xkbButtons.insert(Qt::MidButton, Button2Mask);
99  m_xkbButtons.insert(Qt::RightButton, Button3Mask);
100  m_xkbButtons.insert(Qt::XButton1, Button4Mask);
101  m_xkbButtons.insert(Qt::XButton2, Button5Mask);
102 
103  // get the initial state
104  if (m_xkbAvailable) {
105  XkbStateRec state;
106  XkbGetState(QX11Info::display(), XkbUseCoreKbd, &state);
107  xkbModifierStateChanged(state.mods, state.latched_mods, state.locked_mods);
108  xkbButtonStateChanged(state.ptr_buttons);
109 
110  QCoreApplication::instance()->installNativeEventFilter(this);
111  }
112 }

211 bool KModifierKeyInfoProvider::nativeEventFilter(const QByteArray &eventType, void *message, long int *result)
212 {
213  Q_UNUSED(result)
214  if (!m_xkbAvailable || eventType != "xcb_generic_event_t") {
215  return false;
216  }
217  xcb_generic_event_t *event = static_cast<xcb_generic_event_t *>(message);
218  if ((event->response_type & ~0x80) == m_xkbEv + XkbEventCode) {
219  _xkb_event *kbevt = reinterpret_cast<_xkb_event *>(event);
220  unsigned int stateMask = XkbModifierStateMask | XkbModifierBaseMask |
221  XkbModifierLatchMask | XkbModifierLockMask;
222  if (kbevt->any.xkbType == XkbMapNotify) {
223  xkbUpdateModifierMapping();
224  } else if (kbevt->any.xkbType == XkbStateNotify) {
225  if (kbevt->state_notify.changed & stateMask) {
226  xkbModifierStateChanged(kbevt->state_notify.mods, kbevt->state_notify.latchedMods,
227  kbevt->state_notify.lockedMods);
228  } else if (kbevt->state_notify.changed & XkbPointerButtonMask) {
229  xkbButtonStateChanged(kbevt->state_notify.ptrBtnState);
230  }
231  }
232  }
233  return false;
234 }

236 void KModifierKeyInfoProvider::xkbModifierStateChanged(unsigned char mods,
237  unsigned char latched_mods,
238  unsigned char locked_mods)
239 {
240  // detect keyboard modifiers
241  ModifierStates oldState;
242  ModifierStates newState;
243 
244  QHash<Qt::Key, unsigned int>::const_iterator it;
245  QHash<Qt::Key, unsigned int>::const_iterator end = m_xkbModifiers.constEnd();
246  for (it = m_xkbModifiers.constBegin(); it != end; ++it) {
247  if (!m_modifierStates.contains(it.key())) {
248  continue;
249  }
250  newState = Nothing;
251  oldState = m_modifierStates[it.key()];
252 
253  // determine the new state
254  if (mods & it.value()) {
255  newState |= Pressed;
256  }
257  if (latched_mods & it.value()) {
258  newState |= Latched;
259  }
260  if (locked_mods & it.value()) {
261  newState |= Locked;
262  }
263 
264  if (newState != oldState) {
265  m_modifierStates[it.key()] = newState;
266 
267  if ((newState ^ oldState) & Pressed) {
268  emit keyPressed(it.key(), newState & Pressed);
269  }
270  if ((newState ^ oldState) & Latched) {
271  emit keyLatched(it.key(), newState & Latched);
272  }
273  if ((newState ^ oldState) & Locked) {
274  emit keyLocked(it.key(), newState & Locked);
275  }
276  }
277  }
278 }

280 void KModifierKeyInfoProvider::xkbButtonStateChanged(unsigned short ptr_buttons)
281 {
282  // detect mouse button states
283  bool newButtonState;
284 
285  QHash<Qt::MouseButton, unsigned short>::const_iterator it;
286  QHash<Qt::MouseButton, unsigned short>::const_iterator end = m_xkbButtons.constEnd();
287  for (it = m_xkbButtons.constBegin(); it != end; ++it) {
288  newButtonState = (ptr_buttons & it.value());

292  }

296 void KModifierKeyInfoProvider::xkbUpdateModifierMapping()
297 {
298  if (!m_xkbAvailable) {
299  return;
300  }
301  m_xkbModifiers.clear();
302 
303  QList<ModifierDefinition> srcModifiers;
304  srcModifiers << ModifierDefinition(Qt::Key_Shift, ShiftMask, 0, 0)
305  << ModifierDefinition(Qt::Key_Control, ControlMask, 0, 0)
306  << ModifierDefinition(Qt::Key_Alt, 0, "Alt", XK_Alt_L)
307  // << { 0, 0, I18N_NOOP("Win"), "superkey", "" }
308  << ModifierDefinition(Qt::Key_Meta, 0, "Meta", XK_Meta_L)
309  << ModifierDefinition(Qt::Key_Super_L, 0, "Super", XK_Super_L)
310  << ModifierDefinition(Qt::Key_Hyper_L, 0, "Hyper", XK_Hyper_L)
311  << ModifierDefinition(Qt::Key_AltGr, 0, "AltGr", 0)
312  << ModifierDefinition(Qt::Key_NumLock, 0, "NumLock", XK_Num_Lock)
313  << ModifierDefinition(Qt::Key_CapsLock, LockMask, 0, 0)
314  << ModifierDefinition(Qt::Key_ScrollLock, 0, "ScrollLock", XK_Scroll_Lock);
315 
316  XkbDescPtr xkb = XkbGetKeyboard(QX11Info::display(), XkbAllComponentsMask, XkbUseCoreKbd);
317 
318  QList<ModifierDefinition>::const_iterator it;
319  QList<ModifierDefinition>::const_iterator end = srcModifiers.constEnd();
320  for (it = srcModifiers.constBegin(); it != end; ++it) {
321  unsigned int mask = it->mask;
322  if (mask == 0 && xkb != 0) {
323  // try virtual modifier first
324  if (it->name != 0) {
325  mask = xkbVirtualModifier(xkb, it->name);
326  }
327  if (mask == 0 && it->keysym != 0) {
328  mask = XkbKeysymToModifiers(QX11Info::display(), it->keysym);
329  } else if (mask == 0) {
330  // special case for AltGr
331  mask = XkbKeysymToModifiers(QX11Info::display(), XK_Mode_switch) |
332  XkbKeysymToModifiers(QX11Info::display(), XK_ISO_Level3_Shift) |
333  XkbKeysymToModifiers(QX11Info::display(), XK_ISO_Level3_Latch) |
334  XkbKeysymToModifiers(QX11Info::display(), XK_ISO_Level3_Lock);
335  }
336  }
337 
338  if (mask != 0) {
339  m_xkbModifiers.insert(it->key, mask);
340  // previously unknown modifier
341  if (!m_modifierStates.contains(it->key)) {
342  m_modifierStates.insert(it->key, Nothing);
343  emit keyAdded(it->key);
344  }
345  }
346  }
347 
348  // remove modifiers which are no longer available
349  QMutableHashIterator<Qt::Key, ModifierStates> i(m_modifierStates);
350  while (i.hasNext()) {
351  i.next();
352  if (!m_xkbModifiers.contains(i.key())) {
353  Qt::Key key = i.key();
354  i.remove();
355  emit keyRemoved(key);
356  }
357  }
358 
359  if (xkb != 0) {
360  XkbFreeKeyboard(xkb, 0, true);
361  }
362 }
