92 void KNotificationRestrictions::Private::screensaverFakeKeyEvent()
93 {
94  qCDebug(LOG_KNOTIFICATIONS);
95 #if HAVE_XTEST
96  if (!isX11) {
97  return;
98  }
99  qCDebug(LOG_KNOTIFICATIONS) << "---- using XTestFakeKeyEvent";
100  Display *display = QX11Info::display();
101  XTestFakeKeyEvent(display, XTestKeyCode, true, CurrentTime);
102  XTestFakeKeyEvent(display, XTestKeyCode, false, CurrentTime);
103  XSync(display, false);
104 #endif // HAVE_XTEST
105 }

107 void KNotificationRestrictions::Private::startScreenSaverPrevention()
108 {
109  qCDebug(LOG_KNOTIFICATIONS);
110 
111  QDBusMessage message = QDBusMessage::createMethodCall(
112  QStringLiteral("org.freedesktop.ScreenSaver"), QStringLiteral("/ScreenSaver"), QStringLiteral("org.freedesktop.ScreenSaver"), QStringLiteral("Inhibit"));
113  message << determineProgramName();
114  message << reason;
115  QDBusReply<uint> reply = QDBusConnection::sessionBus().call(message);
116  if (reply.isValid()) {
117  screenSaverDbusCookie = reply.value();
118  return;
119  }
120 #if HAVE_XTEST
121  if (!isX11) {
122  return;
123  }
124  if (!haveXTest) {
125  int a, b, c, e;
126  haveXTest = XTestQueryExtension(QX11Info::display(), &a, &b, &c, &e);
127 
128  if (!haveXTest) {
129  qCDebug(LOG_KNOTIFICATIONS) << "--- No XTEST!";
130  return;
131  }
132  }
133 
134  if (!XTestKeyCode) {
135  XTestKeyCode = XKeysymToKeycode(QX11Info::display(), XK_Shift_L);
136 
137  if (!XTestKeyCode) {
138  qCDebug(LOG_KNOTIFICATIONS) << "--- No XKeyCode for XK_Shift_L!";
139  return;
140  }
141  }
142 
143  if (!screensaverTimer) {
144  screensaverTimer = new QTimer(q);
145  connect(screensaverTimer, SIGNAL(timeout()),
146  q, SLOT(screensaverFakeKeyEvent()));
147  }
148 
149  qCDebug(LOG_KNOTIFICATIONS) << "---- using XTest";
150  // send a fake event right away in case this got started after a period of
151  // innactivity leading to the screensaver set to activate in <55s
152  screensaverFakeKeyEvent();
153  screensaverTimer->start(55000);
154 #endif // HAVE_XTEST
155 }

157 void KNotificationRestrictions::Private::stopScreenSaverPrevention()
158 {
159 
160  if (screenSaverDbusCookie != -1) {
161  QDBusMessage message = QDBusMessage::createMethodCall(
162  QStringLiteral("org.freedesktop.ScreenSaver"), QStringLiteral("/ScreenSaver"), QStringLiteral("org.freedesktop.ScreenSaver"), QStringLiteral("UnInhibit"));
163  message << static_cast<uint>(screenSaverDbusCookie);
164  screenSaverDbusCookie = -1;
165  if (QDBusConnection::sessionBus().send(message)) {
166  return;
167  }
168  }
169 #if HAVE_XTEST
170  if (!isX11) {
171  return;
172  }
173  delete screensaverTimer;
174  screensaverTimer = 0;
175 #endif // HAVE_XTEST
176 }

178 QString KNotificationRestrictions::Private::determineProgramName()
179 {
180  QString appName = QGuiApplication::applicationDisplayName();
181  if (appName.isEmpty()) {
182  appName = QCoreApplication::applicationName();
183  }
184  if (appName.isEmpty()) {
185  appName = tr("Unknown Application");
186  }
187  return appName;
188 }
