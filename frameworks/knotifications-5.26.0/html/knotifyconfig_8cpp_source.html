33 static KSharedConfig::Ptr retrieve_from_cache(const QString &filename, QStandardPaths::StandardLocation type = QStandardPaths::GenericConfigLocation)
34 {
35  QCache<QString, KSharedConfig::Ptr> &cache = *static_cache;
36  if (cache.contains(filename)) {
37  return *cache[filename];
38  }
39 
40  KSharedConfig::Ptr m = KSharedConfig::openConfig(filename, KConfig::NoGlobals, type);
41  cache.insert(filename, new KSharedConfig::Ptr(m));
42 
43  return m;
44 }
