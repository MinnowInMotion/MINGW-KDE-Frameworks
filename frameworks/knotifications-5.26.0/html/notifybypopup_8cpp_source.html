156 NotifyByPopup::NotifyByPopup(QObject *parent)
157  : KNotificationPlugin(parent),
158  d(new NotifyByPopupPrivate(this))
159 {
160  d->animationTimer = 0;
161  d->dbusServiceExists = false;
162  d->dbusServiceCapCacheDirty = true;
163  d->nextPosition = -1;
164 
165  // check if service already exists on plugin instantiation
166  QDBusConnectionInterface *interface = QDBusConnection::sessionBus().interface();
167  d->dbusServiceExists = interface && interface->isServiceRegistered(dbusServiceName);
168 
169  if (d->dbusServiceExists) {
170  onServiceOwnerChanged(dbusServiceName, QString(), QStringLiteral("_")); //connect signals
171  }
172 
173  // to catch register/unregister events from service in runtime
174  QDBusServiceWatcher *watcher = new QDBusServiceWatcher(this);
175  watcher->setConnection(QDBusConnection::sessionBus());
176  watcher->setWatchMode(QDBusServiceWatcher::WatchForOwnerChange);
177  watcher->addWatchedService(dbusServiceName);
178  connect(watcher, SIGNAL(serviceOwnerChanged(QString,QString,QString)),
179  SLOT(onServiceOwnerChanged(QString,QString,QString)));
180 
181  if (!d->dbusServiceExists) {
182  bool startfdo = false;
183 #ifdef Q_WS_WIN
184  startfdo = true;
185 #else
186  if (qEnvironmentVariableIsEmpty("KDE_FULL_SESSION")) {
187  QDBusMessage message = QDBusMessage::createMethodCall(QStringLiteral("org.freedesktop.DBus"),
188  QStringLiteral("/org/freedesktop/DBus"),
189  QStringLiteral("org.freedesktop.DBus"),
190  QStringLiteral("ListActivatableNames"));
191 
192  // FIXME - this should be async
193  QDBusReply<QStringList> reply = QDBusConnection::sessionBus().call(message);
194  if (reply.isValid() && reply.value().contains(dbusServiceName)) {
195  startfdo = true;
196  // We need to set d->dbusServiceExists to true because dbus might be too slow
197  // starting the service and the first call to NotifyByPopup::notify
198  // might not have had the service up, by setting this to true we
199  // guarantee it will still go through dbus and dbus will do the correct
200  // thing and wait for the service to go up
201  d->dbusServiceExists = true;
202  }
203  }
204 #endif
205  if (startfdo) {
206  QDBusConnection::sessionBus().interface()->startService(dbusServiceName);
207  }
208  }
209 }

226 void NotifyByPopup::notify(KNotification *notification, const KNotifyConfig &notifyConfig)
227 {
228  if (d->passivePopups.contains(notification) || d->galagoNotifications.contains(notification->id())) {
229  // notification is alrady on the screen, do nothing
230  finish(notification);
231  return;
232  }
233 
234  // check if Notifications DBus service exists on bus, use it if it does
235  if (d->dbusServiceExists) {
236  if (d->dbusServiceCapCacheDirty) {
237  // if we don't have the server capabilities yet, we need to query for them first;
238  // as that is an async dbus operation, we enqueue the notification and process them
239  // when we receive dbus reply with the server capabilities
240  d->notificationQueue.append(qMakePair(notification, notifyConfig));
241  d->queryPopupServerCapabilities();
242  } else {
243  if (!d->sendNotificationToGalagoServer(notification, notifyConfig)) {
244  finish(notification); //an error ocurred.
245  }
246  }
247  return;
248  }
249 
250  // Persistent => 0 == infinite timeout
251  // CloseOnTimeout => -1 == let the server decide
252  int timeout = notification->flags() & KNotification::Persistent ? 0 : -1;
253 
254  // if Growl can display our popups, use that instead
255  if (NotifyByPopupGrowl::canPopup()) {
256 
257  QString appCaption, iconName;
258  d->getAppCaptionAndIconName(notifyConfig, &appCaption, &iconName);
259  appCaption = d->stripHtml(appCaption);
260 
261  //did the user override the icon name?
262  if (!notification->iconName().isEmpty()) {
263  iconName = notification->iconName();
264  }
265 
266  // Our growl implementation does not support html stuff
267  // so strip it off right away
268  QString text = notification->text();
269  text = d->stripHtml(text);
270 
271  // The first arg is QPixmap*, however that pixmap is not used
272  // at all (it has Q_UNUSED) so just set it to 0
273  NotifyByPopupGrowl::popup(Q_NULLPTR, timeout, appCaption, text);
274 
275  // Finish immediately, because current NotifyByPopupGrowl can't callback
276  finish(notification);
277  return;
278  }
279 
280  // Check if this object lives in the GUI thread and return if it doesn't
281  // as Qt cannot create/handle widgets in non-GUI threads
282  if (QThread::currentThread() != qApp->thread()) {
283  qCWarning(LOG_KNOTIFICATIONS) << "KNotification did not detect any running org.freedesktop.Notifications server and fallback notifications cannot be used from non-GUI thread!";
284  return;
285  }
286 
287  // last fallback - display the popup using KPassivePopup
288  KPassivePopup *pop = new KPassivePopup(notification->widget());
289  d->passivePopups.insert(notification, pop);
290  d->fillPopup(pop, notification, notifyConfig);
291 
292  QRect screen = QGuiApplication::primaryScreen()->availableGeometry();
293  if (d->nextPosition == -1) {
294  d->nextPosition = screen.top();
295  }
296  pop->setAutoDelete(true);
297  connect(pop, SIGNAL(destroyed()), this, SLOT(onPassivePopupDestroyed()));
298 
299  pop->setTimeout(timeout);
300  pop->adjustSize();
301  pop->show(QPoint(screen.left() + screen.width()/2 - pop->width()/2 , d->nextPosition));
302  d->nextPosition += pop->height();
303 }

305 void NotifyByPopup::onPassivePopupDestroyed()
306 {
307  const QObject *destroyedPopup = sender();
308 
309  if (!destroyedPopup) {
310  return;
311  }
312 
313  QMap<KNotification*, KPassivePopup*>::iterator it;
314  for (it = d->passivePopups.begin(); it != d->passivePopups.end(); ++it) {
315  QObject *popup = it.value();
316  if (popup && popup == destroyedPopup) {
317  finish(it.key());
318  d->passivePopups.remove(it.key());
319  break;
320  }
321  }
322 
323  //relocate popup
324  if (!d->animationTimer) {
325  d->animationTimer = startTimer(10);
326  }

329 void NotifyByPopup::timerEvent(QTimerEvent *event)
330 {
331  if (event->timerId() != d->animationTimer) {
332  return KNotificationPlugin::timerEvent(event);
333  }

335  bool cont = false;
336  QRect screen = QGuiApplication::primaryScreen()->availableGeometry();
337  d->nextPosition = screen.top();
338 
339  Q_FOREACH (KPassivePopup *popup, d->passivePopups)
340  {
341  int y = popup->pos().y();
342  if (y > d->nextPosition) {
343  y = qMax(y - 5, d->nextPosition);
344  d->nextPosition = y + popup->height();
345  cont = cont || y != d->nextPosition;
346  popup->move(popup->pos().x(), y);
347  } else {
348  d->nextPosition += popup->height();
349  }
350  }
351 
352  if (!cont) {
353  killTimer(d->animationTimer);
354  d->animationTimer = 0;
355  }

417 void NotifyByPopup::onServiceOwnerChanged(const QString &serviceName, const QString &oldOwner, const QString &newOwner)
418 {
419  Q_UNUSED(serviceName);
420  // close all notifications we currently hold reference to
421  Q_FOREACH (KNotification *n, d->galagoNotifications.values()) {
422  if (n) {
423  finished(n);
424  }
425  }
426  Q_FOREACH (KNotification *n, d->passivePopups.keys()) {
427  finished(n);
428  }
429  d->galagoNotifications.clear();
430  d->passivePopups.clear();
431 
432  d->dbusServiceCapCacheDirty = true;
433  d->popupServerCapabilities.clear();
434  d->notificationQueue.clear();
435 
436  if (newOwner.isEmpty()) {
437  d->dbusServiceExists = false;
438  } else if (oldOwner.isEmpty()) {
439  d->dbusServiceExists = true;
440 
441  // connect to action invocation signals
442  bool connected = QDBusConnection::sessionBus().connect(QString(), // from any service
443  dbusPath,
444  dbusInterfaceName,
445  QStringLiteral("ActionInvoked"),
446  this,
447  SLOT(onGalagoNotificationActionInvoked(uint,QString)));
448  if (!connected) {
449  qCWarning(LOG_KNOTIFICATIONS) << "warning: failed to connect to ActionInvoked dbus signal";
450  }
451 
452  connected = QDBusConnection::sessionBus().connect(QString(), // from any service
453  dbusPath,
454  dbusInterfaceName,
455  QStringLiteral("NotificationClosed"),
456  this,
457  SLOT(onGalagoNotificationClosed(uint,uint)));
458  if (!connected) {
459  qCWarning(LOG_KNOTIFICATIONS) << "warning: failed to connect to NotificationClosed dbus signal";
460  }
461  }

479 void NotifyByPopup::onGalagoNotificationClosed(uint dbus_id, uint reason)
480 {
481  auto iter = d->galagoNotifications.find(dbus_id);
482  if (iter == d->galagoNotifications.end()) {
483  return;
484  }
485  KNotification *n = *iter;
486  d->galagoNotifications.remove(dbus_id);
487 
488  if (n) {
489  finished(n);
490  // The popup bubble is the only user facing part of a notification,
491  // if the user closes the popup, it means he wants to get rid
492  // of the notification completely, including playing sound etc
493  // Therefore we close the KNotification completely after closing
494  // the popup, but only if the reason is 2, which means "user closed"
495  if (reason == 2) {
496  n->close();
497  }
498  }
499 }

612 bool NotifyByPopupPrivate::sendNotificationToGalagoServer(KNotification *notification, const KNotifyConfig &notifyConfig_nocheck, bool update)
613 {
614  uint updateId = galagoNotifications.key(notification, 0);
615 
616  if (update) {
617  if (updateId == 0) {
618  // we have nothing to update; the notification we're trying to update
619  // has been already closed
620  return false;
621  }
622  }
623 
624  QDBusMessage dbusNotificationMessage = QDBusMessage::createMethodCall(dbusServiceName, dbusPath, dbusInterfaceName, QStringLiteral("Notify"));
625 
626  QList<QVariant> args;
627 
628  QString appCaption;
629  QString iconName;
630  getAppCaptionAndIconName(notifyConfig_nocheck, &appCaption, &iconName);
631 
632  //did the user override the icon name?
633  if (!notification->iconName().isEmpty()) {
634  iconName = notification->iconName();
635  }
636 
637  args.append(appCaption); // app_name
638  args.append(updateId); // notification to update
639  args.append(iconName); // app_icon
640 
641  QString title = notification->title().isEmpty() ? appCaption : notification->title();
642  QString text = notification->text();
643 
644  if (!popupServerCapabilities.contains(QStringLiteral("body-markup"))) {
645  if (title.startsWith(QLatin1String("<html>"))) {
646  title = stripHtml(title);
647  }
648  if (text.startsWith(QLatin1String("<html>"))) {
649  text = stripHtml(text);
650  }
651  }
652 
653  args.append(title); // summary
654  args.append(text); // body
655 
656  // galago spec defines action list to be list like
657  // (act_id1, action1, act_id2, action2, ...)
658  //
659  // assign id's to actions like it's done in fillPopup() method
660  // (i.e. starting from 1)
661  QStringList actionList;
662  if (popupServerCapabilities.contains(QStringLiteral("actions"))) {
663  int actId = 0;
664  Q_FOREACH (const QString &actionName, notification->actions()) {
665  actId++;
666  actionList.append(QString::number(actId));
667  actionList.append(actionName);
668  }
669  }
670 
671  args.append(actionList); // actions
672 
673  QVariantMap hintsMap;
674  // Add the application name to the hints.
675  // According to fdo spec, the app_name is supposed to be the applicaton's "pretty name"
676  // but in some places it's handy to know the application name itself
677  if (!notification->appName().isEmpty()) {
678  hintsMap[QStringLiteral("x-kde-appname")] = notification->appName();
679  }
680 
681  if (!notification->eventId().isEmpty()) {
682  hintsMap[QStringLiteral("x-kde-eventId")] = notification->eventId();
683  }
684 
685  if (notification->flags() & KNotification::SkipGrouping) {
686  hintsMap[QStringLiteral("x-kde-skipGrouping")] = 1;
687  }
688 
689  //FIXME - reenable/fix
690  // let's see if we've got an image, and store the image in the hints map
691  if (!notification->pixmap().isNull()) {
692  QByteArray pixmapData;
693  QBuffer buffer(&pixmapData);
694  buffer.open(QIODevice::WriteOnly);
695  notification->pixmap().save(&buffer, "PNG");
696  buffer.close();
697  hintsMap[QStringLiteral("image_data")] = ImageConverter::variantForImage(QImage::fromData(pixmapData));
698  }
699 
700  args.append(hintsMap); // hints
701 
702  // Persistent => 0 == infinite timeout
703  // CloseOnTimeout => -1 == let the server decide
704  int timeout = notification->flags() & KNotification::Persistent ? 0 : -1;
705 
706  args.append(timeout); // expire timout
707 
708  dbusNotificationMessage.setArguments(args);
709 
710  QDBusPendingCall notificationCall = QDBusConnection::sessionBus().asyncCall(dbusNotificationMessage, -1);
711 
712  //parent is set to the notification so that no-one ever accesses a dangling pointer on the notificationObject property
713  QDBusPendingCallWatcher *watcher = new QDBusPendingCallWatcher(notificationCall, notification);
714  watcher->setProperty("notificationObject", QVariant::fromValue<KNotification*>(notification));
715 
716  QObject::connect(watcher, SIGNAL(finished(QDBusPendingCallWatcher*)),
717  q, SLOT(onGalagoServerReply(QDBusPendingCallWatcher*)));
718 
719  return true;
720 }

771 QString NotifyByPopupPrivate::stripHtml(const QString &text)
772 {
773  QXmlStreamReader r("<elem>" + text + "</elem>");
774  HtmlEntityResolver resolver;
775  r.setEntityResolver(&resolver);
776  QString result;
777  while (!r.atEnd()) {
778  r.readNext();
779  if (r.tokenType() == QXmlStreamReader::Characters) {
780  result.append(r.text());
781  } else if (r.tokenType() == QXmlStreamReader::StartElement && r.name() == "br") {
782  result.append("\n");
783  }
784  }
785 
786  if (r.hasError()) {
787  // XML error in the given text, just return the original string
788  qCWarning(LOG_KNOTIFICATIONS) << "Notification to send to backend which does "
789  "not support HTML, contains invalid XML:"
790  << r.errorString() << "line" << r.lineNumber()
791  << "col" << r.columnNumber();
792  return text;
793  }
794 
795  return result;
796 }
