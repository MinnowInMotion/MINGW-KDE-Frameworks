50 void NotifyByAudio::notify(KNotification *notification, KNotifyConfig *config)
51 {
52  if (!m_audioOutput) {
53  m_audioOutput = new Phonon::AudioOutput(Phonon::NotificationCategory, this);
54  }
55  QString soundFilename = config->readEntry(QStringLiteral("Sound"));
56  if (soundFilename.isEmpty()) {
57  qCWarning(LOG_KNOTIFICATIONS) << "Audio notification requested, but no sound file provided in notifyrc file, aborting audio notification";
58 
59  finish(notification);
60  return;
61  }
62 
63  QUrl soundURL;
64  const auto dataLocations = QStandardPaths::standardLocations(QStandardPaths::GenericDataLocation);
65  foreach (const QString &dataLocation, dataLocations) {
66  soundURL = QUrl::fromUserInput(soundFilename,
67  dataLocation + "/sounds",
68  QUrl::AssumeLocalFile);
69  if (soundURL.isLocalFile() && QFile::exists(soundURL.toLocalFile())) {
70  break;
71  } else if (!soundURL.isLocalFile() && soundURL.isValid()) {
72  break;
73  }
74  soundURL.clear();
75  }
76  if (soundURL.isEmpty()) {
77  qCWarning(LOG_KNOTIFICATIONS) << "Audio notification requested, but sound file from notifyrc file was not found, aborting audio notification";
78  finish(notification);
79  return;
80  }
81 
82  Phonon::MediaObject *m;
83 
84  if (m_reusablePhonons.isEmpty()) {
85  m = new Phonon::MediaObject(this);
86  connect(m, SIGNAL(finished()), SLOT(onAudioFinished()));
87  connect(m, SIGNAL(stateChanged(Phonon::State,Phonon::State)), SLOT(stateChanged(Phonon::State,Phonon::State)));
88  Phonon::createPath(m, m_audioOutput);
89  } else {
90  m = m_reusablePhonons.takeFirst();
91  }
92 
93  m->setCurrentSource(soundURL);
94  m->play();
95 
96  if (notification->flags() & KNotification::LoopSound) {
97  // Enqueing essentially prevents the subsystem pipeline from partial teardown
98  // which is the most desired thing in terms of load and delay between loop cycles.
99  // All of this is timing dependent, which is why we want at least one source queued;
100  // in reality the shorter the source the more sources we want to be queued to prevent
101  // the MO from running out of sources.
102  // Point being that all phonon signals are forcefully queued (becuase qthread has problems detecting !pthread threads),
103  // so when you get for example the aboutToFinish signal the MO might already have stopped playing.
104  //
105  // And so we queue it three times at least; doesn't cost anything and keeps us safe.
106 
107  m->enqueue(soundURL);
108  m->enqueue(soundURL);
109  m->enqueue(soundURL);
110 
111  connect(m, SIGNAL(currentSourceChanged(Phonon::MediaSource)), SLOT(onAudioSourceChanged(Phonon::MediaSource)));
112  }
113 
114  m_notifications.insert(m, notification);
115 }

122 void NotifyByAudio::close(KNotification *notification)
123 {
124  Phonon::MediaObject *m = m_notifications.key(notification);
125 
126  if (!m) {
127  return;
128  }
129 
130  // this should call onAudioFinished() which also does finish() on the notification
131  m->stop();
132  m_reusablePhonons.append(m);

135 void NotifyByAudio::onAudioFinished()
136 {
137  Phonon::MediaObject *m = qobject_cast<Phonon::MediaObject*>(sender());
138 
139  if (!m) {
140  return;
141  }
142 
143  if (KNotification *notification = m_notifications.value(m, nullptr)) {
144  //if the sound is short enough, we can't guarantee new sounds are
145  //enqueued before finished is emitted.
146  //so to make sure we are looping restart it when the sound finished
147  if (notification->flags() & KNotification::LoopSound) {
148  m->play();
149  return;
150  }
151 
152  finish(notification);
153  }
154 
155  disconnect(m, SIGNAL(currentSourceChanged(Phonon::MediaSource)), this, SLOT(onAudioSourceChanged(Phonon::MediaSource)));
156 
157  m_notifications.remove(m);
158  m_reusablePhonons.append(m);
159 }
