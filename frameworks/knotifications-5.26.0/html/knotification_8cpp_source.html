246 void KNotification::activate(unsigned int action)
247 {
248  switch (action) {
249  case 0:
250  emit activated();
251  break;
252  case 1:
253  emit action1Activated();
254  break;
255  case 2:
256  emit action2Activated();
257  break;
258  case 3:
259  emit action3Activated();
260  break;
261  }
262 
263  // emitting activated() makes the Manager close all the active plugins
264  // which will deref() the KNotification object, which will result
265  // in closing the notification
266  emit activated(action);
267 }

430 bool KNotification::eventFilter(QObject *watched, QEvent *event)
431 {
432  if (watched == d->widget) {
433  if (event->type() == QEvent::WindowActivate) {
434  if (d->flags & CloseWhenWidgetActivated) {
435  QTimer::singleShot(500, this, SLOT(close()));
436  }
437  }
438  }
439 
440  return false;
441 }

443 QString KNotification::standardEventToEventId(KNotification::StandardEvent event)
444 {
445  QString eventId;
446  switch (event) {
447  case Warning:
448  eventId = QStringLiteral("warning");
449  break;
450  case Error:
451  eventId = QStringLiteral("fatalerror");
452  break;
453  case Catastrophe:
454  eventId = QStringLiteral("catastrophe");
455  break;
456  case Notification: // fall through
457  default:
458  eventId = QStringLiteral("notification");
459  break;
460  }
461  return eventId;
462 }
