552 bool KStatusNotifierItemPrivate::checkVisibility(QPoint pos, bool perform)
553 {
554 #ifdef Q_OS_WIN
555 #if 0
556  // the problem is that we lose focus when the systray icon is activated
557  // and we don't know the former active window
558  // therefore we watch for activation event and use our stopwatch :)
559  if (GetTickCount() - dwTickCount < 300) {
560  // we were active in the last 300ms -> hide it
561  minimizeRestore(false);
562  emit activateRequested(false, pos);
563  } else {
564  minimizeRestore(true);
565  emit activateRequested(true, pos);
566  }
567 #endif
568 #else
569  KWindowInfo info1 = KWindowSystem::windowInfo(associatedWidget->winId(), NET::XAWMState | NET::WMState | NET::WMDesktop);
570  // mapped = visible (but possibly obscured)
571  bool mapped = (info1.mappingState() == NET::Visible) && !info1.isMinimized();
572 
573 // - not mapped -> show, raise, focus
574 // - mapped
575 // - obscured -> raise, focus
576 // - not obscured -> hide
577  //info1.mappingState() != NET::Visible -> window on another desktop?
578  if (!mapped) {
579  if (perform) {
580  minimizeRestore(true);
581  emit q->activateRequested(true, pos);
582  }
583 
584  return true;
585  } else {
586  QListIterator< WId > it(KWindowSystem::stackingOrder());
587  it.toBack();
588  while (it.hasPrevious()) {
589  WId id = it.previous();
590  if (id == associatedWidget->winId()) {
591  break;
592  }
593 
594  KWindowInfo info2 = KWindowSystem::windowInfo(id,
595  NET::WMDesktop | NET::WMGeometry | NET::XAWMState | NET::WMState | NET::WMWindowType);
596 
597  if (info2.mappingState() != NET::Visible) {
598  continue; // not visible on current desktop -> ignore
599  }
600 
601  if (!info2.geometry().intersects(associatedWidget->geometry())) {
602  continue; // not obscuring the window -> ignore
603  }
604 
605  if (!info1.hasState(NET::KeepAbove) && info2.hasState(NET::KeepAbove)) {
606  continue; // obscured by window kept above -> ignore
607  }
608 
609  NET::WindowType type = info2.windowType(NET::NormalMask | NET::DesktopMask
610  | NET::DockMask | NET::ToolbarMask | NET::MenuMask | NET::DialogMask
611  | NET::OverrideMask | NET::TopMenuMask | NET::UtilityMask | NET::SplashMask);
612 
613  if (type == NET::Dock || type == NET::TopMenu) {
614  continue; // obscured by dock or topmenu -> ignore
615  }
616 
617  if (perform) {
618  KWindowSystem::raiseWindow(associatedWidget->winId());
619  KWindowSystem::forceActiveWindow(associatedWidget->winId());
620  emit q->activateRequested(true, pos);
621  }
622 
623  return true;
624  }
625 
626  //not on current desktop?
627  if (!info1.isOnCurrentDesktop()) {
628  if (perform) {
629  KWindowSystem::activateWindow(associatedWidget->winId());
630  emit q->activateRequested(true, pos);
631  }
632 
633  return true;
634  }
635 
636  if (perform) {
637  minimizeRestore(false); // hide
638  emit q->activateRequested(false, pos);
639  }
640 
641  return false;
642  }
643 #endif
644 
645  return true;
646 }

681 void KStatusNotifierItemPrivate::init(const QString &extraId)
682 {
683  qDBusRegisterMetaType<KDbusImageStruct>();
684  qDBusRegisterMetaType<KDbusImageVector>();
685  qDBusRegisterMetaType<KDbusToolTipStruct>();
686 
687  statusNotifierItemDBus = new KStatusNotifierItemDBus(q);
688  q->setAssociatedWidget(qobject_cast<QWidget *>(q->parent()));
689 
690  QDBusServiceWatcher *watcher = new QDBusServiceWatcher(s_statusNotifierWatcherServiceName,
691  QDBusConnection::sessionBus(),
692  QDBusServiceWatcher::WatchForOwnerChange,
693  q);
694  QObject::connect(watcher, SIGNAL(serviceOwnerChanged(QString,QString,QString)),
695  q, SLOT(serviceChange(QString,QString,QString)));
696 
697  //create a default menu, just like in KSystemtrayIcon
698  QMenu *m = new QMenu(associatedWidget);
699 
700  title = QGuiApplication::applicationDisplayName();
701  if (title.isEmpty()) {
702  title = QCoreApplication::applicationName();
703  }
704  titleAction = m->addSection(qApp->windowIcon(), title);
705  m->setTitle(title);
706  q->setContextMenu(m);
707 
708  QAction *action = new QAction(q);
709  action->setText(KStatusNotifierItem::tr("Quit"));

725 void KStatusNotifierItemPrivate::registerToDaemon()
726 {
727  qCDebug(LOG_KNOTIFICATIONS) << "Registering a client interface to the KStatusNotifierWatcher";
728  if (!statusNotifierWatcher) {
729  statusNotifierWatcher = new org::kde::StatusNotifierWatcher(s_statusNotifierWatcherServiceName, QStringLiteral("/StatusNotifierWatcher"),
730  QDBusConnection::sessionBus());
731  QObject::connect(statusNotifierWatcher, SIGNAL(StatusNotifierHostRegistered()),
732  q, SLOT(checkForRegisteredHosts()));
733  QObject::connect(statusNotifierWatcher, SIGNAL(StatusNotifierHostUnregistered()),
734  q, SLOT(checkForRegisteredHosts()));
735  }
736 
737  if (statusNotifierWatcher->isValid()) {
738  // get protocol version in async way
739  QDBusMessage msg = QDBusMessage::createMethodCall(s_statusNotifierWatcherServiceName,
740  QStringLiteral("/StatusNotifierWatcher"),
741  QStringLiteral("org.freedesktop.DBus.Properties"),
742  QStringLiteral("Get"));
743  msg.setArguments(QVariantList{QStringLiteral("org.kde.StatusNotifierWatcher"), QStringLiteral("ProtocolVersion")});
744  QDBusPendingCall async = QDBusConnection::sessionBus().asyncCall(msg);
745  QDBusPendingCallWatcher *watcher = new QDBusPendingCallWatcher(async, q);
746  QObject::connect(watcher, &QDBusPendingCallWatcher::finished, q,
747  [this, watcher] {
748  watcher->deleteLater();
749  QDBusPendingReply<QVariant> reply = *watcher;
750  if (reply.isError()) {
751  qCDebug(LOG_KNOTIFICATIONS) << "Failed to read protocol version of KStatusNotifierWatcher";
752  setLegacySystemTrayEnabled(true);
753  } else {
754  bool ok = false;
755  const int protocolVersion = reply.value().toInt(&ok);
756  if (ok && protocolVersion == s_protocolVersion) {
757  statusNotifierWatcher->RegisterStatusNotifierItem(statusNotifierItemDBus->service());
758  setLegacySystemTrayEnabled(false);
759  } else {
760  qCDebug(LOG_KNOTIFICATIONS) << "KStatusNotifierWatcher has incorrect protocol version";
761  setLegacySystemTrayEnabled(true);
762  }
763  }
764  }
765  );
766  } else {
767  qCDebug(LOG_KNOTIFICATIONS) << "KStatusNotifierWatcher not reachable";
768  setLegacySystemTrayEnabled(true);
769  }
770 }

819 {
820  if (enabled == (systemTrayIcon != 0)) {
821  // already in the correct state
822  return;
823  }
824 
825 
826  if (enabled) {
827  if (!QSystemTrayIcon::isSystemTrayAvailable()) {
828  return;

831  if (!systemTrayIcon && !isKde) {
832  systemTrayIcon = new KStatusNotifierLegacyIcon(associatedWidget);
833  syncLegacySystemTrayIcon();
834  systemTrayIcon->setToolTip(toolTipTitle);
835  systemTrayIcon->show();

838  } else if (isKde) {

840  // but SNI is not available; see bug 350785

847  menu->setWindowFlags(Qt::Popup);
848  }
849  } else {
850  delete systemTrayIcon;
851  systemTrayIcon = 0;
852 
853  if (menu) {
854  menu->setWindowFlags(Qt::Window);
855  }
856  }
857 
858  if (menu) {
859  QMenu *m = menu;
860  menu = 0;
861  q->setContextMenu(m);

969 KDbusImageStruct KStatusNotifierItemPrivate::imageToStruct(const QImage &image)
970 {
971  KDbusImageStruct icon;
972  icon.width = image.size().width();
973  icon.height = image.size().height();
974  if (image.format() == QImage::Format_ARGB32) {
975  icon.data = QByteArray((char *)image.bits(), image.byteCount());
976  } else {
977  QImage image32 = image.convertToFormat(QImage::Format_ARGB32);
978  icon.data = QByteArray((char *)image32.bits(), image32.byteCount());
979  }
980 
981  //swap to network byte order if we are little endian
982  if (QSysInfo::ByteOrder == QSysInfo::LittleEndian) {
983  quint32 *uintBuf = (quint32 *) icon.data.data();
984  for (uint i = 0; i < icon.data.size() / sizeof(quint32); ++i) {
985  *uintBuf = qToBigEndian(*uintBuf);
986  ++uintBuf;
987  }
988  }
989 
990  return icon;
991 }
