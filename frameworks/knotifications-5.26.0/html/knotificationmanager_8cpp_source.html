146 void KNotificationManager::notificationClosed()
147 {
148  KNotification *notification = qobject_cast<KNotification*>(sender());
149  if (!notification) {
150  return;
151  }
152  // We cannot do d->notifications.find(notification->id()); here because the
153  // notification->id() is -1 or -2 at this point, so we need to look for value
154  for (auto iter = d->notifications.begin(); iter != d->notifications.end(); ++iter) {
155  if (iter.value() == notification) {
156  d->notifications.erase(iter);
157  break;
158  }
159  }
160 }

162 void KNotificationManager::close(int id, bool force)
163 {
164  if (force || d->notifications.contains(id)) {
165  KNotification *n = d->notifications.value(id);
166  qCDebug(LOG_KNOTIFICATIONS) << "Closing notification" << id;
167 
168  // Find plugins that are actually acting on this notification
169  // call close() only on those, otherwise each KNotificationPlugin::close()
170  // will call finish() which may close-and-delete the KNotification object
171  // before it finishes calling close on all the other plugins.
172  // For example: Action=Popup is a single actions but there is 5 loaded
173  // plugins, calling close() on the second would already close-and-delete
174  // the notification
175  KNotifyConfig notifyConfig(n->appName(), n->contexts(), n->eventId());
176  QString notifyActions = notifyConfig.readEntry(QStringLiteral("Action"));
177 
178  Q_FOREACH (const QString &action, notifyActions.split('|')) {
179  if (!d->notifyPlugins.contains(action)) {
180  qCDebug(LOG_KNOTIFICATIONS) << "No plugin for action" << action;
181  continue;
182  }
183 
184  d->notifyPlugins[action]->close(n);
185  }
186  }
187 }

189 int KNotificationManager::notify(KNotification *n)
190 {
191  KNotifyConfig notifyConfig(n->appName(), n->contexts(), n->eventId());
192 
193  if (d->dirtyConfigCache.contains(n->appName())) {
194  notifyConfig.reparseSingleConfiguration(n->appName());
195  d->dirtyConfigCache.removeOne(n->appName());
196  }
197 
198  QString notifyActions = notifyConfig.readEntry(QStringLiteral("Action"));
199 
200  if (notifyActions.isEmpty() || notifyActions == QLatin1String("None")) {
201  // this will cause KNotification closing itself fast
202  n->deref();
203  return -1;
204  }
205 
206  d->notifications.insert(d->notifyIdCounter, n);
207 
208  Q_FOREACH (const QString &action, notifyActions.split('|')) {
209  if (!d->notifyPlugins.contains(action)) {
210  qCDebug(LOG_KNOTIFICATIONS) << "No plugin for action" << action;
211  continue;
212  }
213 
214  KNotificationPlugin *notifyPlugin = d->notifyPlugins[action];
215  n->ref();
216  qCDebug(LOG_KNOTIFICATIONS) << "Calling notify on" << notifyPlugin->optionName();
217  notifyPlugin->notify(n, &notifyConfig);
218  }
219 
220  connect(n, &KNotification::closed, this, &KNotificationManager::notificationClosed);
221  return d->notifyIdCounter++;
222 }
