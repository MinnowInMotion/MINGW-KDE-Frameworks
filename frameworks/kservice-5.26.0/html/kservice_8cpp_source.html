58 void KServicePrivate::init(const KDesktopFile *config, KService *q)
59 {
60  const QString entryPath = q->entryPath();
61  bool absPath = !QDir::isRelativePath(entryPath);
62 
63  // TODO: it makes sense to have a KConstConfigGroup I guess
64  const KConfigGroup desktopGroup = const_cast<KDesktopFile *>(config)->desktopGroup();
65  QMap<QString, QString> entryMap = desktopGroup.entryMap();
66 
67  entryMap.remove(QStringLiteral("Encoding")); // reserved as part of Desktop Entry Standard
68  entryMap.remove(QStringLiteral("Version")); // reserved as part of Desktop Entry Standard
69 
70  q->setDeleted(desktopGroup.readEntry("Hidden", false));
71  entryMap.remove(QStringLiteral("Hidden"));
72  if (q->isDeleted()) {
73  m_bValid = false;
74  return;
75  }
76 
77  m_strName = config->readName();
78  entryMap.remove(QStringLiteral("Name"));
79  if (m_strName.isEmpty()) {
80  // Try to make up a name.
81  m_strName = entryPath;
82  int i = m_strName.lastIndexOf(QLatin1Char('/'));
83  m_strName = m_strName.mid(i + 1);
84  i = m_strName.lastIndexOf(QLatin1Char('.'));
85  if (i != -1) {
86  m_strName.truncate(i);
87  }
88  }
89 
90  m_strType = config->readType();
91  entryMap.remove(QStringLiteral("Type"));
92  if (m_strType.isEmpty()) {
93  /*kWarning(servicesDebugArea()) << "The desktop entry file " << entryPath
94  << " has no Type=... entry."
95  << " It should be \"Application\" or \"Service\"";
96  m_bValid = false;
97  return;*/
98  m_strType = QStringLiteral("Application");
99  } else if (m_strType != QLatin1String("Application") && m_strType != QLatin1String("Service")) {
100  qWarning() << "The desktop entry file " << entryPath
101  << " has Type=" << m_strType
102  << " instead of \"Application\" or \"Service\"";
103  m_bValid = false;
104  return;
105  }
106 
107  // NOT readPathEntry, it is not XDG-compliant: it performs
108  // various expansions, like $HOME. Note that the expansion
109  // behaviour still happens if the "e" flag is set, maintaining
110  // backwards compatibility.
111  m_strExec = desktopGroup.readEntry("Exec", QString());
112  entryMap.remove(QStringLiteral("Exec"));
113 
114  if (m_strType == QLatin1String("Application")) {
115  // It's an application? Should have an Exec line then, otherwise we can't run it
116  if (m_strExec.isEmpty()) {
117  qWarning() << "The desktop entry file" << entryPath << "has Type=" << m_strType << "but no Exec line";
118  m_bValid = false;
119  return;
120  }
121  }
122 
123  // In case Try Exec is set, check if the application is available
124  if (!config->tryExec()) {
125  q->setDeleted(true);
126  m_bValid = false;
127  return;
128  }
129 
130  const QStandardPaths::StandardLocation resource = config->resource();
131 
132  if ((m_strType == QLatin1String("Application")) &&
133  (resource != QStandardPaths::ApplicationsLocation) &&
134  !absPath) {
135  qWarning() << "The desktop entry file" << entryPath
136  << "has Type=" << m_strType << "but is located under \"" << QStandardPaths::displayName(resource)
137  << "\" instead of \"Applications\"";
138  m_bValid = false;
139  return;
140  }
141 
142  if ((m_strType == QLatin1String("Service")) &&
143  (resource != QStandardPaths::GenericDataLocation) &&
144  !absPath) {
145  qWarning() << "The desktop entry file" << entryPath
146  << "has Type=" << m_strType << "but is located under \"" << QStandardPaths::displayName(resource)
147  << "\" instead of \"Shared Data\"/kservices5";
148  m_bValid = false;
149  return;
150  }
151 
152  // entryPath To desktopEntryName
153  // (e.g. "/home/x/.qttest/share/kservices5/fakepart2.desktop" --> "fakepart2")
154  QString _name = KServiceUtilPrivate::completeBaseName(entryPath);
155 
156  m_strIcon = config->readIcon();
157  entryMap.remove(QStringLiteral("Icon"));
158  m_bTerminal = desktopGroup.readEntry("Terminal", false); // should be a property IMHO
159  entryMap.remove(QStringLiteral("Terminal"));
160  m_strTerminalOptions = desktopGroup.readEntry("TerminalOptions"); // should be a property IMHO
161  entryMap.remove(QStringLiteral("TerminalOptions"));
162  m_strPath = config->readPath();
163  entryMap.remove(QStringLiteral("Path"));
164  m_strComment = config->readComment();
165  entryMap.remove(QStringLiteral("Comment"));
166  m_strGenName = config->readGenericName();
167  entryMap.remove(QStringLiteral("GenericName"));
168  QString _untranslatedGenericName = desktopGroup.readEntryUntranslated("GenericName");
169  if (!_untranslatedGenericName.isEmpty()) {
170  entryMap.insert(QStringLiteral("UntranslatedGenericName"), _untranslatedGenericName);
171  }
172 
173  m_lstFormFactors = desktopGroup.readEntry("X-KDE-FormFactors", QStringList());
174  entryMap.remove(QStringLiteral("X-KDE-FormFactors"));
175 
176  m_lstKeywords = desktopGroup.readXdgListEntry("Keywords", QStringList());
177  entryMap.remove(QStringLiteral("Keywords"));
178  m_lstKeywords += desktopGroup.readEntry("X-KDE-Keywords", QStringList());
179  entryMap.remove(QStringLiteral("X-KDE-Keywords"));
180  categories = desktopGroup.readXdgListEntry("Categories");
181  entryMap.remove(QStringLiteral("Categories"));
182  // TODO KDE5: only care for X-KDE-Library in Type=Service desktop files
183  // This will prevent people defining a part and an app in the same desktop file
184  // which makes user-preference handling difficult.
185  m_strLibrary = desktopGroup.readEntry("X-KDE-Library");
186  entryMap.remove(QStringLiteral("X-KDE-Library"));
187  if (!m_strLibrary.isEmpty() && m_strType == QLatin1String("Application")) {
188  qWarning() << "The desktop entry file" << entryPath << "has Type=" << m_strType
189  << "but also has a X-KDE-Library key. This works for now,"
190  " but makes user-preference handling difficult, so support for this might"
191  " be removed at some point. Consider splitting it into two desktop files.";
192  }
193 
194  QStringList lstServiceTypes = desktopGroup.readEntry("ServiceTypes", QStringList());
195  entryMap.remove(QStringLiteral("ServiceTypes"));
196  lstServiceTypes += desktopGroup.readEntry("X-KDE-ServiceTypes", QStringList());
197  entryMap.remove(QStringLiteral("X-KDE-ServiceTypes"));
198  lstServiceTypes += desktopGroup.readXdgListEntry("MimeType");
199  entryMap.remove(QStringLiteral("MimeType"));
200 
201  if (m_strType == QLatin1String("Application") && !lstServiceTypes.contains(QLatin1String("Application")))
202  // Applications implement the service type "Application" ;-)
203  {
204  lstServiceTypes += QStringLiteral("Application");
205  }
206 
207  m_initialPreference = desktopGroup.readEntry("InitialPreference", 1);
208  entryMap.remove(QStringLiteral("InitialPreference"));
209 
210  // Assign the "initial preference" to each mimetype/servicetype
211  // (and to set such preferences in memory from kbuildsycoca)
212  m_serviceTypes.reserve(lstServiceTypes.size());
213  QListIterator<QString> st_it(lstServiceTypes);
214  while (st_it.hasNext()) {
215  const QString st = st_it.next();
216  if (st.isEmpty()) {
217  qWarning() << "The desktop entry file" << entryPath << "has an empty mimetype!";
218  continue;
219  }
220  int initialPreference = m_initialPreference;
221  if (st_it.hasNext()) {
222  // TODO better syntax - separate group with mimetype=number entries?
223  bool isNumber;
224  const int val = st_it.peekNext().toInt(&isNumber);
225  if (isNumber) {
226  initialPreference = val;
227  st_it.next();
228  }
229  }
230  m_serviceTypes.push_back(KService::ServiceTypeAndPreference(initialPreference, st));
231  }
232 
233  if (entryMap.contains(QLatin1String("Actions"))) {
234  parseActions(config, q);
235  }
236 
237  QString dbusStartupType = desktopGroup.readEntry("X-DBUS-StartupType").toLower();
238  entryMap.remove(QStringLiteral("X-DBUS-StartupType"));
239  if (dbusStartupType == QLatin1String("unique")) {
240  m_DBUSStartusType = KService::DBusUnique;
241  } else if (dbusStartupType == QLatin1String("multi")) {
242  m_DBUSStartusType = KService::DBusMulti;
243  } else {
244  m_DBUSStartusType = KService::DBusNone;
245  }
246 
247  m_strDesktopEntryName = _name.toLower();
248 
249  m_bAllowAsDefault = desktopGroup.readEntry("AllowDefault", true);
250  entryMap.remove(QStringLiteral("AllowDefault"));
251 
252  // allow plugin users to translate categories without needing a separate key
253  QMap<QString, QString>::Iterator entry = entryMap.find(QStringLiteral("X-KDE-PluginInfo-Category"));
254  if (entry != entryMap.end()) {
255  const QString &key = entry.key();
256  m_mapProps.insert(key, QVariant(desktopGroup.readEntryUntranslated(key)));
257  m_mapProps.insert(key + QStringLiteral("-Translated"), QVariant(*entry));
258  entryMap.erase(entry);
259  }
260 
261  // Store all additional entries in the property map.
262  // A QMap<QString,QString> would be easier for this but we can't
263  // break BC, so we have to store it in m_mapProps.
264 // qDebug("Path = %s", entryPath.toLatin1().constData());
265  QMap<QString, QString>::ConstIterator it = entryMap.constBegin();
266  for (; it != entryMap.constEnd(); ++it) {
267  const QString key = it.key();
268  // do not store other translations like Name[fr]; kbuildsycoca will rerun if we change languages anyway
269  if (!key.contains(QLatin1Char('['))) {
270  //qDebug() << " Key =" << key << " Data =" << *it;
271  m_mapProps.insert(key, QVariant(*it));
272  }
273  }
274 }

276 void KServicePrivate::parseActions(const KDesktopFile *config, KService *q)
277 {
278  const QStringList keys = config->readActions();
279  if (keys.isEmpty()) {
280  return;
281  }
282 
283  QStringList::ConstIterator it = keys.begin();
284  const QStringList::ConstIterator end = keys.end();
285  for (; it != end; ++it) {
286  const QString group = *it;
287  if (group == QLatin1String("_SEPARATOR_")) {
288  m_actions.append(KServiceAction(group, QString(), QString(), QString(), false));
289  continue;
290  }
291 
292  if (config->hasActionGroup(group)) {
293  const KConfigGroup cg = config->actionGroup(group);
294  if (!cg.hasKey("Name") || !cg.hasKey("Exec")) {
295  qWarning() << "The action" << group << "in the desktop file" << q->entryPath() << "has no Name or no Exec key";
296  } else {
297  m_actions.append(KServiceAction(group,
298  cg.readEntry("Name"),
299  cg.readEntry("Icon"),
300  cg.readEntry("Exec"),
301  cg.readEntry("NoDisplay", false)));
302  }
303  } else {
304  qWarning() << "The desktop file" << q->entryPath() << "references the action" << group << "but doesn't define it";
305  }
306  }
307 }

401 bool KService::hasServiceType(const QString &serviceType) const
402 {
403  Q_D(const KService);
404 
405  if (!d->m_bValid) {
406  return false; // (useless) safety test
407  }
408  const KServiceType::Ptr ptr = KServiceType::serviceType(serviceType);
409  if (!ptr) {
410  return false;
411  }
412  const int serviceOffset = offset();
413  // doesn't seem to work:
414  //if ( serviceOffset == 0 )
415  // serviceOffset = serviceByStorageId( storageId() );
416  if (serviceOffset) {
417  KSycoca::self()->ensureCacheValid();
418  return KSycocaPrivate::self()->serviceFactory()->hasOffer(ptr->offset(), ptr->serviceOffersOffset(), serviceOffset);
419  }
420 
421  // fall-back code for services that are NOT from ksycoca
422  // For each service type we are associated with, if it doesn't
423  // match then we try its parent service types.
424  QVector<ServiceTypeAndPreference>::ConstIterator it = d->m_serviceTypes.begin();
425  for (; it != d->m_serviceTypes.end(); ++it) {
426  const QString &st = (*it).serviceType;
427  //qDebug() << " has " << (*it);
428  if (st == ptr->name()) {
429  return true;
430  }
431  // also the case of parent servicetypes
432  KServiceType::Ptr p = KServiceType::serviceType(st);
433  if (p && p->inherits(ptr->name())) {
434  return true;
435  }
436  }
437  return false;
438 }

440 bool KService::hasMimeType(const QString &mimeType) const
441 {
442  Q_D(const KService);
443  QMimeDatabase db;
444  const QString mime = db.mimeTypeForName(mimeType).name();
445  if (mime.isEmpty()) {
446  return false;
447  }
448  int serviceOffset = offset();
449  if (serviceOffset) {
450  KSycoca::self()->ensureCacheValid();
451  KMimeTypeFactory *factory = KSycocaPrivate::self()->mimeTypeFactory();
452  const int mimeOffset = factory->entryOffset(mime);
453  const int serviceOffersOffset = factory->serviceOffersOffset(mime);
454  if (serviceOffersOffset == -1) {
455  return false;
456  }
457  return KSycocaPrivate::self()->serviceFactory()->hasOffer(mimeOffset, serviceOffersOffset, serviceOffset);
458  }
459 
460  // fall-back code for services that are NOT from ksycoca
461  QVector<ServiceTypeAndPreference>::ConstIterator it = d->m_serviceTypes.begin();
462  for (; it != d->m_serviceTypes.end(); ++it) {
463  const QString &st = (*it).serviceType;
464  //qDebug() << " has " << (*it);
465  if (st == mime) {
466  return true;
467  }
468  // TODO: should we handle inherited mimetypes here?
469  // KMimeType was in kio when this code was written, this is the only reason it's not done.
470  // But this should matter only in a very rare case, since most code gets KServices from ksycoca.
471  // Warning, change hasServiceType if you implement this here (and check kbuildservicefactory).
472  }
473  return false;
474 }

497 QVariant KServicePrivate::property(const QString &_name, QVariant::Type t) const
498 {
499  if (_name == QLatin1String("Type")) {
500  return QVariant(m_strType); // can't be null
501  } else if (_name == QLatin1String("Name")) {
502  return QVariant(m_strName); // can't be null
503  } else if (_name == QLatin1String("Exec")) {
504  return makeStringVariant(m_strExec);
505  } else if (_name == QLatin1String("Icon")) {
506  return makeStringVariant(m_strIcon);
507  } else if (_name == QLatin1String("Terminal")) {
508  return QVariant(m_bTerminal);
509  } else if (_name == QLatin1String("TerminalOptions")) {
510  return makeStringVariant(m_strTerminalOptions);
511  } else if (_name == QLatin1String("Path")) {
512  return makeStringVariant(m_strPath);
513  } else if (_name == QLatin1String("Comment")) {
514  return makeStringVariant(m_strComment);
515  } else if (_name == QLatin1String("GenericName")) {
516  return makeStringVariant(m_strGenName);
517  } else if (_name == QLatin1String("ServiceTypes")) {
518  return QVariant(serviceTypes());
519  } else if (_name == QLatin1String("AllowAsDefault")) {
520  return QVariant(m_bAllowAsDefault);
521  } else if (_name == QLatin1String("InitialPreference")) {
522  return QVariant(m_initialPreference);
523  } else if (_name == QLatin1String("Library")) {
524  return makeStringVariant(m_strLibrary);
525  } else if (_name == QLatin1String("DesktopEntryPath")) { // can't be null
526  return QVariant(path);
527  } else if (_name == QLatin1String("DesktopEntryName")) {
528  return QVariant(m_strDesktopEntryName); // can't be null
529  } else if (_name == QLatin1String("Categories")) {
530  return QVariant(categories);
531  } else if (_name == QLatin1String("Keywords")) {
532  return QVariant(m_lstKeywords);
533  } else if (_name == QLatin1String("FormFactors")) {
534  return QVariant(m_lstFormFactors);
535  }
536 
537  // Ok we need to convert the property from a QString to its real type.
538  // Maybe the caller helped us.
539  if (t == QVariant::Invalid) {
540  // No luck, let's ask KServiceTypeFactory what the type of this property
541  // is supposed to be.
542  // ######### this looks in all servicetypes, not just the ones this service supports!
543  KSycoca::self()->ensureCacheValid();
544  t = KSycocaPrivate::self()->serviceTypeFactory()->findPropertyTypeByName(_name);
545  if (t == QVariant::Invalid) {
546  qDebug() << "Request for unknown property" << _name;
547  return QVariant(); // Unknown property: Invalid variant.
548  }
549  }
550 
551  QMap<QString, QVariant>::ConstIterator it = m_mapProps.find(_name);
552  if ((it == m_mapProps.end()) || (!it->isValid())) {
553  //qDebug() << "Property not found " << _name;
554  return QVariant(); // No property set.
555  }
556 
557  if (t == QVariant::String) {
558  return *it; // no conversion necessary
559  } else {
560  // All others
561  // For instance properties defined as StringList, like MimeTypes.
562  // XXX This API is accessible only through a friend declaration.
563  return KConfigGroup::convertToQVariant(_name.toUtf8().constData(), it->toString().toUtf8(), t);
564  }
565 }

567 QStringList KServicePrivate::propertyNames() const
568 {
569  QStringList res;
570 
571  QMap<QString, QVariant>::ConstIterator it = m_mapProps.begin();

649 bool KService::showInCurrentDesktop() const
650 {
651  Q_D(const KService);
652 
653  QStringList currentDesktops(QString::fromLatin1(qgetenv("XDG_CURRENT_DESKTOP")));
654  if (currentDesktops.isEmpty()) {
655  // This could be an old display manager, or e.g. a failsafe session with no desktop name
656  // In doubt, let's say we show KDE stuff.
657  currentDesktops << QStringLiteral("KDE");
658  }
659 
660  // This algorithm is described in the desktop entry spec
661 
662  QMap<QString, QVariant>::ConstIterator it = d->m_mapProps.find(QStringLiteral("OnlyShowIn"));
663  if ((it != d->m_mapProps.end()) && (it->isValid())) {
664  const QStringList aList = it->toString().split(QLatin1Char(';'));
665  foreach (const QString &desktop, currentDesktops) {
666  if (aList.contains(desktop)) {
667  return true;
668  }
669  }
670  return false;
671  }
672 
673  it = d->m_mapProps.find(QStringLiteral("NotShowIn"));
674  if ((it != d->m_mapProps.end()) && (it->isValid())) {
675  const QStringList aList = it->toString().split(QLatin1Char(';'));
676  foreach (const QString &desktop, currentDesktops) {
677  if (aList.contains(desktop)) {
678  return false;
679  }
680  }
681  }
682  return true;
683 }

685 bool KService::showOnCurrentPlatform() const
686 {
687  Q_D(const KService);
688  const QString platform = QCoreApplication::instance()->property("platformName").toString();
689  if (platform.isEmpty()) {
690  return true;
691  }
692 
693  auto it = d->m_mapProps.find(QStringLiteral("X-KDE-OnlyShowOnQtPlatforms"));
694  if ((it != d->m_mapProps.end()) && (it->isValid())) {
695  const QStringList aList = it->toString().split(QLatin1Char(';'));
696  if (!aList.contains(platform)) {
697  return false;
698  }
699  }
700 
701  it = d->m_mapProps.find(QStringLiteral("X-KDE-NotShowOnQtPlatforms"));
702  if ((it != d->m_mapProps.end()) && (it->isValid())) {
703  const QStringList aList = it->toString().split(QLatin1Char(';'));
704  if (aList.contains(platform)) {
705  return false;
706  }
707  }
708  return true;

818 QString KService::newServicePath(bool showInMenu, const QString &suggestedName,
819  QString *menuId, const QStringList *reservedMenuIds)
820 {
821  Q_UNUSED(showInMenu); // TODO KDE5: remove argument
822 
823  QString base = suggestedName;
824  QString result;
825  for (int i = 1; true; i++) {
826  if (i == 1) {
827  result = base + QStringLiteral(".desktop");
828  } else {
829  result = base + QStringLiteral("-%1.desktop").arg(i);
830  }
831 
832  if (reservedMenuIds && reservedMenuIds->contains(result)) {
833  continue;
834  }
835 
836  // Lookup service by menu-id
837  KService::Ptr s = serviceByMenuId(result);
838  if (s) {
839  continue;
840  }
841 
842  if (!QStandardPaths::locate(QStandardPaths::GenericDataLocation, QLatin1String("applications/") + result).isEmpty()) {
843  continue;
844  }
845 
846  break;
847  }
848  if (menuId) {
849  *menuId = result;
850  }
851 
852  return QStandardPaths::writableLocation(QStandardPaths::GenericDataLocation) + QLatin1String("/applications/") + result;
853 }
