117 //This comes from KConfigGroupPrivate::deserializeList()
118 QStringList KPluginInfoPrivate::deserializeList(const QString &data)
119 {
120  if (data.isEmpty()) {
121  return QStringList();
122  }
123  if (data == QLatin1String("\\0")) {
124  return QStringList(QString());
125  }
126  QStringList value;
127  QString val;
128  val.reserve(data.size());
129  bool quoted = false;
130  for (int p = 0; p < data.length(); p++) {
131  if (quoted) {
132  val += data[p];
133  quoted = false;
134  } else if (data[p].unicode() == '\\') {
135  quoted = true;
136  } else if (data[p].unicode() == ',' || data[p].unicode() == ';') {
137  val.squeeze(); // release any unused memory
138  value.append(val);
139  val.clear();
140  val.reserve(data.size() - p);
141  } else {
142  val += data[p];
143  }
144  }
145  value.append(val);
146  return value;
147 }

149 // maps the KService, QVariant and KDesktopFile keys to the new KPluginMetaData keys
150 template<typename T, typename Func>
151 static QJsonObject mapToJsonKPluginKey(const QString &name, const QString &description,
152  const QStringList &dependencies, const QStringList &serviceTypes,
153  const QStringList &formFactors, const T &data, Func accessor)
154 {
155  /* Metadata structure is as follows:
156  "KPlugin": {
157  "Name": "Date and Time",
158  "Description": "Date and time by timezone",
159  "Icon": "preferences-system-time",
160  "Authors": { "Name": "Aaron Seigo", "Email": "aseigo@kde.org" },
161  "Category": "Date and Time",
162  "Dependencies": [],
163  "EnabledByDefault": "true",
164  "License": "LGPL",
165  "Id": "time",
166  "Version": "1.0",
167  "Website": "http://plasma.kde.org/",
168  "ServiceTypes": ["Plasma/DataEngine"]
169  "FormFactors": ["tablet", "handset"]
170  }
171  */
172  QJsonObject kplugin;
173  kplugin[s_nameKey()] = name;
174  kplugin[s_jsonDescriptionKey()] = description;
175  kplugin[s_iconKey()] = accessor(data, s_iconKey());
176  QJsonObject authors;
177  authors[s_nameKey()] = accessor(data, s_authorKey());
178  authors[s_jsonEmailKey()] = accessor(data, s_emailKey());
179  kplugin[s_jsonAuthorsKey()] = authors;
180  kplugin[s_jsonCategoryKey()] = accessor(data, s_categoryKey());
181  QJsonValue enabledByDefault = accessor(data, s_enabledbyDefaultKey());
182  // make sure that enabledByDefault is bool and not string
183  if (!enabledByDefault.isBool()) {
184  enabledByDefault = enabledByDefault.toString().compare(QLatin1String("true"), Qt::CaseInsensitive) == 0;
185  }
186  kplugin[s_jsonEnabledByDefaultKey()] = enabledByDefault;
187  kplugin[s_jsonLicenseKey()] = accessor(data, s_licenseKey());
188  kplugin[s_jsonIdKey()] = accessor(data, s_pluginNameKey());
189  kplugin[s_jsonVersionKey()] = accessor(data, s_versionKey());
190  kplugin[s_jsonWebsiteKey()] = accessor(data, s_websiteKey());
191  kplugin[s_jsonFormFactorsKey()] = QJsonArray::fromStringList(formFactors);
192  kplugin[s_serviceTypesKey()] = QJsonArray::fromStringList(serviceTypes);
193  kplugin[s_jsonDependenciesKey()] = QJsonArray::fromStringList(dependencies);
194  QJsonValue mimeTypes = accessor(data, s_mimeTypeKey());
195  if (mimeTypes.isString()) {
196  QStringList mimeList = KPluginInfoPrivate::deserializeList(mimeTypes.toString());
197  if (!mimeList.isEmpty()) {
198  mimeTypes = QJsonArray::fromStringList(mimeList);
199  } else {
200  mimeTypes = QJsonValue();
201  }
202  }
203  kplugin[s_jsonMimeTypesKey()] = mimeTypes;
204  return kplugin;
205 }

207 // TODO: KF6 remove
208 static KPluginMetaData fromCompatibilityJson(const QJsonObject &json, const QString &lib,
209  const QString &metaDataFile, bool warnOnOldStyle)
210 {
211  // This is not added to KPluginMetaData(QJsonObject, QString) to ensure that all the compatility code
212  // remains in kservice and does not increase the size of kcoreaddons
213  QStringList serviceTypes = KPluginMetaData::readStringList(json, s_xKDEServiceTypes());
214  if (serviceTypes.isEmpty()) {
215  serviceTypes = KPluginMetaData::readStringList(json, s_serviceTypesKey());
216  }
217  QJsonObject obj = json;
218  QString name = KPluginMetaData::readTranslatedString(json, s_nameKey());
219  if (warnOnOldStyle) {
220  qWarning("Constructing a KPluginInfo object from old style JSON. Please use"
221  " kcoreaddons_desktop_to_json() for \"%s\" instead of kservice_desktop_to_json()"
222  " in your CMake code.",
223  qPrintable(lib));
224  }
225  QString description = KPluginMetaData::readTranslatedString(json, s_commentKey());
226  QStringList formfactors = KPluginMetaData::readStringList(json, s_jsonFormFactorsKey());
227  QJsonObject kplugin = mapToJsonKPluginKey(name, description,
228  KPluginMetaData::readStringList(json, s_dependenciesKey()), serviceTypes, formfactors, json,
229  [](const QJsonObject &o, const QString &key) { return o.value(key); });
230  obj.insert(s_jsonKPluginKey(), kplugin);
231  return KPluginMetaData(obj, lib, metaDataFile);
232 }

277 KPluginInfo::KPluginInfo(const QVariantList &args, const QString &libraryPath)
278  : d(new KPluginInfoPrivate)
279 {
280  const QString metaData = QStringLiteral("MetaData");
281  foreach (const QVariant &v, args) {
282  if (v.canConvert<QVariantMap>()) {
283  const QVariantMap &m = v.toMap();
284  const QVariant &_metadata = m.value(metaData);
285  if (_metadata.canConvert<QVariantMap>()) {
286  const QVariantMap &map = _metadata.toMap();
287  if (map.value(s_hiddenKey()).toBool()) {
288  d->hidden = true;
289  break;
290  }
291  d->setMetaData(KPluginMetaData(QJsonObject::fromVariantMap(map), libraryPath), true);
292  break;
293  }
294  }
295  }
296  if (!d->metaData.isValid()) {
297  d.reset();
298  }
299 }

302 KPluginInfo::KPluginInfo(const KService::Ptr service)
303  : d(new KPluginInfoPrivate)
304 {
305  if (!service) {
306  d = 0; // isValid() == false
307  return;
308  }
309  d->service = service;
310  if (service->isDeleted()) {
311  d->hidden = true;
312  return;
313  }
314 
315  KSycoca::self()->ensureCacheValid();
316 
317  QJsonObject json;
318  foreach (const QString &key, service->propertyNames()) {
319  QVariant::Type t = KSycocaPrivate::self()->serviceTypeFactory()->findPropertyTypeByName(key);
320  if (t == QVariant::Invalid) {
321  t = QVariant::String; // default to string if the type is not known
322  }
323  QVariant v = service->property(key, t);
324  if (v.isValid()) {
325  json[key] = QJsonValue::fromVariant(v);
326  }
327  }
328  // reintroduce the separation between MimeType= and X-KDE-ServiceTypes=
329  // we could do this by modifying KService and KSyCoCa, but as this is only compatibility
330  // code we just query QMimeDatabase whether a ServiceType is a valid MIME type.
331  // TODO: should we also make sure invalid MimeType= entries end up in KPluginMetaData::mimeTypes()?
332  const QStringList services = service->serviceTypes();
333  if (!services.isEmpty()) {
334  QMimeDatabase db;
335  QStringList mimeTypes;
336  mimeTypes.reserve(services.size());
337  QStringList newServiceTypes;
338  newServiceTypes.reserve(services.size());
339  foreach (const QString& s, services) {
340  if (db.mimeTypeForName(s).isValid()) {
341  mimeTypes << s;
342  } else {
343  newServiceTypes << s;
344  }
345  }
346  json[s_mimeTypeKey()] = QJsonArray::fromStringList(mimeTypes);
347  json[s_xKDEServiceTypes()] = QJsonArray::fromStringList(newServiceTypes);
348  json[s_serviceTypesKey()] = QJsonValue();
349  }
350 
351  d->setMetaData(KPluginMetaData(json, service->library(), service->entryPath()), false);
352  if (!d->metaData.isValid()) {
353  d.reset();
354  }
355 }
356 #endif

615 QVariant KPluginInfo::property(const QString &key) const
616 {
617  KPLUGININFO_ISVALID_ASSERTION;
618  if (d->service) {
619  return d->service->property(key);
620  }
621  QVariant result = d->metaData.rawData().value(key).toVariant();
622  if (result.isValid()) {
623  KSycoca::self()->ensureCacheValid();
624  const QVariant::Type t = KSycocaPrivate::self()->serviceTypeFactory()->findPropertyTypeByName(key);
625 
626  //special case if we want a stringlist: split values by ',' or ';' and construct the list
627  if (t == QVariant::StringList) {
628  if (result.canConvert<QString>()) {
629  result = KPluginInfoPrivate::deserializeList(result.toString());
630  } else if (result.canConvert<QVariantList>()) {
631  QVariantList list = result.toList();
632  QStringList newResult;
633  foreach (const QVariant &value, list) {
634  newResult += value.toString();
635  }
636  result = newResult;
637  } else
638  qWarning() << "Cannot interpret" << result << "into a string list";
639  }
640  return result;
641  }

644  // These warnings should only happen if JSON was generated with kcoreaddons_desktop_to_json
