132 bool KSycocaPrivate::tryMmap()
133 {
134 #if HAVE_MMAP
135  Q_ASSERT(!m_databasePath.isEmpty());
136  m_mmapFile = new QFile(m_databasePath);
137  const bool canRead = m_mmapFile->open(QIODevice::ReadOnly);
138  Q_ASSERT(canRead);
139  if (!canRead) {
140  return false;
141  }
142  fcntl(m_mmapFile->handle(), F_SETFD, FD_CLOEXEC);
143  sycoca_size = m_mmapFile->size();
144  void *mmapRet = mmap(0, sycoca_size,
145  PROT_READ, MAP_SHARED,
146  m_mmapFile->handle(), 0);
147  /* POSIX mandates only MAP_FAILED, but we are paranoid so check for
148  null pointer too. */
149  if (mmapRet == MAP_FAILED || mmapRet == 0) {
150  qCDebug(SYCOCA).nospace() << "mmap failed. (length = " << sycoca_size << ")";
151  sycoca_mmap = 0;
152  return false;
153  } else {
154  sycoca_mmap = static_cast<const char *>(mmapRet);
155 #if HAVE_MADVISE
156  (void) posix_madvise(mmapRet, sycoca_size, POSIX_MADV_WILLNEED);
157 #endif // HAVE_MADVISE
158  return true;
159  }
160 #else
161  return false;
162 #endif // HAVE_MMAP
163 }

329 void KSycocaPrivate::slotDatabaseChanged()
330 {
331  // We don't have information anymore on what resources changed, so emit them all
332  changeList = QStringList() << QStringLiteral("services") << QStringLiteral("servicetypes") << QStringLiteral("xdgdata-mime") << QStringLiteral("apps");
333 
334  qCDebug(SYCOCA) << QThread::currentThread() << "got a notifyDatabaseChanged signal";
335  // kbuildsycoca tells us the database file changed
336  // We would have found out in the next call to ensureCacheValid(), but for
337  // now keep the call to closeDatabase, to help refcounting to 0 the old mmaped file earlier.
338  closeDatabase();
339  // Start monitoring the new file right away
340  m_databasePath = findDatabase();
341 
342  // Now notify applications
343  emit q->databaseChanged();
344  emit q->databaseChanged(changeList);

497 // If it returns true, we have a valid database and the stream has rewinded to the beginning
498 // and past the version number.
499 bool KSycocaPrivate::checkDatabase(BehaviorsIfNotFound ifNotFound)
500 {
501  if (databaseStatus == DatabaseOK) {
502  if (checkVersion()) { // we know the version is ok, but we must rewind the stream anyway
503  return true;
504  }
505  }
506 
507  closeDatabase(); // close the dummy one
508 
509  // Check if new database already available
510  if (openDatabase(ifNotFound & IfNotFoundOpenDummy)) {
511  if (checkVersion()) {
512  // Database exists, and version is ok, we can read it.
513 
514  if (qAppName() != KBUILDSYCOCA_EXENAME && ifNotFound != IfNotFoundDoNothing) {
515 
516  // Ensure it's uptodate, rebuild if needed
517  checkDirectories();
518 
519  // Don't check again for some time
520  m_lastCheck.start();
521  }
522 
523  return true;
524  }
525  }
526 
527  if (ifNotFound & IfNotFoundRecreate) {
528  return buildSycoca();
529  }
530 
531  return false;
532 }

567 KSycocaHeader KSycocaPrivate::readSycocaHeader()
568 {
569  KSycocaHeader header;
570  // do not try to launch kbuildsycoca from here; this code is also called by kbuildsycoca.
571  if (!checkDatabase(KSycocaPrivate::IfNotFoundDoNothing)) {
572  return header;
573  }
574  QDataStream *str = stream();
575  qint64 oldPos = str->device()->pos();
576 
577  Q_ASSERT(str);
578  qint32 aId;
579  qint32 aOffset;
580  // skip factories offsets
581  while (true) {
582  *str >> aId;
583  if (aId) {
584  *str >> aOffset;
585  } else {
586  break; // just read 0
587  }
588  }
589  // We now point to the header
590  QStringList directoryList;
591  *str >> header >> directoryList;
592  allResourceDirs.clear();
593  for (int i = 0; i < directoryList.count(); ++i) {
594  qint64 mtime;
595  *str >> mtime;
596  allResourceDirs.insert(directoryList.at(i), mtime);
597  }
598 
599  str->device()->seek(oldPos);
600 
601  timeStamp = header.timeStamp;
602 
603  // for the useless public accessors. KF6: remove these two lines, the accessors and the vars.
604  language = header.language;
605  updateSig = header.updateSignature;
606 
607  if (m_globalDatabase) {
608  // The global database doesn't point to the user's local dirs, but we need to check them too
609  // to react on something being created there
610  const QString dataHome = QStandardPaths::writableLocation(QStandardPaths::GenericDataLocation);
611  addLocalResourceDir(dataHome + QLatin1String("/kservices5"));
612  addLocalResourceDir(dataHome + QLatin1String("/kservicetypes5"));
613  addLocalResourceDir(dataHome + QLatin1String("/mime"));
614  addLocalResourceDir(dataHome + QLatin1String("/applications"));
615  }
616 
617  return header;
618 }

621 class TimestampChecker
622 {
623 public:
624  TimestampChecker()
625  : m_now(QDateTime::currentDateTime())
626  {}
627 
628  // Check times of last modification of all directories on which ksycoca depends,
629  // If none of them is newer than the mtime we stored for that directory at the
630  // last rebuild, this means that there's no need to rebuild ksycoca.
631  bool checkTimestamps(const QMap<QString, qint64> &dirs)
632  {
633  Q_ASSERT(!dirs.isEmpty());
634  //qCDebug(SYCOCA) << "checking file timestamps";
635  for (auto it = dirs.begin(); it != dirs.end(); ++it) {
636  const QString dir = it.key();
637  const qint64 lastStamp = it.value();
638 
639  auto visitor = [&] (const QFileInfo &fi) {
640  const QDateTime mtime = fi.lastModified();
641  if (mtime.toMSecsSinceEpoch() > lastStamp) {
642  if (mtime > m_now) {
643  qCDebug(SYCOCA) << fi.filePath() << "has a modification time in the future" << mtime;
644  }
645  qCDebug(SYCOCA) << "timestamp changed:" << fi.filePath() << mtime << ">" << QDateTime::fromMSecsSinceEpoch(lastStamp);
646  // no need to continue search
647  return false;
648  }
649 
650  return true;
651  };
652 
653  if (!KSycocaUtilsPrivate::visitResourceDirectory(dir, visitor)) {
654  return false;
655  }
656  }
657  return true;
658  }
659 
660 private:
661  QDateTime m_now;
662 };

819 void KSycoca::ensureCacheValid()
820 {

822  return;
823  }
824 
825  if (d->databaseStatus != KSycocaPrivate::DatabaseOK) {
826  if (!d->checkDatabase(KSycocaPrivate::IfNotFoundRecreate)) {
827  return;
828  }
829  }
830 
831  if (d->m_lastCheck.isValid() && d->m_lastCheck.elapsed() < ksycoca_ms_between_checks) {
832  return;
833  }
834  d->m_lastCheck.start();
835 
836  // Check if the file on disk was modified since we last checked it.
837  QFileInfo info(d->m_databasePath);
838  if (info.lastModified() == d->m_dbLastModified) {
839  // Check if the watched directories were modified, then the cache needs a rebuild.
840  d->checkDirectories();
841  return;
842  }
843

845  // The next call to any public method will recreate
846  // everything that's needed.
847  d->closeDatabase();
848 }
