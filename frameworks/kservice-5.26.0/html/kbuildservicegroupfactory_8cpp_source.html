66 KServiceGroup::Ptr
67 KBuildServiceGroupFactory::addNew(const QString &menuName, const QString &file, KServiceGroup::Ptr entry, bool isDeleted)
68 {
69  KSycocaEntry::Ptr ptr = m_entryDict->value(menuName);
70  if (ptr) {
71  qCWarning(SYCOCA) << "( " << menuName << ", " << file << " ): menu already exists!";
72  return KServiceGroup::Ptr(static_cast<KServiceGroup*>(ptr.data()));
73  }
74 
75  // Create new group entry
76  if (!entry) {
77  entry = new KServiceGroup(file, menuName);
78  }
79 
80  entry->d_func()->m_childCount = -1; // Recalculate
81 
82  addEntry(KSycocaEntry::Ptr(entry));
83 
84  if (menuName != "/") {
85  // Make sure parent dir exists.
86  QString parent = menuName.left(menuName.length() - 1);
87  int i = parent.lastIndexOf('/');
88  if (i > 0) {
89  parent = parent.left(i + 1);
90  } else {
91  parent = '/';
92  }
93 
94  KServiceGroup::Ptr parentEntry;
95  ptr = m_entryDict->value(parent);
96  if (ptr && ptr->isType(KST_KServiceGroup)) {
97  parentEntry = KServiceGroup::Ptr(static_cast<KServiceGroup*>(ptr.data()));
98  }
99  if (!parentEntry) {
100  qCWarning(SYCOCA) << "( " << menuName << ", " << file << " ): parent menu does not exist!";
101  } else {
102  if (!isDeleted && !entry->isDeleted()) {
103  parentEntry->addEntry(KSycocaEntry::Ptr(entry));
104  }
105  }
106  }
107  return entry;
108 }
