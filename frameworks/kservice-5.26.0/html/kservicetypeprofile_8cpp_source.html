67 void KServiceTypeProfiles::ensureParsed()
68 {
69  if (m_parsed) {
70  return;
71  }
72  m_parsed = true;
73 
74  // Read the service type profiles from servicetype_profilerc
75  // See writeServiceTypeProfile for a description of the file format.
76  // ### Since this new format names groups after servicetypes maybe we can even
77  // avoid doing any init upfront, and just look up the group when asked...
78  KConfig configFile(QStringLiteral("servicetype_profilerc"), KConfig::NoGlobals);
79  const QStringList tmpList = configFile.groupList();
80  for (QStringList::const_iterator aIt = tmpList.begin();
81  aIt != tmpList.end(); ++aIt) {
82  const QString type = *aIt;
83  KConfigGroup config(&configFile, type);
84  const int count = config.readEntry("NumberOfEntries", 0);
85  KServiceTypeProfileEntry *p = this->value(type, 0);
86  if (!p) {
87  p = new KServiceTypeProfileEntry();
88  this->insert(type, p);
89  }
90 
91  for (int i = 0; i < count; ++i) {
92  const QString num = QString::fromLatin1("Entry") + QString::number(i);
93  const QString serviceId = config.readEntry(num + QLatin1String("_Service"), QString());
94  if (!serviceId.isEmpty()) {
95  const int pref = config.readEntry(num + QLatin1String("_Preference"), 0);
96  //qDebug() << "adding service " << serviceId << " to profile for " << type << " with preference " << pref;
97  p->addService(serviceId, pref);
98  }
99  }
100  }
101 }

122 KServiceOfferList KServiceTypeProfile::sortServiceTypeOffers(const KServiceOfferList &list, const QString &serviceType)
123 {
124  QMutexLocker lock(&s_serviceTypeProfiles()->m_mutex);
125  s_serviceTypeProfiles()->ensureParsed();
126  KServiceTypeProfileEntry *profile = s_serviceTypeProfiles()->value(serviceType, 0);
127 
128  KServiceOfferList offers;
129 
130  KServiceOfferList::const_iterator it = list.begin();
131  const KServiceOfferList::const_iterator end = list.end();
132  for (; it != end; ++it) {
133  const KService::Ptr servPtr = (*it).service();
134  //qDebug() << "KServiceTypeProfile::offers considering " << servPtr->storageId();
135  // Look into the profile (if there's one), to find this service's preference.
136  bool foundInProfile = false;
137  if (profile) {
138  QMap<QString, int>::ConstIterator it2 = profile->m_mapServices.constFind(servPtr->storageId());
139  if (it2 != profile->m_mapServices.constEnd()) {
140  const int pref = it2.value();
141  //qDebug() << "found in mapServices pref=" << pref;
142  if (pref > 0) { // 0 disables the service
143  offers.append(KServiceOffer(servPtr, pref, 0, servPtr->allowAsDefault()));
144  }
145  foundInProfile = true;
146  }
147  }
148  if (!foundInProfile) {
149  // This offer isn't in the profile
150  // This can be because we have no profile at all, or because the
151  // services have been installed after the profile was written,
152  // but it's also the case for any service that's neither App nor ReadOnlyPart, e.g. RenameDlg/Plugin
153  //qDebug() << "not found in mapServices. Appending.";
154 
155  // If there's a profile, we use 0 as the preference to ensure new apps don't take over existing apps (which default to 1)
156  offers.append(KServiceOffer(servPtr,
157  profile ? 0 : (*it).preference(),
158  0,
159  servPtr->allowAsDefault()));
160  }
161  }
162 
163  qStableSort(offers);
164 
165  //qDebug() << "KServiceTypeProfile::offers returning " << offers.count() << " offers";
166  return offers;
167 }

174 void KServiceTypeProfile::writeServiceTypeProfile(const QString &serviceType,
175  const KService::List &services,
176  const KService::List &disabledServices)
177 {
178  /*
179  * [ServiceType]
180  * NumEntries=3
181  * Entry0_Service=serv.desktop
182  * Entry0_Preference=10
183  * Entry1_Service=otherserv.desktop
184  * Entry1_Preference=5
185  * Entry2_Service=broken_service.desktop
186  * Entry2_Preference=0
187  */
188 
189  KConfig configFile(QStringLiteral("servicetype_profilerc"), KConfig::SimpleConfig);
190  configFile.deleteGroup(serviceType);
191 
192  KConfigGroup config(&configFile, serviceType);
193  const int count = services.count();
194  config.writeEntry("NumberOfEntries", count + disabledServices.count());
195  KService::List::ConstIterator servit = services.begin();
196  int i = 0;
197  for (; servit != services.end(); ++servit, ++i) {
198  if (*servit) {
199  const QString num = QString::fromLatin1("Entry") + QString::number(i);
200  config.writeEntry(num + QLatin1String("_Service"), (*servit)->storageId());
201  config.writeEntry(num + QLatin1String("_Preference"), count - i);
202  }
203  }
204  servit = disabledServices.begin();
205  for (; servit != disabledServices.end(); ++servit, ++i) {
206  if (*servit) {
207  const QString num = QString::fromLatin1("Entry") + QString::number(i);
208  config.writeEntry(num + QLatin1String("_Service"), (*servit)->storageId());
209  config.writeEntry(num + QLatin1String("_Preference"), 0);
210  }
211  }
212  configFile.sync();
213 
214  // Drop the whole cache...
215  clearCache();
216 }
