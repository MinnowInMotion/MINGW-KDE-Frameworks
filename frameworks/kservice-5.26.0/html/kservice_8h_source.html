517  template <class T>
518  T *createInstance(QWidget *parentWidget, QObject *parent,
519  const QVariantList &args = QVariantList(), QString *error = 0) const
520  {
521  KPluginLoader pluginLoader(*this);
522  KPluginFactory *factory = pluginLoader.factory();
523  if (factory) {
524  QVariantList argsWithMetaData = args;
525  argsWithMetaData << pluginLoader.metaData().toVariantMap();
526  T *o = factory->template create<T>(parentWidget, parent, pluginKeyword(), argsWithMetaData);
527  if (!o && error)
528  *error = QCoreApplication::translate("", "The service '%1' does not provide an interface '%2' with keyword '%3'")
529  .arg(name(), QString::fromLatin1(T::staticMetaObject.className()), pluginKeyword());
530  return o;
531  } else if (error) {
532  *error = pluginLoader.errorString();
533  pluginLoader.unload();
534  }
535  return 0;
536  }
