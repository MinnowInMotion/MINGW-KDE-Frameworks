40 static void crashHandler(int)
41 {
42  // If we crash while reading sycoca, we delete the database
43  // in an attempt to recover.
44  if (KBuildSycoca::sycocaPath()) {
45  unlink(KBuildSycoca::sycocaPath());
46  }
47 }

101  const bool bGlobalDatabase = parser.isSet(QStringLiteral("global"));
102  const bool bMenuTest = parser.isSet(QStringLiteral("menutest"));
103 
104  if (parser.isSet(QStringLiteral("testmode"))) {
105  QStandardPaths::enableTestMode(true);
106  }
107 
108  if (bGlobalDatabase) {
109  // Qt uses XDG_DATA_HOME as first choice for GenericDataLocation so we set it to 2nd entry
110  QStringList paths = QStandardPaths::standardLocations(QStandardPaths::GenericDataLocation);
111  if (paths.size() >= 2) {
112  qputenv("XDG_DATA_HOME", paths.at(1).toLocal8Bit());
113  }
114  }
115 
116  KCrash::setEmergencySaveFunction(crashHandler);
117 
118  fprintf(stderr, "%s running...\n", KBUILDSYCOCA_EXENAME);
119 
120  const bool incremental = !bGlobalDatabase && !parser.isSet(QStringLiteral("noincremental"));
121 
122  KBuildSycoca sycoca(bGlobalDatabase); // Build data base
123  if (parser.isSet(QStringLiteral("track"))) {
124  sycoca.setTrackId(parser.value(QStringLiteral("track")));
125  }
126  sycoca.setMenuTest(bMenuTest);
127  if (!sycoca.recreate(incremental)) {
128  return -1;
129  }
130 
131  return 0;
