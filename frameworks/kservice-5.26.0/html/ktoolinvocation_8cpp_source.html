73 int KToolInvocation::startServiceInternal(const char *_function,
74  const QString &_name, const QStringList &URLs,
75  QString *error, QString *serviceName, int *pid,
76  const QByteArray &startup_id, bool noWait,
77  const QString &workdir)
78 {
79  QString function = QLatin1String(_function);
80  KToolInvocation::ensureKdeinitRunning();
81  QDBusMessage msg = QDBusMessage::createMethodCall(QStringLiteral("org.kde.klauncher5"),
82  QStringLiteral("/KLauncher"),
83  QStringLiteral("org.kde.KLauncher"),
84  function);
85  msg << _name << URLs;
86  if (function == QLatin1String("kdeinit_exec_with_workdir")) {
87  msg << workdir;
88  }
89  // make sure there is id, so that user timestamp exists
90  QStringList envs;
91  QByteArray s = startup_id;
92  emit kapplication_hook(envs, s);
93  msg << envs;
94  msg << QString::fromLatin1(s);
95  if (!function.startsWith(QLatin1String("kdeinit_exec"))) {
96  msg << noWait;
97  }
98 
99  QDBusMessage reply = QDBusConnection::sessionBus().call(msg, QDBus::Block, INT_MAX);
100  if (reply.type() != QDBusMessage::ReplyMessage) {
101  QDBusReply<QString> replyObj(reply);
102  if (replyObj.error().type() == QDBusError::NoReply) {
103  printError(i18n("Error launching %1. Either KLauncher is not running anymore, or it failed to start the application.", _name), error);
104  } else {
105  const QString rpl = reply.arguments().count() > 0 ? reply.arguments().at(0).toString() : reply.errorMessage();
106  printError(i18n("KLauncher could not be reached via D-Bus. Error when calling %1:\n%2\n", function, rpl), error);
107  }
108  //qDebug() << reply;
109  return EINVAL;
110  }
111 
112  if (noWait) {
113  return 0;
114  }
115 
116  Q_ASSERT(reply.arguments().count() == 4);
117  if (serviceName) {
118  *serviceName = reply.arguments().at(1).toString();
119  }
120  if (error) {
121  *error = reply.arguments().at(2).toString();
122  }
123  if (pid) {
124  *pid = reply.arguments().at(3).toInt();
125  }
126  return reply.arguments().at(0).toInt();
127 }

257 void KToolInvocation::invokeMailer(const QUrl &mailtoURL, const QByteArray &startup_id, bool allowAttachments)
258 {
259  if (!isMainThreadActive()) {
260  return;
261  }
262 
263  QString address = mailtoURL.path();
264  QString subject;
265  QString cc;
266  QString bcc;
267  QString body;
268 
269  QList<QPair<QString, QString> > queryItems = QUrlQuery(mailtoURL).queryItems();
270  const QChar comma = QChar::fromLatin1(',');
271  QStringList attachURLs;
272  for (int i = 0; i < queryItems.count(); ++i) {
273  const QString q = queryItems.at(i).first.toLower();
274  const QString value = queryItems.at(i).second;
275  if (q == QLatin1String("subject")) {
276  subject = value;
277  } else if (q == QLatin1String("cc")) {
278  cc = cc.isEmpty() ? value : cc + comma + value;
279  } else if (q == QLatin1String("bcc")) {
280  bcc = bcc.isEmpty() ? value : bcc + comma + value;
281  } else if (q == QLatin1String("body")) {
282  body = value;
283  } else if (allowAttachments && q == QLatin1String("attach")) {
284  attachURLs.push_back(value);
285  } else if (allowAttachments && q == QLatin1String("attachment")) {
286  attachURLs.push_back(value);
287  } else if (q == QLatin1String("to")) {
288  address = address.isEmpty() ? value : address + comma + value;
289  }
290  }
291 
292  invokeMailer(address, cc, bcc, subject, body, QString(), attachURLs, startup_id);
293 }
