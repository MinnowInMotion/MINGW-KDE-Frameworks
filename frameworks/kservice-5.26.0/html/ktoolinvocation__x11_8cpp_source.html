48 static QStringList splitEmailAddressList(const QString &aStr)
49 {
50  // This is a copy of KPIM::splitEmailAddrList().
51  // Features:
52  // - always ignores quoted characters
53  // - ignores everything (including parentheses and commas)
54  // inside quoted strings
55  // - supports nested comments
56  // - ignores everything (including double quotes and commas)
57  // inside comments
58 
59  QStringList list;
60 
61  if (aStr.isEmpty()) {
62  return list;
63  }
64 
65  QString addr;
66  uint addrstart = 0;
67  int commentlevel = 0;
68  bool insidequote = false;
69 
70  for (int index = 0; index < aStr.length(); index++) {
71  // the following conversion to latin1 is o.k. because
72  // we can safely ignore all non-latin1 characters
73  switch (aStr[index].toLatin1()) {
74  case '"' : // start or end of quoted string
75  if (commentlevel == 0) {
76  insidequote = !insidequote;
77  }
78  break;
79  case '(' : // start of comment
80  if (!insidequote) {
81  commentlevel++;
82  }
83  break;
84  case ')' : // end of comment
85  if (!insidequote) {
86  if (commentlevel > 0) {
87  commentlevel--;
88  } else {
89  //qDebug() << "Error in address splitting: Unmatched ')'"
90  // << endl;
91  return list;
92  }
93  }
94  break;
95  case '\\' : // quoted character
96  index++; // ignore the quoted character
97  break;
98  case ',' :
99  if (!insidequote && (commentlevel == 0)) {
100  addr = aStr.mid(addrstart, index - addrstart);
101  if (!addr.isEmpty()) {
102  list += addr.simplified();
103  }
104  addrstart = index + 1;
105  }
106  break;
107  }
108  }
109  // append the last address to the list
110  if (!insidequote && (commentlevel == 0)) {
111  addr = aStr.mid(addrstart, aStr.length() - addrstart);
112  if (!addr.isEmpty()) {
113  list += addr.simplified();
114  }
115  }
116  //else
117  // qDebug() << "Error in address splitting: "
118  // << "Unexpected end of address list"
119  // << endl;
120 
121  return list;
122 }

124 void KToolInvocation::invokeMailer(const QString &_to, const QString &_cc, const QString &_bcc,
125  const QString &subject, const QString &body,
126  const QString & /*messageFile TODO*/, const QStringList &attachURLs,
127  const QByteArray &startup_id)
128 {
129  if (!isMainThreadActive()) {
130  return;
131  }
132 
133  KConfig config(QStringLiteral("emaildefaults"));
134  KConfigGroup defaultsGrp(&config, "Defaults");
135 
136  QString group = defaultsGrp.readEntry("Profile", "Default");
137 
138  KConfigGroup profileGrp(&config, QStringLiteral("PROFILE_%1").arg(group));
139  QString command = profileGrp.readPathEntry("EmailClient", QString());
140 
141  QString to, cc, bcc;
142  if (command.isEmpty() || command == QLatin1String("kmail")
143  || command.endsWith(QLatin1String("/kmail"))) {
144  command = QStringLiteral("kmail --composer -s %s -c %c -b %b --body %B --attach %A -- %t");
145  if (!_to.isEmpty()) {
146  QUrl url;
147  url.setScheme(QStringLiteral("mailto"));
148  url.setPath(_to);
149  to = QString::fromLatin1(url.toEncoded());
150  }
151  if (!_cc.isEmpty()) {
152  QUrl url;
153  url.setScheme(QStringLiteral("mailto"));
154  url.setPath(_cc);
155  cc = QString::fromLatin1(url.toEncoded());
156  }
157  if (!_bcc.isEmpty()) {
158  QUrl url;
159  url.setScheme(QStringLiteral("mailto"));
160  url.setPath(_bcc);
161  bcc = QString::fromLatin1(url.toEncoded());
162  }
163  } else {
164  to = _to;
165  cc = _cc;
166  bcc = _bcc;
167  if (!command.contains(QLatin1Char('%'))) {
168  command += QLatin1String(" %u");
169  }
170  }
171 
172  if (profileGrp.readEntry("TerminalClient", false)) {
173  KConfigGroup confGroup(KSharedConfig::openConfig(), "General");
174  QString preferredTerminal = confGroup.readPathEntry("TerminalApplication", QStringLiteral("konsole"));
175  command = preferredTerminal + QString::fromLatin1(" -e ") + command;
176  }
177 
178  QStringList cmdTokens = KShell::splitArgs(command);
179  QString cmd = cmdTokens.takeFirst();
180 
181  QUrl url;
182  QUrlQuery query;
183  if (!to.isEmpty()) {
184  QStringList tos = splitEmailAddressList(to);
185  url.setPath(tos.first());
186  tos.erase(tos.begin());
187  for (QStringList::ConstIterator it = tos.constBegin(); it != tos.constEnd(); ++it) {
188  query.addQueryItem(QStringLiteral("to"), *it);
189  }
190  }
191  const QStringList ccs = splitEmailAddressList(cc);
192  for (QStringList::ConstIterator it = ccs.constBegin(); it != ccs.constEnd(); ++it) {
193  query.addQueryItem(QStringLiteral("cc"), *it);
194  }
195  const QStringList bccs = splitEmailAddressList(bcc);
196  for (QStringList::ConstIterator it = bccs.constBegin(); it != bccs.constEnd(); ++it) {
197  query.addQueryItem(QStringLiteral("bcc"), *it);
198  }
199  for (QStringList::ConstIterator it = attachURLs.constBegin(); it != attachURLs.constEnd(); ++it) {
200  query.addQueryItem(QStringLiteral("attach"), *it);
201  }
202  if (!subject.isEmpty()) {
203  query.addQueryItem(QStringLiteral("subject"), subject);
204  }
205  if (!body.isEmpty()) {
206  query.addQueryItem(QStringLiteral("body"), body);
207  }
208 
209  url.setQuery(query);
210 
211  if (!(to.isEmpty() && (!url.hasQuery()))) {
212  url.setScheme(QStringLiteral("mailto"));
213  }
214 
215  QHash<QChar, QString> keyMap;
216  keyMap.insert(QLatin1Char('t'), to);
217  keyMap.insert(QLatin1Char('s'), subject);
218  keyMap.insert(QLatin1Char('c'), cc);
219  keyMap.insert(QLatin1Char('b'), bcc);
220  keyMap.insert(QLatin1Char('B'), body);
221  keyMap.insert(QLatin1Char('u'), url.toString());
222 
223  QString attachlist = attachURLs.join(QLatin1Char(','));
224  attachlist.prepend(QLatin1Char('\''));
225  attachlist.append(QLatin1Char('\''));
226  keyMap.insert(QLatin1Char('A'), attachlist);
227  for (int i = 0; i < cmdTokens.count(); ++i) {
228  if (cmdTokens.at(i) == QLatin1String("%A")) {
229  if (attachURLs.isEmpty()) {
230  cmdTokens.removeAt(i);
231  } else {
232  const QString previousStr = cmdTokens.at(i-1);
233  cmdTokens.removeAt(i);
234  const int currentPos = i;
235  Q_FOREACH(const QString &url, attachURLs) {
236  cmdTokens.insert(currentPos, previousStr);
237  cmdTokens.insert(currentPos, url);
238  i += 2;
239  }
240  }
241  } else {
242  const QString str = KMacroExpander::expandMacros(cmdTokens.at(i), keyMap);
243  cmdTokens[i] = str;
244  }
245  }
246  QString error;
247  // TODO this should check if cmd has a .desktop file, and use data from it, together
248  // with sending more ASN data
249  if (kdeinitExec(cmd, cmdTokens, &error, NULL, startup_id)) {
250  KMessage::message(KMessage::Error,
251  i18n("Could not launch the mail client:\n\n%1", error),
252  i18n("Could not launch Mail Client"));
253  }
254 }

256 void KToolInvocation::invokeBrowser(const QString &url, const QByteArray &startup_id)
257 {
258  if (!isMainThreadActive()) {
259  return;
260  }
261 
262  QStringList args;
263  args << url;
264  QString error;
265 
266  // This method should launch a webbrowser, preferably without doing a mimetype
267  // check first, like KRun (i.e. kde-open) would do.
268 
269  // In a KDE session, honour BrowserApplication if set, otherwise use preferred app for text/html if any,
270  // otherwise xdg-open, otherwise kde-open (which does a mimetype check first though).
271 
272  // Outside KDE, call xdg-open if present, otherwise fallback to the above logic.
273 
274  QString exe; // the binary we are going to launch.
275 
276  const QString xdg_open = QStandardPaths::findExecutable(QStringLiteral("xdg-open"));
277  if (qEnvironmentVariableIsEmpty("KDE_FULL_SESSION")) {
278  exe = xdg_open;
279  }
280 
281  if (exe.isEmpty()) {
282  // We're in a KDE session (or there's no xdg-open installed)
283  KConfigGroup config(KSharedConfig::openConfig(), "General");
284  const QString browserApp = config.readPathEntry("BrowserApplication", QString());
285  if (!browserApp.isEmpty()) {
286  exe = browserApp;
287  if (exe.startsWith(QLatin1Char('!'))) {
288  exe = exe.mid(1); // Literal command
289  QStringList cmdTokens = KShell::splitArgs(exe);
290  exe = cmdTokens.takeFirst();
291  args = cmdTokens + args;
292  } else {
293  // desktop file ID
294  KService::Ptr service = KService::serviceByStorageId(exe);
295  if (service) {
296  //qDebug() << "Starting service" << service->entryPath();
297  if (startServiceByDesktopPath(service->entryPath(), args,
298  &error, 0, 0, startup_id)) {
299  KMessage::message(KMessage::Error,
300  // TODO: i18n("Could not launch %1:\n\n%2", exe, error),
301  i18n("Could not launch the browser:\n\n%1", error),
302  i18n("Could not launch Browser"));
303  }
304  return;
305  }
306  }
307  } else {
308  const KService::Ptr htmlApp = KMimeTypeTrader::self()->preferredService(QStringLiteral("text/html"));
309  if (htmlApp) {
310  // WORKAROUND: For bugs 264562 and 265474:
311  // In order to correctly handle non-HTML urls we change the service
312  // desktop file name to "kfmclient.desktop" whenever the above query
313  // returns "kfmclient_html.desktop".Otherwise, the hard coded mime-type
314  // "text/html" mime-type parameter in the kfmclient_html will cause all
315  // URLs to be treated as if they are HTML page.
316  QString entryPath = htmlApp->entryPath();
317  if (entryPath.endsWith(QLatin1String("kfmclient_html.desktop"))) {
318  entryPath.remove(entryPath.length() - 13, 5);
319  }
320  QString error;
321  int pid = 0;
322  int err = startServiceByDesktopPath(entryPath, url, &error, 0, &pid, startup_id);
323  if (err != 0) {
324  KMessage::message(KMessage::Error,
325  // TODO: i18n("Could not launch %1:\n\n%2", htmlApp->exec(), error),
326  i18n("Could not launch the browser:\n\n%1", error),
327  i18n("Could not launch Browser"));
328  } else { // success
329  return;
330  }
331  } else {
332  exe = xdg_open;
333  }
334  }
335  }
336 
337  if (exe.isEmpty()) {
338  exe = QStringLiteral("kde-open"); // it's from kdebase-runtime, it has to be there.
339  }
340 
341  //qDebug() << "Using" << exe << "to open" << url;
342  if (kdeinitExec(exe, args, &error, NULL, startup_id)) {
343  KMessage::message(KMessage::Error,
344  // TODO: i18n("Could not launch %1:\n\n%2", exe, error),
345  i18n("Could not launch the browser:\n\n%1", error),
346  i18n("Could not launch Browser"));
347  }
348 }

350 void KToolInvocation::invokeTerminal(const QString &command,
351  const QString &workdir,
352  const QByteArray &startup_id)
353 {
354  if (!isMainThreadActive()) {
355  return;
356  }
357 
358  KConfigGroup confGroup(KSharedConfig::openConfig(), "General");
359  QString exec = confGroup.readPathEntry("TerminalApplication", QStringLiteral("konsole"));
360 
361  if (!command.isEmpty()) {
362  if (exec == QLatin1String("konsole")) {
363  exec += QString::fromLatin1(" --noclose");
364  } else if (exec == QLatin1String("xterm")) {
365  exec += QString::fromLatin1(" -hold");
366  }
367 
368  exec += QString::fromLatin1(" -e ") + command;
369  }
370 
371  QStringList cmdTokens = KShell::splitArgs(exec);
372  QString cmd = cmdTokens.takeFirst();
373 
374  if (exec == QLatin1String("konsole") && !workdir.isEmpty()) {
375  cmdTokens << QStringLiteral("--workdir");
376  cmdTokens << workdir;
377  // For other terminals like xterm, we'll simply change the working
378  // directory before launching them, see below.
379  }
380 
381  QString error;
382  if (self()->startServiceInternal("kdeinit_exec_with_workdir",
383  cmd, cmdTokens, &error, 0, NULL, startup_id, false, workdir)) {
384  KMessage::message(KMessage::Error,
385  i18n("Could not launch the terminal client:\n\n%1", error),
386  i18n("Could not launch Terminal Client"));
387  }
388 }
