52 void KMimeAssociations::parseAllMimeAppsList()
53 {
54  QStringList mimeappsFileNames;
55  // make the list of possible filenames from the spec ($desktop-mimeapps.list, then mimeapps.list)
56  const QString desktops = QString::fromLocal8Bit(qgetenv("XDG_CURRENT_DESKTOP"));
57  foreach (const QString &desktop, desktops.split(QLatin1Char(':'), QString::SkipEmptyParts)) {
58  mimeappsFileNames.append(desktop.toLower() + QLatin1String("-mimeapps.list"));
59  }
60  mimeappsFileNames.append(QStringLiteral("mimeapps.list"));
61  // list the dirs in the order of the spec (XDG_CONFIG_HOME, XDG_CONFIG_DIRS, XDG_DATA_HOME, XDG_DATA_DIRS)
62  const QStringList mimeappsDirs = QStandardPaths::standardLocations(QStandardPaths::GenericConfigLocation)
63  + QStandardPaths::standardLocations(QStandardPaths::ApplicationsLocation);
64  QStringList mimeappsFiles;
65  // collect existing files
66  foreach (const QString &dir, mimeappsDirs) {
67  foreach (const QString &file, mimeappsFileNames) {
68  const QString filePath = dir + '/' + file;
69  if (QFile::exists(filePath)) {
70  mimeappsFiles.append(filePath);
71  }
72  }
73  }
74  //qDebug() << "FILE LIST:" << mimeappsFiles;
75 
76  int basePreference = 1000; // start high :)
77  QListIterator<QString> mimeappsIter(mimeappsFiles);
78  mimeappsIter.toBack();
79  while (mimeappsIter.hasPrevious()) { // global first, then local.
80  const QString mimeappsFile = mimeappsIter.previous();
81  //qDebug() << "Parsing" << mimeappsFile;
82  parseMimeAppsList(mimeappsFile, basePreference);
83  basePreference += 50;
84  }
85 }

87 void KMimeAssociations::parseMimeAppsList(const QString &file, int basePreference)
88 {
89  KConfig profile(file, KConfig::SimpleConfig);
90  if (file.endsWith(QLatin1String("/mimeapps.list"))) { // not for $desktop-mimeapps.list
91  parseAddedAssociations(KConfigGroup(&profile, "Added Associations"), file, basePreference);
92  parseRemovedAssociations(KConfigGroup(&profile, "Removed Associations"), file);
93 
94  // KDE extension for parts and plugins, see settings/filetypes/mimetypedata.cpp
95  parseAddedAssociations(KConfigGroup(&profile, "Added KDE Service Associations"), file, basePreference);
96  parseRemovedAssociations(KConfigGroup(&profile, "Removed KDE Service Associations"), file);
97  }
98 
99  // TODO "Default Applications" is a separate query and a separate algorithm, says the spec.
100  // For now this is better than nothing though.
101  parseAddedAssociations(KConfigGroup(&profile, "Default Applications"), file, basePreference);
102 }

104 void KMimeAssociations::parseAddedAssociations(const KConfigGroup &group, const QString &file, int basePreference)
105 {
106  Q_UNUSED(file) // except in debug statements
107  QMimeDatabase db;
108  Q_FOREACH (const QString &mimeName, group.keyList()) {
109  const QStringList services = group.readXdgListEntry(mimeName);
110  const QString resolvedMimeName = mimeName.startsWith("x-scheme-handler/") ? mimeName : db.mimeTypeForName(mimeName).name();
111  if (resolvedMimeName.isEmpty()) {
112  qCDebug(SYCOCA) << file << "specifies unknown mimeType" << mimeName << "in" << group.name();
113  } else {
114  int pref = basePreference;
115  Q_FOREACH (const QString &service, services) {
116  KService::Ptr pService = m_serviceFactory->findServiceByStorageId(service);
117  if (!pService) {
118  qDebug(SYCOCA) << file << "specifies unknown service" << service << "in" << group.name();
119  } else {
120  //qDebug() << "adding mime" << resolvedMimeName << "to service" << pService->entryPath() << "pref=" << pref;
121  m_offerHash.addServiceOffer(resolvedMimeName, KServiceOffer(pService, pref, 0, pService->allowAsDefault()));
122  --pref;
123  }
124  }
125  }
126  }
127 }

129 void KMimeAssociations::parseRemovedAssociations(const KConfigGroup &group, const QString &file)
130 {
131  Q_UNUSED(file) // except in debug statements
132  Q_FOREACH (const QString &mime, group.keyList()) {
133  const QStringList services = group.readXdgListEntry(mime);
134  Q_FOREACH (const QString &service, services) {
135  KService::Ptr pService = m_serviceFactory->findServiceByStorageId(service);
136  if (!pService) {
137  //qDebug() << file << "specifies unknown service" << service << "in" << group.name();

140  m_offerHash.removeServiceOffer(mime, pService);

146 void KOfferHash::addServiceOffer(const QString &serviceType, const KServiceOffer &offer)
147 {
148  KService::Ptr service = offer.service();
149  //qDebug() << "Adding" << service->entryPath() << "to" << serviceType << offer.preference();
150  ServiceTypeOffersData &data = m_serviceTypeData[serviceType]; // find or create
151  QList<KServiceOffer> &offers = data.offers;
152  QSet<KService::Ptr> &offerSet = data.offerSet;
153  if (!offerSet.contains(service)) {
154  offers.append(offer);
155  offerSet.insert(service);
156  } else {
157  //qDebug() << service->entryPath() << "already in" << serviceType;
158  // This happens when mimeapps.list mentions a service (to make it preferred)
159  // Update initialPreference to qMax(existing offer, new offer)
160  QMutableListIterator<KServiceOffer> sfit(data.offers);
161  while (sfit.hasNext()) {
162  if (sfit.next().service() == service) { // we can compare KService::Ptrs because they are from the memory hash
163  sfit.value().setPreference(qMax(sfit.value().preference(), offer.preference()));
164  }
165  }
166  }
167 }

169 void KOfferHash::removeServiceOffer(const QString &serviceType, KService::Ptr service)
170 {
171  ServiceTypeOffersData &data = m_serviceTypeData[serviceType]; // find or create
172  data.removedOffers.insert(service);
173  data.offerSet.remove(service);
174  QMutableListIterator<KServiceOffer> sfit(data.offers);
175  while (sfit.hasNext()) {
176  if (sfit.next().service()->storageId() == service->storageId()) {
177  sfit.remove();
178  }
179  }
180 }
