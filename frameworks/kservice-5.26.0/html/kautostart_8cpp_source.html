50 void KAutostartPrivate::copyIfNeeded()
51 {
52  if (copyIfNeededChecked) {
53  return;
54  }
55 
56  const QString local = QStandardPaths::writableLocation(QStandardPaths::GenericConfigLocation) + QLatin1String("/autostart/") + name;
57 
58  if (!QFile::exists(local)) {
59  const QString global = QStandardPaths::locate(QStandardPaths::GenericConfigLocation, QLatin1String("autostart/") + name);
60  if (!global.isEmpty()) {
61  KDesktopFile *newDf = df->copyTo(local);
62  delete df;
63  delete newDf; //Force sync-to-disk
64  df = new KDesktopFile(QStandardPaths::GenericConfigLocation, QStringLiteral("autostart/") + name); //Recreate from disk
65  }
66  }
67 
68  copyIfNeededChecked = true;
69 }

71 KAutostart::KAutostart(const QString &entryName, QObject *parent)
72  : QObject(parent),

111 bool KAutostart::autostarts(const QString &environment, Conditions check) const
112 {
113  // check if this is actually a .desktop file
114  bool starts = d->df->desktopGroup().exists();
115 
116  // check the hidden field
117  starts = starts && !d->df->desktopGroup().readEntry("Hidden", false);
118 
119  if (!environment.isEmpty()) {
120  starts = starts && checkAllowedEnvironment(environment);
121  }
122 
123  if (check & CheckCommand) {
124  starts = starts && d->df->tryExec();
125  }
126 
127  if (check & CheckCondition) {
128  starts = starts && checkStartCondition();
129  }
130 
131  return starts;
132 }

134 bool KAutostart::checkStartCondition() const
135 {
136  QString condition = d->df->desktopGroup().readEntry("X-KDE-autostart-condition");
137  if (condition.isEmpty()) {
138  return true;
139  }
140 
141  const QStringList list = condition.split(QLatin1Char(':'));
142  if (list.count() < 4) {
143  return true;
144  }
145 
146  if (list[0].isEmpty() || list[2].isEmpty()) {
147  return true;
148  }
149 
150  KConfig config(list[0], KConfig::NoGlobals);
151  KConfigGroup cg(&config, list[1]);
152 
153  const bool defaultValue = (list[3].toLower() == QLatin1String("true"));
154  return cg.readEntry(list[2], defaultValue);
155 }

157 bool KAutostart::checkAllowedEnvironment(const QString &environment) const
158 {
159  const QStringList allowed = allowedEnvironments();
160  if (!allowed.isEmpty()) {
161  return allowed.contains(environment);
162  }
163 
164  const QStringList excluded = excludedEnvironments();
165  if (!excluded.isEmpty()) {
166  return !excluded.contains(environment);
167  }
168 
169  return true;
170 }

223 // do not specialize the readEntry template -
224 // http://connect.microsoft.com/VisualStudio/feedback/ViewFeedback.aspx?FeedbackID=100911
225 static KAutostart::StartPhase readEntry(const KConfigGroup &group, const char *key, const KAutostart::StartPhase &aDefault)
226 {
227  const QByteArray data = group.readEntry(key, QByteArray());
228 
229  if (data.isNull()) {
230  return aDefault;
231  }
232 
233  if (data == "0" || data == "BaseDesktop") {
234  return KAutostart::BaseDesktop;
235  } else if (data == "1" || data == "DesktopServices") {
236  return KAutostart::DesktopServices;
237  } else if (data == "2" || data == "Applications") {
238  return KAutostart::Applications;
239  }
240 
241  return aDefault;
242 }

249 void KAutostart::setStartPhase(KAutostart::StartPhase phase)
250 {
251  QString data = QStringLiteral("Applications");
252 
253  switch (phase) {
254  case BaseDesktop:
255  data = QStringLiteral("BaseDesktop");
256  break;
257  case DesktopServices:
258  data = QStringLiteral("DesktopServices");
259  break;
260  case Applications: // This is the default
261  break;
262  }
263 
264  if (d->df->desktopGroup().readEntry("X-KDE-autostart-phase", QString()) == data) {
265  return;
266  }
267 
268  d->copyIfNeeded();
269  d->df->desktopGroup().writeEntry("X-KDE-autostart-phase", data);
270 }
