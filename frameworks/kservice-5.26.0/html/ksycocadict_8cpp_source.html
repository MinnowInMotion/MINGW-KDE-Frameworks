235 uint KSycocaDictPrivate::hashKey(const QString &key) const
236 {
237  int len = key.length();
238  uint h = 0;
239 
240  for (int i = 0; i < hashList.count(); i++) {
241  int pos = hashList[i];
242  if (pos == 0) {
243  continue;
244  } else if (pos < 0) {
245  pos = -pos;
246  if (pos < len) {
247  h = ((h * 13) + (key[len - pos].cell() % 29)) & 0x3ffffff;
248  }
249  } else {
250  pos = pos - 1;
251  if (pos < len) {
252  h = ((h * 13) + (key[pos].cell() % 29)) & 0x3ffffff;
253  }
254  }
255  }
256  return h;
257 }

259 // If we have the strings
260 // hello
261 // world
262 // kde
263 // Then we end up with
264 // ABCDE
265 // where A = diversity of 'h' + 'w' + 'k' etc.
266 // Also, diversity(-2) == 'l'+'l'+'d' (second character from the end)
267 
268 // The hasList is used for hashing:
269 // hashList = (-2, 1, 3) means that the hash key comes from
270 // the 2nd character from the right, then the 1st from the left, then the 3rd from the left.
271 
272 // Calculate the diversity of the strings at position 'pos'
273 // NOTE: this code is slow, it takes 12% of the _overall_ `kbuildsycoca5 --noincremental` running time
274 static int
275 calcDiversity(KSycocaDictStringList* stringlistp, int inPos, uint sz)
276 {
277  if (inPos == 0) {
278  return 0;
279  }
280  KSycocaDictStringList& stringlist = *stringlistp;
281  QBitArray matrix(sz);
282  int pos;
283 
284  //static const int s_maxItems = 50;
285  //int numItem = 0;
286 
287  if (inPos < 0) {
288  pos = -inPos;
289  for (auto it = stringlist.constBegin(), end = stringlist.constEnd(); it != end; ++it) {
290  string_entry *entry = *it;
291  int rpos = entry->length - pos;
292  if (rpos > 0) {
293  uint hash = ((entry->hash * 13) + (entry->key[rpos].cell() % 29)) & 0x3ffffff;
294  matrix.setBit(hash % sz, true);
295  }
296  //if (++numItem == s_maxItems)
297  // break;
298  }
299  } else {
300  pos = inPos - 1;
301  for (auto it = stringlist.constBegin(), end = stringlist.constEnd(); it != end; ++it) {
302  string_entry *entry = *it;
303  if (pos < entry->length) {
304  uint hash = ((entry->hash * 13) + (entry->key[pos].cell() % 29)) & 0x3ffffff;
305  matrix.setBit(hash % sz, true);
306  }
307  //if (++numItem == s_maxItems)
308  // break;
309  }
310  }
311  return matrix.count(true);
312 }

317 addDiversity(KSycocaDictStringList* stringlistp, int pos)
318 {
319  if (pos == 0) {
320  return;
321  }
322  KSycocaDictStringList& stringlist = *stringlistp;
323  if (pos < 0) {
324  pos = -pos;
325  for (auto it = stringlist.constBegin(), end = stringlist.constEnd(); it != end; ++it) {
326  string_entry *entry = *it;
327  int rpos = entry->length - pos;
328  if (rpos > 0) {
329  entry->hash = ((entry->hash * 13) + (entry->key[rpos].cell() % 29)) & 0x3fffffff;
330  }
331  }
332  } else {
333  pos = pos - 1;
334  for (auto it = stringlist.constBegin(), end = stringlist.constEnd(); it != end; ++it) {
335  string_entry *entry = *it;
336  if (pos < entry->length) {
337  entry->hash = ((entry->hash * 13) + (entry->key[pos].cell() % 29)) & 0x3fffffff;
338  }

343 void
344 KSycocaDict::save(QDataStream &str)
345 {
346  if (count() == 0) {
347  d->hashTableSize = 0;
348  d->hashList.clear();
349  str << d->hashTableSize;
350  str << d->hashList;
351  return;
352  }
353 
354  d->offset = str.device()->pos();
355 
356  //qDebug() << "KSycocaDict:" << count() << "entries.";
357 
358  //qDebug() << "Calculating hash keys..";
359 
360  int maxLength = 0;
361  //qDebug() << "Finding maximum string length";
362  for (KSycocaDictStringList::const_iterator it = d->stringlist.constBegin(); it != d->stringlist.constEnd(); ++it) {
363  string_entry *entry = *it;
364  entry->hash = 0;
365  if (entry->length > maxLength) {
366  maxLength = entry->length;
367  }
368  }
369 
370  //qDebug() << "Max string length=" << maxLength << "existing hashList=" << d->hashList;
371 
372  // use "almost prime" number for sz (to calculate diversity) and later
373  // for the table size of big tables
374  // int sz = d->stringlist.count()*5-1;
375  unsigned int sz = count() * 4 + 1;
376  while (!(((sz % 3) && (sz % 5) && (sz % 7) && (sz % 11) && (sz % 13)))) {
377  sz += 2;
378  }
379 
380  d->hashList.clear();
381 
382  // Times (with warm caches, i.e. after multiple runs)
383  // kbuildsycoca5 --noincremental 2.83s user 0.20s system 95% cpu 3.187 total
384  // kbuildsycoca5 --noincremental 2.74s user 0.25s system 93% cpu 3.205 total
385  // unittest: 0.50-60 msec per iteration / 0.40-50 msec per iteration
386 
387  // Now that MimeTypes are not parsed anymore:
388  // kbuildsycoca5 --noincremental 2.18s user 0.30s system 91% cpu 2.719 total
389  // kbuildsycoca5 --noincremental 2.07s user 0.34s system 89% cpu 2.681 total
390 
391  // If I enabled s_maxItems = 50, it goes down to
392  // but I don't know if that's a good idea.
393  // kbuildsycoca5 --noincremental 1.73s user 0.31s system 85% cpu 2.397 total
394  // kbuildsycoca5 --noincremental 1.84s user 0.29s system 95% cpu 2.230 total
395 
396  // try to limit diversity scan by "predicting" positions
397  // with high diversity
398  QVector<int> oldvec(maxLength * 2 + 1);
399  oldvec.fill(0);
400  int mindiv = 0;
401  int lastDiv = 0;
402 
403  while (true) {
404  int divsum = 0, divnum = 0;
405 
406  int maxDiv = 0;
407  int maxPos = 0;
408  for (int pos = -maxLength; pos <= maxLength; ++pos) {
409  // cut off
410  if (oldvec[pos + maxLength] < mindiv) {
411  oldvec[pos + maxLength] = 0;
412  continue;
413  }
414 
415  const int diversity = calcDiversity(&(d->stringlist), pos, sz);
416  if (diversity > maxDiv) {
417  maxDiv = diversity;
418  maxPos = pos;
419  }
420  oldvec[pos + maxLength] = diversity;
421  divsum += diversity;
422  ++divnum;
423  }
424  // arbitrary cut-off value 3/4 of average seems to work
425  if (divnum) {
426  mindiv = (3 * divsum) / (4 * divnum);
427  }
428 
429  if (maxDiv <= lastDiv) {
430  break;
431  }
432  //qDebug() << "Max Div=" << maxDiv << "at pos" << maxPos;
433  lastDiv = maxDiv;
434  addDiversity(&(d->stringlist), maxPos);
435  d->hashList.append(maxPos);
436  }
437 
438  for (auto it = d->stringlist.constBegin(); it != d->stringlist.constEnd(); ++it) {
439  (*it)->hash = d->hashKey((*it)->keyStr);
440  }
441 // fprintf(stderr, "Calculating minimum table size..\n");
442 
443  d->hashTableSize = sz;
444 
445  //qDebug() << "hashTableSize=" << sz << "hashList=" << d->hashList << "oldvec=" << oldvec;
446 
447  struct hashtable_entry {
448  string_entry *entry;
449  QList<string_entry *> *duplicates;
450  qint64 duplicate_offset;
451  };
452 
453  hashtable_entry *hashTable = new hashtable_entry[ sz ];
454 
455  //qDebug() << "Clearing hashtable...";
456  for (unsigned int i = 0; i < sz; i++) {
457  hashTable[i].entry = 0;
458  hashTable[i].duplicates = 0;
459  }
460 
461  //qDebug() << "Filling hashtable...";
462  for (auto it = d->stringlist.constBegin(); it != d->stringlist.constEnd(); ++it) {
463  string_entry *entry = *it;
464  //qDebug() << "entry keyStr=" << entry->keyStr << entry->payload.data() << entry->payload->entryPath();
465  int hash = entry->hash % sz;
466  if (!hashTable[hash].entry) {
467  // First entry
468  hashTable[hash].entry = entry;
469  } else {
470  if (!hashTable[hash].duplicates) {
471  // Second entry, build duplicate list.
472  hashTable[hash].duplicates = new QList<string_entry *>;
473  hashTable[hash].duplicates->append(hashTable[hash].entry);
474  hashTable[hash].duplicate_offset = 0;
475  }
476  hashTable[hash].duplicates->append(entry);
477  }
478  }
479 
480  str << d->hashTableSize;
481  str << d->hashList;
482 
483  d->offset = str.device()->pos(); // d->offset points to start of hashTable
484  //qDebug() << QString("Start of Hash Table, offset = %1").arg(d->offset,8,16);
485 
486  // Write the hashtable + the duplicates twice.
487  // The duplicates are after the normal hashtable, but the offset of each
488  // duplicate entry is written into the normal hashtable.
489  for (int pass = 1; pass <= 2; pass++) {
490  str.device()->seek(d->offset);
491  //qDebug() << QString("Writing hash table (pass #%1)").arg(pass);
492  for (uint i = 0; i < d->hashTableSize; i++) {
493  qint32 tmpid;
494  if (!hashTable[i].entry) {
495  tmpid = 0;
496  } else if (!hashTable[i].duplicates) {
497  tmpid = hashTable[i].entry->payload->offset(); // Positive ID
498  } else {
499  tmpid = - hashTable[i].duplicate_offset; // Negative ID
500  }
501  str << tmpid;
502  //qDebug() << QString("Hash table : %1").arg(tmpid,8,16);
503  }
504  //qDebug() << QString("End of Hash Table, offset = %1").arg(str.device()->at(),8,16);
505 
506  //qDebug() << QString("Writing duplicate lists (pass #%1)").arg(pass);
507  for (uint i = 0; i < d->hashTableSize; i++) {
508  const QList<string_entry *> *dups = hashTable[i].duplicates;
509  if (dups) {
510  hashTable[i].duplicate_offset = str.device()->pos();
511 
512  /*qDebug() << QString("Duplicate lists: Offset = %1 list_size = %2") .arg(hashTable[i].duplicate_offset,8,16).arg(dups->count());
513  */
514  Q_FOREACH (string_entry* dup, *dups) {
515  const qint32 offset = dup->payload->offset();
516  if (!offset) {
517  const QString storageId = dup->payload->storageId();
518  qDebug() << "about to assert! dict=" << this << "storageId=" << storageId << dup->payload.data();
519  if (dup->payload->isType(KST_KService)) {
520  KService::Ptr service(static_cast<KService*>(dup->payload.data()));
521  qDebug() << service->storageId() << service->entryPath();
522  }
523  // save() must have been called on the entry
524  Q_ASSERT_X(offset, "KSycocaDict::save",
525  QByteArray("entry offset is 0, save() was not called on "
526  + dup->payload->storageId().toLatin1()
527  + " entryPath="
528  + dup->payload->entryPath().toLatin1()).constData()
529  );
530  }
531  str << offset; // Positive ID
532  str << dup->keyStr; // Key (QString)
533  }
534  str << qint32(0); // End of list marker (0)
535  }
536  }
537  //qDebug() << QString("End of Dict, offset = %1").arg(str.device()->at(),8,16);
538  }
539 
540  //qDebug() << "Cleaning up hash table.";
541  for (uint i = 0; i < d->hashTableSize; i++) {
542  delete hashTable[i].duplicates;
543  }
544  delete [] hashTable;
545 }

547 qint32 KSycocaDictPrivate::offsetForKey(const QString &key) const
548 {
549  if (!stream || !offset) {
550  qCWarning(SYCOCA) << "No ksycoca database available! Tried running" << KBUILDSYCOCA_EXENAME << "?";
551  return 0;
552  }
553 
554  if (hashTableSize == 0) {
555  return 0; // Unlikely to find anything :-]
556  }
557 
558  // Read hash-table data
559  const uint hash = hashKey(key) % hashTableSize;
560  //qDebug() << "hash is" << hash;
561 
562  const qint32 off = offset + sizeof(qint32) * hash;
563  //qDebug() << QString("off is %1").arg(off,8,16);
564  stream->device()->seek(off);
