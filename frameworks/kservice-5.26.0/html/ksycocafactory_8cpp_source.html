52  : m_resourceList(0), m_entryDict(0), m_str(0), m_sycoca(sycoca), d(new KSycocaFactoryPrivate)
53 {
54  if (!m_sycoca->isBuilding() && (m_str = m_sycoca->findFactory(factory_id))) {
55  // Read position of index tables....

77  }

97 void
98 KSycocaFactory::save(QDataStream &str)
99 {
100  if (!m_entryDict) {
101  return; // Error! Function should only be called when
102  }
103  // building database
104  if (!d->m_sycocaDict) {
105  return; // Error!
106  }
107 
108  d->mOffset = str.device()->pos(); // store position in member variable
109  d->m_sycocaDictOffset = 0;
110 
111  // Write header (pass #1)
112  saveHeader(str);
113 
114  d->m_beginEntryOffset = str.device()->pos();
115 
116  // Write all entries.
117  int entryCount = 0;
118  Q_FOREACH(KSycocaEntry::Ptr entry, *m_entryDict) {
119  entry->d_ptr->save(str);
120  entryCount++;
121  }
122 
123  d->m_endEntryOffset = str.device()->pos();
124 
125  // Write indices...
126  // Linear index
127  str << qint32(entryCount);
128  Q_FOREACH(KSycocaEntry::Ptr entry, *m_entryDict) {
129  str << qint32(entry.data()->offset());
130  }
131 
132  // Dictionary index
133  d->m_sycocaDictOffset = str.device()->pos();
134  d->m_sycocaDict->save(str);
135 
136  int endOfFactoryData = str.device()->pos();
137 
138  // Update header (pass #2)
139  saveHeader(str);
140 
141  // Seek to end.
142  str.device()->seek(endOfFactoryData);
143 }

145 void
146 KSycocaFactory::addEntry(const KSycocaEntry::Ptr &newEntry)
147 {
148  if (!m_entryDict) {
149  return; // Error! Function should only be called when
150  }
151  // building database
152 
153  if (!d->m_sycocaDict) {
154  return; // Error!
155  }
156 
157  KSycocaEntry::Ptr oldEntry = m_entryDict->value(newEntry->storageId());
158  if (oldEntry) {
159  // Already exists -> replace
160  // We found a more-local override, e.g. ~/.local/share/applications/kde5/foo.desktop
161  // So forget about the more global file.
162  //
163  // This can also happen with two .protocol files using the same protocol= entry.
164  // If we didn't remove one here, we would end up asserting because save()
165  // wasn't called on one of the entries.
166  //qDebug() << "removing" << oldEntry.data() << oldEntry->entryPath() << "because of" << newEntry->entryPath() << "they have the same storageId" << newEntry->storageId();
167  removeEntry(newEntry->storageId());
168  }
169 
170  const QString name = newEntry->storageId();
171  m_entryDict->insert(name, newEntry);
172  d->m_sycocaDict->add(name, newEntry);
173 }

191 KSycocaEntry::List KSycocaFactory::allEntries() const
192 {
193  KSycocaEntry::List list;
194 
195  // Assume we're NOT building a database
196 
197  QDataStream *str = stream();
198  if (!str) {
199  return list;
200  }
201  str->device()->seek(d->m_endEntryOffset);
202  qint32 entryCount;
203  (*str) >> entryCount;
204 
205  if (entryCount > 8192) {
206  qCWarning(SYCOCA) << QThread::currentThread() << "error detected in factory" << this;
207  KSycoca::flagError();
208  return list;
209  }
210 
211  // offsetList is needed because createEntry() modifies the stream position
212  qint32 *offsetList = new qint32[entryCount];
213  for (int i = 0; i < entryCount; i++) {
214  (*str) >> offsetList[i];
215  }
216 
217  for (int i = 0; i < entryCount; i++) {
218  KSycocaEntry *newEntry = createEntry(offsetList[i]);
219  if (newEntry) {
220  list.append(KSycocaEntry::Ptr(newEntry));
221  }
222  }
223  delete [] offsetList;
224  return list;
225 }
