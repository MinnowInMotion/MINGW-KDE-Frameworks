74 bool KServiceOffer::operator< (const KServiceOffer &_o) const
75 {
76  // First check mimetype inheritance level.
77  // Direct mimetype association is preferred above association via parent mimetype
78  // So, the smaller the better.
79  if (d->mimeTypeInheritanceLevel != _o.d->mimeTypeInheritanceLevel) {
80  return d->mimeTypeInheritanceLevel < _o.d->mimeTypeInheritanceLevel;
81  }
82 
83  // Put offers allowed as default FIRST.
84  if (_o.d->bAllowAsDefault && !d->bAllowAsDefault) {
85  return false; // _o is default and not 'this'.
86  }
87  if (!_o.d->bAllowAsDefault && d->bAllowAsDefault) {
88  return true; // 'this' is default but not _o.
89  }
90  // Both offers are allowed or not allowed as default
91 
92  // Finally, use preference to sort them
93  // The bigger the better, but we want the better FIRST
94  return _o.d->preference < d->preference;
95 }
