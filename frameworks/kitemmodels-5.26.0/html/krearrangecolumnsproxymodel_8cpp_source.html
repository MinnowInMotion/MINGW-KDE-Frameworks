50 int KRearrangeColumnsProxyModel::rowCount(const QModelIndex &parent) const
51 {
52  Q_ASSERT(parent.isValid() ? parent.model() == this : true);
53  // The parent in the source model is on column 0, whatever swapping we are doing
54  const QModelIndex sourceParent = mapToSource(parent).sibling(parent.row(), 0);
55  return sourceModel()->rowCount(sourceParent);
56 }

61 QModelIndex KRearrangeColumnsProxyModel::index(int row, int column, const QModelIndex &parent) const
62 {
63  Q_ASSERT(parent.isValid() ? parent.model() == this : true);
64  Q_ASSERT(row >= 0);
65  Q_ASSERT(column >= 0);
66 
67  // The parent in the source model is on column 0, whatever swapping we are doing
68  const QModelIndex sourceParent = mapToSource(parent).sibling(parent.row(), 0);
69 
70  // Find the child in the source model, we need its internal pointer
71  const QModelIndex sourceIndex = sourceModel()->index(row, sourceColumnForProxyColumn(column), sourceParent);
72  if (!sourceIndex.isValid()) {
73  return QModelIndex();
74  }
75 
76  return createIndex(row, column, sourceIndex.internalPointer());
77 }

100 QModelIndex KRearrangeColumnsProxyModel::mapFromSource(const QModelIndex &sourceIndex) const
101 {
102  if (!sourceIndex.isValid()) {
103  return QModelIndex();
104  }
105  Q_ASSERT(sourceIndex.model() == sourceModel());
106  const int proxyColumn = proxyColumnForSourceColumn(sourceIndex.column());
107  return createIndex(sourceIndex.row(), proxyColumn, sourceIndex.internalPointer());
108 }

110 QModelIndex KRearrangeColumnsProxyModel::mapToSource(const QModelIndex &proxyIndex) const
111 {
112  if (!proxyIndex.isValid()) {
113  return QModelIndex();
114  }
115  // This is just an indirect way to call sourceModel->createIndex(row, sourceColumn, pointer)
116  const QModelIndex fakeIndex = createIndex(proxyIndex.row(), sourceColumnForProxyColumn(proxyIndex.column()), proxyIndex.internalPointer());
117  return QIdentityProxyModel::mapToSource(fakeIndex);
118 }
