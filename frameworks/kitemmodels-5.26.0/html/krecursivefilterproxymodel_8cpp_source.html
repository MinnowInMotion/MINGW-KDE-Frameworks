24 // Maintainability note:
25 // This class invokes some Q_PRIVATE_SLOTs in QSortFilterProxyModel which are
26 // private API and could be renamed or removed at any time.
27 // If they are renamed, the invocations can be updated with an #if (QT_VERSION(...))
28 // If they are removed, then layout{AboutToBe}Changed Q_SIGNALS should be used when the source model
29 // gets new rows or has rowsremoved or moved. The Q_PRIVATE_SLOT invocation is an optimization
30 // because layout{AboutToBe}Changed is expensive and causes the entire mapping of the tree in QSFPM
31 // to be cleared, even if only a part of it is dirty.
32 // Stephen Kelly, 30 April 2010.

34 inline static bool passRolesToDataChanged()
35 {
36  // runtime check to ensure a KF5 built against Qt < 5.5 keeps working when Qt is updated to 5.5 and above but KF5 is not rebuild

152 void KRecursiveFilterProxyModelPrivate::sourceDataChanged(const QModelIndex &source_top_left, const QModelIndex &source_bottom_right, const QVector<int> &roles)
153 {
154  QModelIndex source_parent = source_top_left.parent();
155  Q_ASSERT(source_bottom_right.parent() == source_parent); // don't know how to handle different parents in this code...
156 
157  // Tell the world.
158  invokeDataChanged(source_top_left, source_bottom_right, roles);
159 
160  // We can't find out if the change really matters to us or not, for a lack of a dataAboutToBeChanged signal (or a cache).
161  // TODO: add a set of roles that we care for, so we can at least ignore the rest.
162 
163  // Even if we knew the visibility was just toggled, we also can't find out what
164  // was the last filtered out ascendant (on show, like sourceRowsAboutToBeInserted does)
165  // or the last to-be-filtered-out ascendant (on hide, like sourceRowsRemoved does)
166  // So we have to refresh all parents.
167  QModelIndex sourceParent = source_parent;
168  while (sourceParent.isValid()) {
169  invokeDataChanged(sourceParent, sourceParent, roles);
170  sourceParent = sourceParent.parent();
171  }
172 }

174 QModelIndex KRecursiveFilterProxyModelPrivate::lastFilteredOutAscendant(const QModelIndex &idx)
175 {
176  Q_Q(KRecursiveFilterProxyModel);

178  QModelIndex index = idx.parent();
179  while (index.isValid() && !q->filterAcceptsRow(index.row(), index.parent())) {
180  last = index;
181  index = index.parent();

186 void KRecursiveFilterProxyModelPrivate::sourceRowsAboutToBeInserted(const QModelIndex &source_parent, int start, int end)
187 {
188  Q_Q(KRecursiveFilterProxyModel);
189 
190  if (!source_parent.isValid() || q->filterAcceptsRow(source_parent.row(), source_parent.parent())) {
191  // If the parent is already in the model (directly or indirectly), we can just pass on the signal.
192  invokeRowsAboutToBeInserted(source_parent, start, end);
193  completeInsert = true;
194  } else {
195  // OK, so parent is not in the model.
196  // Maybe the grand parent neither.. Go up until the first one that is.
197  lastHiddenAscendantForInsert = lastFilteredOutAscendant(source_parent);
198  }
199 }

201 void KRecursiveFilterProxyModelPrivate::sourceRowsInserted(const QModelIndex &source_parent, int start, int end)
202 {
203  Q_Q(KRecursiveFilterProxyModel);
204 
205  if (completeInsert) {
206  // If the parent is already in the model, we can just pass on the signal.
207  completeInsert = false;
208  invokeRowsInserted(source_parent, start, end);
209  return;
210  }
211 
212  bool requireRow = false;
213  for (int row = start; row <= end; ++row) {
214  if (q->filterAcceptsRow(row, source_parent)) {
215  requireRow = true;
216  break;
217  }
218  }
219 
220  if (!requireRow) {
221  // The new rows doesn't have any descendants that match the filter. Filter them out.
222  return;
223  }
224 
225  // Make QSFPM realize that lastHiddenAscendantForInsert should be shown now
226  invokeDataChanged(lastHiddenAscendantForInsert, lastHiddenAscendantForInsert);
227 }

234 void KRecursiveFilterProxyModelPrivate::sourceRowsRemoved(const QModelIndex &source_parent, int start, int end)
235 {
236  Q_Q(KRecursiveFilterProxyModel);
237 
238  invokeRowsRemoved(source_parent, start, end);
239 
240  // Find out if removing this visible row means that some ascendant
241  // row can now be hidden.
242  // We go up until we find a row that should still be visible
243  // and then make QSFPM re-evaluate the last one we saw before that, to hide it.
244 
245  QModelIndex toHide;
246  QModelIndex sourceAscendant = source_parent;
247  while (sourceAscendant.isValid()) {
248  if (q->filterAcceptsRow(sourceAscendant.row(), sourceAscendant.parent())) {
249  break;
250  }
251  toHide = sourceAscendant;
252  sourceAscendant = sourceAscendant.parent();
253  }
254  if (toHide.isValid()) {
255  invokeDataChanged(toHide, toHide);
256  }
257 }

270 bool KRecursiveFilterProxyModel::filterAcceptsRow(int sourceRow, const QModelIndex &sourceParent) const
271 {
272  // TODO: Implement some caching so that if one match is found on the first pass, we can return early results
273  // when the subtrees are checked by QSFPM.
274  if (acceptRow(sourceRow, sourceParent)) {
275  return true;
276  }
277 
278  QModelIndex source_index = sourceModel()->index(sourceRow, 0, sourceParent);
279  Q_ASSERT(source_index.isValid());
280  bool accepted = false;
281 
282  const int numChildren = sourceModel()->rowCount(source_index);
283  for (int row = 0, rows = numChildren; row < rows; ++row) {
284  if (filterAcceptsRow(row, source_index)) {
285  accepted = true;
286  break;
287  }
288  }
289 
290  return accepted;
291 }

293 QModelIndexList KRecursiveFilterProxyModel::match(const QModelIndex &start, int role, const QVariant &value, int hits, Qt::MatchFlags flags) const
294 {
295  if (role < Qt::UserRole) {
296  return QSortFilterProxyModel::match(start, role, value, hits, flags);
297  }

299  QModelIndexList list;
300  if (!sourceModel())
301  return list;
302 
303  QModelIndex proxyIndex;
304  Q_FOREACH (const QModelIndex &idx, sourceModel()->match(mapToSource(start), role, value, hits, flags)) {
305  proxyIndex = mapFromSource(idx);
306  if (proxyIndex.isValid()) {
307  list << proxyIndex;
308  }
309  }
310

346  // Disconnect in the QSortFilterProxyModel. These methods will be invoked manually
347  // in invokeDataChanged, invokeRowsInserted etc.
348  //
349  // The reason for that is that when the source model adds new rows for example, the new rows
350  // May not match the filter, but maybe their child items do match.
351  //
352  // Source model before insert:
353  //
354  // - A
355  // - B
356  // - - C
357  // - - D
358  // - - - E
359  // - - - F
360  // - - - G
361  // - H
362  // - I
363  //
364  // If the A F and L (which doesn't exist in the source model yet) match the filter
365  // the proxy will be:
366  //
367  // - A
368  // - B
369  // - - D
370  // - - - F
371  //
372  // New rows are inserted in the source model below H:
373  //
374  // - A
375  // - B
376  // - - C
377  // - - D
378  // - - - E
379  // - - - F
380  // - - - G
381  // - H
382  // - - J
383  // - - K
384  // - - - L
385  // - I
386  //
387  // As L matches the filter, it should be part of the KRecursiveFilterProxyModel.
388  //
389  // - A
390  // - B
391  // - - D
392  // - - - F
393  // - H
394  // - - K
395  // - - - L
396  //
397  // when the QSortFilterProxyModel gets a notification about new rows in H, it only checks
398  // J and K to see if they match, ignoring L, and therefore not adding it to the proxy.
399  // To work around that, we make sure that the QSFPM slot which handles that change in
400  // the source model (_q_sourceRowsAboutToBeInserted) does not get called directly.
401  // Instead we connect the sourceModel signal to our own slot in *this (sourceRowsAboutToBeInserted)
402  // Inside that method, the entire new subtree is queried (J, K *and* L) to see if there is a match,
403  // then the relevant Q_SLOTS in QSFPM are invoked.
404  // In the example above, we need to tell the QSFPM that H should be queried again to see if
405  // it matches the filter. It did not before, because L did not exist before. Now it does. That is
406  // achieved by telling the QSFPM that the data changed for H, which causes it to requery this class
407  // to see if H matches the filter (which it now does as L now exists).
408  // That is done in sourceRowsInserted.
409 
410  if (!model) {
411  return;
412  }
413 
414  if (passRolesToDataChanged()) {
415  disconnect(model, SIGNAL(dataChanged(QModelIndex,QModelIndex,QVector<int>)),
416  this, SLOT(_q_sourceDataChanged(QModelIndex,QModelIndex,QVector<int>)));
417  } else {
418  disconnect(model, SIGNAL(dataChanged(QModelIndex,QModelIndex)),
419  this, SLOT(_q_sourceDataChanged(QModelIndex,QModelIndex)));
420  }
421 
422  disconnect(model, SIGNAL(rowsAboutToBeInserted(QModelIndex,int,int)),
423  this, SLOT(_q_sourceRowsAboutToBeInserted(QModelIndex,int,int)));
424 
425  disconnect(model, SIGNAL(rowsInserted(QModelIndex,int,int)),
426  this, SLOT(_q_sourceRowsInserted(QModelIndex,int,int)));
427 
428  disconnect(model, SIGNAL(rowsAboutToBeRemoved(QModelIndex,int,int)),
429  this, SLOT(_q_sourceRowsAboutToBeRemoved(QModelIndex,int,int)));
430 
431  disconnect(model, SIGNAL(rowsRemoved(QModelIndex,int,int)),
432  this, SLOT(_q_sourceRowsRemoved(QModelIndex,int,int)));
433 
434  // Slots for manual invoking of QSortFilterProxyModel methods.
435  if (passRolesToDataChanged()) {
436  connect(model, SIGNAL(dataChanged(QModelIndex,QModelIndex,QVector<int>)),
437  this, SLOT(sourceDataChanged(QModelIndex,QModelIndex,QVector<int>)));
438  } else {
439  connect(model, SIGNAL(dataChanged(QModelIndex,QModelIndex)),
440  this, SLOT(sourceDataChanged(QModelIndex,QModelIndex)));
441  }
442 
443  connect(model, SIGNAL(rowsAboutToBeInserted(QModelIndex,int,int)),
444  this, SLOT(sourceRowsAboutToBeInserted(QModelIndex,int,int)));
445
