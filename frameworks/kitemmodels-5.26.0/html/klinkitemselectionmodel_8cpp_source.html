114 void KLinkItemSelectionModel::setLinkedItemSelectionModel(QItemSelectionModel *selectionModel)
115 {
116  Q_D(KLinkItemSelectionModel);
117  if (d->m_linkedItemSelectionModel != selectionModel) {
118 
119  if (d->m_linkedItemSelectionModel) {
120  disconnect(d->m_linkedItemSelectionModel);
121  }
122 
123  d->m_linkedItemSelectionModel = selectionModel;
124 
125  if (d->m_linkedItemSelectionModel) {
126  connect(d->m_linkedItemSelectionModel, SIGNAL(selectionChanged(QItemSelection,QItemSelection)), SLOT(sourceSelectionChanged(QItemSelection,QItemSelection)));
127  connect(d->m_linkedItemSelectionModel, SIGNAL(currentChanged(QModelIndex,QModelIndex)), SLOT(sourceCurrentChanged(QModelIndex)));
128 
129 #if QT_VERSION >= QT_VERSION_CHECK(5, 5, 0)
130  connect(d->m_linkedItemSelectionModel, &QItemSelectionModel::modelChanged, this, [this] {
131  d_ptr->reinitializeIndexMapper();

134  }
135  d->reinitializeIndexMapper();
136  Q_EMIT linkedItemSelectionModelChanged();
137  }

140 void KLinkItemSelectionModel::select(const QModelIndex &index, QItemSelectionModel::SelectionFlags command)
141 {
142  Q_D(KLinkItemSelectionModel);
143  // When an item is removed, the current index is set to the top index in the model.
144  // That causes a selectionChanged signal with a selection which we do not want.
145  if (d->m_ignoreCurrentChanged) {
146  return;
147  }
148  // Do *not* replace next line with: QItemSelectionModel::select(index, command)
149  //
150  // Doing so would end up calling KLinkItemSelectionModel::select(QItemSelection, QItemSelectionModel::SelectionFlags)
151  //
152  // This is because the code for QItemSelectionModel::select(QModelIndex, QItemSelectionModel::SelectionFlags) looks like this:
153  // {
154  // QItemSelection selection(index, index);
155  // select(selection, command);
156  // }
157  // So it calls KLinkItemSelectionModel overload of
158  // select(QItemSelection, QItemSelectionModel::SelectionFlags)
159  //
160  // When this happens and the selection flags include Toggle, it causes the
161  // selection to be toggled twice.
162  QItemSelectionModel::select(QItemSelection(index, index), command);
163  if (index.isValid()) {
164  d->m_linkedItemSelectionModel->select(d->m_indexMapper->mapSelectionLeftToRight(QItemSelection(index, index)), command);
165  } else {
166  d->m_linkedItemSelectionModel->clearSelection();
167  }
168 }

170 // QAbstractProxyModel::mapSelectionFromSource creates invalid ranges to we filter
171 // those out manually in a loop. Hopefully fixed in Qt 4.7.2, so we ifdef it out.
172 // http://qt.gitorious.org/qt/qt/merge_requests/2474
173 // http://qt.gitorious.org/qt/qt/merge_requests/831
174 #if QT_VERSION < 0x040702
175 #define RANGE_FIX_HACK
176 #endif
177 
178 #ifdef RANGE_FIX_HACK
179 static QItemSelection klink_removeInvalidRanges(const QItemSelection &selection)
180 {
181  QItemSelection result;
182  Q_FOREACH (const QItemSelectionRange &range, selection) {
183  if (!range.isValid()) {
184  continue;
185  }
186  result << range;
187  }
188  return result;
189 }
190 #endif

192 void KLinkItemSelectionModel::select(const QItemSelection &selection, QItemSelectionModel::SelectionFlags command)
193 {
194  Q_D(KLinkItemSelectionModel);
195  d->m_ignoreCurrentChanged = true;
196 #ifdef RANGE_FIX_HACK
197  QItemSelection _selection = klink_removeInvalidRanges(selection);
198 #else
199  QItemSelection _selection = selection;
200 #endif
201  QItemSelectionModel::select(_selection, command);
202  Q_ASSERT(d->assertSelectionValid(_selection));
203  QItemSelection mappedSelection = d->m_indexMapper->mapSelectionLeftToRight(_selection);
204  Q_ASSERT(d->assertSelectionValid(mappedSelection));
205  d->m_linkedItemSelectionModel->select(mappedSelection, command);
206  d->m_ignoreCurrentChanged = false;
207 }

218 void KLinkItemSelectionModelPrivate::sourceSelectionChanged(const QItemSelection &selected, const QItemSelection &deselected)
219 {
220  Q_Q(KLinkItemSelectionModel);
221 #ifdef RANGE_FIX_HACK
222  QItemSelection _selected = klink_removeInvalidRanges(selected);
223  QItemSelection _deselected = klink_removeInvalidRanges(deselected);
224 #else
225  QItemSelection _selected = selected;
226  QItemSelection _deselected = deselected;
227 #endif
228  Q_ASSERT(assertSelectionValid(_selected));
229  Q_ASSERT(assertSelectionValid(_deselected));
230  const QItemSelection mappedDeselection = m_indexMapper->mapSelectionRightToLeft(_deselected);
231  const QItemSelection mappedSelection = m_indexMapper->mapSelectionRightToLeft(_selected);
232 
233  q->QItemSelectionModel::select(mappedDeselection, QItemSelectionModel::Deselect);
234  q->QItemSelectionModel::select(mappedSelection, QItemSelectionModel::Select);
235 }
