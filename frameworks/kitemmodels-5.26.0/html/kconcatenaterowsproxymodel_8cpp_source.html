74 QModelIndex KConcatenateRowsProxyModel::mapToSource(const QModelIndex &proxyIndex) const
75 {
76  if (!proxyIndex.isValid()) {

173 void KConcatenateRowsProxyModel::addSourceModel(QAbstractItemModel *sourceModel)
174 {
175  Q_ASSERT(sourceModel);
176  Q_ASSERT(!d->m_models.contains(sourceModel));
177  connect(sourceModel, SIGNAL(dataChanged(QModelIndex,QModelIndex,QVector<int>)), this, SLOT(slotDataChanged(QModelIndex,QModelIndex,QVector<int>)));
178  connect(sourceModel, SIGNAL(rowsInserted(QModelIndex,int,int)), this, SLOT(slotRowsInserted(QModelIndex,int,int)));
179  connect(sourceModel, SIGNAL(rowsRemoved(QModelIndex,int,int)), this, SLOT(slotRowsRemoved(QModelIndex,int,int)));
180  connect(sourceModel, SIGNAL(rowsAboutToBeInserted(QModelIndex,int,int)), this, SLOT(slotRowsAboutToBeInserted(QModelIndex,int,int)));
181  connect(sourceModel, SIGNAL(rowsAboutToBeRemoved(QModelIndex,int,int)), this, SLOT(slotRowsAboutToBeRemoved(QModelIndex,int,int)));
182 
183  connect(sourceModel, SIGNAL(columnsInserted(QModelIndex,int,int)), this, SLOT(slotColumnsInserted(QModelIndex,int,int)));
184  connect(sourceModel, SIGNAL(columnsRemoved(QModelIndex,int,int)), this, SLOT(slotColumnsRemoved(QModelIndex,int,int)));
185  connect(sourceModel, SIGNAL(columnsAboutToBeInserted(QModelIndex,int,int)), this, SLOT(slotColumnsAboutToBeInserted(QModelIndex,int,int)));
186  connect(sourceModel, SIGNAL(columnsAboutToBeRemoved(QModelIndex,int,int)), this, SLOT(slotColumnsAboutToBeRemoved(QModelIndex,int,int)));
187 
188  connect(sourceModel, SIGNAL(layoutAboutToBeChanged(QList<QPersistentModelIndex>, QAbstractItemModel::LayoutChangeHint)),
189  this, SLOT(slotSourceLayoutAboutToBeChanged(QList<QPersistentModelIndex>, QAbstractItemModel::LayoutChangeHint)));
190  connect(sourceModel, SIGNAL(layoutChanged(QList<QPersistentModelIndex>, QAbstractItemModel::LayoutChangeHint)),
191  this, SLOT(slotSourceLayoutChanged(QList<QPersistentModelIndex>, QAbstractItemModel::LayoutChangeHint)));
192  connect(sourceModel, SIGNAL(modelAboutToBeReset()), this, SLOT(slotModelAboutToBeReset()));
193  connect(sourceModel, SIGNAL(modelReset()), this, SLOT(slotModelReset()));
194 
195  const int newRows = sourceModel->rowCount();
196  if (newRows > 0) {
197  beginInsertRows(QModelIndex(), d->m_rowCount, d->m_rowCount + newRows - 1);
198  }
199  d->m_rowCount += newRows;
200  d->m_models.append(sourceModel);
201  if (newRows > 0) {
202  endInsertRows();
203  }
204 }

206 void KConcatenateRowsProxyModel::removeSourceModel(QAbstractItemModel *sourceModel)
207 {
208  Q_ASSERT(d->m_models.contains(sourceModel));
209  disconnect(sourceModel, 0, this, 0);
210 
211  const int rowsRemoved = sourceModel->rowCount();
212  const int rowsPrior = d->computeRowsPrior(sourceModel); // location of removed section
213 
214  if (rowsRemoved > 0) {
215  beginRemoveRows(QModelIndex(), rowsPrior, rowsPrior + rowsRemoved - 1);
216  }
217  d->m_models.removeOne(sourceModel);
218  d->m_rowCount -= rowsRemoved;
219  if (rowsRemoved > 0) {
220  endRemoveRows();
221  }
222 }

304 void KConcatenateRowsProxyModelPrivate::slotSourceLayoutAboutToBeChanged(const QList<QPersistentModelIndex> &sourceParents, QAbstractItemModel::LayoutChangeHint hint)
305 {
306  const QModelIndexList persistentIndexList = q->persistentIndexList();
307  layoutChangePersistentIndexes.reserve(persistentIndexList.size());
308 
309  foreach (const QPersistentModelIndex &proxyPersistentIndex, persistentIndexList) {
310  proxyIndexes << proxyPersistentIndex;
311  Q_ASSERT(proxyPersistentIndex.isValid());
312  const QPersistentModelIndex srcPersistentIndex = q->mapToSource(proxyPersistentIndex);
313  Q_ASSERT(srcPersistentIndex.isValid());
314  layoutChangePersistentIndexes << srcPersistentIndex;
315  }
316 
317  QList<QPersistentModelIndex> parents;
318  parents.reserve(sourceParents.size());
319  foreach (const QPersistentModelIndex &parent, sourceParents) {
320  if (!parent.isValid()) {
321  parents << QPersistentModelIndex();
322  continue;
323  }
324  const QModelIndex mappedParent = q->mapFromSource(parent);
325  Q_ASSERT(mappedParent.isValid());
326  parents << mappedParent;
327  }
328 
329  emit q->layoutAboutToBeChanged(parents, hint);
330 }

332 void KConcatenateRowsProxyModelPrivate::slotSourceLayoutChanged(const QList<QPersistentModelIndex> &sourceParents, QAbstractItemModel::LayoutChangeHint hint)
333 {
334  for (int i = 0; i < proxyIndexes.size(); ++i) {
335  const QModelIndex proxyIdx = proxyIndexes.at(i);
336  QModelIndex newProxyIdx = q->mapFromSource(layoutChangePersistentIndexes.at(i));
337  q->changePersistentIndex(proxyIdx, newProxyIdx);
338  }
339 
340  layoutChangePersistentIndexes.clear();
341  proxyIndexes.clear();
342 
343  QList<QPersistentModelIndex> parents;
344  parents.reserve(sourceParents.size());
345  foreach (const QPersistentModelIndex &parent, sourceParents) {
346  if (!parent.isValid()) {
347  parents << QPersistentModelIndex();
348  continue;
349  }
350  const QModelIndex mappedParent = q->mapFromSource(parent);
351  Q_ASSERT(mappedParent.isValid());
352  parents << mappedParent;
353  }
354  emit q->layoutChanged(parents, hint);
355 }

385 int KConcatenateRowsProxyModelPrivate::computeRowsPrior(const QAbstractItemModel *sourceModel) const
386 {
387  int rowsPrior = 0;
388  foreach (const QAbstractItemModel *model, m_models) {
389  if (model == sourceModel) {
390  break;
391  }
392  rowsPrior += model->rowCount();
393  }
394  return rowsPrior;
395 }

397 QAbstractItemModel *KConcatenateRowsProxyModelPrivate::sourceModelForRow(int row, int *sourceRow) const
398 {
399  int rowCount = 0;
400  QAbstractItemModel *selection = NULL;
401  foreach (QAbstractItemModel *model, m_models) {
402  const int subRowCount = model->rowCount();
403  if (rowCount + subRowCount > row) {
404  selection = model;
405  break;
406  }
407  rowCount += subRowCount;
408  }
409  *sourceRow = row - rowCount;
410  return selection;
411 }
