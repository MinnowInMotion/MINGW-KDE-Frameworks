86 void KExtraColumnsProxyModel::setSourceModel(QAbstractItemModel *model)
87 {
88  if (sourceModel()) {
89  disconnect(sourceModel(), SIGNAL(layoutAboutToBeChanged(QList<QPersistentModelIndex>,QAbstractItemModel::LayoutChangeHint)),
90  this, SLOT(_ec_sourceLayoutAboutToBeChanged(QList<QPersistentModelIndex>,QAbstractItemModel::LayoutChangeHint)));
91  disconnect(sourceModel(), SIGNAL(layoutChanged(QList<QPersistentModelIndex>,QAbstractItemModel::LayoutChangeHint)),
92  this, SLOT(_ec_sourceLayoutChanged(QList<QPersistentModelIndex>,QAbstractItemModel::LayoutChangeHint)));
93  }
94 
95  QIdentityProxyModel::setSourceModel(model);
96 
97  if (model) {
98  // The handling of persistent model indexes assumes mapToSource can be called for any index
99  // This breaks for the extra column, so we'll have to do it ourselves
100  disconnect(model, SIGNAL(layoutAboutToBeChanged(QList<QPersistentModelIndex>,QAbstractItemModel::LayoutChangeHint)),
101  this, SLOT(_q_sourceLayoutAboutToBeChanged(QList<QPersistentModelIndex>,QAbstractItemModel::LayoutChangeHint)));
102  disconnect(model, SIGNAL(layoutChanged(QList<QPersistentModelIndex>,QAbstractItemModel::LayoutChangeHint)),
103  this, SLOT(_q_sourceLayoutChanged(QList<QPersistentModelIndex>,QAbstractItemModel::LayoutChangeHint)));
104  connect(model, SIGNAL(layoutAboutToBeChanged(QList<QPersistentModelIndex>,QAbstractItemModel::LayoutChangeHint)),
105  this, SLOT(_ec_sourceLayoutAboutToBeChanged(QList<QPersistentModelIndex>,QAbstractItemModel::LayoutChangeHint)));
106  connect(model, SIGNAL(layoutChanged(QList<QPersistentModelIndex>,QAbstractItemModel::LayoutChangeHint)),
107  this, SLOT(_ec_sourceLayoutChanged(QList<QPersistentModelIndex>,QAbstractItemModel::LayoutChangeHint)));
108  }
109 }

111 QModelIndex KExtraColumnsProxyModel::mapToSource(const QModelIndex &proxyIndex) const
112 {
113  if (!proxyIndex.isValid()) { // happens in e.g. rowCount(mapToSource(parent))
114  return QModelIndex();
115  }
116  const int column = proxyIndex.column();
117  if (column >= sourceModel()->columnCount()) {
118  qDebug() << "Returning invalid index in mapToSource";
119  return QModelIndex();
120  }
121  return QIdentityProxyModel::mapToSource(proxyIndex);
122 }

141 QItemSelection KExtraColumnsProxyModel::mapSelectionToSource(const QItemSelection &selection) const
142 {
143  QItemSelection sourceSelection;
144 
145  if (!sourceModel()) {
146  return sourceSelection;
147  }
148 
149  // mapToSource will give invalid index for our additional columns, so truncate the selection
150  // to the columns known by the source model
151  const int sourceColumnCount = sourceModel()->columnCount();
152  QItemSelection::const_iterator it = selection.constBegin();
153  const QItemSelection::const_iterator end = selection.constEnd();
154  for (; it != end; ++it) {
155  Q_ASSERT(it->model() == this);
156  QModelIndex topLeft = it->topLeft();
157  Q_ASSERT(topLeft.isValid());
158  Q_ASSERT(topLeft.model() == this);
159  topLeft = topLeft.sibling(topLeft.row(), 0);
160  QModelIndex bottomRight = it->bottomRight();
161  Q_ASSERT(bottomRight.isValid());
162  Q_ASSERT(bottomRight.model() == this);
163  if (bottomRight.column() >= sourceColumnCount) {
164  bottomRight = bottomRight.sibling(bottomRight.row(), sourceColumnCount - 1);
165  }
166  // This can lead to duplicate source indexes, so use merge().
167  const QItemSelectionRange range(mapToSource(topLeft), mapToSource(bottomRight));
168  QItemSelection newSelection; newSelection << range;
169  sourceSelection.merge(newSelection, QItemSelectionModel::Select);
170  }
171 
172  return sourceSelection;
173 }

271 void KExtraColumnsProxyModelPrivate::_ec_sourceLayoutAboutToBeChanged(const QList<QPersistentModelIndex> &sourceParents, QAbstractItemModel::LayoutChangeHint hint)
272 {
273  Q_Q(KExtraColumnsProxyModel);
274  const QModelIndexList persistentIndexList = q->persistentIndexList();
275  layoutChangePersistentIndexes.reserve(persistentIndexList.size());
276  layoutChangeProxyColumns.reserve(persistentIndexList.size());
277 
278  foreach (QPersistentModelIndex proxyPersistentIndex, persistentIndexList) {
279  proxyIndexes << proxyPersistentIndex;
280  Q_ASSERT(proxyPersistentIndex.isValid());
281  const int column = proxyPersistentIndex.column();
282  layoutChangeProxyColumns << column;
283  if (column >= q->sourceModel()->columnCount()) {
284  proxyPersistentIndex = proxyPersistentIndex.sibling(proxyPersistentIndex.row(), 0);
285  }
286  const QPersistentModelIndex srcPersistentIndex = q->mapToSource(proxyPersistentIndex);
287  Q_ASSERT(srcPersistentIndex.isValid());
288  layoutChangePersistentIndexes << srcPersistentIndex;
289  }
290 
291  QList<QPersistentModelIndex> parents;
292  parents.reserve(sourceParents.size());
293  foreach (const QPersistentModelIndex &parent, sourceParents) {
294  if (!parent.isValid()) {
295  parents << QPersistentModelIndex();
296  continue;
297  }
298  const QModelIndex mappedParent = q->mapFromSource(parent);
299  Q_ASSERT(mappedParent.isValid());
300  parents << mappedParent;
301  }
302 
303  emit q->layoutAboutToBeChanged(parents, hint);
304 }

306 void KExtraColumnsProxyModelPrivate::_ec_sourceLayoutChanged(const QList<QPersistentModelIndex> &sourceParents, QAbstractItemModel::LayoutChangeHint hint)
307 {
308  Q_Q(KExtraColumnsProxyModel);
309  for (int i = 0; i < proxyIndexes.size(); ++i) {
310  const QModelIndex proxyIdx = proxyIndexes.at(i);
311  QModelIndex newProxyIdx = q->mapFromSource(layoutChangePersistentIndexes.at(i));
312  if (proxyIdx.column() >= q->sourceModel()->columnCount()) {
313  newProxyIdx = newProxyIdx.sibling(newProxyIdx.row(), layoutChangeProxyColumns.at(i));
314  }
315  q->changePersistentIndex(proxyIdx, newProxyIdx);
316  }
317 
318  layoutChangePersistentIndexes.clear();
319  layoutChangeProxyColumns.clear();
320  proxyIndexes.clear();
321 
322  QList<QPersistentModelIndex> parents;
323  parents.reserve(sourceParents.size());
324  foreach (const QPersistentModelIndex &parent, sourceParents) {
325  if (!parent.isValid()) {
326  parents << QPersistentModelIndex();
327  continue;
328  }
329  const QModelIndex mappedParent = q->mapFromSource(parent);
330  Q_ASSERT(mappedParent.isValid());
331  parents << mappedParent;
332  }
333 
334  emit q->layoutChanged(parents, hint);
335 }
