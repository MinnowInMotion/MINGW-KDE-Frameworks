81 QVariant ServiceModel::data(const QModelIndex &index, int role) const
82 {
83  if (!index.isValid()) {
84  return QVariant();
85  }
86  if (!hasIndex(index.row(), index.column(), index.parent())) {
87  return QVariant();
88  }
89  const QList<RemoteService::Ptr> srv = d->m_browser->services();
90  switch ((uint)role) {
91  case Qt::DisplayRole:
92  switch (index.column()) {
93  case ServiceName: return srv[index.row()]->serviceName();
94  case Host: return srv[index.row()]->hostName();
95  case Port: return srv[index.row()]->port();
96  }
97  case ServicePtrRole: QVariant ret;
98  ret.setValue(srv[index.row()]);
99  return ret;
100  }
101  return QVariant();
102 }
