156 bool PublicServicePrivate::fillEntryGroup()
157 {
158  registerTypes();
159  if (!m_group) {
160  QDBusReply<QDBusObjectPath> rep = m_server->EntryGroupNew();
161  if (!rep.isValid()) {
162  return false;
163  }
164  m_group = new org::freedesktop::Avahi::EntryGroup("org.freedesktop.Avahi", rep.value().path(), QDBusConnection::systemBus());
165  connect(m_group, SIGNAL(StateChanged(int,QString)), this, SLOT(groupStateChanged(int,QString)));
166  }
167  if (m_serviceName.isNull()) {
168  QDBusReply<QString> rep = m_server->GetHostName();
169  if (!rep.isValid()) {
170  return false;
171  }
172  m_serviceName = rep.value();
173  }
174 
175  QList<QByteArray> txt;
176  QMap<QString, QByteArray>::ConstIterator itEnd = m_textData.constEnd();
177  for (QMap<QString, QByteArray>::ConstIterator it = m_textData.constBegin(); it != itEnd; ++it)
178  if (it.value().isNull()) {
179  txt.append(it.key().toLatin1());
180  } else {
181  txt.append(it.key().toLatin1() + '=' + it.value());
182  }
183 
184  for (;;) {
185  QDBusReply<void> ret = m_group->AddService(-1, -1, 0, m_serviceName, m_type, domainToDNS(m_domain),
186  m_hostName, m_port, txt);
187  if (ret.isValid()) {
188  break;
189  }
190 
191  // serious error, bail out
192  if (ret.error().name() != QLatin1String("org.freedesktop.Avahi.CollisionError")) {
193  return false;
194  }
195 
196  // name collision, try another
197  QDBusReply<QString> rep = m_server->GetAlternativeServiceName(m_serviceName);
198  if (rep.isValid()) {
199  m_serviceName = rep.value();
200  } else {
201  return false;
202  }
203  }
204 
205  Q_FOREACH (const QString &subtype, m_subtypes) {
206  m_group->AddServiceSubtype(-1, -1, 0, m_serviceName, m_type, domainToDNS(m_domain), subtype);
207  }
208  return true;
209 }

211 void PublicServicePrivate::serverStateChanged(int s, const QString &)
212 {
213  if (!m_running) {
214  return;
215  }
216  switch (s) {
217  case AVAHI_SERVER_INVALID:
218  m_parent->stop();
219  emit m_parent->published(false);
220  break;
221  case AVAHI_SERVER_REGISTERING:
222  case AVAHI_SERVER_COLLISION:
223  if (m_group) {
224  m_group->Reset();
225  }
226  m_collision = true;
227  break;
228  case AVAHI_SERVER_RUNNING:
229  if (m_collision) {
230  m_collision = false;
231  tryApply();
232  }
233  }
234 }

236 void PublicService::publishAsync()
237 {
238  K_D;
239  if (d->m_running) {
240  stop();
241  }
242 
243  if (!d->m_server) {
244  d->m_server = new org::freedesktop::Avahi::Server("org.freedesktop.Avahi", "/", QDBusConnection::systemBus());
245  connect(d->m_server, SIGNAL(StateChanged(int,QString)), d, SLOT(serverStateChanged(int,QString)));
246  }
247 
248  int state = AVAHI_SERVER_INVALID;
249  QDBusReply<int> rep = d->m_server->GetState();
250 
251  if (rep.isValid()) {
252  state = rep.value();
253  }
254  d->m_running = true;
255  d->m_collision = true; // make it look like server is getting out of collision to force registering
256  d->serverStateChanged(state, QString());
257 }

259 void PublicServicePrivate::groupStateChanged(int s, const QString &reason)
260 {
261  switch (s) {
262  case AVAHI_ENTRY_GROUP_COLLISION: {
263  QDBusReply<QString> rep = m_server->GetAlternativeServiceName(m_serviceName);
264  if (rep.isValid()) {
265  m_parent->setServiceName(rep.value());
266  } else {
267  serverStateChanged(AVAHI_SERVER_INVALID, reason);
268  }
269  break;
270  }
271  case AVAHI_ENTRY_GROUP_ESTABLISHED:
272  m_published = true;
273  emit m_parent->published(true);
274  break;
275  case AVAHI_ENTRY_GROUP_FAILURE:
276  serverStateChanged(AVAHI_SERVER_INVALID, reason);
277  default:
278  break;
279  }
280 }
