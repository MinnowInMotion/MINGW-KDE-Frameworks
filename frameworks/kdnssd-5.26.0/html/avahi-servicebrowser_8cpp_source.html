57 void ServiceBrowser::startBrowse()
58 {
59  if (d->m_running) {
60  return;
61  }
62  org::freedesktop::Avahi::Server s("org.freedesktop.Avahi", "/", QDBusConnection::systemBus());
63  QString fullType = d->m_type;
64  if (!d->m_subtype.isEmpty()) {
65  fullType = d->m_subtype + "._sub." + d->m_type;
66  }
67  QDBusReply<QDBusObjectPath> rep = s.ServiceBrowserNew(-1, -1, fullType, domainToDNS(d->m_domain), 0);
68 
69  if (!rep.isValid()) {
70  emit finished();
71  return;
72  }
73  d->m_running = true;
74  d->m_browserFinished = true;
75  org::freedesktop::Avahi::ServiceBrowser *b = new org::freedesktop::Avahi::ServiceBrowser("org.freedesktop.Avahi", rep.value().path(),
76  QDBusConnection::systemBus());
77  connect(b, SIGNAL(ItemNew(int,int,QString,QString,QString,uint)), d,
78  SLOT(gotNewService(int,int,QString,QString,QString,uint)));
79  connect(b, SIGNAL(ItemRemove(int,int,QString,QString,QString,uint)), d,
80  SLOT(gotRemoveService(int,int,QString,QString,QString,uint)));
81  connect(b, SIGNAL(AllForNow()), d, SLOT(browserFinished()));
82  d->m_browser = b;
83  connect(&d->m_timer, SIGNAL(timeout()), d, SLOT(browserFinished()));
84  d->m_timer.start(domainIsLocal(d->m_domain) ? TIMEOUT_LAST_SERVICE : TIMEOUT_START_WAN);
85 }

87 void ServiceBrowserPrivate::serviceResolved(bool success)
88 {
89  QObject *sender_obj = const_cast<QObject *>(sender());
90  RemoteService *svr = static_cast<RemoteService *>(sender_obj);
91  disconnect(svr, SIGNAL(resolved(bool)), this, SLOT(serviceResolved(bool)));
92  QList<RemoteService::Ptr>::Iterator it = m_duringResolve.begin();
93  QList<RemoteService::Ptr>::Iterator itEnd = m_duringResolve.end();
94  while (it != itEnd && svr != (*it).data()) {
95  ++it;
96  }
97  if (it != itEnd) {
98  if (success) {
99  m_services += (*it);
100  emit m_parent->serviceAdded(RemoteService::Ptr(svr));
101  }
102  m_duringResolve.erase(it);
103  queryFinished();
104  }
105 }

115 void ServiceBrowserPrivate::gotNewService(int, int, const QString &name, const QString &type, const QString &domain, uint)
116 {
117  m_timer.start(TIMEOUT_LAST_SERVICE);
118  RemoteService::Ptr svr(new RemoteService(name, type, domain));
119  if (m_autoResolve) {
120  connect(svr.data(), SIGNAL(resolved(bool)), this, SLOT(serviceResolved(bool)));
121  m_duringResolve += svr;
122  svr->resolveAsync();
123  } else {
124  m_services += svr;
125  emit m_parent->serviceAdded(svr);
126  }
127 }

129 void ServiceBrowserPrivate::gotRemoveService(int, int, const QString &name, const QString &type, const QString &domain, uint)
130 {
131  m_timer.start(TIMEOUT_LAST_SERVICE);
132  RemoteService::Ptr tmpl(new RemoteService(name, type, domain));
133  RemoteService::Ptr found = find(tmpl, m_duringResolve);
134  if (found) {
135  m_duringResolve.removeAll(found);
136  return;
137  }
138  found = find(tmpl, m_services);
139  if (!found) {
140  return;
141  }
142 
143  emit m_parent->serviceRemoved(found);
144  m_services.removeAll(found);
145 }
